21/**
22 * \brief Checks that a value is the name of an enum constant and
23 * returns that constant.
24 * \param l A Lua state.
25 * \param index Index of the value in the stack.
26 * \return The corresponding enum value.
27 */
28template<typename E>
29E check_enum(
30 lua_State* l,
31 int index
32) {
33 return check_enum(l, index, EnumInfoTraits<E>::names);
34}

87/**
88 * \brief Function to be used at the Lua to C++ boundary.
89 *
90 * The whole code of any function that can directly be called from Lua must be
91 * wrapped in a lambda passed to exception_boundary_handle, including the
92 * declaration of variables and the return instructions.
93 * Any exception is caught and generates a Lua error instead.
94 *
95 * \param l The Lua state.
96 * \param func The lambda wrapping the code.
97 * \return The return value of the wrapped code.
98 */
99template<typename Callable>
100int exception_boundary_handle(
101 lua_State* l,
102 Callable&& func
103) {
104 try {
105 return func();
106 }
107 catch (const LuaException& ex) {
108 // Error in user script, detected by Solarus.
109 luaL_error(l, ex.what());
110 }
111 catch (const SolarusFatal& ex) {
112 // Internal Solarus bug.
113 luaL_error(l, (std::string("Internal error: ") + ex.what()).c_str());
114 }
115 catch (const std::exception& ex) {
116 // Probably also a Solarus bug.
117 luaL_error(l, (std::string("Internal error: ") + ex.what()).c_str());
118 }
119 // If we get another exception type, just don't catch it.
120 // Because with LuaJIT, it can already be a Lua error properly wrapped in a
121 // C++ exception. Generating a new Lua error from it would repeat the
122 // file and line number in the error message.
123
124 return 0;
125}

127/**
128 * \brief Checks whether a value is the name of an enumeration value and
129 * returns this value.
130 *
131 * Throws a LuaException if the value is not a string or if the string is not
132 * a valid name for the enum.
133 * This is a useful function for mapping strings to C or C++ enums.
134 *
135 * This function is similar to luaL_checkoption() except that it accepts an
136 * std::map<E, std::string> instead of an array of char*,
137 * and returns a value of enumerated type E instead of int.
138 *
139 * \param l A Lua state.
140 * \param index Index of a string in the Lua stack.
141 * \param names A mapping of enum values to strings to search in.
142 * \return The enumerated value corresponding to this string.
143 */
144template<typename E>
145E check_enum(
146 lua_State* l,
147 int index,
148 const std::map<E, std::string>& names
149) {
150 const std::string& name = LuaTools::check_string(l, index);
151 for (const auto& kvp: names) {
152 if (kvp.second == name) {
153 return kvp.first;
154 }
155 }
156
157 // The value was not found. Build an error message with possible values.
158 std::string allowed_names;
159 for (const auto& kvp: names) {
160 allowed_names += "\"" + kvp.second + "\", ";
161 }
162 allowed_names = allowed_names.substr(0, allowed_names.size() - 2);
163
164 arg_error(l, index,
165 std::string("Invalid name '") + name + "'. Allowed names are: "
166 + allowed_names
167 );
168 return E(); // Make sure the compiler is happy.
169}

171/**
172 * \brief Checks that a table field is the name of an enumeration value and
173 * returns this value.
174 *
175 * This function acts like lua_getfield() followed by check_enum().
176 *
177 * \param l A Lua state.
178 * \param table_index Index of a table in the stack.
179 * \param key Key of the field to get in that table.
180 * \param names A mapping of enum values to strings to search in.
181 * \return The enumerated value corresponding to this string.
182 */
183template<typename E>
184E check_enum_field(
185 lua_State* l,
186 int table_index,
187 const std::string& key,
188 const std::map<E, std::string>& names
189) {
190 lua_getfield(l, table_index, key.c_str());
191 if (!lua_isstring(l, -1)) {
192 arg_error(l, table_index,
193 std::string("Bad field '") + key + "' (string expected, got "
194 + luaL_typename(l, -1)
195 );
196 }
197
198 E value = check_enum<E>(l, -1, names);
199 lua_pop(l, 1);
200 return value;
201}

203/**
204 * \brief Like LuaTools::check_enum() but with a default value.
205 *
206 * \param l A Lua state.
207 * \param index Index of a string in the Lua stack.
208 * \param names A mapping of enum values to strings to search in.
209 * \param default_value The default value to return.
210 * \return The enumerated value corresponding to this string.
211 */
212template<typename E>
213E opt_enum(
214 lua_State* l,
215 int index,
216 const std::map<E, std::string>& names,
217 E default_value
218) {
219 if (lua_isnoneornil(l, index)) {
220 return default_value;
221 }
222
223 return check_enum<E>(l, index, names);
224}
225
226/**
227 * \brief Like LuaTools::check_enum_field() but with a default value.
228 * \param l A Lua state.
229 * \param table_index Index of a table in the stack.
230 * \param key Key of the field to get in that table.
231 * \param names A mapping of enum values to strings to search in.
232 * \param default_value The default value to return.
233 * \return The enumerated value corresponding to this string.
234 */
235template<typename E>
236E opt_enum_field(
237 lua_State* l,
238 int table_index,
239 const std::string& key,
240 const std::map<E, std::string>& names,
241 E default_value
242) {
243 lua_getfield(l, table_index, key.c_str());
244 if (lua_isnil(l, -1)) {
245 lua_pop(l, 1);
246 return default_value;
247 }
248
249 if (!lua_isstring(l, -1)) {
250 arg_error(l, table_index,
251 std::string("Bad field '") + key + "' (string expected, got "
252 + luaL_typename(l, -1) + ")"
253 );
254 }
255 E value = check_enum<E>(l, -1, names);
256 lua_pop(l, 1);
257 return value;
258}
