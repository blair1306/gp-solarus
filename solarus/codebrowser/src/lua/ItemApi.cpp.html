38/**
39 * \brief Initializes the item features provided to Lua.
40 */
41void LuaContext::register_item_module() {
42
43 static const luaL_Reg methods[] = {
44 { "get_name", item_api_get_name },
45 { "get_game", item_api_get_game },
46 { "get_map", item_api_get_map },
47 { "get_savegame_variable", item_api_get_savegame_variable },
48 { "set_savegame_variable", item_api_set_savegame_variable },
49 { "get_amount_savegame_variable", item_api_get_amount_savegame_variable },
50 { "set_amount_savegame_variable", item_api_set_amount_savegame_variable },
51 { "is_obtainable", item_api_is_obtainable },
52 { "set_obtainable", item_api_set_obtainable },
53 { "is_assignable", item_api_is_assignable },
54 { "set_assignable", item_api_set_assignable },
55 { "get_can_disappear", item_api_get_can_disappear },
56 { "set_can_disappear", item_api_set_can_disappear },
57 { "get_brandish_when_picked", item_api_get_brandish_when_picked },
58 { "set_brandish_when_picked", item_api_set_brandish_when_picked },
59 { "get_shadow", item_api_get_shadow },
60 { "set_shadow", item_api_set_shadow },
61 { "get_sound_when_picked", item_api_get_sound_when_picked },
62 { "set_sound_when_picked", item_api_set_sound_when_picked },
63 { "get_sound_when_brandished", item_api_get_sound_when_brandished },
64 { "set_sound_when_brandished", item_api_set_sound_when_brandished },
65 { "has_variant", item_api_has_variant },
66 { "get_variant", item_api_get_variant },
67 { "set_variant", item_api_set_variant },
68 { "has_amount", item_api_has_amount },
69 { "get_amount", item_api_get_amount },
70 { "set_amount", item_api_set_amount },
71 { "add_amount", item_api_add_amount },
72 { "remove_amount", item_api_remove_amount },
73 { "get_max_amount", item_api_get_max_amount },
74 { "set_max_amount", item_api_set_max_amount },
75 { "set_finished", item_api_set_finished },
76 { nullptr, nullptr }
77 };
78
79 static const luaL_Reg metamethods[] = {
80 { "__gc", userdata_meta_gc },
81 { "__newindex", userdata_meta_newindex_as_table },
82 { "__index", userdata_meta_index_as_table },
83 { nullptr, nullptr }
84 };
85
86 register_type(item_module_name, nullptr, methods, metamethods);
87}

89/**
90 * \brief Returns whether a value is a userdata of type item.
91 * \param l A Lua context.
92 * \param index An index in the stack.
93 * \return true if the value at this index is an item.
94 */
95bool LuaContext::is_item(lua_State* l, int index) {
96 return is_userdata(l, index, item_module_name);
97}
98
99/**
100 * \brief Checks that the userdata at the specified index of the stack is an
101 * equipment item and returns it.
102 * \param l A Lua context.
103 * \param index An index in the stack.
104 * \return The equipment item.
105 */
106std::shared_ptr<EquipmentItem> LuaContext::check_item(lua_State* l, int index) {
107 return std::static_pointer_cast<EquipmentItem>(check_userdata(
108 l, index, item_module_name
109 ));
110}

112/**
113 * \brief Pushes an equipment item userdata onto the stack.
114 * \param l A Lua context.
115 * \param item An item.
116 */
117void LuaContext::push_item(lua_State* l, EquipmentItem& item) {
118
119 push_userdata(l, item);
120}

194/**
195 * \brief Implementation of item:set_savegame_variable().
196 * \param l The Lua context that is calling this function.
197 * \return Number of values to return to Lua.
198 */
199int LuaContext::item_api_set_savegame_variable(lua_State* l) {
200
201 return LuaTools::exception_boundary_handle(l, [&] {
202 EquipmentItem& item = *check_item(l, 1);
203 std::string savegame_variable;
204 if (!lua_isnil(l, 2)) {
205 savegame_variable = LuaTools::check_string(l, 2);
206 }
207
208 if (!savegame_variable.empty()
209 && !LuaTools::is_valid_lua_identifier(savegame_variable)) {
210 LuaTools::arg_error(l, 2,
211 std::string("savegame variable identifier expected, got '")
212 + savegame_variable + "'");
213 }
214
215 item.set_savegame_variable(savegame_variable);
216
217 return 0;
218 });
219}

723 if (!item.has_amount()) {
724 LuaTools::error(l, std::string("Item '") + item.get_name() + "' has no amount");
725 }
726
727 if (max_amount < 0) {
728 LuaTools::arg_error(l, 2, "Invalid amount value: must be positive or zero");
729 }
730
731 item.set_max_amount(max_amount);
732
733 return 0;
734 });
735}

737/**
738 * \brief Implementation of item:set_finished().
739 * \param l The Lua context that is calling this function.
740 * \return Number of values to return to Lua.
741 */
742int LuaContext::item_api_set_finished(lua_State* l) {
743
744 return LuaTools::exception_boundary_handle(l, [&] {
745 EquipmentItem& item = *check_item(l, 1);
746
747 // Retrieve the equipment item from the hero.
748 Hero& hero = *item.get_game()->get_hero();
749 if (hero.is_using_item()) { // Do nothing if the script has already changed the hero's state.
750
751 EquipmentItemUsage& item_usage = hero.get_item_being_used();
752 item_usage.set_finished();
753 }
754
755 return 0;
756 });
757}
