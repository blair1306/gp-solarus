24using namespace Solarus;

26namespace {
27
28class Element {
29
30 public:
31
32 Element() :
33 rectangle() {
34 }
35
36 explicit Element(const Rectangle& rectangle) :
37 rectangle(rectangle) {
38 }
39
40 const Rectangle& get_bounding_box() const {
41 return rectangle;
42 }
43
44 Rectangle& get_bounding_box() {
45 return rectangle;
46 }
47
48 private:
49
50 Rectangle rectangle;
51};
52
53using ElementPtr = std::shared_ptr<Element>;
54
55/**
56 * \brief Checks the number of elements of a quadtree.
57 */
58void check_num_elements(const Quadtree<ElementPtr>& quadtree, int expected) {
59
60 if (quadtree.get_num_elements() != expected) {
61 std::ostringstream oss;
62 oss << "Wrong number of elements: expected " << expected << ", got " << quadtree.get_num_elements();
63 Debug::die(oss.str());
64 }
65}
66
67/**
68 * \brief Checks that an element was found after a quadtree query.
69 */
70void check_found(const std::vector<ElementPtr>& found_elements, ElementPtr& expected) {
71
72 Debug::check_assertion(
73 std::find(found_elements.begin(), found_elements.end(), expected) != found_elements.end(),
74 "Element not found"
75 );
76}
77
78/**
79 * \brief Creates an element with the given coordinates and adds it to a quatree.
80 */
81ElementPtr add(Quadtree<ElementPtr>& quadtree, const Rectangle& bounding_box) {
82
83 int num_elements = quadtree.get_num_elements();
84 ElementPtr element(std::make_shared<Element>(bounding_box));
85 Debug::check_assertion(quadtree.add(element, element->get_bounding_box()), "Failed to add element");
86 check_num_elements(quadtree, num_elements + 1);
87 return element;
88}
89
90/**
91 * \brief Removes an element from a quatree.
92 */
93void remove(Quadtree<ElementPtr>& quadtree, const ElementPtr& element) {
94
95 int num_elements = quadtree.get_num_elements();
96 Debug::check_assertion(quadtree.remove(element), "Failed to remove element");
97 check_num_elements(quadtree, num_elements - 1);
98}
99
100/**
101 * \brief Moves an element to its new coordinates in a quatree.
102 */
103void move(Quadtree<ElementPtr>& quadtree, const ElementPtr& element) {
104
105 Debug::check_assertion(quadtree.contains(element), "Element not in quadtree");
106
107 int num_elements = quadtree.get_num_elements();
108 Debug::check_assertion(quadtree.move(element, element->get_bounding_box()), "Failed to move element");
109
110 // The number of elements should be unchanged.
111 check_num_elements(quadtree, num_elements);
112}

114/**
115 * \brief Tests an empty quadtree.
116 */
117void test_empty(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
118
119 check_num_elements(quadtree, 0);
120}

122/**
123 * \brief Tests adding elements to a quadtree.
124 */
125void test_add(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
126
127 const std::vector<Rectangle> rectangles = {
128 Rectangle(100, 40, 16, 16),
129 Rectangle(200, 10, 16, 16),
130 Rectangle(250, 20, 16, 16),
131 Rectangle(300, 30, 16, 16),
132 Rectangle(300, 50, 16, 16),
133 Rectangle(800, 40, 16, 16),
134 Rectangle(500, 60, 16, 16),
135 Rectangle(600, 100, 16, 16),
136 Rectangle(400, 300, 16, 16),
137 Rectangle(700, 400, 16, 16)
138 };
139
140 std::vector<ElementPtr> added_elements;
141 for (const Rectangle& rectangle : rectangles) {
142 ElementPtr element = add(quadtree, rectangle);
143 added_elements.push_back(element);
144 }
145
146 Rectangle region(220, 10, 100, 100);
147 std::vector<ElementPtr> found_elements = quadtree.get_elements(region);
148
149 Debug::check_assertion(found_elements.size() == 3, "Expected 3 elements found");
150 check_found(found_elements, added_elements[2]);
151 check_found(found_elements, added_elements[3]);
152 check_found(found_elements, added_elements[4]);
153}

155/**
156 * \brief Tests removing elements from a quadtree.
157 */
158void test_remove(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
159
160 // Get all elements.
161 std::vector<ElementPtr> elements = quadtree.get_elements(quadtree.get_space());
162
163 // Remove some of them.
164 Debug::check_assertion(quadtree.get_num_elements() > 5, "Wrong number of elements");
165 remove(quadtree, elements[0]);
166}

168/**
169 * \brief Tests adding elements whose size overlaps several cells.
170 */
171void test_add_big_size(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
172
173 add(quadtree, Rectangle(25, 25, 600, 600));
174 add(quadtree, Rectangle(100, 0, 16, 960));
175}

177/**
178 * \brief Tests adding elements near the limit or outside the quadtree space.
179 */
180void test_add_limit(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
181
182 add(quadtree, Rectangle(-16, 0, 16, 960));
183
184 // Try to add an element outside the quadtree space.
185 add(quadtree, Rectangle(-160, 0, 16, 960));
186}

188/**
189 * \brief Tests adding an element partially outside the quadtree space, with the center outside.
190 */
191void test_add_center_outside(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
192
193 add(quadtree, Rectangle(-480, 32, 640, 640));
194 add(quadtree, Rectangle(1300, 32, 640, 640));
195}

197/**
198 * \brief Tests moving elements in a quadtree.
199 */
200void test_move(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
201
202 // Get all elements.
203 std::vector<ElementPtr> elements = quadtree.get_elements(quadtree.get_space());
204
205 // Move some of them.
206 Debug::check_assertion(quadtree.get_num_elements() > 4, "Wrong number of elements");
207 ElementPtr element = elements[0];
208 element->get_bounding_box().set_xy(128, 256);
209 move(quadtree, element);
210}
211
212/**
213 * \brief Tests moving elements to outside and from outside a quadtree.
214 */
215void test_move_limit(TestEnvironment& /* env */, Quadtree<ElementPtr>& quadtree) {
216
217 // Get all elements.
218 std::vector<ElementPtr> elements = quadtree.get_elements(quadtree.get_space());
219
220 // Move an element outside the bounds.
221 int num_elements = quadtree.get_num_elements();
222 Debug::check_assertion(num_elements > 4, "Wrong number of elements");
223
224 ElementPtr element = elements[0];
225 element->get_bounding_box().set_xy(-50000, -50000);
226 move(quadtree, element);
227 Debug::check_assertion(quadtree.get_num_elements() == num_elements, "Wrong number of elements");
228
229 // Come back in the bounds.
230 element->get_bounding_box().set_xy(42, 64);
231 move(quadtree, element);
232 Debug::check_assertion(quadtree.get_num_elements() == num_elements, "Wrong number of elements");
233}

237/**
238 * Tests for the path movement.
239 */
240int main(int argc, char** argv) {
241
242 TestEnvironment env(argc, argv);
243
244 int margin = 64;
245 Rectangle space(-margin, -margin, 1280 + 2 * margin, 960 + 2 * margin);
246 Quadtree<ElementPtr> quadtree(space);
247
248 test_empty(env, quadtree);
249 test_add(env, quadtree);
250 test_add_big_size(env, quadtree);
251 test_add_limit(env, quadtree);
252 test_add_center_outside(env, quadtree);
253 test_remove(env, quadtree);
254 test_move(env, quadtree);
255 test_move_limit(env, quadtree);
256
257 return 0;
258}
