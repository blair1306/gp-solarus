44/**
45 * \brief Creates a new chest with the specified treasure.
46 * \param name Name identifying this chest.
47 * \param layer Layer of the chest to create on the map.
48 * \param xy Coordinates of the chest to create.
49 * \param sprite_name Name of the animation set of the
50 * sprite to create for the chest. It must have animations "open" and "closed".
51 * \param treasure The treasure in the chest.
52 */
53Chest::Chest(
54 const std::string& name,
55 int layer,
56 const Point& xy,
57 const std::string& sprite_name,
58 const Treasure& treasure):
59
60 Entity(name, 0, layer, xy, Size(16, 16)),
61 treasure(treasure),
62 open(treasure.is_found()),
63 treasure_given(open),
64 treasure_date(0),
65 opening_method(OpeningMethod::BY_INTERACTION),
66 opening_condition_consumed(false) {
67
68 set_collision_modes(CollisionMode::COLLISION_FACING);
69
70 // Create the sprite.
71 const SpritePtr& sprite = create_sprite(sprite_name);
72 std::string animation = is_open() ? "open" : "closed";
73 sprite->set_current_animation(animation);
74
75 set_origin(get_width() / 2, get_height() - 3);
76
77 // TODO set this as the default drawn_in_y_order for Entity
78 set_drawn_in_y_order(sprite->get_max_size().height > get_height());
79}

134/**
135 * \brief Sets whether the chest is open.
136 *
137 * If you don't change the chest state, this function has no effect.
138 * If you make the chest open, its sprite is updated but this function does
139 * not give any treasure to the player.
140 * If you close the chest, its sprite is updated and the chest will contain
141 * its initial treasure again.
142 *
143 * \param open true to open the chest, false to close it
144 */
145void Chest::set_open(bool open) {
146
147 if (open != this->open) {
148
149 this->open = open;
150
151 const SpritePtr& sprite = get_sprite();
152 if (open) {
153 // open the chest
154 if (sprite != nullptr) {
155 sprite->set_current_animation("open");
156 }
157 }
158 else {
159 // close the chest
160 if (sprite != nullptr) {
161 sprite->set_current_animation("closed");
162 }
163 treasure_given = false;
164 }
165 }
166}

168/**
169 * \brief Returns whether the player is able to open this chest now.
170 * \return \c true if this the player can open the chest.
171 */
172bool Chest::can_open() {
173
174 switch (get_opening_method()) {
175
176 case OpeningMethod::BY_INTERACTION:
177 // No condition: the hero can always open the chest.
178 return true;
179
180 case OpeningMethod::BY_INTERACTION_IF_SAVEGAME_VARIABLE:
181 {
182 // The hero can open the chest if a savegame variable is set.
183 const std::string& required_savegame_variable = get_opening_condition();
184 if (required_savegame_variable.empty()) {
185 return false;
186 }
187
188 Savegame& savegame = get_savegame();
189 if (savegame.is_boolean(required_savegame_variable)) {
190 return savegame.get_boolean(required_savegame_variable);
191 }
192
193 if (savegame.is_integer(required_savegame_variable)) {
194 return savegame.get_integer(required_savegame_variable) > 0;
195 }
196
197 if (savegame.is_string(required_savegame_variable)) {
198 return !savegame.get_string(required_savegame_variable).empty();
199 }
200
201 return false;
202 }
203
204 case OpeningMethod::BY_INTERACTION_IF_ITEM:
205 {
206 // The hero can open the chest if he has an item.
207 const std::string& required_item_name = get_opening_condition();
208 if (required_item_name.empty()) {
209 return false;
210 }
211 const EquipmentItem& item = get_equipment().get_item(required_item_name);
212 return item.is_saved()
213 && item.get_variant() > 0
214 && (!item.has_amount() || item.get_amount() > 0);
215 }
216
217 default:
218 return false;
219 }
220}

283/**
284 * \brief Returns whether opening this chest consumes the condition that
285 * was required.
286 *
287 * If this setting is \c true, here is the behavior when the hero opens
288 * the chest:
289 * - If the opening method is OPENING_BY_INTERACTION_IF_SAVEGAME_VARIABLE,
290 * then the required savegame variable is either:
291 * - set to \c false if it is a boolean,
292 * - decremented if it is an integer,
293 * - set to an empty string if it is a string.
294 * - If the opening method is OPENING_BY_INTERACTION_IF_ITEM, then:
295 * - if the required item has an amount, the amount is decremented.
296 * - if the required item has no amount, its possession state is set to zero.
297 * - With other opening methods, this setting has no effect.
298 *
299 * \return \c true if opening this chest consumes the condition that was
300 * required.
301 */
302bool Chest::is_opening_condition_consumed() const {
303 return opening_condition_consumed;
304}

372/**
373 * \brief Updates the chest.
374 *
375 * This function is called repeatedly by the map.
376 * This is a redefinition of Entity::update()
377 * the handle the chest opening.
378 */
379void Chest::update() {
380
381 if (is_open() && !is_suspended()) {
382
383 if (!treasure_given && treasure_date != 0 && System::now() >= treasure_date) {
384
385 treasure_date = 0;
386 treasure_given = true;
387
388 if (treasure.is_saved()) {
389 // Mark the chest as open in the savegame.
390 get_savegame().set_boolean(treasure.get_savegame_variable(), true);
391 }
392
393 // Notify scripts.
394 bool done = get_lua_context()->chest_on_opened(*this, treasure);
395 if (!done) {
396 if (treasure.is_empty() ||
397 !treasure.is_obtainable()
398 ) {
399 // No treasure and the script does not define any behavior:
400 // unfreeze the hero.
401 get_hero().start_free();
402 }
403 else {
404 // Give the treasure to the player.
405 get_hero().start_treasure(treasure, ScopedLuaRef());
406 }
407 }
408 }
409 }
410
411 Entity::update();
412}

414/**
415 * \copydoc Entity::notify_action_command_pressed
416 */
417bool Chest::notify_action_command_pressed() {
418
419 if (is_enabled() &&
420 get_hero().is_free() &&
421 get_commands_effects().get_action_key_effect() != CommandsEffects::ACTION_KEY_NONE
422 ) {
423
424 if (can_open()) {
425 Sound::play("chest_open");
426 set_open(true);
427 treasure_date = System::now() + 300;
428
429 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
430 get_hero().start_frozen();
431 }
432 else if (!get_cannot_open_dialog_id().empty()) {
433 Sound::play("wrong");
434 get_game().start_dialog(get_cannot_open_dialog_id(), ScopedLuaRef(), ScopedLuaRef());
435 }
436
437 return true;
438 }
439
440 return false;
441}

443/**
444 * \brief This function is called by the map when the game is suspended or resumed.
445 *
446 * This is a redefinition of Entity::set_suspended() to suspend the timer
447 * of the chest being opened.
448 *
449 * \param suspended true to suspend the entity, false to resume it
450 */
451void Chest::set_suspended(bool suspended) {
452
453 Entity::set_suspended(suspended);
454
455 if (!suspended && treasure_date != 0) {
456 // restore the timer
457 treasure_date = System::now() + (treasure_date - get_when_suspended());
458 }
459}
