40/**
41 * \brief The hero entity.
42 *
43 * This class represents the hero. It coordinates his position on the map and his state.
44 * The hero is animated by several sprites that are handled by the HeroSprites class.
45 */
46class Hero: public Entity {
47
48 public:
49
50 static constexpr EntityType ThisType = EntityType::HERO;

53 * \name Creation and destruction.

118 /**
119 * \name Movement.
120 *
121 * Information about the movement of the hero.

139 virtual void notify_movement_changed() override;
140 virtual void notify_movement_finished() override;
141 void reset_movement();

144 * \name Ground under the hero.

146 * Functions to handle the ground of the tiles under the hero.

288 private:
289
290 // state
291 class PlayerMovementState; /**< base class for states whose movement is controlled by the player */
292 class FreeState; /**< the hero is free to move (stopped or walking) and can interact with entities */
293 class CarryingState; /**< the hero can walk but he is carrying a pot or a bush */
294 class SwordLoadingState; /**< the hero can walk but his sword is loading for a spin attack */
295 class SwimmingState; /**< the hero is swimming in deep water */
296 class PushingState; /**< the hero is trying to push an obstacle */
297 class SwordTappingState; /**< the hero is tapping his sword on a wall */
298 class PullingState; /**< the hero is pulling an object */
299 class GrabbingState; /**< the hero is grabbing an object and can pull it */
300 class SwordSwingingState; /**< the hero is swinging his sword */
301 class SpinAttackState; /**< the hero is releasing a spin attack */
302 class LiftingState; /**< the hero is lifting an destroyable item (a pot, a bush, etc.) */
303 class TreasureState; /**< the hero is brandishing a treasure */
304 class RunningState; /**< the hero is running */
305 class ForcedWalkingState; /**< the hero is walking with a predetermined path */
306 class JumpingState; /**< the hero is jumping */
307 class HurtState; /**< the hero is hurt */
308 class PlungingState; /**< the hero is plunging into water */
309 class FallingState; /**< the hero is falling into a hole */
310 class BackToSolidGroundState; /**< the hero is getting back to solid ground (e.g. after he drowned
311 * in deep water or falled into a hole) */
312 class StairsState; /**< the hero is being moved by stairs */
313 class VictoryState; /**< the hero is make a victory sequence with his sword */
314 class UsingItemState; /**< the hero is currently using an equipment item */
315 class BoomerangState; /**< the hero is shooting a boomerang */
316 class HookshotState; /**< the hero is throwing his hookshot */
317 class BowState; /**< the hero is shooting an arrow with a bow */
318 class FrozenState; /**< the hero cannot move for various possible reasons,
319 * including an instruction from the script */

329 // ground
330 void update_ground_effects();
331 void update_ice();
332 void stop_ice_movement();
333
334 // life
335 void check_gameover();
336 void update_invincibility();
337
338 // state
339 bool invincible; /**< Whether the hero is temporarily invincible. */
340 uint32_t end_invincible_date; /**< When stopping the invincibility (0 means infinite). */
341
342 // sprites
343 std::unique_ptr<HeroSprites>
344 sprites; /**< the hero's sprites (note that we don't use the sprites structure from Entity) */
345
346 // position
347 int normal_walking_speed; /**< speed when normally walking */
348 int walking_speed; /**< current walking speed (possibly changed by the ground) */
349
350 // state specific
351 std::shared_ptr<Teletransporter>
352 delayed_teletransporter; /**< a teletransporter that will be activated when the hero finishes
353 * a special behavior, such as falling into a hole or walking on stairs */
354 bool on_raised_blocks; /**< indicates that the hero is currently on
355 * raised crystal blocks */
356
357 // ground
358 Point last_solid_ground_coords; /**< coordinates of the last hero position on a ground
359 * where he can walk (e.g. before jumping or falling into a hole) */
360 int last_solid_ground_layer; /**< layer of the last hero position on a solid ground */
361 ScopedLuaRef
362 target_solid_ground_callback; /**< Function that gives the position where the hero will go back if he falls
363 * into a hole (or some other bad ground), or an empty ref to indicate
364 * that the hero will just return to the last solid ground coordinates. */
365 uint32_t next_ground_date; /**< when something will happen with the ground next time (a sound or a movement) */
366 uint32_t next_ice_date; /**< when recomputing the additional movement on ice */
367 int ice_movement_direction8; /**< wanted movement direction a while ago */
368 Point ground_dxy; /**< additional movement with special ground (hole or ice) */
369
370};
