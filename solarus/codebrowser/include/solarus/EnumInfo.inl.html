21namespace Solarus {
22
23template <typename E>
24const std::list<E> EnumInfo<E>::enums() {
25 std::list<E> enums;
26
27 std::transform(EnumInfoTraits<E>::names.begin(), EnumInfoTraits<E>::names.end(), std::back_inserter(enums),
28 [] (typename decltype(EnumInfoTraits<E>::names)::const_reference val) {
29 return val.first;
30 }
31 );
32 return enums;
33}
34
35template <typename E>
36const std::list<std::string> EnumInfo<E>::names() {
37 std::list<std::string> names;
38
39 std::transform(EnumInfoTraits<E>::names.begin(), EnumInfoTraits<E>::names.end(), std::back_inserter(enums),
40 [] (typename decltype(EnumInfoTraits<E>::names)::const_reference val) {
41 return val.second;
42 }
43 );
44 return names;
45}
46
47template <typename E>
48const std::string& enum_to_name(E value) {
49
50 const auto& it = EnumInfoTraits<E>::names.find(value);
51 if (it == EnumInfoTraits<E>::names.end()) {
52 Debug::die(std::string("Invalid ") + EnumInfoTraits<E>::pretty_name +
53 " number: " + String::to_string(static_cast<int>(value)));
54 }
55
56 return it->second;
57}
58
59template <typename E>
60E name_to_enum(const std::string& name) {
61
62 for (const auto& kvp : EnumInfoTraits<E>::names) {
63 if (kvp.second == name) {
64 return kvp.first;
65 }
66 }
67
68 Debug::die(std::string("Invalid ") + EnumInfoTraits<E>::pretty_name + " name: '" + name + "'");
69}
70
71template <typename E>
72E name_to_enum(const std::string& name, E default_value) {
73
74 for (const auto& kvp : EnumInfoTraits<E>::names) {
75 if (kvp.second == name) {
76 return kvp.first;
77 }
78 }
79
80 return default_value;
81}
