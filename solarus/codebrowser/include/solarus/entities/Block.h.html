17#ifndef SOLARUS_BLOCK_H
18#define SOLARUS_BLOCK_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/lowlevel/Point.h"
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief A entity that the hero can push or pull.
29 *
30 * Some blocks can be moved only once or in only one direction.
31 */
32class Block: public Entity {
33
34 public:
35
36 static constexpr EntityType ThisType = EntityType::BLOCK;
37
38 Block(const std::string& name,
39 int layer,
40 const Point& xy,
41 int direction,
42 const std::string& sprite_name,
43 bool can_be_pushed,
44 bool can_be_pulled,
45 int maximum_moves
46 );
47
48 EntityType get_type() const override;
49 bool is_ground_observer() const override;
50
51 bool is_obstacle_for(Entity& other) override;
52 bool is_hole_obstacle() const override;
53 bool is_teletransporter_obstacle(Teletransporter& teletransporter) override;
54 bool is_hero_obstacle(Hero& hero) override;
55 bool is_enemy_obstacle(Enemy& enemy) override;
56 bool is_destructible_obstacle(Destructible& destructible) override;
57 void notify_created() override;
58 void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
59 void notify_collision_with_switch(Switch& sw, CollisionMode collision_mode) override;
60 bool notify_action_command_pressed() override;
61 bool start_movement_by_hero() override;
62 void stop_movement_by_hero() override;
63 void notify_moving_by(Entity& entity) override;
64 void notify_moved_by(Entity& entity) override;
65 void notify_position_changed() override;
66 void notify_obstacle_reached() override;
67 void notify_ground_below_changed() override;
68
69 void reset();
70 bool is_pushable() const;
71 void set_pushable(bool pushable);
72 bool is_pullable() const;
73 void set_pullable(bool pullable);
74 int get_maximum_moves() const;
75 void set_maximum_moves(int maximum_moves);
76
77 private:
78
79 void movement_by_hero_finished();
80
81 int maximum_moves; /**< indicates whether the block can be pushed
82 * (0: none, 1: once: 2: infinite) */
83 bool sound_played; /**< true if the block sound was played while pulling it */
84 uint32_t when_can_move; /**< date when the hero can move the block again */
85
86 Point last_position; /**< last position of the block before moving */
87 Point initial_position; /**< position of the block when created */
88 int initial_maximum_moves; /**< value of maximum_moves when the block was created */
89 bool can_be_pushed; /**< indicates that the hero can push this block */
90 bool can_be_pulled; /**< indicates that the hero can pull this block */
91
92 static constexpr uint32_t moving_delay = 500; /**< delay between two successive moves of a block */
93
94};
95
96}
97
98#endif
