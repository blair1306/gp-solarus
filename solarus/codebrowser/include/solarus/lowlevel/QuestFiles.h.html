30/**
31 * \brief Provides access to data files of the current quest.
32 *
33 * This namespace provides some functions to read and write data files.
34 * It encapsulates how the access to data files is done.
35 * This is this only place that accesses data files of the
36 * current quest (including the language-specific ones)
37 * and that calls the PHYSFS library to get data files from
38 * the data archive when necessary.
39 */
40namespace QuestFiles {
41
42/**
43 * \brief Describes the possible physical locations of a data file
44 * of the quest.
45 */
46enum class DataFileLocation {
47 LOCATION_NONE,
48 LOCATION_DATA_DIRECTORY,
49 LOCATION_DATA_ARCHIVE,
50 LOCATION_WRITE_DIRECTORY,
51};
