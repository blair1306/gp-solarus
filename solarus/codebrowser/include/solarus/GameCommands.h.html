17#ifndef SOLARUS_GAME_COMMANDS_H
18#define SOLARUS_GAME_COMMANDS_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/InputEvent.h"
22#include "solarus/lua/ScopedLuaRef.h"
23#include "solarus/GameCommand.h"
24#include <map>
25#include <set>
26#include <string>
27
28namespace Solarus {
29
30class Game;
31class Savegame;
32
33/**
34 * \brief Stores the mapping between the in-game high-level commands
35 * and their keyboard and joypad bindings.
36 *
37 * This class receives the low-level keyboard and joypad events that occur
38 * during the game and then notifies the appropriate objects that a built-in
39 * game command was pressed or released.
40 * What we call a game command is a high-level notion such as "action" or
41 * "attack".
42 * The corresponding low-level input event can be a keyboard event or a
43 * joypad event.
44 */
45class GameCommands {
46
47 public:
48
49 explicit GameCommands(Game& game);
50
51 InputEvent::KeyboardKey get_keyboard_binding(GameCommand command) const;
52 void set_keyboard_binding(GameCommand command, InputEvent::KeyboardKey keyboard_key);
53 const std::string& get_joypad_binding(GameCommand command) const;
54 void set_joypad_binding(GameCommand command, const std::string& joypad_string);
55
56 void notify_input(const InputEvent& event);
57 bool is_command_pressed(GameCommand command) const;
58 int get_wanted_direction8() const;
59
60 void customize(GameCommand command, const ScopedLuaRef& callback_ref);
61 bool is_customizing() const;
62 GameCommand get_command_to_customize() const;
63
64 static bool is_joypad_string_valid(const std::string& joypad_string);
65 static const std::string& get_command_name(GameCommand command);
66 static GameCommand get_command_by_name(const std::string& command_name);
67
68 static const std::map<GameCommand, std::string> command_names;
69
70 // High-level resulting commands.
71 void game_command_pressed(GameCommand command);
72 void game_command_released(GameCommand command);
73
74 private:
75
76 Savegame& get_savegame();
77 const Savegame& get_savegame() const;
78
79 // Keyboard mapping.
80 void keyboard_key_pressed(InputEvent::KeyboardKey keyboard_key_pressed);
81 void keyboard_key_released(InputEvent::KeyboardKey keyboard_key_released);
82 const std::string& get_keyboard_binding_savegame_variable(GameCommand command) const;
83 InputEvent::KeyboardKey get_saved_keyboard_binding(GameCommand command) const;
84 void set_saved_keyboard_binding(GameCommand command, InputEvent::KeyboardKey key);
85 GameCommand get_command_from_keyboard(InputEvent::KeyboardKey key) const;
86
87 // Joypad mapping.
88 void joypad_button_pressed(int button);
89 void joypad_button_released(int button);
90 void joypad_axis_moved(int axis, int direction);
91 void joypad_hat_moved(int hat, int direction);
92 const std::string& get_joypad_binding_savegame_variable(GameCommand command) const;
93 std::string get_saved_joypad_binding(GameCommand command) const;
94 void set_saved_joypad_binding(GameCommand command, const std::string& joypad_string);
95 GameCommand get_command_from_joypad(const std::string& joypad_string) const;
96
97 void do_customization_callback();
98
99 Game& game; /**< The game we are controlling. */
100 std::map<InputEvent::KeyboardKey, GameCommand>
101 keyboard_mapping; /**< Associates each game command to the
102 * keyboard key that triggers it. */
103 std::map<std::string, GameCommand>
104 joypad_mapping; /**< Associates each game command to the
105 * joypad action that triggers it. */
106 std::set<GameCommand>
107 commands_pressed; /**< Memorizes the state of each game command. */
108
109 bool customizing; /**< Indicates that the next keyboard or
110 * joypad event will be considered as the
111 * new binding for a game command. */
112 GameCommand command_to_customize; /**< The game command being customized
113 * when customizing is true. */
114 ScopedLuaRef customize_callback_ref; /**< Lua ref to a function to call
115 * when the customization finishes. */
116
117 static const uint16_t
118 direction_masks[4]; /**< Bit mask associated to each direction:
119 * this allows to store any combination of
120 * the four directions into an integer. */
121 static const int
122 masks_to_directions8[16]; /**< Associates to each combination of
123 * directional command a direction between 0
124 * and 7 (-1 means none). */
125 static const std::string
126 direction_names[4]; /**< English name of each arrow direction,
127 * used to save a joypad action as a string. */
128};
129
130}
131
132#endif
