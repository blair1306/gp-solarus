47/**
48 * \brief Updates this state.
49 */
50void Hero::GrabbingState::update() {
51
52 if (is_suspended()) {
53 return;
54 }
55
56 // the hero is grabbing an obstacle: check the direction pressed
57
58 int wanted_direction8 = get_commands().get_wanted_direction8();
59 int sprite_direction8 = get_sprites().get_animation_direction8();
60
61 // release the obstacle
62 Hero& hero = get_entity();
63 if (!get_commands().is_command_pressed(GameCommand::ACTION)) {
64 hero.set_state(new FreeState(hero));
65 }
66
67 // push the obstacle
68 else if (wanted_direction8 == sprite_direction8) {
69 hero.set_state(new PushingState(hero));
70 }
71
72 // pull the obstacle
73 else if (wanted_direction8 == (sprite_direction8 + 4) % 8) {
74 hero.set_state(new PullingState(hero));
75 }
76}
