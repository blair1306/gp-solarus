17#ifndef SOLARUS_SPC_DECODER_H
18#define SOLARUS_SPC_DECODER_H
19
20#include "solarus/Common.h"
21#include "solarus/third_party/snes_spc/spc.h"
22#include <cstddef> // size_t
23#include <cstdint>
24#include <memory>
25
26namespace Solarus {
27
28/**
29 * \brief Encapsulates the SPC music decoding.
30 *
31 * This class allows the Music class to be independent of the SPC library.
32 * To decode SPC music files, the code of the Snes_SPC library is used.
33 * Snes_SPC is a recent SPC decoding library that exists in two versions:
34 * - a fast version, that renders music with a very nice quality;
35 * - a highly accurate version, that renders music with a better accuracy, but about three times slower.
36 * The source code of Solarus include the fast version of Snes_SPC, which I recommend
37 * since the sound generated is good enough.
38 */
39class SpcDecoder {
40
41 public:
42
43 SpcDecoder();
44
45 void load(int16_t* sound_data, size_t sound_size);
46 void decode(int16_t* decoded_data, int nb_samples);
47
48 private:
49
50 struct SNES_SPC_Deleter {
51 void operator()(SNES_SPC* snes_spc_manager) {
52 spc_delete(snes_spc_manager);
53 }
54 };
55
56 struct SPC_Filter_Deleter {
57 void operator()(SPC_Filter* snes_spc_filter) {
58 spc_filter_delete(snes_spc_filter);
59 }
60 };
61
62 // Snes_SPC specific data
63 std::unique_ptr<SNES_SPC, SNES_SPC_Deleter>
64 snes_spc_manager; /**< the snes_spc object encapsulated */
65 std::unique_ptr<SPC_Filter, SPC_Filter_Deleter>
66 snes_spc_filter; /**< the snes_spc filter object */
67
68};
69
70}
71
72#endif
