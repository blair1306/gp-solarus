24/**
25 * \brief Constructor.
26 * \param name Name of the destination to create.
27 * \param layer The layer.
28 * \param xy Coordinates of the destination to create.
29 * \param hero_direction Initial direction of the hero in this state
30 * (0 to 3, or -1 to indicate that the hero's direction is not changed).
31 * \param sprite_name Animation set id of a sprite to represent this
32 * destination, or an empty string.
33 * \param is_default \c true to make this destination the default one.
34 */
35Destination::Destination(
36 const std::string& name,
37 int layer,
38 const Point& xy,
39 int hero_direction,
40 const std::string& sprite_name,
41 bool is_default
42):
43 Entity(name, hero_direction, layer, xy, Size(16, 16)),
44 starting_location_mode(StartingLocationMode::WHEN_WORLD_CHANGES),
45 is_default_destination(is_default) {
46
47 set_origin(8, 13);
48
49 if (!sprite_name.empty()) {
50 create_sprite(sprite_name);
51 }
52}
53
54/**
55 * \brief Returns the type of entity.
56 * \return the type of entity
57 */
58EntityType Destination::get_type() const {
59 return ThisType;
60}
61
62/**
63 * \brief Returns whether this destination is the default one of the map.
64 * \return \c true if this destination is the default one.
65 */
66bool Destination::is_default() const {
67 return is_default_destination;
68}
69
70/**
71 * \brief Returns whether this destination updates the starting location.
72 * \return The starting location mode.
73 */
74StartingLocationMode Destination::get_starting_location_mode() const {
75 return starting_location_mode;
76}
77
78/**
79 * \brief Sets whether this destination updates the starting location.
80 * \param mode The starting location mode.
81 */
82void Destination::set_starting_location_mode(StartingLocationMode mode) {
83 this->starting_location_mode = mode;
84}
