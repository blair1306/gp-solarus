17#ifndef SOLARUS_ARROW_H
18#define SOLARUS_ARROW_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/entities/EntityPtr.h"
23
24namespace Solarus {
25
26/**
27 * \brief An arrow thrown by the bow on the map.
28 */
29class Arrow: public Entity {
30
31 public:
32
33 static constexpr EntityType ThisType = EntityType::ARROW;
34
35 explicit Arrow(const Hero& hero);
36
37 virtual EntityType get_type() const override;
38
39 virtual bool is_teletransporter_obstacle(Teletransporter& teletransporter) override;
40 virtual bool is_stream_obstacle(Stream& stream) override;
41 virtual bool is_stairs_obstacle(Stairs& stairs) override;
42 virtual bool is_low_wall_obstacle() const override;
43 virtual bool is_deep_water_obstacle() const override;
44 virtual bool is_hole_obstacle() const override;
45 virtual bool is_lava_obstacle() const override;
46 virtual bool is_prickle_obstacle() const override;
47 virtual bool is_ladder_obstacle() const override;
48 virtual bool is_switch_obstacle(Switch& sw) override;
49 virtual bool is_raised_block_obstacle(CrystalBlock& raised_block) override;
50 virtual bool is_crystal_obstacle(Crystal& crystal) override;
51 virtual bool is_npc_obstacle(Npc& npc) override;
52 virtual bool is_jumper_obstacle(Jumper& jumper, const Rectangle& candidate_position) override;
53
54 // state
55 virtual void update() override;
56 virtual void set_suspended(bool suspended) override;
57 void stop();
58 bool is_stopped() const;
59 bool is_flying() const;
60 void attach_to(Entity& entity_reached);
61
62 // collisions
63 virtual void notify_collision_with_switch(Switch& sw, CollisionMode collision_mode) override;
64 virtual void notify_collision_with_crystal(Crystal& crystal, CollisionMode collision_mode) override;
65 virtual void notify_collision_with_destructible(Destructible& destructible, CollisionMode collision_mode) override;
66 virtual void notify_collision_with_enemy(Enemy& enemy, Sprite& enemy_sprite, Sprite& this_sprite) override;
67 virtual void notify_attacked_enemy(
68 EnemyAttack attack,
69 Enemy& victim,
70 const Sprite* victim_sprite,
71 EnemyReaction::Reaction& result,
72 bool killed
73 ) override;
74 bool has_reached_map_border();
75
76 private:
77
78 const Hero& hero; /**< the hero */
79 uint32_t disappear_date; /**< date when the arrow disappears */
80 bool stop_now; /**< true to make the arrow stop now */
81 EntityPtr
82 entity_reached; /**< a dynamic entity reached by the arrow
83 * (i.e. an enemy or a crystal, not a usual wall) */
84
85};
86
87}
88
89#endif
