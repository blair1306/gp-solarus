18#include "solarus/lowlevel/PixelFilter.h"
19#include "solarus/lowlevel/Debug.h"
20#include "solarus/lowlevel/shaders/ShaderContext.h"
21#include <utility>
22
23namespace Solarus {
24
25/**
26 * \brief Creates a video mode with the specified properties.
27 *
28 * \c software_filter and \c hardware_filter cannot be both set.
29 *
30 * \param name Lua name of the video mode.
31 * \param initial_window_size Default size of the window when selecting this video mode.
32 * \param software_filter Software filter to apply to the quest image or nullptr.
33 * \param hardware_filter Scaling shader to apply to the quest image or nullptr.
34 */
35VideoMode::VideoMode(
36 const std::string& name,
37 const Size& initial_window_size,
38 std::unique_ptr<PixelFilter> software_filter,
39 std::unique_ptr<Shader> hardware_filter
40):
41 name(name),
42 initial_window_size(initial_window_size),
43 software_filter(std::move(software_filter)),
44 hardware_filter(std::move(hardware_filter)) {
45
46 Debug::check_assertion(software_filter == nullptr || hardware_filter == nullptr,
47 "Video mode can have at most one filter");
48}
49
50/**
51 * \brief Returns the Lua name of this video mode.
52 * \return The video mode name.
53 */
54const std::string& VideoMode::get_name() const {
55 return name;
56}
57
58/**
59 * \brief Returns the default size of the window when selecting this mode.
60 * \return Default size of the window when selecting this video mode.
61 */
62const Size& VideoMode::get_initial_window_size() const {
63 return initial_window_size;
64}
65
66/**
67 * \brief Returns the software filter applied to the quest image in this mode.
68 * \return Software filter or nullptr.
69 */
70const PixelFilter* VideoMode::get_software_filter() const {
71 return software_filter.get();
72}
73
74/**
75 * \brief Returns the hardware filter applied to the quest image in this mode.
76 * \return Hardware filter or nullptr.
77 */
78const Shader* VideoMode::get_hardware_filter() const {
79 return hardware_filter.get();
80}
