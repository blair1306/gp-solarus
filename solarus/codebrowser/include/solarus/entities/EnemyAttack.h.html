17#ifndef SOLARUS_ENEMY_ATTACK_H
18#define SOLARUS_ENEMY_ATTACK_H
19
20namespace Solarus {
21
22/**
23 * \brief Defines the types of attacks an enemy can be victim of.
24 */
25enum class EnemyAttack {
26 SWORD, /**< attacked by the sword (for a spin attack, the life lost is multiplied by 2) */
27 THROWN_ITEM, /**< hit by an item thrown (bush, pot, stone...) */
28 EXPLOSION, /**< explosion of a bomb or something else */
29 ARROW, /**< hit by an arrow */
30 HOOKSHOT, /**< hit by the hookshot */
31 BOOMERANG, /**< hit by the boomerang */
32 FIRE, /**< burned by fire */
33 SCRIPT /**< a script decided to hurt the enemy */
34};
35
36}
