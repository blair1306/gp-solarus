33/**
34 * \brief Initializes the surface features provided to Lua.
35 */
36void LuaContext::register_surface_module() {
37
38 static const luaL_Reg functions[] = {
39 { "create", surface_api_create },
40 { nullptr, nullptr }
41 };
42
43 static const luaL_Reg methods[] = {
44 { "get_size", surface_api_get_size },
45 { "clear", surface_api_clear },
46 { "fill_color", surface_api_fill_color },
47 { "get_opacity", surface_api_get_opacity },
48 { "set_opacity", surface_api_set_opacity },
49 { "get_pixels", surface_api_get_pixels },
50 { "set_pixels", surface_api_set_pixels },
51 { "draw", drawable_api_draw },
52 { "draw_region", drawable_api_draw_region },
53 { "get_blend_mode", drawable_api_get_blend_mode },
54 { "set_blend_mode", drawable_api_set_blend_mode },
55 { "fade_in", drawable_api_fade_in },
56 { "fade_out", drawable_api_fade_out },
57 { "get_xy", drawable_api_get_xy },
58 { "set_xy", drawable_api_set_xy },
59 { "get_movement", drawable_api_get_movement },
60 { "stop_movement", drawable_api_stop_movement },
61 { nullptr, nullptr }
62 };
63
64 static const luaL_Reg metamethods[] = {
65 { "__gc", drawable_meta_gc },
66 { nullptr, nullptr }
67 };
68
69 register_type(surface_module_name, functions, methods, metamethods);
70}

72/**
73 * \brief Returns whether a value is a userdata of type surface.
74 * \param l A Lua context.
75 * \param index An index in the stack.
76 * \return true if the value at this index is a surface.
77 */
78bool LuaContext::is_surface(lua_State* l, int index) {
79 return is_userdata(l, index, surface_module_name);
80}
81
82/**
83 * \brief Checks that the userdata at the specified index of the stack is a
84 * surface and returns it.
85 * \param l a Lua context
86 * \param index an index in the stack
87 * \return the surface
88 */
89SurfacePtr LuaContext::check_surface(lua_State* l, int index) {
90 return std::static_pointer_cast<Surface>(check_userdata(
91 l, index, surface_module_name
92 ));
93}

104/**
105 * \brief Implementation of sol.surface.create().
106 * \param l The Lua context that is calling this function.
107 * \return Number of values to return to Lua.
108 */
109int LuaContext::surface_api_create(lua_State* l) {
110
111 return LuaTools::exception_boundary_handle(l, [&] {
112 SurfacePtr surface;
113 if (lua_gettop(l) == 0) {
114 // create an empty surface with the screen size
115 surface = Surface::create(Video::get_quest_size());
116 }
117 else if (lua_type(l, 1) == LUA_TNUMBER) {
118 // create an empty surface with the specified size
119 int width = LuaTools::check_int(l, 1);
120 int height = LuaTools::check_int(l, 2);
121 surface = Surface::create(width, height);
122 }
123 else if (lua_type(l, 1) == LUA_TSTRING) {
124 // load from a file
125 const std::string& file_name = lua_tostring(l, 1);
126 bool language_specific = LuaTools::opt_boolean(l, 2, false);
127 surface = Surface::create(file_name, language_specific ?
128 Surface::DIR_LANGUAGE : Surface::DIR_SPRITES);
129 }
130 else {
131 LuaTools::type_error(l, 1, "number, string or no value");
132 }
133
134 if (surface == nullptr) {
135 // Image file not found or not valid.
136 lua_pushnil(l);
137 }
138 else {
139 get_lua_context(l).add_drawable(surface);
140 push_surface(l, *surface);
141 }
142 return 1;
143 });
144}

146/**
147 * \brief Implementation of surface:get_size().
148 * \param l The Lua context that is calling this function.
149 * \return Number of values to return to Lua.
150 */
151int LuaContext::surface_api_get_size(lua_State* l) {
152
153 return LuaTools::exception_boundary_handle(l, [&] {
154 Surface& surface = *check_surface(l, 1);
155
156 lua_pushinteger(l, surface.get_width());
157 lua_pushinteger(l, surface.get_height());
158 return 2;
159 });
160}

178/**
179 * \brief Implementation of surface:fill_color().
180 * \param l The Lua context that is calling this function.
181 * \return Number of values to return to Lua.
182 */
183int LuaContext::surface_api_fill_color(lua_State* l) {
184
185 return LuaTools::exception_boundary_handle(l, [&] {
186 Surface& surface = *check_surface(l, 1);
187 Color color = LuaTools::check_color(l, 2);
188
189 if (lua_gettop(l) >= 3) {
190 int x = LuaTools::check_int(l, 3);
191 int y = LuaTools::check_int(l, 4);
192 int width = LuaTools::check_int(l, 5);
193 int height = LuaTools::check_int(l, 6);
194 Rectangle where(x, y, width, height);
195 surface.fill_with_color(color, where);
196 }
197 else {
198 surface.fill_with_color(color);
199 }
200
201 return 0;
202 });
203}
