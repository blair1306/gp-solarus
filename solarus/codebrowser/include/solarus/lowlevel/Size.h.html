17#ifndef SOLARUS_SIZE_H
18#define SOLARUS_SIZE_H
19
20#include "solarus/Common.h"
21#include <iosfwd>
22
23namespace Solarus {
24
25/**
26 * \brief Represents a 2D size.
27 *
28 * This class represents a generic size with
29 * width and height properties.
30 */
31class Size {
32
33 public:
34 ////////////////////////////////////////////////////////////
35 // Constructors
36
37 constexpr Size();
38 constexpr Size(int width, int height);
39
40 ////////////////////////////////////////////////////////////
41 // Operators
42
43 Size& operator+=(const Size& other);
44 Size& operator-=(const Size& other);
45
46 Size& operator*=(int factor);
47 Size& operator/=(int divisor);
48
49 ////////////////////////////////////////////////////////////
50 // Miscellaneous functions
51
52 constexpr bool is_flat() const;
53 constexpr bool is_square() const;
54
55 ////////////////////////////////////////////////////////////
56 // Member data
57
58 int width = 0; /**< Width of an object */
59 int height = 0; /**< Height of an object */
60
61};
62
63////////////////////////////////////////////////////////////
64// Free functions
65
66// Comparison operators
67constexpr bool operator==(const Size& lhs, const Size& rhs);
68constexpr bool operator!=(const Size& lhs, const Size& rhs);
69
70// Arithmetic operators
71constexpr Size operator+(const Size& lhs, const Size& rhs);
72constexpr Size operator-(const Size& lhs, const Size& rhs);
73
74constexpr Size operator*(const Size& size, int factor);
75constexpr Size operator*(int factor, const Size& size);
76constexpr Size operator/(const Size& size, int divisor);
77
78// Output to stream
79std::ostream& operator<<(std::ostream& stream, const Size& size);
80
81}
