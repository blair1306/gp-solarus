42/**
43 * \brief Loads an encoded sound from memory.
44 *
45 * This function respects the prototype specified by libvorbisfile.
46 *
47 * \param ptr pointer to a buffer to load
48 * \param size size
49 * \param nb_bytes number of bytes to load
50 * \param datasource source of the data to read
51 * \return number of bytes loaded
52 */
53size_t cb_read(void* ptr, size_t /* size */, size_t nb_bytes, void* datasource) {
54
55 Sound::SoundFromMemory* mem = static_cast<Sound::SoundFromMemory*>(datasource);
56
57 const size_t total_size = mem->data.size();
58 if (mem->position >= total_size) {
59 if (mem->loop) {
60 mem->position = 0;
61 }
62 else {
63 return 0;
64 }
65 }
66 else if (mem->position + nb_bytes >= total_size) {
67 nb_bytes = total_size - mem->position;
68 }
69
70 std::memcpy(ptr, mem->data.data() + mem->position, nb_bytes);
71 mem->position += nb_bytes;
72
73 return nb_bytes;
74}

76/**
77 * \brief Seeks the sound stream to the specified offset.
78 *
79 * This function respects the prototype specified by libvorbisfile.
80 *
81 * \param datasource Source of the data to read.
82 * \param offset Where to seek.
83 * \param whence How to seek: SEEK_SET, SEEK_CUR or SEEK_END.
84 * \return 0 in case of success, -1 in case of error.
85 */
86int cb_seek(void* datasource, ogg_int64_t offset, int whence) {
87
88 Sound::SoundFromMemory* mem = static_cast<Sound::SoundFromMemory*>(datasource);
89
90 switch (whence) {
91
92 case SEEK_SET:
93 mem->position = offset;
94 break;
95
96 case SEEK_CUR:
97 mem->position += offset;
98 break;
99
100 case SEEK_END:
101 mem->position = mem->data.size() - offset;
102 break;
103 }
104
105 if (mem->position >= mem->data.size()) {
106 mem->position = mem->data.size();
107 }
108
109 return 0;
110}
111
112/**
113 * \brief Returns the current position in a sound stream.
114 *
115 * This function respects the prototype specified by libvorbisfile.
116 *
117 * \param datasource Source of the data to read.
118 * \return The current position.
119 */
120long cb_tell(void* datasource) {
121
122 Sound::SoundFromMemory* mem = static_cast<Sound::SoundFromMemory*>(datasource);
123 return mem->position;
124}

154/**
155 * \brief Destroys the sound.
156 */
157Sound::~Sound() {
158
159 if (is_initialized() && buffer != AL_NONE) {
160
161 // stop the sources where this buffer is attached
162 for (ALuint source: sources) {
163 alSourceStop(source);
164 alSourcei(source, AL_BUFFER, 0);
165 alDeleteSources(1, &source);
166 }
167 alDeleteBuffers(1, &buffer);
168 current_sounds.remove(this);
169 }
170}

172/**
173 * \brief Initializes the audio (music and sound) system.
174 *
175 * This method should be called when the application starts.
176 * If the argument -no-audio is provided, this function has no effect and
177 * there will be no sound.
178 *
179 * \param args Command-line arguments.
180 */
181void Sound::initialize(const Arguments& args) {
182
183 // Check the -no-audio option.
184 const bool disable = args.has_argument("-no-audio");
185 if (disable) {
186 return;
187 }
188
189 // Initialize OpenAL.
190
191 device = alcOpenDevice(nullptr);
192 if (!device) {
193 Debug::error("Cannot open audio device");
194 return;
195 }
196
197 ALCint attr[] = { ALC_FREQUENCY, 32000, 0 }; // 32 KHz is the SPC output sampling rate
198 context = alcCreateContext(device, attr);
199 if (!context) {
200 Debug::error("Cannot create audio context");
201 alcCloseDevice(device);
202 return;
203 }
204 if (!alcMakeContextCurrent(context)) {
205 Debug::error("Cannot activate audio context");
206 alcDestroyContext(context);
207 alcCloseDevice(device);
208 return;
209 }
210
211 alGenBuffers(0, nullptr); // Necessary on some systems to avoid errors with the first sound loaded.
212
213 initialized = true;
214 set_volume(100);
215
216 // initialize the music system
217 Music::initialize();
218}

220/**
221 * \brief Closes the audio (music and sound) system.
222 *
223 * This method should be called when exiting the application.
224 */
225void Sound::quit() {
226
227 if (is_initialized()) {
228
229 // uninitialize the music subsystem
230 Music::quit();
231
232 // clear the sounds
233 all_sounds.clear();
234
235 // uninitialize OpenAL
236
237 alcMakeContextCurrent(nullptr);
238 alcDestroyContext(context);
239 context = nullptr;
240 alcCloseDevice(device);
241 device = nullptr;
242 volume = 1.0;
243
244 initialized = false;
245 }
246}

256/**
257 * \brief Loads and decodes all sounds listed in the game database.
258 */
259void Sound::load_all() {
260
261 if (is_initialized() && !sounds_preloaded) {
262
263 const std::map<std::string, std::string>& sound_elements =
264 CurrentQuest::get_resources(ResourceType::SOUND);
265 for (const auto& kvp: sound_elements) {
266 const std::string& sound_id = kvp.first;
267
268 all_sounds[sound_id] = Sound(sound_id);
269 all_sounds[sound_id].load();
270 }
271
272 sounds_preloaded = true;
273 }
274}

321/**
322 * \brief Updates the audio (music and sound) system.
323 *
324 * This function is called repeatedly by the game.
325 */
326void Sound::update() {
327
328 // update the playing sounds
329 std::list<Sound*> sounds_to_remove;
330 for (Sound* sound: current_sounds) {
331 if (!sound->update_playing()) {
332 sounds_to_remove.push_back(sound);
333 }
334 }
335
336 for (Sound* sound: sounds_to_remove) {
337 current_sounds.remove(sound);
338 }
339
340 // also update the music
341 Music::update();
342}

344/**
345 * \brief Updates this sound when it is playing.
346 * \return true if the sound is still playing, false if it is finished.
347 */
348bool Sound::update_playing() {
349
350 // See if this sound is still playing.
351 if (sources.empty()) {
352 return false;
353 }
354
355 ALuint source = *sources.begin();
356 ALint status;
357 alGetSourcei(source, AL_SOURCE_STATE, &status);
358
359 if (status != AL_PLAYING) {
360 sources.pop_front();
361 alSourcei(source, AL_BUFFER, 0);
362 alDeleteSources(1, &source);
363 }
364
365 return !sources.empty();
366}

368/**
369 * \brief Loads and decodes the sound into memory.
370 */
371void Sound::load() {
372
373 if (alGetError() != AL_NONE) {
374 Debug::error("Previous audio error not cleaned");
375 }
376
377 std::string file_name = std::string("sounds/" + id);
378 if (id.find(".") == std::string::npos) {
379 file_name += ".ogg";
380 }
381
382 // Create an OpenAL buffer with the sound decoded by the library.
383 buffer = decode_file(file_name);
384
385 // buffer is now AL_NONE if there was an error.
386}

388/**
389 * \brief Plays the sound.
390 * \return true if the sound was loaded successfully, false otherwise
391 */
392bool Sound::start() {
393
394 bool success = false;
395
396 if (is_initialized()) {
397
398 if (buffer == AL_NONE) { // first time: load and decode the file
399 load();
400 }
401
402 if (buffer != AL_NONE) {
403
404 // create a source
405 ALuint source;
406 alGenSources(1, &source);
407 alSourcei(source, AL_BUFFER, buffer);
408 alSourcef(source, AL_GAIN, volume);
409
410 // play the sound
411 int error = alGetError();
412 if (error != AL_NO_ERROR) {
413 std::ostringstream oss;
414 oss << "Cannot attach buffer " << buffer
415 << " to the source to play sound '" << id << "': error " << error;
416 Debug::error(oss.str());
417 alDeleteSources(1, &source);
418 }
419 else {
420 sources.push_back(source);
421 current_sounds.remove(this); // to avoid duplicates
422 current_sounds.push_back(this);
423 alSourcePlay(source);
424 error = alGetError();
425 if (error != AL_NO_ERROR) {
426 std::ostringstream oss;
427 oss << "Cannot play sound '" << id << "': error " << error;
428 Debug::error(oss.str());
429 }
430 else {
431 success = true;
432 }
433 }
434 }
435 }
436 return success;
437}

439/**
440 * \brief Loads the specified sound file and decodes its content into an OpenAL buffer.
441 * \param file_name name of the file to open
442 * \return the buffer created, or AL_NONE if the sound could not be loaded
443 */
444ALuint Sound::decode_file(const std::string& file_name) {
445
446 ALuint buffer = AL_NONE;
447
448 if (!QuestFiles::data_file_exists(file_name)) {
449 Debug::error(std::string("Cannot find sound file '") + file_name + "'");
450 return AL_NONE;
451 }
452
453 // load the sound file
454 SoundFromMemory mem;
455 mem.loop = false;
456 mem.position = 0;
457 mem.data = QuestFiles::data_file_read(file_name);
458
459 OggVorbis_File file;
460 int error = ov_open_callbacks(&mem, &file, nullptr, 0, ogg_callbacks);
461
462 if (error) {
463 std::ostringstream oss;
464 oss << "Cannot load sound file '" << file_name
465 << "' from memory: error " << error;
466 Debug::error(oss.str());
467 }
468 else {
469
470 // read the encoded sound properties
471 vorbis_info* info = ov_info(&file, -1);
472 ALsizei sample_rate = ALsizei(info->rate);
473
474 ALenum format = AL_NONE;
475 if (info->channels == 1) {
476 format = AL_FORMAT_MONO16;
477 }
478 else if (info->channels == 2) {
479 format = AL_FORMAT_STEREO16;
480 }
481
482 if (format == AL_NONE) {
483 Debug::error(std::string("Invalid audio format for sound file '")
484 + file_name + "'");
485 }
486 else {
487 // decode the sound with vorbisfile
488 std::vector<char> samples;
489 int bitstream;
490 long bytes_read;
491 long total_bytes_read = 0;
492 const int buffer_size = 16384;
493 char samples_buffer[buffer_size];
494 do {
495 bytes_read = ov_read(&file, samples_buffer, buffer_size, 0, 2, 1, &bitstream);
496 if (bytes_read < 0) {
497 std::ostringstream oss;
498 oss << "Error while decoding ogg chunk in sound file '"
499 << file_name << "': " << bytes_read;
500 Debug::error(oss.str());
501 }
502 else {
503 total_bytes_read += bytes_read;
504 if (format == AL_FORMAT_STEREO16) {
505 samples.insert(samples.end(), samples_buffer, samples_buffer + bytes_read);
506 }
507 else {
508 // mono sound files make no sound on some machines
509 // workaround: convert them on-the-fly into stereo sounds
510 // TODO find a better solution
511 for (int i = 0; i < bytes_read; i += 2) {
512 samples.insert(samples.end(), samples_buffer + i, samples_buffer + i + 2);
513 samples.insert(samples.end(), samples_buffer + i, samples_buffer + i + 2);
514 }
515 total_bytes_read += bytes_read;
516 }
517 }
518 }
519 while (bytes_read > 0);
520
521 // copy the samples into an OpenAL buffer
522 alGenBuffers(1, &buffer);
523 if (alGetError() != AL_NO_ERROR) {
524 Debug::error("Failed to generate audio buffer");
525 }
526 alBufferData(buffer,
527 AL_FORMAT_STEREO16,
528 reinterpret_cast<ALshort*>(samples.data()),
529 ALsizei(total_bytes_read),
530 sample_rate);
531 ALenum error = alGetError();
532 if (error != AL_NO_ERROR) {
533 std::ostringstream oss;
534 oss << "Cannot copy the sound samples of '"
535 << file_name << "' into buffer " << buffer
536 << ": error " << error;
537 Debug::error(oss.str());
538 buffer = AL_NONE;
539 }
540 }
541 ov_clear(&file);
542 }
543
544 mem.data.clear();
545
546 return buffer;
547}
