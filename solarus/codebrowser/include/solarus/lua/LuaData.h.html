17#ifndef SOLARUS_LUA_DATA_FILE_H
18#define SOLARUS_LUA_DATA_FILE_H
19
20#include "solarus/Common.h"
21#include <iosfwd>
22#include <string>
23
24struct lua_State;
25
26namespace Solarus {
27
28/**
29 * \brief Abstract class for data the can be loaded and optionally saved as Lua.
30 */
31class SOLARUS_API LuaData {
32
33 public:
34
35 LuaData() = default;
36 virtual ~LuaData() = default;
37
38 virtual bool import_from_lua(lua_State* l) = 0;
39 virtual bool export_to_lua(std::ostream& out) const; // Optional.
40
41 bool import_from_buffer(const std::string& buffer, const std::string& file_name);
42 bool import_from_file(const std::string& file_name);
43 bool import_from_quest_file(
44 const std::string& quest_file_name,
45 bool language_specific = false
46 );
47
48 bool export_to_buffer(std::string& buffer) const;
49 bool export_to_file(const std::string& file_name) const;
50
51 static std::string escape_string(std::string value);
52 static std::string escape_multiline_string(std::string value);
53 static std::string unescape_multiline_string(std::string value);
54
55};
56
57}
58
59#endif
