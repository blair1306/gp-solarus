44/**
45 * \brief Comparator that sorts entities according to their stacking order
46 * on the map (layer and then Z index).
47 */
48class ZOrderComparator {
49
50 public:
51
52 /**
53 * \brief Creates a Z order comparator.
54 * \param entities The map entities to work with.
55 */
56 explicit ZOrderComparator(const Entities& entities) :
57 entities(entities) {
58
59 }
60
61 /**
62 * \brief Compares two con entities.
63 * \param first An entity.
64 * \param second Another entity.
65 * \return \c true if the first entity's Z index is lower than the second one's.
66 */
67 bool operator()(const ConstEntityPtr& first, const ConstEntityPtr& second) const {
68
69 if (first->get_layer() < second->get_layer()) {
70 return true;
71 }
72
73 if (first->get_layer() > second->get_layer()) {
74 return false;
75 }
76
77 // Same layer.
78 return entities.get_entity_relative_z_order(first) < entities.get_entity_relative_z_order(second);
79 }
80
81 private:
82
83 const Entities& entities;
84
85};
86
87/**
88 * \brief Comparator that sorts entities in drawing order.
89 * on the map (layer and then Z index).
90 */
91class DrawingOrderComparator {
92
93 public:
94
95 /**
96 * \brief Compares two entities.
97 * \param first An entity.
98 * \param second Another entity.
99 * \return \c true if the first entity should be drawn before the secone one.
100 */
101 bool operator()(const EntityPtr& first, const EntityPtr& second) const {
102
103 if (first->get_layer() < second->get_layer()) {
104 return true;
105 }
106
107 if (first->get_layer() > second->get_layer()) {
108 return false;
109 }
110
111 // Same layer.
112 // All entities displayed in Z order are displayed before entities displayed in Y order.
113 if (!first->is_drawn_in_y_order() && second->is_drawn_in_y_order()) {
114 return true;
115 }
116
117 if (first->is_drawn_in_y_order() && !second->is_drawn_in_y_order()) {
118 return false;
119 }
120
121 if (first->is_drawn_in_y_order()) {
122 // Both entities are displayed in Y order.
123 return first->get_y() < second->get_y();
124 }
125
126 // Both entities are displayed in Z order.
127 const Entities& entities = first->get_entities();
128 return entities.get_entity_relative_z_order(first) < entities.get_entity_relative_z_order(second);
129 }
130
131};

135/**
136 * \brief Constructor.
137 * \param game The game.
138 * \param map The map.
139 */
140Entities::Entities(Game& game, Map& map):
141 game(game),
142 map(map),
143 map_width8(0),
144 map_height8(0),
145 tiles_grid_size(0),
146 tiles_ground(),
147 non_animated_regions(),
148 tiles_in_animated_regions(),
149 hero(game.get_hero()),
150 camera(nullptr),
151 named_entities(),
152 all_entities(),
153 quadtree(),
154 z_caches(),
155 entities_drawn_not_at_their_position(),
156 entities_to_draw(),
157 entities_to_remove(),
158 default_destination(nullptr) {
159
160 // Initialize the size.
161 initialize_layers();
162 map_width8 = map.get_width8();
163 map_height8 = map.get_height8();
164 tiles_grid_size = map.get_width8() * map.get_height8();
165 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
166
167 Ground initial_ground = (layer == map.get_min_layer()) ? Ground::TRAVERSABLE : Ground::EMPTY;
168 tiles_ground[layer].assign(tiles_grid_size, initial_ground);
169
170 non_animated_regions[layer] = std::unique_ptr<NonAnimatedRegions>(
171 new NonAnimatedRegions(map, layer)
172 );
173 }
174
175 // Initialize the quadtree.
176 const int margin = 64;
177 Rectangle quadtree_space(-margin, -margin, map.get_width() + 2 * margin, map.get_height() + 2 * margin);
178 quadtree.initialize(quadtree_space);
179
180 // Create the camera.
181 add_entity(std::make_shared<Camera>(map));
182}

184/**
185 * \brief Creates live entities from the given data.
186 */
187void Entities::create_entities(const MapData& data) {
188
189 // Create entities from the map data file.
190 LuaContext& lua_context = map.get_lua_context();
191 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
192 for (int i = 0; i < data.get_num_entities(layer); ++i) {
193 const EntityData& entity_data = data.get_entity({ layer, i });
194 EntityType type = entity_data.get_type();
195 if (!EntityTypeInfo::can_be_stored_in_map_file(type)) {
196 Debug::error("Illegal entity type in map data: " + enum_to_name(type));
197 }
198 if (lua_context.create_map_entity_from_data(map, entity_data)) {
199 lua_pop(lua_context.get_internal_state(), 1); // Discard the created entity on the stack.
200 }
201 }
202 }
203}

244/**
245 * \brief Sets the tile ground property of an 8*8 square of the map.
246 *
247 * Coordinates outside the range of the map are not an error:
248 * in this case, this function does nothing.
249 *
250 * \param layer Layer of the square.
251 * \param x8 X coordinate of the square (divided by 8).
252 * \param y8 Y coordinate of the square (divided by 8).
253 * \param ground The ground property to set.
254 */
255void Entities::set_tile_ground(int layer, int x8, int y8, Ground ground) {
256
257 if (x8 >= 0 && x8 < map_width8 && y8 >= 0 && y8 < map_height8) {
258 int index = y8 * map_width8 + x8;
259 tiles_ground[layer][index] = ground;
260 }
261}

288 * given name.

290EntityPtr Entities::find_entity(const std::string& name) {
291
292 auto it = named_entities.find(name);
293 if (it == named_entities.end()) {
294 return nullptr;
295 }
296
297 const EntityPtr& entity = it->second;
298
299 if (entity->is_being_removed()) {
300 return nullptr;
301 }
302
303 return entity;

306/**
307 * \brief Returns the entities of the map having the specified name prefix.
308 *
309 * The hero is included if the prefix matches.
310 *
311 * \param prefix Prefix of the name.
312 * \return The entities having this prefix in their name, in arbitrary order.
313 */
314EntityVector Entities::get_entities_with_prefix(const std::string& prefix) {
315
316 EntityVector entities;
317
318 if (prefix.empty()) {
319 // No prefix: return all entities no matter their name.
320 for (const EntityPtr& entity: all_entities) {
321 if (!entity->is_being_removed()) {
322 entities.push_back(entity);
323 }
324 }
325 entities.push_back(hero);
326 return entities;
327 }
328
329 // Normal case: add entities whose name starts with the prefix.
330 for (const auto& kvp: named_entities) {
331 const EntityPtr& entity = kvp.second;
332 if (entity->has_prefix(prefix) &&
333 !entity->is_being_removed()) {
334 entities.push_back(entity);
335 }
336 }
337
338 return entities;
339}

341/**
342 * \brief Like get_entities_with_prefix(const std::string&), but sorts entities according to
343 * their Z index on the map.
344 * \param prefix Prefix of the name.
345 * \return The entities having this prefix in their name, in Z order.
346 */
347EntityVector Entities::get_entities_with_prefix_sorted(const std::string& prefix) {
348
349 EntityVector entities = get_entities_with_prefix(prefix);
350 std::sort(entities.begin(), entities.end(), ZOrderComparator(*this));
351
352 return entities;

355/**
356 * \brief Returns the entities of the map with the specified type and having
357 * the specified name prefix.
358 * \param type Type of entity.
359 * \param prefix Prefix of the name.
360 * \return The entities of this type and having this prefix in their name, in arbitrary order.
361 */
362EntityVector Entities::get_entities_with_prefix(
363 EntityType type, const std::string& prefix) {
364
365 EntityVector entities;
366
367 if (prefix.empty()) {
368 // No prefix: return all entities of the type, no matter their name.
369 for (const EntityPtr& entity: get_entities_by_type(type)) {
370 if (!entity->is_being_removed()) {
371 entities.push_back(entity);
372 }
373 }
374 return entities;
375 }
376
377 // Normal case: add entities whose name starts with the prefix.
378 for (const auto& kvp: named_entities) {
379 const EntityPtr& entity = kvp.second;
380 if (entity->get_type() == type &&
381 entity->has_prefix(prefix) &&
382 !entity->is_being_removed()
383 ) {
384 entities.push_back(entity);
385 }
386 }
387
388 return entities;
389}

478/**
479 * \brief Returns all entities in the same separator region as the given point.
480 *
481 * Regions are assumed to be rectangular (convex: no "L" shape).
482 *
483 * \param[in] xy A point.
484 * \param[out] result The entities in the same region as the point,
485 * where regions are delimited by separators and map limits.
486 */
487void Entities::get_entities_in_region(
488 const Point& xy, EntityVector& result
489) {
490
491 // Find the bounding box of the region.
492 Rectangle region_box = get_region_box(xy);
493
494 // Get entities in that rectangle.
495 get_entities_in_rectangle(region_box, result);
496}

498/**
499 * \brief Determines the bounding box of a same separator region.
500 *
501 * Regions are assumed to be rectangular (convex: no "L" shape).
502 *
503 * \param point A point.
504 * \return The box of the region.
505 */
506Rectangle Entities::get_region_box(const Point& point) const {
507
508 // Start with a rectangle of the whole map.
509 int top = 0;
510 int bottom = map.get_height();
511 int left = 0;
512 int right = map.get_width();
513
514 // Find the closest separator in each direction.
515
516 const std::set<ConstSeparatorPtr>& separators =
517 get_entities_by_type<Separator>();
518 for (const ConstSeparatorPtr& separator: separators) {
519
520 const Point& separator_center = separator->get_center_point();
521
522 if (separator->is_vertical()) {
523
524 // Vertical separation.
525 if (point.y < separator->get_top_left_y() ||
526 point.y >= separator->get_top_left_y() + separator->get_height()) {
527 // This separator is irrelevant: the point is not in either side,
528 // it is too much to the north or to the south.
529 //
530 // |
531 // |
532 // |
533 //
534 // x
535 //
536 continue;
537 }
538
539 if (separator_center.x <= point.x) {
540 // The separator is on the left.
541 left = std::max(left, separator_center.x);
542 }
543 else {
544 // The separator is on the right.
545 right = std::min(right, separator_center.x);
546 }
547 }
548 else {
549 // Horizontal separation.
550 if (point.x < separator->get_top_left_x() ||
551 point.x >= separator->get_top_left_x() + separator->get_width()) {
552 // This separator is irrelevant: the point is not in either side.
553 continue;
554 }
555
556 if (separator_center.y <= point.y) {
557 // The separator is on the top.
558 top = std::max(top, separator_center.y);
559 }
560 else {
561 // The separator is on the bottom.
562 bottom = std::min(bottom, separator_center.y);
563 }
564 }
565 }
566
567 Debug::check_assertion(top < bottom && left < right, "Invalid region rectangle");
568
569 return Rectangle(left, top, right - left, bottom - top);
570}

617/**
618 * \brief Returns all entities of a type on the given layer.
619 * \param type Type of entities to get.
620 * \param layer The layer to get entities from.
621 * \return All entities of the type on this layer.
622 */
623EntitySet Entities::get_entities_by_type(EntityType type, int layer) {
624
625 Debug::check_assertion(map.is_valid_layer(layer), "Invalid layer");
626
627 EntitySet result;
628
629 const auto& it = entities_by_type.find(type);
630 if (it == entities_by_type.end()) {
631 return result;
632 }
633
634 const ByLayer<EntitySet>& sets = it->second;
635 const auto& layer_it = sets.find(layer);
636 if (layer_it == sets.end()) {
637 return result;
638 }
639 for (const EntityPtr& entity : layer_it->second) {
640 result.insert(entity);
641 }
642 return result;
643}

645/**
646 * \brief Returns a hint on the Z order of this entity.
647 *
648 * It can be used to know if an entity is above or below another one on the
649 * same layer. It is unique for each entity on the same layer.
650 *
651 * This is an arbitrary value: it is not necessarily between 0 and the number
652 * of entities on the layer. Its value only makes sense when compared to the
653 * value of other entities.
654 * Due to entities being removed from the map, entities whose layer changes,
655 * and bring_to_front()/bring_to_back() calls,
656 * there may be holes in the sequence.
657 * The returned value can even be negative.
658 *
659 * \param entity An entity of the map.
660 * \return Its relative Z order.
661 */
662int Entities::get_entity_relative_z_order(const ConstEntityPtr& entity) const {
663
664 const int layer = entity->get_layer();
665 return z_caches.at(layer).get_z(entity);
666}

690/**
691 * \brief Notifies all entities of the map that the map has just become active.
692 */
693void Entities::notify_map_started() {
694
695 // Setup non-animated tiles pre-drawing.
696 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
697 std::vector<TileInfo> tiles_in_animated_regions_info;
698 non_animated_regions.at(layer)->build(tiles_in_animated_regions_info);
699 for (const TileInfo& tile_info : tiles_in_animated_regions_info) {
700 // This tile is non-optimizable, create it for real.
701 TilePtr tile = std::make_shared<Tile>(tile_info);
702 tiles_in_animated_regions.at(layer).push_back(tile);
703 add_entity(tile);
704 }
705 }
706
707 // Now, tiles_in_animated_regions contains the tiles that won't be optimized.
708 // Notify entities.
709 for (const EntityPtr& entity: all_entities) {
710 entity->notify_map_started();
711 entity->notify_tileset_changed();
712 }
713 hero->notify_map_started();
714 hero->notify_tileset_changed();
715}

756/**
757 * \brief Creates the internal layer structures.
758 *
759 * The min and max layers in the map must be known at this point.
760 */
761void Entities::initialize_layers() {
762
763 Debug::check_assertion(z_caches.empty(), "Layers already initialized");
764
765 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
766 tiles_ground[layer] = std::vector<Ground>();
767 non_animated_regions[layer] = std::unique_ptr<NonAnimatedRegions>();
768 tiles_in_animated_regions[layer] = std::vector<TilePtr>();
769 z_caches[layer] = ZCache();
770 }
771}

773/**
774 * \brief Adds tile creation info to the map.
775 *
776 * If possible, the actual tile will never be created for performance reasons:
777 * instead, only its picture and its obstacle info are stored.
778 *
779 * This function is called for each tile when loading the map.
780 *
781 * \param tile_info The tile info to add.
782 */
783void Entities::add_tile_info(const TileInfo& tile_info) {
784
785 const Rectangle& box = tile_info.box;
786 const int layer = tile_info.layer;
787 Debug::check_assertion(map.is_valid_layer(layer),
788 "Invalid layer");
789
790 Debug::check_assertion(tile_info.pattern != nullptr,
791 "Missing tile pattern");
792 const TilePattern& pattern = *tile_info.pattern;
793
794 // The size of a runtime tile should be the size of its pattern
795 // for performance reasons, to optimize away more tiles.
796 Debug::check_assertion(
797 box.get_width() == pattern.get_width() &&
798 box.get_height() == pattern.get_height(),
799 "Static tile size must match tile pattern size");
800
801 // Update the animated regions manager.
802 non_animated_regions[tile_info.layer]->add_tile(tile_info);
803
804 // Update the ground list.
805 const Ground ground = pattern.get_ground();
806
807 const int tile_x8 = box.get_x() / 8;
808 const int tile_y8 = box.get_y() / 8;
809 const int tile_width8 = box.get_width() / 8;
810 const int tile_height8 = box.get_height() / 8;
811
812 int i, j;
813 Ground non_obstacle_triangle;
814
815 switch (ground) {
816
817 // If the obstacle property is the same for all points inside the tile
818 // pattern, then all 8x8 squares of the extended tile have the same
819 // property.
820 case Ground::TRAVERSABLE:
821 case Ground::LOW_WALL:
822 case Ground::SHALLOW_WATER:
823 case Ground::DEEP_WATER:
824 case Ground::GRASS:
825 case Ground::HOLE:
826 case Ground::ICE:
827 case Ground::LAVA:
828 case Ground::PRICKLE:
829 case Ground::LADDER:
830 case Ground::WALL:
831 for (i = 0; i < tile_height8; i++) {
832 for (j = 0; j < tile_width8; j++) {
833 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, ground);
834 }
835 }
836 break;
837
838 // If the top right corner of the tile is an obstacle,
839 // then the top right 8x8 squares are Ground::WALL, the bottom left 8x8
840 // squares are Ground::TRAVERSABLE or Ground::DEEP_WATER and the 8x8 squares
841 // on the diagonal are Ground::WALL_TOP_RIGHT.
842 case Ground::WALL_TOP_RIGHT:
843 case Ground::WALL_TOP_RIGHT_WATER:
844
845 non_obstacle_triangle = (ground == Ground::WALL_TOP_RIGHT) ?
846 Ground::TRAVERSABLE : Ground::DEEP_WATER;
847
848 // We traverse each row of 8x8 squares on the tile.
849 for (i = 0; i < tile_height8; i++) {
850
851 // 8x8 square on the diagonal.
852 set_tile_ground(layer, tile_x8 + i, tile_y8 + i, Ground::WALL_TOP_RIGHT);
853
854 // Left part of the row: we are in the bottom-left corner.
855 for (j = 0; j < i; j++) {
856 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, non_obstacle_triangle);
857 }
858
859 // Right part of the row: we are in the top-right corner.
860 for (j = i + 1; j < tile_width8; j++) {
861 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL);
862 }
863 }
864 break;
865
866 case Ground::WALL_TOP_LEFT:
867 case Ground::WALL_TOP_LEFT_WATER:
868
869 non_obstacle_triangle = (ground == Ground::WALL_TOP_LEFT) ?
870 Ground::TRAVERSABLE : Ground::DEEP_WATER;
871
872 // We traverse each row of 8x8 squares on the tile.
873 for (i = 0; i < tile_height8; i++) {
874
875 // Right part of the row: we are in the bottom-right corner.
876 for (j = tile_width8 - i; j < tile_width8; j++) {
877 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, non_obstacle_triangle);
878 }
879
880 // Left part of the row: we are in the top-left corner.
881 for (j = 0; j < tile_width8 - i - 1; j++) {
882 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL);
883 }
884
885 // 8x8 square on the diagonal.
886 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL_TOP_LEFT);
887 }
888 break;
889
890 case Ground::WALL_BOTTOM_LEFT:
891 case Ground::WALL_BOTTOM_LEFT_WATER:
892
893 non_obstacle_triangle = (ground == Ground::WALL_BOTTOM_LEFT) ?
894 Ground::TRAVERSABLE : Ground::DEEP_WATER;
895
896 // We traverse each row of 8x8 squares on the tile.
897 for (i = 0; i < tile_height8; i++) {
898
899 // Right part of the row: we are in the top-right corner.
900 for (j = i + 1; j < tile_width8; j++) {
901 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, non_obstacle_triangle);
902 }
903 // Left part of the row: we are in the bottom-left corner.
904 for (j = 0; j < i; j++) {
905 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL);
906 }
907
908 // 8x8 square on the diagonal.
909 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL_BOTTOM_LEFT);
910 }
911 break;
912
913 case Ground::WALL_BOTTOM_RIGHT:
914 case Ground::WALL_BOTTOM_RIGHT_WATER:
915
916 non_obstacle_triangle = (ground == Ground::WALL_BOTTOM_RIGHT) ?
917 Ground::TRAVERSABLE : Ground::DEEP_WATER;
918
919 // We traverse each row of 8x8 squares on the tile.
920 for (i = 0; i < tile_height8; i++) {
921
922 // 8x8 square on the diagonal
923 set_tile_ground(layer, tile_x8 + tile_width8 - i - 1, tile_y8 + i, Ground::WALL_BOTTOM_RIGHT);
924
925 // Left part of the row: we are in the top-left corner.
926 for (j = 0; j < tile_width8 - i - 1; j++) {
927 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, non_obstacle_triangle);
928 }
929
930 // Right part of the row: we are in the bottom-right corner.
931 for (j = tile_width8 - i; j < tile_width8; j++) {
932 set_tile_ground(layer, tile_x8 + j, tile_y8 + i, Ground::WALL);
933 }
934 }
935 break;
936
937 case Ground::EMPTY:
938 // Keep the ground property from any tile placed before.
939 break;
940 }
941}

943/**
944 * \brief Adds an entity to the map.
945 *
946 * This function can be called when loading the map or,
947 * except for tiles, when the map is already loaded.
948 * If the entity specified is nullptr
949 * (because some entity creation functions may return nullptr),
950 * nothing is done.
951 *
952 * \param entity The entity to add or nullptr.
953 */
954void Entities::add_entity(const EntityPtr& entity) {
955
956 if (entity == nullptr) {
957 return;
958 }
959
960 Debug::check_assertion(map.is_valid_layer(entity->get_layer()),
961 "No such layer on this map");
962
963 const EntityType type = entity->get_type();
964 if (type != EntityType::TILE) { // Tiles are optimized specifically.
965 const int layer = entity->get_layer();
966
967 // Update the quadtree.
968 quadtree.add(entity, entity->get_max_bounding_box());
969
970 // Update the specific entities lists.
971 switch (entity->get_type()) {
972
973 case EntityType::CAMERA:
974 Debug::check_assertion(camera == nullptr, "Only one camera is supported");
975 camera = std::static_pointer_cast<Camera>(entity);
976 break;
977
978 case EntityType::DESTINATION:
979 {
980 std::shared_ptr<Destination> destination =
981 std::static_pointer_cast<Destination>(entity);
982 if (this->default_destination == nullptr || destination->is_default()) {
983 this->default_destination = destination;
984 }
985 }
986 break;
987
988 default:
989 break;
990 }
991
992 // Update the list of entities drawn not at their position.
993 if (!entity->is_drawn_at_its_position()) {
994 entities_drawn_not_at_their_position[layer].push_back(entity);
995 }
996
997 // Track the insertion order.
998 z_caches[layer].add(entity);
999
1000 // Update the list of entities by type.
1001 auto it = entities_by_type.find(type);
1002 if (it == entities_by_type.end()) {
1003 it = entities_by_type.emplace(type, ByLayer<EntitySet>()).first;
1004 }
1005 ByLayer<EntitySet>& sets = it->second;
1006 sets[layer].insert(entity);
1007
1008 // Update the list of all entities.
1009 if (type != EntityType::HERO) {
1010 all_entities.push_back(entity);
1011 }
1012 }
1013
1014 // Rename the entity if there is already an entity with the same name.
1015 std::string name = entity->get_name();
1016 if (!name.empty()) {
1017
1018 if (named_entities.find(name) != named_entities.end()) {
1019 // This name is already used by another entity. Add a suffix.
1020 std::ostringstream oss;
1021 std::istringstream iss;
1022 int suffix_number = 1;
1023 std::string prefix = name;
1024 size_t index = name.rfind('_');
1025 if (index != std::string::npos) {
1026 // If there is already a numbered suffix, we will increment it.
1027 const std::string& suffix = name.substr(index + 1);
1028 iss.clear();
1029 iss.str(suffix);
1030 if (iss >> suffix_number) {
1031 prefix = name.substr(0, index);
1032 }
1033 }
1034
1035 // Now we have the final prefix. Find the first available suffix.
1036 do {
1037 ++suffix_number;
1038 oss.str("");
1039 oss.clear();
1040 oss << prefix << '_' << suffix_number;
1041 name = oss.str();
1042 }
1043 while (named_entities.find(name) != named_entities.end());
1044
1045 entity->set_name(name);
1046 }
1047 named_entities[name] = entity;
1048 }
1049
1050 // Notify the entity.
1051 if (type != EntityType::HERO) {
1052 entity->set_map(map);
1053 }
1054}

1065 entities_to_remove.push_back(shared_entity);
1066

1069
1070 // Remove the entity from the by name list
1071 // to allow users to create a new one with
1072 // the same name right now.
1073 if (!entity.get_name().empty()) {
1074 named_entities.erase(entity.get_name());
1075 }
1076 }

1103/**
1104 * \brief Removes and destroys the entities placed in the entities_to_remove list.
1105 */
1106void Entities::remove_marked_entities() {
1107
1108 // Remove the marked entities.
1109 for (const EntityPtr& entity: entities_to_remove) {
1110
1111 const EntityType type = entity->get_type();
1112 const int layer = entity->get_layer();
1113
1114 // Remove it from the quadtree.
1115 quadtree.remove(entity);
1116
1117 // Remove it from the whole list.
1118 all_entities.remove(entity);
1119 const std::string& name = entity->get_name();
1120 if (!name.empty()) {
1121 named_entities.erase(name);
1122 }
1123
1124 // Update the specific entities lists.
1125 switch (type) {
1126
1127 case EntityType::CAMERA:
1128 camera = nullptr;
1129 break;
1130
1131 default:
1132 break;
1133 }
1134
1135 // Track the insertion order.
1136 z_caches.at(layer).remove(entity);
1137
1138 // Update the list of entities by type.
1139 const auto& it = entities_by_type.find(type);
1140 if (it != entities_by_type.end()) {
1141 ByLayer<EntitySet>& sets = it->second;
1142 sets[layer].erase(entity);
1143 }
1144
1145 // Destroy it.
1146 notify_entity_removed(*entity);
1147 }
1148 entities_to_remove.clear();

1176void Entities::update() {

1202/**
1203 * \brief Draws the entities on the map surface.
1204 */
1205void Entities::draw() {
1206
1207 const CameraPtr& camera = get_camera();
1208 if (camera == nullptr) {
1209 return;
1210 }
1211
1212 const SurfacePtr& camera_surface = camera->get_surface();
1213
1214 // Lazily build the list of entities to draw.
1215 if (entities_to_draw.empty()) {
1216
1217 // Add entities in the camera,
1218 // or nearby because of possible
1219 // on_pre_draw()/on_draw()/on_post_draw() reimplementations.
1220 // TODO it would probably be better to detect entities with
1221 // such events and make their is_drawn_at_its_position()
1222 // method return false.
1223 EntityVector entities_in_camera;
1224 Rectangle around_camera(
1225 Point(
1226 camera->get_x() - camera->get_size().width,
1227 camera->get_y() - camera->get_size().height
1228 ),
1229 camera->get_size() * 3
1230 );
1231 get_entities_in_rectangle(around_camera, entities_in_camera);
1232
1233 for (const EntityPtr& entity : entities_in_camera) {
1234 int layer = entity->get_layer();
1235 Debug::check_assertion(map.is_valid_layer(layer), "Invalid layer");
1236 if (entity->is_enabled() &&
1237 entity->is_visible()) {
1238 entities_to_draw[layer].push_back(entity);
1239 }
1240 }
1241
1242 // Add entities displayed even when out of the camera.
1243 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
1244 for (const EntityPtr& entity : entities_drawn_not_at_their_position[layer]) {
1245 if (entity->is_enabled() &&
1246 entity->is_visible()) {
1247 entities_to_draw[layer].push_back(entity);
1248 }
1249 }
1250
1251 // Sort them and remove duplicates.
1252 // Duplicate drawings are a problem for entities with semi-transparency.
1253 // Using an std::set would be slower because duplicates are rare:
1254 // there are not often a lot of dynamic entities displayed out of the camera.
1255 std::sort(entities_to_draw[layer].begin(), entities_to_draw[layer].end(), DrawingOrderComparator());
1256 entities_to_draw[layer].erase(
1257 std::unique(entities_to_draw[layer].begin(), entities_to_draw[layer].end()),
1258 entities_to_draw[layer].end()
1259 );
1260 }
1261
1262 }
1263
1264 for (int layer = map.get_min_layer(); layer <= map.get_max_layer(); ++layer) {
1265
1266 // Draw the animated tiles and the tiles that overlap them:
1267 // in other words, draw all regions containing animated tiles
1268 // (and maybe more, but we don't care because non-animated tiles
1269 // will be drawn later).
1270 for (unsigned int i = 0; i < tiles_in_animated_regions[layer].size(); i++) {
1271 Tile& tile = *tiles_in_animated_regions[layer][i];
1272 if (tile.overlaps(*camera) || !tile.is_drawn_at_its_position()) {
1273 tile.draw_on_map();
1274 }
1275 }
1276
1277 // Draw the non-animated tiles (with transparent rectangles on the regions of animated tiles
1278 // since they are already drawn).
1279 non_animated_regions[layer]->draw_on_map();
1280
1281 // Draw dynamic entities, ordered by their data structure.
1282 for (const EntityPtr& entity: entities_to_draw[layer]) {
1283 entity->draw_on_map();
1284 }
1285 }
1286
1287 if (EntityTree::debug_quadtrees) {
1288 // Draw the quadtree structure for debugging.
1289 quadtree.draw(camera_surface, -camera->get_top_left_xy());
1290 }
1291}

1295 * \param entity An entity.
1296 * \param layer The new layer.
1297 */
1298void Entities::set_entity_layer(Entity& entity, int layer) {
1299
1300 int old_layer = entity.get_layer();
1301
1302 if (layer != old_layer) {
1303
1304 const EntityPtr& shared_entity = std::static_pointer_cast<Entity>(entity.shared_from_this());
1305
1306 // Track the insertion order.
1307 z_caches.at(old_layer).remove(shared_entity);
1308 z_caches.at(layer).add(shared_entity);
1309
1310 // Update the list of entities by type and layer.
1311 const EntityType type = entity.get_type();
1312 const auto& it = entities_by_type.find(type);
1313 if (it != entities_by_type.end()) {
1314 ByLayer<EntitySet>& sets = it->second;
1315 sets[old_layer].erase(shared_entity);
1316 sets[layer].insert(shared_entity);
1317 }
1318
1319 // Update the entity after the lists because this function might be called again.
1320 entity.set_layer(layer);
1321 }

1341 * \param layer The layer to check.
1342 * \param rectangle A rectangle.
1343 * \return \c true if this rectangle overlaps a raised crystal block.

1345bool Entities::overlaps_raised_blocks(int layer, const Rectangle& rectangle) {
