228template <typename T>
229QDataStream& operator>>(QDataStream& s, QList<T>& l)
230{
231 l.clear();
232 quint32 c;
233 s >> c;
234 l.reserve(c);
235 for(quint32 i = 0; i < c; ++i)
236 {
237 T t;
238 s >> t;
239 l.append(t);
240 if (s.atEnd())
241 break;
242 }
243 return s;
244}
245
246template <typename T>
247QDataStream& operator<<(QDataStream& s, const QList<T>& l)
248{
249 s << quint32(l.size());
250 for (int i = 0; i < l.size(); ++i)
251 s << l.at(i);
252 return s;
253}

334template <class Key, class T>
335Q_OUTOFLINE_TEMPLATE QDataStream &operator>>(QDataStream &in, QHash<Key, T> &hash)
336{
337 QDataStream::Status oldStatus = in.status();
338 in.resetStatus();
339 hash.clear();
340
341 quint32 n;
342 in >> n;
343
344 for (quint32 i = 0; i < n; ++i) {
345 if (in.status() != QDataStream::Ok)
346 break;
347
348 Key k;
349 T t;
350 in >> k >> t;
351 hash.insertMulti(k, t);
352 }
353
354 if (in.status() != QDataStream::Ok)
355 hash.clear();
356 if (oldStatus != QDataStream::Ok)
357 in.setStatus(oldStatus);
358 return in;
359}
360
361template <class Key, class T>
362Q_OUTOFLINE_TEMPLATE QDataStream &operator<<(QDataStream &out, const QHash<Key, T>& hash)
363{
364 out << quint32(hash.size());
365 typename QHash<Key, T>::ConstIterator it = hash.end();
366 typename QHash<Key, T>::ConstIterator begin = hash.begin();
367 while (it != begin) {
368 --it;
369 out << it.key() << it.value();
370 }
371 return out;
372}
373#ifdef Q_QDOC
374template <class Key, class T>
375Q_OUTOFLINE_TEMPLATE QDataStream &operator>>(QDataStream &in, QMap<Key, T> &map)
376#else
377template <class aKey, class aT>
378Q_OUTOFLINE_TEMPLATE QDataStream &operator>>(QDataStream &in, QMap<aKey, aT> &map)
379#endif
380{
381 QDataStream::Status oldStatus = in.status();
382 in.resetStatus();
383 map.clear();
384
385 quint32 n;
386 in >> n;
387
388 map.detach();
389 for (quint32 i = 0; i < n; ++i) {
390 if (in.status() != QDataStream::Ok)
391 break;
392
393 aKey key;
394 aT value;
395 in >> key >> value;
396 map.insertMulti(key, value);
397 }
398 if (in.status() != QDataStream::Ok)
399 map.clear();
400 if (oldStatus != QDataStream::Ok)
401 in.setStatus(oldStatus);
402 return in;
403}
