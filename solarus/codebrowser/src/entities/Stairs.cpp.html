29/**
30 * \brief Creates a new stairs entity.
31 * \param name Name of the entity to create.
32 * \param layer Layer of the entity to create on the map.
33 * \param xy Coordinates of the entity to create.
34 * \param direction Direction of the stairs (0 to 3).
35 * \param subtype The subtype of stairs.
36 */
37Stairs::Stairs(
38 const std::string& name,
39 int layer,
40 const Point& xy,
41 int direction,
42 Subtype subtype):
43 Entity(name, direction, layer, xy, Size(16, 16)),
44 subtype(subtype) {
45
46 set_collision_modes(COLLISION_TOUCHING | COLLISION_OVERLAPPING);
47
48 if (is_inside_floor()) {
49 set_layer_independent_collisions(true);
50 }
51 else {
52 set_size(16, 8);
53 if (direction == 3) { // Down.
54 set_origin(0, -8);
55 }
56 }
57}

151/**
152 * \brief Returns the direction of the movement an entity would take when
153 * activating these stairs.
154 * \param way The way you intend to take these stairs.
155 * \return The movement direction an entity should take on these stairs
156 * (0 to 7).
157 */
158int Stairs::get_movement_direction(Way way) const {
159
160 int movement_direction = get_direction() * 2;
161 if (way == REVERSE_WAY) {
162 movement_direction = (movement_direction + 4) % 8;
163 }
164
165 return movement_direction;
166}
167
168/**
169 * \brief Returns the direction of the animation an entity should take when
170 * walking these stairs.
171 *
172 * For spiral stairs, the direction returned is diagonal.
173 *
174 * \param way The way you are taking these stairs.
175 * \return The direction of animation (0 to 7).
176 */
177int Stairs::get_animation_direction(Way way) const {
178
179 int basic_direction = get_direction() * 2;
180 int result = basic_direction;
181
182 if (subtype == SPIRAL_UPSTAIRS) {
183 result = (basic_direction == 2) ? 1 : 5;
184 }
185 else if (subtype == SPIRAL_DOWNSTAIRS) {
186 result = (basic_direction == 2) ? 3 : 7;
187 }
188
189 if (way == REVERSE_WAY) {
190 result = (result + 4) % 8;
191 }
192
193 return result;
194}

196/**
197 * \brief Plays a sound corresponding to these stairs.
198 *
199 * When an entity collides with the stairs (usually the hero),
200 * it can call this function to play the appropriate stairs sound.
201 *
202 * \param way The way you are taking these stairs.
203 */
204void Stairs::play_sound(Way way) const {
205
206 std::string sound_id;
207 if (is_inside_floor()) {
208 // Choose the sound depending on whether we are going
209 // upstairs or downstairs.
210 sound_id = (way == NORMAL_WAY) ? "stairs_up_end" : "stairs_down_end";
211 }
212 else {
213 // Choose the sound depending on whether we are on the old floor or the
214 // new floor, and going upstairs or downstairs.
215 if (subtype == SPIRAL_UPSTAIRS || subtype == STRAIGHT_UPSTAIRS) {
216 sound_id = (way == NORMAL_WAY) ? "stairs_up_start" : "stairs_down_end";
217 }
218 else {
219 sound_id = (way == NORMAL_WAY) ? "stairs_down_start" : "stairs_up_end";
220 }
221 }
222
223 if (Sound::exists(sound_id)) {
224 Sound::play(sound_id);
225 }
226}

228/**
229 * \brief Returns the path an entity should follow to take these stairs.
230 *
231 * When an entity collides with the stairs (usually the hero),
232 * it can call this function to know the path it should take to make the
233 * appropriate movement.
234 *
235 * \param way The way you are taking these stairs.
236 * \return The corresponding path to make.
237 */
238std::string Stairs::get_path(Way way) const {
239
240 // Determine the movement direction.
241 int initial_direction = get_direction() * 2;
242 std::string path = "";
243 int nb_steps;
244 if (is_inside_floor()) {
245 nb_steps = 5;
246 }
247 else {
248 nb_steps = (get_direction() == 1) ? 1 : 2;
249 }
250 for (int i = 0; i < nb_steps; i++) {
251 path += '0' + initial_direction;
252 }
253
254 if (!is_inside_floor()) {
255
256 // Second direction to take for each subtype of stairs
257 // (assuming the direction is north).
258 static const int second_directions[] = {
259 0, 4, 2, 2
260 };
261 int second_direction = second_directions[subtype];
262 if (get_direction() == 3) { // Direction south.
263 second_direction = (second_direction + 4) % 8;
264 }
265 char c = '0' + second_direction;
266 path = path + c;
267 if (subtype == SPIRAL_UPSTAIRS || subtype == SPIRAL_DOWNSTAIRS) {
268 path = path + c;
269 }
270 }
271
272 if (way == REVERSE_WAY) {
273 std::string inverse_path = "";
274 std::string::reverse_iterator it;
275 for (it = path.rbegin(); it != path.rend(); ++it) {
276 int direction = *it - '0';
277 direction = (direction + 4) % 8;
278 inverse_path += '0' + direction;
279 }
280 path = inverse_path;
281 }
282
283 return path;
284}

286/**
287 * \brief Returns the subarea in which an entity tooking these stairs can be
288 * displayed.
289 * \param way The way you are taking these stairs.
290 * \return The subarea of the map where the entity displaying should be
291 * restricted to.
292 */
293Rectangle Stairs::get_clipping_rectangle(Way /* way */) const {
294
295 if (subtype == INSIDE_FLOOR || subtype == STRAIGHT_UPSTAIRS) {
296 return Rectangle(0, 0, 0, 0); // no restriction
297 }
298
299 Rectangle clipping_rectangle(0, 0, get_map().get_width(), get_map().get_height());
300 bool north = get_direction() == 1; // north or south
301
302 // Hide the hero after the north of south edge.
303 if (north) {
304 clipping_rectangle.set_y(get_top_left_y() - 8);
305 clipping_rectangle.set_height(48);
306 }
307 else { // south
308 clipping_rectangle.set_y(get_top_left_y() - 32);
309 clipping_rectangle.set_height(48);
310 }
311
312 // Spiral staircase: also hide a side.
313 if ((subtype == SPIRAL_DOWNSTAIRS && north)
314 || (subtype == SPIRAL_UPSTAIRS && !north)) {
315 // north downstairs or south upstairs: hide the west side
316 clipping_rectangle.set_x(get_top_left_x());
317 clipping_rectangle.set_width(32);
318 }
319 else if ((subtype == SPIRAL_UPSTAIRS && north)
320 || (subtype == SPIRAL_DOWNSTAIRS && !north)) {
321 // north downstairs or south upstairs: hide the east side
322 clipping_rectangle.set_x(get_top_left_x() - 16);
323 clipping_rectangle.set_width(32);
324 }
325
326 return clipping_rectangle;
327}

329/**
330 * \brief Notifies this entity that it was just enabled or disabled.
331 * \param enabled \c true if the entity is now enabled.
332 *
333 * All dynamic tiles whose prefix is "stairsname_enabled"
334 * and "stairsame_disabled" will be updated depending on the stairs state
335 * (where stairsname is the name of the stairs).
336 */
337void Stairs::notify_enabled(bool enabled) {
338
339 Entity::notify_enabled(enabled);
340
341 update_dynamic_tiles();
342}
343
344/**
345 * \brief Enables or disables the dynamic tiles related to these stairs.
346 *
347 * The dynamic tiles impacted by this function are the ones whose prefix is
348 * the stairs's name
349 * followed by "_enabled" or "_disabled", depending on the stairs state.
350 */
351void Stairs::update_dynamic_tiles() {
352
353 std::vector<EntityPtr> tiles = get_entities().get_entities_with_prefix(
354 EntityType::DYNAMIC_TILE, get_name() + "_enabled");
355 for (const EntityPtr& tile: tiles) {
356 tile->set_enabled(is_enabled());
357 }
358
359 tiles = get_entities().get_entities_with_prefix(
360 EntityType::DYNAMIC_TILE, get_name() + "_disabled");
361 for (const EntityPtr& tile: tiles) {
362 tile->set_enabled(!is_enabled());
363 }
364}
