34/**
35 * \brief Associates to each movement direction the possible directions of the hero's sprites.
36 *
37 * This array indicates the possible directions of the hero's animation (from 0 to 3, or -1 for no change)
38 * for each movement direction.
39 * Each combination of directional keys can be associated to one or two directions.
40 * When one direction is specified (i.e. the second one is set to -1), the hero's sprite always takes
41 * this direction.
42 * When two directions are specified,
43 * the hero sprite takes the first direction, unless it is already under the second one
44 * (then it stays in the second one). This permits a natural behavior for diagonal directions.
45 */
46const int HeroSprites::animation_directions[][2] = {
47 { 0, -1}, // right
48 { 0, 1}, // right-up: right or up
49 { 1, -1}, // up
50 { 2, 1}, // left-up: left or up
51 { 2, -1}, // left
52 { 2, 3}, // left-down: left or down
53 { 3, -1}, // down
54 { 0, 3} // right-down: right or down
55};

96/**
97 * \brief Loads (or reloads) the sprites and sounds of the hero and his
98 * equipment.
99 *
100 * The sprites and sounds loaded may depend on his tunic, sword and shield as
101 * specified in the savegame.
102 * This function must be called at the game beginning
103 * and as soon as the hero's equipment is changed.
104 */
105void HeroSprites::rebuild_equipment() {
106
107 // Make the tunic sprite be the default one.
108 hero.set_default_sprite_name("tunic");
109
110 int animation_direction = -1;
111 if (tunic_sprite != nullptr) {
112 // Save the direction.
113 animation_direction = tunic_sprite->get_current_direction();
114 }
115
116 // The hero's body.
117 if (has_default_tunic_sprite) {
118 set_tunic_sprite_id(get_default_tunic_sprite_id());
119 }
120
121 // The hero's shadow.
122 if (shadow_sprite == nullptr) {
123 shadow_sprite = hero.create_sprite("entities/shadow", "shadow");
124 shadow_sprite->set_current_animation("big");
125 }
126
127 // The hero's sword.
128 if (has_default_sword_sprite) {
129 set_sword_sprite_id(get_default_sword_sprite_id());
130 }
131
132 if (has_default_sword_sound) {
133 set_sword_sound_id(get_default_sword_sound_id());
134 }
135
136 const int sword_number = equipment.get_ability(Ability::SWORD);
137 if (sword_number > 0) {
138 // TODO make this sprite depend on the sword sprite: sword_sprite_id + "_stars"
139 std::ostringstream oss;
140 oss << "hero/sword_stars" << sword_number;
141 sword_stars_sprite = hero.create_sprite(oss.str(), "sword_stars");
142 sword_stars_sprite->stop_animation();
143 }
144
145 // The hero's shield.
146 if (has_default_shield_sprite) {
147 set_shield_sprite_id(get_default_shield_sprite_id());
148 }
149
150 // The trail.
151 trail_sprite = hero.create_sprite("hero/trail", "trail");
152 trail_sprite->stop_animation();
153
154 // Restore the animation direction.
155 if (animation_direction != -1) {
156 set_animation_direction(animation_direction);
157 }
158
159 reorder_sprites();
160}

171/**
172 * \brief Sets the animation set id to use for the tunic sprite.
173 * \param sprite_id The tunic sprite animation set id.
174 */
175void HeroSprites::set_tunic_sprite_id(const std::string& sprite_id) {
176
177 if (sprite_id == this->tunic_sprite_id) {
178 return;
179 }
180
181 this->tunic_sprite_id = sprite_id;
182
183 std::string animation;
184 int direction = -1;
185 if (tunic_sprite != nullptr) {
186 // Delete the previous sprite, but save its animation and direction.
187 animation = tunic_sprite->get_current_animation();
188 direction = tunic_sprite->get_current_direction();
189 hero.remove_sprite(*tunic_sprite);
190 tunic_sprite = nullptr;
191 }
192
193 tunic_sprite = hero.create_sprite(sprite_id, "tunic");
194 tunic_sprite->enable_pixel_collisions();
195 if (!animation.empty()) {
196 set_tunic_animation(animation);
197 tunic_sprite->set_current_direction(direction);
198 }
199
200 has_default_tunic_sprite = (sprite_id == get_default_tunic_sprite_id());
201
202 // Synchronize other sprites to the new tunic sprite.
203 if (sword_sprite != nullptr) {
204 sword_sprite->set_synchronized_to(tunic_sprite);
205 }
206 if (shield_sprite != nullptr) {
207 shield_sprite->set_synchronized_to(tunic_sprite);
208 }
209
210 reorder_sprites();
211 recompute_sprites_bounding_box();
212}

238/**
239 * \brief Sets the animation set id to use for the sword sprite.
240 * \param sprite_id The sword sprite animation set id.
241 * An empty string means no sword sprite.
242 */
243void HeroSprites::set_sword_sprite_id(const std::string& sprite_id) {
244
245 if (sprite_id == this->sword_sprite_id) {
246 return;
247 }
248
249 this->sword_sprite_id = sprite_id;
250
251 std::string animation;
252 int direction = -1;
253 if (sword_sprite != nullptr) {
254 // Delete the previous sprite, but save its animation and direction.
255 if (sword_sprite->is_animation_started()) {
256 animation = sword_sprite->get_current_animation();
257 direction = sword_sprite->get_current_direction();
258 }
259 hero.remove_sprite(*sword_sprite);
260 sword_sprite = nullptr;
261 }
262
263 if (!sprite_id.empty()) {
264 // There is a sword sprite specified.
265 sword_sprite = hero.create_sprite(sprite_id, "sword");
266 sword_sprite->enable_pixel_collisions();
267 sword_sprite->set_synchronized_to(tunic_sprite);
268 if (animation.empty()) {
269 sword_sprite->stop_animation();
270 }
271 else {
272 sword_sprite->set_current_animation(animation);
273 sword_sprite->set_current_direction(direction);
274 }
275 }
276
277 has_default_sword_sprite = (sprite_id == get_default_sword_sprite_id());
278
279 reorder_sprites();
280 recompute_sprites_bounding_box();
281}

432/**
433 * \brief Makes sure that hero sprites are stored in the correct order.
434 * This function should be called whenever some sprites are created.
435 */
436void HeroSprites::reorder_sprites() {
437
438 // Shadow, tunic, trail, ground, sword, sword stars, shield.
439
440 if (shadow_sprite != nullptr) {
441 hero.bring_sprite_to_front(*shadow_sprite);
442 }
443 if (tunic_sprite != nullptr) {
444 hero.bring_sprite_to_front(*tunic_sprite);
445 }
446 if (trail_sprite != nullptr) {
447 hero.bring_sprite_to_front(*trail_sprite);
448 }
449 if (ground_sprite != nullptr) {
450 hero.bring_sprite_to_front(*ground_sprite);
451 }
452 if (sword_sprite != nullptr) {
453 hero.bring_sprite_to_front(*sword_sprite);
454 }
455 if (sword_stars_sprite != nullptr) {
456 hero.bring_sprite_to_front(*sword_stars_sprite);
457 }
458 if (shield_sprite != nullptr) {
459 hero.bring_sprite_to_front(*shield_sprite);
460 }
461}

579/**
580 * \brief Makes the hero blink for a while.
581 * \param duration Delay before stopping blinking.
582 * 0 means infinite.
583 */
584void HeroSprites::blink(uint32_t duration) {
585
586 const uint32_t blink_delay = 50;
587
588 blinking = true;
589 tunic_sprite->set_blinking(blink_delay);
590
591 if (equipment.has_ability(Ability::SHIELD)) {
592 shield_sprite->set_blinking(blink_delay);
593 }
594 if (equipment.has_ability(Ability::SWORD)) {
595 sword_sprite->set_blinking(blink_delay);
596 }
597 trail_sprite->set_blinking(blink_delay);
598
599 if (duration == 0) {
600 // No end date.
601 end_blink_date = 0;
602 }
603 else {
604 end_blink_date = System::now() + duration;
605 }
606}

676/**
677 * \brief Returns the direction that the hero's sprites should take.
678 *
679 * This function returns the direction that the hero's sprites should take depending on the direction wanted
680 * by the player and the real movement direction.
681 * For diagonal directions, the direction returned depends
682 * on the current real direction of the hero's sprites.
683 *
684 * \param keys_direction the direction defined by the combination of directional keys pressed (0 to 7),
685 * or -1 if this is not a valid direction
686 * \param real_movement_direction the direction of the hero's actual movement (may be different from keys_direction)
687 * \return the direction of the sprites corresponding to these arrows (0 to 3),
688 * or -1 if the directional keys combination is illegal
689 */
690int HeroSprites::get_animation_direction(
691 int keys_direction, int real_movement_direction) const {
692
693 int result;
694
695 if (keys_direction == -1) {
696 // the player is not pressing a valid combination of directional keys: don't change the sprite's direction
697 result = -1;
698 }
699 else if (keys_direction % 2 == 0) {
700 // the player is pressing only one of the four main directions
701 // (i.e. he is not trying to make a diagonal move, even if the real movement is diagonal)
702 // we just give the sprite this direction
703 result = keys_direction / 2;
704 }
705 // the movement is diagonal: we have to choose between two directions
706 else if (animation_directions[real_movement_direction][1] == get_animation_direction()) {
707 // we choose the second direction if it was already the sprite's direction
708 result = animation_directions[real_movement_direction][1];
709 }
710 else {
711 // otherwise we choose the first direction
712 result = animation_directions[real_movement_direction][0];
713 }
714
715 return result;
716}

718/**
719 * \brief Changes the direction of the hero's sprites.
720 *
721 * It is different from the movement direction.
722 *
723 * \param direction the direction to set (0 to 3)
724 */
725void HeroSprites::set_animation_direction(int direction) {
726
727 Debug::check_assertion(direction >= 0 && direction < 4,
728 "Invalid direction for set_animation_direction");
729
730 tunic_sprite->set_current_direction(direction);
731
732 if (is_sword_visible()) {
733 sword_sprite->set_current_direction(direction);
734 }
735
736 if (is_sword_stars_visible()) {
737 sword_stars_sprite->set_current_direction(direction);
738 }
739
740 if (is_shield_visible()) {
741 shield_sprite->set_current_direction(direction);
742 }
743
744 if (is_trail_visible()) {
745 trail_sprite->set_current_direction(direction);
746 }
747
748 if (lifted_item != nullptr) {
749 const SpritePtr& sprite = lifted_item->get_sprite();
750 sprite->restart_animation();
751 }
752}

756 *
757 * The direction specified is one of the 8 possible movement directions of the hero.
758 * The hero's sprites only have four directions, so when
759 * the specified direction is a diagonal one,
760 * one of the two closest main directions is picked.

762 * \param direction the movement direction (0 to 7)
763 */
764void HeroSprites::set_animation_direction8(int direction) {
765
766 if (get_animation_direction() != animation_directions[direction][1]) {
767 set_animation_direction(animation_directions[direction][0]);
768 }
769}

804/**
805 * \brief Updates the animation of the hero's sprites if necessary.
806 */
807void HeroSprites::update() {
808
809 // Hero sprites are all updated here, Entity::update() is overridden for the hero.
810
811 // Keep the current sprites here in case they change from a script during the operation.
812 SpritePtr tunic_sprite = this->tunic_sprite;
813 SpritePtr sword_sprite = this->sword_sprite;
814
815 // Update the frames.
816 tunic_sprite->update();
817
818 if (is_sword_visible()) {
819 sword_sprite->update();
820 sword_sprite->set_current_frame(tunic_sprite->get_current_frame());
821 hero.check_collision_with_detectors(*sword_sprite);
822 }
823 hero.check_collision_with_detectors(*tunic_sprite);
824
825 if (is_sword_stars_visible()) {
826 // The stars are not synchronized with the other sprites.
827 sword_stars_sprite->update();
828 }
829
830 if (is_shield_visible()) {
831 shield_sprite->update();
832 if (walking) {
833 shield_sprite->set_current_frame(tunic_sprite->get_current_frame());
834 }
835 }
836
837 if (is_trail_visible()) {
838 trail_sprite->update();
839 }
840
841 if (is_ground_visible()) {
842 ground_sprite->update();
843 }
844
845 // Blinking.
846 if (is_blinking()
847 && end_blink_date != 0
848 && System::now() >= end_blink_date) {
849 stop_blinking();
850 }
851
852 // Lua callback.
853 if (tunic_sprite->is_animation_finished() &&
854 !animation_callback_ref.is_empty()) {
855 animation_callback_ref.clear_and_call("hero animation callback");
856 }
857}

864 int x = hero.get_x();

905 * \brief Suspends or resumes the animation of the hero's sprites.
906 *
907 * This function is called by the map when the game is suspended or resumed.
908 *
909 * \param suspended true to suspend the hero's sprites, false to resume them
910 */
911void HeroSprites::set_suspended(bool suspended) {
912
913 tunic_sprite->set_suspended(suspended);
914
915 if (equipment.has_ability(Ability::SWORD) && sword_sprite != nullptr) {
916 sword_sprite->set_suspended(suspended);
917 sword_stars_sprite->set_suspended(suspended);
918 }
919
920 if (equipment.has_ability(Ability::SHIELD) && shield_sprite != nullptr) {
921 shield_sprite->set_suspended(suspended);
922 }
923
924 trail_sprite->set_suspended(suspended);
925
926 if (is_ground_visible()) {
927 ground_sprite->set_suspended(suspended);
928 }
929
930 // Timer.
931 uint32_t now = System::now();
932 if (suspended) {
933 when_suspended = now;
934 }
935 else if (end_blink_date != 0) {
936 end_blink_date += now - when_suspended;
937 }
938}
