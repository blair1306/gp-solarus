40/**
41 * \brief Starts this state.
42 * \param previous_state the previous state
43 */
44void Hero::PushingState::start(const State* previous_state) {
45
46 HeroState::start(previous_state);
47
48 pushing_direction4 = get_sprites().get_animation_direction();
49 get_sprites().set_animation_pushing();
50}

52/**
53 * \brief Stops this state.
54 */
55void Hero::PushingState::stop(const State* next_state) {
56
57 HeroState::stop(next_state);
58
59 if (is_moving_grabbed_entity()) {
60 get_entity().clear_movement();
61 pushed_entity->update();
62 stop_moving_pushed_entity();
63 }
64}
65
66/**
67 * \brief Updates this state.
68 */
69void Hero::PushingState::update() {
70
71 HeroState::update();
72
73 Hero& hero = get_entity();
74 if (!is_moving_grabbed_entity()) { // the hero is pushing a fixed obstacle
75
76 // stop pushing if there is no more obstacle
77 if (!hero.is_facing_obstacle()) {
78 hero.set_state(new FreeState(hero));
79 }
80
81 // stop pushing if the player changes his direction
82 else if (get_commands().get_wanted_direction8() != pushing_direction4 * 2) {
83
84 if (get_commands().is_command_pressed(GameCommand::ACTION)) {
85 hero.set_state(new GrabbingState(hero));
86 }
87 else {
88 hero.set_state(new FreeState(hero));
89 }
90 }
91
92 // see if the obstacle pushed is an entity that the hero can move
93 else {
94
95 Entity* facing_entity = hero.get_facing_entity();
96 if (facing_entity != nullptr) { // the obstacle pushed is an entity
97
98 if (facing_entity->get_type() == EntityType::BLOCK) { // it can be moved by the hero (TODO dynamic binding)
99 hero.try_snap_to_facing_entity();
100 }
101
102 if (facing_entity->start_movement_by_hero()) {
103
104 std::string path = " ";
105 path[0] = path[1] = '0' + pushing_direction4 * 2;
106
107 pushing_movement = std::make_shared<PathMovement>(
108 path, 40, false, false, false
109 );
110 hero.set_movement(pushing_movement);
111 pushed_entity = facing_entity;
112 pushed_entity->notify_moving_by(hero);
113 }
114 }
115 }
116 }
117}

150/**
151 * \brief Notifies this state that the movement if finished.
152 */
153void Hero::PushingState::notify_movement_finished() {
154
155 if (is_moving_grabbed_entity()) {
156 // the 16 pixels of the path are completed
157 pushed_entity->update();
158 stop_moving_pushed_entity();
159 }
160}

162/**
163 * \brief Notifies this state that the hero has just failed to change its
164 * position because of obstacles.
165 */
166void Hero::PushingState::notify_obstacle_reached() {
167
168 if (is_moving_grabbed_entity()) {
169 // an obstacle is reached before the 16 pixels are completed
170 pushed_entity->update();
171 stop_moving_pushed_entity();
172 }
173}

175/**
176 * \brief Notifies this state that the hero has just changed its
177 * position.
178 */
179void Hero::PushingState::notify_position_changed() {
180
181 if (is_moving_grabbed_entity()) {
182 // if the entity has made more than 8 pixels and is aligned on the grid,
183 // we stop the movement
184
185 bool horizontal = pushing_direction4 % 2 == 0;
186 bool has_reached_grid = pushing_movement->get_total_distance_covered() > 8
187 && ((horizontal && pushed_entity->is_aligned_to_grid_x())
188 || (!horizontal && pushed_entity->is_aligned_to_grid_y()));
189
190 if (has_reached_grid) {
191 stop_moving_pushed_entity();
192 }
193 }
194}

196/**
197 * \brief Makes the hero stop pushing the entity he is facing.
198 */
199void Hero::PushingState::stop_moving_pushed_entity() {
200
201 Hero& hero = get_entity();
202 if (pushed_entity != nullptr) {
203 pushed_entity->stop_movement_by_hero();
204
205 // the hero may have moved one or several pixels too much
206 // because he moved before the block, not knowing that the block would not follow him
207
208 switch (pushing_direction4) {
209
210 case 0:
211 // east
212 hero.set_x(pushed_entity->get_x() - 16);
213 break;
214
215 case 1:
216 // north
217 hero.set_y(pushed_entity->get_y() + 16);
218 break;
219
220 case 2:
221 // west
222 hero.set_x(pushed_entity->get_x() + 16);
223 break;
224
225 case 3:
226 // south
227 hero.set_y(pushed_entity->get_y() - 16);
228 break;
229 }
230
231 hero.clear_movement();
232 pushing_movement = nullptr;
233 Entity* entity_just_moved = pushed_entity;
234 pushed_entity = nullptr;
235 entity_just_moved->notify_moved_by(hero);
236 }
237
238 if (!is_current_state()) {
239 // Another state is already starting (for example TreasureState).
240 // In this case, don't overwrite it.
241 return;
242 }
243
244 if (!get_commands().is_command_pressed(GameCommand::ACTION)) {
245 // The hero was pushing an entity without grabbing it.
246
247 // Stop the animation pushing if his direction changed.
248 if (get_commands().get_wanted_direction8() != pushing_direction4 * 2) {
249 hero.set_state(new FreeState(hero));
250 }
251 }
252 else {
253 // The hero was pushing an entity and grabbing it.
254 hero.set_state(new GrabbingState(hero));
255 }
256}
