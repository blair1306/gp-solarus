38/**
39 * \brief Creates a quadtree and initializes with the given size.
40 * \param space Rectangle representing the space to create partitions of.
41 */
42template<typename T>
43Quadtree<T>::Quadtree(const Rectangle& space) :
44 elements(),
45 root(*this) {
46
47 initialize(space);
48}

50/**
51 * \brief Removes all elements of the quadtree.
52 */
53template<typename T>
54void Quadtree<T>::clear() {
55
56 elements.clear();
57 elements_outside.clear();
58 root.clear();
59}

61/**
62 * \brief Clears the quadtree and initializes it with a new size.
63 * \param space Rectangle representing the space to create partitions of.
64 */
65template<typename T>
66void Quadtree<T>::initialize(const Rectangle& space) {
67
68 clear();
69
70 // Expand the space so that it is square.
71 Rectangle square = space;
72 if (space.get_width() > space.get_height()) {
73 square.set_y(square.get_center().y - square.get_width() / 2);
74 square.set_height(square.get_width());
75 }
76 else {
77 square.set_x(square.get_center().x - square.get_height() / 2);
78 square.set_width(square.get_height());
79 }
80
81 root.initialize(square);
82}

84/**
85 * \brief Returns the space partitioned by this quadtree.
86 * \return The partitioned space.
87 */
88template<typename T>
89Rectangle Quadtree<T>::get_space() const {
90 return root.get_cell();
91}

93/**
94 * \brief Adds an element to the quadtree.
95 *
96 * It is allowed to add it outside the space delimited by the quadtree,
97 * for example if it can move inside later.
98 *
99 * \param element The element to add.
100 * \param bounding_box Bounding box of the element.
101 * \return \c true in case of success.
102 */
103template<typename T>
104bool Quadtree<T>::add(const T& element, const Rectangle& bounding_box) {
105
106 if (contains(element)) {
107 // Element already in the quadtree.
108 return false;
109 }
110
111 if (!bounding_box.overlaps(get_space())) {
112 // Out of the space of the quadtree.
113 elements_outside.insert(element);
114 }
115 else if (!root.add(element, bounding_box)) {
116 // Add failed.
117 return false;
118 }
119
120 ElementInfo info = { bounding_box };
121 elements.emplace(element, info);
122
123 return true;
124}

126/**
127 * \brief Removes an element from the quadtree.
128 * \param element The element to remove.
129 * \return \c true in case of success.
130 */
131template<typename T>
132bool Quadtree<T>::remove(const T& element) {
133
134 const auto& it = elements.find(element);
135 if (it == elements.end()) {
136 // Unknown element.
137 return false;
138 }
139
140 Rectangle box = it->second.bounding_box;
141 elements.erase(element);
142 if (elements_outside.erase(element) > 0) {
143 // It was outside the quadtree space.
144 return true;
145 }
146
147 // Normal case.
148 return root.remove(element, box);
149}
150
151/**
152 * \brief Moves the element in the quadtree.
153 *
154 * This function should be called when the position or size or the element
155 * is changed.
156 *
157 * It is allowed for an element to go to or come from outside the space of the
158 * quadtree.
159 *
160 * \param element The element to move. If the element is not in the quadtree,
161 * does nothing and returns \c false.
162 * \param bounding_box New bounding box of the element.
163 * \return \c true in case of success.
164 */
165template<typename T>
166bool Quadtree<T>::move(const T& element, const Rectangle& bounding_box) {
167
168 const auto& it = elements.find(element);
169 if (it == elements.end()) {
170 // Not in the quadtree: error.
171 return false;
172 }
173 else {
174 if (it->second.bounding_box == bounding_box) {
175 // Already in the quadtree and no change.
176 return true;
177 }
178
179 if (!remove(element)) {
180 // Failed to remove.
181 return false;
182 }
183 }
184
185 if (!add(element, bounding_box)) {
186 // Failed to add.
187 return false;
188 }
189 return true;
190}

192/**
193 * \brief Returns the total number of elements in the quadtree.
194 * \return The number of elements, including elements outside the quadtree
195 * space.
196 */
197template<typename T>
198int Quadtree<T>::get_num_elements() const {
199 return root.get_num_elements() + elements_outside.size();
200}

202/**
203 * \brief Gets the elements intersecting the given rectangle.
204 * \param region The rectangle to check.
205 * The rectangle should be entirely contained in the quadtree space.
206 * \return A list of elements intersecting the rectangle, in arbitrary order.
207 * Elements outside the quadtree space are not added there.
208 */
209template<typename T>
210std::vector<T> Quadtree<T>::get_elements(
211 const Rectangle& region
212) const {
213 std::set<T> element_set;
214 root.get_elements(region, element_set);
215 return std::vector<T>(element_set.begin(), element_set.end());
216}

218/**
219 * \brief Returns whether an element is in the quadtree.
220 * \param element The element to check.
221 * \return \c true if it is in the quadtree, even if it is
222 * outside the quadtree space.
223 */
224template<typename T>
225bool Quadtree<T>::contains(const T& element) const {
226
227 const auto& it = elements.find(element);
228 return it != elements.end();
229}

318/**
319 * \brief Adds an element to this node if its bounding box intersects it.
320 *
321 * Splits the node if necessary when the threshold is exceeded.
322 *
323 * \param element The element to add.
324 * \param bounding_box Bounding box of the element.
325 * \return \c true in case of success.
326 */
327template<typename T>
328bool Quadtree<T>::Node::add(
329 const T& element,
330 const Rectangle& bounding_box
331) {
332 if (!get_cell().overlaps(bounding_box)) {
333 // Nothing to do.
334 return false;
335 }
336
337 if (!is_split()) {
338
339 // See if it is time to split.
340 if (is_main_cell(bounding_box)) {
341 // We are the main cell of this element: it counts in the total.
342 if (get_num_elements() >= max_in_cell &&
343 get_cell_size().width > min_cell_size &&
344 get_cell_size().height > min_cell_size) {
345 split();
346 }
347 }
348 }
349
350 if (!is_split()) {
351 // Add it to the current node.
352 elements.emplace_back(element, bounding_box);
353 return true;
354 }
355
356 // Add it to children cells.
357 for (const std::unique_ptr<Node>& child : children) {
358 child->add(element, bounding_box);
359 }
360 return true;
361}
362
363/**
364 * \brief Removes an element from this node if its bounding box intersects it.
365 *
366 * Merges nodes when necessary.
367 *
368 * \param element The element to remove.
369 * \param bounding_box Bounding box of the element.
370 * \return \c true in the element was found and removed.
371 */
372template<typename T>
373bool Quadtree<T>::Node::remove(
374 const T& element,
375 const Rectangle& bounding_box
376) {
377 if (!get_cell().overlaps(bounding_box)) {
378 // Nothing to do.
379 return false;
380 }
381
382 if (!is_split()) {
383 // Remove from this cell.
384 const auto& it = std::find(elements.begin(), elements.end(), std::make_pair(element, bounding_box));
385 if (it == elements.end()) {
386 // The element was not here.
387 return false;
388 }
389 elements.erase(it);
390 return true;
391 }
392
393 // Remove from children cells.
394 bool removed = false;
395 for (const std::unique_ptr<Node>& child : children) {
396 removed |= child->remove(element, bounding_box);
397 }
398
399 if (removed &&
400 !children[0]->is_split() // We are the parent node of where the element was removed.
401 ) {
402 // See if it is time to merge.
403 int num_elements_in_children = get_num_elements();
404 if (num_elements_in_children < min_in_4_cells) {
405 merge();
406 }
407 }
408 return removed;
409}

411/**
412 * \brief Returns whether this node is split or is a leaf cell.
413 * \return \c true if the node is split.
414 */
415template<typename T>
416bool Quadtree<T>::Node::is_split() const {
417
418 return children[0] != nullptr;
419}
420
421/**
422 * \brief Splits this cell in four parts and moves its elements to them.
423 */
424template<typename T>
425void Quadtree<T>::Node::split() {
426
427 Debug::check_assertion(!is_split(), "Quadtree node already split");
428
429 // Create 4 children cells.
430 const Rectangle& cell = get_cell();
431 const Point& center = cell.get_center();
432 children[0] = std::unique_ptr<Node>(
433 new Node(quadtree, Rectangle(cell.get_top_left(), center))
434 );
435 children[1] = std::unique_ptr<Node>(
436 new Node(quadtree, Rectangle(Point(center.x, cell.get_top()), Point(cell.get_right(), center.y)))
437 );
438 children[2] = std::unique_ptr<Node>(
439 new Node(quadtree, Rectangle(Point(cell.get_left(), center.y), Point(center.x, cell.get_bottom())))
440 );
441 children[3] = std::unique_ptr<Node>(
442 new Node(quadtree, Rectangle(center, cell.get_bottom_right()))
443 );
444
445 // Move existing elements into them.
446 for (const std::pair<T, Rectangle>& pair : elements) {
447 for (const std::unique_ptr<Node>& child : children) {
448 child->add(pair.first, pair.second);
449 }
450 }
451 elements.clear();
452
453 Debug::check_assertion(is_split(), "Quadtree node split failed");
454}

456/**
457 * \brief Merges the four children cell into this one and destroys them.
458 *
459 * The children must already be leaves.
460 */
461template<typename T>
462void Quadtree<T>::Node::merge() {
463
464 Debug::check_assertion(is_split(), "Quadtree node already merged");
465
466 // We want to avoid duplicates while preserving a deterministic order.
467 std::set<T> merged_elements;
468 for (const std::unique_ptr<Node>& child : children) {
469 Debug::check_assertion(!child->is_split(), "Quadtree node child is not a leaf");
470 for (const std::pair<T, Rectangle>& pair : child->elements) {
471 const T& element = pair.first;
472 if (merged_elements.insert(element).second) {
473 elements.push_back(pair);
474 }
475 }
476 }
477
478 std::fill(std::begin(children), std::end(children), nullptr);
479
480 Debug::check_assertion(!is_split(), "Quadtree node merge failed");
481}

483/**
484 * \brief Returns whether this cell contains a box and is also its main cell.
485 *
486 * The main cell is used to ensure uniqueness, for example when counting
487 * elements.
488 */
489template<typename T>
490bool Quadtree<T>::Node::is_main_cell(const Rectangle& bounding_box) const {
491
492 if (!get_cell().overlaps(bounding_box)) {
493 // Not overlapping this cell.
494 return false;
495 }
496
497 // The bounding box is in this cell. See if this is the main cell.
498 Point center = bounding_box.get_center();
499
500 // Clamp the center to the quadtree space,
501 // in case the center it actually outside.
502 const Rectangle& quadtree_space = quadtree.get_space();
503 center = {
504 std::max(quadtree_space.get_left(), std::min(quadtree_space.get_right() - 1, center.x)),
505 std::max(quadtree_space.get_top(), std::min(quadtree_space.get_bottom() - 1, center.y))
506 };
507
508 Debug::check_assertion(quadtree_space.contains(center), "Wrong center position");
509
510 return get_cell().contains(center);
511}

513/**
514 * \brief Returns the number of elements whose center is under this node.
515 * \return The number of elements under this node.
516 */
517template<typename T>
518int Quadtree<T>::Node::get_num_elements() const {
519
520 int num_elements = 0;
521 if (!is_split()) {
522 // Some elements can overlap several cells.
523 // To avoid duplicates, we count an element if this cell is its main cell.
524 // TODO This information could be stored for better performance.
525 for (const std::pair<T, Rectangle>& pair : elements) {
526 const Rectangle& box = pair.second;
527 if (is_main_cell(box)) {
528 ++num_elements;
529 }
530 }
531 }
532 else {
533 // Ask children.
534 for (const std::unique_ptr<Node>& child : children) {
535 num_elements += child->get_num_elements();
536 }
537 }
538 return num_elements;
539}

541/**
542 * \brief Gets the elements intersecting the given rectangle under this node.
543 * \param[in] region The rectangle to check.
544 * \param[in/out] result A list that will be filled with elements.
545 */
546template<typename T>
547void Quadtree<T>::Node::get_elements(
548 const Rectangle& region,
549 std::set<T>& result
550) const {
551
552 if (!get_cell().overlaps(region)) {
553 // Nothing here.
554 return;
555 }
556
557 if (!is_split()) {
558 for (const std::pair<T, Rectangle>& pair : elements) {
559 if (pair.second.overlaps(region)) {
560 result.insert(pair.first);
561 }
562 }
563 }
564 else {
565 // Get from from children cells.
566 for (const std::unique_ptr<Node>& child : children) {
567 child->get_elements(region, result);
568 }
569 }
570}

572/**
573 * \brief Draws the node on a surface for debugging purposes.
574 * \param dst_surface The destination surface.
575 * \param dst_position Where to draw on that surface.
576 */
577template<typename T>
578void Quadtree<T>::Node::draw(const SurfacePtr& dst_surface, const Point& dst_position) {
579
580 if (!is_split()) {
581 // Draw the rectangle of the node.
582 draw_rectangle(get_cell(), color, dst_surface, dst_position);
583
584 // Draw bounding boxes of elements.
585 for (const std::pair<T, Rectangle>& pair : elements) {
586 const Rectangle& bounding_box = pair.second;
587 if (is_main_cell(bounding_box)) {
588 draw_rectangle(bounding_box, color, dst_surface, dst_position);
589 }
590 }
591 }
592 else {
593 // Draw children nodes.
594 for (const std::unique_ptr<Node>& child : children) {
595 child->draw(dst_surface, dst_position);
596 }
597 }
598}

600/**
601 * \brief Draws the border of a rectangle on a surface for debugging purposes.
602 * \param rectangle The rectangle to draw.
603 * \param line_color The color to use.
604 * \param dst_surface The destination surface.
605 * \param dst_position Where to draw on that surface.
606 */
607template<typename T>
608void Quadtree<T>::Node::draw_rectangle(
609 const Rectangle& rectangle,
610 const Color& line_color,
611 const SurfacePtr& dst_surface,
612 const Point& dst_position
613) {
614 // TODO remove this function when the draw line API is available
615
616 Rectangle where = rectangle;
617 where.set_xy(where.get_xy() + dst_position);
618 dst_surface->fill_with_color(line_color, Rectangle(
619 where.get_top_left(), Size(where.get_width(), 1)
620 ));
621 dst_surface->fill_with_color(line_color, Rectangle(
622 where.get_bottom_left() + Point(0, -1), Size(where.get_width(), 1)
623 ));
624 dst_surface->fill_with_color(line_color, Rectangle(
625 where.get_top_left(), Size(1, where.get_height())
626 ));
627 dst_surface->fill_with_color(line_color, Rectangle(
628 where.get_top_right() + Point(-1, 0), Size(1, where.get_height())
629 ));
630}
