18#define SOLARUS_HERO_FREE_STATE_H

20#include "solarus/hero/PlayerMovementState.h"
21#include <cstdint>
22
23namespace Solarus {
24
25/**
26 * \brief State of the hero where he can walk normally and interact with entities.
27 */
28class Hero::FreeState: public Hero::PlayerMovementState {
29
30 public:
31
32 explicit FreeState(Hero& hero);
33
34 virtual void start(const State* previous_state) override;
35 virtual void stop(const State* next_state) override;
36 virtual void update() override;
37 virtual void set_suspended(bool suspended) override;
38 virtual void notify_action_command_pressed() override;
39 virtual void notify_obstacle_reached() override;
40
41 virtual bool is_free() const override;
42 virtual bool can_start_sword() const override;
43 virtual bool can_start_item(EquipmentItem& item) const override;
44 virtual bool can_take_stairs() const override;
45 virtual CarriedObject::Behavior get_previous_carried_object_behavior() const override;
46 virtual void set_animation_stopped() override;
47 virtual void set_animation_walking() override;
48
49 private:
50
51 int pushing_direction4; /**< direction where the hero is trying to
52 * push an obstacle (0 to 3) or -1 */
53 uint32_t start_pushing_date; /**< date when the state pushing starts */
54
55};
56
57}
