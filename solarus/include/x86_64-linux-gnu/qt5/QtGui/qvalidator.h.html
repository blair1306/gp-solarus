57class QValidatorPrivate;
58
59class Q_GUI_EXPORT QValidator : public QObject
60{
61 Q_OBJECT
62public:
63 explicit QValidator(QObject * parent = 0);
64 ~QValidator();
65
66 enum State {
67 Invalid,
68 Intermediate,
69 Acceptable
70 };
71
72 void setLocale(const QLocale &locale);
73 QLocale locale() const;
74
75 virtual State validate(QString &, int &) const = 0;
76 virtual void fixup(QString &) const;
77
78Q_SIGNALS:
79 void changed();
80
81protected:
82 QValidator(QObjectPrivate &d, QObject *parent);
83 QValidator(QValidatorPrivate &d, QObject *parent);
84
85private:
86 Q_DISABLE_COPY(QValidator)
87 Q_DECLARE_PRIVATE(QValidator)
88};
89
90class Q_GUI_EXPORT QIntValidator : public QValidator
91{
92 Q_OBJECT
93 Q_PROPERTY(int bottom READ bottom WRITE setBottom NOTIFY bottomChanged)
94 Q_PROPERTY(int top READ top WRITE setTop NOTIFY topChanged)
95
96public:
97 explicit QIntValidator(QObject * parent = 0);
98 QIntValidator(int bottom, int top, QObject *parent = 0);
99 ~QIntValidator();
100
101 QValidator::State validate(QString &, int &) const;
102 void fixup(QString &input) const;
103
104 void setBottom(int);
105 void setTop(int);
106 virtual void setRange(int bottom, int top);
107
108 int bottom() const { return b; }
109 int top() const { return t; }
110Q_SIGNALS:
111 void bottomChanged(int bottom);
112 void topChanged(int top);
113
114private:
115 Q_DISABLE_COPY(QIntValidator)
116
117 int b;
118 int t;
119};

123class QDoubleValidatorPrivate;
124
125class Q_GUI_EXPORT QDoubleValidator : public QValidator
126{
127 Q_OBJECT
128 Q_PROPERTY(double bottom READ bottom WRITE setBottom NOTIFY bottomChanged)
129 Q_PROPERTY(double top READ top WRITE setTop NOTIFY topChanged)
130 Q_PROPERTY(int decimals READ decimals WRITE setDecimals NOTIFY decimalsChanged)
131 Q_ENUMS(Notation)
132 Q_PROPERTY(Notation notation READ notation WRITE setNotation NOTIFY notationChanged)
133
134public:
135 explicit QDoubleValidator(QObject * parent = 0);
136 QDoubleValidator(double bottom, double top, int decimals, QObject *parent = 0);
137 ~QDoubleValidator();
138
139 enum Notation {
140 StandardNotation,
141 ScientificNotation
142 };
143 QValidator::State validate(QString &, int &) const;
144
145 virtual void setRange(double bottom, double top, int decimals = 0);
146 void setBottom(double);
147 void setTop(double);
148 void setDecimals(int);
149 void setNotation(Notation);
150
151 double bottom() const { return b; }
152 double top() const { return t; }
153 int decimals() const { return dec; }
154 Notation notation() const;
155
156Q_SIGNALS:
157 void bottomChanged(double bottom);
158 void topChanged(double top);
159 void decimalsChanged(int decimals);
160 void notationChanged(QDoubleValidator::Notation notation);
161
162private:
163 Q_DECLARE_PRIVATE(QDoubleValidator)
164 Q_DISABLE_COPY(QDoubleValidator)
165
166 double b;
167 double t;
168 int dec;
169};
