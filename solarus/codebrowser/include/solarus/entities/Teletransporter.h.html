17#ifndef SOLARUS_TELETRANSPORTER_H
18#define SOLARUS_TELETRANSPORTER_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/Transition.h"
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief A detector that sends the hero to a destination.
29 *
30 * A teletransporter is a detector placed on a map.
31 * When the hero walks on this detector, he is placed on a destination
32 * on the same map or another one.
33 */
34class Teletransporter: public Entity {
35
36 public:
37
38 static constexpr EntityType ThisType = EntityType::TELETRANSPORTER;
39
40 Teletransporter(
41 const std::string& name,
42 int layer,
43 const Point& xy,
44 const Size& size,
45 const std::string& sprite_name,
46 const std::string& sound_id,
47 Transition::Style transition_style,
48 const std::string& destination_map_id,
49 const std::string& destination_name
50 );
51
52 virtual EntityType get_type() const override;
53 virtual void notify_creating() override;
54
55 const std::string& get_sound_id() const;
56 void set_sound_id(const std::string& sound_id);
57 Transition::Style get_transition_style() const;
58 void set_transition_style(Transition::Style transition_style);
59 const std::string& get_destination_map_id() const;
60 void set_destination_map_id(const std::string& map_id);
61 const std::string& get_destination_name() const;
62 void set_destination_name(const std::string& destination_name);
63 bool is_on_map_side() const;
64
65 virtual bool is_obstacle_for(Entity& other) override;
66 virtual bool test_collision_custom(Entity& entity) override;
67 virtual void notify_collision(
68 Entity& entity_overlapping,
69 CollisionMode collision_mode
70 ) override;
71 void transport_hero(Hero& hero);
72
73 private:
74
75 std::string sound_id; /**< Sound played when this teletransporter is used
76 * (an empty string means no sound). */
77 Transition::Style transition_style; /**< Style of transition between the two maps. */
78 std::string destination_map_id; /**< Id of the destination map. */
79 std::string destination_name; /**< Destination on that map, or "_same" to keep the hero's coordinates,
80 * or "_side" to place the hero on the appropriate side of the map.
81 * An empty string means the default destination entity of the map. */
82 int destination_side; /**< When the destination is "_side", indicates which side
83 * of the destination map this teletransporters leads to.
84 * This depends on the teletransporter position on the original map.
85 * -1 means a teletransporter other than "_side". */
86 int transition_direction; /**< When the destination is "_side", indicates the direction
87 * of the transition between the two maps (this is the opposite
88 * direction of destination_side). */
89 bool transporting_hero; /**< Whether the hero is currently being transported
90 * by this teletransporter. */
91
92};
93
94}
95
96#endif
