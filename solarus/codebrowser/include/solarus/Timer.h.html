17#ifndef SOLARUS_TIMER_H
18#define SOLARUS_TIMER_H
19
20#include "solarus/Common.h"
21#include "solarus/lua/ExportableToLua.h"
22#include <string>
23
24namespace Solarus {
25
26/**
27 * \brief Represents a timer that any class can start.
28 *
29 * Timers are mostly used by the scripts.
30 */
31class Timer: public ExportableToLua {
32
33 public:
34
35 explicit Timer(uint32_t duration);
36
37 bool is_with_sound() const;
38 void set_with_sound(bool with_sound);
39 bool is_suspended() const;
40 void set_suspended(bool suspended);
41 bool is_suspended_with_map() const;
42 void set_suspended_with_map(bool suspend_with_map);
43 bool is_finished() const;
44 uint32_t get_initial_duration() const;
45 uint32_t get_expiration_date() const;
46 void set_expiration_date(uint32_t expiration_date);
47
48 void update();
49 void notify_map_suspended(bool suspended);
50
51 virtual const std::string& get_lua_type_name() const override;
52
53 private:
54
55 uint32_t expiration_date; /**< Date when the timer is finished. */
56 uint32_t duration; /**< Initial duration of this timer. */
57 bool finished; /**< Whether the timer is finished. */
58 bool suspended_with_map; /**< Whether the timer should be suspended when the map is. */
59 bool suspended; /**< Whether the timer is suspended. */
60 uint32_t when_suspended; /**< Date when the timer was suspended. */
61 uint32_t next_sound_date; /**< Date when the next clock sound effect is played. */
62
63};
64
65}
66
67#endif
