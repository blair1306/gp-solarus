30/**
31 * \brief Function called by Lua when parsing the quest{ } part of the file.
32 */
33int l_quest(lua_State* l) {
34
35 return LuaTools::exception_boundary_handle(l, [&] {
36
37 QuestProperties& properties = *(static_cast<QuestProperties*>(
38 lua_touserdata(l, lua_upvalueindex(1))
39 ));
40
41 // Retrieve the quest properties from the table parameter.
42 LuaTools::check_type(l, 1, LUA_TTABLE);
43 const std::string& solarus_version =
44 LuaTools::check_string_field(l, 1, "solarus_version");
45 const std::string& quest_write_dir =
46 LuaTools::opt_string_field(l, 1, "write_dir", "");
47
48 // Accept the old title_bar value (1.4 and before) as a quest title
49 // so that the quest.dat file of 1.4 quests and earlier
50 // is still recognized, even though the quest cannot be run.
51 const std::string& title_bar =
52 LuaTools::opt_string_field(l, 1, "title_bar", "");
53 const std::string& title =
54 LuaTools::opt_string_field(l, 1, "title", title_bar);
55
56 const std::string& short_description =
57 LuaTools::opt_string_field(l, 1, "short_description", "");
58 std::string long_description =
59 LuaTools::opt_string_field(l, 1, "long_description", "");
60 long_description = LuaData::unescape_multiline_string(long_description);
61 const std::string& author =
62 LuaTools::opt_string_field(l, 1, "author", "");
63 const std::string& quest_version =
64 LuaTools::opt_string_field(l, 1, "quest_version", "");
65 const std::string& release_date =
66 LuaTools::opt_string_field(l, 1, "release_date", "");
67 const std::string& website =
68 LuaTools::opt_string_field(l, 1, "website", "");
69
70 const std::string& normal_quest_size_string =
71 LuaTools::opt_string_field(l, 1, "normal_quest_size", "320x240");
72 const std::string& min_quest_size_string =
73 LuaTools::opt_string_field(l, 1, "min_quest_size", normal_quest_size_string);
74 const std::string& max_quest_size_string =
75 LuaTools::opt_string_field(l, 1, "max_quest_size", normal_quest_size_string);
76
77 properties.set_solarus_version(solarus_version);
78 properties.set_quest_write_dir(quest_write_dir);
79 properties.set_title(title);
80 properties.set_short_description(short_description);
81 properties.set_long_description(long_description);
82 properties.set_author(author);
83 properties.set_website(website);
84 properties.set_quest_version(quest_version);
85 properties.set_release_date(release_date);
86
87 Size normal_quest_size, min_quest_size, max_quest_size;
88 bool success = Video::parse_size(normal_quest_size_string, normal_quest_size);
89 if (!success) {
90 LuaTools::arg_error(l, 1, std::string(
91 "Bad field 'normal_quest_size' (not a valid size string: '")
92 + normal_quest_size_string + "')");
93 }
94
95 success = Video::parse_size(min_quest_size_string, min_quest_size);
96 if (!success) {
97 LuaTools::arg_error(l, 1, std::string(
98 "Bad field 'min_quest_size' (not a valid size string: '")
99 + min_quest_size_string + "')");
100 }
101
102 success = Video::parse_size(max_quest_size_string, max_quest_size);
103 if (!success) {
104 LuaTools::arg_error(l, 1, std::string(
105 "Bad field 'max_quest_size' (not a valid size string: '")
106 + max_quest_size_string + "')");
107 }
108
109 if (normal_quest_size.width < min_quest_size.width
110 || normal_quest_size.height < min_quest_size.height
111 || normal_quest_size.width > max_quest_size.width
112 || normal_quest_size.height > max_quest_size.height) {
113 LuaTools::arg_error(l, 1, "Invalid range of quest sizes");
114 }
115
116 properties.set_normal_quest_size(normal_quest_size);
117 properties.set_min_quest_size(min_quest_size);
118 properties.set_max_quest_size(max_quest_size);
119
120 return 0;
121 });
122}

132/**
133 * \copydoc LuaData::import_from_lua
134 */
135bool QuestProperties::import_from_lua(lua_State* l) {
136
137 lua_pushlightuserdata(l, this);
138 lua_pushcclosure(l, l_quest, 1);
139 lua_setglobal(l, "quest");
140 if (lua_pcall(l, 0, 0, 0) != 0) {
141 // Runtime error in quest.dat.
142 Debug::error(std::string("Failed to parse quest.dat: ") + lua_tostring(l, -1));
143 return false;
144 }
145
146 return true;
147}

149/**
150 * \copydoc LuaData::export_to_lua
151 */
152bool QuestProperties::export_to_lua(std::ostream& out) const {
153
154 out << "quest{\n"
155 << " solarus_version = \"" << solarus_version << "\",\n"
156 << " write_dir = \"" << escape_string(quest_write_dir) << "\",\n"
157 << " title = \"" << escape_string(title) << "\",\n"
158 << " short_description = \"" << escape_string(short_description) << "\",\n"
159 << " long_description = [[\n" << escape_multiline_string(long_description) << "]],\n"
160 << " author = \"" << escape_string(author) << "\",\n"
161 << " quest_version = \"" << escape_string(quest_version) << "\",\n"
162 << " release_date = \"" << escape_string(release_date) << "\",\n"
163 << " website = \"" << escape_string(website) << "\",\n"
164 << " normal_quest_size = \"" << normal_quest_size.width << 'x' << normal_quest_size.height << "\",\n"
165 << " min_quest_size = \"" << min_quest_size.width << 'x' << min_quest_size.height << "\",\n"
166 << " max_quest_size = \"" << max_quest_size.width << 'x' << max_quest_size.height << "\",\n"
167 << "}\n\n";
168
169 return true;
170}
