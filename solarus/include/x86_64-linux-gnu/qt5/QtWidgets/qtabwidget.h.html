42#ifndef QTABWIDGET_H
43#define QTABWIDGET_H
44
45#include <QtWidgets/qwidget.h>
46#include <QtGui/qicon.h>
47
48QT_BEGIN_NAMESPACE
49
50
51#ifndef QT_NO_TABWIDGET
52
53class QTabBar;
54class QTabWidgetPrivate;
55class QStyleOptionTabWidgetFrame;
56
57class Q_WIDGETS_EXPORT QTabWidget : public QWidget
58{
59 Q_OBJECT
60 Q_ENUMS(TabPosition TabShape)
61 Q_PROPERTY(TabPosition tabPosition READ tabPosition WRITE setTabPosition)
62 Q_PROPERTY(TabShape tabShape READ tabShape WRITE setTabShape)
63 Q_PROPERTY(int currentIndex READ currentIndex WRITE setCurrentIndex NOTIFY currentChanged)
64 Q_PROPERTY(int count READ count)
65 Q_PROPERTY(QSize iconSize READ iconSize WRITE setIconSize)
66 Q_PROPERTY(Qt::TextElideMode elideMode READ elideMode WRITE setElideMode)
67 Q_PROPERTY(bool usesScrollButtons READ usesScrollButtons WRITE setUsesScrollButtons)
68 Q_PROPERTY(bool documentMode READ documentMode WRITE setDocumentMode)
69 Q_PROPERTY(bool tabsClosable READ tabsClosable WRITE setTabsClosable)
70 Q_PROPERTY(bool movable READ isMovable WRITE setMovable)
71
72public:
73 explicit QTabWidget(QWidget *parent = 0);
74 ~QTabWidget();
75
76 int addTab(QWidget *widget, const QString &);
77 int addTab(QWidget *widget, const QIcon& icon, const QString &label);
78
79 int insertTab(int index, QWidget *widget, const QString &);
80 int insertTab(int index, QWidget *widget, const QIcon& icon, const QString &label);
81
82 void removeTab(int index);
83
84 bool isTabEnabled(int index) const;
85 void setTabEnabled(int index, bool);
86
87 QString tabText(int index) const;
88 void setTabText(int index, const QString &);
89
90 QIcon tabIcon(int index) const;
91 void setTabIcon(int index, const QIcon & icon);
92
93#ifndef QT_NO_TOOLTIP
94 void setTabToolTip(int index, const QString & tip);
95 QString tabToolTip(int index) const;
96#endif
97
98#ifndef QT_NO_WHATSTHIS
99 void setTabWhatsThis(int index, const QString &text);
100 QString tabWhatsThis(int index) const;
101#endif
102
103 int currentIndex() const;
104 QWidget *currentWidget() const;
105 QWidget *widget(int index) const;
106 int indexOf(QWidget *widget) const;
107 int count() const;
108
109 enum TabPosition { North, South, West, East };
110 TabPosition tabPosition() const;
111 void setTabPosition(TabPosition);
112
113 bool tabsClosable() const;
114 void setTabsClosable(bool closeable);
115
116 bool isMovable() const;
117 void setMovable(bool movable);
118
119 enum TabShape { Rounded, Triangular };
120 TabShape tabShape() const;
121 void setTabShape(TabShape s);
122
123 QSize sizeHint() const;
124 QSize minimumSizeHint() const;
125 int heightForWidth(int width) const;
126 bool hasHeightForWidth() const;
127
128 void setCornerWidget(QWidget * w, Qt::Corner corner = Qt::TopRightCorner);
129 QWidget * cornerWidget(Qt::Corner corner = Qt::TopRightCorner) const;
130
131 Qt::TextElideMode elideMode() const;
132 void setElideMode(Qt::TextElideMode);
133
134 QSize iconSize() const;
135 void setIconSize(const QSize &size);
136
137 bool usesScrollButtons() const;
138 void setUsesScrollButtons(bool useButtons);
139
140 bool documentMode() const;
141 void setDocumentMode(bool set);
142
143 void clear();
144
145 QTabBar* tabBar() const;
146
147public Q_SLOTS:
148 void setCurrentIndex(int index);
149 void setCurrentWidget(QWidget *widget);
150
151Q_SIGNALS:
152 void currentChanged(int index);
153 void tabCloseRequested(int index);
154 void tabBarClicked(int index);
155 void tabBarDoubleClicked(int index);
156
157protected:
158 virtual void tabInserted(int index);
159 virtual void tabRemoved(int index);
160
161 void showEvent(QShowEvent *);
162 void resizeEvent(QResizeEvent *);
163 void keyPressEvent(QKeyEvent *);
164 void paintEvent(QPaintEvent *);
165 void setTabBar(QTabBar *);
166 void changeEvent(QEvent *);
167 bool event(QEvent *);
168 void initStyleOption(QStyleOptionTabWidgetFrame *option) const;
169
170
171private:
172 Q_DECLARE_PRIVATE(QTabWidget)
173 Q_DISABLE_COPY(QTabWidget)
174 Q_PRIVATE_SLOT(d_func(), void _q_showTab(int))
175 Q_PRIVATE_SLOT(d_func(), void _q_removeTab(int))
176 Q_PRIVATE_SLOT(d_func(), void _q_tabMoved(int, int))
177 void setUpLayout(bool = false);
178};
179
180#endif // QT_NO_TABWIDGET

182QT_END_NAMESPACE
