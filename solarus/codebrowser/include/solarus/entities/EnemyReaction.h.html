17#ifndef SOLARUS_ENEMY_REACTION_H
18#define SOLARUS_ENEMY_REACTION_H
19
20#include "solarus/Common.h"
21#include "solarus/EnumInfo.h"
22#include <map>
23#include <string>
24
25namespace Solarus {
26
27class Sprite;
28
29/**
30 * \brief Describes how an enemy reacts when it receives an attack.
31 *
32 * The reaction may be different between different sprites of the enemy.
33 */
34class SOLARUS_API EnemyReaction {
35
36 public:
37
38 /**
39 * \brief Types of reactions to an attack.
40 */
41 enum class ReactionType {
42 HURT, /**< the enemy is hurt and loses some life points */
43 IGNORED, /**< nothing happens */
44 PROTECTED, /**< the attack is stopped */
45 IMMOBILIZED, /**< the enemy is temporarily immobilized */
46 CUSTOM /**< the enemy's script decides what to do */
47 };
48
49 /**
50 * \brief Represents a reaction to an attack.
51 */
52 struct Reaction {
53 ReactionType type; /**< type of reaction */
54 int life_lost; /**< number of life points lost (possibly zero) */
55
56 Reaction() :
57 type(ReactionType::IGNORED),
58 life_lost(0) {
59 }
60 };
61
62 EnemyReaction();
63
64 void set_default_reaction();
65 void set_general_reaction(ReactionType reaction, int life_lost = 0);
66 void set_sprite_reaction(const Sprite* sprite, ReactionType reaction, int life_lost = 0);
67 const Reaction& get_reaction(const Sprite* sprite) const;
68
69 private:
70
71 Reaction general_reaction; /**< reaction to make unless sprite-specific override */
72 std::map<const Sprite*, Reaction> sprite_reactions; /**< sprite-specific reaction (overrides the default one) */
73
74};
75
76template <>
77struct SOLARUS_API EnumInfoTraits<EnemyReaction::ReactionType> {
78 static const std::string pretty_name;
79
80 static const EnumInfo<EnemyReaction::ReactionType>::names_type names;
81};
82
83}
84
85#endif
