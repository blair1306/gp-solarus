35/**
36 * \brief Constructor.
37 * \param hero The hero controlled by this state.
38 * \param stairs The stairs to take.
39 * \param way The way you are taking the stairs.
40 */
41Hero::StairsState::StairsState(
42 Hero& hero,
43 const std::shared_ptr<const Stairs>& stairs,
44 Stairs::Way way
45):
46 HeroState(hero, "stairs"),
47 stairs(stairs),
48 way(way),
49 phase(0),
50 next_phase_date(0),
51 carried_object(nullptr) {
52
53 if (get_previous_carried_object_behavior() == CarriedObject::BEHAVIOR_KEEP) {
54 // Keep the carried object of the previous state.
55 carried_object = hero.get_carried_object();
56 }
57}

59/**
60 * \brief Changes the map.
61 * \param map the new map
62 */
63void Hero::StairsState::set_map(Map& map) {
64
65 HeroState::set_map(map);
66
67 // the hero may go to another map while taking stairs and carrying an item
68 if (carried_object != nullptr) {
69 carried_object->set_map(map);
70 }
71}

73/**
74 * \brief Starts this state.
75 * \param previous_state the previous state
76 */
77void Hero::StairsState::start(const State* previous_state) {
78
79 HeroState::start(previous_state);
80
81 // movement
82 int speed = stairs->is_inside_floor() ? 40 : 24;
83 std::string path = stairs->get_path(way);
84 std::shared_ptr<PathMovement> movement = std::make_shared<PathMovement>(
85 path, speed, false, true, false
86 );
87
88 // sprites and sound
89 HeroSprites& sprites = get_sprites();
90 if (carried_object == nullptr) {
91 sprites.set_animation_walking_normal();
92 }
93 else {
94 sprites.set_lifted_item(carried_object);
95 sprites.set_animation_walking_carrying();
96 }
97 sprites.set_animation_direction((path[0] - '0') / 2);
98 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
99
100 Hero& hero = get_entity();
101 if (stairs->is_inside_floor()) {
102 if (way == Stairs::NORMAL_WAY) {
103 // Toward a higher layer: change the layer now.
104 int layer = stairs->get_layer();
105 Debug::check_assertion(get_map().is_valid_layer(layer), "Invalid stairs layer");
106 get_entities().set_entity_layer(hero, layer + 1);
107 }
108 }
109 else {
110 sprites.set_clipping_rectangle(stairs->get_clipping_rectangle(way));
111 if (way == Stairs::REVERSE_WAY) {
112 Point dxy = movement->get_xy_change();
113 int fix_y = 8;
114 if (path[path.size() - 1] == '2') {
115 fix_y *= -1;
116 }
117 hero.set_xy(hero.get_x() - dxy.x, hero.get_y() - dxy.y + fix_y);
118 }
119 }
120 hero.set_movement(movement);
121}

123/**
124 * \brief Stops this state.
125 * \param next_state the next state
126 */
127void Hero::StairsState::stop(const State* next_state) {
128
129 HeroState::stop(next_state);
130
131 if (carried_object != nullptr) {
132
133 switch (next_state->get_previous_carried_object_behavior()) {
134
135 case CarriedObject::BEHAVIOR_THROW:
136 carried_object->throw_item(get_sprites().get_animation_direction());
137 get_entities().add_entity(carried_object);
138 carried_object = nullptr;
139 get_sprites().set_lifted_item(nullptr);
140 break;
141
142 case CarriedObject::BEHAVIOR_DESTROY:
143 get_sprites().set_lifted_item(nullptr);
144 break;
145
146 case CarriedObject::BEHAVIOR_KEEP:
147 carried_object = nullptr;
148 break;
149
150 default:
151 Debug::die("Invalid carried object behavior");
152 }
153 }
154}

156/**
157 * \brief Updates this state.
158 */
159void Hero::StairsState::update() {
160
161 HeroState::update();
162
163 if (is_suspended()) {
164 return;
165 }
166
167 // first time: we play the sound and initialize
168 if (phase == 0) {
169 stairs->play_sound(way);
170 next_phase_date = System::now() + 450;
171 phase++;
172 }
173
174 // update the carried object if any
175 if (carried_object != nullptr) {
176 carried_object->update();
177 }
178
179 Hero& hero = get_entity();
180 if (stairs->is_inside_floor()) {
181
182 // inside a single floor: return to normal state as soon as the movement is finished
183 if (hero.get_movement()->is_finished()) {
184
185 if (way == Stairs::REVERSE_WAY) {
186 get_entities().set_entity_layer(hero, stairs->get_layer());
187 }
188 hero.clear_movement();
189 if (carried_object == nullptr) {
190 hero.set_state(new FreeState(hero));
191 }
192 else {
193 hero.set_state(new CarryingState(hero, carried_object));
194 }
195 }
196 }
197 else {
198 // stairs between two different floors: more complicated
199
200 HeroSprites& sprites = get_sprites();
201 if (hero.get_movement()->is_finished()) {
202 hero.clear_movement();
203
204 if (carried_object == nullptr) {
205 hero.set_state(new FreeState(hero));
206 }
207 else {
208 hero.set_state(new CarryingState(hero, carried_object));
209 }
210
211 if (way == Stairs::NORMAL_WAY) {
212 // we are on the old floor:
213 // there must be a teletransporter associated with these stairs,
214 // otherwise the hero would get stuck into the walls
215 std::shared_ptr<Teletransporter> teletransporter = hero.get_delayed_teletransporter();
216 if (teletransporter == nullptr) {
217 Logger::error("Teletransporter expected with the stairs");
218 }
219 else {
220 teletransporter->transport_hero(hero);
221 }
222 }
223 else {
224 // we are on the new floor: everything is finished
225 sprites.set_clipping_rectangle();
226 }
227 }
228 else { // movement not finished yet
229
230 uint32_t now = System::now();
231 if (now >= next_phase_date) {
232 phase++;
233 next_phase_date += 350;
234
235 // main movement direction corresponding to each animation direction while taking stairs
236 static constexpr int movement_directions[] = { 0, 0, 2, 4, 4, 4, 6, 0 };
237
238 int animation_direction = stairs->get_animation_direction(way);
239 if (phase == 2) { // the first phase of the movement is finished
240 if (animation_direction % 2 != 0) {
241 // if the stairs are spiral, take a diagonal direction of animation
242 sprites.set_animation_walking_diagonal(animation_direction);
243 }
244 else {
245 // otherwise, take a usual direction
246 sprites.set_animation_direction(animation_direction / 2);
247 sprites.set_animation_walking_normal();
248 }
249 }
250 else if (phase == 3) { // the second phase of the movement (possibly diagonal) is finished
251 sprites.set_animation_walking_normal();
252
253 if (way == Stairs::NORMAL_WAY) {
254 // on the old floor, take a direction towards the next floor
255 sprites.set_animation_direction(movement_directions[animation_direction] / 2);
256 }
257 else {
258 // on the new floor, take the opposite direction from the stairs
259 sprites.set_animation_direction((stairs->get_direction() + 2) % 4);
260 }
261 }
262 }
263 }
264 }

267/**
268 * \brief Notifies this state that the game was just suspended or resumed.
269 * \param suspended true if the game is suspended
270 */
271void Hero::StairsState::set_suspended(bool suspended) {
272
273 HeroState::set_suspended(suspended);
274
275 if (carried_object != nullptr) {
276 carried_object->set_suspended(suspended);
277 }
278
279 if (!suspended) {
280 next_phase_date += System::now() - get_when_suspended();
281 }
282}

284/**
285 * \brief Returns whether the hero is touching the ground in the current state.
286 * \return \c true if the hero is touching the ground in the current state.
287 */
288bool Hero::StairsState::is_touching_ground() const {
289 // To avoid falling to the lower layer when going upstairs
290 // with INSIDE_FLOOR stairs.
291 return false;
292}
293
294/**
295 * \brief Returns whether the hero's current position can be considered
296 * as a place to come back after a bad ground (hole, deep water, etc).
297 * \return true if the hero can come back here
298 */
299bool Hero::StairsState::can_come_from_bad_ground() const {
300 return false;
301}
302
303/**
304 * \brief Returns whether the effect of teletransporters is delayed in this state.
305 *
306 * When overlapping a teletransporter, if this function returns true, the teletransporter
307 * will not be activated immediately. The state then has to activate it when it is ready.
308 *
309 * \return true if the effect of teletransporters is delayed in this state
310 */
311bool Hero::StairsState::is_teletransporter_delayed() const {
312 return true;
313}
314
315/**
316 * \brief Returns the direction of the hero's movement as defined by the controls applied by the player
317 * and the movements allowed is the current state.
318 *
319 * If he is not moving, -1 is returned.
320 * This direction may be different from the real movement direction because of obstacles.
321 *
322 * \return the hero's wanted direction between 0 and 7, or -1 if he is stopped
323 */
324int Hero::StairsState::get_wanted_movement_direction8() const {
325 return get_sprites().get_animation_direction8();
326}

328/**
329 * \brief Returns the item currently carried by the hero in this state, if any.
330 * \return the item carried by the hero, or nullptr
331 */
332std::shared_ptr<CarriedObject> Hero::StairsState::get_carried_object() const {
333 return carried_object;
334}
335
336/**
337 * \copydoc Entity::State::get_previous_carried_object_behavior
338 */
339CarriedObject::Behavior Hero::StairsState::get_previous_carried_object_behavior() const {
340
341 if (stairs->is_inside_floor()) {
342 return CarriedObject::BEHAVIOR_KEEP;
343 }
344 return CarriedObject::BEHAVIOR_DESTROY;
345}

347/**
348 * \brief Notifies this state that the layer has changed.
349 */
350void Hero::StairsState::notify_layer_changed() {
351
352 if (carried_object != nullptr) {
353 carried_object->set_layer(get_entity().get_layer());
354 }
