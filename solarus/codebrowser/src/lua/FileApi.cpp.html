23/**
24 * Name of the Lua table representing the file module.
25 */
26const std::string LuaContext::file_module_name = "sol.file";
27
28/**
29 * \brief Initializes the file features provided to Lua.
30 */
31void LuaContext::register_file_module() {
32
33 static const luaL_Reg functions[] = {
34 { "open", file_api_open },
35 { "exists", file_api_exists },
36 { "remove", file_api_remove },
37 { "mkdir", file_api_mkdir },
38 { nullptr, nullptr }
39 };
40 register_functions(file_module_name, functions);
41
42 // Store the original io.open function in the registry.
43 // We will need to access it from sol.file.open().
44 // --
45 lua_getglobal(l, "io");
46 // io
47 lua_getfield(l, -1, "open");
48 // io open
49 Debug::check_assertion(lua_isfunction(l, -1), "Could not find io.open");
50 lua_setfield(l, LUA_REGISTRYINDEX, "io.open");
51 // io
52 lua_pop(l, 1);
53 // --
54}

56/**
57 * \brief Implementation of sol.file.open().
58 * \param l The Lua context that is calling this function.
59 * \return Number of values to return to Lua.
60 */
61int LuaContext::file_api_open(lua_State* l) {
62
63 return LuaTools::exception_boundary_handle(l, [&] {
64 const std::string& file_name = LuaTools::check_string(l, 1);
65 const std::string& mode = LuaTools::opt_string(l, 2, "r");
66
67 const bool writing = mode != "r" && mode != "rb";
68
69 // file_name is relative to the data directory, the data archive or the
70 // quest write directory.
71 // Let's determine the real file name and pass it to io.open().
72 std::string real_file_name;
73 if (writing) {
74
75 // Writing a file.
76 if (QuestFiles::get_quest_write_dir().empty()) {
77 LuaTools::error(l,
78 "Cannot open file in writing: no write directory was specified in quest.dat");
79 }
80
81 real_file_name = QuestFiles::get_full_quest_write_dir() + "/" + file_name;
82 }
83 else {
84 // Reading a file.
85 QuestFiles::DataFileLocation location = QuestFiles::data_file_get_location(file_name);
86
87 switch (location) {
88
89 case QuestFiles::DataFileLocation::LOCATION_NONE:
90 // Not found.
91 lua_pushnil(l);
92 push_string(l, std::string("Cannot find file '") + file_name
93 + "' in the quest write directory, in data/, data.solarus or in data.solarus.zip");
94 return 2;
95
96 case QuestFiles::DataFileLocation::LOCATION_WRITE_DIRECTORY:
97 // Found in the quest write directory.
98 real_file_name = QuestFiles::get_full_quest_write_dir() + "/" + file_name;
99 break;
100
101 case QuestFiles::DataFileLocation::LOCATION_DATA_DIRECTORY:
102 // Found in the data directory.
103 real_file_name = QuestFiles::get_quest_path() + "/data/" + file_name;
104 break;
105
106 case QuestFiles::DataFileLocation::LOCATION_DATA_ARCHIVE:
107 {
108 // Found in the data archive.
109 // To call io.open(), we need a regular file, so let's create
110 // a temporary one.
111 const std::string& buffer = QuestFiles::data_file_read(file_name);
112 real_file_name = QuestFiles::create_temporary_file(buffer);
113 break;
114 }
115 }
116 }
117
118 // Call io.open.
119 lua_getfield(l, LUA_REGISTRYINDEX, "io.open");
120 push_string(l, real_file_name);
121 push_string(l, mode);
122
123 bool called = LuaTools::call_function(l, 2, 2, "io.open");
124 if (!called) {
125 LuaTools::error(l, "Unexpected error: failed to call io.open()");
126 }
127
128 return 2;
129 });
130}

132/**
133 * \brief Implementation of sol.file.exists().
134 * \param l The Lua context that is calling this function.
135 * \return Number of values to return to Lua.
136 */
137int LuaContext::file_api_exists(lua_State* l) {
138
139 return LuaTools::exception_boundary_handle(l, [&] {
140 const std::string& file_name = LuaTools::check_string(l, 1);
141
142 lua_pushboolean(l, QuestFiles::data_file_exists(file_name, false));
143
144 return 1;
145 });
146}
147
148/**
149 * \brief Implementation of sol.file.remove().
150 * \param l The Lua context that is calling this function.
151 * \return Number of values to return to Lua.
152 */
153int LuaContext::file_api_remove(lua_State* l) {
154
155 return LuaTools::exception_boundary_handle(l, [&] {
156 const std::string& file_name = LuaTools::check_string(l, 1);
157
158 bool success = QuestFiles::data_file_delete(file_name);
159
160 if (!success) {
161 lua_pushnil(l);
162 push_string(l, std::string("Failed to delete file '") + file_name + "'");
163 return 2;
164 }
165
166 lua_pushboolean(l, true);
167 return 1;
168 });
169}

171/**
172 * \brief Implementation of sol.file.mkdir().
173 * \param l The Lua context that is calling this function.
174 * \return Number of values to return to Lua.
175 */
176int LuaContext::file_api_mkdir(lua_State* l) {
177
178 return LuaTools::exception_boundary_handle(l, [&] {
179 const std::string& dir_name = LuaTools::check_string(l, 1);
180
181 bool success = QuestFiles::data_file_mkdir(dir_name);
182
183 if (!success) {
184 lua_pushnil(l);
185 push_string(l, std::string("Failed to create directory '") + dir_name + "'");
186 return 2;
187 }
188
189 lua_pushboolean(l, true);
190 return 1;
191 });
192}
