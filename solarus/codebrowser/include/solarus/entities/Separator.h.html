17#ifndef SOLARUS_SEPARATOR_H
18#define SOLARUS_SEPARATOR_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <string>
23
24namespace Solarus {
25
26/**
27 * \brief An horizontal or vertical separation between two parts of the map.
28 *
29 * The camera cannot cross this separation except when the hero touches it:
30 * in this case, scrolling is performed.
31 * This entity allows to separate parts of the map like if there was several
32 * maps.
33 */
34class Separator: public Entity {
35
36 public:
37
38 static constexpr EntityType ThisType = EntityType::SEPARATOR;
39
40 Separator(
41 const std::string& name,
42 int layer,
43 const Point& xy,
44 const Size& size
45 );
46
47 virtual EntityType get_type() const override;
48
49 bool is_horizontal() const;
50 bool is_vertical() const;
51
52 virtual bool is_obstacle_for(Entity& other) override;
53 virtual bool test_collision_custom(Entity& entity) override;
54 virtual void notify_collision(
55 Entity& entity_overlapping,
56 CollisionMode collision_mode
57 ) override;
58 void notify_activating(int direction4);
59 void notify_activated(int direction4);
60
61};
62
63}
64
65#endif
