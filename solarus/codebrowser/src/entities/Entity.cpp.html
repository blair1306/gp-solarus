163/**
164 * \brief Inform entities sensible to their ground that it may have just
165 * changed because of this entity.
166 *
167 * This does the work even if this entity is not a ground modifier,
168 * because this is necessary in case it just stopped being one.
169 */
170void Entity::update_ground_observers() {
171
172 // Update overlapping entities that are sensible to their ground.
173 const Rectangle& box = get_bounding_box();
174 std::vector<EntityPtr> entities_nearby;
175 get_entities().get_entities_in_rectangle(box, entities_nearby);
176 for (const EntityPtr& entity_nearby: entities_nearby) {
177
178 if (!entity_nearby->is_ground_observer()) {
179 // The entity does not care about the ground below it.
180 continue;
181 }
182
183 // Update the ground of observers that overlap or were just overlapping this one.
184 if (entity_nearby->get_layer() == get_layer() &&
185 (overlaps(entity_nearby->get_ground_point()) || overlaps(*entity_nearby))
186 // FIXME this is not precise and does not work for entities that disappear.
187 ) {
188 entity_nearby->update_ground_below();
189 }
190 }
191}

193/**
194 * \brief Returns the ground where this entity is.
195 * \return The ground under this entity.
196 */
197Ground Entity::get_ground_below() const {
198
199 if (is_ground_observer()) {
200 // The information is already known.
201 return ground_below;
202 }
203
204 return get_map().get_ground(
205 get_layer(), get_ground_point(), this
206 );
207}

209/**
210 * \brief Recomputes the kind of ground below this entity.
211 *
212 * This function does nothing if the entity is not sensible to its ground.
213 */
214void Entity::update_ground_below() {
215
216 if (!is_ground_observer()) {
217 // This entity does not care about the ground below it.
218 return;
219 }
220
221 if (!is_enabled() ||
222 is_being_removed()) {
223 return;
224 }
225 // Note that even if the entity is suspended,
226 // the user might want to know the ground below it.
227
228 if (map->test_collision_with_border(get_ground_point())) {
229 // If the entity is outside the map, which is legal during a scrolling
230 // transition, don't try to determine any ground.
231 return;
232 }
233
234 Ground previous_ground = this->ground_below;
235 this->ground_below = get_map().get_ground(
236 get_layer(), get_ground_point(), this
237 );
238 if (this->ground_below != previous_ground) {
239 notify_ground_below_changed();
240 }
241}

306/**
307 * \brief Sets the map where this entity is.
308 *
309 * Warning: when this function is called during the initialization of a new map,
310 * the current map of the game is still the old one.
311 *
312 * \param map The map.
313 */
314void Entity::set_map(Map& map) {
315
316 this->main_loop = &map.get_game().get_main_loop();
317 this->map = &map;
318 set_lua_context(&main_loop->get_lua_context());
319 if (&get_game().get_current_map() == &map) {
320 notify_tileset_changed();
321 }
322
323 this->ground_below = Ground::EMPTY;
324
325 if (!initialized && map.is_loaded()) {
326 // The entity is being created on a map already running.
327 // In this case, we are ready to finish the initialization right now.
328 finish_initialization();
329 }
330}

356/**
357 * \brief Finishes the initialization of this entity once the map is loaded.
358 *
359 * This function must be called once the map is ready.
360 * It calls notify_creating(), then the Lua event entity:on_created(),
361 * and then notify_created().
362 */
363void Entity::finish_initialization() {
364
365 Debug::check_assertion(!initialized, "Entity is already initialized");
366 Debug::check_assertion(is_on_map(), "Missing map");
367 Debug::check_assertion(get_map().is_loaded(), "Map is not ready");

370 get_lua_context()->entity_on_created(*this);
371 notify_created();
372
373 initialized = true;
374}

589/**
590 * \brief This function is called when the layer of this entity has just changed.
591 *
592 * Redefine it if you need to be notified.
593 */
594void Entity::notify_layer_changed() {
595
596 if (!is_on_map()) {
597 return;
598 }
599
600 if (is_detector()) {
601 // Since this entity is a detector, all entities need to check
602 // their collisions with it.
603 get_map().check_collision_from_detector(*this);
604 }
605
606 // Check collisions between this entity and other detectors.
607 check_collision_with_detectors();
608
609 // Update the ground.
610 if (is_ground_modifier()) {
611 update_ground_observers();
612 }
613 update_ground_below();
614
615 // Notify Lua.
616 if (are_movement_notifications_enabled()) {
617 get_lua_context()->entity_on_position_changed(*this, get_xy(), get_layer());
618 }
619}

1006/**
1007 * \brief Returns the coordinates of the point the entity is looking at.
1008 *
1009 * By default, this point depends on the direction of the sprite if any.
1010 * If there is no sprite, or if the sprite has not 4 directions,
1011 * then the movement is considered.
1012 * If there is no movement either, the entity is assumed to look to the north.
1013 * You can redefine this method to change what is a facing point for your
1014 * subclass.
1015 *
1016 * \return The coordinates of the point the entity is looking at.
1017 */
1018Point Entity::get_facing_point() const {
1019
1020 int direction4 = 1; // North by default.
1021 const SpritePtr& sprite = get_sprite();
1022 if (sprite != nullptr && sprite->get_nb_directions() == 4) {
1023 // Use the sprite to decide where the entity is looking to.
1024 direction4 = sprite->get_current_direction();
1025 }
1026 else {
1027 if (get_movement() != nullptr) {
1028 // Otherwise use the movement.
1029 direction4 = get_movement()->get_displayed_direction4();
1030 }
1031 }
1032
1033 return get_touching_point(direction4);
1034}
1035
1036/**
1037 * \brief Returns the coordinates of the point the entity would be facing
1038 * if it was looking towards the specified direction.
1039 * \param direction A direction (0 to 3).
1040 * \return The point the entity touching this direction.
1041 */
1042Point Entity::get_touching_point(int direction) const {
1043
1044 Point touching_point = get_center_point();
1045
1046 switch (direction) {
1047
1048 // right
1049 case 0:
1050 touching_point.x += get_width() / 2;
1051 break;
1052
1053 // up
1054 case 1:
1055 touching_point.y += -get_height() / 2 - 1;
1056 break;
1057
1058 // left
1059 case 2:
1060 touching_point.x += -get_width() / 2 - 1;
1061 break;
1062
1063 // down
1064 case 3:
1065 touching_point.y += get_height() / 2;
1066 break;
1067
1068 default:
1069 Debug::die("Invalid direction for Entity::get_touching_point()");
1070 }
1071 return touching_point;
1072}

1241/**
1242 * \brief Returns a sprite of this entity.
1243 * \param sprite_name Name of the sprite to get, or an empty string to
1244 * get the default sprite.
1245 * \return The corresponding sprite, or nullptr if there is no such sprite.
1246 */
1247SpritePtr Entity::get_sprite(const std::string& sprite_name) const {
1248
1249 if (sprites.empty()) {
1250 // This entity has no sprite.
1251 return nullptr;
1252 }
1253
1254 std::string valid_sprite_name;
1255 if (sprite_name.empty()) {
1256 // No sprite name specified: use the default one if any.
1257 if (default_sprite_name.empty()) {
1258 // No default sprite either: return the first one.
1259 for (const NamedSprite& named_sprite : sprites) {
1260 if (!named_sprite.removed) {
1261 return named_sprite.sprite;
1262 }
1263 }
1264 }
1265 else {
1266 valid_sprite_name = default_sprite_name;
1267 }
1268 }
1269 else {
1270 valid_sprite_name = sprite_name;
1271 }
1272
1273 for (const NamedSprite& named_sprite : sprites) {
1274 // Find a sprite with the given name and that was not removed.
1275 if (named_sprite.name == valid_sprite_name &&
1276 !named_sprite.removed) {
1277 return named_sprite.sprite;
1278 }
1279 }
1280 return nullptr;
1281}

1736/**

1745 * test_collision_* functions are called.
1746 * If there is a collision, the notify_collision() method is called.
1747 *
1748 * \param other The entity to check.
1749 */
1750void Entity::check_collision(Entity& other) {
1751
1752 if (!is_detector()) {
1753 // No collision kind to detect.
1754 return;
1755 }
1756
1757 if (&other == this) {
1758 return;
1759 }
1760
1761 if (get_layer() != other.get_layer() && !has_layer_independent_collisions()) {
1762 // Not the same layer: no collision.
1763 return;
1764 }
1765
1766 // Detect the collision depending on the collision modes.
1767
1768 if (has_collision_mode(CollisionMode::COLLISION_OVERLAPPING) && test_collision_rectangle(other)) {
1769 notify_collision(other, CollisionMode::COLLISION_OVERLAPPING);
1770 }
1771
1772 if (has_collision_mode(CollisionMode::COLLISION_CONTAINING) && test_collision_inside(other)) {
1773 notify_collision(other, CollisionMode::COLLISION_CONTAINING);
1774 }
1775
1776 if (has_collision_mode(CollisionMode::COLLISION_ORIGIN) && test_collision_origin_point(other)) {
1777 notify_collision(other, CollisionMode::COLLISION_ORIGIN);
1778 }
1779
1780 if (has_collision_mode(CollisionMode::COLLISION_FACING) && test_collision_facing_point(other)) {
1781
1782 if (other.get_facing_entity() == nullptr) {
1783 // Make sure only one entity can think "I am the facing entity".
1784 other.set_facing_entity(this);
1785 }
1786 notify_collision(other, CollisionMode::COLLISION_FACING);
1787 }
1788
1789 if (has_collision_mode(CollisionMode::COLLISION_TOUCHING) && test_collision_touching(other)) {
1790 notify_collision(other, CollisionMode::COLLISION_TOUCHING);
1791 }
1792
1793 if (has_collision_mode(CollisionMode::COLLISION_CENTER) && test_collision_center(other)) {
1794 notify_collision(other, CollisionMode::COLLISION_CENTER);
1795 }
1796
1797 if (has_collision_mode(CollisionMode::COLLISION_CUSTOM) && test_collision_custom(other)) {

3493 if (old_state != nullptr) {

3496

3500 // This is not a normal situation since we only called stop() to allow
3501 // new_state to start.
