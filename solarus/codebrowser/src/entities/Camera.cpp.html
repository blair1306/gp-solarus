36/**
37 * \brief State of the camera when centered on an entity.
38 */
39class TrackingState: public Entity::State {
40
41public:
42
43 TrackingState(Camera& camera, const EntityPtr& tracked_entity);
44
45 void update() override;
46 bool is_traversing_separator() const;
47 void traverse_separator(Separator& separator);
48
49 const EntityPtr& get_tracked_entity() const;
50
51private:
52
53 EntityPtr tracked_entity; /**< Entity the camera is tracking. */
54 Rectangle separator_scrolling_position; /**< Current camera position while crossing a separator. */
55 Rectangle separator_target_position; /**< Target camera position when crossing a separator. */
56 Point separator_scrolling_delta; /**< increment to the camera position when crossing a separator. */
57 uint32_t separator_next_scrolling_date; /**< Next camera position change when crossing a separator. */
58 int separator_scrolling_direction4; /**< Direction when scrolling. */
59 std::shared_ptr<Separator>
60 separator_traversed; /**< Separator currently being traversed or nullptr. */
61
62};

65 * \brief Creates a camera tracking state.

87/**
88 * \brief Updates the position of the camera when the camera is tracking
89 * an entity.
90 */
91void TrackingState::update() {
92
93 Camera& camera = get_entity<Camera>();
94 if (separator_next_scrolling_date == 0) {
95 // Normal case: not traversing a separator.
96
97 // First compute camera coordinates ignoring map limits and separators.
98 Rectangle next = camera.get_bounding_box();
99 next.set_center(tracked_entity->get_center_point());
100
101 // Then apply constraints of both separators and map limits.
102 camera.set_bounding_box(camera.apply_separators_and_map_bounds(next));
103 camera.notify_bounding_box_changed();
104 }
105 else {
106 // The tracked entity is currently traversing a separator.
107 // Update camera coordinates.
108 uint32_t now = System::now();
109 bool finished = false;
110 while (separator_next_scrolling_date != 0
111 && now >= separator_next_scrolling_date) {
112 separator_scrolling_position.add_xy(separator_scrolling_delta);
113
114 separator_next_scrolling_date += 1;
115
116 if (separator_scrolling_position == separator_target_position) {
117 // Finished.
118 finished = true;
119 }
120 }
121
122 if (finished) {
123 separator_next_scrolling_date = 0;
124 separator_traversed->notify_activated(separator_scrolling_direction4);
125 separator_traversed = nullptr;
126 separator_scrolling_direction4 = 0;
127 }
128
129 // Then only apply map limit constraints.
130 // Ignore separators since we are currently crossing one of them.
131 camera.set_bounding_box(camera.apply_map_bounds(separator_scrolling_position));
132 camera.notify_bounding_box_changed();
133 }
134}

144/**
145 * \brief Starts traversing a separator.
146 *
147 * The tracked entity must touch the separator when you call this function.
148 *
149 * \param separator The separator to traverse.
150 */
151void TrackingState::traverse_separator(Separator& separator) {
152
153 Camera& camera = get_entity<Camera>();
154
155 // Save the current position of the camera.
156 separator_scrolling_position = camera.get_bounding_box();
157
158 // Start scrolling.
159 separator_traversed = std::static_pointer_cast<Separator>(
160 separator.shared_from_this()
161 );
162 separator_scrolling_delta = Point();
163 separator_target_position = separator_scrolling_position;
164
165 const Point& tracked_entity_center = tracked_entity->get_center_point();
166 const Point& separator_center = separator.get_center_point();
167 if (separator.is_horizontal()) {
168 if (tracked_entity_center.y < separator_center.y) {
169 separator_scrolling_direction4 = 3;
170 separator_scrolling_delta.y = 1;
171 separator_target_position.add_y(camera.get_height());
172 }
173 else {
174 separator_scrolling_direction4 = 1;
175 separator_scrolling_delta.y = -1;
176 separator_target_position.add_y(-camera.get_height());
177 }
178 }
179 else {
180 if (tracked_entity_center.x < separator_center.x) {
181 separator_scrolling_direction4 = 0;
182 separator_scrolling_delta.x = 1;
183 separator_target_position.add_x(camera.get_width());
184 }
185 else {
186 separator_scrolling_direction4 = 2;
187 separator_scrolling_delta.x = -1;
188 separator_target_position.add_x(-camera.get_width());
189 }
190 }
191
192 separator.notify_activating(separator_scrolling_direction4);
193 separator_next_scrolling_date = System::now();
194
195 // Move the tracked entity two pixels ahead to avoid to traverse the separator again.
196 tracked_entity->set_xy(tracked_entity->get_xy() + 2 * separator_scrolling_delta);
197 tracked_entity->notify_bounding_box_changed();
198}

222/**
223 * \brief Creates a camera.
224 * \param map The map.
225 */
226Camera::Camera(Map& map):
227 Entity("", 0, map.get_max_layer(), Point(0, 0), Video::get_quest_size()),
228 surface(nullptr),
229 position_on_screen(0, 0) {
230
231 create_surface();
232 set_map(map);
233 const HeroPtr& hero = get_game().get_hero();
234 Debug::check_assertion(hero != nullptr, "Missing hero when initializing camera");
235 start_tracking(hero);
236}

302/**
303 * \brief Returns the position this camera should take to track the specified point.
304 * \param tracked_xy A point in map coordinates.
305 * \return Where this camera should be placed to be focused on this point,
306 * respecting separators and map limits.
307 */
308Point Camera::get_position_to_track(const Point& tracked_xy) const {
309
310 Point top_left_xy = tracked_xy - Point(get_width() / 2, get_height() / 2);
311 Rectangle box(top_left_xy, get_size());
312
313 return apply_separators_and_map_bounds(box).get_xy();
314}

397/**
398 * \brief Ensures that a rectangle does not cross map limits.
399 * \param area The rectangle to check. Is should not be entirely inside the map.
400 * It can be bigger than the map: in such a case, the resulting rectangle cannot
401 * avoid to cross map limits, and it will be centered.
402 * \return A rectangle corresponding to the first one but stopping on map limits.
403 */
404Rectangle Camera::apply_map_bounds(const Rectangle& area) const {
405
406 int x = area.get_x(); // Top-left corner.
407 int y = area.get_y();
408 const int width = area.get_width();
409 const int height = area.get_height();
410
411 const Size& map_size = get_map().get_size();
412 if (map_size.width < width) {
413 x = (map_size.width - width) / 2;
414 }
415 else {
416 x = std::min(std::max(x, 0),
417 map_size.width - width);
418 }
419
420 if (map_size.height < height) {
421 y = (map_size.height - height) / 2;
422 }
423 else {
424 y = std::min(std::max(y, 0),
425 map_size.height - height);
426 }
427 return Rectangle(x, y, width, height);
428}

430/**
431 * \brief Ensures that a rectangle does not cross separators.
432 * \param area The rectangle to check.
433 * \return A rectangle corresponding to the first one but stopping on separators.
434 */
435Rectangle Camera::apply_separators(const Rectangle& area) const {
436
437 int x = area.get_x(); // Top-left corner.
438 int y = area.get_y();
439 const int width = area.get_width();
440 const int height = area.get_height();
441
442 // TODO simplify: treat horizontal separators first and then all vertical ones.
443 int adjusted_x = x; // Updated coordinates after applying separators.
444 int adjusted_y = y;
445 std::vector<std::shared_ptr<const Separator>> applied_separators;
446 const std::set<std::shared_ptr<const Separator>>& separators =
447 get_entities().get_entities_by_type<Separator>();
448 for (const std::shared_ptr<const Separator>& separator: separators) {
449
450 if (separator->is_vertical()) {
451 // Vertical separator.
452 int separation_x = separator->get_x() + 8;
453
454 if (x < separation_x && separation_x < x + width
455 && separator->get_y() < y + height
456 && y < separator->get_y() + separator->get_height()) {
457 int left = separation_x - x;
458 int right = x + width - separation_x;
459 if (left > right) {
460 adjusted_x = separation_x - width;
461 }
462 else {
463 adjusted_x = separation_x;
464 }
465 applied_separators.push_back(separator);
466 }
467 }
468 else {
469 Debug::check_assertion(separator->is_horizontal(), "Invalid separator shape");
470
471 // Horizontal separator.
472 int separation_y = separator->get_y() + 8;
473 if (y < separation_y && separation_y < y + height
474 && separator->get_x() < x + width
475 && x < separator->get_x() + separator->get_width()) {
476 int top = separation_y - y;
477 int bottom = y + height - separation_y;
478 if (top > bottom) {
479 adjusted_y = separation_y - height;
480 }
481 else {
482 adjusted_y = separation_y;
483 }
484 applied_separators.push_back(separator);
485 }
486 }
487 } // End for each separator.
488
489 bool must_adjust_x = true;
490 bool must_adjust_y = true;
491 if (adjusted_x != x && adjusted_y != y) {
492 // Both directions were modified. Maybe it is a T configuration where
493 // a separator deactivates another one.
494
495 must_adjust_x = false;
496 must_adjust_y = false;
497 for (const std::shared_ptr<const Separator>& separator: applied_separators) {
498
499 if (separator->is_vertical()) {
500 // Vertical separator.
501 int separation_x = separator->get_x() + 8;
502
503 if (x < separation_x && separation_x < x + width
504 && separator->get_y() < adjusted_y + height
505 && adjusted_y < separator->get_y() + separator->get_height()) {
506 must_adjust_x = true;
507 }
508 }
509 else {
510 // Horizontal separator.
511 int separation_y = separator->get_y() + 8;
512
513 if (y < separation_y && separation_y < y + height
514 && separator->get_x() < adjusted_x + width
515 && adjusted_x < separator->get_x() + separator->get_width()) {
516 must_adjust_y = true;
517 }
518 }
519 } // End for each separator applied.
520 } // End if both directions.
521
522 if (must_adjust_x) {
523 x = adjusted_x;
524 }
525 if (must_adjust_y) {
526 y = adjusted_y;
527 }
528
529 return Rectangle(x, y, width, height);
530}
531
532/**
533 * \brief Ensures that a rectangle does not cross separators nor map bounds.
534 * \param area The rectangle to check.
535 * It is the responsibility of quest makers to put enough space between
536 * separators (the space should be at least the quest size).
537 * If separators are too close to each other for the rectangle to fit,
538 * the camera will cross some of them.
539 * If separators are too close to a limit of the map, the
540 * the camera will cross them but will never cross map limits.
541 * \return A rectangle corresponding to the first one but stopping on
542 * separators and map bounds.
543 */
544Rectangle Camera::apply_separators_and_map_bounds(const Rectangle& area) const {
545 return apply_map_bounds(apply_separators(area));
546}
