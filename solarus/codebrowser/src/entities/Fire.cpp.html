17#include "solarus/entities/Fire.h"
18#include "solarus/lowlevel/Point.h"

22namespace Solarus {
23
24/**
25 * \brief Creates some fire.
26 * \param name Name identifying the entity on the map or an empty string.
27 * \param layer layer of the fire entity to create
28 * \param xy coordinates of the fire
29 */
30Fire::Fire(const std::string& name, int layer, const Point& xy):
31 Entity(name, 0, layer, xy, Size(16, 16)) {
32
33 // initialize the entity
34 set_collision_modes(CollisionMode::COLLISION_OVERLAPPING | CollisionMode::COLLISION_SPRITE);
35
36 const SpritePtr& sprite = create_sprite("entities/fire");
37 sprite->enable_pixel_collisions();
38
39 set_origin(8, 13);
40}
41
42/**
43 * \brief Returns the type of entity.
44 * \return the type of entity
45 */
46EntityType Fire::get_type() const {
47 return ThisType;
48}
49
50/**
51 * \brief Updates this entity.
52 */
53void Fire::update() {
54
55 Entity::update();
56
57 const SpritePtr& sprite = get_sprite();
58 if (sprite != nullptr && sprite->is_animation_finished()) {
59 remove_from_map();
60 }
61 else {
62 check_collision_with_detectors();
63 }
64}
65
66/**
67 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
68 */
69void Fire::notify_collision(
70 Entity& other_entity,
71 Sprite& /* this_sprite */,
72 Sprite& other_sprite
73) {
74 other_entity.notify_collision_with_fire(*this, other_sprite);
75}
