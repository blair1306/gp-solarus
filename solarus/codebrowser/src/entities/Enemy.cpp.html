71/**
72 * \brief Creates an enemy.
73 * \param game The game.
74 * \param name Name identifying the enemy or an empty string.
75 * \param layer The layer on the map.
76 * \param xy Coordinates on the map.
77 * \param breed Breed of the enemy.
78 * \param treasure Treasure dropped when the enemy is killed.
79 */
80Enemy::Enemy(
81 Game& /* game */,
82 const std::string& name,
83 int layer,
84 const Point& xy,
85 const std::string& breed,
86 const Treasure& treasure
87):
88 Entity(name, 0, layer, xy, Size(0, 0)),
89 breed(breed),
90 damage_on_hero(1),
91 life(1),
92 hurt_style(HurtStyle::NORMAL),
93 pushed_back_when_hurt(true),
94 push_hero_on_sword(false),
95 can_hurt_hero_running(false),
96 minimum_shield_needed(0),
97 savegame_variable(),
98 traversable(true),
99 obstacle_behavior(ObstacleBehavior::NORMAL),
100 being_hurt(false),
101 stop_hurt_date(0),
102 invulnerable(false),
103 vulnerable_again_date(0),
104 can_attack(true),
105 can_attack_again_date(0),
106 immobilized(false),
107 start_shaking_date(0),
108 end_shaking_date(0),
109 dying_animation_started(false),
110 treasure(treasure),
111 exploding(false),
112 nb_explosions(0),
113 next_explosion_date(0) {
114
115 set_collision_modes(CollisionMode::COLLISION_OVERLAPPING | CollisionMode::COLLISION_SPRITE);
116 set_size(16, 16);
117 set_origin(8, 13);
118 set_drawn_in_y_order(true);
119}

121/**
122 * \brief Creates an enemy.
123 *
124 * This method acts like a constructor, and usually returns an enemy.
125 * However, if the enemy is already dead and cannot be killed again,
126 * this function returns:
127 * - nullptr if the enemy has no treasure (or its treasure was already picked)
128 * - or a pickable treasure if the enemy has one
129 *
130 * \param game the current game
131 * \param breed breed of the enemy to create
132 * \param name a name identifying the enemy
133 * \param savegame_variable name of the boolean variable indicating that the enemy is dead
134 * \param layer layer of the enemy on the map
135 * \param xy Coordinates of the enemy on the map.
136 * \param direction initial direction of the enemy on the map (0 to 3)
137 * \param treasure the pickable item that the enemy drops
138 * \return the enemy created (may also be a Pickable or nullptr)
139 */
140EntityPtr Enemy::create(
141 Game& game,
142 const std::string& breed,
143 const std::string& savegame_variable,
144 const std::string& name,
145 int layer,
146 const Point& xy,
147 int direction,
148 const Treasure& treasure
149) {
150 // see if the enemy is dead
151 if (!savegame_variable.empty()
152 && game.get_savegame().get_boolean(savegame_variable)) {
153
154 // the enemy is dead: create its pickable treasure (if any) instead
155 if (treasure.is_saved() && !game.get_savegame().get_boolean(treasure.get_savegame_variable())) {
156 return Pickable::create(game, "", layer, xy, treasure, FALLING_NONE, true);
157 }
158 return nullptr;
159 }
160
161 // create the enemy
162 std::shared_ptr<Enemy> enemy = std::make_shared<Enemy>(
163 game, name, layer, xy, breed, treasure
164 );
165
166 // initialize the fields
167 enemy->set_direction(direction);
168 enemy->savegame_variable = savegame_variable;
169 enemy->set_default_attack_consequences();
170
171 return enemy;
172}

198/**
199 * \copydoc Entity::notify_created
200 */
201void Enemy::notify_created() {
202
203 Entity::notify_created();
204
205 // At this point, enemy:on_created() was called.
206 enable_pixel_collisions();
207
208 // Give sprites their initial direction.
209 int initial_direction = get_direction();
210 for (const SpritePtr& sprite: get_sprites()) {
211 sprite->set_current_direction(initial_direction);
212 }
213
214 if (is_enabled()) {
215 restart();
216 }
217}

247/**
248 * \brief Returns whether this entity is an obstacle for another one.
249 * \param other Another entity.
250 * \return \c true if this entity is an obstacle for the other one.
251 */
252bool Enemy::is_obstacle_for(Entity& other) {
253
254 if (!is_enabled()) {
255 return false;
256 }
257
258 return !is_traversable() || other.is_enemy_obstacle(*this);
259}

272/**
273 * \brief Returns whether a destructible item is currently considered as an obstacle for this entity.
274 * \param destructible a destructible item
275 * \return true if the destructible item is currently an obstacle this entity
276 */
277bool Enemy::is_destructible_obstacle(Destructible& destructible) {
278
279 // The destructible object is an obstacle unless the enemy is already
280 // overlapping it, which is possible with destructibles that can regenerate.
281 if (this->overlaps(destructible)) {
282 return false;
283 }
284 return obstacle_behavior != ObstacleBehavior::FLYING;
285}

338/**
339 * \brief Returns whether a deep water tile is currently considered as an obstacle by this entity.
340 * \return true if the deep water tiles are currently an obstacle for this entity
341 */
342bool Enemy::is_deep_water_obstacle() const {
343
344 if (obstacle_behavior == ObstacleBehavior::FLYING
345 || obstacle_behavior == ObstacleBehavior::SWIMMING) {
346 return false;
347 }
348
349 if (is_being_hurt()) {
350 return false;
351 }
352
353 const int layer = get_layer();
354 const int x = get_top_left_x();
355 const int y = get_top_left_y();
356 if (get_map().get_ground(layer, x, y, this) == Ground::DEEP_WATER
357 || get_map().get_ground(layer, x + get_width() - 1, y, this) == Ground::DEEP_WATER
358 || get_map().get_ground(layer, x, y + get_height() - 1, this) == Ground::DEEP_WATER
359 || get_map().get_ground(layer, x + get_width() - 1, y + get_height() - 1, this) == Ground::DEEP_WATER) {
360 return false;
361 }
362
363 return true;
364}

616/**
617 * \brief Returns the reaction corresponding to an attack on a sprite of this enemy.
618 * \param attack an attack this enemy receives
619 * \param this_sprite the sprite attacked, or nullptr if the attack does not come from
620 * a pixel-precise collision test
621 * \return the corresponding reaction
622 */
623EnemyReaction::Reaction Enemy::get_attack_consequence(
624 EnemyAttack attack,
625 const Sprite* this_sprite) const {
626
627 const auto& it = attack_reactions.find(attack);
628 if (it == attack_reactions.end()) {
629 // Attack consequence was not initialized. Return a default value.
630 return EnemyReaction::Reaction();
631 }
632 return it->second.get_reaction(this_sprite);
633}

635/**
636 * \brief Sets how the enemy reacts to an attack.
637 *
638 * This reaction will be used unless the attack is received by a sprite
639 * that has a specific reaction (see set_attack_consequence_sprite()).
640 *
641 * \param attack an attack
642 * \param reaction how the enemy will react
643 * \param life_lost number of life points lost with this attack (possibly zero)
644 */
645void Enemy::set_attack_consequence(
646 EnemyAttack attack,
647 EnemyReaction::ReactionType reaction,
648 int life_lost) {
649
650 if (life_lost < 0) {
651 std::ostringstream oss;
652 oss << "Invalid amount of life: " << life_lost;
653 Debug::die(oss.str());
654 }
655 attack_reactions[attack].set_general_reaction(reaction, life_lost);
656}

665void Enemy::set_attack_consequence_sprite(

701/**
702 * \brief Sets some default values for the reactions of the attacks.
703 */
704void Enemy::set_default_attack_consequences() {
705
706 for (const auto& kvp : attack_names) {
707 EnemyAttack attack = kvp.first;
708 attack_reactions[attack].set_default_reaction();
709 }
710 set_attack_consequence(EnemyAttack::SWORD, EnemyReaction::ReactionType::HURT, 1); // multiplied by the sword strength
711 set_attack_consequence(EnemyAttack::THROWN_ITEM, EnemyReaction::ReactionType::HURT, 1); // multiplied depending on the item
712 set_attack_consequence(EnemyAttack::EXPLOSION, EnemyReaction::ReactionType::HURT, 2);
713 set_attack_consequence(EnemyAttack::ARROW, EnemyReaction::ReactionType::HURT, 2);
714 set_attack_consequence(EnemyAttack::HOOKSHOT, EnemyReaction::ReactionType::IMMOBILIZED);
715 set_attack_consequence(EnemyAttack::BOOMERANG, EnemyReaction::ReactionType::IMMOBILIZED);
716 set_attack_consequence(EnemyAttack::FIRE, EnemyReaction::ReactionType::HURT, 3);
717}

766/**
767 * \brief Updates the enemy.
768 */
769void Enemy::update() {
770
771 Entity::update();
772
773 if (is_suspended() || !is_enabled()) {
774 return;
775 }
776
777 uint32_t now = System::now();
778
779 if (being_hurt) {
780
781 // see if we should stop the animation "hurt"
782 if (now >= stop_hurt_date) {
783 being_hurt = false;
784 set_movement_notifications_enabled(true);
785
786 if (life <= 0) {
787 kill();
788 }
789 else if (is_immobilized()) {
790 clear_movement();
791 set_animation("immobilized");
792 notify_immobilized();
793 }
794 else {
795 clear_movement();
796 restart();
797 }
798 }
799 }
800
801 if (life > 0 && invulnerable && now >= vulnerable_again_date && !being_hurt) {
802 invulnerable = false;
803 }
804
805 if (life > 0 && !can_attack && !is_immobilized()
806 && can_attack_again_date != 0 && now >= can_attack_again_date) {
807 can_attack = true;
808 }
809
810 if (is_immobilized() &&
811 !is_killed() &&
812 now >= end_shaking_date &&
813 get_animation() == "shaking"
814 ) {
815 restart();
816 }
817
818 if (is_immobilized() &&
819 !is_killed() &&
820 !is_being_hurt() &&
821 now >= start_shaking_date &&
822 !get_animation().empty() &&
823 get_animation() != "shaking"
824 ) {
825
826 end_shaking_date = now + 2000;
827 set_animation("shaking");
828 }
829
830 if (exploding) {
831 uint32_t now = System::now();
832 if (now >= next_explosion_date) {
833
834 // create an explosion
835 Point xy;
836 xy.x = get_top_left_x() + Random::get_number(get_width());
837 xy.y = get_top_left_y() + Random::get_number(get_height());
838 get_entities().add_entity(std::make_shared<Explosion>(
839 "", get_map().get_max_layer(), xy, false
840 ));
841 Sound::play("explosion");
842
843 next_explosion_date = now + 200;
844 nb_explosions++;
845
846 if (nb_explosions >= 15) {
847 exploding = false;
848 }
849 }
850 }
851
852 if (is_killed() && is_dying_animation_finished()) {
853
854 // Create the pickable treasure if any.
855 get_entities().add_entity(Pickable::create(
856 get_game(),
857 "",
858 get_layer(),
859 get_xy(),
860 treasure,
861 FALLING_HIGH,
862 false
863 ));
864
865 // Remove the enemy.
866 remove_from_map();
867
868 // Notify Lua that this enemy is dead.
869 // We need to do this after remove_from_map() so that this enemy is
870 // considered dead in functions like map:has_entities(prefix).
871 notify_dead();
872 }
873
874 get_lua_context()->entity_on_update(*this);
875}

1024/**
1025 * \brief Attacks the hero if possible.
1026 *
1027 * This function is called when there is a collision between the enemy and the hero.
1028 *
1029 * \param hero The hero to attack.
1030 * \param this_sprite The sprite of this enemy that detected the collision with the hero,
1031 * or nullptr if it was not a pixel-precise collision.
1032 */
1033void Enemy::attack_hero(Hero& hero, Sprite* this_sprite) {
1034
1035 if (!is_immobilized()
1036 && can_attack
1037 && hero.can_be_hurt(this)) {
1038
1039 bool hero_protected = false;
1040 if (minimum_shield_needed != 0
1041 && get_equipment().has_ability(Ability::SHIELD, minimum_shield_needed)
1042 && hero.can_use_shield()) {
1043
1044 // Compute the direction corresponding to the angle between the enemy and the hero.
1045 double angle = hero.get_angle(*this, nullptr, this_sprite);
1046 int protected_direction4 = (int) ((angle + Geometry::PI_OVER_2 / 2.0) * 4 / Geometry::TWO_PI);
1047 protected_direction4 = (protected_direction4 + 4) % 4;
1048
1049 // Also get the direction of the enemy's sprite.
1050 int sprite_opposite_direction4 = -1;
1051 if (this_sprite != nullptr) {
1052 sprite_opposite_direction4 = (this_sprite->get_current_direction() + 2) % 4;
1053 }
1054
1055 // The hero is protected if he is facing the opposite of one of these directions.
1056 hero_protected = hero.is_facing_direction4(protected_direction4) ||
1057 hero.is_facing_direction4(sprite_opposite_direction4);
1058 }
1059
1060 if (hero_protected) {
1061 attack_stopped_by_hero_shield();
1062 }
1063 else {
1064 // Let the enemy script handle this if it wants.
1065 const bool handled = get_lua_context()->enemy_on_attacking_hero(
1066 *this, hero, this_sprite
1067 );
1068 if (!handled) {
1069 // Scripts did not customize the attack:
1070 // do the built-in hurt state of the hero.
1071 hero.hurt(*this, this_sprite, damage_on_hero);
1072 }
1073 }
1074 }
1075}

1167/**
1168 * \brief Makes the enemy receive an attack.
1169 *
1170 * He might resist to the attack or get hurt.
1171 *
1172 * \param attack type of attack
1173 * \param source the entity attacking the enemy (often the hero)
1174 * \param this_sprite the sprite of this enemy that received the attack, or nullptr
1175 * if the attack comes from a non pixel-precise collision test
1176 */
1177void Enemy::try_hurt(EnemyAttack attack, Entity& source, Sprite* this_sprite) {
1178
1179 EnemyReaction::Reaction reaction = get_attack_consequence(attack, this_sprite);
1180 if (invulnerable || reaction.type == EnemyReaction::ReactionType::IGNORED) {
1181 // ignore the attack
1182 return;
1183 }
1184
1185 invulnerable = true;
1186 vulnerable_again_date = System::now() + 500;
1187
1188 switch (reaction.type) {
1189
1190 case EnemyReaction::ReactionType::PROTECTED:
1191 // attack failure sound
1192 Sound::play("sword_tapping");
1193 break;
1194
1195 case EnemyReaction::ReactionType::IMMOBILIZED:
1196 // get immobilized
1197 being_hurt = true;
1198 hurt(source, this_sprite);
1199 immobilize();
1200 break;
1201
1202 case EnemyReaction::ReactionType::CUSTOM:
1203 // custom attack (defined in the script)
1204 if (is_in_normal_state()) {
1205 custom_attack(attack, this_sprite);
1206 }
1207 else {
1208 // no attack was made: notify the source correctly
1209 reaction.type = EnemyReaction::ReactionType::IGNORED;
1210 invulnerable = false;
1211 }
1212 break;
1213
1214 case EnemyReaction::ReactionType::HURT:
1215
1216 if (is_immobilized() && get_animation() == "shaking") {
1217 stop_immobilized();
1218 }
1219
1220 // Compute the number of health points lost by the enemy.
1221
1222 being_hurt = true;
1223 if (attack == EnemyAttack::SWORD) {
1224
1225 Hero& hero = static_cast<Hero&>(source);
1226
1227 // Sword attacks only use pixel-precise collisions.
1228 Debug::check_assertion(this_sprite != nullptr,
1229 "Missing enemy sprite for sword attack"
1230 );
1231
1232 // For a sword attack, the damage may be something customized.
1233 bool customized = get_lua_context()->enemy_on_hurt_by_sword(
1234 *this, hero, *this_sprite);
1235
1236 if (customized) {
1237 reaction.life_lost = 0; // Already done by the script.
1238 }
1239 else {
1240 // If this is not customized, the default it to multiply the reaction
1241 // by a factor that depends on the sword.
1242 reaction.life_lost *= hero.get_sword_damage_factor();
1243 }
1244 }
1245 else if (attack == EnemyAttack::THROWN_ITEM) {
1246 reaction.life_lost *= static_cast<CarriedObject&>(source).get_damage_on_enemies();
1247 }
1248 life -= reaction.life_lost;
1249
1250 hurt(source, this_sprite);
1251 notify_hurt(source, attack);
1252 break;
1253
1254 case EnemyReaction::ReactionType::IGNORED:
1255 return;
1256 }
1257
1258 // notify the source
1259 source.notify_attacked_enemy(
1260 attack,
1261 *this,
1262 this_sprite,
1263 reaction,
1264 get_life() <= 0
1265 );
1266}

1269 * \brief Hurts the enemy.
1270 *
1271 * Updates its state, its sprite and plays the sound.
1272 *
1273 * \param source The entity attacking the enemy (often the hero).
1274 * \param this_sprite The sprite of this enemy that received the attack, or nullptr
1275 * if the attack comes from a non pixel-precise collision test.
1276 */

1338/**
1339 * \brief Kills the enemy.
1340 *
1341 * This function is called when the enemy has no more health points.
1342 */
1343void Enemy::kill() {
1344
1345 // stop any movement and disable attacks
1346 set_collision_modes(COLLISION_NONE);
1347 clear_movement();
1348 invulnerable = true;
1349 can_attack = false;
1350 can_attack_again_date = 0;
1351 dying_animation_started = true;
1352
1353 if (hurt_style == HurtStyle::BOSS) {
1354 // A boss: create some explosions.
1355 exploding = true;
1356 nb_explosions = 0;
1357 next_explosion_date = System::now() + 2000;
1358 }
1359 else {
1360 // Replace the enemy sprites.
1361 clear_sprites();
1362 bool special_ground = false;
1363 switch (get_ground_below()) {
1364
1365 case Ground::HOLE:
1366 if (get_obstacle_behavior() != ObstacleBehavior::FLYING) {
1367 // TODO animation of falling into a hole.
1368 special_ground = true;
1369 Sound::play("jump");
1370 clear_treasure();
1371 }
1372 break;
1373
1374 case Ground::DEEP_WATER:
1375 if (get_obstacle_behavior() != ObstacleBehavior::FLYING &&
1376 get_obstacle_behavior() != ObstacleBehavior::SWIMMING) {
1377 // TODO water animation.
1378 special_ground = true;
1379 Sound::play("splash");
1380 clear_treasure();
1381 }
1382 break;
1383
1384 case Ground::LAVA:
1385 if (get_obstacle_behavior() != ObstacleBehavior::FLYING &&
1386 get_obstacle_behavior() != ObstacleBehavior::SWIMMING) {
1387 // TODO lava animation.
1388 special_ground = true;
1389 Sound::play("splash");
1390 clear_treasure();
1391 }
1392 break;
1393
1394 default:
1395 break;
1396 }
1397
1398 if (!special_ground) {
1399 // Normal dying animation.
1400 create_sprite("enemies/enemy_killed");
1401 Sound::play("enemy_killed");
1402 }
1403 }
1404
1405 // save the enemy state if required
1406 if (is_saved()) {
1407 get_savegame().set_boolean(savegame_variable, true);
1408 }
1409}
