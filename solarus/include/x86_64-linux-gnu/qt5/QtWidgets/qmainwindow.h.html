45#include <QtWidgets/qwidget.h>

51#ifndef QT_NO_MAINWINDOW
52
53class QDockWidget;
54class QMainWindowPrivate;
55class QMenuBar;
56class QStatusBar;
57class QToolBar;
58class QMenu;
59
60class Q_WIDGETS_EXPORT QMainWindow : public QWidget
61{
62 Q_OBJECT
63
64 Q_ENUMS(DockOption)
65 Q_FLAGS(DockOptions)
66 Q_PROPERTY(QSize iconSize READ iconSize WRITE setIconSize)
67 Q_PROPERTY(Qt::ToolButtonStyle toolButtonStyle READ toolButtonStyle WRITE setToolButtonStyle)
68#ifndef QT_NO_DOCKWIDGET
69 Q_PROPERTY(bool animated READ isAnimated WRITE setAnimated)
70#ifndef QT_NO_TABBAR
71 Q_PROPERTY(bool documentMode READ documentMode WRITE setDocumentMode)
72#endif // QT_NO_TABBAR
73#ifndef QT_NO_TABWIDGET
74 Q_PROPERTY(QTabWidget::TabShape tabShape READ tabShape WRITE setTabShape)
75#endif // QT_NO_TABWIDGET
76 Q_PROPERTY(bool dockNestingEnabled READ isDockNestingEnabled WRITE setDockNestingEnabled)
77#endif // QT_NO_DOCKWIDGET
78 Q_PROPERTY(DockOptions dockOptions READ dockOptions WRITE setDockOptions)
79#ifndef QT_NO_TOOLBAR
80 Q_PROPERTY(bool unifiedTitleAndToolBarOnMac READ unifiedTitleAndToolBarOnMac WRITE setUnifiedTitleAndToolBarOnMac)
81#endif
82
83public:
84 enum DockOption {
85 AnimatedDocks = 0x01,
86 AllowNestedDocks = 0x02,
87 AllowTabbedDocks = 0x04,
88 ForceTabbedDocks = 0x08, // implies AllowTabbedDocks, !AllowNestedDocks
89 VerticalTabs = 0x10 // implies AllowTabbedDocks
90 };
91 Q_DECLARE_FLAGS(DockOptions, DockOption)
92
93 explicit QMainWindow(QWidget *parent = 0, Qt::WindowFlags flags = 0);
94 ~QMainWindow();
95
96 QSize iconSize() const;
97 void setIconSize(const QSize &iconSize);
98
99 Qt::ToolButtonStyle toolButtonStyle() const;
100 void setToolButtonStyle(Qt::ToolButtonStyle toolButtonStyle);
101
102#ifndef QT_NO_DOCKWIDGET
103 bool isAnimated() const;
104 bool isDockNestingEnabled() const;
105#endif
106
107#ifndef QT_NO_TABBAR
108 bool documentMode() const;
109 void setDocumentMode(bool enabled);
110#endif
111
112#ifndef QT_NO_TABWIDGET
113 QTabWidget::TabShape tabShape() const;
114 void setTabShape(QTabWidget::TabShape tabShape);
115 QTabWidget::TabPosition tabPosition(Qt::DockWidgetArea area) const;
116 void setTabPosition(Qt::DockWidgetAreas areas, QTabWidget::TabPosition tabPosition);
117#endif // QT_NO_TABWIDGET
118
119 void setDockOptions(DockOptions options);
120 DockOptions dockOptions() const;
121
122 bool isSeparator(const QPoint &pos) const;
123
124#ifndef QT_NO_MENUBAR
125 QMenuBar *menuBar() const;
126 void setMenuBar(QMenuBar *menubar);
127
128 QWidget *menuWidget() const;
129 void setMenuWidget(QWidget *menubar);
130#endif
131
132#ifndef QT_NO_STATUSBAR
133 QStatusBar *statusBar() const;
134 void setStatusBar(QStatusBar *statusbar);
135#endif
136
137 QWidget *centralWidget() const;
138 void setCentralWidget(QWidget *widget);
139
140 QWidget *takeCentralWidget();
141
142#ifndef QT_NO_DOCKWIDGET
143 void setCorner(Qt::Corner corner, Qt::DockWidgetArea area);
144 Qt::DockWidgetArea corner(Qt::Corner corner) const;
145#endif
146
147#ifndef QT_NO_TOOLBAR
148 void addToolBarBreak(Qt::ToolBarArea area = Qt::TopToolBarArea);
149 void insertToolBarBreak(QToolBar *before);
150
151 void addToolBar(Qt::ToolBarArea area, QToolBar *toolbar);
152 void addToolBar(QToolBar *toolbar);
153 QToolBar *addToolBar(const QString &title);
154 void insertToolBar(QToolBar *before, QToolBar *toolbar);
155 void removeToolBar(QToolBar *toolbar);
156 void removeToolBarBreak(QToolBar *before);
157
158 void setUnifiedTitleAndToolBarOnMac(bool set);
159 bool unifiedTitleAndToolBarOnMac() const;
160
161 Qt::ToolBarArea toolBarArea(QToolBar *toolbar) const;
162 bool toolBarBreak(QToolBar *toolbar) const;
163#endif
164#ifndef QT_NO_DOCKWIDGET
165 void addDockWidget(Qt::DockWidgetArea area, QDockWidget *dockwidget);
166 void addDockWidget(Qt::DockWidgetArea area, QDockWidget *dockwidget,
167 Qt::Orientation orientation);
168 void splitDockWidget(QDockWidget *after, QDockWidget *dockwidget,
169 Qt::Orientation orientation);
170 void tabifyDockWidget(QDockWidget *first, QDockWidget *second);
171 QList<QDockWidget*> tabifiedDockWidgets(QDockWidget *dockwidget) const;
172 void removeDockWidget(QDockWidget *dockwidget);
173 bool restoreDockWidget(QDockWidget *dockwidget);
174
175 Qt::DockWidgetArea dockWidgetArea(QDockWidget *dockwidget) const;
176#endif // QT_NO_DOCKWIDGET
177
178 QByteArray saveState(int version = 0) const;
179 bool restoreState(const QByteArray &state, int version = 0);
180
181#ifndef QT_NO_MENU
182 virtual QMenu *createPopupMenu();
183#endif
184
185
186#ifndef QT_NO_DOCKWIDGET
187public Q_SLOTS:
188 void setAnimated(bool enabled);
189 void setDockNestingEnabled(bool enabled);
190#endif
191
192Q_SIGNALS:
193 void iconSizeChanged(const QSize &iconSize);
194 void toolButtonStyleChanged(Qt::ToolButtonStyle toolButtonStyle);
195
196protected:
197#ifndef QT_NO_CONTEXTMENU
198 void contextMenuEvent(QContextMenuEvent *event);
199#endif
200 bool event(QEvent *event);
201
202private:
203 Q_DECLARE_PRIVATE(QMainWindow)
204 Q_DISABLE_COPY(QMainWindow)
205};

209#endif // QT_NO_MAINWINDOW
