55class Q_CORE_EXPORT QDate
56{
57public:
58 enum MonthNameType {
59 DateFormat = 0,
60 StandaloneFormat
61 };
62public:
63 QDate() { jd = nullJd(); }
64 QDate(int y, int m, int d);
65
66 bool isNull() const { return !isValid(); }
67 bool isValid() const { return jd >= minJd() && jd <= maxJd(); }
68
69 int year() const;
70 int month() const;
71 int day() const;
72 int dayOfWeek() const;
73 int dayOfYear() const;
74 int daysInMonth() const;
75 int daysInYear() const;
76 int weekNumber(int *yearNum = 0) const;
77
78#ifndef QT_NO_TEXTDATE
79 static QString shortMonthName(int month, MonthNameType type = DateFormat);
80 static QString shortDayName(int weekday, MonthNameType type = DateFormat);
81 static QString longMonthName(int month, MonthNameType type = DateFormat);
82 static QString longDayName(int weekday, MonthNameType type = DateFormat);
83#endif // QT_NO_TEXTDATE
84#ifndef QT_NO_DATESTRING
85 QString toString(Qt::DateFormat f = Qt::TextDate) const;
86 QString toString(const QString &format) const;
87#endif
88#if QT_DEPRECATED_SINCE(5,0)
89QT_DEPRECATED inline bool setYMD(int y, int m, int d)
90{ if (uint(y) <= 99) y += 1900; return setDate(y, m, d); }
91#endif
92
93 bool setDate(int year, int month, int day);
94
95 void getDate(int *year, int *month, int *day);
96
97 QDate addDays(qint64 days) const;
98 QDate addMonths(int months) const;
99 QDate addYears(int years) const;
100 qint64 daysTo(const QDate &) const;
101
102 bool operator==(const QDate &other) const { return jd == other.jd; }
103 bool operator!=(const QDate &other) const { return jd != other.jd; }
104 bool operator<(const QDate &other) const { return jd < other.jd; }
105 bool operator<=(const QDate &other) const { return jd <= other.jd; }
106 bool operator>(const QDate &other) const { return jd > other.jd; }
107 bool operator>=(const QDate &other) const { return jd >= other.jd; }
108
109 static QDate currentDate();
110#ifndef QT_NO_DATESTRING
111 static QDate fromString(const QString &s, Qt::DateFormat f = Qt::TextDate);
112 static QDate fromString(const QString &s, const QString &format);
113#endif
114 static bool isValid(int y, int m, int d);
115 static bool isLeapYear(int year);
116
117 static inline QDate fromJulianDay(qint64 jd)
118 { QDate d; if (jd >= minJd() && jd <= maxJd()) d.jd = jd; return d; }
119 inline qint64 toJulianDay() const { return jd; }
120
121private:
122 static inline qint64 nullJd() { return std::numeric_limits<qint64>::min(); }
123 static inline qint64 minJd() { return Q_INT64_C(-784350574879); }
124 static inline qint64 maxJd() { return Q_INT64_C( 784354017364); }
125
126 qint64 jd;
127
128 friend class QDateTime;
129 friend class QDateTimePrivate;
130#ifndef QT_NO_DATASTREAM
131 friend Q_CORE_EXPORT QDataStream &operator<<(QDataStream &, const QDate &);
132 friend Q_CORE_EXPORT QDataStream &operator>>(QDataStream &, QDate &);
133#endif
134};
135Q_DECLARE_TYPEINFO(QDate, Q_MOVABLE_TYPE);

137class Q_CORE_EXPORT QTime
138{
139public:
140 QTime(): mds(NullTime)
141#if defined(Q_OS_WINCE)
142 , startTick(NullTime)
143#endif
144 {}
145 QTime(int h, int m, int s = 0, int ms = 0);
146
147 bool isNull() const { return mds == NullTime; }
148 bool isValid() const;
149
150 int hour() const;
151 int minute() const;
152 int second() const;
153 int msec() const;
154#ifndef QT_NO_DATESTRING
155 QString toString(Qt::DateFormat f = Qt::TextDate) const;
156 QString toString(const QString &format) const;
157#endif
158 bool setHMS(int h, int m, int s, int ms = 0);
159
160 QTime addSecs(int secs) const;
161 int secsTo(const QTime &) const;
162 QTime addMSecs(int ms) const;
163 int msecsTo(const QTime &) const;
164
165 bool operator==(const QTime &other) const { return mds == other.mds; }
166 bool operator!=(const QTime &other) const { return mds != other.mds; }
167 bool operator<(const QTime &other) const { return mds < other.mds; }
168 bool operator<=(const QTime &other) const { return mds <= other.mds; }
169 bool operator>(const QTime &other) const { return mds > other.mds; }
170 bool operator>=(const QTime &other) const { return mds >= other.mds; }
171
172 static inline QTime fromMSecsSinceStartOfDay(int msecs) { QTime t; t.mds = msecs; return t; }
173 inline int msecsSinceStartOfDay() const { return mds == NullTime ? 0 : mds; }
174
175 static QTime currentTime();
176#ifndef QT_NO_DATESTRING
177 static QTime fromString(const QString &s, Qt::DateFormat f = Qt::TextDate);
178 static QTime fromString(const QString &s, const QString &format);
179#endif
180 static bool isValid(int h, int m, int s, int ms = 0);
181
182 void start();
183 int restart();
184 int elapsed() const;
185private:
186 enum TimeFlag { NullTime = -1 };
187 inline int ds() const { return mds == -1 ? 0 : mds; }
188 int mds;
189#if defined(Q_OS_WINCE)
190 int startTick;
191#endif
192
193 friend class QDateTime;
194 friend class QDateTimePrivate;
195#ifndef QT_NO_DATASTREAM
196 friend Q_CORE_EXPORT QDataStream &operator<<(QDataStream &, const QTime &);
197 friend Q_CORE_EXPORT QDataStream &operator>>(QDataStream &, QTime &);
198#endif
199};
