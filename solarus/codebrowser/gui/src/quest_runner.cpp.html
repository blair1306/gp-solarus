26/**
27 * @brief Creates a quest runner.
28 * @param parent The parent object of the thread.
29 */
30QuestRunner::QuestRunner(QObject* parent) :
31 QObject(parent),
32 process(this),
33 last_command_id(-1) {
34
35 // Connect to QProcess signals to know when the quest is running and finished.
36 connect(&process, SIGNAL(started()),
37 this, SIGNAL(running()));
38 connect(&process, SIGNAL(finished(int)),
39 this, SLOT(on_finished()));
40 connect(&process, SIGNAL(error(QProcess::ProcessError)),
41 this, SLOT(on_finished())); // TODO report the error
42 connect(&process, SIGNAL(readyReadStandardOutput()),
43 this, SLOT(standard_output_data_available()));
44
45 // Workaround to make the quest process close properly instead of hanging
46 // while reading on its stdin on windows.
47 QTimer* timer = new QTimer(this);
48 connect(timer, &QTimer::timeout, [this] () {
49 if (is_started()) {
50 process.write("\n");
51 }
52 });
53 timer->start(100);
54}
55
56/**
57 * @brief Destroys the quest runner.
58 *
59 * If a quest is running, terminates it.
60 */
61QuestRunner::~QuestRunner() {
62
63 if (process.state() == QProcess::Running) {
64 // Give a chance to the quest process to finish properly.
65 process.terminate();
66 if (!process.waitForFinished(1000)) {
67 // Kill it after a delay.
68 process.kill();
69 }
70 }
71}

73/**
74 * @brief Creates and returns the list of arguments to pass to the process.
75 * @param quest_path The path of the quest to run.
76 */
77QStringList QuestRunner::create_arguments(const QString& quest_path) const {
78
79 QStringList arguments;
80
81 Settings settings;
82
83 // -run quest_path
84 arguments << "-run";
85 arguments << quest_path;
86
87 // no-audio
88 if (settings.value("no_audio", false).toBool()) {
89 arguments << "-no-audio";
90 }
91
92 // video-acceleration
93 const bool video_acceleration =
94 settings.value("video_acceleration", true).toBool();
95 arguments << "-video-acceleration=" + QString(video_acceleration ? "yes" : "no");
96
97 // quest-size
98 const QSize size = settings.value("quest_size").toSize();
99 if (size.isValid()) {
100 QString size_str = QString::number(size.width()) + "x" +
101 QString::number(size.height());
102 arguments << "-quest-size=" + size_str;
103 }
104
105 // Path of the quest.
106 arguments << quest_path;
107
108 return arguments;
109}

111/**
112 * @brief Returns whether the quest is started.
113 *
114 * The quest is started as soon as you call start(),
115 * but it then takes a slight delay for the process to actually load and run.
116 * The signal running() is emitted when the process is running.
117 *
118 * @return @c true if the quest is started.
119 */
120bool QuestRunner::is_started() const {
121
122 return process.state() != QProcess::NotRunning;
123}
124
125/**
126 * @brief Returns whether the quest is running.
127 * @return @c true if the quest is running.
128 */
129bool QuestRunner::is_running() const {
130
131 return process.state() == QProcess::Running;
132}
133
134/**
135 * @brief Runs a specific quest.
136 * @param quest_path The path of the quest to run.
137 * Does nothing if the path is empty or if a quest is already running.
138 *
139 * This function returns immediately.
140 * The signal running() is emitted when the process actually runs.
141 */
142void QuestRunner::start(const QString& quest_path) {
143
144 if (quest_path.isEmpty()) {
145 return;
146 }
147
148 if (is_started()) {
149 return;
150 }
151
152 // Run the current executable itself with the special option "-run quest_path".
153 QStringList editor_arguments = QApplication::arguments();
154 if (editor_arguments.isEmpty()) {
155 QMessageBox::warning(
156 nullptr,
157 tr("Failed to run quest"),
158 tr("Cannot start quest process: no program name")
159 );
160 }
161 QString program_name = editor_arguments.at(0);
162 QStringList arguments = create_arguments(quest_path);
163
164 process.start(program_name, arguments);
165
166}

168/**
169 * @brief Stops the current quest.
170 *
171 * Returns immediately.
172 * The signal finished() is emitted when the process is finished.
173 */
174void QuestRunner::stop() {
175
176 if (is_started()) {
177 process.terminate();
178 }
179}

181/**
182 * @brief Slot called when lines are written on the standard output of the
183 * quest process.
184 */
185void QuestRunner::standard_output_data_available() {
186
187 // Read the UTF-8 data available.
188 QStringList lines;
189 QByteArray line_utf8 = process.readLine();
190 while (!line_utf8.isEmpty()) {
191 QString line(line_utf8);
192 line = line.trimmed(); // Remove the final '\n'.
193 if (!line.isEmpty()) {
194 lines << line;
195 }
196 line_utf8 = process.readLine();
197 }
198
199 if (!lines.isEmpty()) {
200 emit output_produced(lines);
201 }
202}

204/**
205 * @brief Executes some Lua code in the quest process.
206 * @param command The Lua code.
207 * @return The id of the command executed, or -1 if it could not be sent
208 * to the process.
209 */
210int QuestRunner::execute_command(const QString& command) {
211
212 if (!is_running()) {
213 return -1;
214 }
215
216 if (command.isEmpty()) {
217 return -1;
218 }
219
220 QByteArray command_utf8 = command.toUtf8();
221 command_utf8.append("\n");
222 qint64 bytes_written = process.write(command_utf8);
223 if (bytes_written != command_utf8.size()) {
224 return -1;
225 }
226
227 ++last_command_id;
228 return last_command_id;
229}
230
231/**
232 * @brief Slot called when the execution finishes.
233 */
234void QuestRunner::on_finished() {
235
236 last_command_id = -1;
237 emit finished();
238}
