30 */
31Hero::FallingState::FallingState(Hero& hero):
32 HeroState(hero, "falling") {
33
34}

36/**
37 * \brief Starts this state.
38 * \param previous_state the previous state
39 */
40void Hero::FallingState::start(const State* previous_state) {
41
42 HeroState::start(previous_state);
43
44 get_entity().delayed_teletransporter = nullptr;
45 get_sprites().save_animation_direction();
46 get_sprites().set_animation_falling();
47 Sound::play("hero_falls");
48}
49
50/**
51 * \brief Stops this state.
52 * \param next_state the next state
53 */
54void Hero::FallingState::stop(const State* next_state) {
55
56 HeroState::stop(next_state);
57
58 get_sprites().set_animation_stopped_normal();
59 get_sprites().restore_animation_direction();
60}
61
62/**
63 * \brief Updates this state.
64 */
65void Hero::FallingState::update() {
66
67 HeroState::update();
68
69 Hero& hero = get_entity();
70 if (!is_suspended() && get_sprites().is_animation_finished()) {
71
72 // the hero has just finished falling
73 std::shared_ptr<Teletransporter> teletransporter = hero.get_delayed_teletransporter();
74 if (teletransporter != nullptr) {
75 // special hole with a teletransporter
76 teletransporter->transport_hero(hero);
77 }
78 else {
79 // normal hole that hurts the hero
80 get_equipment().remove_life(2);
81 hero.set_state(new BackToSolidGroundState(hero, true));
82 }
83 }
84}
