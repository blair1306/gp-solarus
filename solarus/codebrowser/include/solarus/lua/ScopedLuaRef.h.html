17#ifndef SOLARUS_SCOPED_LUA_REF_H
18#define SOLARUS_SCOPED_LUA_REF_H
19
20#include "solarus/Common.h"
21#include <string>
22
23struct lua_State;
24
25namespace Solarus {
26
27/**
28 * \brief Holds a Lua ref and does luaL_unref() on destruction.
29 *
30 * This class is meant to make the usage of Lua refs safer, by making sure
31 * that for each luaL_ref() call, there is exactly one luaL_unref() call.
32 * This avoids memory leaks and duplicate luaL_unref() calls.
33 *
34 * It is recommended to use this class rather than calling luaL_unref()
35 * directly.
36 */
37class ScopedLuaRef {
38
39 public:
40
41 ScopedLuaRef();
42 ScopedLuaRef(lua_State* l, int ref);
43 ScopedLuaRef(const ScopedLuaRef& other);
44 ScopedLuaRef(ScopedLuaRef&& other);
45 ~ScopedLuaRef();
46
47 ScopedLuaRef& operator=(const ScopedLuaRef& other);
48 ScopedLuaRef& operator=(ScopedLuaRef&& other);
49
50 lua_State* get_lua_state() const;
51 bool is_empty() const;
52 int get() const;
53 void clear();
54
55 void push() const;
56 void call(const std::string& function_name) const;
57 void clear_and_call(const std::string& function_name);
58
59 private:
60
61 lua_State* l; /**< The Lua state. nullptr means no ref. */
62 int ref; /**< Lua ref to a value. */
63
64};
65
66}
67
68#endif
