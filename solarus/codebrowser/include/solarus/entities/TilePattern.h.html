17#ifndef SOLARUS_TILE_PATTERN_H
18#define SOLARUS_TILE_PATTERN_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Ground.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/lowlevel/Size.h"
24
25namespace Solarus {
26
27class Point;
28class Rectangle;
29class Surface;
30class Tileset;
31
32/**
33 * \brief Abstract class for a tile pattern.
34 *
35 * A tile pattern defines a rectangle image in a tileset and has a ground property.
36 * It may have a special animation.
37 * The width and the height of a tile pattern are always multiples or 8.
38 */
39class TilePattern {
40
41 public:
42
43 virtual ~TilePattern();
44
45 int get_width() const;
46 int get_height() const;
47 const Size& get_size() const;
48 Ground get_ground() const;
49
50 static void initialize();
51 static void quit();
52 static void update();
53 void fill_surface(
54 const SurfacePtr& dst_surface,
55 const Rectangle& dst_position,
56 const Tileset& tileset,
57 const Point& viewport
58 ) const;
59
60 /**
61 * \brief Draws the tile image on a surface.
62 * \param dst_surface The surface to draw.
63 * \param dst_position Position where the tile pattern should be drawn.
64 * \param tileset The tileset of this tile.
65 * \param viewport Coordinates of the top-left corner of dst_surface
66 * relative to the map (may be used for scrolling tiles).
67 */
68 virtual void draw(
69 const SurfacePtr& dst_surface,
70 const Point& dst_position,
71 const Tileset& tileset,
72 const Point& viewport
73 ) const = 0;
74 virtual bool is_animated() const;
75 virtual bool is_drawn_at_its_position() const;
76
77 protected:
78
79 TilePattern(Ground ground, const Size& size);
80
81 const Ground ground; /**< Kind of tile. */
82
83 Size size; /**< Pattern size (multiple of 8). */
84
85};
86
87}
88
89#endif
