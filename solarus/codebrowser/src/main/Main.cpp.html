29/**
30 * \brief Prints the usage of the program.
31 * \param argc number of command-line arguments
32 * \param argv command-line arguments
33 */
34void print_help(const Arguments& args) {
35
36 std::string binary_name = args.get_program_name();
37 if (binary_name.empty()) {
38 binary_name = "solarus";
39 }
40 std::cout << "Usage: " << binary_name << " [options] [quest_path]"
41 << std::endl << std::endl
42 << "The quest path is the name of a directory that contains either the data"
43 << std::endl
44 << "directory or the data archive (data.solarus or data.solarus.zip) of the game to run."
45 << std::endl
46 << "If the quest path is not specified, the default directory will be: '"
47 << SOLARUS_DEFAULT_QUEST << "'."
48 << std::endl
49 << std::endl
50 << "Options:"
51 << std::endl
52 << " -help shows this help message and exits"
53 << std::endl
54 << " -no-audio disables sounds and musics"
55 << std::endl
56 << " -no-video disables displaying"
57 << std::endl
58 << " -video-acceleration=yes|no enables or disables accelerated graphics (default yes)"
59 << std::endl
60 << " -quest-size=<width>x<height> sets the size of the drawing area (if compatible with the quest)"
61 << std::endl
62 << " -lua-console=yes|no accepts standard input lines as Lua commands (default yes)"
63 << std::endl
64 << " -turbo=yes|no runs as fast as possible rather than simulating real time (default no)"
65 << std::endl
66 << " -lag=X slows down each frame of X milliseconds to simulate slower systems for debugging (default 0)"
67 << std::endl;
68}
69
70} // Anonymous namespace.
71
72} // namespace Solarus.

74/**
75 * \brief Usual entry point of the program.
76 *
77 * Usage: solarus [options] [quest_path]
78 *
79 * The quest path is the name of a directory that contains either the data
80 * directory ("data") or the data archive ("data.solarus" or
81 * "data.solarus.zip").
82 * If the quest path is not specified, it is set to the preprocessor constant
83 * SOLARUS_DEFAULT_QUEST, which is the current directory "." by default.
84 * In all cases, this quest path is relative to the working directory,
85 * or to the solarus executable directory if no quest is found in the working
86 * directory.
87 *
88 * The following options are supported:
89 * -help Shows a help message.
90 * -no-audio Disables sounds and musics.
91 * -no-video Disables displaying (used for unit tests).
92 * -video-acceleration=yes|no Enables or disables 2D accelerated graphics if available (default: yes).
93 * -quest-size=<width>x<height> Sets the size of the drawing area (if compatible with the quest).
94 * -lua-console=yes|no Accepts lines from standard input as Lua commands (default: yes).
95 * -turbo=yes|no Runs as fast as possible rather than simulating real time (default: no).
96 * -lag=X (Advanced) Artificially slows down each frame of X milliseconds
97 * to simulate slower systems for debugging (default: 0).
98 *
99 * \param argc Number of command-line arguments.
100 * \param argv Command-line arguments.
101 */
102int main(int argc, char** argv) {
103
104 using namespace Solarus;
105
106 Debug::set_abort_on_die(true); // Better for debugging (get a callstack).
107
108 // Store the command-line arguments.
109 const Arguments args(argc, argv);
110
111 // Check the -help option.
112 if (args.has_argument("-help")) {
113 // Print a help message.
114 print_help(args);
115 }
116 else {
117 // Run the main loop.
118 MainLoop(args).run();
119 }
120
121 return 0;
122}
