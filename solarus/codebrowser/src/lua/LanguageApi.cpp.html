25/**
26 * Name of the Lua table representing the language module.
27 */
28const std::string LuaContext::language_module_name = "sol.language";
29
30/**
31 * \brief Pushes a dialog onto the stack.
32 * \param l A Lua context.
33 * \param dialog A dialog.
34 */
35void LuaContext::push_dialog(lua_State* l, const Dialog& dialog) {
36
37 lua_newtable(l);
38
39 // Dialog id.
40 push_string(l, dialog.get_id());
41 lua_setfield(l, -2, "id");
42
43 // Text.
44 push_string(l, dialog.get_text());
45 lua_setfield(l, -2, "text");
46
47 // User properties.
48 const std::map<std::string, std::string>& properties =
49 dialog.get_properties();
50 for (const auto& kvp: properties) {
51 push_string(l, kvp.second);
52 lua_setfield(l, -2, kvp.first.c_str());
53 }
54}

56/**
57 * \brief Initializes the language features provided to Lua.
58 */
59void LuaContext::register_language_module() {
60
61 static const luaL_Reg functions[] = {
62 { "get_language", language_api_get_language },
63 { "set_language", language_api_set_language },
64 { "get_language_name", language_api_get_language_name },
65 { "get_languages", language_api_get_languages },
66 { "get_string", language_api_get_string },
67 { "get_dialog", language_api_get_dialog },
68 { nullptr, nullptr }
69 };
70
71 register_functions(language_module_name, functions);
72}

74/**
75 * \brief Implementation of sol.language.get_language().
76 * \param l the Lua context that is calling this function
77 * \return number of values to return to Lua
78 */
79int LuaContext::language_api_get_language(lua_State* l) {
80
81 return LuaTools::exception_boundary_handle(l, [&] {
82 const std::string& language = CurrentQuest::get_language();
83
84 if (language.empty()) { // Return nil if no language is set.
85 lua_pushnil(l);
86 }
87 else { // Return the language code.
88 push_string(l, language);
89 }
90 return 1;
91 });
92}

94/**
95 * \brief Implementation of sol.language.set_language().
96 * \param l the Lua context that is calling this function
97 * \return number of values to return to Lua
98 */
99int LuaContext::language_api_set_language(lua_State* l) {
100
101 return LuaTools::exception_boundary_handle(l, [&] {
102 const std::string& language_code = LuaTools::check_string(l, 1);
103
104 if (!CurrentQuest::has_language(language_code)) {
105 LuaTools::arg_error(l, 1, std::string("No such language: '") + language_code + "'");
106 }
107 CurrentQuest::set_language(language_code);
108
109 return 0;
110 });
111}

113/**
114 * \brief Implementation of sol.language.get_language_name().
115 * \param l the Lua context that is calling this function
116 * \return number of values to return to Lua
117 */
118int LuaContext::language_api_get_language_name(lua_State* l) {
119
120 return LuaTools::exception_boundary_handle(l, [&] {
121 std::string language_code;
122 if (lua_gettop(l) >= 1) {
123 language_code = LuaTools::check_string(l, 1);
124 if (!CurrentQuest::has_language(language_code)) {
125 LuaTools::arg_error(l, 1, std::string("No such language: '") + language_code + "'");
126 }
127 }
128 else {
129 language_code = CurrentQuest::get_language();
130 if (language_code.empty()) {
131 LuaTools::error(l, "No language is set");
132 }
133 }
134
135 const std::string& name = CurrentQuest::get_language_name(language_code);
136 push_string(l, name);
137
138 return 1;
139 });
140}

142/**
143 * \brief Implementation of sol.language.get_languages().
144 * \param l The Lua context that is calling this function.
145 * \return Number of values to return to Lua.
146 */
147int LuaContext::language_api_get_languages(lua_State* l) {
148
149 return LuaTools::exception_boundary_handle(l, [&] {
150 const std::map<std::string, std::string>& languages =
151 CurrentQuest::get_resources(ResourceType::LANGUAGE);
152
153 lua_newtable(l);
154 int i = 1;
155 for (const auto& kvp: languages) {
156 const std::string& language_code = kvp.first;
157 push_string(l, language_code);
158 lua_rawseti(l, -2, i);
159 ++i;
160 }
161
162 return 1;
163 });
164}

166/**
167 * \brief Implementation of sol.language.get_string().
168 * \param l the Lua context that is calling this function
169 * \return number of values to return to Lua
170 */
171int LuaContext::language_api_get_string(lua_State* l) {
172
173 return LuaTools::exception_boundary_handle(l, [&] {
174 const std::string& key = LuaTools::check_string(l, 1);
175
176 if (!CurrentQuest::string_exists(key)) {
177 lua_pushnil(l);
178 }
179 else {
180 push_string(l, CurrentQuest::get_string(key));
181 }
182 return 1;
183 });
184}

197 lua_pushnil(l);
