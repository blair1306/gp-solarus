42#ifndef QTIMER_H
43#define QTIMER_H
44
45#include <QtCore/qglobal.h>
46
47#ifndef QT_NO_QOBJECT
48
49#include <QtCore/qbasictimer.h> // conceptual inheritance
50#include <QtCore/qobject.h>
51
52QT_BEGIN_NAMESPACE
53
54
55class Q_CORE_EXPORT QTimer : public QObject
56{
57 Q_OBJECT
58 Q_PROPERTY(bool singleShot READ isSingleShot WRITE setSingleShot)
59 Q_PROPERTY(int interval READ interval WRITE setInterval)
60 Q_PROPERTY(int remainingTime READ remainingTime)
61 Q_PROPERTY(Qt::TimerType timerType READ timerType WRITE setTimerType)
62 Q_PROPERTY(bool active READ isActive)
63public:
64 explicit QTimer(QObject *parent = 0);
65 ~QTimer();
66
67 inline bool isActive() const { return id >= 0; }
68 int timerId() const { return id; }
69
70 void setInterval(int msec);
71 int interval() const { return inter; }
72
73 int remainingTime() const;
74
75 void setTimerType(Qt::TimerType atype) { this->type = atype; }
76 Qt::TimerType timerType() const { return Qt::TimerType(type); }
77
78 inline void setSingleShot(bool singleShot);
79 inline bool isSingleShot() const { return single; }
80
81 static void singleShot(int msec, const QObject *receiver, const char *member);
82 static void singleShot(int msec, Qt::TimerType timerType, const QObject *receiver, const char *member);
83
84public Q_SLOTS:
85 void start(int msec);
86
87 void start();
88 void stop();
89
90Q_SIGNALS:
91 void timeout(
92#if !defined(Q_QDOC)
93 QPrivateSignal
94#endif
95 );
96
97protected:
98 void timerEvent(QTimerEvent *);
99
100private:
101 Q_DISABLE_COPY(QTimer)
102
103 inline int startTimer(int){ return -1;}
104 inline void killTimer(int){}
105
106 int id, inter, del;
107 uint single : 1;
108 uint nulltimer : 1;
109 uint type : 2;
110 // reserved : 28
111};
112
113inline void QTimer::setSingleShot(bool asingleShot) { single = asingleShot; }
114
115QT_END_NAMESPACE
116
117#endif // QT_NO_QOBJECT
118
119#endif // QTIMER_H
