73#ifndef QT_NO_CURSOR
74
75class QCursorData;
76class QBitmap;
77class QPixmap;
78
79
80class Q_GUI_EXPORT QCursor
81{
82public:
83 QCursor();
84 QCursor(Qt::CursorShape shape);
85 QCursor(const QBitmap &bitmap, const QBitmap &mask, int hotX=-1, int hotY=-1);
86 QCursor(const QPixmap &pixmap, int hotX=-1, int hotY=-1);
87 QCursor(const QCursor &cursor);
88 ~QCursor();
89 QCursor &operator=(const QCursor &cursor);
90#ifdef Q_COMPILER_RVALUE_REFS
91 inline QCursor &operator=(QCursor &&other)
92 { qSwap(d, other.d); return *this; }
93#endif
94 operator QVariant() const;
95
96 Qt::CursorShape shape() const;
97 void setShape(Qt::CursorShape newShape);
98
99 const QBitmap *bitmap() const;
100 const QBitmap *mask() const;
101 QPixmap pixmap() const;
102 QPoint hotSpot() const;
103
104 static QPoint pos();
105 static QPoint pos(const QScreen *screen);
106 static void setPos(int x, int y);
107 static void setPos(QScreen *screen, int x, int y);
108 inline static void setPos(const QPoint &p) { setPos(p.x(), p.y()); }
109 inline static void setPos(QScreen *screen, const QPoint &p) { setPos(screen, p.x(), p.y()); }
110
111private:
112 QCursorData *d;
113};
