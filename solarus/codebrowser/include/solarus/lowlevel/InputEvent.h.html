35/**
36 * \brief Represents a low-level event.
37 *
38 * This low-level class provides an API for input events (keyboard, joypad and window events).
39 * It encapsulates the library-dependent events.

187 /**
188 * Enumeration of mouse buttons.
189 * This is a Solarus dedicated enumeration.
190 * To make the SDL encapsulation easier, this enumeration is compatible
191 * with SDL_BUTTON definitions from SDL.
192 * If you intend to replace SDL by another library one day, the API of the
193 * Input class will not have to change
194 * (only its implementation will).
195 */
196 enum class MouseButton {
197 NONE = -1,
198 LEFT = SDL_BUTTON_LEFT,
199 MIDDLE = SDL_BUTTON_MIDDLE,
200 RIGHT = SDL_BUTTON_RIGHT,
201 X1 = SDL_BUTTON_X1,
202 X2 = SDL_BUTTON_X2,
203 };
