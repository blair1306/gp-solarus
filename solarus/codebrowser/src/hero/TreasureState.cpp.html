31/**
32 * \brief Constructor.
33 * \param hero The hero controlled by this state.
34 * \param treasure The treasure to give to the hero. It must be obtainable.
35 * \param callback_ref Lua ref to a function to call when the
36 * treasure's dialog finishes (possibly an empty ref).
37 */
38Hero::TreasureState::TreasureState(
39 Hero& hero,
40 const Treasure& treasure,
41 const ScopedLuaRef& callback_ref
42):
43
44 HeroState(hero, "treasure"),
45 treasure(treasure),
46 treasure_sprite(),
47 callback_ref(callback_ref) {
48
49 treasure.check_obtainable();
50 treasure_sprite = treasure.create_sprite();
51}
52
53/**
54 * \brief Starts this state.
55 * \param previous_state the previous state
56 */
57void Hero::TreasureState::start(const State* previous_state) {
58
59 HeroState::start(previous_state);
60
61 // Show the animation.
62 get_sprites().save_animation_direction();
63 get_sprites().set_animation_brandish();
64
65 // Play the sound.
66 const std::string& sound_id = treasure.get_item().get_sound_when_brandished();
67 if (!sound_id.empty()) {
68 Sound::play(sound_id);
69 }
70
71 // Give the treasure.
72 treasure.give_to_player();
73
74 // Show a dialog (Lua does the job after this).
75 ScopedLuaRef callback_ref = this->callback_ref;
76 this->callback_ref.clear();
77 get_lua_context().notify_hero_brandish_treasure(treasure, callback_ref);
78}
79
80/**
81 * \brief Stops this state.
82 * \param next_state the next state
83 */
84void Hero::TreasureState::stop(const State* next_state) {
85
86 HeroState::stop(next_state);
87
88 // Restore the sprite's direction.
89 get_sprites().restore_animation_direction();
90 callback_ref.clear();
91}
92
93/**
94 * \copydoc Hero::HeroState::update
95 */
96void Hero::TreasureState::update() {
97
98 HeroState::update();
99
100 treasure_sprite->update();
101}
102
103/**
104 * \brief Draws this state.
105 */
106void Hero::TreasureState::draw_on_map() {
107
108 HeroState::draw_on_map();
109
110 const Hero& hero = get_entity();
111 int x = hero.get_x();
112 int y = hero.get_y();
113
114 const CameraPtr& camera = get_map().get_camera();
115 if (camera == nullptr) {
116 return;
117 }
118 treasure_sprite->draw(get_map().get_camera_surface(),
119 x - camera->get_top_left_x(),
120 y - 24 - camera->get_top_left_y());
121}

123/**
124 * \copydoc Entity::State::get_previous_carried_object_behavior
125 */
126CarriedObject::Behavior Hero::TreasureState::get_previous_carried_object_behavior() const {
127 return CarriedObject::BEHAVIOR_DESTROY;
128}
129
130/**
131 * \brief Returns whether the hero is brandishing a treasure in this state.
132 * \return \c true if the hero is brandishing a treasure.
133 */
134bool Hero::TreasureState::is_brandishing_treasure() const {
135 return true;
136}
