26/**
27 * \brief Creates an Ogg decoder.
28 */
29OggDecoder::OggDecoder():
30 ogg_file(),
31 ogg_mem(),
32 ogg_info(nullptr),
33 loop_start_pcm(-1),
34 loop_end_pcm(-1) {
35
36}
37
38/**
39 * \brief Loads an OGG file from the memory.
40 * \param ogg_data The memory area to read.
41 * \param loop Whether the music should loop if reaching the end.
42 * \return \c true in case of success.
43 */
44bool OggDecoder::load(std::string&& ogg_data, bool loop) {
45
46 ogg_file = OggFileUniquePtr(new OggVorbis_File());
47
48 ogg_mem.position = 0;
49 ogg_mem.loop = loop;
50 ogg_mem.data = ogg_data;
51 // Now, ogg_mem contains the encoded data.
52
53 int error = ov_open_callbacks(&ogg_mem, ogg_file.get(), nullptr, 0, Sound::ogg_callbacks);
54
55 if (error != 0) {
56 return false;
57 }
58
59 ogg_info = ov_info(ogg_file.get(), -1);
60
61 // Read loop info from the header.
62 vorbis_comment* header = ov_comment(ogg_file.get(), -1);
63 if (header == nullptr) {
64 return false;
65 }
66
67 const char* loop_start_string = vorbis_comment_query(header, "LOOPSTART", 0);
68 const char* loop_end_string = vorbis_comment_query(header, "LOOPEND", 0);
69 const char* loop_length_string = vorbis_comment_query(header, "LOOPLENGTH", 0);
70
71 if (loop_start_string != nullptr) {
72 std::istringstream iss(loop_start_string);
73 ogg_int64_t pcm = -1;
74 if (iss >> pcm && pcm >= 0) {
75 // There is a loop.
76 loop_start_pcm = pcm;
77 loop_end_pcm = ov_pcm_total(ogg_file.get(), -1); // Default loop end value.
78 }
79
80 if (loop_end_string != nullptr) {
81 iss.clear();
82 iss.str(loop_end_string);
83 if (iss >> pcm && pcm > 0) {
84 // There is a loop end info.
85 loop_end_pcm = pcm;
86 }
87 }
88 else if (loop_length_string != nullptr) {
89 iss.clear();
90 iss.str(loop_length_string);
91 if (iss >> pcm && pcm > 0) {
92 // There is a loop end info from a length value.
93 loop_end_pcm = loop_start_pcm + pcm;
94 }
95 }
96 }
97
98 return true;
99}

101/**
102 * \brief Unloads the OGG data from memory.
103 */
104void OggDecoder::unload() {
105 ogg_file = nullptr;
106 ogg_mem.data.clear();
107 ogg_info = nullptr;
108 loop_start_pcm = -1;
109 loop_end_pcm = -1;
110}
111
112/**
113 * \brief Decodes a chunk of the previously loaded OGG data into PCM data
114 * and plays it.
115 * \param decoded_data Pointer to where you want the decoded data to be written.
116 * \param nb_samples Number of samples to write.
117 */
118void OggDecoder::decode(ALuint destination_buffer, ALsizei nb_samples) {
119
120 if (ogg_info == nullptr) {
121 return;
122 }
123
124 // Read the encoded music properties.
125 const int num_channels = ogg_info->channels;
126 ALsizei sample_rate = ALsizei(ogg_info->rate);
127 ALenum al_format = AL_NONE;
128 if (num_channels == 1) {
129 al_format = AL_FORMAT_MONO16;
130 }
131 else if (num_channels == 2) {
132 al_format = AL_FORMAT_STEREO16;
133 }
134
135 const ogg_int64_t loop_end_byte = loop_end_pcm * num_channels * sizeof(ALshort);
136
137 // Decode the OGG data.
138 std::vector<ALshort> raw_data(nb_samples * num_channels);
139 int bitstream = 0;
140 long bytes_read = 0;
141 long total_bytes_read = 0;
142 long remaining_bytes = nb_samples * num_channels * sizeof(ALshort);
143
144 do {
145 long max_bytes_to_read = remaining_bytes;
146 ogg_int64_t current_pcm = ov_pcm_tell(ogg_file.get());
147 ogg_int64_t current_byte = current_pcm * num_channels * sizeof(ALshort);
148
149 if (loop_end_pcm != -1 &&
150 current_byte + max_bytes_to_read > loop_end_byte) {
151 // Don't read after the loop point.
152 max_bytes_to_read = (long) loop_end_byte - current_byte;
153 }
154
155 bytes_read = ov_read(
156 ogg_file.get(),
157 ((char*) raw_data.data()) + total_bytes_read,
158 max_bytes_to_read,
159 0,
160 2,
161 1,
162 &bitstream
163 );
164
165 if (bytes_read < 0) {
166 if (bytes_read != OV_HOLE) { // OV_HOLE is normal when the music loops
167 std::ostringstream oss;
168 oss << "Error while decoding ogg chunk: " << bytes_read;
169 Debug::error(oss.str());
170 return;
171 }
172 }
173 else {
174 total_bytes_read += bytes_read;
175 remaining_bytes -= bytes_read;
176
177 // Check if we should loop now.
178 current_pcm = ov_pcm_tell(ogg_file.get());
179
180 if (loop_end_pcm != -1 &&
181 loop_start_pcm != -1 &&
182 current_pcm == loop_end_pcm) {
183 int error = ov_pcm_seek(ogg_file.get(), loop_start_pcm);
184 if (error != 0) {
185 std::ostringstream oss;
186 oss << "Failed to loop in OGG file: error " << error;
187 Debug::error(oss.str());
188 }
189 }
190 }
191 }
192 while (remaining_bytes > 0 && bytes_read > 0);
193
194 // Put this decoded data into the buffer.
195 alBufferData(destination_buffer, al_format, raw_data.data(), ALsizei(total_bytes_read), sample_rate);
196
197 int error = alGetError();
198 if (error != AL_NO_ERROR) {
199 std::ostringstream oss;
200 oss << "Failed to fill the audio buffer with decoded OGG data: error " << error;
201 Debug::error(oss.str());
202 }
203}
