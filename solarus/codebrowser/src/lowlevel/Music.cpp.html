52/**
53 * \brief Creates an empty music.
54 */
55Music::Music():
56 id(none),
57 format(NO_FORMAT),
58 loop(false),
59 callback_ref(),
60 source(AL_NONE) {
61
62 for (int i = 0; i < nb_buffers; i++) {
63 buffers[i] = AL_NONE;
64 }
65}

67/**
68 * \brief Creates a new music.
69 * \param music_id Id of the music (file name without extension).
70 * \param loop Whether the music should loop when reaching its end.
71 * \param callback_ref Lua function to call when the music ends
72 * or an empty ref.
73 * There cannot be both a loop and a callback at the same time.
74 */
75Music::Music(
76 const std::string& music_id,
77 bool loop,
78 const ScopedLuaRef& callback_ref):
79 id(music_id),
80 format(OGG),
81 loop(loop),
82 callback_ref(callback_ref),
83 source(AL_NONE) {
84
85 Debug::check_assertion(!loop || callback_ref.is_empty(),
86 "Attempt to set both a loop and a callback to music"
87 );
88
89 for (int i = 0; i < nb_buffers; i++) {
90 buffers[i] = AL_NONE;
91 }
92}

94/**
95 * \brief Initializes the music system.
96 */
97void Music::initialize() {
98
99 // initialize the decoding features
100 spc_decoder = std::unique_ptr<SpcDecoder>(new SpcDecoder());
101 it_decoder = std::unique_ptr<ItDecoder>(new ItDecoder());
102 ogg_decoder = std::unique_ptr<OggDecoder>(new OggDecoder());
103
104 set_volume(100);
105}

107/**
108 * \brief Exits the music system.
109 */
110void Music::quit() {
111
112 if (is_initialized()) {
113 current_music = nullptr;
114 spc_decoder = nullptr;
115 it_decoder = nullptr;
116 volume = 1.0;
117 }
118}

150/**
151 * \brief Sets the volume of musics.
152 * \param volume the new volume (0 to 100)
153 */
154void Music::set_volume(int volume) {
155
156 volume = std::min(100, std::max(0, volume));
157 Music::volume = volume / 100.0;
158
159 if (current_music != nullptr) {
160 alSourcef(current_music->source, AL_GAIN, Music::volume);
161 }
162
163 Logger::info(std::string("Music volume: ") + String::to_string(get_volume()));
164}

244 * \brief Returns the id of the music currently playing.
245 * \return the id of the current music, or "none" if no music is being played

247const std::string& Music::get_current_music_id() {
248 return current_music != nullptr ? current_music->id : none;

251/**
252 * \brief Tries to find a music file from a music id.
253 * \param music_id Id of the music to find (file name without
254 * directory or extension). Cannot be Music::none or Music::unchanged.
255 * \param file_name Resulting file name with its extension
256 * (empty string if not found).
257 * \param format Resulting music format.
258 */
259void Music::find_music_file(const std::string& music_id,
260 std::string& file_name, Format& format) {
261
262 file_name = "";
263 format = OGG;
264
265 std::string file_name_start = std::string("musics/" + music_id);
266 if (QuestFiles::data_file_exists(file_name_start + ".ogg")) {
267 format = OGG;
268 file_name = file_name_start + ".ogg";
269 }
270 else if (QuestFiles::data_file_exists(file_name_start + ".it")) {
271 format = IT;
272 file_name = file_name_start + ".it";
273 }
274 else if (QuestFiles::data_file_exists(file_name_start + ".spc")) {
275 format = SPC;
276 file_name = file_name_start + ".spc";
277 }
278}

280/**
281 * \brief Returns whether a music exists.
282 * \param music_id Id of the music to test. Music::none and Music::unchanged
283 * are also considered valid.
284 * \return true If this music exists.
285 */
286bool Music::exists(const std::string& music_id) {
287
288 if (music_id == none || music_id == unchanged) {
289 return true;
290 }
291
292 std::string file_name;
293 Format format;
294 find_music_file(music_id, file_name, format);
295
296 return !file_name.empty();
297}

299/**
300 * \brief Plays a music.
301 *
302 * If the music is different from the current one, the current one is stopped.
303 * The music specified can also be Music::none_id (then the current music is
304 * just stopped).
305 * or even Music::unchanged_id (the current music continues in this case).
306 *
307 * \param music_id Id of the music to play (file name without extension).
308 * \param loop Whether the music should loop when reaching its end
309 * (if there is an end).
310 */
311void Music::play(const std::string& music_id, bool loop) {
312
313 play(music_id, loop, ScopedLuaRef());
314}
315
316/**
317 * \brief Plays a music with an optional Lua callback to call when it finishes.
318 * \param music_id Id of the music to play (file name without extension).
319 * \param loop Whether the music should loop when reaching its end
320 * (if there is an end).
321 * \param callback_ref Lua function to call when the music finishes or
322 * an empty ref. There cannot be both a loop and a callback at the same time.
323 */
324void Music::play(
325 const std::string& music_id,
326 bool loop,
327 const ScopedLuaRef& callback_ref
328) {
329 if (music_id != unchanged && music_id != get_current_music_id()) {
330 // The music is changed.
331
332 if (current_music != nullptr) {
333 // Stop the music that was played.
334 current_music->stop();
335 current_music = nullptr;
336 }
337
338 if (music_id != none) {
339 // Play another music.
340 current_music = std::unique_ptr<Music>(
341 new Music(music_id, loop, callback_ref)
342 );
343 if (!current_music->start()) {
344 // Could not play the music.
345 current_music = nullptr;
346 }
347 }
348 }
349}

361/**
362 * \brief Updates the music system.
363 *
364 * When a music is playing, this function continues the streaming.
365 */
366void Music::update() {
367
368 if (!is_initialized()) {
369 return;
370 }
371
372 if (current_music != nullptr) {
373 bool playing = current_music->update_playing();
374 if (!playing) {
375 // Music is finished.
376 ScopedLuaRef callback_ref = current_music->callback_ref;
377 current_music->stop();
378 current_music = nullptr;
379 callback_ref.call("music callback");
380 }
381 }
382}

384/**
385 * \brief Updates this music when it is playing.
386 *
387 * This function handles the double buffering.
388 *
389 * \return \c true if the music keeps playing, \c false if the end is reached.
390 */
391bool Music::update_playing() {
392
393 // Get the empty buffers.
394 ALint nb_empty;
395 alGetSourcei(source, AL_BUFFERS_PROCESSED, &nb_empty);
396
397 // Refill them.
398 for (int i = 0; i < nb_empty; i++) {
399 ALuint buffer;
400 alSourceUnqueueBuffers(source, 1, &buffer); // Unqueue the buffer.
401
402 // Fill it by decoding more data.
403 switch (format) {
404
405 case SPC:
406 decode_spc(buffer, 16384);
407 break;
408
409 case IT:
410 decode_it(buffer, 16384);
411 break;
412
413 case OGG:
414 decode_ogg(buffer, 16384);
415 break;
416
417 case NO_FORMAT:
418 Debug::die("Invalid music format");
419 break;
420 }
421
422 alSourceQueueBuffers(source, 1, &buffer); // Queue it again.
423 }
424
425 // Check whether there is still something playing.
426 ALint status;
427 alGetSourcei(source, AL_SOURCE_STATE, &status);
428 if (status != AL_PLAYING) {
429 // The end of the file is reached, or we need to decode more data.
430 alSourcePlay(source);
431 }
432
433 alGetSourcei(source, AL_SOURCE_STATE, &status);
434 return status == AL_PLAYING;
435}

437/**
438 * \brief Decodes a chunk of SPC data into PCM data for the current music.
439 * \param destination_buffer the destination buffer to write
440 * \param nb_samples number of samples to write
441 */
442void Music::decode_spc(ALuint destination_buffer, ALsizei nb_samples) {
443
444 // decode the SPC data
445 std::vector<ALushort> raw_data(nb_samples);
446 spc_decoder->decode((int16_t*) raw_data.data(), nb_samples);
447
448 // put this decoded data into the buffer
449 alBufferData(destination_buffer, AL_FORMAT_STEREO16, raw_data.data(), nb_samples * 2, 32000);
450
451 int error = alGetError();
452 if (error != AL_NO_ERROR) {
453 std::ostringstream oss;
454 oss << "Failed to fill the audio buffer with decoded SPC data for music file '"
455 << file_name << ": error " << error;
456 Debug::error(oss.str());
457 }
458}
459
460/**
461 * \brief Decodes a chunk of IT data into PCM data for the current music.
462 * \param destination_buffer the destination buffer to write
463 * \param nb_samples number of samples to write
464 */
465void Music::decode_it(ALuint destination_buffer, ALsizei nb_samples) {
466
467 // Decode the IT data.
468 std::vector<ALushort> raw_data(nb_samples);
469 int bytes_read = it_decoder->decode(raw_data.data(), nb_samples);
470
471 if (bytes_read == 0) {
472 // End of file.
473 alBufferData(destination_buffer, AL_FORMAT_STEREO16, raw_data.data(), 0, 44100);
474 }
475 else {
476 // Put this decoded data into the buffer.
477 alBufferData(destination_buffer, AL_FORMAT_STEREO16, raw_data.data(), nb_samples, 44100);
478 }
479 int error = alGetError();
480 if (error != AL_NO_ERROR) {
481 std::ostringstream oss;
482 oss << "Failed to fill the audio buffer with decoded IT data for music file '"
483 << file_name << ": error " << error;
484 Debug::error(oss.str());
485 }
486}

488/**

493void Music::decode_ogg(ALuint destination_buffer, ALsizei nb_samples) {

498/**
499 * \brief Loads the file and starts playing this music.
500 *
501 * No other music should be playing.
502 *
503 * \return true if the music was loaded successfully
504 */
505bool Music::start() {
506
507 if (!is_initialized()) {
508 return false;
509 }
510
511 // First time: find the file.
512 if (file_name.empty()) {
513 find_music_file(id, file_name, format);
514
515 if (file_name.empty()) {
516 Debug::error(std::string("Cannot find music file 'musics/")
517 + id + "' (tried with extensions .ogg, .it and .spc)"
518 );
519 return false;
520 }
521 }
522
523 bool success = true;
524
525 // create the buffers and the source
526 alGenBuffers(nb_buffers, buffers);
527 alGenSources(1, &source);
528 alSourcef(source, AL_GAIN, volume);
529
530 // load the music into memory
531 std::string sound_buffer;
532 switch (format) {
533
534 case SPC:
535
536 sound_buffer = QuestFiles::data_file_read(file_name);
537
538 // Give the SPC data into the SPC decoder.
539 spc_decoder->load((int16_t*) sound_buffer.data(), sound_buffer.size());
540
541 for (int i = 0; i < nb_buffers; i++) {
542 decode_spc(buffers[i], 16384);
543 }
544 break;
545
546 case IT:
547
548 sound_buffer = QuestFiles::data_file_read(file_name);
549
550 // Give the IT data to the IT decoder
551 it_decoder->load(sound_buffer);
552
553 for (int i = 0; i < nb_buffers; i++) {
554 decode_it(buffers[i], 16384);
555 }
556 break;
557
558 case OGG:
559
560 sound_buffer = QuestFiles::data_file_read(file_name);
561
562 // Give the OGG data to the OGG decoder.
563 success = ogg_decoder->load(std::move(sound_buffer), this->loop);
564 if (success) {
565 for (int i = 0; i < nb_buffers; i++) {
566 decode_ogg(buffers[i], 16384);
567 }
568 }
569 break;
570
571 case NO_FORMAT:
572 Debug::die("Invalid music format");
573 break;
574 }
575
576 if (!success) {
577 Debug::error("Cannot load music file '" + file_name + "'");
578 }
579
580 // start the streaming
581 alSourceQueueBuffers(source, nb_buffers, buffers);
582 int error = alGetError();
583 if (error != AL_NO_ERROR) {
584 std::ostringstream oss;
585 oss << "Cannot initialize buffers for music '"
586 << file_name << "': error " << error;
587 Debug::error(oss.str());
588 success = false;
589 }
590
591 alSourcePlay(source);
592
593 // The update() function will then take care of filling the buffers
594
595 return success;
596}

598/**
599 * \brief Stops playing the music.
600 *
601 * The callback if any is not called.
602 */
603void Music::stop() {
604
605 if (!is_initialized()) {
606 return;
607 }
608
609 // Release the callback if any.
610 callback_ref.clear();
611
612 // empty the source
613 alSourceStop(source);
614
615 ALint nb_queued;
616 ALuint buffer;
617 alGetSourcei(source, AL_BUFFERS_QUEUED, &nb_queued);
618 for (int i = 0; i < nb_queued; i++) {
619 alSourceUnqueueBuffers(source, 1, &buffer);
620 }
621 alSourcei(source, AL_BUFFER, 0);
622
623 // delete the source
624 alDeleteSources(1, &source);
625
626 // delete the buffers
627 alDeleteBuffers(nb_buffers, buffers);
628
629 switch (format) {
630
631 case SPC:
632 break;
633
634 case IT:
635 it_decoder->unload();
636 break;
637
638 case OGG:
639 ogg_decoder->unload();
640 break;
641
642 case NO_FORMAT:
643 Debug::die("Invalid music format");
644 break;
645 }
646}

648/**
649 * \brief Returns whether the music is paused.
650 * \return true if the music is paused, false otherwise
651 */
652bool Music::is_paused() {
653
654 if (!is_initialized()) {
655 return false;
656 }
657
658 ALint status;
659 alGetSourcei(source, AL_SOURCE_STATE, &status);
660 return status == AL_PAUSED;
661}

663/**
664 * \brief Pauses or resumes the music.
665 * \param pause true to pause the music, false to resume it
666 */
667void Music::set_paused(bool pause) {
668
669 if (!is_initialized()) {
670 return;
671 }
672
673 if (pause) {
674 alSourcePause(source);
675 }
676 else {
677 alSourcePlay(source);
678 }
