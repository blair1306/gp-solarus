53/*
54 The catch-all template.
55*/

57template <typename T>
58class QTypeInfo
59{
60public:
61 enum {
62 isPointer = false,
63 isIntegral = QtPrivate::is_integral<T>::value,
64 isComplex = true,
65 isStatic = true,
66 isLarge = (sizeof(T)>sizeof(void*)),
67 isDummy = false,
68 sizeOf = sizeof(T)
69 };
70};
71
72template<>
73class QTypeInfo<void>
74{
75public:
76 enum {
77 isPointer = false,
78 isIntegral = false,
79 isComplex = false,
80 isStatic = false,
81 isLarge = false,
82 isDummy = false,
83 sizeOf = 0
84 };
85};

100};

102/*!
103 \class QTypeInfoMerger
104 \inmodule QtCore
105 \internal
106
107 \brief QTypeInfoMerger merges the QTypeInfo flags of T1, T2... and presents them
108 as a QTypeInfo<T> would do.
109
110 Let's assume that we have a simple set of structs:
111
112 \snippet code/src_corelib_global_qglobal.cpp 50
113
114 To create a proper QTypeInfo specialization for A struct, we have to check
115 all sub-components; B, C and D, then take the lowest common denominator and call
116 Q_DECLARE_TYPEINFO with the resulting flags. An easier and less fragile approach is to
117 use QTypeInfoMerger, which does that automatically. So struct A would have
118 the following QTypeInfo definition:
119
120 \snippet code/src_corelib_global_qglobal.cpp 51
121*/
122template <class T, class T1, class T2 = T1, class T3 = T1, class T4 = T1>
123class QTypeInfoMerger
124{
125public:
126 enum {
127 isComplex = QTypeInfo<T1>::isComplex || QTypeInfo<T2>::isComplex || QTypeInfo<T3>::isComplex || QTypeInfo<T4>::isComplex,
128 isStatic = QTypeInfo<T1>::isStatic || QTypeInfo<T2>::isStatic || QTypeInfo<T3>::isStatic || QTypeInfo<T4>::isStatic,
129 isLarge = sizeof(T) > sizeof(void*),
130 isPointer = false,
131 isIntegral = false,
132 isDummy = false,
133 sizeOf = sizeof(T)
134 };
135};
136
137#define Q_DECLARE_MOVABLE_CONTAINER(CONTAINER) \
138template <typename T> class CONTAINER; \
139template <typename T> \
140class QTypeInfo< CONTAINER<T> > \
141{ \
142public: \
143 enum { \
144 isPointer = false, \
145 isIntegral = false, \
146 isComplex = true, \
147 isStatic = false, \
148 isLarge = (sizeof(CONTAINER<T>) > sizeof(void*)), \
149 isDummy = false, \
150 sizeOf = sizeof(CONTAINER<T>) \
151 }; \
152};

163/*
164 Specialize a specific type with:
165
166 Q_DECLARE_TYPEINFO(type, flags);
167
168 where 'type' is the name of the type to specialize and 'flags' is
169 logically-OR'ed combination of the flags below.
170*/
171enum { /* TYPEINFO flags */
172 Q_COMPLEX_TYPE = 0,
173 Q_PRIMITIVE_TYPE = 0x1,
174 Q_STATIC_TYPE = 0,
175 Q_MOVABLE_TYPE = 0x2,
176 Q_DUMMY_TYPE = 0x4
177};
178
179#define Q_DECLARE_TYPEINFO_BODY(TYPE, FLAGS) \
180class QTypeInfo<TYPE > \
181{ \
182public: \
183 enum { \
184 isComplex = (((FLAGS) & Q_PRIMITIVE_TYPE) == 0), \
185 isStatic = (((FLAGS) & (Q_MOVABLE_TYPE | Q_PRIMITIVE_TYPE)) == 0), \
186 isLarge = (sizeof(TYPE)>sizeof(void*)), \
187 isPointer = false, \
188 isIntegral = QtPrivate::is_integral< TYPE >::value, \
189 isDummy = (((FLAGS) & Q_DUMMY_TYPE) != 0), \
190 sizeOf = sizeof(TYPE) \
191 }; \
192 static inline const char *name() { return #TYPE; } \
193}
194
195#define Q_DECLARE_TYPEINFO(TYPE, FLAGS) \
196template<> \
197Q_DECLARE_TYPEINFO_BODY(TYPE, FLAGS)
198
199/* Specialize QTypeInfo for QFlags<T> */
200template<typename T> class QFlags;
201template<typename T>
202Q_DECLARE_TYPEINFO_BODY(QFlags<T>, Q_PRIMITIVE_TYPE);
203
204/*
205 Specialize a shared type with:
206
207 Q_DECLARE_SHARED(type)
208
209 where 'type' is the name of the type to specialize. NOTE: shared
210 types must define a member-swap, and be defined in the same
211 namespace as Qt for this to work.
212*/
213#define Q_DECLARE_SHARED_STL(TYPE) \
214QT_END_NAMESPACE \
215namespace std { \
216 template<> inline void swap< QT_PREPEND_NAMESPACE(TYPE) >(QT_PREPEND_NAMESPACE(TYPE) &value1, QT_PREPEND_NAMESPACE(TYPE) &value2) \
217 { value1.swap(value2); } \
218} \
219QT_BEGIN_NAMESPACE
220
221#define Q_DECLARE_SHARED(TYPE) \
222Q_DECLARE_TYPEINFO(TYPE, Q_MOVABLE_TYPE); \
223template <> inline void qSwap<TYPE>(TYPE &value1, TYPE &value2) \
224{ value1.swap(value2); } \
225Q_DECLARE_SHARED_STL(TYPE)
