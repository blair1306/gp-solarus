38/**
39 * \brief Starts this state.
40 * \param previous_state the previous state
41 */
42void Hero::HookshotState::start(const State* previous_state) {
43
44 HeroState::start(previous_state);
45
46 get_sprites().set_animation("hookshot");
47 hookshot = std::make_shared<Hookshot>(get_entity());
48 get_entities().add_entity(hookshot);
49}
50
51/**
52 * \brief Ends this state.
53 * \param next_state the next state (for information)
54 */
55void Hero::HookshotState::stop(const State* next_state) {
56
57 HeroState::stop(next_state);
58
59 if (!hookshot->is_being_removed()) {
60 // the hookshot state was stopped by something other than the hookshot (e.g. an enemy)
61 hookshot->remove_from_map();
62 get_entity().clear_movement();
63 }
64}

184/**
185 * \brief Notifies this state that the hero has just failed to change its
186 * position because of obstacles.
187 */
188void Hero::HookshotState::notify_obstacle_reached() {
189
190 // the movement of the hero has finished normally or an early obstacle was
191 // reached (e.g. an NPC who moved after the hookshot passed)
192 finish_movement();
193}
194
195/**
196 * \brief Returns control to the hero after its hookshot movement.
197 *
198 * This function is called when the hero has finished the hookshot movement.
199 * It checks the validity of the destination position.
200 */
201void Hero::HookshotState::finish_movement() {
202
203 Hero& hero = get_entity();
204 const Rectangle& hero_position = hero.get_bounding_box();
205 int layer = hero.get_layer();
206 Map& map = get_map();
207 Entities& entities = get_entities();
208
209 if (layer == map.get_min_layer() || !map.has_empty_ground(layer, hero_position)) {
210 // the hero is totally on the same layer: no problem
211 hero.start_state_from_ground();
212 }
213 else {
214 // a part of the hero is on empty tiles: this is often illegal, especially
215 // if there are jumpers; allow this only if tiles on
216 // the lower layer are not obstacles, and go to this layer
217 --layer;
218 if (!map.test_collision_with_obstacles(layer, hero_position, hero)) {
219 Sound::play("hero_lands");
220 entities.set_entity_layer(hero, layer);
221 hero.start_state_from_ground();
222 }
223 else {
224 // illegal position: get back to the start point
225 // TODO: get back to the closest valid point from the destination instead
226 Sound::play("hero_hurt");
227 hero.set_state(new BackToSolidGroundState(hero, false, 0, true));
228 }
229 }
