42#ifndef QFILEDEVICE_H
43#define QFILEDEVICE_H
44
45#include <QtCore/qiodevice.h>
46#include <QtCore/qstring.h>
47
48QT_BEGIN_NAMESPACE
49
50class QFileDevicePrivate;
51
52class Q_CORE_EXPORT QFileDevice : public QIODevice
53{
54#ifndef QT_NO_QOBJECT
55 Q_OBJECT
56#endif
57 Q_DECLARE_PRIVATE(QFileDevice)
58
59public:
60 enum FileError {
61 NoError = 0,
62 ReadError = 1,
63 WriteError = 2,
64 FatalError = 3,
65 ResourceError = 4,
66 OpenError = 5,
67 AbortError = 6,
68 TimeOutError = 7,
69 UnspecifiedError = 8,
70 RemoveError = 9,
71 RenameError = 10,
72 PositionError = 11,
73 ResizeError = 12,
74 PermissionsError = 13,
75 CopyError = 14
76 };
77
78 enum Permission {
79 ReadOwner = 0x4000, WriteOwner = 0x2000, ExeOwner = 0x1000,
80 ReadUser = 0x0400, WriteUser = 0x0200, ExeUser = 0x0100,
81 ReadGroup = 0x0040, WriteGroup = 0x0020, ExeGroup = 0x0010,
82 ReadOther = 0x0004, WriteOther = 0x0002, ExeOther = 0x0001
83 };
84 Q_DECLARE_FLAGS(Permissions, Permission)
85
86 enum FileHandleFlag {
87 AutoCloseHandle = 0x0001,
88 DontCloseHandle = 0
89 };
90 Q_DECLARE_FLAGS(FileHandleFlags, FileHandleFlag)
91
92 ~QFileDevice();
93
94 FileError error() const;
95 void unsetError();
96
97 virtual void close();
98
99 bool isSequential() const;
100
101 int handle() const;
102 virtual QString fileName() const;
103
104 qint64 pos() const;
105 bool seek(qint64 offset);
106 bool atEnd() const;
107 bool flush();
108
109 qint64 size() const;
110
111 virtual bool resize(qint64 sz);
112 virtual Permissions permissions() const;
113 virtual bool setPermissions(Permissions permissionSpec);
114
115 enum MemoryMapFlags {
116 NoOptions = 0
117 };
118
119 uchar *map(qint64 offset, qint64 size, MemoryMapFlags flags = NoOptions);
120 bool unmap(uchar *address);
121
122protected:
123 QFileDevice();
124#ifdef QT_NO_QOBJECT
125 QFileDevice(QFileDevicePrivate &dd);
126#else
127 explicit QFileDevice(QObject *parent);
128 QFileDevice(QFileDevicePrivate &dd, QObject *parent = 0);
129#endif
130
131 qint64 readData(char *data, qint64 maxlen);
132 qint64 writeData(const char *data, qint64 len);
133 qint64 readLineData(char *data, qint64 maxlen);
134
135private:
136 Q_DISABLE_COPY(QFileDevice)
137};
138
139Q_DECLARE_OPERATORS_FOR_FLAGS(QFileDevice::Permissions)
140
141QT_END_NAMESPACE
142
143#endif // QFILEDEVICE_H
