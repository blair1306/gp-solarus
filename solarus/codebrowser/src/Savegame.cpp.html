76/**
77 * \brief Creates a savegame with a specified file name, existing or not.
78 * \param main_loop The Solarus root object.
79 * \param file_name Name of the savegame file (can be a new file),
80 * relative to the quest write directory, with its extension.
81 */
82Savegame::Savegame(MainLoop& main_loop, const std::string& file_name):
83 ExportableToLua(),
84 empty(true),
85 file_name(file_name),
86 main_loop(main_loop),
87 equipment(*this),
88 game(nullptr) {
89
90 // Don't call initialize() manually because the shared_ptr does not exist
91 // at this point, but is needed by initialize() when calling item scripts.
92}
93
94/**
95 * \brief Initializes the data from the file or from initial values if the file
96 * does not exist.
97 *
98 * This function should be called before using the object.
99 * The constructor does not call it because it involves Lua calls to initialize
100 * the equipment.
101 */
102void Savegame::initialize() {
103
104 const std::string& quest_write_dir = QuestFiles::get_quest_write_dir();
105 Debug::check_assertion(!quest_write_dir.empty(),
106 "The quest write directory for savegames was not set in quest.dat");
107
108 if (!QuestFiles::data_file_exists(file_name)) {
109 // This save does not exist yet.
110 empty = true;
111 set_initial_values();
112 }
113 else {
114 // A save already exists, let's load it.
115 empty = false;
116 import_from_file();
117 }
118
119 get_equipment().load_items();
120}

194/**
195 * \brief Import the savegame data from the file.
196 */
197void Savegame::import_from_file() {
198
199 // Try to parse as Lua first.
200 lua_State* l = luaL_newstate();
201 const std::string& buffer = QuestFiles::data_file_read(file_name);
202 const int load_result = luaL_loadbuffer(l, buffer.data(), buffer.size(), file_name.c_str());
203
204 // Call the Lua savegame file.
205 if (load_result == 0) {
206 // The buffer was successfully loaded.
207
208 // Make the Lua world aware of this savegame object.
209 lua_pushlightuserdata(l, this);
210 lua_setfield(l, LUA_REGISTRYINDEX, "savegame");
211
212 // Set a special environment to catch every variable declaration.
213 lua_newtable(l);
214 // fun env
215 lua_newtable(l);
216 // fun env env_mt
217 lua_pushcfunction(l, l_newindex);
218 // fun env env_mt __newindex
219 lua_setfield(l, -2, "__newindex");
220 // fun env env_mt
221 lua_setmetatable(l, -2);
222 // fun env
223 lua_setfenv(l, -2);
224 // fun
225
226 if (lua_pcall(l, 0, 0, 0) != 0) {
227 Debug::die(std::string("Failed to load savegame file '")
228 + file_name + "': " + lua_tostring(l, -1));
229 }
230 }
231 else if (load_result == LUA_ERRSYNTAX) {
232 // Apparently it was not a Lua file.
233 // Let's try the obsolete format of Solarus 0.9.
234 SavegameConverterV1 converter(file_name);
235 converter.convert_to_v2(*this);
236 }
237
238 lua_close(l);
239}

241/**
242 * \brief __newindex function of the environment of the savegame file.
243 *
244 * This special __newindex function catches declaration of global variables
245 * to store them into the savegame.
246 *
247 * \param l The Lua context that is calling this function.
248 * \return Number of values to return to Lua.
249 */
250int Savegame::l_newindex(lua_State* l) {
251
252 return LuaTools::exception_boundary_handle(l, [&] {
253 lua_getfield(l, LUA_REGISTRYINDEX, "savegame");
254 Savegame* savegame = static_cast<Savegame*>(lua_touserdata(l, -1));
255 lua_pop(l, 1);
256
257 const std::string& key = LuaTools::check_string(l, 2);
258
259 switch (lua_type(l, 3)) {
260
261 case LUA_TBOOLEAN:
262 savegame->set_boolean(key, lua_toboolean(l, 3));
263 break;
264
265 case LUA_TNUMBER:
266 savegame->set_integer(key, (int) lua_tointeger(l, 3));
267 break;
268
269 case LUA_TSTRING:
270 savegame->set_string(key, lua_tostring(l, 3));
271 break;
272
273 default:
274 LuaTools::type_error(l, 3, "string, number or boolean");
275 }
276
277 return 0;
278 });
279}

281/**
282 * \brief Saves the data into a file.
283 */
284void Savegame::save() {
285
286 std::ostringstream oss;
287 for (const auto& kvp: saved_values) {
288 const std::string& key = kvp.first;
289 oss << key << " = ";
290 const SavedValue& value = kvp.second;
291 if (value.type == SavedValue::VALUE_BOOLEAN) {
292 oss << (value.int_data ? "true" : "false");
293 }
294 else if (value.type == SavedValue::VALUE_INTEGER) {
295 oss << value.int_data;
296 }
297 else { // String.
298 oss << "\"" << value.string_data << "\"";
299 }
300 oss << "\n";
301 }
302
303 const std::string& text = oss.str();
304 QuestFiles::data_file_save(file_name, text);
305 empty = false;
306}

380/**
381 * \brief Returns whether a saved value is a string.
382 * \param key Name of the value to get.
383 * \return true if this value exists and is a string.
384 */
385bool Savegame::is_string(const std::string& key) const {
386
387 SOLARUS_ASSERT(LuaTools::is_valid_lua_identifier(key),
388 std::string("Savegame variable '") + key + "' is not a valid key");
389
390 bool result = false;
391 const auto& it = saved_values.find(key);
392 if (it != saved_values.end()) {
393 const SavedValue& value = it->second;
394 result = (value.type == SavedValue::VALUE_STRING);
395 }
396 return result;
397}

399/**
400 * \brief Returns a string value saved.
401 * \param key Name of the value to get.
402 * \return The string value associated with this key or an empty string.
403 */
404std::string Savegame::get_string(const std::string& key) const {
405
406 SOLARUS_ASSERT(LuaTools::is_valid_lua_identifier(key),
407 std::string("Savegame variable '") + key + "' is not a valid key");
408
409 const auto& it = saved_values.find(key);
410 if (it == saved_values.end()) {
411 return "";
412 }
413
414 const SavedValue& value = it->second;
415 if (value.type != SavedValue::VALUE_STRING) {
416 Debug::error(std::string("Value '") + key + "' is not a string");
417 return "";
418 }
419
420 return value.string_data;
421}
