28/**
29 * \brief Constructor.
30 * \param speed Speed of the movement in pixels per seconds.
31 */
32RandomMovement::RandomMovement(int speed):
33 RandomMovement(speed, 0) {
34
35}

37/**
38 * \brief Constructor.
39 * \param speed Speed of the movement in pixels per seconds.
40 * \param max_radius if the object goes further than this distance, it will come back
41 */
42RandomMovement::RandomMovement(int speed, int max_radius):
43 StraightMovement(false, false),
44 normal_speed(speed),
45 max_radius(max_radius),
46 next_direction_change_date(0) {
47
48 set_speed(speed);
49 set_next_direction();
50}

52/**
53 * \brief Notifies this movement that the object it controls has changed.
54 */
55void RandomMovement::notify_object_controlled() {
56
57 StraightMovement::notify_object_controlled();
58 set_max_radius(max_radius);
59}

61/**
62 * \brief Returns the speed applied when the movement is started.
63 * \return The speed when moving.

65double RandomMovement::get_normal_speed() const {
66 return normal_speed;
67}

69/**
70 * \brief Sets the speed to apply when the movement is started.
71 * \param normal_speed The new speed.
72 */
73void RandomMovement::set_normal_speed(double normal_speed) {
74
75 this->normal_speed = normal_speed;
76
77 if (is_started()) {
78 StraightMovement::set_speed(normal_speed);
79 }
80}
81
82/**
83 * \brief Returns the maximum radius.
84 * \return The distance above which the object comes back.
85 */
86int RandomMovement::get_max_radius() const {
87 return max_radius;
88}
89
90/**
91 * \brief Sets the maximum radius.
92 * \param max_radius If the object goes further than this distance,
93 * it will come back (0 means no limit).
94 */
95void RandomMovement::set_max_radius(int max_radius) {
96
97 if (max_radius < 0) {
98 std::ostringstream oss;
99 oss << "Invalid max radius: " << max_radius;
100 Debug::die(oss.str());
101 }
102 this->max_radius = max_radius;
103
104 // restrict the movement in a rectangle
105 bounds.set_xy(get_xy());
106 bounds.add_xy(-max_radius, -max_radius);
107 bounds.set_size(max_radius * 2, max_radius * 2);
108}
109
110/**
111 * \brief Chooses a new direction for the movement.
112 */
113void RandomMovement::set_next_direction() {
114
115 set_speed(normal_speed);
116
117 double angle;
118 if (get_entity() == nullptr
119 || max_radius == 0 // means no limit
120 || bounds.contains(get_xy())) {
121
122 // we are inside the bounds (or there is no bound): pick a random direction
123 angle = Geometry::degrees_to_radians(Random::get_number(8) * 45 + 22.5);
124 }
125 else {
126
127 // we are outside the bounds: get back into the rectangle to avoid going too far
128 angle = Geometry::get_angle(get_xy(), bounds.get_center());
129 }
130 set_angle(angle);
131
132 next_direction_change_date = System::now() + 500 + Random::get_number(1500); // change again in 0.5 to 2 seconds
133
134 notify_movement_changed();
135}
136
137/**
138 * \brief Updates the random movement.
139 *
140 * This function is called repeatedly by the map.
141 * This is a redefinition of Movement::update()
142 * to change the direction sometimes.
143 */
144void RandomMovement::update() {
145
146 StraightMovement::update();
147
148 if (!is_suspended()) {
149
150 uint32_t now = System::now();
151
152 if (now >= next_direction_change_date) {
153 set_next_direction();
154 }
155 }
156}
157
158/**
159 * \brief Suspends or resumes this movement.
160 * \param suspended true to suspend the movement, false to resume it
161 */
162void RandomMovement::set_suspended(bool suspended) {
163
164 StraightMovement::set_suspended(suspended);
165
166 if (!suspended) {
167 next_direction_change_date += System::now() - get_when_suspended();
168 }
169}
170
171/**
172 * \brief Notifies this movement that it just failed to apply
173 * because of obstacles.
174 */
175void RandomMovement::notify_obstacle_reached() {
176
177 StraightMovement::notify_obstacle_reached();
178 set_next_direction();
179}
