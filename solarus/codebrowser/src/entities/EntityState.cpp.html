181/**
182 * \brief Starts this state.
183 *
184 * This function is called automatically when this state becomes the active
185 * state of the entity.
186 * The initializations should be done here rather than in the constructor.
187 *
188 * \param previous_state The previous state or nullptr if this is the first state
189 * (for information).
190 */
191void Entity::State::start(const State* /* previous_state */) {
192
193 set_suspended(entity.is_suspended());
194
195 // Notify Lua.
196 if (entity.is_on_map()) {
197 get_lua_context().entity_on_state_changed(entity, get_name());
198 }
199}
200
201/**
202 * \brief Ends this state.
203 *
204 * This function is called automatically when this state is not the active
205 * state anymore.
206 * You should here close everything the start() function has opened.
207 * The destructor will be called at the next cycle.
208 *
209 * \param next_state The next state (for information).
210 */
211void Entity::State::stop(const State* /* next_state */) {
212
213 Debug::check_assertion(!is_stopping(),
214 std::string("This state is already stopping: ") + get_name());
215
216 this->stopping = true;
217}

247/**
248 * \brief Notifies this state that the game was just suspended or resumed.
249 * \param suspended true if the game is suspended
250 */
251void Entity::State::set_suspended(bool suspended) {
252
253 if (suspended != this->suspended) {
254
255 this->suspended = suspended;
256
257 // remember the date if the state is being suspended
258 if (suspended) {
259 when_suspended = System::now();
260 }
261 }
262}

272/**
273 * \brief This function is called when a game command is pressed and the game
274 * is not suspended.
275 * \param command The command pressed.
276 */
277void Entity::State::notify_command_pressed(GameCommand command) {
278
279 switch (command) {
280
281 // action key
282 case GameCommand::ACTION:
283 notify_action_command_pressed();
284 break;
285
286 // sword key
287 case GameCommand::ATTACK:
288 notify_attack_command_pressed();
289 break;
290
291 // move the entity
292 case GameCommand::RIGHT:
293 notify_direction_command_pressed(0);
294 break;
295
296 case GameCommand::UP:
297 notify_direction_command_pressed(1);
298 break;
299
300 case GameCommand::LEFT:
301 notify_direction_command_pressed(2);
302 break;
303
304 case GameCommand::DOWN:
305 notify_direction_command_pressed(3);
306 break;
307
308 // use an equipment item
309 case GameCommand::ITEM_1:
310 notify_item_command_pressed(1);
311 break;
312
313 case GameCommand::ITEM_2:
314 notify_item_command_pressed(2);
315 break;
316
317 default:
318 break;
319 }
320}

322/**
323 * \brief This function is called when a command is released if the game is
324 * not suspended.
325 * \param command The command released.

327void Entity::State::notify_command_released(GameCommand command) {
328
329 switch (command) {
330
331 case GameCommand::ACTION:
332 notify_action_command_released();
333 break;
334
335 case GameCommand::ATTACK:
336 notify_attack_command_released();
337 break;

343 case GameCommand::UP:
344 notify_direction_command_released(1);
345 break;
346
347 case GameCommand::LEFT:
348 notify_direction_command_released(2);

350
351 case GameCommand::DOWN:
352 notify_direction_command_released(3);
353 break;
354
355 case GameCommand::ITEM_1:
356 notify_item_command_released(0);
357 break;
358
359 case GameCommand::ITEM_2:
360 notify_item_command_released(1);
361 break;
362
363 default:
364 break;
365 }
366}

772/**
773 * \brief Returns whether some stairs are considered as obstacle in this state.
774 * \param stairs Some stairs.
775 * \return \c true if the stairs are obstacle in this state.
776 */
777bool Entity::State::is_stairs_obstacle(const Stairs& stairs) const {
778
779 // The entity may overlap stairs in rare cases,
780 // for example if he arrived by swimming over them
781 // and thus did not activate them.
782 // This is allowed and can be used to leave water pools for example.
783 if (entity.overlaps(stairs)) {
784 return false;
785 }
786
787 return true;
788}
