318/**
319 * \brief Returns whether a value is a userdata of type movement.
320 * \param l A Lua context.
321 * \param index An index in the stack.
322 * \return true if the value at this index is a entity.
323 */
324bool LuaContext::is_movement(lua_State* l, int index) {
325 return is_straight_movement(l, index)
326 || is_random_movement(l, index)
327 || is_target_movement(l, index)
328 || is_path_movement(l, index)
329 || is_random_path_movement(l, index)
330 || is_path_finding_movement(l, index)
331 || is_circle_movement(l, index)
332 || is_jump_movement(l, index)
333 || is_pixel_movement(l, index);
334}

336/**
337 * \brief Checks that the userdata at the specified index of the stack is a
338 * movement (of any subtype) and returns it.
339 * \param l a Lua context
340 * \param index an index in the stack
341 * \return the sprite
342 */
343std::shared_ptr<Movement> LuaContext::check_movement(lua_State* l, int index) {
344
345 if (is_movement(l, index)) {
346 const ExportableToLuaPtr& userdata = *(static_cast<ExportableToLuaPtr*>(
347 lua_touserdata(l, index)
348 ));
349 return std::static_pointer_cast<Movement>(userdata);
350 }
351 else {
352 LuaTools::type_error(l, index, "movement");
353 throw;
354 }
355}

367/**
368 * \brief Starts moving an x,y point.
369 *
370 * The point is a Lua table with two fields x and y.
371 * Fields x and y may be initially missing.
372 *
373 * \param movement The movement to apply to the points.
374 * \param point_index Index of the x,y table in the Lua stack.
375 */
376void LuaContext::start_movement_on_point(
377 const std::shared_ptr<Movement>& movement, int point_index
378) {
379 int x = 0;
380 int y = 0;
381 // ...
382 lua_getfield(l, LUA_REGISTRYINDEX, "sol.movements_on_points");
383 // ... movements
384 push_movement(l, *movement);
385 // ... movements movement
386 lua_pushvalue(l, point_index);
387 // ... movements movement xy
388 lua_getfield(l, -1, "x");
389 // ... movements movement xy x/nil
390 if (lua_isnil(l, -1)) {
391 // ... movements movement xy nil
392 lua_pop(l, 1);
393 // ... movements movement xy
394 lua_pushinteger(l, 0);
395 // ... movements movement xy 0
396 lua_setfield(l, -2, "x");
397 // ... movements movement xy
398 }
399 else {
400 // ... movements movement xy x
401 x = LuaTools::check_int(l, -1);
402 lua_pop(l, 1);
403 // ... movements movement xy
404 }
405 lua_getfield(l, -1, "y");
406 // ... movements movement xy y/nil
407 if (lua_isnil(l, -1)) {
408 // ... movements movement xy nil
409 lua_pop(l, 1);
410 // ... movements movement xy
411 lua_pushinteger(l, 0);
412 // ... movements movement xy 0
413 lua_setfield(l, -2, "y");
414 // ... movements movement xy
415 movement->set_y(0);
416 }
417 else {
418 // ... movements movement xy y
419 y = LuaTools::check_int(l, -1);
420 lua_pop(l, 1);
421 // ... movements movement xy
422 }
423
424 lua_settable(l, -3);
425 // ... movements
426 lua_pop(l, 1);
427 // ...
428 movement->set_xy(x, y);
429}

450/**
451 * \brief Updates all movements applied to x,y points.
452 *
453 * Movements applied to map entities or drawables are already updated
454 * by the entity or the drawable.
455 * This may change in the future in order to unify the handling of movements.
456 */
457void LuaContext::update_movements() {
458
459 lua_getfield(l, LUA_REGISTRYINDEX, "sol.movements_on_points");
460 lua_pushnil(l); // First key.
461 while (lua_next(l, -2)) {
462 Movement& movement = *check_movement(l, -2);
463 movement.update();
464 lua_pop(l, 1); // Pop the value, keep the key for next iteration.
465 }
466 lua_pop(l, 1); // Pop the movements table.
467}

469/**
470 * \brief Implementation of sol.movement.create().
471 * \param l the Lua context that is calling this function
472 * \return number of values to return to Lua
473 */
474int LuaContext::movement_api_create(lua_State* l) {
475
476 return LuaTools::exception_boundary_handle(l, [&] {
477 LuaContext& lua_context = get_lua_context(l);
478 const std::string& type = LuaTools::check_string(l, 1);
479
480 std::shared_ptr<Movement> movement;
481 if (type == "straight") {
482 std::shared_ptr<StraightMovement> straight_movement =
483 std::make_shared<StraightMovement>(false, true);
484 straight_movement->set_speed(32);
485 movement = straight_movement;
486 }
487 else if (type == "random") {
488 movement = std::make_shared<RandomMovement>(32);
489 }
490 else if (type == "target") {
491 Game* game = lua_context.get_main_loop().get_game();
492 if (game != nullptr) {
493 // If we are on a map, the default target is the hero.
494 movement = std::make_shared<TargetMovement>(
495 game->get_hero(), 0, 0, 96, false
496 );
497 }
498 else {
499 movement = std::make_shared<TargetMovement>(
500 nullptr, 0, 0, 32, false
501 );
502 }
503 }
504 else if (type == "path") {
505 movement = std::make_shared<PathMovement>(
506 "", 32, false, false, false
507 );
508 }
509 else if (type == "random_path") {
510 movement = std::make_shared<RandomPathMovement>(32);
511 }
512 else if (type == "path_finding") {
513 std::shared_ptr<PathFindingMovement> path_finding_movement =
514 std::make_shared<PathFindingMovement>(32);
515 Game* game = lua_context.get_main_loop().get_game();
516 if (game != nullptr) {
517 // If we are on a map, the default target is the hero.
518 path_finding_movement->set_target(game->get_hero());
519 }
520 movement = path_finding_movement;
521 }
522 else if (type == "circle") {
523 movement = std::make_shared<CircleMovement>(false);
524 }
525 else if (type == "jump") {
526 movement = std::make_shared<JumpMovement>(0, 0, 0, false);
527 }
528 else if (type == "pixel") {
529 movement = std::make_shared<PixelMovement>("", 30, false, false);
530 }
531 else {
532 LuaTools::arg_error(l, 1, "should be one of: "
533 "\"straight\", "
534 "\"random\", "
535 "\"target\", "
536 "\"path\", "
537 "\"random_path\", "
538 "\"path_finding\", "
539 "\"circle\", "
540 "\"jump\" or "
541 "\"pixel\"");
542 }
543
544 push_movement(l, *movement);
545 return 1;
546 });
547}

584/**
585 * \brief Implementation of movement:start().
586 * \param l the Lua context that is calling this function
587 * \return number of values to return to Lua
588 */
589int LuaContext::movement_api_start(lua_State* l) {
590
591 return LuaTools::exception_boundary_handle(l, [&] {
592 LuaContext& lua_context = get_lua_context(l);
593
594 std::shared_ptr<Movement> movement = check_movement(l, 1);
595 movement_api_stop(l); // First, stop any previous movement.
596
597 ScopedLuaRef callback_ref = LuaTools::opt_function(l, 3);
598
599 if (lua_type(l, 2) == LUA_TTABLE) {
600 lua_context.start_movement_on_point(movement, 2);
601 }
602 else if (is_entity(l, 2)) {
603 Entity& entity = *check_entity(l, 2);
604 entity.clear_movement();
605 entity.set_movement(movement);
606 }
607 else if (is_drawable(l, 2)) {
608 Drawable& drawable = *check_drawable(l, 2);
609 drawable.start_movement(movement);
610 }
611 else {
612 LuaTools::type_error(l, 2, "table, entity or drawable");
613 }
614 movement->set_finished_callback(callback_ref);
615
616 return 0;
617 });
618}

620/**
621 * \brief Implementation of movement:stop().
622 * \param l the Lua context that is calling this function
623 * \return number of values to return to Lua
624 */
625int LuaContext::movement_api_stop(lua_State* l) {
626
627 return LuaTools::exception_boundary_handle(l, [&] {
628 LuaContext& lua_context = get_lua_context(l);
629
630 std::shared_ptr<Movement> movement = check_movement(l, 1);
631
632 Entity* entity = movement->get_entity();
633 if (entity != nullptr) {
634 // The object controlled is a map entity.
635 entity->clear_movement();
636 }
637 else {
638 Drawable* drawable = movement->get_drawable();
639 if (drawable != nullptr) {
640 // The object controlled is a drawable.
641 drawable->stop_movement();
642 }
643 else {
644 // The object controlled is a point.
645 lua_context.stop_movement_on_point(movement);
646 }
647 }
648
649 return 0;
650 });
651}
