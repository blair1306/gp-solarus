42#ifndef QFILEINFO_H
43#define QFILEINFO_H
44
45#include <QtCore/qfile.h>
46#include <QtCore/qlist.h>
47#include <QtCore/qshareddata.h>
48#include <QtCore/qmetatype.h>
49
50QT_BEGIN_NAMESPACE
51
52
53class QDir;
54class QDirIteratorPrivate;
55class QDateTime;
56class QFileInfoPrivate;
57
58class Q_CORE_EXPORT QFileInfo
59{
60 friend class QDirIteratorPrivate;
61public:
62 explicit QFileInfo(QFileInfoPrivate *d);
63
64 QFileInfo();
65 QFileInfo(const QString &file);
66 QFileInfo(const QFile &file);
67 QFileInfo(const QDir &dir, const QString &file);
68 QFileInfo(const QFileInfo &fileinfo);
69 ~QFileInfo();
70
71 QFileInfo &operator=(const QFileInfo &fileinfo);
72#ifdef Q_COMPILER_RVALUE_REFS
73 inline QFileInfo&operator=(QFileInfo &&other)
74 { qSwap(d_ptr, other.d_ptr); return *this; }
75#endif
76
77 inline void swap(QFileInfo &other)
78 { qSwap(d_ptr, other.d_ptr); }
79
80 bool operator==(const QFileInfo &fileinfo) const;
81 inline bool operator!=(const QFileInfo &fileinfo) const { return !(operator==(fileinfo)); }
82
83 void setFile(const QString &file);
84 void setFile(const QFile &file);
85 void setFile(const QDir &dir, const QString &file);
86 bool exists() const;
87 static bool exists(const QString &file);
88 void refresh();
89
90 QString filePath() const;
91 QString absoluteFilePath() const;
92 QString canonicalFilePath() const;
93 QString fileName() const;
94 QString baseName() const;
95 QString completeBaseName() const;
96 QString suffix() const;
97 QString bundleName() const;
98 QString completeSuffix() const;
99
100 QString path() const;
101 QString absolutePath() const;
102 QString canonicalPath() const;
103 QDir dir() const;
104 QDir absoluteDir() const;
105
106 bool isReadable() const;
107 bool isWritable() const;
108 bool isExecutable() const;
109 bool isHidden() const;
110 bool isNativePath() const;
111
112 bool isRelative() const;
113 inline bool isAbsolute() const { return !isRelative(); }
114 bool makeAbsolute();
115
116 bool isFile() const;
117 bool isDir() const;
118 bool isSymLink() const;
119 bool isRoot() const;
120 bool isBundle() const;
121
122 QString readLink() const;
123 inline QString symLinkTarget() const { return readLink(); }
124
125 QString owner() const;
126 uint ownerId() const;
127 QString group() const;
128 uint groupId() const;
129
130 bool permission(QFile::Permissions permissions) const;
131 QFile::Permissions permissions() const;
132
133 qint64 size() const;
134
135 QDateTime created() const;
136 QDateTime lastModified() const;
137 QDateTime lastRead() const;
138
139 bool caching() const;
140 void setCaching(bool on);
141
142protected:
143 QSharedDataPointer<QFileInfoPrivate> d_ptr;
144
145private:
146 QFileInfoPrivate* d_func();
147 inline const QFileInfoPrivate* d_func() const
148 {
149 return d_ptr.constData();
150 }
151};
152
153Q_DECLARE_SHARED(QFileInfo)
154
155typedef QList<QFileInfo> QFileInfoList;

157QT_END_NAMESPACE
