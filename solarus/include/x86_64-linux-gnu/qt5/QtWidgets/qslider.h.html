42#ifndef QSLIDER_H
43#define QSLIDER_H
44
45#include <QtWidgets/qabstractslider.h>
46
47QT_BEGIN_NAMESPACE
48
49
50#ifndef QT_NO_SLIDER
51
52class QSliderPrivate;
53class QStyleOptionSlider;
54class Q_WIDGETS_EXPORT QSlider : public QAbstractSlider
55{
56 Q_OBJECT
57
58 Q_ENUMS(TickPosition)
59 Q_PROPERTY(TickPosition tickPosition READ tickPosition WRITE setTickPosition)
60 Q_PROPERTY(int tickInterval READ tickInterval WRITE setTickInterval)
61
62public:
63 enum TickPosition {
64 NoTicks = 0,
65 TicksAbove = 1,
66 TicksLeft = TicksAbove,
67 TicksBelow = 2,
68 TicksRight = TicksBelow,
69 TicksBothSides = 3
70 };
71
72 explicit QSlider(QWidget *parent = 0);
73 explicit QSlider(Qt::Orientation orientation, QWidget *parent = 0);
74
75 ~QSlider();
76
77 QSize sizeHint() const;
78 QSize minimumSizeHint() const;
79
80 void setTickPosition(TickPosition position);
81 TickPosition tickPosition() const;
82
83 void setTickInterval(int ti);
84 int tickInterval() const;
85
86 bool event(QEvent *event);
87
88protected:
89 void paintEvent(QPaintEvent *ev);
90 void mousePressEvent(QMouseEvent *ev);
91 void mouseReleaseEvent(QMouseEvent *ev);
92 void mouseMoveEvent(QMouseEvent *ev);
93 void initStyleOption(QStyleOptionSlider *option) const;
94
95
96private:
97 friend Q_WIDGETS_EXPORT QStyleOptionSlider qt_qsliderStyleOption(QSlider *slider);
98
99 Q_DISABLE_COPY(QSlider)
100 Q_DECLARE_PRIVATE(QSlider)
101};
102
103#endif // QT_NO_SLIDER
104
105QT_END_NAMESPACE
106
107#endif // QSLIDER_H
