260struct AbstractComparatorFunction
261{
262 typedef bool (*LessThan)(const AbstractComparatorFunction *, const void *, const void *);
263 typedef bool (*Equals)(const AbstractComparatorFunction *, const void *, const void *);
264 typedef void (*Destroy)(AbstractComparatorFunction *);
265 explicit AbstractComparatorFunction(LessThan lt = 0, Equals e = 0, Destroy d = 0)
266 : lessThan(lt), equals(e), destroy(d) {}
267 Q_DISABLE_COPY(AbstractComparatorFunction)
268 LessThan lessThan;
269 Equals equals;
270 Destroy destroy;
271};
272
273template<typename T>
274struct BuiltInComparatorFunction : public AbstractComparatorFunction
275{
276 BuiltInComparatorFunction()
277 : AbstractComparatorFunction(lessThan, equals, destroy) {}
278 static bool lessThan(const AbstractComparatorFunction *, const void *l, const void *r)
279 {
280 const T *lhs = static_cast<const T *>(l);
281 const T *rhs = static_cast<const T *>(r);
282 return *lhs < *rhs;
283 }
284
285 static bool equals(const AbstractComparatorFunction *, const void *l, const void *r)
286 {
287 const T *lhs = static_cast<const T *>(l);
288 const T *rhs = static_cast<const T *>(r);
289 return *lhs == *rhs;
290 }
291
292 static void destroy(AbstractComparatorFunction *_this)
293 {
294 delete static_cast<BuiltInComparatorFunction *>(_this);
295 }
296};

307template<typename From, typename To>
308struct ConverterMemberFunction : public AbstractConverterFunction
309{
310 explicit ConverterMemberFunction(To(From::*function)() const)
311 : AbstractConverterFunction(convert),
312 m_function(function) {}
313 ~ConverterMemberFunction();
314 static bool convert(const AbstractConverterFunction *_this, const void *in, void *out)
315 {
316 const From *f = static_cast<const From *>(in);
317 To *t = static_cast<To *>(out);
318 const ConverterMemberFunction *_typedThis =
319 static_cast<const ConverterMemberFunction *>(_this);
320 *t = (f->*_typedThis->m_function)();
321 return true;
322 }
323
324 To(From::* const m_function)() const;
325};

327template<typename From, typename To>
328struct ConverterMemberFunctionOk : public AbstractConverterFunction
329{
330 explicit ConverterMemberFunctionOk(To(From::*function)(bool *) const)
331 : AbstractConverterFunction(convert),
332 m_function(function) {}
333 ~ConverterMemberFunctionOk();
334 static bool convert(const AbstractConverterFunction *_this, const void *in, void *out)
335 {
336 const From *f = static_cast<const From *>(in);
337 To *t = static_cast<To *>(out);
338 bool ok = false;
339 const ConverterMemberFunctionOk *_typedThis =
340 static_cast<const ConverterMemberFunctionOk *>(_this);
341 *t = (f->*_typedThis->m_function)(&ok);
342 if (!ok)
343 *t = To();
344 return ok;
345 }
346
347 To(From::* const m_function)(bool*) const;
348};

351struct ConverterFunctor : public AbstractConverterFunction

361 const ConverterFunctor *_typedThis =

365 }
