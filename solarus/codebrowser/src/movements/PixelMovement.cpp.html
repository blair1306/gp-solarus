26/**
27 * \brief Creates a pixel movement object.
28 * \param trajectory_string a string describing the succession of translations that compose this movement,
29 * with the syntax "dx1 dy1 dx2 dy2 dx3 dy3 ..." (the number of spaces between values does not matter)
30 * \param delay delay in milliseconds between two translations
31 * \param loop true to make the movement return to the beginning once finished
32 * \param ignore_obstacles true to make the movement ignore obstacles
33 */
34PixelMovement::PixelMovement(
35 const std::string& trajectory_string,
36 uint32_t delay,
37 bool loop,
38 bool ignore_obstacles):
39 Movement(ignore_obstacles),
40 next_move_date(0),
41 delay(delay),
42 loop(loop),
43 nb_steps_done(0),
44 finished(false) {
45
46 set_trajectory(trajectory_string);
47}

57/**
58 * \brief Sets the trajectory of this movement.
59 *
60 * This function can be called even if the object was moving with a previous trajectory.
61 * The old trajectory is replaced and the movement starts the from beginning of the
62 * new trajectory.
63 *
64 * \param trajectory a list of points describing the succession of translations that compose this movement,
65 * where each point is an xy value representing a translation
66 */
67void PixelMovement::set_trajectory(const std::list<Point>& trajectory) {
68
69 this->trajectory = trajectory;
70 this->trajectory_string = ""; // will be computed only on demand
71
72 restart();
73}
74
75/**
76 * \brief Sets the trajectory of this movement from a string.
77 *
78 * This function can be called even if the object was moving with a previous trajectory.
79 * The old trajectory is replaced and the movement starts the from beginning of the
80 * new trajectory.
81 *
82 * \param trajectory_string a string describing the succession of translations that compose this movement,
83 * with the syntax "dx1 dy1 dx2 dy2 dx3 dy3 ..." (the number of spaces between values does not matter)
84 */
85void PixelMovement::set_trajectory(const std::string& trajectory_string) {
86
87 int dx = 0;
88 int dy = 0;
89
90 trajectory.clear();
91 std::istringstream iss(trajectory_string);
92 while (iss >> dx) {
93 if (!(iss >> dy)) {
94 Debug::die(std::string("Invalid trajectory string: '")
95 + trajectory_string + "'");
96 }
97 trajectory.emplace_back(dx, dy);
98 }
99 this->trajectory_string = trajectory_string;
100
101 restart();
102}

128/**
129 * \brief Sets whether this movement loops when the end of the trajectory is reached.
130 *
131 * Is the movement was finished and loop is set to true, the movement restarts.
132 *
133 * \param loop true to make the movement loop
134 */
135void PixelMovement::set_loop(bool loop) {
136
137 this->loop = loop;
138
139 if (finished && loop) {
140 restart();
141 }
142}

144/**
145 * \brief Restarts this movement to the beginning.
146 */
147void PixelMovement::restart() {
148
149 if (get_length() == 0) {
150 finished = true;
151 }
152 else {
153 nb_steps_done = 0;
154 finished = false;
155 trajectory_iterator = trajectory.begin();
156
157 if (next_move_date == 0) {
158 // Keep the previous date if we just looped.
159 next_move_date = System::now();
160 }
161 next_move_date += delay;
162
163 notify_movement_changed();
164 }
165}

167/**
168 * \brief Updates the position.
169 */
170void PixelMovement::update() {
171
172 uint32_t now = System::now();
173
174 while (now >= next_move_date &&
175 !is_suspended() &&
176 !finished &&
177 (get_entity() == nullptr || get_entity()->get_movement().get() == this)
178 ) {
179 Point old_xy = get_xy();
180 make_next_step();
181
182 bool success = (get_xy() != old_xy);
183 if (!success) {
184 notify_obstacle_reached();
185 }
186 }
187
188 // Do this at last so that Movement::update() knows whether we are finished.
189 Movement::update();
190}

192/**
193 * \brief Suspends or resumes this movement.
194 * \param suspended true to suspend the movement, false to resume it
195 */
196void PixelMovement::set_suspended(bool suspended) {
197
198 Movement::set_suspended(suspended);
199
200 if (!suspended
201 && get_when_suspended() != 0
202 && next_move_date != 0) {
203 next_move_date += System::now() - get_when_suspended();
204 }
205}

207/**
208 * \brief Makes a move in the trajectory.
209 *
210 * This function must be called only when the trajectory is not finished yet.
211 */
212void PixelMovement::make_next_step() {
213
214 bool success = false;
215 const Point& dxy = *trajectory_iterator;
216
217 if (!test_collision_with_obstacles(dxy)) {
218 translate_xy(dxy);
219 success = true;
220 }
221
222 ++trajectory_iterator;
223
224 if (trajectory_iterator == trajectory.end()) {
225 if (loop) {
226 trajectory_iterator = trajectory.begin();
227 }
228 else {
229 finished = true;
230 }
231 }
232
233 if (!finished) {
234 // Add the delay unless we are finished, because we may be looping.
235 next_move_date += delay;
236 }
237
238 int step_index = nb_steps_done;
239 nb_steps_done++;
240 notify_step_done(step_index, success);
241}

249}

259/**

263bool PixelMovement::is_started() const {
264 return !finished;
265}

267/**
