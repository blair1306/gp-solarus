17#ifndef SOLARUS_RANDOM_MOVEMENT_H
18#define SOLARUS_RANDOM_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/movements/StraightMovement.h"
22#include "solarus/lowlevel/Rectangle.h"
23#include <cstdint>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief A movement that makes successive steps of random directions
30 * and (possibly) stays inside a limited rectangle.
31 */
32class RandomMovement: public StraightMovement {
33
34 public:
35
36 explicit RandomMovement(int speed);
37 RandomMovement(int speed, int max_radius);
38
39 virtual void notify_object_controlled() override;
40 virtual void update() override;
41 virtual void set_suspended(bool suspended) override;
42
43 double get_normal_speed() const;
44 void set_normal_speed(double normal_speed);
45 int get_max_radius() const;
46 void set_max_radius(int max_radius);
47
48 virtual const std::string& get_lua_type_name() const override;
49
50 protected:
51
52 virtual void notify_obstacle_reached() override;
53
54 private:
55
56 void set_next_direction();
57
58 double normal_speed; /** Speed applied when the movement is started. */
59 int max_radius; /**< if the object goes further than this distance in x or y, it will come back
60 * (this is not a hard limit, there is no guarantee) */
61 Rectangle bounds; /**< a rectangle the object should not escape from */
62 uint32_t next_direction_change_date; /**< date of the next direction change */
63
64};
65
66}
67
68#endif
