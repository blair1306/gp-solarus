36void QuestsItemDelegate::paint(QPainter *painter,
37 const QStyleOptionViewItem &option,
38 const QModelIndex &index) const {
39
40 // Paint the default background
41 //QStyledItemDelegate::paint(painter, option, index);
42
43 // Save painter initial state before any modification
44 painter->save();
45 painter->setRenderHint(QPainter::Antialiasing, true);
46 painter->setRenderHint(QPainter::TextAntialiasing, true);
47
48 const QRect& global_rect = option.rect;
49 const int padding = 4;
50 const QFont &default_font = option.font;
51
52 // Retrieve quest info
53 const QVariant& data = index.model()->data(index, Qt::DisplayRole);
54 const QuestsModel::QuestInfo& quest_info = data.value<QuestsModel::QuestInfo>();
55
56 // Retrieve info from option
57 QStyle::State state = option.state;
58 const QStyleOptionViewItem::ViewItemFeatures& features = option.features;
59 const QPalette& palette = option.palette;
60 bool active = state.testFlag(QStyle::State_Active);
61 bool enabled = state.testFlag(QStyle::State_Enabled);
62 bool selected = state.testFlag(QStyle::State_Selected);
63 bool alternate_rows = features.testFlag(QStyleOptionViewItem::Alternate);
64 bool should_alternate = alternate_rows && index.row() % 2;
65 //auto focused = state.testFlag(QStyle::State_HasFocus);
66 //auto mouse_over = state.testFlag(QStyle::StateFlag::State_MouseOver);
67
68 // Compute colors
69 QPalette::ColorGroup color_group =
70 !enabled ? QPalette::Disabled
71 : (active ? QPalette::Active : QPalette::Inactive);
72 QPalette::ColorRole color_role =
73 selected ? QPalette::Highlight
74 : (should_alternate ? QPalette::AlternateBase : QPalette::Light);
75 const QBrush& background = palette.brush(color_group, color_role);
76
77 // Paint background
78 painter->fillRect(option.rect, background);
79
80 // Paint icon
81 QRect icon_rect(
82 global_rect.left() + padding * 2,
83 global_rect.top() + (global_rect.height() - _iconSize.height()) / 2,
84 _iconSize.width(),
85 _iconSize.height());
86 QIcon::Mode icon_mode = enabled ? QIcon::Normal : QIcon::Disabled;
87 const QPixmap& pixmap = quest_info.icon.pixmap(_iconSize, icon_mode, QIcon::On);
88 painter->drawPixmap(icon_rect, pixmap);
89
90 // Compute title font
91 QFont title_font = default_font;
92 title_font.setPointSize(title_font.pointSize() * 1.25);
93 title_font.setWeight(QFont::DemiBold);
94
95 // Compute author font
96 QFont author_font = default_font;
97 author_font.setWeight(QFont::Normal);
98
99 // Set temporarily the font to compute the title text height
100 painter->save();
101 painter->setFont(title_font);
102 const QFontMetrics& title_font_metrics = painter->fontMetrics();
103 painter->restore();
104
105 // Set temporarily the font to compute the author text height
106 painter->save();
107 painter->setFont(author_font);
108 const QFontMetrics& author_font_metrics = painter->fontMetrics();
109 painter->restore();
110
111 // Compute author and title text rects
112 int title_rect_left = icon_rect.right() + padding * 3;
113 int title_rect_width = global_rect.width() - title_rect_left - padding;
114 int title_rect_height = title_font_metrics.height();
115
116 int author_rect_left = title_rect_left;
117 int author_rect_width = title_rect_width;
118 int author_rect_height = author_font_metrics.height();
119
120 QRect title_rect(title_rect_left, 0, title_rect_width, title_rect_height);
121 QRect author_rect(author_rect_left, title_rect_height, author_rect_width,
122 author_rect_height);
123
124 int global_text_height = author_rect.bottom() - title_rect.top();
125 int global_text_top = (global_rect.height() - global_text_height) / 2;
126
127 int y_translate = global_rect.top() + global_text_top;
128 title_rect.translate(0, y_translate);
129 author_rect.translate(0, y_translate);
130
131 // Compute title text color
132 QPalette::ColorRole pen_color_role = selected ? QPalette::HighlightedText
133 : QPalette::Text;
134 const QColor& title_pen_color = palette.brush(color_group, pen_color_role).color();
135
136 // Paint title text
137 const QString& title_text = QString::fromStdString(quest_info.properties.get_title());
138 if (!title_text.isEmpty()) {
139 QString title_elided_text = title_font_metrics.elidedText(title_text,
140 Qt::ElideRight,
141 title_rect_width);
142 painter->setPen(title_pen_color);
143 painter->setFont(title_font);
144 painter->drawText(title_rect, title_elided_text);
145 }
146
147 // Compute title text color
148 const QColor& author_pen_color =
149 palette.brush(QPalette::Disabled, pen_color_role).color();
150
151 // Paint author text
152 QString separator = QString(" %1 ").arg(QChar(0x2022)); // bullet
153
154 QStringList secondary_info;
155 QString author_text =
156 QString::fromStdString(quest_info.properties.get_author());
157 QString date_string =
158 QString::fromStdString(quest_info.properties.get_release_date());
159 QDate date = QDate::fromString(date_string, "yyyyMMdd");
160 QString date_text = date.toString("yyyy");
161 if (!date_text.isEmpty()) {
162 secondary_info << date_text;
163 }
164 if (!author_text.isEmpty()) {
165 secondary_info << author_text;
166 }
167
168 if (!secondary_info.isEmpty()){
169 QString secondary_text = secondary_info.join(separator);
170 QString secondary_elided_text = author_font_metrics.elidedText(
171 secondary_text, Qt::ElideRight, author_rect_width);
172 painter->setPen(author_pen_color);
173 painter->setFont(author_font);
174 painter->drawText(author_rect, secondary_elided_text);
175 }
176 // Restore painter initial state
177 painter->restore();
178}
179
180QSize QuestsItemDelegate::sizeHint(const QStyleOptionViewItem &,
181 const QModelIndex &) const {
182 return QSize(150, _iconSize.height() + 16);
183}
184
185const QSize &QuestsItemDelegate::iconSize() const {
186 return _iconSize;
187}
188
189void QuestsItemDelegate::setIconSize(const QSize &iconSize) {
190 _iconSize = iconSize;
191}
