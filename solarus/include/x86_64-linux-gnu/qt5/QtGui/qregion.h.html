63class Q_GUI_EXPORT QRegion
64{
65public:
66 enum RegionType { Rectangle, Ellipse };
67
68 QRegion();
69 QRegion(int x, int y, int w, int h, RegionType t = Rectangle);
70 QRegion(const QRect &r, RegionType t = Rectangle);
71 QRegion(const QPolygon &pa, Qt::FillRule fillRule = Qt::OddEvenFill);
72 QRegion(const QRegion &region);
73 QRegion(const QBitmap &bitmap);
74 ~QRegion();
75 QRegion &operator=(const QRegion &);
76#ifdef Q_COMPILER_RVALUE_REFS
77 inline QRegion &operator=(QRegion &&other)
78 { qSwap(d, other.d); return *this; }
79#endif
80 inline void swap(QRegion &other) { qSwap(d, other.d); }
81 bool isEmpty() const;
82 bool isNull() const;
83
84 bool contains(const QPoint &p) const;
85 bool contains(const QRect &r) const;
86
87 void translate(int dx, int dy);
88 inline void translate(const QPoint &p) { translate(p.x(), p.y()); }
89 QRegion translated(int dx, int dy) const;
90 inline QRegion translated(const QPoint &p) const { return translated(p.x(), p.y()); }
91
92 QRegion united(const QRegion &r) const;
93 QRegion united(const QRect &r) const;
94 QRegion intersected(const QRegion &r) const;
95 QRegion intersected(const QRect &r) const;
96 QRegion subtracted(const QRegion &r) const;
97 QRegion xored(const QRegion &r) const;
98
99#if QT_DEPRECATED_SINCE(5, 0)
100 inline QT_DEPRECATED QRegion unite(const QRegion &r) const { return united(r); }
101 inline QT_DEPRECATED QRegion unite(const QRect &r) const { return united(r); }
102 inline QT_DEPRECATED QRegion intersect(const QRegion &r) const { return intersected(r); }
103 inline QT_DEPRECATED QRegion intersect(const QRect &r) const { return intersected(r); }
104 inline QT_DEPRECATED QRegion subtract(const QRegion &r) const { return subtracted(r); }
105 inline QT_DEPRECATED QRegion eor(const QRegion &r) const { return xored(r); }
106#endif
107
108 bool intersects(const QRegion &r) const;
109 bool intersects(const QRect &r) const;
110
111 QRect boundingRect() const;
112 QVector<QRect> rects() const;
113 void setRects(const QRect *rect, int num);
114 int rectCount() const;
115
116 const QRegion operator|(const QRegion &r) const;
117 const QRegion operator+(const QRegion &r) const;
118 const QRegion operator+(const QRect &r) const;
119 const QRegion operator&(const QRegion &r) const;
120 const QRegion operator&(const QRect &r) const;
121 const QRegion operator-(const QRegion &r) const;
122 const QRegion operator^(const QRegion &r) const;
123 QRegion& operator|=(const QRegion &r);
124 QRegion& operator+=(const QRegion &r);
125 QRegion& operator+=(const QRect &r);
126 QRegion& operator&=(const QRegion &r);
127 QRegion& operator&=(const QRect &r);
128 QRegion& operator-=(const QRegion &r);
129 QRegion& operator^=(const QRegion &r);
130
131 bool operator==(const QRegion &r) const;
132 inline bool operator!=(const QRegion &r) const { return !(operator==(r)); }
133 operator QVariant() const;
134
135#ifndef QT_NO_DATASTREAM
136 friend Q_GUI_EXPORT QDataStream &operator<<(QDataStream &, const QRegion &);
137 friend Q_GUI_EXPORT QDataStream &operator>>(QDataStream &, QRegion &);
138#endif
139private:
140 QRegion copy() const; // helper of detach.
141 void detach();
142Q_GUI_EXPORT
143 friend bool qt_region_strictContains(const QRegion &region,
144 const QRect &rect);
145 friend struct QRegionPrivate;

147#ifndef QT_NO_DATASTREAM
148 void exec(const QByteArray &ba, int ver = 0, QDataStream::ByteOrder byteOrder = QDataStream::BigEndian);

156 static void cleanUp(QRegionData *x);
157};
