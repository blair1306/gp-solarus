43 /**
44 * \brief Implementation of the resource() function.
45 * \param l The Lua state of the quest resource file.
46 * \return Number of values to return to Lua.
47 */
48 int l_resource_element(lua_State* l) {
49
50 return LuaTools::exception_boundary_handle(l, [&] {
51
52 lua_getfield(l, LUA_REGISTRYINDEX, "resources");
53 QuestResources& resources = *(static_cast<QuestResources*>(
54 lua_touserdata(l, -1)
55 ));
56 lua_pop(l, 1);
57
58 ResourceType resource_type =
59 LuaTools::check_enum<ResourceType>(l, 1);
60 const std::string& id = LuaTools::check_string_field(l, 2, "id");
61 const std::string& description = LuaTools::check_string_field(l, 2, "description");
62
63 resources.add(resource_type, id, description);
64
65 return 0;
66 });
67 }
68
69}

71/**
72 * \brief Creates an empty quest resources object.
73 */
74QuestResources::QuestResources() {
75
76 for (size_t i = 0 ; i < EnumInfoTraits<ResourceType>::names.size(); ++i) {
77 resource_maps.emplace(static_cast<ResourceType>(i), ResourceMap());
78 }
79}
80
81/**
82 * \brief Removes all resource elements.
83 */
84void QuestResources::clear() {
85
86 for (size_t i = 0 ; i < EnumInfoTraits<ResourceType>::names.size(); ++i) {
87 resource_maps[static_cast<ResourceType>(i)].clear();
88 }
89}

91/**
92 * \brief Returns whether there exists an element with the specified id.
93 * \param resource_type A type of resource.
94 * \param id The id to look for.
95 * \return \c true if there exists an element with the specified id in this
96 * resource type.
97 */
98bool QuestResources::exists(ResourceType resource_type, const std::string& id) const {
99
100 const ResourceMap& resource = get_elements(resource_type);
101 return resource.find(id) != resource.end();
102}
103
104/**
105 * \brief Returns the list of element IDs of the specified resource type.
106 * \param resource_type A type of resource.
107 * \return The ids of all declared element of this type
108 */
109const QuestResources::ResourceMap& QuestResources::get_elements(
110 ResourceType resource_type) const {
111
112 return resource_maps.find(resource_type)->second;
113}
114
115/**
116 * \brief Returns the list of element IDs of the specified resource type.
117 * \param resource_type A type of resource.
118 * \return The ids of all declared element of this type
119 */
120QuestResources::ResourceMap& QuestResources::get_elements(
121 ResourceType resource_type) {
122
123 return resource_maps.find(resource_type)->second;
124}
125
126/**
127 * \brief Adds a resource element to the list.
128 * \param resource_type A type of resource.
129 * \param id Id of the element to add.
130 * \param description Description the element to add.
131 * \return \c true if the element was added, \c false if an element with
132 * this id already exists.
133 */
134bool QuestResources::add(
135 ResourceType resource_type,
136 const std::string& id,
137 const std::string& description
138) {
139 ResourceMap& resource = get_elements(resource_type);
140 auto result = resource.emplace(id, description);
141 return result.second;
142}
143
144/**
145 * \brief Removes a resource element from the list.
146 * \param resource_type A type of resource.
147 * \param id Id of the element to remove.
148 * \return \c true if the element was removed, \c false if such an element
149 * did not exist.
150 */
151bool QuestResources::remove(
152 ResourceType resource_type,
153 const std::string& id
154) {
155 ResourceMap& resource = get_elements(resource_type);
156 return resource.erase(id) > 0;
157}
158
159/**
160 * \brief Changes the id of a resource element from the list.
161 * \param resource_type A type of resource.
162 * \param old_id Id of the element to change.
163 * \param new_id The new id to set.
164 * \return \c true in case of success, \c false if the old id does not
165 * exist or if the new id already exists.
166 */
167bool QuestResources::rename(
168 ResourceType resource_type,
169 const std::string& old_id,
170 const std::string& new_id
171) {
172 if (!exists(resource_type, old_id)) {
173 return false;
174 }
175 if (exists(resource_type, new_id)) {
176 return false;
177 }
178 const std::string& description = get_description(resource_type, old_id);
179 remove(resource_type, old_id);
180 add(resource_type, new_id, description);
181 return true;
182}
183
184/**
185 * \brief Returns the description of a resource element.
186 * \param resource_type A type of resource.
187 * \param id Id of the element to get.
188 * \return description The element description.
189 * Returns an empty string if the element does not exist.
190 */
191std::string QuestResources::get_description(
192 ResourceType resource_type,
193 const std::string& id
194) const {
195
196 const ResourceMap& resource = get_elements(resource_type);
197
198 const auto& it = resource.find(id);
199 if (it == resource.end()) {
200 return "";
201 }
202 return it->second;
203}
204
205/**
206 * \brief Sets the description of a resource element.
207 * \param resource_type A type of resource.
208 * \param id Id of the element to modify.
209 * \param description The new description.
210 * \return \c true in case of success, \c false if such an element does not
211 * exist.
212 */
213bool QuestResources::set_description(
214 ResourceType resource_type,
215 const std::string& id,
216 const std::string& description
217) {
218 if (!exists(resource_type, id)) {
219 return false;
220 }
221
222 ResourceMap& resource = get_elements(resource_type);
223 resource[id] = description;
224 return true;
225}

227/**
228 * \copydoc LuaData::import_from_lua
229 */
230bool QuestResources::import_from_lua(lua_State* l) {
231
232 lua_pushlightuserdata(l, this);
233 lua_setfield(l, LUA_REGISTRYINDEX, "resources");
234
235 // We register only one C function for all resource types.
236 lua_register(l, "resource", l_resource_element);
237 for (const auto& kvp: EnumInfoTraits<ResourceType>::names) {
238 std::ostringstream oss;
239 oss << "function " << kvp.second << "(t) resource('"
240 << kvp.second << "', t) end";
241 luaL_dostring(l, oss.str().c_str());
242 }
243
244 if (lua_pcall(l, 0, 0, 0) != 0) {
245 Debug::error(std::string("Failed to load quest resource list 'project_db.dat': ") + lua_tostring(l, -1));
246 lua_pop(l, 1);
247 return false;
248 }
249
250 return true;
251}

253/**
254 * \copydoc LuaData::export_to_lua
255 */
256bool QuestResources::export_to_lua(std::ostream& out) const {
257
258 // Save each resource.
259 for (const auto& kvp: EnumInfoTraits<ResourceType>::names) {
260
261 const ResourceMap& resource = get_elements(kvp.first);
262 for (const auto& element: resource) {
263
264 const std::string& id = element.first;
265 const std::string& description = element.second;
266
267 out << kvp.second
268 << "{ id = \""
269 << escape_string(id)
270 << "\", description = \""
271 << escape_string(description)
272 << "\" }\n";
273 }
274 out << "\n";
275 }
276
277 return true;
278}
