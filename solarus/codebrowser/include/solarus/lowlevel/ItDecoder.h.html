17#ifndef SOLARUS_IT_DECODER_H
18#define SOLARUS_IT_DECODER_H
19
20#include "solarus/Common.h"
21#include <cstddef>
22#include <memory>
23#include <string>
24#include <modplug.h>
25
26namespace Solarus {
27
28/**
29 * \brief Encapsulates the Impulse Tracker music decoding.
30 *
31 * This class allows the Music class to be independent of the Impulse Tracker decoding library.
32 */
33class ItDecoder {
34
35 public:
36
37 ItDecoder();
38
39 void load(const std::string& sound_buffer);
40 void unload();
41 int decode(void* decoded_data, int nb_samples);
42
43 int get_num_channels() const;
44 int get_channel_volume(int channel) const;
45 void set_channel_volume(int channel, int volume);
46 int get_tempo() const;
47 void set_tempo(int tempo);
48 bool loops() const;
49 void set_loops(bool loops);
50
51 private:
52
53 struct ModPlugFileDeleter {
54 void operator()(ModPlugFile* modplug_file) {
55 ModPlug_Unload(modplug_file);
56 }
57 };
58 using ModPlugFileUniquePtr = std::unique_ptr<ModPlugFile, ModPlugFileDeleter>;
59
60 ModPlugFileUniquePtr modplug_file;
61
62};
63
64}
65
66#endif
