51/**
52 * \brief Creates a door.
53 * \param game The game.
54 * \param name Name identifying this entity.
55 * \param layer Layer of the entity to create.
56 * \param xy Coordinates of the entity to create.
57 * \param direction Direction of the door.
58 * \param sprite_name Name of the animation set of the
59 * sprite to create for the door.
60 * The sprite must have an animation \c "closed", that will be shown while
61 * the door is closed. Optionally, they sprite can also have animations
62 * \c "opening" and \c "closing", that will be shown (if they exist)
63 * while the door is being opened or closed, respectively.
64 * If they don't exist, the door will get opened or closed instantly.
65 * \param savegame_variable Boolean variable where the door's state is saved
66 * (an empty string lets the door unsaved).
67 */
68Door::Door(Game& game,
69 const std::string& name,
70 int layer,
71 const Point& xy,
72 int direction,
73 const std::string& sprite_name,
74 const std::string& savegame_variable):
75 Entity(name, 0, layer, xy, Size(16, 16)),
76 savegame_variable(savegame_variable),
77 opening_method(OpeningMethod::NONE),
78 opening_condition(),
79 opening_condition_consumed(false),
80 cannot_open_dialog_id(),
81 state(OPEN),
82 initialized(false),
83 next_hint_sound_date(0) {
84
85 set_collision_modes(CollisionMode::COLLISION_FACING | CollisionMode::COLLISION_SPRITE);
86
87 const SpritePtr& sprite = create_sprite(sprite_name);
88 sprite->enable_pixel_collisions();
89 sprite->set_ignore_suspend(true); // Continue the animation while the camera is moving.
90 set_direction(direction);
91
92 if (is_saved()) {
93 set_open(game.get_savegame().get_boolean(savegame_variable));
94 }
95 else {
96 set_open(false);
97 }
98 sprite->set_current_direction(direction);
99}

161/**
162 * \brief Makes the door immediately open or closed.
163 * \param door_open true to make it opened, false to make it closed.
164 */
165void Door::set_open(bool door_open) {
166
167 state = door_open ? OPEN : CLOSED;
168
169 if (door_open) {
170 set_collision_modes(COLLISION_NONE); // to avoid being the hero's facing entity
171 }
172 else {
173
174 const SpritePtr& sprite = get_sprite();
175 if (sprite != nullptr) {
176 sprite->set_current_animation("closed");
177 }
178 set_collision_modes(COLLISION_FACING | COLLISION_SPRITE);
179
180 // ensure that we are not closing the door on the hero
181 if (is_on_map() && overlaps(get_hero())) {
182 get_hero().avoid_collision(*this, (get_direction() + 2) % 4);
183 }
184 }
185
186 if (is_on_map()) {
187 update_dynamic_tiles();
188
189 if (is_saved()) {
190 get_savegame().set_boolean(savegame_variable, door_open);
191 }
192
193 if (door_open) {
194 get_lua_context()->door_on_opened(*this);
195 }
196 else {
197 get_lua_context()->door_on_closed(*this);
198 }
199 }

202/**
203 * \brief Enables or disables the dynamic tiles related to this door.
204 *
205 * The dynamic tiles impacted by this function are the ones whose prefix is the door's name
206 * followed by "_closed" or "_open", depending on the door state.
207 */
208void Door::update_dynamic_tiles() {
209
210 std::vector<EntityPtr> tiles =
211 get_entities().get_entities_with_prefix(EntityType::DYNAMIC_TILE, get_name() + "_closed");
212 for (const EntityPtr& tile: tiles) {
213 tile->set_enabled(is_closed() || is_opening());
214 }
215
216 tiles = get_entities().get_entities_with_prefix(EntityType::DYNAMIC_TILE, get_name() + "_open");
217 for (const EntityPtr& tile: tiles) {
218 tile->set_enabled(is_open() || is_closing());
219 }
220}

222/**
223 * \brief Notifies this detector that a collision was just detected with another entity.
224 *
225 * This function is called by the engine when there is a collision with another entity.
226 *
227 * \param entity_overlapping the entity overlapping the detector
228 * \param collision_mode the collision mode that detected the collision
229 */
230void Door::notify_collision(Entity& entity_overlapping, CollisionMode /* collision_mode */) {
231
232 if (is_closed() && entity_overlapping.is_hero()) {
233
234 Hero& hero = static_cast<Hero&>(entity_overlapping);
235
236 if (get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_NONE
237 && hero.is_free()) {
238
239 if (can_open()) {
240 // The action command opens the door.
241 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_OPEN);
242 }
243 else if (!get_cannot_open_dialog_id().empty()) {
244 // The action command shows a dialog.
245 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_LOOK);
246 }
247 }
248 }
249}

413/**
414 * \brief Returns whether the required conditions are met to open this door.
415 * \return \c true if the hero can currently open the door by pressing the
416 * action command.
417 */
418bool Door::can_open() const {
419
420 switch (get_opening_method()) {
421
422 case OpeningMethod::BY_INTERACTION:
423 // No condition: the hero can always open the door.
424 return true;
425
426 case OpeningMethod::BY_INTERACTION_IF_SAVEGAME_VARIABLE:
427 {
428 // The hero can open the door if a savegame variable is set.
429 const std::string& required_savegame_variable = get_opening_condition();
430 if (required_savegame_variable.empty()) {
431 return false;
432 }
433
434 const Savegame& savegame = get_savegame();
435 if (savegame.is_boolean(required_savegame_variable)) {
436 return savegame.get_boolean(required_savegame_variable);
437 }
438
439 if (savegame.is_integer(required_savegame_variable)) {
440 return savegame.get_integer(required_savegame_variable) > 0;
441 }
442
443 if (savegame.is_string(required_savegame_variable)) {
444 return !savegame.get_string(required_savegame_variable).empty();
445 }
446
447 return false;
448 }
449
450 case OpeningMethod::BY_INTERACTION_IF_ITEM:
451 {
452 // The hero can open the door if he has an item.
453 const std::string& required_item_name = get_opening_condition();
454 if (required_item_name.empty()) {
455 return false;
456 }
457 const EquipmentItem& item = get_equipment().get_item(required_item_name);
458 return item.is_saved()
459 && item.get_variant() > 0
460 && (!item.has_amount() || item.get_amount() > 0);
461 }
462
463 default:
464 return false;
465 }
466}

502/**
503 * \brief Updates the entity.
504 */
505void Door::update() {
506
507 Entity::update();
508
509 if (!initialized) {
510 update_dynamic_tiles();
511 initialized = true;
512 }
513
514 if (is_closed()
515 && get_opening_method() == OpeningMethod::BY_EXPLOSION
516 && get_equipment().has_ability(Ability::DETECT_WEAK_WALLS)
517 && Geometry::get_distance(get_center_point(), get_hero().get_center_point()) < 40
518 && !is_suspended()
519 && System::now() >= next_hint_sound_date) {
520 Sound::play("cane");
521 next_hint_sound_date = System::now() + 500;
522 }
523
524 const SpritePtr& sprite = get_sprite();
525 if (is_changing() &&
526 sprite != nullptr &&
527 sprite->is_animation_finished()) {
528 // Toggle door_open when the changing animation finishes.
529 set_open(is_opening());
530 }
531
532 if (is_saved() && !is_changing()) {
533 bool open_in_savegame = get_savegame().get_boolean(savegame_variable);
534 if (open_in_savegame && is_closed()) {
535 set_opening();
536 }
537 else if (!open_in_savegame && is_open()) {
538 set_closing();
539 }
540 }
541}

553/**
554 * \copydoc Entity::notify_action_command_pressed
555 */
556bool Door::notify_action_command_pressed() {
557
558 if (get_hero().is_free() &&
559 is_closed() &&
560 get_commands_effects().get_action_key_effect() != CommandsEffects::ACTION_KEY_NONE
561 ) {
562
563 if (can_open()) {
564 Sound::play("door_unlocked");
565 Sound::play("door_open");
566
567 if (is_saved()) {
568 get_savegame().set_boolean(savegame_variable, true);
569 }
570
571 if (is_opening_condition_consumed()) {
572 consume_opening_condition();
573 }
574
575 set_opening();
576
577 get_hero().check_position();
578 }
579 else if (!cannot_open_dialog_id.empty()) {
580 Sound::play("wrong");
581 get_game().start_dialog(cannot_open_dialog_id, ScopedLuaRef(), ScopedLuaRef());
582 }
583
584 return true;
585 }
586
587 return false;
588}

590/**
591 * \brief Consumes the savegame variable or the equipment item that was required
592 * to open the door.
593 */
594void Door::consume_opening_condition() {
595
596 switch (get_opening_method()) {
597
598 case OpeningMethod::BY_INTERACTION_IF_SAVEGAME_VARIABLE:
599 {
600 // Reset or decrement the savegame variable that was required.
601 const std::string& required_savegame_variable = get_opening_condition();
602 Savegame& savegame = get_savegame();
603 if (!required_savegame_variable.empty()) {
604 if (savegame.is_boolean(required_savegame_variable)) {
605 savegame.set_boolean(required_savegame_variable, false);
606 }
607 else if (savegame.is_integer(required_savegame_variable)) {
608 int current_value = savegame.get_integer(required_savegame_variable);
609 savegame.set_integer(required_savegame_variable, current_value - 1);
610 }
611 else if (savegame.is_string(required_savegame_variable)) {
612 savegame.set_string(required_savegame_variable, "");
613 }
614 }
615 break;
616 }
617
618 case OpeningMethod::BY_INTERACTION_IF_ITEM:
619 {
620 // Remove the equipment item that was required.
621 if (!opening_condition.empty()) {
622 EquipmentItem& item = get_equipment().get_item(opening_condition);
623 if (item.is_saved() && item.get_variant() > 0) {
624 if (item.has_amount()) {
625 item.set_amount(item.get_amount() - 1);
626 }
627 else {
628 item.set_variant(0);
629 }
630 }
631 }
632 break;
633 }
634
635 default:
636 // Ignored.
637 break;
638 }
639}
