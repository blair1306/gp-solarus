33/**
34 * \brief Stores the tree of what surfaces have to be drawn on other surfaces.
35 *
36 * When a drawing is requested, if the destination surface is in GPU, no
37 * drawing actually occurs: instead, the information is stored into this tree.
38 * At rendering time, the tree is traversed to perform all drawings
39 * accelerated in GPU.
40 *
41 * The root node of the tree is the screen, and the children nodes are the
42 * surfaces drawn on it.
43 *
44 * Each node represents a source surface drawn somewhere, and the list of
45 * surfaces drawn on itself.
46 */
47class Surface::SubSurfaceNode {
48
49 public:
50
51 /**
52 * Creates a subsurface node.
53 * \param src_surface The surface to draw.
54 * \param src_rect Region of the surface to draw.
55 * \param dst_rect The rectangle where to draw the surface, relative to
56 * the parent surface.
57 * \param subsurfaces Surfaces drawn onto src_surface.
58 */
59 SubSurfaceNode(
60 SurfacePtr src_surface,
61 const Rectangle& src_rect,
62 const Rectangle& dst_rect,
63 const std::vector<SubSurfaceNodePtr>& subsurfaces
64 ):
65 src_surface(src_surface),
66 src_rect(src_rect),
67 dst_rect(dst_rect),
68 subsurfaces(subsurfaces) {
69
70 // Clip the source rectangle to the size of the source surface.
71 // Otherwise, SDL_RenderCopy() will stretch the image.
72 // FIXME still buggy with software renderer for now but should be fixed soon :
73 // https://bugzilla.libsdl.org/show_bug.cgi?id=1968
74 if (this->src_rect.get_x() < 0) {
75 this->src_rect.set_x(0);
76 this->src_rect.set_width(this->src_rect.get_width() + src_rect.get_x());
77 this->dst_rect.add_x(-src_rect.get_x());
78 }
79 if (this->src_rect.get_x() + this->src_rect.get_width() > src_surface->get_width()) {
80 this->src_rect.set_width(src_surface->get_width() - this->src_rect.get_x());
81 }
82 if (src_rect.get_y() < 0) {
83 this->src_rect.set_y(0);
84 this->src_rect.set_height(this->src_rect.get_height() + src_rect.get_y());
85 this->dst_rect.add_y(-src_rect.get_y());
86 }
87 if (this->src_rect.get_y() + this->src_rect.get_height() > src_surface->get_height()) {
88 this->src_rect.set_height(src_surface->get_height() - this->src_rect.get_y());
89 }
90
91 }
92
93 SurfacePtr src_surface; /**< Surface to draw. */
94 Rectangle src_rect; /**< Region of the surface to draw. */
95 Rectangle dst_rect; /**< The rectangle where to draw the surface, relative to the parent surface. */
96 std::vector<SubSurfaceNodePtr> subsurfaces; /**< Subsurfaces drawn onto src_surface. */
97};

179/**
180 * \brief Creates a surface from the specified image file name.
181 *
182 * This function acts like a constructor excepts that it returns nullptr if the
183 * file does not exist or is not a valid image.
184 *
185 * \param file_name Name of the image file to load, relative to the base directory specified.
186 * \param base_directory The base directory to use.
187 * \return The surface created, or nullptr if the file could not be loaded.
188 */
189SurfacePtr Surface::create(const std::string& file_name,
190 ImageDirectory base_directory) {
191
192 SDL_Surface* sdl_surface = get_surface_from_file(file_name, base_directory);
193
194 if (sdl_surface == nullptr) {
195 return nullptr;
196 }
197
198 SurfacePtr surface = std::make_shared<Surface>(sdl_surface);
199 return surface;
200}

202/**
203 * \brief Creates an SDL surface corresponding to the requested file.
204 *
205 * The returned SDL surface has to be manually deleted.
206 *
207 * \param file_name Name of the image file to load, relative to the base directory specified.
208 * \param base_directory The base directory to use.
209 * \return The SDL surface created.
210 */
211SDL_Surface* Surface::get_surface_from_file(
212 const std::string& file_name,
213 ImageDirectory base_directory) {
214
215 std::string prefix;
216 bool language_specific = false;
217
218 if (base_directory == DIR_SPRITES) {
219 prefix = "sprites/";
220 }
221 else if (base_directory == DIR_LANGUAGE) {
222 language_specific = true;
223 prefix = "images/";
224 }
225 std::string prefixed_file_name = prefix + file_name;
226
227 if (!QuestFiles::data_file_exists(prefixed_file_name, language_specific)) {
228 // File not found.
229 return nullptr;
230 }
231
232 const std::string& buffer = QuestFiles::data_file_read(prefixed_file_name, language_specific);
233 SDL_RWops* rw = SDL_RWFromMem(const_cast<char*>(buffer.data()), (int) buffer.size());
234
235 SDL_Surface* software_surface = IMG_Load_RW(rw, 0);
236
237 SDL_RWclose(rw);
238
239 Debug::check_assertion(software_surface != nullptr,
240 std::string("Cannot load image '") + prefixed_file_name + "'");
241
242 SDL_PixelFormat* pixel_format = Video::get_pixel_format();
243 if (software_surface->format->format == pixel_format->format) {
244 return software_surface;
245 }
246
247 // Convert to the preferred pixel format.
248 uint8_t opacity;
249 SDL_GetSurfaceAlphaMod(software_surface, &opacity);
250 SDL_Surface* converted_surface = SDL_ConvertSurface(
251 software_surface,
252 pixel_format,
253 0
254 );
255 Debug::check_assertion(converted_surface != nullptr,
256 std::string("Failed to convert software surface: ") + SDL_GetError());
257 SDL_FreeSurface(software_surface);
258
259 SDL_SetSurfaceAlphaMod(converted_surface, opacity); // Re-apply the alpha.
260 SDL_SetSurfaceBlendMode(converted_surface, SDL_BLENDMODE_BLEND);
261
262 return converted_surface;
263}

265/**
266 * \brief Creates a hardware texture from the software surface.
267 *
268 * Also converts the software surface to a preferred format if necessary.
269 */
270void Surface::create_texture_from_surface() {
271
272 SDL_Renderer* main_renderer = Video::get_renderer();
273 if (main_renderer != nullptr) {
274
275 Debug::check_assertion(internal_surface != nullptr,
276 "Missing software surface to create texture from");
277
278 // Create the texture.
279 internal_texture = SDL_Texture_UniquePtr(
280 SDL_CreateTexture(
281 main_renderer,
282 Video::get_pixel_format()->format,
283 SDL_TEXTUREACCESS_STATIC,
284 internal_surface->w,
285 internal_surface->h
286 )
287 );
288 SDL_SetTextureBlendMode(internal_texture.get(), get_sdl_blend_mode());
289
290 // Copy the pixels of the software surface to the GPU texture.
291 SDL_UpdateTexture(internal_texture.get(), nullptr, internal_surface->pixels, internal_surface->pitch);
292 SDL_GetSurfaceAlphaMod(internal_surface.get(), &opacity);
293 }
294}

332void Surface::set_opacity(uint8_t opacity) {
333
334 this->opacity = opacity;
335
336 if (software_destination // The destination surface is in RAM.

338 ) {
339 if (internal_surface == nullptr) {

341 }
342
343 int error = SDL_SetSurfaceAlphaMod(internal_surface.get(), opacity);
344 if (error != 0) {
345 Debug::error(SDL_GetError());
346 }
347 is_rendered = false; // The surface has changed.
348 }

350 // If this is a hardware surface, the opacity is applied later.
351}

353/**
354 * \brief Returns a buffer of the raw pixels of this surface.
355 *
356 * Pixels returned have the RGBA 32-bit format.
357 *
358 * \return The pixel buffer.
359 */
360std::string Surface::get_pixels() const {
361
362 if (!software_destination &&
363 Video::is_acceleration_enabled()) {
364 // The surface is in GPU.
365 return ""; // TODO
366 }
367
368 const int num_pixels = get_width() * get_height();
369 if (internal_surface == nullptr) {
370 // No surface: this may be a color.
371
372 if (internal_color == nullptr) {
373 // No color either: fully transparent surface.
374 return std::string(num_pixels * 4, (size_t) 0);
375 }
376
377 uint8_t r, g, b, a;
378 internal_color->get_components(r, g, b, a);
379 std::string pixel;
380 pixel += r;
381 pixel += g;
382 pixel += b;
383 pixel += a;
384 std::ostringstream oss;
385 for (int i = 0; i < num_pixels; ++i) {
386 oss << pixel;
387 }
388 return oss.str();
389 }
390
391 if (internal_surface->format->format == SDL_PIXELFORMAT_ABGR8888) {
392 // No conversion needed.
393 const char* buffer = static_cast<const char*>(internal_surface->pixels);
394 return std::string(buffer, num_pixels * 4);
395 }
396
397 // Convert to RGBA format.
398 SDL_PixelFormat* format = SDL_AllocFormat(SDL_PIXELFORMAT_ABGR8888); // TODO keep this object
399 SDL_Surface_UniquePtr converted_surface(SDL_ConvertSurface(
400 internal_surface.get(),
401 format,
402 0
403 ));
404 SDL_FreeFormat(format);
405 Debug::check_assertion(converted_surface != nullptr,
406 std::string("Failed to convert pixels to RGBA format") + SDL_GetError());
407 const char* buffer = static_cast<const char*>(converted_surface->pixels);
408 return std::string(buffer, num_pixels * 4);
409}

411/**
412 * When this surface is used as the destination of a drawing operation,
413 * returns whether the drawing operation is performed in RAM or by the GPU.
414 *
415 * By default, this setting is true and all drawing operations are performed
416 * in RAM.
417 * Otherwise, when 2D acceleration is active, drawing operations are delayed
418 * and performed by the GPU at rendering time.
419 *
420 * \return Whether this surface is a software destination surface.
421 */
422bool Surface::is_software_destination() const {
423 return software_destination;
424}
425
426/**
427 * When this surface is used as the destination of a drawing operation,
428 * sets whether the drawing operation is performed immediately in RAM or later
429 * by the GPU.
430 *
431 * By default, this setting is true and all drawing operations are performed
432 * in RAM.
433 * Otherwise, when 2D acceleration is active, drawing operations are delayed
434 * and performed by the GPU at rendering time.
435 *
436 * You should leave this to \c true if your surface is built from lots of source
437 * surfaces that don't change often.
438 *
439 * Hardware destinations are intended to be used for internal optimizations of
440 * the engine. They do not support all operations that software ones do.
441 * In particular, a hardware surface can never be drawn on a software surface.
442 * Also, when you draw on a hardware surface after it was rendered, previous
443 * drawings on this surface get automatically cleared.
444 *
445 * Use hardware surfaces only if you know what you are doing.
446 * If in doubt, leave this to \c true.
447 *
448 * \param software_destination Whether this surface is a software destination
449 * surface.
450 */
451void Surface::set_software_destination(bool software_destination) {
452
453 this->software_destination = software_destination;
454 // The software surface if any will be created lazily.
455}

457/**
458 * \brief Creates an internal surface in software mode for this surface.
459 */
460void Surface::create_software_surface() {
461
462 Debug::check_assertion(internal_surface == nullptr,
463 "Software surface already exists");
464
465 // Create a surface with the appropriate pixel format.
466 SDL_PixelFormat* format = Video::get_pixel_format();
467 internal_surface = SDL_Surface_UniquePtr(
468 SDL_CreateRGBSurface(
469 0,
470 width,
471 height,
472 32,
473 format->Rmask,
474 format->Gmask,
475 format->Bmask,
476 format->Amask
477 )
478 );
479
480 Debug::check_assertion(internal_surface != nullptr,
481 std::string("Failed to create software surface: ") + SDL_GetError());
482
483 SDL_SetSurfaceBlendMode(internal_surface.get(), get_sdl_blend_mode());
484 is_rendered = false;
485}

487/**
488 * \brief Clears this surface.
489 *
490 * The surface becomes fully transparent and its size remains unchanged.
491 * The opacity property of the surface is preserved.
492 */
493void Surface::clear() {
494
495 clear_subsurfaces();
496
497 internal_color = nullptr;
498
499 if (internal_texture != nullptr) {
500 internal_texture = nullptr;
501 }
502
503 if (internal_surface != nullptr) {
504 if (software_destination) {
505 SDL_FillRect(
506 internal_surface.get(),
507 nullptr,
508 get_color_value(Color::transparent)
509 );
510 }
511 else {
512 internal_surface = nullptr;
513 }
514 }
515}

529
530 if (internal_surface == nullptr) {
531 // Nothing to do.

533 }
534
535 SDL_FillRect(
536 internal_surface.get(),
537 where.get_internal_rect(),
538 get_color_value(Color::transparent)
539 );
540 is_rendered = false; // The surface has changed.
541}

574/**
575 * \brief Add a SubSurface to draw on this surface.
576 * \param src_surface The Surface to draw.
577 * \param region The subrectangle to draw in the source surface.
578 * \param dst_position Coordinates on this surface.
579 */
580void Surface::add_subsurface(
581 const SurfacePtr& src_surface,
582 const Rectangle& region,
583 const Point& dst_position) {
584
585 SubSurfaceNodePtr node(new SubSurfaceNode(
586 src_surface,
587 region,
588 Rectangle(dst_position),
589 src_surface->subsurfaces
590 ));
591
592 // Clear the subsurface queue if the current dst_surface has already been rendered.
593 if (is_rendered) {
594 clear_subsurfaces();
595 }
596
597 subsurfaces.push_back(node);
598}

604
605 subsurfaces.clear();

619/**
620 * \brief Draws a subrectangle of this surface on another surface.
621 * \param region The subrectangle to draw in this object.
622 * \param dst_surface The destination surface.
623 * \param dst_position Coordinates on the destination surface.
624 */
625void Surface::raw_draw_region(
626 const Rectangle& region,
627 Surface& dst_surface,
628 const Point& dst_position) {
629
630 if (dst_surface.software_destination // The destination surface is in RAM.
631 || !Video::is_acceleration_enabled() // The rendering is in RAM.
632 ) {
633
634 if (dst_surface.internal_surface == nullptr) {
635 dst_surface.create_software_surface();
636 }
637
638 // First, draw subsurfaces if any.
639 // They can exist if the video mode recently switched from an accelerated
640 // one to a software one.
641 if (!subsurfaces.empty()) {
642
643 if (this->internal_surface == nullptr) {
644 create_software_surface();
645 }
646
647 std::vector<SubSurfaceNodePtr> subsurfaces = this->subsurfaces;
648 this->subsurfaces.clear(); // Avoid infinite recursive calls if there are cycles.
649
650 for (SubSurfaceNodePtr& subsurface: subsurfaces) {
651
652 // TODO draw the subsurfaces of the whole tree recursively instead.
653 // The current version is not correct because it handles only one level
654 // (it ignores subsurface->subsurfaces).
655 // Plus it needs the workaround above to avoid a stack overflow.
656 subsurface->src_surface->raw_draw_region(
657 subsurface->src_rect,
658 *this,
659 subsurface->dst_rect.get_xy()
660 );
661 subsurface = nullptr;
662 }
663 clear_subsurfaces();
664 }
665
666 if (this->internal_surface != nullptr) {
667 // The source surface is not empty: draw it onto the destination.
668
669 SDL_SetSurfaceBlendMode(
670 this->internal_surface.get(),
671 get_sdl_blend_mode()
672 );
673 SDL_BlitSurface(
674 this->internal_surface.get(),
675 region.get_internal_rect(),
676 dst_surface.internal_surface.get(),
677 Rectangle(dst_position).get_internal_rect()
678 );
679 }
680 else if (internal_color != nullptr) { // No internal surface to draw: this may be a color.
681
682 if (get_blend_mode() == BlendMode::BLEND && internal_color->get_alpha() == 255) {
683 // Fill with opaque color: we can directly modify the destination pixels.
684 Rectangle dst_rect(
685 dst_position,
686 region.get_size()
687 );
688 SDL_FillRect(
689 dst_surface.internal_surface.get(),
690 dst_rect.get_internal_rect(),
691 get_color_value(*internal_color)
692 );
693 }
694 else {
695 // Fill with semi-transparent pixels: perform alpha-blending.
696 create_software_surface();
697 SDL_FillRect(
698 this->internal_surface.get(),
699 nullptr,
700 get_color_value(*internal_color)
701 );
702 SDL_BlitSurface(
703 this->internal_surface.get(),
704 region.get_internal_rect(),
705 dst_surface.internal_surface.get(),
706 Rectangle(dst_position).get_internal_rect()
707 );
708 }
709 }
710 }
711 else {
712 // The destination is a GPU surface (a texture).
713 // Do not draw anything, just store the operation in the tree instead.
714 // The actual drawing will be done at rendering time in GPU.
715
716 SurfacePtr src_surface = std::static_pointer_cast<Surface>(shared_from_this());
717 dst_surface.add_subsurface(src_surface, region, dst_position);
718 }
719
720 dst_surface.is_rendered = false;
721}

731/**
732 * \brief Draws this software surface with a pixel filter on another software
733 * surface.
734 * \param pixel_filter The pixel filter to apply.
735 * \param dst_surface The destination surface. It must have the size of the
736 * this surface multiplied by the scaling factor of the filter.
737 */
738void Surface::apply_pixel_filter(
739 const PixelFilter& pixel_filter, Surface& dst_surface) {
740
741 const int factor = pixel_filter.get_scaling_factor();
742 Debug::check_assertion(dst_surface.get_width() == get_width() * factor,
743 "Wrong destination surface size");
744 Debug::check_assertion(dst_surface.get_height() == get_height() * factor,
745 "Wrong destination surface size");
746
747 SDL_Surface* src_internal_surface = this->internal_surface.get();
748 SDL_Surface* dst_internal_surface = dst_surface.internal_surface.get();
749
750 if (src_internal_surface == nullptr) {
751 // This is possible if nothing was drawn on the surface yet.
752 return;
753 }
754
755 Debug::check_assertion(dst_internal_surface != nullptr,
756 "Missing software destination surface for pixel filter");
757
758 SDL_LockSurface(src_internal_surface);
759 SDL_LockSurface(dst_internal_surface);
760
761 uint32_t* src = static_cast<uint32_t*>(src_internal_surface->pixels);
762 uint32_t* dst = static_cast<uint32_t*>(dst_internal_surface->pixels);
763
764 pixel_filter.filter(src, get_width(), get_height(), dst);
765
766 SDL_UnlockSurface(dst_internal_surface);
767 SDL_UnlockSurface(src_internal_surface);
768
769 // The destination surface has changed.
770 dst_surface.is_rendered = false;
771}

785 * \brief Renders the internal texture if any, and all subsurfaces that are
786 * drawn onto it.
787 * \param renderer The renderer where to draw.
788 * \param src_rect The subrectangle of the texture to draw.
789 * \param dst_rect The position where to draw on the renderer.
790 * \param clip_rect A portion of the renderer where to restrict the drawing.
791 * \param opacity The opacity of the parent surface.
792 * \param subsurfaces The subsurfaces drawn onto this texture. They will be
793 * renderered recursively.
794 */
795void Surface::render(
796 SDL_Renderer* renderer,
797 const Rectangle& src_rect,
798 const Rectangle& dst_rect,
799 const Rectangle& clip_rect,
800 uint8_t opacity,
801 const std::vector<SubSurfaceNodePtr>& subsurfaces
802) {
803
804 //FIXME SDL_RenderSetClipRect is buggy for now, but should be fixed soon.
805 // It means that software and hardware surface doesn't have the exact same behavior for now.
806 // Uncomment the two lines using it when https://bugzilla.libsdl.org/show_bug.cgi?id=2336 will be solved.
807
808 // Accelerate the internal software surface.
809 if (internal_surface != nullptr) {
810
811 if (internal_texture == nullptr) {
812 create_texture_from_surface();
813 }
814
815 // If the software surface has changed, update the hardware texture.
816 else if (
817 (software_destination || !Video::is_acceleration_enabled())
818 && !is_rendered) {
819 SDL_UpdateTexture(
820 internal_texture.get(),
821 nullptr,
822 internal_surface->pixels,
823 internal_surface->pitch
824 );
825 SDL_GetSurfaceAlphaMod(internal_surface.get(), &this->opacity);
826 }
827 }
828
829 const uint8_t current_opacity = std::min(this->opacity, opacity);
830
831 // Draw the internal color as background color.
832 if (internal_color != nullptr) {
833 uint8_t r, g, b, a;
834 internal_color->get_components(r, g, b, a);
835
836 SDL_SetRenderDrawColor(renderer, r, g, b, std::min((uint8_t) a, current_opacity));
837 //SDL_RenderSetClipRect(renderer, clip_rect.get_internal_rect());
838 SDL_RenderFillRect(renderer, clip_rect.get_internal_rect());
839 }
840
841 // Draw the internal texture.
842 if (internal_texture != nullptr) {
843 SDL_SetTextureAlphaMod(internal_texture.get(), current_opacity);
844 //SDL_RenderSetClipRect(renderer, clip_rect.get_internal_rect());
845
846 SDL_RenderCopy(
847 renderer,
848 internal_texture.get(),
849 src_rect.get_internal_rect(),
850 dst_rect.get_internal_rect()
851 );
852 }
853
854 // The surface is rendered. Now draw all subtextures.
855 for (const SubSurfaceNodePtr& subsurface: subsurfaces) {
856
857 // subsurface has to be drawn on this surface
858
859 // Calculate absolute destination subrectangle position on screen.
860 Rectangle subsurface_dst_rect(
861 dst_rect.get_xy() + subsurface->dst_rect.get_xy() - src_rect.get_xy(),
862 subsurface->src_rect.get_size()
863 );
864
865 // Set the intersection of the subsurface destination and this surface's clip as clipping rectangle.
866 Rectangle superimposed_clip_rect;
867 if (SDL_IntersectRect(subsurface_dst_rect.get_internal_rect(),
868 clip_rect.get_internal_rect(),
869 superimposed_clip_rect.get_internal_rect())) {
870
871 // If there is an intersection, render the subsurface.
872 subsurface->src_surface->render(
873 renderer,
874 subsurface->src_rect,
875 subsurface_dst_rect,
876 superimposed_clip_rect,
877 current_opacity,
878 subsurface->subsurfaces
879 );
880 }
881 }
882
883 is_rendered = true;
884}

895/**
896 * \brief Returns a pixel value of this surface.
897 *
898 * The pixel format is preserved: if it is lower than 32 bpp, then the unused
899 * upper bits of the value are is padded with zeros.
900 *
901 * \param index Index of the pixel to get.
902 * \return The value of this pixel.
903 */
904uint32_t Surface::get_pixel(int index) const {
905
906 SOLARUS_ASSERT(internal_surface != nullptr,
907 "Attempt to read a pixel on a hardware or a buffer surface.");
908
909 SDL_PixelFormat* format = internal_surface->format;
910
911 // Test from the most common to the most exotic.
912 switch (format->BytesPerPixel) {
913
914 case 1:
915 {
916 uint8_t* pixels = static_cast<uint8_t*>(internal_surface->pixels);
917 return pixels[index];
918 }
919
920 case 4:
921 {
922 uint32_t* pixels = static_cast<uint32_t*>(internal_surface->pixels);
923 return pixels[index];
924 }
925
926 case 2:
927 {
928 uint16_t* pixels = static_cast<uint16_t*>(internal_surface->pixels);
929 return pixels[index];
930 }
931
932 case 3:
933 {
934 // Manual cast of the pixel into uint32_t.
935 uint8_t* bytes = static_cast<uint8_t*>(internal_surface->pixels);
936 return *reinterpret_cast<uint32_t*>(&bytes[index * 3]) & 0xffffff00 >> 8;
937 }
938 }
939
940 std::ostringstream oss;
941 oss << "Unknown pixel depth: " << format->BitsPerPixel;
942 Debug::die(oss.str());
943}

945/**
946 * \brief Returns whether a pixel is transparent.
947 *
948 * A pixel is transparent if it corresponds to the colorkey
949 * or if its alpha channel is equal to 0.
950 *
951 * \param index The index of the pixel to test.
952 * \return \c true if the pixel is transparent.
953 */
954bool Surface::is_pixel_transparent(int index) const {
955
956 uint32_t pixel = get_pixel(index);
957 uint32_t colorkey;
958 bool with_colorkey = SDL_GetColorKey(internal_surface.get(), &colorkey) == 0;
959
960 if (with_colorkey && pixel == colorkey) {
961 // The pixel has the transparency color.
962 return true;
963 }
964
965 if (internal_surface->format->Amask != 0 // There exists an alpha channel.
966 && (pixel & internal_surface->format->Amask) == 0 // The pixel is fully transparent.
967 ) {
968 return true;
969 }
970
971 return false;
972}
