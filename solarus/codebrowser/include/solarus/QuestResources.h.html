17#ifndef SOLARUS_QUEST_RESOURCE_H
18#define SOLARUS_QUEST_RESOURCE_H
19
20#include "solarus/Common.h"
21#include "solarus/EnumInfo.h"
22#include "solarus/lua/LuaData.h"
23#include "solarus/ResourceType.h"
24#include <iosfwd>
25#include <map>
26#include <string>
27
28namespace Solarus {
29
30template <>
31struct SOLARUS_API EnumInfoTraits<ResourceType> {
32 static const std::string pretty_name;
33
34 static const EnumInfo<ResourceType>::names_type names;
35};
36
37/**
38 * \brief Stores the resource list of a quest.
39 *
40 * This class basically stores the content of a resource list file
41 * project_db.dat.
42 * It does not create, remove or rename any resource file.
43 */
44class SOLARUS_API QuestResources : public LuaData {
45
46 public:
47
48 /**
49 * Mapping of resource element ids to their description.
50 */
51 using ResourceMap = std::map<std::string, std::string>;
52
53 QuestResources();
54
55 void clear();
56
57 bool exists(ResourceType resource_type, const std::string& id) const;
58 const ResourceMap& get_elements(
59 ResourceType resource_type
60 ) const;
61 ResourceMap& get_elements(
62 ResourceType resource_type
63 );
64
65 bool add(
66 ResourceType resource_type,
67 const std::string& id,
68 const std::string& description
69 );
70 bool remove(
71 ResourceType resource_type,
72 const std::string& id
73 );
74 bool rename(
75 ResourceType resource_type,
76 const std::string& old_id,
77 const std::string& new_id
78 );
79 std::string get_description(
80 ResourceType resource_type,
81 const std::string& id
82 ) const;
83 bool set_description(
84 ResourceType resource_type,
85 const std::string& id,
86 const std::string& description
87 );
88
89 virtual bool import_from_lua(lua_State* l) override;
90 virtual bool export_to_lua(std::ostream& out) const override;
91
92 private:
93
94 std::map<ResourceType, ResourceMap> resource_maps;
95
96};
97
98}
