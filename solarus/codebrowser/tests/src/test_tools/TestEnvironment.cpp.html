29/**
30 * \brief Creates a test environment.
31 */
32TestEnvironment::TestEnvironment(int argc, char** argv):
33 arguments(argc, argv),
34 main_loop(arguments),
35 map_id("traversable") {
36
37 Debug::set_show_popup_on_die(false);
38 Debug::set_die_on_error(true);
39 Debug::set_abort_on_die(true);
40}

42/**
43 * \brief Returns the runtime arguments that were passed to the test.
44 * \return The runtime arguments.
45 */
46const Arguments& TestEnvironment::get_arguments() const {
47 return arguments;
48}

50/**
51 * \brief Returns the main loop.
52 */
53MainLoop& TestEnvironment::get_main_loop() {
54 return main_loop;
55}
56
57/**
58 * \brief Returns the game, creating it if necessary.
59 */
60Game& TestEnvironment::get_game() {
61
62 if (main_loop.get_game() == nullptr) {
63 std::shared_ptr<Savegame> savegame = std::make_shared<Savegame>(
64 main_loop, "save_initial.dat"
65 );
66 savegame->initialize();
67 savegame->set_string(Savegame::KEY_STARTING_MAP, map_id);
68 Game* game = new Game(main_loop, savegame);
69 main_loop.set_game(game);
70 step(); // Advance one tick to start the game.
71 }
72
73 Debug::check_assertion(main_loop.get_game() != nullptr, "Missing game");
74
75 return *main_loop.get_game();
76}
77
78/**
79 * \brief Returns the current map, creating it if necessary.
80 */
81Map& TestEnvironment::get_map() {
82
83 if (!get_game().has_current_map()) {
84 step(); // Advance one tick to start the map.
85 Debug::check_assertion(get_game().has_current_map(), "Missing map");
86 }
87
88 return get_game().get_current_map();
89}

107/**
108 * \brief Runs the main loop on the specified map.
109 * \param map_id Id of the map to open.
110 */
111void TestEnvironment::run_map(const std::string& map_id) {
112
113 this->map_id = map_id;
114 get_game(); // Create a game and start it on the map.
115 get_main_loop().run();
116}

118/**
119 * \brief Creates a custom entity on the map and returns it.
120 * \param xy Coordinates of the entity to create.
121 * \param layer Layer of the entity to create.
122 */
123template<>
124std::shared_ptr<CustomEntity> TestEnvironment::make_entity<CustomEntity>(
125 const Point& xy,
126 int layer
127) {
128
129 std::shared_ptr<CustomEntity> entity = std::make_shared<CustomEntity>(
130 get_game(),
131 "",
132 0,
133 layer,
134 xy,
135 Size(16, 16),
136 "",
137 ""
138 );
139 get_entities().add_entity(entity);
140
141 return entity;
142}
143
144/**
145 * \brief Creates an NPC on the map and returns it.
146 * \param xy Coordinates of the entity to create.
147 * \param layer Layer of the entity to create.
148 */
149template<>
150std::shared_ptr<Npc> TestEnvironment::make_entity<Npc>(
151 const Point& xy,
152 int layer
153) {
154
155 std::shared_ptr<Npc> entity = std::make_shared<Npc>(
156 get_game(),
157 "",
158 layer,
159 xy,
160 Npc::Subtype::GENERALIZED_NPC,
161 "",
162 0,
163 "map"
164 );
165 get_entities().add_entity(entity);
166
167 return entity;
168}
