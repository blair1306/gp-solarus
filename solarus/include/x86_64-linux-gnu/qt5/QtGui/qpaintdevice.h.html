42#ifndef QPAINTDEVICE_H
43#define QPAINTDEVICE_H
44
45#include <QtGui/qwindowdefs.h>
46#include <QtCore/qrect.h>
47
48QT_BEGIN_NAMESPACE
49
50
51
52class QPaintEngine;
53class QPaintDevicePrivate;
54
55class Q_GUI_EXPORT QPaintDevice // device for QPainter
56{
57public:
58 enum PaintDeviceMetric {
59 PdmWidth = 1,
60 PdmHeight,
61 PdmWidthMM,
62 PdmHeightMM,
63 PdmNumColors,
64 PdmDepth,
65 PdmDpiX,
66 PdmDpiY,
67 PdmPhysicalDpiX,
68 PdmPhysicalDpiY,
69 PdmDevicePixelRatio
70 };
71
72 virtual ~QPaintDevice();
73
74 virtual int devType() const;
75 bool paintingActive() const;
76 virtual QPaintEngine *paintEngine() const = 0;
77
78 int width() const { return metric(PdmWidth); }
79 int height() const { return metric(PdmHeight); }
80 int widthMM() const { return metric(PdmWidthMM); }
81 int heightMM() const { return metric(PdmHeightMM); }
82 int logicalDpiX() const { return metric(PdmDpiX); }
83 int logicalDpiY() const { return metric(PdmDpiY); }
84 int physicalDpiX() const { return metric(PdmPhysicalDpiX); }
85 int physicalDpiY() const { return metric(PdmPhysicalDpiY); }
86 int devicePixelRatio() const { return metric(PdmDevicePixelRatio); }
87 int colorCount() const { return metric(PdmNumColors); }
88 int depth() const { return metric(PdmDepth); }
89
90protected:
91 QPaintDevice();
92 virtual int metric(PaintDeviceMetric metric) const;
93 virtual void initPainter(QPainter *painter) const;
94 virtual QPaintDevice *redirected(QPoint *offset) const;
95 virtual QPainter *sharedPainter() const;
96
97 ushort painters; // refcount
98
99private:
100 Q_DISABLE_COPY(QPaintDevice)
101
102 QPaintDevicePrivate *reserved;
103
104 friend class QPainter;
105 friend class QPainterPrivate;
106 friend class QFontEngineMac;
107 friend class QX11PaintEngine;
108 friend Q_GUI_EXPORT int qt_paint_device_metric(const QPaintDevice *device, PaintDeviceMetric metric);
109};
110
111/*****************************************************************************
112 Inline functions
113 *****************************************************************************/
114
115inline int QPaintDevice::devType() const
116{ return QInternal::UnknownDevice; }
117
118inline bool QPaintDevice::paintingActive() const
119{ return painters != 0; }
120
121QT_END_NAMESPACE
122
123#endif // QPAINTDEVICE_H
