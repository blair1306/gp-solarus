77/**
78 * \brief Implementation of the callback used to return solid ground coordinates.
79 * \param l A Lua state.
80 * \return Number of values to return to Lua.
81 */
82int l_solid_ground_callback(lua_State* l) {
83
84 int x = LuaTools::check_int(l, lua_upvalueindex(1));
85 int y = LuaTools::check_int(l, lua_upvalueindex(2));
86 int layer = LuaTools::check_int(l, lua_upvalueindex(3));
87
88 lua_pushinteger(l, x);
89 lua_pushinteger(l, y);
90 lua_pushinteger(l, layer);
91 return 3;
92}

159 if (!suspended) {

203 // is not touching ground.
204 }
205 else {
206 on_raised_blocks = get_entities().overlaps_raised_blocks(get_layer(), get_bounding_box());

208
209 if (get_movement() != nullptr) {
210 get_movement()->update();

212 // TODO clear_old_movements() is missing
213
214 if (has_stream_action()) {
215 get_stream_action()->update();
216 if (!get_stream_action()->is_active()) {
217 stop_stream_action();

222/**
223 * \brief Updates the effects (if any) of the ground below the hero.
224 *
225 * This function is called repeatedly.
226 */
227void Hero::update_ground_effects() {
228
229 // see if it's time to do something (depending on the ground)
230 uint32_t now = System::now();
231 if (now >= next_ground_date) {
232
233 if (is_ground_visible() && get_movement() != nullptr) {
234
235 // a special ground is displayed under the hero and it's time to play a sound
236 const std::shared_ptr<StraightMovement> movement =
237 std::dynamic_pointer_cast<StraightMovement>(get_movement());
238 if (movement != nullptr) {
239 // TODO replace the dynamic_pointer_cast by a virtual method get_speed() in Movement.
240 double speed = movement->get_speed();
241 next_ground_date = now + std::max(150, (int) (20000 / speed));
242 if (sprites->is_walking() && get_state().is_touching_ground()) {
243 sprites->play_ground_sound();
244 }
245 }
246 }
247
248 else {
249
250 Ground ground = get_ground_below();
251 if (ground == Ground::HOLE && !get_state().can_avoid_hole()) {
252 // the hero is being attracted by a hole and it's time to move one more pixel into the hole
253
254 next_ground_date = now + 60;
255
256 if (get_distance(last_solid_ground_coords) >= 8) {
257 // too far from the solid ground: make the hero fall
258 set_walking_speed(normal_walking_speed);
259 set_state(new FallingState(*this));
260 }
261 else {
262 // not too far yet
263 apply_additional_ground_movement();
264 }
265 }
266 else if (ground == Ground::ICE) {
267
268 // Slide on ice.
269 if (!get_state().can_avoid_ice()) {
270 apply_additional_ground_movement();
271 }
272
273 next_ground_date = now + 30;
274
275 if (now >= next_ice_date) {
276 // Time to update the additional movement.
277 update_ice();
278 ice_movement_direction8 = get_wanted_movement_direction8();
279 }
280 }
281 }
282 }

285/**
286 * \brief Updates the additional movement applied when the hero is on ice ground.
287 */
288void Hero::update_ice() {
289
290 uint32_t now = System::now();
291 int wanted_movement_direction8 = get_wanted_movement_direction8();
292 if (wanted_movement_direction8 == -1) {
293 // The player wants to stop.
294 if (ice_movement_direction8 == -1) {
295 // And he does for a while so stop.
296 ground_dxy = { 0, 0 };
297 next_ice_date = now + 300;
298 }
299 else {
300 // But he was just moving on ice: continue the ice movement.
301 ground_dxy = direction_to_xy_move(ice_movement_direction8) * 2;
302 next_ice_date = now + 300;
303 }
304 }
305 else {
306 // The player wants to move.
307 if (ice_movement_direction8 == -1) {
308 // But he was not just moving on ice: resist to the wanted movement.
309 ground_dxy = direction_to_xy_move((wanted_movement_direction8 + 4) % 8);
310 next_ice_date = now + 300;
311 }
312 else if (ice_movement_direction8 != wanted_movement_direction8) {
313 // He changed his direction: continue the ice movement strongly.
314 ground_dxy = direction_to_xy_move(ice_movement_direction8) * 2;
315 next_ice_date = now + 300;
316 }
317 else {
318 // He continues in the same direction.
319 ground_dxy = direction_to_xy_move(wanted_movement_direction8);
320 next_ice_date = now + 300;
321 }
322 }

334/**
335 * \brief Changes the position of the hero as an effect of his current ground
336 * (like hole or ice).
337 */
338void Hero::apply_additional_ground_movement() {
339
340 if (ground_dxy.x == 0 && ground_dxy.y == 0) {
341 return;
342 }
343
344 bool moved = false;
345 Rectangle collision_box = get_bounding_box();
346 collision_box.add_xy(ground_dxy);
347
348 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
349 set_bounding_box(collision_box);
350 notify_position_changed();
351 moved = true;
352 }
353
354 if (!moved && ground_dxy.x != 0) { // try x only
355 collision_box = get_bounding_box();
356 collision_box.add_xy(ground_dxy.x, 0);
357 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
358 set_bounding_box(collision_box);
359 notify_position_changed();
360 moved = true;
361 }
362 }
363
364 if (!moved && ground_dxy.y != 0) { // try y only
365 collision_box = get_bounding_box();
366 collision_box.add_xy(0, ground_dxy.y);
367 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
368 set_bounding_box(collision_box);
369 notify_position_changed();
370 moved = true;
371 }
372 }
373
374 if (!moved) {
375 if (get_ground_below() == Ground::HOLE) {
376 // the hero cannot be moved towards the direction previously calculated
377 set_walking_speed(normal_walking_speed);
378 set_state(new FallingState(*this));
379 }
380 }

534/**
535 * \brief Puts the hero on a map.
536 *
537 * This function is called when the current map is changed.
538 * Does nothing if the hero is already on this map.
539 *
540 * \param map The map.
541 */
542void Hero::place_on_map(Map& map) {
543
544 if (is_on_map() &&
545 &get_map() == &map
546 ) {
547 // No change.
548 return;
549 }
550
551 // Add the hero to the map.
552 const HeroPtr& shared_hero = std::static_pointer_cast<Hero>(shared_from_this());
553 map.get_entities().add_entity(shared_hero);
554
555 last_solid_ground_coords = { -1, -1 };
556 last_solid_ground_layer = 0;
557 reset_target_solid_ground_callback();
558 get_hero_sprites().set_clipping_rectangle();
559
560 get_state().set_map(map);
561
562 Entity::set_map(map);
563}

565/**
566 * \brief Places the hero on the map specified and at its destination point selected.
567 * \param map The new map.
568 * \param previous_map_location Position of the previous map in its world
569 * (because the previous map is already destroyed).
570 */
571void Hero::place_on_destination(Map& map, const Rectangle& previous_map_location) {
572
573 const std::string& destination_name = map.get_destination_name();
574
575 if (destination_name == "_same") {
576
577 // The hero's coordinates are the same as on the previous map
578 // but 1) we need to take into account the location of both maps
579 // and 2) we may have to change the layer.
580
581 const Rectangle& next_map_location = map.get_location();
582 int x = get_x() - next_map_location.get_x() + previous_map_location.get_x();
583 int y = get_y() - next_map_location.get_y() + previous_map_location.get_y();
584
585 int layer = map.get_max_layer();
586 while (layer > map.get_min_layer() &&
587 map.get_ground(layer, x, y, this) == Ground::EMPTY) {
588 // TODO check the whole hero's bounding box rather than just a point.
589 --layer;
590 }
591 place_on_map(map);
592 set_xy(x, y);
593 map.get_entities().notify_entity_bounding_box_changed(*this);
594 map.get_entities().set_entity_layer(*this, layer);
595 last_solid_ground_coords = { x, y };
596 last_solid_ground_layer = get_layer();
597
598 start_free();
599 check_position(); // To appear initially swimming, for example.
600 }
601 else {
602 int side = map.get_destination_side();
603
604 if (side != -1) {
605
606 // Go to a side of the other map.
607 place_on_map(map);
608
609 switch (side) {
610
611 case 0: // Right side.
612 set_x(map.get_width());
613 set_y(get_y() - map.get_location().get_y() + previous_map_location.get_y());
614 break;
615
616 case 1: // Top side.
617 set_y(5);
618 set_x(get_x() - map.get_location().get_x() + previous_map_location.get_x());
619 break;
620
621 case 2: // Left side.
622 set_x(0);
623 set_y(get_y() - map.get_location().get_y() + previous_map_location.get_y());
624 break;
625
626 case 3: // Bottom side.
627 set_y(map.get_height() + 5);
628 set_x(get_x() - map.get_location().get_x() + previous_map_location.get_x());
629 break;
630
631 default:
632 Debug::die("Invalid destination side");
633 }
634 map.get_entities().notify_entity_bounding_box_changed(*this);
635 last_solid_ground_coords = get_xy();
636 last_solid_ground_layer = get_layer();
637 // Note that we keep the hero's state from the previous map.
638 }
639 else {
640
641 // Normal case: the location is specified by a destination point object.
642
643 Destination* destination = map.get_destination();
644
645 if (destination == nullptr) {
646 // This is embarrassing: there is no valid destination that we can use.
647 // The map is probably in an early development phase.
648 // For now, let's place the hero at the top-left corner of the map.
649 Debug::error(
650 std::string("No valid destination on map '") + map.get_id()
651 + "'. Placing the hero at (0,0) instead."
652 );
653 place_on_map(map);
654 sprites->set_animation_direction(3);
655 set_top_left_xy(0, 0);
656 map.get_entities().notify_entity_bounding_box_changed(*this);
657 map.get_entities().set_entity_layer(*this, map.get_max_layer());
658 }
659 else {
660 // Normal case.
661 place_on_map(map);
662 if (destination->get_direction() != -1) {
663 sprites->set_animation_direction(destination->get_direction());
664 }
665 set_xy(destination->get_xy());
666 map.get_entities().notify_entity_bounding_box_changed(*this);
667 map.get_entities().set_entity_layer(*this, destination->get_layer());
668 }
669 last_solid_ground_coords = get_xy();
670 last_solid_ground_layer = get_layer();
671
672 // Remove boomerangs in case the map remains the same.
673 const std::set<std::shared_ptr<Boomerang>>& boomerangs =
674 map.get_entities().get_entities_by_type<Boomerang>();
675 for (const std::shared_ptr<Boomerang>& boomerang : boomerangs) {
676 boomerang->remove_from_map();
677 }
678
679 if (destination != nullptr) {
680 get_lua_context()->destination_on_activated(*destination);
681 }
682
683 const std::shared_ptr<const Stairs> stairs = get_stairs_overlapping();
684 if (stairs != nullptr) {
685 // The hero arrived on the map by stairs.
686 set_state(new StairsState(*this, stairs, Stairs::REVERSE_WAY));
687 }
688 else {
689 // The hero arrived on the map by a usual destination point.
690 start_free();
691 check_position(); // To appear initially swimming, for example.
692 }
693 }
694 }

697/**
698 * \brief This function is called when the opening transition of the map is finished.
699 *
700 * The position of the hero is changed if necessary.
701 */
702void Hero::notify_map_opening_transition_finished() {
703
704 Entity::notify_map_opening_transition_finished();
705
706 int side = get_map().get_destination_side();
707 if (side != -1) {
708 // the hero was placed on the side of the map:
709 // there was a scrolling between the previous map and this one
710
711 switch (side) {
712
713 case 0: // right side
714 set_x(get_map().get_width() - 8);
715 break;
716
717 case 1: // top side
718 set_y(13);
719 break;
720
721 case 2: // left side
722 set_x(8);
723 break;
724
725 case 3: // bottom side
726 set_y(get_map().get_height() - 3);
727 break;
728
729 default:
730 Debug::die("Invalid destination side");
731 }
732 }
733 check_position();
734 if (get_state().is_touching_ground()) { // Don't change the state during stairs.
735 start_state_from_ground();
736 }
737}

931/**
932 * \brief Returns the actual direction of the hero's movement.
933 *
934 * This function returns the actual direction of the hero's movement, which can be different from the one
935 * defined by the directional keys pressed by the player because we consider obstacles here.
936 * If he does not want to move, -1 is returned. If he is trying to move but cannot because of obstacles,
937 * the direction he is trying to move toward is returned.
938 * This function is not used to compute the hero's movement (PlayerMovement does that) but only
939 * to decide what direction to give to its sprites once the movement is already computed.
940 *
941 * \return the hero's actual direction between 0 and 7, or -1 if he is stopped
942 */
943int Hero::get_real_movement_direction8() {
944
945 int result;
946
947 int wanted_direction8 = get_wanted_movement_direction8();
948 if (wanted_direction8 == -1) {
949 // the hero does not want to move
950 result = -1;
951 }
952 else {
953 // the hero wants to move
954
955 Rectangle collision_box(get_bounding_box());
956
957 // if we can move towards the wanted direction, no problem
958 Point xy_move = direction_to_xy_move(wanted_direction8);
959 collision_box.add_xy(xy_move);
960 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
961 result = wanted_direction8;
962 }
963 else {
964 // otherwise, see if he can move in one of the two closest directions (i.e. he is sliding)
965
966 int alternative_direction8 = (wanted_direction8 + 1) % 8;
967 collision_box = get_bounding_box();
968 xy_move = direction_to_xy_move(alternative_direction8);
969 collision_box.add_xy(xy_move);
970 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
971 result = alternative_direction8;
972 }
973 else {
974 alternative_direction8 = (wanted_direction8 - 1) % 8;
975 collision_box = get_bounding_box();
976 xy_move = direction_to_xy_move(alternative_direction8);
977 collision_box.add_xy(xy_move);
978 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
979 result = alternative_direction8;
980 }
981 else {
982 // he is not sliding, he wants to move but can't
983 result = wanted_direction8;
984 }
985 }
986 }
987 }
988
989 return result;
990}

1074/**
1075 * \brief Checks collisions with detectors, determines the facing entity
1076 * and the ground below the hero in its current position.
1077 *
1078 * This function is called when these checks have to be done again,
1079 * e.g. when the position, the direction or the state of the hero changes.
1080 */
1081void Hero::check_position() {
1082
1083 if (!is_on_map()) {
1084 return;
1085 }
1086
1087 if (get_state().are_collisions_ignored()) {
1088 // Do not take care of the ground or detectors.
1089 return;
1090 }
1091
1092 // Recompute the facing entity.
1093 set_facing_entity(nullptr);
1094 check_collision_with_detectors();
1095
1096 if (is_suspended()
1097 && get_map().test_collision_with_border(get_ground_point())) {
1098 // When suspended, the hero may have invalid coordinates
1099 // (e.g. transition between maps).
1100 return;
1101 }
1102
1103 // Determine the new ground if it has changed.
1104 update_ground_below();
1105
1106 // Save the hero's last valid position.
1107 Ground ground = get_ground_below();
1108 if (ground != Ground::DEEP_WATER
1109 && ground != Ground::HOLE
1110 && ground != Ground::LAVA
1111 && ground != Ground::PRICKLE
1112 && ground != Ground::EMPTY
1113 && get_state().can_come_from_bad_ground()
1114 && (get_xy() != last_solid_ground_coords)) {
1115
1116 last_solid_ground_coords = get_xy();
1117 last_solid_ground_layer = get_layer();
1118 }
1119
1120 // With empty ground, possibly go to the lower layer.
1121 if (ground == Ground::EMPTY && get_state().is_touching_ground()) {
1122
1123 int x = get_top_left_x();
1124 int y = get_top_left_y();
1125 int layer = get_layer();
1126
1127 if (layer > get_map().get_min_layer()
1128 && get_map().get_ground(layer, x, y, this) == Ground::EMPTY
1129 && get_map().get_ground(layer, x + 15, y, this) == Ground::EMPTY
1130 && get_map().get_ground(layer, x, y + 15, this) == Ground::EMPTY
1131 && get_map().get_ground(layer, x + 15, y + 15, this) == Ground::EMPTY) {
1132
1133 get_entities().set_entity_layer(*this, layer - 1);
1134 Ground new_ground = get_map().get_ground(get_layer(), x, y, this);
1135 if (get_state().is_free() &&
1136 (new_ground == Ground::TRAVERSABLE
1137 || new_ground == Ground::GRASS
1138 || new_ground == Ground::LADDER)) {
1139 Sound::play("hero_lands");
1140 }
1141 }
1142 }
1143}

1204/**
1205 * \brief Starts activating the new ground of the hero.
1206 */
1207void Hero::notify_ground_below_changed() {
1208
1209 const bool suspended = get_game().is_suspended();
1210
1211 Entity::notify_ground_below_changed();
1212
1213 switch (get_ground_below()) {
1214
1215 case Ground::TRAVERSABLE:
1216 // Traversable ground: remove any special sprite displayed under the hero.
1217 sprites->destroy_ground();
1218 set_walking_speed(normal_walking_speed);
1219 break;
1220
1221 case Ground::DEEP_WATER:
1222 // Deep water: plunge if the hero is not jumping.
1223 if (!get_state().can_avoid_deep_water()) {
1224
1225 if (suspended) {
1226 // During a transition, it is okay to start swimming
1227 // but we don't want to start plunging right now.
1228 if (get_state().is_touching_ground()) {
1229 start_deep_water();
1230 }
1231 }
1232 else {
1233 start_deep_water();
1234 }
1235 }
1236 break;
1237
1238 case Ground::HOLE:
1239 // Hole: fall into the hole or get attracted to it.
1240 // But wait for the teletransporter opening transition to finish if any.
1241 if (!suspended
1242 && !get_state().can_avoid_hole()) {
1243 start_hole();
1244 }
1245 break;
1246
1247 case Ground::ICE:
1248 // Ice: make the hero slide.
1249 if (!get_state().can_avoid_ice()) {
1250 start_ice();
1251 }
1252 break;
1253
1254 case Ground::LAVA:
1255 // Lava: plunge into lava.
1256 if (!suspended
1257 && !get_state().can_avoid_lava()) {
1258 start_lava();
1259 }
1260 break;
1261
1262 case Ground::PRICKLE:
1263 // Prickles.
1264 if (!suspended
1265 && !get_state().can_avoid_prickle()) {
1266 start_prickle(500);
1267 }
1268 break;
1269
1270 case Ground::SHALLOW_WATER:
1271 start_shallow_water();
1272 break;
1273
1274 case Ground::GRASS:
1275 start_grass();
1276 break;
1277
1278 case Ground::LADDER:
1279 set_walking_speed(normal_walking_speed * 3 / 5);
1280 break;
1281
1282 case Ground::WALL:
1283 case Ground::LOW_WALL:
1284 case Ground::WALL_TOP_RIGHT:
1285 case Ground::WALL_TOP_LEFT:
1286 case Ground::WALL_BOTTOM_LEFT:
1287 case Ground::WALL_BOTTOM_RIGHT:
1288 case Ground::WALL_TOP_RIGHT_WATER:
1289 case Ground::WALL_TOP_LEFT_WATER:
1290 case Ground::WALL_BOTTOM_LEFT_WATER:
1291 case Ground::WALL_BOTTOM_RIGHT_WATER:
1292 // The hero is stuck in a wall. Damn.
1293 // This is the fault of the quest maker, unless there is a bug in Solarus.
1294 // The user will have to save and quit his game.
1295 // TODO maybe we could use the back to solid ground mechanism here?
1296 break;
1297
1298 case Ground::EMPTY:
1299 break;
1300 }
1301
1302 // Notify the state.
1303 get_state().notify_ground_changed();
1304}

1617/**
1618 * \copydoc Entity::notify_collision_with_stream
1619 */
1620void Hero::notify_collision_with_stream(
1621 Stream& stream, int dx, int dy) {
1622
1623 if (has_stream_action()) {
1624 // There is already an active stream.
1625 return;
1626 }
1627
1628 if (get_state().can_avoid_stream(stream)) {
1629 // Streams are ignored in the current state of the hero.
1630 return;
1631 }
1632
1633 // Check that the feet of the hero are on the stream.
1634 if (!stream.overlaps(get_ground_point())) {
1635 return;
1636 }
1637
1638 bool activate_stream = false;
1639
1640 if (stream.get_allow_movement()) {
1641 // The stream can be forced: there is no risk.
1642 // Even if the stream is faster than the player's movement,
1643 // he can still move perpendicularly.
1644 activate_stream = true;
1645 }
1646 else if (dx != 0 && dy != 0) {
1647 // Case of a diagonal stream.
1648 // TODO Diagonal streams are always activated for now, this could be improved.
1649 activate_stream = true;
1650 }
1651 else {
1652 // This is a blocking stream.
1653 // Check that the hero can go in the stream's direction.
1654 // Otherwise the hero, would be trapped forever if there
1655 // is an obstacle after the stream.
1656 Map& map = get_map();
1657 Rectangle collision_box(0, 0, 16, 16);
1658
1659 // First check that the exit of the stream is clear.
1660 collision_box.set_xy(stream.get_bounding_box().get_xy());
1661 collision_box.add_xy(dx, dy);
1662 if (!map.test_collision_with_obstacles(get_layer(), collision_box, *this)) {
1663 // The stream's exit is clear.
1664 // Check that we can make at least one pixel with the stream.
1665 // Because maybe we have already passed the exit and cannot do more.
1666 const bool horizontal = dx != 0;
1667 if (horizontal) {
1668 collision_box.set_xy(
1669 get_top_left_x() + dx,
1670 get_top_left_y()
1671 );
1672 }
1673 else {
1674 collision_box.set_xy(
1675 get_top_left_x(),
1676 get_top_left_y() + dy
1677 );
1678 }
1679
1680 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, *this)) {
1681 // We can make at least one pixel with the stream.
1682 activate_stream = true;
1683 }
1684 else {
1685 // We would be blocked if we took the stream straight.
1686 // Maybe we won't get blocked when getting aligned with the stream.
1687 // Let's check that.
1688
1689 if (horizontal) {
1690 // Horizontal stream: try to align the Y coordinate.
1691 collision_box.set_xy(
1692 get_top_left_x() + dx,
1693 stream.get_top_left_y()
1694 );
1695 }
1696 else {
1697 // Vertical stream: try to align the X coordinate.
1698 collision_box.set_xy(
1699 stream.get_top_left_x(),
1700 get_top_left_y() + dy
1701 );
1702 }
1703
1704 if (!map.test_collision_with_obstacles(get_layer(), collision_box, *this)) {
1705 // We can move into the direction when correctly aligned.
1706 activate_stream = true;
1707 }
1708 }
1709 }
1710 }
1711
1712 if (activate_stream) {
1713 stream.activate(*this);
1714 if (!get_state().can_persist_on_stream(stream)) {
1715 start_free();
1716 }
1717 }
1718}

1946 set_top_left_x(entity.get_center_point().x - 8);
