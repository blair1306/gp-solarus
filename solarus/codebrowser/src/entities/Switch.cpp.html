37/**
38 * \brief Constructor.
39 * \param name Name of the entity.
40 * \param layer Layer of the entity.
41 * \param xy Coordinates the entity.
42 * \param subtype The subtype of switch.
43 * \param sprite_name Sprite animation set id to use, or an empty string.
44 * \param sound_id Sound to play when activating the switch,
45 * or an empty string.
46 * \param needs_block \c true if a block is required to activate this switch.
47 * \param inactivate_when_leaving \c true to inactivate the switch when the
48 * hero or the block leaves it.
49 */
50Switch::Switch(
51 const std::string& name,
52 int layer,
53 const Point& xy,
54 Subtype subtype,
55 const std::string& sprite_name,
56 const std::string& sound_id,
57 bool needs_block,
58 bool inactivate_when_leaving):
59 Entity(name, 0, layer, xy, Size(16, 16)),
60 subtype(subtype),
61 sound_id(sound_id),
62 activated(false),
63 locked(false),
64 needs_block(needs_block),
65 inactivate_when_leaving(inactivate_when_leaving),
66 entity_overlapping(nullptr),
67 entity_overlapping_still_present(false) {
68
69 // Sprite.
70 if (!sprite_name.empty()) {
71 const SpritePtr& sprite = create_sprite(sprite_name);
72 sprite->set_current_animation("inactivated");
73 }
74
75 // Collisions.
76 if (is_walkable()) {
77 set_collision_modes(CollisionMode::COLLISION_CUSTOM);
78 }
79 else if (subtype == Subtype::ARROW_TARGET) {
80 set_collision_modes(CollisionMode::COLLISION_FACING);
81 }
82 else if (subtype == Subtype::SOLID) {
83 set_collision_modes(CollisionMode::COLLISION_SPRITE | CollisionMode::COLLISION_OVERLAPPING);
84 }
85}

138/**
139 * \brief Activates the switch, playing a sound and notifying the map script.
140 *
141 * This function does nothing if the switch is locked or already activated.
142 */
143void Switch::activate() {
144
145 if (!activated && !locked) {
146
147 set_activated(true);
148
149 if (!sound_id.empty()) {
150 Sound::play(sound_id);
151 }
152
153 get_lua_context()->switch_on_activated(*this);
154 }
155}
156
157/**
158 * \brief Activates or inactivates the switch, not playing any sound.
159 *
160 * This function can change the switch state even if the switch is locked.
161 *
162 * \param activated true to make the switch on, false to make it off
163 */
164void Switch::set_activated(bool activated) {
165
166 if (activated != this->activated) {
167 this->activated = activated;
168
169 const SpritePtr& sprite = get_sprite();
170 if (sprite != nullptr) {
171 if (activated) {
172 sprite->set_current_animation("activated");
173 }
174 else {
175 sprite->set_current_animation("inactivated");
176 }
177 }
178 }
179}

210/**
211 * \brief Updates this switch.
212 */
213void Switch::update() {
214
215 Entity::update();
216
217 if (is_enabled() &&
218 is_walkable() &&
219 entity_overlapping != nullptr) {
220
221 // if an entity was on the switch, see if it is still there
222 entity_overlapping_still_present = false;
223 check_collision(*entity_overlapping);
224
225 if (!entity_overlapping_still_present) {
226 // the entity just left the switch or disappeared from the map
227 // (it may even have been freed)
228
229 entity_overlapping = nullptr;
230 if (is_activated() && inactivate_when_leaving && !locked) {
231 set_activated(false);
232 get_lua_context()->switch_on_inactivated(*this);
233 }
234 get_lua_context()->switch_on_left(*this);
235 }
236 }
237}

239/**
240 * \brief Tests whether an entity's collides with this entity.
241 * \param entity an entity
242 * \return true if the entity's collides with this entity
243 */
244bool Switch::test_collision_custom(Entity& entity) {
245
246 // This collision test is performed for walkable switches only.
247
248 const Rectangle& entity_rectangle = entity.get_bounding_box();
249 int x1 = entity_rectangle.get_x() + 4;
250 int x2 = x1 + entity_rectangle.get_width() - 9;
251 int y1 = entity_rectangle.get_y() + 4;
252 int y2 = y1 + entity_rectangle.get_height() - 9;
253
254 return overlaps(x1, y1) && overlaps(x2, y1) &&
255 overlaps(x1, y2) && overlaps(x2, y2);
256}

258/**
259 * \brief This function is called by the engine when an entity overlaps the switch.
260 * \param entity_overlapping the entity overlapping the detector
261 * \param collision_mode the collision mode that detected the collision
262 */
263void Switch::notify_collision(
264 Entity& entity_overlapping, CollisionMode collision_mode) {
265
266 if (&entity_overlapping == this->entity_overlapping) {
267 // already overlapping
268 entity_overlapping_still_present = true;
269 return;
270 }
271
272 if (!locked) {
273 entity_overlapping.notify_collision_with_switch(*this, collision_mode);
274 }
275}

277/**
278 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
279 */
280void Switch::notify_collision(
281 Entity& other_entity,
282 Sprite& /* this_sprite */,
283 Sprite& other_sprite
284) {
285 if (!locked) {
286 other_entity.notify_collision_with_switch(*this, other_sprite);
287 }

290/**
291 * \brief This function is called when the hero overlaps this switch.
292 *
293 * The switch is activated if its properties allow it.
294 *
295 * \param hero the hero
296 */
297void Switch::try_activate(Hero& hero) {
298
299 if (is_walkable() &&
300 !needs_block &&
301 !is_activated()) {
302 // this switch allows the hero to activate it
303 activate();
304 }
305 this->entity_overlapping = &hero;
306}
307
308/**
309 * \brief This function is called when a block overlaps this switch.
310 *
311 * The switch is activated if its properties allow it.
312 *
313 * \param block the block overlapping this switch
314 */
315void Switch::try_activate(Block& block) {
316
317 if (is_walkable() &&
318 !is_activated()) {
319 // blocks can only activate walkable, visible switches
320 activate();
321 }
322 this->entity_overlapping = &block;
323}
324
325/**
326 * \brief This function is called when an arrow overlaps this switch.
327 *
328 * The switch is activated if its properties allow it.
329 *
330 * \param arrow the arrow overlapping this switch
331 */
332void Switch::try_activate(Arrow& /* arrow */) {
333
334 if ((subtype == Subtype::ARROW_TARGET || subtype == Subtype::SOLID)
335 && !is_activated()) {
336 // arrows can only activate arrow targets and solid switches
337 activate();
338 }
339}
340
341/**
342 * \brief This function is called when any entity overlaps this switch.
343 *
344 * Only solid switches can be activated this way.
345 * The switch is activated if its properties allow it.
346 */
347void Switch::try_activate() {
348
349 // arbitrary entities can activate solid switches
350 if (subtype == Subtype::SOLID && !is_activated()) {
351 activate();
352 }
353}
