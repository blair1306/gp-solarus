17#include "solarus/lowlevel/Debug.h"
18#include "test_tools/TestEnvironment.h"
19
20using namespace Solarus;
21
22/**
23 * \brief Runs a test written in Lua.
24 *
25 * The Lua test is implemented as a map of the testing quest.
26 */
27int main(int argc, char** argv) {
28
29 TestEnvironment env(argc, argv);
30
31 const std::string& map_id = env.get_arguments().get_argument_value("-map");
32 Debug::check_assertion(!map_id.empty(), "No map specified");
33
34 env.run_map(map_id);
35
36 return 0;
37}
