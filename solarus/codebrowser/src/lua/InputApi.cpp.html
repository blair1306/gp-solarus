26 * Name of the Lua table representing the input module.
27 */
28const std::string LuaContext::input_module_name = "sol.input";

30/**
31 * \brief Initializes the input features provided to Lua.
32 */
33void LuaContext::register_input_module() {
34
35 static const luaL_Reg functions[] = {
36 { "is_joypad_enabled", input_api_is_joypad_enabled },
37 { "set_joypad_enabled", input_api_set_joypad_enabled },
38 { "is_key_pressed", input_api_is_key_pressed },
39 { "get_key_modifiers", input_api_get_key_modifiers },
40 { "is_joypad_button_pressed", input_api_is_joypad_button_pressed },
41 { "get_joypad_axis_state", input_api_get_joypad_axis_state },
42 { "get_joypad_hat_direction", input_api_get_joypad_hat_direction },
43 { "is_mouse_button_pressed", input_api_is_mouse_button_pressed },
44 { "get_mouse_position", input_api_get_mouse_position },
45 { "is_finger_pressed", input_api_is_finger_pressed },
46 { "is_finger_released", input_api_is_finger_released },
47 { "get_finger_position", input_api_get_finger_position },
48 { "get_finger_pressure", input_api_get_finger_pressure },
49 { "simulate_key_pressed", input_api_simulate_key_pressed },
50 { "simulate_key_released", input_api_simulate_key_released },
51 { nullptr, nullptr }
52 };
53
54 register_functions(input_module_name, functions);
55}

57/**
58 * \brief Implementation of sol.input.is_joypad_enabled().
59 * \param l The Lua context that is calling this function.
60 * \return Number of values to return to Lua.
61 */
62int LuaContext::input_api_is_joypad_enabled(lua_State* l) {
63
64 return LuaTools::exception_boundary_handle(l, [&] {
65 lua_pushboolean(l, InputEvent::is_joypad_enabled());
66 return 1;
67 });
68}

86/**
87 * \brief Implementation of sol.input.is_key_pressed().
88 * \param l The Lua context that is calling this function.
89 * \return Number of values to return to Lua.
90 */
91int LuaContext::input_api_is_key_pressed(lua_State* l) {
92
93 return LuaTools::exception_boundary_handle(l, [&] {
94 const std::string& key_name = LuaTools::check_string(l, 1);
95 InputEvent::KeyboardKey key = name_to_enum(key_name, InputEvent::KeyboardKey::NONE);
96
97 if (key == InputEvent::KeyboardKey::NONE) {
98 LuaTools::arg_error(l, 1, std::string(
99 "Unknown keyboard key name: '") + key_name + "'");
100 }
101
102 lua_pushboolean(l, InputEvent::is_key_down(key));
103 return 1;
104 });
105}

107/**
108 * \brief Implementation of sol.input.get_key_modifiers().
109 * \param l The Lua context that is calling this function.
110 * \return Number of values to return to Lua.
111 */
112int LuaContext::input_api_get_key_modifiers(lua_State* l) {
113
114 return LuaTools::exception_boundary_handle(l, [&] {
115 const bool shift = InputEvent::is_shift_down();
116 const bool control = InputEvent::is_control_down();
117 const bool alt = InputEvent::is_alt_down();
118 const bool caps_lock = InputEvent::is_caps_lock_on();
119 const bool num_lock = InputEvent::is_num_lock_on();
120
121 lua_newtable(l);
122 if (shift) {
123 lua_pushboolean(l, 1);
124 lua_setfield(l, -2, "shift");
125 }
126 if (control) {
127 lua_pushboolean(l, 1);
128 lua_setfield(l, -2, "control");
129 }
130 if (alt) {
131 lua_pushboolean(l, 1);
132 lua_setfield(l, -2, "alt");
133 }
134 if (caps_lock) {
135 lua_pushboolean(l, 1);
136 lua_setfield(l, -2, "caps lock");
137 }
138 if (num_lock) {
139 lua_pushboolean(l, 1);
140 lua_setfield(l, -2, "num lock");
141 }
142
143 return 1;
144 });
145}

234/**
235 * \brief Implementation of sol.input.get_mouse_position().
236 * \param l The Lua context that is calling this function.
237 * \return Number of values to return to Lua.
238 */
239int LuaContext::input_api_get_mouse_position(lua_State* l) {
240
241 return LuaTools::exception_boundary_handle(l, [&] {
242 Point mouse_xy = InputEvent::get_global_mouse_position();
243 const Size window_size = Video::get_window_size();
244
245 if (mouse_xy.x < 0
246 || mouse_xy.y < 0
247 || mouse_xy.x >= window_size.width
248 || mouse_xy.y >= window_size.height) {
249 lua_pushnil(l);
250 return 1;
251 }
252
253 lua_pushinteger(l, mouse_xy.x);
254 lua_pushinteger(l, mouse_xy.y);
255 return 2;
256 });
257}

293/**
294 * \brief Implementation of sol.input.get_finger_position().
295 * \param l The Lua context that is calling this function.
296 * \return Number of values to return to Lua.
297 */
298int LuaContext::input_api_get_finger_position(lua_State* l) {
299
300 return LuaTools::exception_boundary_handle(l, [&] {
301 int finger_id = LuaTools::check_int(l, 1);
302 Point finger_xy;
303
304 if (!InputEvent::get_global_finger_position(finger_id, finger_xy)) {
305 lua_pushnil(l);
306 return 1;
307 }
308
309 lua_pushinteger(l, finger_xy.x);
310 lua_pushinteger(l, finger_xy.y);
311 return 2;
312 });
313}
