32/**
33 * @brief Export GUI settings to the Solarus settings.dat file of a quest.
34 *
35 * System settings like audio volume and the video mode are written to
36 * settings.dat.
37 * It allows quests that use settings.dat to apply what the user chose
38 * in the GUI.
39 *
40 * @param quest_path Path of the quest where to write a settings file.
41 */
42void Settings::export_to_quest(const QString& quest_path) const {
43
44 if (Solarus::QuestFiles::is_open()) {
45 Solarus::QuestFiles::close_quest();
46 }
47
48 QStringList arguments = QApplication::arguments();
49 const QString& program_name = arguments.isEmpty() ? QString() : arguments.first();
50 if (!Solarus::QuestFiles::open_quest(program_name.toStdString(), quest_path.toStdString())) {
51 return;
52 }
53
54 const std::string file_name = "settings.dat";
55 Solarus::Settings solarus_settings;
56 if (Solarus::QuestFiles::data_file_exists(file_name)) {
57 // First, load the existing settings.dat to preserve unmodified values.
58 solarus_settings.load(file_name);
59 }
60
61 QVariant video_mode = value("quest_video_mode");
62 if (video_mode.isValid()) {
63 solarus_settings.set_string(
64 Solarus::Settings::key_video_mode, video_mode.toString().toStdString());
65 }
66
67 QVariant fullscreen = value("quest_fullscreen");
68 if (fullscreen.isValid()) {
69 solarus_settings.set_boolean(
70 Solarus::Settings::key_fullscreen, fullscreen.toBool());
71 }
72
73 QVariant sound_volume = value("quest_sound_volume");
74 if (sound_volume.isValid()) {
75 solarus_settings.set_integer(
76 Solarus::Settings::key_sound_volume, sound_volume.toInt());
77 }
78
79 QVariant music_volume = value("quest_music_volume");
80 if (music_volume.isValid()) {
81 solarus_settings.set_integer(
82 Solarus::Settings::key_music_volume, music_volume.toInt());
83 }
84
85 QVariant language = value("quest_language");
86 if (language.isValid()) {
87 solarus_settings.set_string(
88 Solarus::Settings::key_language, language.toString().toStdString());
89 }
90
91 QVariant joypad_enabled = value("quest_joypad_enabled");
92 if (joypad_enabled.isValid()) {
93 solarus_settings.set_boolean(
94 Solarus::Settings::key_joypad_enabled, joypad_enabled.toBool());
95 }
96
97 solarus_settings.save(file_name);
98}
