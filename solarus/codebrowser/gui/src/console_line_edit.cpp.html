32constexpr int max_history_size = 100;
33
34/**
35 * @brief A QValidator that checks Lua syntax.
36 *
37 * Only the syntax of the code is checked (via lua_load()), not its execution.
38 */
39class LuaSyntaxValidator : public QValidator {
40
41public :
42
43 explicit LuaSyntaxValidator(QObject* parent = nullptr);
44
45 State validate(QString& input, int& pos) const override;
46
47private:
48
49 struct LuaStateDeleter {
50 void operator()(lua_State* l) {
51 lua_close(l);
52 }
53 };
54
55 std::unique_ptr<lua_State, LuaStateDeleter> l; /**< Lua context. */
56 mutable QHash<QString, State> result_cache; /**< Cache of previous results. */
57
58};
59

62/**
63 * @brief Creates a console line field.
64 * @param parent Parent object or nullptr.
65 */
66ConsoleLineEdit::ConsoleLineEdit(QWidget* parent) :
67 QLineEdit(parent),
68 history(),
69 history_position(0),
70 current_command() {
71
72 Settings settings;
73 history = settings.value("console_history").toStringList();
74
75 set_history_position(history.size()); // Start after the history.
76
77 // Set a validator.
78 setValidator(new LuaSyntaxValidator(this));
79 connect(this, &ConsoleLineEdit::textChanged, [this](const QString& text) {
80 QString text_copy = text;
81 int cursor_position = cursorPosition();
82 if (validator()->validate(text_copy, cursor_position) == QValidator::Acceptable) {
83 setStyleSheet("");
84 }
85 else {
86 setStyleSheet("background-color: #ffffc0");
87 }
88 });
89
90 // Set a completer.
91 QStringList unique_recent_commands = history;
92 unique_recent_commands.removeDuplicates();
93 std::reverse(unique_recent_commands.begin(), unique_recent_commands.end());
94 completer_model = new QStringListModel(unique_recent_commands, this);
95 QCompleter* completer = new QCompleter(completer_model, this);
96 completer->setCompletionMode(QCompleter::InlineCompletion);
97 setCompleter(completer);
98}

100/**
101 * @brief Reimplementation to navigate in the command history with up/down arrows.
102 * @param event The event to handle.
103 */
104void ConsoleLineEdit::keyPressEvent(QKeyEvent* event) {
105
106 switch (event->key()) {
107
108 case Qt::Key_Up:
109 history_previous();
110 event->accept();
111 return;
112
113 case Qt::Key_Down:
114 history_next();
115 event->accept();
116 return;
117
118 }
119
120 QLineEdit::keyPressEvent(event);
121}

123/**
124 * @brief Shows the command at the given history position.
125 * @param history_position The new history position to set.
126 */
127void ConsoleLineEdit::set_history_position(int history_position) {
128
129 Q_ASSERT(history_position >= 0 && history_position <= history.size());
130
131 this->history_position = history_position;
132
133 if (history_position == history.size()) {
134 // Typing a new command.
135 setText(current_command);
136 }
137 else {
138 // Browsing the history.
139 setText(history[history_position]);
140 }
141}
142
143/**
144 * @brief Sets the text from the previous command in the history.
145 *
146 * Does nothing if there is no such command.
147 */
148void ConsoleLineEdit::history_previous() {
149
150 if (history_position <= 0) {
151 return;
152 }
153
154 if (history_position == history.size()) {
155 // Remember the new command being typed.
156 current_command = text();
157 }
158
159 set_history_position(history_position - 1);
160}
161
162/**
163 * @brief Sets the text from the next command in the history.
164 *
165 * Does nothing if there is no such command.
166 */
167void ConsoleLineEdit::history_next() {
168
169 if (history_position >= history.size()) {
170 return;
171 }
172
173 set_history_position(history_position + 1);
174}

176/**
177 * @brief Adds a command to the history.
178 * @param command The command just executed.
179 */
180void ConsoleLineEdit::command_executed(const QString& command) {
181
182 // Avoid consecutive duplicates.
183 if (history.size() == 0 || command != history.last()) {
184 history.append(command);
185
186 // Limit the history size.
187 while (history.size() > max_history_size) {
188 history.removeFirst();
189 }
190
191 // Save the modified history.
192 Settings settings;
193 settings.setValue("console_history", history);
194 }
195
196 // Update the history position.
197 current_command.clear();
198 set_history_position(history.size());
199
200 // Update the completer.
201 if (!completer_model->stringList().contains(command)) {
202 completer_model->insertRow(0);
203 QModelIndex index = completer_model->index(0, 0);
204 completer_model->setData(index, command);
205 }
206}

218/**
219 * @brief Validates the Lua syntax of an input.
220 * @param input The input to check.
221 * @param pos The cursor position.
222 * @return The syntax validity: Acceptable or Intermediate.
223 */
224QValidator::State LuaSyntaxValidator::validate(QString& input, int& pos) const {
225
226 Q_UNUSED(pos);
227
228 const auto& it = result_cache.find(input);
229 if (it != result_cache.end()) {
230 // We already know the result.
231 return it.value();
232 }
233
234 // New input: ask Lua.
235 QByteArray input_utf8 = input.toUtf8();
236 const int load_result = luaL_loadstring(l.get(), input_utf8.data());
237
238 State result = (load_result == 0) ? State::Acceptable : State::Intermediate;
239 result_cache.insert(input, result);
240 return result;
241}
