29namespace Solarus {
30
31class TilePattern;
32class TilePatternData;
33
34/**
35 * \brief A set of tile patterns that are used to compose a map.
36 *
37 * A tileset represents the skin of a map.
38 */
39class Tileset {
40
41 public:
42
43 explicit Tileset(const std::string& id);
44
45 void load();
46 void unload();
47
48 const std::string& get_id() const;
49 const Color& get_background_color() const;
50 bool is_loaded() const;
51 const SurfacePtr& get_tiles_image() const;
52 const SurfacePtr& get_entities_image() const;
53 const TilePattern& get_tile_pattern(const std::string& id) const;
54 void set_images(const std::string& other_id);
55
56 private:
57
58 void add_tile_pattern(
59 const std::string& id,
60 const TilePatternData& pattern_data
61 );
62
63 const std::string id; /**< Id of the tileset. */
64 std::unordered_map<std::string, std::unique_ptr<TilePattern>>
65 tile_patterns; /**< Tile patterns in this tileset. */
66 Color background_color; /**< Background color of the tileset. */
67 SurfacePtr tiles_image; /**< Image from which the tile patterns are extracted. */
68 SurfacePtr entities_image; /**< Image from which the tileset-dependent sprites are
69 * extracted (optional, nullptr if none). */
70
71};
72
73}
