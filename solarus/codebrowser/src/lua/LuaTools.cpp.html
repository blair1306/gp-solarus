28/**
29 * \brief For an index in the Lua stack, returns an equivalent positive index.
30 *
31 * Pseudo-indexes are unchanged.
32 *
33 * \param l A Lua state.
34 * \param index An index in the stack.
35 * \return The corresponding positive index.
36 */
37int get_positive_index(lua_State* l, int index) {
38
39 int positive_index = index;
40 if (index < 0 && index >= -lua_gettop(l)) {
41 positive_index = lua_gettop(l) + index + 1;
42 }
43
44 return positive_index;
45}

47/**
48 * \brief Returns whether the specified name is a valid Lua identifier.
49 * \param name The name to check.
50 * \return true if the name only contains alphanumeric characters or '_' and
51 * does not start with a digit.
52 */
53bool is_valid_lua_identifier(const std::string& name) {
54
55 if (name.empty() || std::isdigit(name[0])) {
56 return false;
57 }
58
59 for (char character: name) {
60 if (!std::isalnum(character) && character != '_') {
61 return false;
62 }
63 }
64 return true;
65}

69 * this value.

79/**
80 * \brief Creates a reference to the Lua value at the specified index.
81 * \param l A Lua state.
82 * \param index An index in the Lua stack.
83 * \return The reference created, wrapped in an object that manages its
84 * lifetime.
85 */
86ScopedLuaRef create_ref(lua_State* l, int index) {
87
88 lua_pushvalue(l, index);
89 return ScopedLuaRef(l, luaL_ref(l, LUA_REGISTRYINDEX));

92/**
93 * \brief Calls the Lua function with its arguments on top of the stack.
94 *
95 * This function is like lua_pcall, except that it additionally handles the
96 * error message if an error occurs in the Lua code (the error is printed).
97 * This function leaves the results on the stack if there is no error,
98 * and leaves nothing on the stack in case of error.
99 *
100 * \param l A Lua state.
101 * \param nb_arguments Number of arguments placed on the Lua stack above the
102 * function to call.
103 * \param nb_results Number of results expected (you get them on the stack if
104 * there is no error).
105 * \param function_name A name describing the Lua function (only used to print
106 * the error message if any).
107 * This is not a <tt>const std::string&</tt> but a <tt>const char*</tt> on
108 * purpose to avoid costly conversions as this function is called very often.
109 * \return \c true in case of success.
110 */
111bool call_function(
112 lua_State* l,
113 int nb_arguments,
114 int nb_results,
115 const char* function_name
116) {
117 if (lua_pcall(l, nb_arguments, nb_results, 0) != 0) {
118 Debug::error(std::string("In ") + function_name + ": "
119 + lua_tostring(l, -1)
120 );
121 lua_pop(l, 1);
122 return false;
123 }
124
125 return true;
126}

128/**
129 * \brief Loads and executes some Lua code.
130 * \param l A Lua state.
131 * \param code The code to execute.
132 * \param chunk_name A name describing the Lua chunk
133 * (only used to print the error message if any).
134 * \return \c true in case of success.
135 */
136bool do_string(
137 lua_State* l,
138 const std::string& code,
139 const std::string& chunk_name
140) {
141 int load_result = luaL_loadstring(l, code.c_str());
142
143 if (load_result != 0) {
144 Debug::error(std::string("In ") + chunk_name + ": "
145 + lua_tostring(l, -1));
146 lua_pop(l, 1);
147 return false;
148 }
149
150 return call_function(l, 0, 0, chunk_name.c_str());
151}

165/**
166 * \brief Similar to luaL_argerror() but throws a LuaException.
167 *
168 * This function never returns.
169 *
170 * \param l A Lua state.
171 * \param arg_index Index of an argument in the stack.
172 * \param message Error message.
173 */
174void arg_error(lua_State* l, int arg_index, const std::string& message) {
175
176 // The code below is very similar to luaL_argerror(), but ends with
177 // an exception instead of a luaL_error() call.
178
179 std::ostringstream oss;
180 lua_Debug info;
181 if (!lua_getstack(l, 0, &info)) {
182 // No stack frame.
183 oss << "bad argument #" << arg_index << " (" << message << ")";
184 error(l, oss.str());
185 }
186
187 lua_getinfo(l, "n", &info);
188 if (std::string(info.namewhat) == "method") {
189 arg_index--; // Do not count self.
190 if (arg_index == 0) {
191 // Error is in the self argument itself.
192 oss << "calling " << info.name << " on bad self (" << message << ")";
193 error(l, oss.str());
194 }
195 }
196
197 if (info.name == nullptr) {
198 info.name = "?";
199 }
200 oss << "bad argument #" << arg_index << " to " << info.name << " (" << message << ")";
201 error(l, oss.str());
202}

253/**
254 * \brief Checks that a value is a number and returns it as an integer.
255 *
256 * This function acts like luaL_checkint() except that it throws a LuaException
257 * in case of error.
258 *
259 * \param l A Lua state.
260 * \param index Index of a table in the stack.
261 * \return The value as an integer.
262 */
263int check_int(
264 lua_State* l,
265 int index
266) {
267 if (!lua_isnumber(l, index)) {
268 arg_error(l, index,
269 std::string("integer expected, got ")
270 + luaL_typename(l, index) + ")"
271 );
272 }
273
274 return (int) lua_tointeger(l, index);
275}

277/**
278 * \brief Checks that a table field is a number and returns it as an integer.
279 *
280 * This function acts like lua_getfield() followed by LuaTools::check_int().
281 *
282 * \param l A Lua state.
283 * \param table_index Index of a table in the stack.
284 * \param key Key of the field to get in that table.
285 * \return The wanted field as an integer.
286 */
287int check_int_field(
288 lua_State* l,
289 int table_index,
290 const std::string& key
291) {
292 lua_getfield(l, table_index, key.c_str());
293 if (!lua_isnumber(l, -1)) {
294 arg_error(l, table_index,
295 std::string("Bad field '") + key + "' (integer expected, got "
296 + luaL_typename(l, -1) + ")"
297 );
298 }
299
300 int value = (int) lua_tointeger(l, -1);
301 lua_pop(l, 1);
302 return value;
303}

776 * \brief Checks that the value at the given index is a valid layer and returns it.
777 * \param l A Lua state.
778 * \param index An index in the Lua stack.
779 * \param map The map whose number of layer matters.
780 * \return The layer at this index.
781 */
782int check_layer(
783 lua_State* l,
784 int index,
785 const Map& map
786) {
787 if (!is_layer(l, index, map)) {
788 std::ostringstream oss;

790 oss << "Invalid layer";
791 }

884/**
885 * \brief Returns whether a value is a color.
886 * \param l A Lua context.
887 * \param index An index in the stack.
888 * \return true if the value is a color, that is, an array with three integers.
889 */
890bool is_color(lua_State* l, int index) {
891
892 index = get_positive_index(l, index);
893
894 if (lua_type(l, index) != LUA_TTABLE) {
895 return false;
896 }
897
898 lua_rawgeti(l, index, 1);
899 lua_rawgeti(l, index, 2);
900 lua_rawgeti(l, index, 3);
901 lua_rawgeti(l, index, 4);
902 bool result = lua_isnumber(l, -4) &&
903 lua_isnumber(l, -3) &&
904 lua_isnumber(l, -2) &&
905 (lua_isnumber(l, -1) || lua_isnil(l, -1));
906 lua_pop(l, 4);
907 return result;
908}

911 * \brief Checks that the value at the given index is a color and returns it.
912 *
913 * Set opaque by default if alpha channel is not specified.
914 *
915 * \param l a Lua state
916 * \param index an index in the Lua stack
917 * \return the color at this index
918 */
919Color check_color(lua_State* l, int index) {
920
921 index = get_positive_index(l, index);
922
923 check_type(l, index, LUA_TTABLE);
924 lua_rawgeti(l, index, 1);
925 lua_rawgeti(l, index, 2);
926 lua_rawgeti(l, index, 3);
927 lua_rawgeti(l, index, 4);
928 Color color(
929 check_int(l, -4),
930 check_int(l, -3),
931 check_int(l, -2),
932 opt_int(l, -1, 255)
933 );
934 lua_pop(l, 4);
935
936 return color;
937}
