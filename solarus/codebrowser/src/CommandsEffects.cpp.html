93/**
94 * \brief Returns whether the action key is enabled.
95 * \return true if the action key is enabled, false otherwise

97bool CommandsEffects::is_action_key_enabled() {
98 return action_key_enabled;
99}
100
101/**
102 * \brief Sets whether the action key is enabled.
103 * \param enable true to enable the action key, false to disable it
104 */
105void CommandsEffects::set_action_key_enabled(bool enable) {
106 this->action_key_enabled = enable;
107}
108
109/**
110 * \brief Saves the current effect of the action key.
111 *
112 * Call restore_action_key_effect() to restore the action key saved here.
113 */
114void CommandsEffects::save_action_key_effect() {
115 this->action_key_effect_saved = get_action_key_effect();
116}
117
118/**
119 * \brief Restores the action key effect saved by the last
120 * call to save_action_key_effect().
121 */
122void CommandsEffects::restore_action_key_effect() {
123 this->action_key_effect = action_key_effect_saved;
124}
125
126/**
127 * \brief Returns whether the action key is currently taking effect
128 * on the entity the hero is facing.
129 *
130 * Such action key effects are ACTION_KEY_LOOK, ACTION_KEY_OPEN,
131 * ACTION_KEY_LIFT, ACTION_KEY_SPEAK and ACTION_KEY_GRAB.
132 *
133 * \return true if the action key is acting on the facing entity
134 */
135bool CommandsEffects::is_action_key_acting_on_facing_entity() {
136 return action_key_effect == ACTION_KEY_LOOK
137 || action_key_effect == ACTION_KEY_OPEN
138 || action_key_effect == ACTION_KEY_LIFT
139 || action_key_effect == ACTION_KEY_SPEAK
140 || action_key_effect == ACTION_KEY_GRAB;
141}
