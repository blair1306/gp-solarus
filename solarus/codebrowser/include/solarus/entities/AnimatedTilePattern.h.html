17#ifndef SOLARUS_ANIMATED_TILE_PATTERN_H
18#define SOLARUS_ANIMATED_TILE_PATTERN_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/TilePattern.h"
22#include "solarus/lowlevel/Rectangle.h"
23
24namespace Solarus {
25
26/**
27 * \brief Animated tile pattern.
28 *
29 * Unlike simple tile patterns that always display the same image,
30 * an animated tile pattern displays three frames alternatively.
31 */
32class AnimatedTilePattern: public TilePattern {
33
34 public:
35
36 /**
37 * \brief Tile animation sequence mode: 0-1-2 or 0-1-2-1.
38 */
39 enum AnimationSequence {
40 ANIMATION_SEQUENCE_012 = 1,
41 ANIMATION_SEQUENCE_0121 = 2
42 };
43
44 AnimatedTilePattern(Ground ground, AnimationSequence sequence,
45 const Size& size, int x1, int y1, int x2, int y2, int x3, int y3,
46 bool parallax);
47
48 static void initialize();
49 static void update();
50 static void quit();
51
52 virtual void draw(
53 const SurfacePtr& dst_surface,
54 const Point& dst_position,
55 const Tileset& tileset,
56 const Point& viewport
57 ) const override;
58 virtual bool is_drawn_at_its_position() const override;
59
60 private:
61
62 // static variables to handle the animations of all tiles
63 static int frame_counter; /**< Frame counter (0 to 11), increased every 250 ms. */
64 static int current_frames[3]; /**< Current frame (0 to 2) for both sequences. */
65 static uint32_t next_frame_date; /**< Date of the next frame change. */
66
67 const AnimationSequence sequence; /**< Animation sequence type of this tile pattern: 0-1-2-1 or 0-1-2. */
68
69 Rectangle position_in_tileset[3]; /**< Array of 3 rectangles representing the 3 animation frames
70 * of this tile pattern in the tileset image.
71 * The 3 frames should have the same width and height. */
72 bool parallax; /**< Indicates that the tile pattern also makes parallax scrolling */
73
74};
75
76}
77
78#endif
