26bool FontResource::fonts_loaded = false;
27std::map<std::string, FontResource::FontFile> FontResource::fonts;

29/**
30 * \brief Initializes the font system.
31 */
32void FontResource::initialize() {
33
34 TTF_Init();
35}

38 * \brief Closes the font system.

47/**
48 * \brief Loads the fonts declared in the quest resource list.
49 */
50void FontResource::load_fonts() {
51
52 // Get the list of available fonts.
53 const std::map<std::string, std::string>& font_resource =
54 CurrentQuest::get_resources(ResourceType::FONT);
55
56 for (const auto& kvp: font_resource) {
57
58 const std::string& font_id = kvp.first;
59
60 FontFile font;
61
62 // Load the font.
63
64 bool bitmap_font = false;
65 const std::string file_name_start = std::string("fonts/") + font_id;
66 if (QuestFiles::data_file_exists(file_name_start + ".png")) {
67 font.file_name = file_name_start + ".png";
68 bitmap_font = true;
69 }
70 else if (QuestFiles::data_file_exists(file_name_start + ".PNG")) {
71 font.file_name = file_name_start + ".PNG";
72 bitmap_font = true;
73 }
74 else if (QuestFiles::data_file_exists(file_name_start + ".ttf")) {
75 font.file_name = file_name_start + ".ttf";
76 }
77 else if (QuestFiles::data_file_exists(file_name_start + ".TTF")) {
78 font.file_name = file_name_start + ".TTF";
79 }
80 else if (QuestFiles::data_file_exists(file_name_start + ".ttc")) {
81 font.file_name = file_name_start + ".ttc";
82 }
83 else if (QuestFiles::data_file_exists(file_name_start + ".TTC")) {
84 font.file_name = file_name_start + ".TTC";
85 }
86 else if (QuestFiles::data_file_exists(file_name_start + ".fon")) {
87 font.file_name = file_name_start + ".fon";
88 }
89 else if (QuestFiles::data_file_exists(file_name_start + ".FON")) {
90 font.file_name = file_name_start + ".FON";
91 }
92 else {
93 Debug::error(std::string("Cannot find font file 'fonts/")
94 + font_id + "' (tried with extensions .png, .ttf, .ttc and .fon)"
95 );
96 continue;
97 }
98
99 if (bitmap_font) {
100 // It's a bitmap font.
101 font.bitmap_font = Surface::create(font.file_name, Surface::DIR_DATA);
102 }
103
104 else {
105 // It's an outline font.
106 font.buffer = QuestFiles::data_file_read(font.file_name);
107 font.bitmap_font = nullptr;
108 }
109
110 fonts.emplace(font_id, std::move(font));
111 }
112
113 fonts_loaded = true;
114}

116/**
117 * \brief Returns the id of default font.
118 * \return Id of the first font in alphabetical order, or an empty string
119 * if there is no font at all.
120 */
121std::string FontResource::get_default_font_id() {
122
123 if (!fonts_loaded) {
124 load_fonts();
125 }
126
127 if (fonts.empty()) {
128 return "";
129 }
130
131 return fonts.begin()->first;
132}

134/**
135 * \brief Returns whether the specified font exists.
136 * \param font_id The id to test.
137 * \return \c true if there is a valid font with this id in the quest resource
138 * list.
139 */
140bool FontResource::exists(const std::string& font_id) {
141
142 if (!fonts_loaded) {
143 load_fonts();
144 }
145
146 return fonts.find(font_id) != fonts.end();
147}
148
149/**
150 * \brief Returns whether the specified font is a bitmap or an outline font.
151 * \param font_id Id of the font to test. It must exist.
152 * \return \c true if this is a bitmap font, \c false if this is an outline font.
153 */
154bool FontResource::is_bitmap_font(const std::string& font_id) {
155
156 if (!fonts_loaded) {
157 load_fonts();
158 }
159
160 const auto& kvp = fonts.find(font_id);
161 Debug::check_assertion(kvp != fonts.end(), std::string("No such font: '") + font_id + "'");
162 return kvp->second.bitmap_font != nullptr;
163}

165/**
166 * \brief Returns the surface image of a bitmap font.
167 * \param font_id Id of the bitmap font to get. It must exist.
168 * \return The bitmap.
169 */
170SurfacePtr FontResource::get_bitmap_font(const std::string& font_id) {
171
172 if (!fonts_loaded) {
173 load_fonts();
174 }
175
176 const auto& kvp = fonts.find(font_id);
177 Debug::check_assertion(kvp != fonts.end(), std::string("No such font: '") + font_id + "'");
178 Debug::check_assertion(kvp->second.bitmap_font != nullptr, std::string("This is not a bitmap font: '") + font_id + "'");
179 return kvp->second.bitmap_font;
180}

182/**
183 * \brief Returns an outline font with the specified size.
184 * \param font_id Id of the outline font to get. It must exist.
185 * \param size Size to use.
186 * \return The font.
187 */
188TTF_Font& FontResource::get_outline_font(const std::string& font_id, int size) {
189
190 if (!fonts_loaded) {
191 load_fonts();
192 }
193
194 const auto& kvp = fonts.find(font_id);
195 Debug::check_assertion(kvp != fonts.end(), std::string("No such font: '") + font_id + "'");
196 FontFile& font = kvp->second;
197 Debug::check_assertion(font.bitmap_font == nullptr, std::string("This is not an outline font: '") + font_id + "'");
198
199 std::map<int, OutlineFontReader>& outline_fonts = kvp->second.outline_fonts;
200
201 const auto& kvp2 = outline_fonts.find(size);
202 if (kvp2 != outline_fonts.end()) {
203 return *kvp2->second.outline_font;
204 }
205
206 // First time we want this font with this particular size.
207 SDL_RWops_UniquePtr rw = SDL_RWops_UniquePtr(SDL_RWFromMem(
208 const_cast<char*>(font.buffer.data()),
209 (int) font.buffer.size()
210 ));
211 TTF_Font_UniquePtr outline_font(TTF_OpenFontRW(rw.get(), 0, size));
212 Debug::check_assertion(outline_font != nullptr,
213 std::string("Cannot load font from file '") + font.file_name
214 + "': " + TTF_GetError()
215 );
216 OutlineFontReader reader = { std::move(rw), std::move(outline_font) };
217 outline_fonts.emplace(size, std::move(reader));
218 return *outline_fonts.at(size).outline_font;
219}
