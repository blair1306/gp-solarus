38/**
39 * \brief Creates a hookshot.
40 * \param hero the hero
41 */
42Hookshot::Hookshot(const Hero& hero):
43 Entity("", 0, hero.get_layer(), Point(0, 0), Size(0, 0)),
44 next_sound_date(System::now()),
45 has_to_go_back(false),
46 going_back(false),
47 entity_reached(nullptr),
48 link_sprite(std::make_shared<Sprite>("entities/hookshot")) {
49
50 // initialize the entity
51 int direction = hero.get_animation_direction();
52 const SpritePtr& sprite = create_sprite("entities/hookshot");
53 sprite->enable_pixel_collisions();
54 sprite->set_current_direction(direction);
55 link_sprite->set_current_animation("link");
56
57 set_size(16, 16);
58 set_origin(8, 13);
59 set_drawn_in_y_order(true);
60 set_xy(hero.get_xy());
61
62 std::string path = " ";
63 path[0] = '0' + (direction * 2);
64 set_movement(std::make_shared<PathMovement>(path, 192, true, false, false));
65}

165/**
166 * \brief Updates this entity.
167 */
168void Hookshot::update() {
169
170 Entity::update();
171
172 if (is_suspended()) {
173 return;
174 }
175
176 uint32_t now = System::now();
177 if (now >= next_sound_date) {
178 Sound::play("hookshot");
179 next_sound_date = now + 150;
180 }
181
182 if (entity_reached == nullptr) {
183 if (!going_back) {
184
185 if (has_to_go_back) {
186 going_back = true;
187 std::shared_ptr<Movement> movement = std::make_shared<TargetMovement>(
188 std::static_pointer_cast<Hero>(get_hero().shared_from_this()),
189 0,
190 0,
191 192,
192 true
193 );
194 clear_movement();
195 set_movement(movement);
196 }
197 else if (get_distance(get_hero()) >= 120) {
198 go_back();
199 }
200 }
201 else if (get_distance(get_hero()) == 0 ||
202 (get_movement() != nullptr && get_movement()->is_finished())) {
203 remove_from_map();
204 get_hero().start_state_from_ground();
205 }
206 }
207}

209/**
210 * \brief Draws the entity on the map.
211 */
212void Hookshot::draw_on_map() {
213
214 static constexpr int nb_links = 7;
215 static constexpr Point dxy[] = {
216 { 16, -5 },
217 { 0, -13 },
218 { -16, -5 },
219 { 0, 7 }
220 };
221
222 Entity::draw_on_map();
223
224 // also draw the links
225 const SpritePtr& sprite = get_sprite();
226 if (sprite == nullptr) {
227 return;
228 }
229 int direction = sprite->get_current_direction();
230 if (direction > 4) {
231 return;
232 }
233 int x1 = get_hero().get_x() + dxy[direction].x;
234 int y1 = get_hero().get_y() + dxy[direction].y;
235 int x2 = get_x();
236 int y2 = get_y() - 5;
237
238 Point link_xy;
239 for (int i = 0; i < nb_links; i++) {
240 link_xy.x = x1 + (x2 - x1) * i / nb_links;
241 link_xy.y = y1 + (y2 - y1) * i / nb_links;
242 get_map().draw_visual(*link_sprite, link_xy);
243 }
244}

246/**
247 * \brief Returns whether the hookshot is currently flying.
248 * \return true if the hookshot was shot, is not going back and has not reached any target yet
249 */
250bool Hookshot::is_flying() const {
251 return !is_going_back() && entity_reached == nullptr;
252}
253
254/**
255 * \brief Returns whether the hookshot is going back towards the hero, i.e. if go_back() has been called.
256 * \return true if the hookshot is going back
257 */
258bool Hookshot::is_going_back() const {
259 return has_to_go_back || going_back;
260}
261
262/**
263 * \brief Makes the hookshot go back towards the hero.
264 */
265void Hookshot::go_back() {
266
267 Debug::check_assertion(!is_going_back(), "The hookshot is already going back");
268
269 has_to_go_back = true;
270}
271
272/**
273 * \brief Attachs the hookshot to an entity and makes the hero move towards this entity.
274 * \param entity_reached the entity to attach the hookshot to
275 */
276void Hookshot::attach_to(Entity& entity_reached) {
277
278 Debug::check_assertion(this->entity_reached == nullptr,
279 "The hookshot is already attached to an entity");
280
281 this->entity_reached = &entity_reached;
282 clear_movement();
283 const SpritePtr& sprite = get_sprite();
284 if (sprite == nullptr) {
285 return;
286 }
287 int direction = sprite->get_current_direction();
288 if (direction > 4) {
289 return;
290 }
291 std::string path = " ";
292 path[0] = '0' + (direction * 2);
293 get_hero().set_movement(std::make_shared<PathMovement>(
294 path, 192, true, false, false
295 ));
296}

298/**
299 * \brief Notifies this entity that it has just failed to change its position
300 * because of obstacles.
301 */
302void Hookshot::notify_obstacle_reached() {
303
304 if (is_flying()) {
305 if (!get_map().test_collision_with_border(
306 get_movement()->get_last_collision_box_on_obstacle())) {
307 // play a sound unless the obstacle is the map border
308 Sound::play("sword_tapping");
309 }
310 go_back();
311 }
312}

314/**
315 * \brief This function is called when an enemy's sprite collides with a sprite of this entity.
316 * \param enemy the enemy
317 * \param enemy_sprite the enemy's sprite that overlaps the hero
318 * \param this_sprite the arrow sprite
319 */
320void Hookshot::notify_collision_with_enemy(
321 Enemy& enemy, Sprite& enemy_sprite, Sprite& /* this_sprite */) {
322
323 if (!overlaps(get_hero())) {
324 enemy.try_hurt(EnemyAttack::HOOKSHOT, *this, &enemy_sprite);
325 }
326}

354/**
355 * \brief This function is called when a destructible item detects a non-pixel precise collision with this entity.
356 * \param destructible the destructible item
357 * \param collision_mode the collision mode that detected the event
358 */
359void Hookshot::notify_collision_with_destructible(Destructible& destructible, CollisionMode /* collision_mode */) {
360
361 if (destructible.is_obstacle_for(*this) && is_flying()) {
362
363 if (destructible.get_can_explode()) {
364 destructible.explode();
365 go_back();
366 }
367 else {
368 attach_to(destructible);
369 }
370 }
371}

384/**
385 * \brief This function is called when a switch detects a collision with this entity.
386 * \param sw the switch
387 * \param collision_mode the collision mode that detected the event
388 */
389void Hookshot::notify_collision_with_switch(Switch& sw, CollisionMode collision_mode) {
390
391 if (is_flying() && collision_mode == COLLISION_OVERLAPPING) {
392
393 sw.try_activate();
394 if (!is_going_back()) {
395 go_back();
396 Sound::play("sword_tapping");
397 }
398 }
399}
