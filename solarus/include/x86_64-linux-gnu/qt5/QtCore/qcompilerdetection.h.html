50/*
51 The compiler, must be one of: (Q_CC_x)
52
53 SYM - Digital Mars C/C++ (used to be Symantec C++)
54 MSVC - Microsoft Visual C/C++, Intel C++ for Windows
55 BOR - Borland/Turbo C++
56 WAT - Watcom C++
57 GNU - GNU C++
58 COMEAU - Comeau C++
59 EDG - Edison Design Group C++
60 OC - CenterLine C++
61 SUN - Forte Developer, or Sun Studio C++
62 MIPS - MIPSpro C++
63 DEC - DEC C++
64 HPACC - HP aC++
65 USLC - SCO OUDK and UDK
66 CDS - Reliant C++
67 KAI - KAI C++
68 INTEL - Intel C++ for Linux, Intel C++ for Windows
69 HIGHC - MetaWare High C/C++
70 PGI - Portland Group C++
71 GHS - Green Hills Optimizing C++ Compilers
72 RVCT - ARM Realview Compiler Suite
73 CLANG - C++ front-end for the LLVM compiler
74
75
76 Should be sorted most to least authoritative.
77*/

213/* IBM compiler versions are a bit messy. There are actually two products:
214 the C product, and the C++ product. The C++ compiler is always packaged
215 with the latest version of the C compiler. Version numbers do not always
216 match. This little table (I'm not sure it's accurate) should be helpful:
217
218 C++ product C product
219
220 C Set 3.1 C Compiler 3.0
221 ... ...
222 C++ Compiler 3.6.6 C Compiler 4.3
223 ... ...
224 Visual Age C++ 4.0 ...
225 ... ...
226 Visual Age C++ 5.0 C Compiler 5.0
227 ... ...
228 Visual Age C++ 6.0 C Compiler 6.0
229
230 Now:
231 __xlC__ is the version of the C compiler in hexadecimal notation
232 is only an approximation of the C++ compiler version
233 __IBMCPP__ is the version of the C++ compiler in decimal notation
234 but it is not defined on older compilers like C Set 3.1 */
235#elif defined(__xlC__)
236# define Q_CC_XLC
237# define Q_FULL_TEMPLATE_INSTANTIATION
238# if __xlC__ < 0x400
239# error "Compiler not supported"
240# elif __xlC__ >= 0x0600
241# define Q_ALIGNOF(type) __alignof__(type)
242# define Q_TYPEOF(expr) __typeof__(expr)
243# define Q_DECL_ALIGN(n) __attribute__((__aligned__(n)))
244# define Q_PACKED __attribute__((__packed__))
245# endif

247/* Older versions of DEC C++ do not define __EDG__ or __EDG - observed
248 on DEC C++ V5.5-004. New versions do define __EDG__ - observed on
249 Compaq C++ V6.3-002.
250 This compiler is different enough from other EDG compilers to handle
251 it separately anyway. */
252#elif defined(__DECCXX) || defined(__DECC)
253# define Q_CC_DEC
254/* Compaq C++ V6 compilers are EDG-based but I'm not sure about older
255 DEC C++ V5 compilers. */
256# if defined(__EDG__)
257# define Q_CC_EDG
258# endif
259/* Compaq has disabled EDG's _BOOL macro and uses _BOOL_EXISTS instead
260 - observed on Compaq C++ V6.3-002.
261 In any case versions prior to Compaq C++ V6.0-005 do not have bool. */
262# if !defined(_BOOL_EXISTS)
263# error "Compiler not supported"
264# endif
265/* Spurious (?) error messages observed on Compaq C++ V6.5-014. */
266# define Q_NO_USING_KEYWORD
267/* Apply to all versions prior to Compaq C++ V6.0-000 - observed on
268 DEC C++ V5.5-004. */

271# endif

280# define Q_CC_EDG
