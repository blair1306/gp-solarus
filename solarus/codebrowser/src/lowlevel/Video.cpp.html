68/**
69 * \brief Creates the window but does not show it.
70 * \param args Command-line arguments.
71 */
72void create_window() {
73
74 Debug::check_assertion(main_window == nullptr, "Window already exists");
75
76#if SOLARUS_HAVE_OPENGL == 1
77 // Set OpenGL as the default renderer driver when available, to avoid using Direct3d.
78 SDL_SetHintWithPriority(SDL_HINT_RENDER_DRIVER, "opengl", SDL_HINT_DEFAULT);
79
80 // Set the default OpenGL built-in shader (nearest)
81 SDL_SetHint(SDL_HINT_RENDER_OPENGL_SHADERS, "0");
82#endif
83
84 std::string title = std::string("Solarus ") + SOLARUS_VERSION;
85 main_window = SDL_CreateWindow(
86 title.c_str(),
87 SDL_WINDOWPOS_CENTERED,
88 SDL_WINDOWPOS_CENTERED,
89 wanted_quest_size.width,
90 wanted_quest_size.height,
91 SDL_WINDOW_HIDDEN | SDL_WINDOW_RESIZABLE
92#if SOLARUS_HAVE_OPENGL == 1
93 | SDL_WINDOW_OPENGL
94#endif
95 );
96 Debug::check_assertion(main_window != nullptr,
97 std::string("Cannot create the window: ") + SDL_GetError());
98
99 int acceleration_flag = acceleration_enabled ?
100 SDL_RENDERER_ACCELERATED : SDL_RENDERER_SOFTWARE;
101
102 main_renderer = SDL_CreateRenderer(main_window, -1, acceleration_flag
103#if SOLARUS_HAVE_OPENGL != 1
104 | SDL_RENDERER_PRESENTVSYNC
105#endif
106 );
107
108 if (main_renderer == nullptr) {
109 // Try without vsync.
110 main_renderer = SDL_CreateRenderer(main_window, -1, acceleration_flag);
111 if (main_renderer == nullptr && acceleration_enabled) {
112 // Try without acceleration.
113 acceleration_enabled = false;
114 acceleration_flag = SDL_RENDERER_SOFTWARE;
115 main_renderer = SDL_CreateRenderer(main_window, -1, acceleration_flag);
116 }
117 }
118
119 Debug::check_assertion(main_renderer != nullptr,
120 std::string("Cannot create the renderer: ") + SDL_GetError());
121
122 // Allow blending mode for direct drawing primitives.
123 SDL_SetRenderDrawBlendMode(main_renderer, SDL_BLENDMODE_BLEND);
124
125 // Get the first renderer format which supports alpha channel and is not a unique format.
126 SDL_RendererInfo renderer_info;
127 SDL_GetRendererInfo(main_renderer, &renderer_info);
128 for (unsigned i = 0; i < renderer_info.num_texture_formats; ++i) {
129
130 if (!SDL_ISPIXELFORMAT_FOURCC(renderer_info.texture_formats[i])
131 && SDL_ISPIXELFORMAT_ALPHA(renderer_info.texture_formats[i])) {
132 pixel_format = SDL_AllocFormat(renderer_info.texture_formats[i]);
133 break;
134 }
135 }
136
137 Debug::check_assertion(pixel_format != nullptr, "No compatible pixel format");
138
139 // Check renderer's flags
140 rendering_driver_name = renderer_info.name;
141 rendertarget_supported = (renderer_info.flags & SDL_RENDERER_TARGETTEXTURE) != 0;
142 acceleration_enabled = acceleration_enabled
143 && (renderer_info.flags & SDL_RENDERER_ACCELERATED) != 0;
144 if (acceleration_enabled) {
145 // Solarus uses accelerated graphics as of version 1.2 with SDL2.
146 Logger::info("2D acceleration: yes");
147 }
148 else {
149 // Acceleration may be disabled because the user decided so or because the
150 // system does not support it.
151 // This is not a problem: the engine runs perfectly in software mode.
152 Logger::info("2D acceleration: no");
153 }
154}

156/**
157 * \brief Creates the list of video modes.
158 *
159 * Some video modes are hardcoded, and some others may be added by the quest.
160 */
161void initialize_video_modes() {
162
163 // Decide whether we enable shaders.
164 shaders_enabled = rendertarget_supported && Video::is_acceleration_enabled() && ShaderContext::initialize();
165
166 // Initialize hardcoded video modes.
167 all_video_modes.emplace_back(
168 "normal",
169 quest_size * 2,
170 nullptr,
171 nullptr
172 );
173 all_video_modes.emplace_back(
174 "scale2x",
175 quest_size * 2,
176 std::unique_ptr<PixelFilter>(new Scale2xFilter()),
177 nullptr
178 );
179 all_video_modes.emplace_back(
180 "hq2x",
181 quest_size * 2,
182 std::unique_ptr<PixelFilter>(new Hq2xFilter()),
183 nullptr
184 );
185 all_video_modes.emplace_back(
186 "hq3x",
187 quest_size * 3,
188 std::unique_ptr<PixelFilter>(new Hq3xFilter()),
189 nullptr
190 );
191 all_video_modes.emplace_back(
192 "hq4x",
193 quest_size * 4,
194 std::unique_ptr<PixelFilter>(new Hq4xFilter()),
195 nullptr
196 );
197 default_video_mode = &all_video_modes[0];
198 // TODO If shaders are enabled, use a C++ shader version of Scale2x and Hq4x instead.
199
200 // Initialize quest custom video modes. These can only include shaded modes.
201 if (shaders_enabled) {
202
203 // Initialize the render target
204 render_target = SDL_CreateTexture(
205 main_renderer,
206 pixel_format->format,
207 SDL_TEXTUREACCESS_TARGET,
208 quest_size.width,
209 quest_size.height
210 );
211 SDL_SetTextureBlendMode(render_target, SDL_BLENDMODE_BLEND);
212
213 // Get all shaders of the quest's shader/videomodes folder.
214 std::vector<std::string> shader_names =
215 QuestFiles::data_files_enumerate("shaders/videomodes/", true, false);
216 // FIXME don't enumerate data files, use the quest resource system like always.
217
218 for (unsigned i = 0; i < shader_names.size(); ++i) {
219
220 // Load the shader and add the corresponding video mode.
221 std::unique_ptr<Shader> video_mode_shader =
222 ShaderContext::create_shader(shader_names.at(i));
223 if (video_mode_shader != nullptr && video_mode_shader->is_valid()) {
224
225 const std::string& video_mode_name = video_mode_shader->get_name();
226 if (Video::get_video_mode_by_name(video_mode_name) != nullptr) {
227 Debug::error("There is already a video mode with name '" + video_mode_name);
228 continue;
229 }
230
231 const Size scaled_quest_size = quest_size * video_mode_shader->get_default_window_scale();
232 all_video_modes.emplace_back(
233 video_mode_shader->get_name(),
234 scaled_quest_size,
235 nullptr,
236 std::move(video_mode_shader)
237 );
238 }
239 }
240 }
241
242 // Everything is ready now.
243 Video::set_default_video_mode();
244}

248/**
249 * \brief Initializes the video system.
250 *
251 * This method should be called when the program starts.
252 * Options recognized:
253 * -no-video
254 * -video-acceleration=yes|no
255 * -quest-size=WIDTHxHEIGHT
256 *
257 * \param args Command-line arguments.
258 */
259void Video::initialize(const Arguments& args) {
260
261 Debug::check_assertion(!is_initialized(), "Video system already initialized");
262
263 // Check the -no-video and the -quest-size options.
264 const std::string& quest_size_string = args.get_argument_value("-quest-size");
265 disable_window = args.has_argument("-no-video");
266
267 wanted_quest_size = {
268 SOLARUS_DEFAULT_QUEST_WIDTH,
269 SOLARUS_DEFAULT_QUEST_HEIGHT
270 };
271
272 if (!quest_size_string.empty()) {
273 if (!parse_size(quest_size_string, wanted_quest_size)) {
274 Debug::error(std::string("Invalid quest size: '") + quest_size_string + "'");
275 }
276 }
277
278 if (args.get_argument_value("-video-acceleration") == "no") {
279 acceleration_enabled = false;
280 }
281 else {
282 // Accelerated by default.
283 acceleration_enabled = true;
284 }
285
286 if (disable_window) {
287 // Create a pixel format anyway to make surface and color operations work,
288 // even though nothing will ever be rendered.
289 pixel_format = SDL_AllocFormat(SDL_PIXELFORMAT_ABGR8888);
290 }
291 else {
292 create_window();
293 }
294}

296/**
297 * \brief Closes the video system.
298 */
299void Video::quit() {
300
301 if (!is_initialized()) {
302 return;
303 }
304
305 ShaderContext::quit();
306
307 if (is_fullscreen()) {
308 // Get back on desktop before destroy the window.
309 SDL_SetWindowFullscreen(main_window, 0);
310 }
311
312 all_video_modes.clear();
313
314 if (pixel_format != nullptr) {
315 SDL_FreeFormat(pixel_format);
316 pixel_format = nullptr;
317 }
318 if (main_renderer != nullptr) {
319 SDL_DestroyRenderer(main_renderer);
320 main_renderer = nullptr;
321 }
322 if (main_window != nullptr) {
323 SDL_DestroyWindow(main_window);
324 main_window = nullptr;
325 }
326
327 // TODO wrap in a class
328 rendering_driver_name = "";
329 disable_window = false;
330 fullscreen_window = false;
331 rendertarget_supported = false;
332 shaders_enabled = false;
333 acceleration_enabled = false;
334 scaled_surface = nullptr;
335 video_mode = nullptr;
336 default_video_mode = nullptr;
337 normal_quest_size = Size();
338 min_quest_size = Size();
339 max_quest_size = Size();
340 quest_size = Size();
341 wanted_quest_size = Size();
342 window_size = Size();
343
344}

403/**
404 * \brief Returns whether 2D hardware acceleration is currently enabled.
405 *
406 * 2D acceleration is enabled if the system supports it, except when the
407 * current video mode needs to work in software mode
408 * (typically because of a pixel filter implemented in software).
409 *
410 * \return \c true if GPU acceleration is active.
411 */
412bool Video::is_acceleration_enabled() {
413
414 const PixelFilter* software_filter = nullptr;
415 if (video_mode != nullptr) {
416 software_filter = video_mode->get_software_filter();
417 }
418
419 return acceleration_enabled // 2D acceleration must be available on the system.
420 && software_filter == nullptr; // No software pixel filter must be active.
421}

491/**
492 * \brief Sets the next video mode.
493 *
494 * The fullscreen flag is preserved.
495 * The window size is reset to the default size of the new video mode.
496 */
497void Video::switch_video_mode() {
498
499 if (all_video_modes.size() <= 1) {
500 return;
501 }
502
503 // Find the current video mode in the list and traverse the list from there.
504 std::vector<VideoMode>::const_iterator it;
505 for (it = all_video_modes.begin(); it != all_video_modes.end(); ++it) {
506 if (it->get_name() == video_mode->get_name()) {
507 break;
508 }
509 };
510
511 const VideoMode* mode = nullptr;
512 do {
513 if (it != all_video_modes.end()) {
514 ++it;
515 }
516 if (it == all_video_modes.end()) {
517 it = all_video_modes.begin();
518 }
519 mode = &(*it);
520 } while (mode == nullptr || !is_mode_supported(*mode));
521
522 set_video_mode(*mode);
523}

535/**
536 * \brief Sets the video mode and the fullscreen setting.
537 *
538 * If the mode changes, the window size is reset to its default value.
539 *
540 * \param mode The video mode to set.
541 * \param fullscreen \c true to set the window to fullscreen.
542 * \return true in case of success, false if this mode is not supported.
543 */
544bool Video::set_video_mode(const VideoMode& mode, bool fullscreen) {
545
546 bool mode_changed = video_mode == nullptr
547 || mode.get_name() != video_mode->get_name();
548
549 if (!is_mode_supported(mode)) {
550 return false;
551 }
552
553 Uint32 fullscreen_flag;
554 if (fullscreen) {
555 fullscreen_flag = SDL_WINDOW_FULLSCREEN_DESKTOP;
556 window_size = get_window_size(); // Store the window size before fullscreen.
557 }
558 else {
559 fullscreen_flag = 0;
560 }
561
562 video_mode = &mode;
563 fullscreen_window = fullscreen;
564
565 if (!disable_window) {
566
567 scaled_surface = nullptr;
568
569 Size render_size = quest_size;
570
571 const PixelFilter* software_filter = mode.get_software_filter();
572 if (software_filter != nullptr) {
573 int factor = software_filter->get_scaling_factor();
574 render_size = quest_size * factor;
575 scaled_surface = Surface::create(render_size);
576 scaled_surface->fill_with_color(Color::black); // To initialize the internal surface.
577 }
578
579 // Initialize the window.
580 // Set fullscreen flag first to set the size on the right mode.
581 SDL_SetWindowFullscreen(main_window, fullscreen_flag);
582 SDL_RenderSetLogicalSize(
583 main_renderer,
584 render_size.width,
585 render_size.height);
586 SDL_ShowCursor(visible_cursor);
587
588 if (mode_changed) {
589 reset_window_size();
590 }
591 }
592
593 if (mode_changed) {
594 Logger::info(std::string("Video mode: ") + video_mode->get_name());
595
596 }
597
598 return true;
599}

644/**
645 * \brief Draws the quest surface on the screen with the current video mode.
646 * \param quest_surface The quest surface to render on the screen.
647 */
648void Video::render(const SurfacePtr& quest_surface) {
649
650 if (disable_window) {
651 return;
652 }
653
654 Debug::check_assertion(video_mode != nullptr,
655 "Missing video mode");
656
657 // See if there is a filter to apply.
658 const Shader* hardware_filter = video_mode->get_hardware_filter();
659 const PixelFilter* software_filter = video_mode->get_software_filter();
660 if (hardware_filter != nullptr) {
661 // OpenGL rendering.
662 hardware_filter->render(quest_surface);
663 }
664 else {
665 // SDL rendering, with acceleration if supported, and optionally with
666 // a software filter.
667 Surface* surface_to_render = nullptr;
668 if (software_filter != nullptr) {
669 Debug::check_assertion(scaled_surface != nullptr,
670 "Missing destination surface for scaling");
671 quest_surface->apply_pixel_filter(*software_filter, *scaled_surface);
672 surface_to_render = scaled_surface.get();
673 }
674 else {
675 surface_to_render = quest_surface.get();
676 }
677
678 SDL_SetRenderDrawColor(main_renderer, 0, 0, 0, 255);
679 SDL_RenderSetClipRect(main_renderer, nullptr);
680 SDL_RenderClear(main_renderer);
681 surface_to_render->render(main_renderer);
682 SDL_RenderPresent(main_renderer);
683 }
684}

704/**
705 * \brief Gets the width and the height values from a size string of the form
706 * "320x240".
707 * \param size_string The input string.
708 * \param size The resulting size. Unchanged in case of failure.
709 * \return true in case of success, false if the string is not a valid size.
710 */
711bool Video::parse_size(const std::string& size_string, Size& size) {
712
713 size_t index = size_string.find('x');
714 if (index == std::string::npos || index + 1 >= size_string.size()) {
715 return false;
716 }
717
718 const std::string& width_string = size_string.substr(0, index);
719 const std::string& height_string = size_string.substr(index + 1);
720
721 int width = 0;
722 int height = 0;
723 std::istringstream iss(width_string);
724 if (!(iss >> width) || width < 0) {
725 return false;
726 }
727
728 iss.str(height_string);
729 iss.clear();
730 if (!(iss >> height) || height < 0) {
731 return false;
732 }
733
734 size = { width, height };
735 return true;
736}

769 * \param min_size Minimum size for this quest.
770 * \param max_size Maximum size for this quest.
771 */
772void Video::set_quest_size_range(
773 const Size& normal_size,
774 const Size& min_size,
775 const Size& max_size) {
776
777 Debug::check_assertion(
778 normal_size.width >= min_size.width
779 && normal_size.height >= min_size.height
780 && normal_size.width <= max_size.width
781 && normal_size.height <= max_size.height,
782 "Invalid quest size range");
783
784 normal_quest_size = normal_size;
785 min_quest_size = min_size;
786 max_quest_size = max_size;
787
788 if (wanted_quest_size.width < min_size.width
789 || wanted_quest_size.height < min_size.height
790 || wanted_quest_size.width > max_size.width
791 || wanted_quest_size.height > max_size.height) {
792 std::ostringstream oss;
793 oss << "Cannot use quest size "
794 << wanted_quest_size.width << "x" << wanted_quest_size.height
795 << ": this quest only supports "
796 << min_size.width << "x" << min_size.height
797 << " to "
798 << max_size.width << "x" << max_size.height
799 << ". Using "
800 << normal_size.width << "x" << normal_size.height
801 << " instead.";
802 Debug::warning(oss.str());
803 quest_size = normal_size;
804 }
805 else {
806 quest_size = wanted_quest_size;
807 }
808
809 // We know the quest size: finish the initialization.
810 initialize_video_modes();
811}

897/**
898 * \brief Converts window coordinates to quest size coordinates.
899 * \param window_xy A position relative to the window, not including
900 * the title bar.
901 * \return The position in quest size coordinate.
902 */
903Point Video::window_to_quest_coordinates(const Point& window_xy) {
904
905 Rectangle viewport = get_viewport();
906
907 float scale_x = 0.0;
908 float scale_y = 0.0;
909 SDL_RenderGetScale(get_renderer(), &scale_x, &scale_y);
910
911 const double x_position = window_xy.x - viewport.get_x() * scale_x;
912 const double y_position = window_xy.y - viewport.get_y() * scale_y;
913 const double quest_size_width = quest_size.width;
914 const double quest_size_height = quest_size.height;
915 const double window_width = viewport.get_width() * scale_x;
916 const double window_height = viewport.get_height() * scale_y;
917
918 Debug::check_assertion(!quest_size.is_flat(), "Quest size is not initialized");
919 Debug::check_assertion(!viewport.is_flat(), "Viewport is not initialized");
920
921 return Point(
922 (int) (x_position * quest_size_width / window_width),
923 (int) (y_position * quest_size_height / window_height));
924}

926/**
927 * \brief Converts logical renderer coordinates to quest size coordinates.
928 * \param[in] renderer_xy A position in renderer coordinates, without window scaling.
929 * \param[out] quest_xy The corresponding value in quest coordinates.
930 * \return \c false if the position is not inside the renderer.
931 */
932bool Video::renderer_to_quest_coordinates(
933 const Point& renderer_xy,
934 Point& quest_xy
935) {
936
937 int renderer_width = 0;
938 int renderer_height = 0;
939 SDL_RenderGetLogicalSize(get_renderer(), &renderer_width, &renderer_height);
940
941 const double quest_width = quest_size.width;
942 const double quest_height = quest_size.height;
943
944 quest_xy = {
945 (int) (renderer_xy.x * quest_width / renderer_width),
946 (int) (renderer_xy.y * quest_height / renderer_height)
947 };
948
949 if (quest_xy.x < 0
950 || quest_xy.y < 0
951 || quest_xy.x >= quest_width
952 || quest_xy.y >= quest_height
953 ) {
954 return false;
955 }
956
957 return true;
958}
