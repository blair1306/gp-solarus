21/**
22 * \brief Creates a fatal error that should stop Solarus.
23 * \param error_message A message describing the error.
24 */
25SolarusFatal::SolarusFatal(const std::string& error_message):
26 error_message(error_message) {
27
28}
29
30/**
31 * \brief Returns the error message.
32 * \return A message describing the error.
33 */
34const char* SolarusFatal::what() const noexcept {
35 return error_message.c_str();
36}
