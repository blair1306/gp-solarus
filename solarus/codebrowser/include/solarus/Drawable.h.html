17#ifndef SOLARUS_DRAWABLE_H
18#define SOLARUS_DRAWABLE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/BlendMode.h"
22#include "solarus/lowlevel/Point.h"
23#include "solarus/lowlevel/Size.h"
24#include "solarus/lowlevel/SurfacePtr.h"
25#include "solarus/lua/ExportableToLua.h"
26#include "solarus/lua/ScopedLuaRef.h"
27#include <memory>
28
29namespace Solarus {
30
31class Movement;
32class Transition;
33class Rectangle;
34
35/**
36 * \brief An object that can be drawn on a surface with additional effects.
37 *
38 * Possible additional effects are movements and transitions.
39 * Lua callbacks may be associated to the end of these effects.
40 */
41class Drawable: public ExportableToLua {
42
43 public:
44
45 virtual ~Drawable();
46
47 virtual Point get_origin() const;
48
49 /**
50 * \brief Returns the size of this drawable.
51 * \return The size of this drawable.
52 */
53 virtual Size get_size() const = 0;
54
55 // dynamic effects
56 void start_movement(const std::shared_ptr<Movement>& movement);
57 void stop_movement();
58 const std::shared_ptr<Movement>& get_movement();
59 const Point& get_xy() const;
60 void set_xy(const Point& xy);
61
62 void start_transition(
63 std::unique_ptr<Transition> transition,
64 const ScopedLuaRef& callback_ref
65 );
66 void stop_transition();
67 Transition* get_transition();
68
69 // drawing with effects
70
71 void draw(const SurfacePtr& dst_surface);
72 void draw(const SurfacePtr& dst_surface, int x, int y);
73 void draw(const SurfacePtr& dst_surface, const Point& dst_position);
74 void draw_region(const Rectangle& region, const SurfacePtr& dst_surface);
75 void draw_region(const Rectangle& region,
76 const SurfacePtr& dst_surface, const Point& dst_position);
77
78 /**
79 * \brief Draws this object without applying dynamic effects.
80 *
81 * Redefine this function to draw your object onto the destination
82 * surface.
83 *
84 * \param dst_surface The destination surface.
85 * \param dst_position Coordinates on the destination surface.
86 */
87 virtual void raw_draw(
88 Surface& dst_surface,
89 const Point& dst_position
90 ) = 0;
91
92 /**
93 * \brief Draws a subrectangle of this object without applying dynamic
94 * effects.
95 *
96 * Redefine this function to draw your object onto the destination
97 * surface.
98 *
99 * \param region The subrectangle to draw in this object.
100 * \param dst_surface The destination surface.
101 * \param dst_position Coordinates on the destination surface.
102 */
103 virtual void raw_draw_region(
104 const Rectangle& region,
105 Surface& dst_surface,
106 const Point& dst_position
107 ) = 0;
108
109 /**
110 * \brief Draws a transition effect on this drawable object.
111 *
112 * Redefine this function to apply the transition effect on the surface
113 * of your object.
114 *
115 * \param transition The transition effect to apply.
116 *
117 * TODO Since there is get_transition_surface() now, this function can probably be removed.
118 */
119 virtual void draw_transition(Transition& transition) = 0;
120
121 /**
122 * \brief Returns the surface where transitions on this drawable object
123 * are applied.
124 * \return The surface for transitions.
125 */
126 virtual Surface& get_transition_surface() = 0;
127
128 virtual void update();
129 bool is_suspended() const;
130 virtual void set_suspended(bool suspended);
131
132 BlendMode get_blend_mode() const;
133 void set_blend_mode(BlendMode blend_mode);
134
135 protected:
136
137 Drawable();
138
139 private:
140
141 Point xy; /**< Current position of this object
142 * (result of movements). */
143 std::shared_ptr<Movement>
144 movement; /**< A movement applied or nullptr. */
145 std::unique_ptr<Transition>
146 transition; /**< A transition applied or nullptr. */
147 ScopedLuaRef transition_callback_ref;
148 /**< Lua registry ref of a function to call
149 * when the transition finishes */
150 bool suspended; /**< Whether this object is suspended. */
151 BlendMode blend_mode; /**< How to draw this object on a surface. */
152};
153
154}
155
156#endif
