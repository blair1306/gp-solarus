42#ifndef QTRANSLATOR_H
43#define QTRANSLATOR_H
44
45#include <QtCore/qobject.h>
46#include <QtCore/qbytearray.h>
47
48QT_BEGIN_NAMESPACE
49
50
51#ifndef QT_NO_TRANSLATION
52
53class QLocale;
54class QTranslatorPrivate;
55
56class Q_CORE_EXPORT QTranslator : public QObject
57{
58 Q_OBJECT
59public:
60 explicit QTranslator(QObject *parent = 0);
61 ~QTranslator();
62
63 virtual QString translate(const char *context, const char *sourceText,
64 const char *disambiguation = 0, int n = -1) const;
65
66 virtual bool isEmpty() const;
67
68 bool load(const QString & filename,
69 const QString & directory = QString(),
70 const QString & search_delimiters = QString(),
71 const QString & suffix = QString());
72 bool load(const QLocale & locale,
73 const QString & filename,
74 const QString & prefix = QString(),
75 const QString & directory = QString(),
76 const QString & suffix = QString());
77 bool load(const uchar *data, int len, const QString &directory = QString());
78
79private:
80 Q_DISABLE_COPY(QTranslator)
81 Q_DECLARE_PRIVATE(QTranslator)
82};
83
84#endif // QT_NO_TRANSLATION
85
86QT_END_NAMESPACE
87
88#endif // QTRANSLATOR_H
