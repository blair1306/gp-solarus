72/**
73 * \brief Starts this state.
74 *
75 * This function is called automatically when this state becomes the active
76 * state of the hero.
77 *
78 * \param previous_state The previous state (for information).
79 */
80void Hero::PlayerMovementState::start(const State* previous_state) {
81 Hero& hero = get_entity();
82
83 HeroState::start(previous_state);
84
85 player_movement = std::make_shared<PlayerMovement>(
86 hero.get_walking_speed()
87 );
88 hero.set_movement(player_movement);
89
90 if (is_current_state()) { // yes, the state may have already changed
91 get_player_movement()->compute_movement();
92 if (is_current_state()) {
93 if (get_wanted_movement_direction8() != -1) {
94 set_animation_walking();
95 }
96 else {
97 set_animation_stopped();
98 }
99 }
100 }
101}
102
103/**
104 * \brief Ends this state.
105 *
106 * This function is called automatically when this state is not the active state anymore.
107 * You should here close everything the start() function has opened.
108 * The destructor will be called at the next cycle.
109 *
110 * \param next_state The next state (for information).
111 */
112void Hero::PlayerMovementState::stop(const State* next_state) {
113
114 HeroState::stop(next_state);
115
116 get_entity().clear_movement();
117 get_sprites().set_animation_stopped_normal();
118 cancel_jumper();
119 player_movement = nullptr;
120}

135/**
136 * \brief Updates this state.
137 *
138 * This function may change the state of the hero,
139 * so when you call it from a subclass, call is_current_state() then.
140 */
141void Hero::PlayerMovementState::update() {
142 Hero& hero = get_entity();
143
144 HeroState::update();
145
146 if (!is_suspended()) {
147
148 if (current_jumper != nullptr) {
149
150 const int jump_direction8 = current_jumper->get_direction();
151 if (!current_jumper->is_enabled()
152 || current_jumper->is_being_removed()
153 || !current_jumper->is_in_jump_position(hero, hero.get_bounding_box(), false)) {
154
155 // Cancel the jumper preparation.
156 current_jumper = nullptr;
157 jumper_start_date = 0;
158 }
159 else if (System::now() >= jumper_start_date) {
160 // Time to make the jump and everything is okay.
161 hero.start_jumping(
162 jump_direction8, current_jumper->get_jump_length(), true, true);
163 }
164 }
165 }
166}

168/**
169 * \brief Notifies this state that the game was just suspended or resumed.
170 * \param suspended \c true if the game is suspended.
171 */
172void Hero::PlayerMovementState::set_suspended(bool suspended) {
173
174 HeroState::set_suspended(suspended);
175
176 if (!suspended) {
177 if (jumper_start_date != 0) {
178 jumper_start_date += System::now() - get_when_suspended();
179 }
180 }
181}

195/**
196 * \brief Returns whether the player can control his movements in the current state.
197 * \return true if the player can control his movements
198 */
199bool Hero::PlayerMovementState::can_control_movement() const {
200
201 // The player has control, unless a script has set another movement.
202 return get_entity().get_movement().get() == get_player_movement().get();
203}

205/**
206 * \brief Returns the direction of the hero's movement as defined by the controls applied by the player
207 * and the movements allowed is the current state.
208 *
209 * If he is not moving, -1 is returned.
210 * This direction may be different from the real movement direction because of obstacles.
211 *
212 * \return the hero's wanted direction between 0 and 7, or -1 if he is stopped

214int Hero::PlayerMovementState::get_wanted_movement_direction8() const {
215 return get_player_movement()->get_wanted_direction8();
216}

231/**
232 * \brief Notifies this state that the movement has changed.
233 *
234 * This function is called when the hero's movement direction changes (for instance
235 * because the player pressed or released a directional key, or the hero just reached an obstacle).
236 * The animations and collisions should be updated according to the new movement.
237 */
238void Hero::PlayerMovementState::notify_movement_changed() {
239
240 // the movement has changed: update the animation of the sprites
241
242 bool movement_walking = get_wanted_movement_direction8() != -1;
243 bool sprites_walking = get_sprites().is_walking();
244
245 if (movement_walking && !sprites_walking) {
246 set_animation_walking();
247 }
248 else if (!movement_walking && sprites_walking) {
249 set_animation_stopped();
250 }
251}

301/**
302 * \brief Notifies this state that the hero is activating a jumper.
303 * \param jumper The jumper activated.
304 */
305void Hero::PlayerMovementState::notify_jumper_activated(Jumper& jumper) {
306
307 if (&jumper == current_jumper.get()) {
308 // We already know.
309 return;
310 }
311
312 // Add a small delay before jumping.
313 current_jumper = std::static_pointer_cast<Jumper>(jumper.shared_from_this());
314 jumper_start_date = System::now() + 200;
315}
316
317/**
318 * \brief Cancels the jump preparation that was ongoing if any.
319 */
320void Hero::PlayerMovementState::cancel_jumper() {
321
322 if (current_jumper != nullptr) {
323 current_jumper = nullptr;
324 jumper_start_date = 0;
325 }
326}
