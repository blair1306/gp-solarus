323if ((sq.section(QChar::fromLatin1('.'),0,0).toInt()<<16)+\
324(sq.section(QChar::fromLatin1('.'),1,1).toInt()<<8)+\
325sq.section(QChar::fromLatin1('.'),2,2).toInt()<(s.section(QChar::fromLatin1('.'),0,0).toInt()<<16)+\
326(s.section(QChar::fromLatin1('.'),1,1).toInt()<<8)+\
327s.section(QChar::fromLatin1('.'),2,2).toInt()) { \
328if (!qApp){ \
329 new QApplication(argc,argv); \
330} \

333str)).arg(QString::fromLatin1(qVersion())); QMessageBox::critical(0, QApplication::tr(\
334"Incompatible Qt Library Error"), s, QMessageBox::Abort, 0); qFatal("%s", s.toLatin1().data()); }}
