26namespace Solarus {
27
28/**
29 * \brief Constructor.
30 * \param hero The hero controlled by this state.
31 */
32Hero::BowState::BowState(Hero& hero):
33 HeroState(hero, "bow") {
34}
35
36/**
37 * \brief Starts this state.
38 * \param previous_state the previous state
39 */
40void Hero::BowState::start(const State* previous_state) {
41
42 HeroState::start(previous_state);
43 get_sprites().set_animation("bow");
44}
45
46/**
47 * \brief Updates this state.
48 */
49void Hero::BowState::update() {
50
51 HeroState::update();
52
53 Hero& hero = get_entity();
54 if (get_sprites().is_animation_finished()) {
55 Sound::play("bow");
56 get_entities().add_entity(std::make_shared<Arrow>(hero));
57 hero.set_state(new FreeState(hero));
58 }
59}
60
61/**
62 * \copydoc Entity::State::can_avoid_stream
63 */
64bool Hero::BowState::can_avoid_stream(const Stream& /* stream */) const {
65 return true;
66}
