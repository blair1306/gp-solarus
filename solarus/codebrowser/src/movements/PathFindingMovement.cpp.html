27/**
28 * \brief Creates a chase movement.
29 * \param speed speed of the movement in pixels per second
30 */
31PathFindingMovement::PathFindingMovement(int speed):
32 PathMovement("", speed, false, false, true),
33 target(),
34 next_recomputation_date(0) {
35
36}
37
38/**
39 * \brief Sets the entity to target with this movement.
40 */
41void PathFindingMovement::set_target(const EntityPtr& target) {
42
43 this->target = target;
44 next_recomputation_date = System::now() + 100;
45}
46
47/**
48 * \brief Updates the position.
49 */
50void PathFindingMovement::update() {
51
52 PathMovement::update();
53
54 if (target != nullptr && target->is_being_removed()) {
55 target = nullptr;
56 }
57
58 if (is_suspended()) {
59 return;
60 }
61
62 if (PathMovement::is_finished()) {
63
64 // there was a collision or the path was made
65 if (target != nullptr
66 && System::now() >= next_recomputation_date
67 && get_entity()->is_aligned_to_grid()) {
68 recompute_movement();
69 }
70 else {
71 set_path(create_random_path());
72 }
73 }
74}
75
76/**
77 * \brief Calculates the direction and the speed of the movement
78 * depending on the target.
79 */
80void PathFindingMovement::recompute_movement() {
81
82 if (target != nullptr) {
83 PathFinding path_finding(get_entity()->get_map(), *get_entity(), *target);
84 std::string path = path_finding.compute_path();
85
86 uint32_t min_delay;
87 if (path.size() == 0) {
88 // the target is too far or there is no path
89 path = create_random_path();
90
91 // no path was found: no need to try again very soon
92 // (note that the A* algorithm is very costly when it explores all nodes without finding a solution)
93 min_delay = 3000;
94 }
95 else {
96 // a path was found: we need to update it frequently (and the A* algorithm is much faster in general when there is a solution)
97 min_delay = 300;
98 }
99 // compute a new path every random delay to avoid
100 // having all path-finding entities of the map compute a path at the same time
101 next_recomputation_date = System::now() + min_delay + Random::get_number(200);
102
103 set_path(path);
104 }
105}

107/**
108 * \brief Returns whether the movement is finished.
109 * \return always false because the movement is restarted as soon as the path is finished
110 * or an obstacle is reached
111 */
112bool PathFindingMovement::is_finished() const {
113 return false;
114}
