64/**
65 * \brief Sets the directory where the engine can write files.
66 *
67 * Initially, this directory is set to the preprocessor constant
68 * SOLARUS_WRITE_DIR (by default ".solarus").
69 * You normally don't need to change this, it should have been set correctly
70 * at compilation time to a value that depends on the target system.
71 *
72 * \param solarus_write_dir The directory where the engine can write files,
73 * relative to the base write directory.
74 */
75void set_solarus_write_dir(const std::string& solarus_write_dir) {
76
77 // This setting never changes at runtime.
78 // Allowing to change it would be complex and we don't need that.
79 Debug::check_assertion(solarus_write_dir_.empty(),
80 "The Solarus write directory is already set");
81
82 solarus_write_dir_ = solarus_write_dir;
83
84 // First check that we can write in a directory.
85 if (!PHYSFS_setWriteDir(get_base_write_dir().c_str())) {
86 Debug::die(std::string("Cannot write in user directory '")
87 + get_base_write_dir().c_str() + "': " + PHYSFS_getLastError());
88 }
89
90 // Create the directory.
91 PHYSFS_mkdir(solarus_write_dir.c_str());
92
93 const std::string& full_write_dir = get_base_write_dir() + "/" + solarus_write_dir;
94 if (!PHYSFS_setWriteDir(full_write_dir.c_str())) {
95 Debug::die(std::string("Cannot set Solarus write directory to '")
96 + full_write_dir + "': " + PHYSFS_getLastError());
97 }
98
99 // The quest subdirectory may be new, create it if needed.
100 if (!quest_write_dir_.empty()) {
101 set_quest_write_dir(quest_write_dir_);
102 }
103}

107/**
108 * \brief Opens a quest.
109 *
110 * Only one quest can be open at a time.
111 * If a quest is already open, it is automatically closed.
112 *
113 * \param program_name Program name passed as first command-line argument.
114 * Necessary for the quest file management system.
115 * \param quest_path Path of the quest to open.
116 * \return \c true if the quest was found.
117 */
118SOLARUS_API bool open_quest(const std::string& program_name, const std::string& quest_path) {
119
120 if (is_open()) {
121 close_quest();
122 }
123
124 if (program_name.empty()) {
125 PHYSFS_init(nullptr);
126 }
127 else {
128 PHYSFS_init(program_name.c_str());
129 }
130
131 quest_path_ = quest_path;
132 std::string dir_quest_path = quest_path + "/data";
133 std::string archive_quest_path_1 = quest_path + "/data.solarus";
134 std::string archive_quest_path_2 = quest_path + "/data.solarus.zip";
135
136 const std::string& base_dir = PHYSFS_getBaseDir();
137 PHYSFS_addToSearchPath(dir_quest_path.c_str(), 1); // data directory
138 PHYSFS_addToSearchPath(archive_quest_path_1.c_str(), 1); // data.solarus archive
139 PHYSFS_addToSearchPath(archive_quest_path_2.c_str(), 1); // data.solarus.zip archive
140 PHYSFS_addToSearchPath((base_dir + "/" + dir_quest_path).c_str(), 1);
141 PHYSFS_addToSearchPath((base_dir + "/" + archive_quest_path_1).c_str(), 1);
142 PHYSFS_addToSearchPath((base_dir + "/" + archive_quest_path_2).c_str(), 1);
143
144 // Set the engine root write directory.
145 set_solarus_write_dir(SOLARUS_WRITE_DIR);
146
147 if (!quest_exists()) {
148 return false;
149 }
150
151 // Set the quest write directory.
152 CurrentQuest::initialize();
153 set_quest_write_dir(CurrentQuest::get_properties().get_quest_write_dir());
154
155 return true;
156}

158/**
159 * \brief Closes the quest.
160 */
161SOLARUS_API void close_quest() {
162
163 if (!is_open()) {
164 return;
165 }
166
167 CurrentQuest::quit();
168
169 remove_temporary_files();
170
171 quest_path_ = "";
172 solarus_write_dir_ = "";
173 quest_write_dir_ = "";
174
175 PHYSFS_deinit();
176}

209/**
210 * \brief Returns the physical location of a data file.
211 *
212 * This function is not often necessary since the whole point of this class
213 * is to access files independently of their real file location
214 * (in the quest write directory, in the data directory or in the data archive).
215 *
216 * \param file_name The file to look for.
217 * \return Where it is actually located, or LOCATION_NONE if not found.
218 */
219SOLARUS_API DataFileLocation data_file_get_location(
220 const std::string& file_name) {
221
222 const char* path_ptr = PHYSFS_getRealDir(file_name.c_str());
223 std::string path = path_ptr == nullptr ? "" : path_ptr;
224 if (path.empty()) {
225 // File does not exist.
226 return DataFileLocation::LOCATION_NONE;
227 }
228
229 if (!get_quest_write_dir().empty() && path == PHYSFS_getWriteDir()) {
230 return DataFileLocation::LOCATION_WRITE_DIRECTORY;
231 }
232
233 if (path.rfind("data") == path.size() - 4) {
234 return DataFileLocation::LOCATION_DATA_DIRECTORY;
235 }
236
237 if (path.rfind("data.solarus") == path.size() - 12
238 || path.rfind("data.solarus.zip") == path.size() - 16) {
239 return DataFileLocation::LOCATION_DATA_ARCHIVE;
240 }
241
242 Debug::die(std::string("Unexpected search path element: " + path));
243}

245/**
246 * \brief Returns whether a file exists in the quest data directory or
247 * in Solarus write directory, and is not a directory.
248 * \param file_name A file name relative to the quest data directory,
249 * to the current language directory or to Solarus write directory.
250 * \param language_specific \c true if the file is relative to the current
251 * language directory.
252 * \return \c true if this file exists and is not a directory.
253 */
254SOLARUS_API bool data_file_exists(const std::string& file_name,
255 bool language_specific) {
256
257 std::string full_file_name;
258 if (language_specific) {
259 if (CurrentQuest::get_language().empty()) {
260 return false;
261 }
262 full_file_name = std::string("languages/") +
263 CurrentQuest::get_language() + "/" + file_name;
264 }
265 else {
266 full_file_name = file_name;
267 }
268
269 return PHYSFS_exists(full_file_name.c_str()) && !PHYSFS_isDirectory(full_file_name.c_str());
270}

272/**
273 * \brief Opens a data file an loads its content into memory.
274 * \param file_name Name of the file to open.
275 * \param language_specific \c true if the file is specific to the current language.
276 * \return The content of the file.
277 */
278SOLARUS_API std::string data_file_read(
279 const std::string& file_name,
280 bool language_specific
281) {
282 std::string full_file_name;
283 if (language_specific) {
284 Debug::check_assertion(!CurrentQuest::get_language().empty(),
285 std::string("Cannot open language-specific file '") + file_name
286 + "': no language was set"
287 );
288 full_file_name = std::string("languages/") +
289 CurrentQuest::get_language() + "/" + file_name;
290 }
291 else {
292 full_file_name = file_name;
293 }
294
295 // open the file
296 Debug::check_assertion(PHYSFS_exists(full_file_name.c_str()),
297 std::string("Data file '") + full_file_name + "' does not exist"
298 );
299 PHYSFS_file* file = PHYSFS_openRead(full_file_name.c_str());
300 Debug::check_assertion(file != nullptr,
301 std::string("Cannot open data file '") + full_file_name + "'"
302 );
303
304 // load it into memory
305 size_t size = static_cast<size_t>(PHYSFS_fileLength(file));
306 std::vector<char> buffer(size);
307
308 PHYSFS_read(file, buffer.data(), 1, (PHYSFS_uint32) size);
309 PHYSFS_close(file);
310
311 return std::string(buffer.data(), size);
312}

314/**
315 * \brief Saves a buffer into a data file.
316 * \param file_name Name of the file to write, relative to Solarus write directory.
317 * \param buffer The buffer to save.
318 *
319 */
320SOLARUS_API void data_file_save(
321 const std::string& file_name,
322 const std::string& buffer
323) {
324 // open the file to write
325 PHYSFS_file* file = PHYSFS_openWrite(file_name.c_str());
326 if (file == nullptr) {
327 Debug::die(std::string("Cannot open file '") + file_name
328 + "' for writing: " + PHYSFS_getLastError()
329 );
330 }
331
332 // save the memory buffer
333 if (PHYSFS_write(file, buffer.data(), (PHYSFS_uint32) buffer.size(), 1) == -1) {
334 Debug::die(std::string("Cannot write file '") + file_name + "': "
335 + PHYSFS_getLastError());
336 }
337 PHYSFS_close(file);
338}

370/**
371 * \brief Enumerate files of a directory.
372 *
373 * Symbolic links are never returned.
374 *
375 * \param dir_path Name of the directory to list, relative to the quest data
376 * directory.
377 * \param list_files Whether regular files should be included in the result.
378 * \param list_directories Whether directories should be included in the result.
379 * \return The files in this directory.
380 */
381SOLARUS_API std::vector<std::string> data_files_enumerate(
382 const std::string& dir_path,
383 bool list_files,
384 bool list_directories
385) {
386
387 std::vector<std::string> result;
388
389 if (PHYSFS_exists(dir_path.c_str())) {
390 char** files = PHYSFS_enumerateFiles(dir_path.c_str());
391
392 for (char** file = files; *file != nullptr; file++) {
393 bool is_directory = PHYSFS_isDirectory((dir_path + "/" + *file).c_str());
394
395 if (!PHYSFS_isSymbolicLink(*file)
396 && ((list_files && !is_directory)
397 || (list_directories && is_directory))) {
398 result.push_back(std::string(*file));
399 }
400 }
401
402 PHYSFS_freeList(files);
403 }
404
405 return result;
406}

427/**
428 * \brief Sets the subdirectory where files specific to the quest are
429 * saved, like savegames and configuration files.
430 *
431 * You have to call this function before loading or saving savegames and
432 * configuration files.
433 * This directory should typically be named like your quest, to be sure other
434 * quests will not interfere.
435 *
436 * \param quest_write_dir The quest write directory, relative to the Solarus
437 * write directory.
438 */
439SOLARUS_API void set_quest_write_dir(const std::string& quest_write_dir) {
440
441 if (!quest_write_dir_.empty()) {
442 // There was already a previous quest subdirectory: remove it from the
443 // search path.
444 PHYSFS_removeFromSearchPath(PHYSFS_getWriteDir());
445 }
446
447 quest_write_dir_ = quest_write_dir;
448
449 // Reset the write directory to the Solarus directory
450 // so that we can create the new quest subdirectory.
451 std::string full_write_dir = get_base_write_dir() + "/" + solarus_write_dir_;
452 if (!PHYSFS_setWriteDir(full_write_dir.c_str())) {
453 Debug::die(std::string("Cannot set Solarus write directory to '")
454 + full_write_dir + "': " + PHYSFS_getLastError());
455 }
456
457 if (!quest_write_dir.empty()) {
458 // Create the quest subdirectory (if not existing)
459 // in the Solarus write directory.
460 PHYSFS_mkdir(quest_write_dir.c_str());
461
462 // Set the write directory to this new place.
463 full_write_dir = get_base_write_dir() + "/" + solarus_write_dir_ + "/" + quest_write_dir;
464 PHYSFS_setWriteDir(full_write_dir.c_str());
465
466 // Also allow the quest to read savegames, settings and data files there.
467 PHYSFS_addToSearchPath(PHYSFS_getWriteDir(), 0);
468 }
469}

491/**
492 * \brief Creates a temporary file with the specified content and closes it.
493 * \param content Content of the file to create.
494 * \return Full name of the file created, or an empty string in case of failure.
495 */
496SOLARUS_API std::string create_temporary_file(const std::string& content) {
497
498 // Determine the name of our temporary file.
499 std::string file_name;
500
501#ifdef HAVE_MKSTEMP
502 // mkstemp+close is safer than tmpnam, but POSIX only.
503 char name_template[] = "/tmp/solarus.XXXXXX";
504 int file_descriptor = mkstemp(name_template);
505 if (file_descriptor == -1) {
506 // Failure.
507 return "";
508 }
509 close(file_descriptor);
510 file_name = name_template;
511#else
512 file_name = std::tmpnam(nullptr);
513#endif
514
515 std::ofstream out(file_name.c_str());
516
517 if (!out) {
518 return "";
519 }
520
521 // File successfully created.
522 temporary_files_.push_back(file_name);
523
524 if (!content.empty()) {
525 out.write(content.data(), content.size());
526 if (!out) {
527 file_name = "";
528 }
529 }
530
531 out.close();
532
533 return file_name;
534}

536/**
537 * \brief Deletes all files previously created with create_temporary_file().
538 * \return \c true in case of success, \c false if at least one file could not
539 * be deleted.
540 */
541SOLARUS_API bool remove_temporary_files() {
542
543 bool success = true;
544 for (const std::string& file_name: temporary_files_) {
545 if (std::remove(file_name.c_str()) != 0) {
546 success = false;
547 }
548 }
549 temporary_files_.clear();
550
551 return success;
552}
