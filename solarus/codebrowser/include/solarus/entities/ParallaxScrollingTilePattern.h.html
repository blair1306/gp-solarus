17#ifndef SOLARUS_PARALLAX_SCROLLING_TILE_PATTERN_H
18#define SOLARUS_PARALLAX_SCROLLING_TILE_PATTERN_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/SimpleTilePattern.h"
22
23namespace Solarus {
24
25/**
26 * \brief Tile pattern with a parallax scrolling effect.
27 *
28 * The pattern moves from 1 pixel when the camera moves from 2 pixels.
29 * This gives an illusion of depth. Only the position of where the pattern
30 * is displayed changes: the real position of the tile never changes.
31 * Therefore, placing such tiles on your map can be tricky because at runtime,
32 * they are displayed elsewhere (except in the particular case where the
33 * camera is in (0,0), the top-left corner of the map).
34 * Note that collisions properties remain applied to the
35 * real position of the tile and not to where they are displayed.
36 * Usually, parallax scrolling is used for decoration only, for example to make
37 * nice scrolling backgrounds.
38 */
39class ParallaxScrollingTilePattern: public SimpleTilePattern {
40
41 public:
42
43 ParallaxScrollingTilePattern(Ground ground, const Point& xy, const Size& size);
44
45 virtual void draw(
46 const SurfacePtr& dst_surface,
47 const Point& dst_position,
48 const Tileset& tileset,
49 const Point& viewport
50 ) const override;
51
52 virtual bool is_animated() const override;
53 virtual bool is_drawn_at_its_position() const override;
54
55 static constexpr int ratio = 2; /**< Distance made by the viewport to move the tile pattern of 1 pixel. */
56
57};
58
59}
60
61#endif
