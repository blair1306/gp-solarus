35/**
36 * \copydoc PixelFilter::filter
37 */
38void Scale2xFilter::filter(
39 const uint32_t* src,
40 int src_width,
41 int src_height,
42 uint32_t* dst) const {
43
44 const int dst_width = src_width * 2;
45
46 int e1 = 0;
47 int e2, e3, e4;
48 int b, d, e = 0, f, h;
49 for (int row = 0; row < src_height; row++) {
50 for (int col = 0; col < src_width; col++) {
51
52 // compute a to i
53
54 b = e - src_width;
55 d = e - 1;
56 f = e + 1;
57 h = e + src_width;
58
59 if (row == 0) {
60 b = e;
61 }
62 if (row == src_height - 1) {
63 h = e;
64 }
65 if (col == 0) {
66 d = e;
67 }
68 if (col == src_width - 1) {
69 f = e;
70 }
71
72 // compute e1 to e4
73 e2 = e1 + 1;
74 e3 = e1 + dst_width;
75 e4 = e3 + 1;
76
77 // compute the color
78
79 if (src[b] != src[h] && src[d] != src[f]) {
80 dst[e1] = src[(src[d] == src[b]) ? d : e];
81 dst[e2] = src[(src[b] == src[f]) ? f : e];
82 dst[e3] = src[(src[d] == src[h]) ? d : e];
83 dst[e4] = src[(src[h] == src[f]) ? f : e];
84 }
85 else {
86 dst[e1] = dst[e2] = dst[e3] = dst[e4] = src[e];
87 }
88 e1 += 2;
89 e++;
90 }
91 e1 += dst_width;
92 }
93
94}
