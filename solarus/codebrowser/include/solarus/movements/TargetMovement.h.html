17#ifndef SOLARUS_TARGET_MOVEMENT_H
18#define SOLARUS_TARGET_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityPtr.h"
22#include "solarus/lowlevel/Point.h"
23#include "solarus/movements/StraightMovement.h"
24#include <cstdint>
25#include <string>
26
27namespace Solarus {
28
29/**
30 * \brief Movement of an object that goes to a target point.
31 *
32 * The target point may be a fixed point or a moving entity.
33 */
34class TargetMovement: public StraightMovement {
35
36 public:
37
38 TargetMovement(
39 const EntityPtr& target_entity,
40 int x,
41 int y,
42 int moving_speed,
43 bool ignore_obstacles
44 );
45
46 TargetMovement(
47 const EntityPtr& target_entity,
48 const Point& xy,
49 int moving_speed,
50 bool ignore_obstacles
51 );
52
53 void set_target(
54 const EntityPtr& target_entity,
55 const Point& xy
56 );
57
58 int get_moving_speed() const;
59 void set_moving_speed(int moving_speed);
60
61 virtual void notify_object_controlled() override;
62 virtual bool is_finished() const override;
63 virtual void update() override;
64
65 virtual const std::string& get_lua_type_name() const override;
66
67 private:
68
69 void recompute_movement();
70
71 Point target; /**< Coordinates of the point or entity to track. */
72 EntityPtr target_entity; /**< The entity to track (nullptr if only
73 * a point is targeted) */
74 Point entity_offset; /**< Value to add to the entity's coordinates. */
75
76 int sign_x; /**< Sign of the x movement (1: right, -1: left) */
77 int sign_y; /**< Sign of the y movement (1: down, -1: up) */
78 int moving_speed; /**< Speed when moving */
79
80 uint32_t next_recomputation_date; /**< Date when the movement is recalculated. */
81 bool finished; /**< \c true if the target is reached. */
82
83};
84
85}
86
87#endif
