29/**
30 * Checks writing to text and reading again some existing entity data.
31 */
32void check_entity(const EntityData& entity) {
33
34 // Export the entity to a string and import it again.
35 std::string exported_entity_buffer;
36 bool success = entity.export_to_buffer(exported_entity_buffer);
37 Debug::check_assertion(success, "Entity export failed");
38
39 EntityData imported_entity;
40 success = imported_entity.import_from_buffer(exported_entity_buffer, "entity");
41 Debug::check_assertion(success, "Entity import failed");
42
43 Debug::check_assertion(imported_entity.get_type() == entity.get_type(), "Entity type differs");
44 Debug::check_assertion(imported_entity.get_name() == entity.get_name(), "Entity name differs");
45 Debug::check_assertion(imported_entity.get_layer() == entity.get_layer(), "Entity layer differs");
46 Debug::check_assertion(imported_entity.get_xy() == entity.get_xy(), "Entity xy differs");
47 Debug::check_assertion(imported_entity.get_fields() == entity.get_fields(), "Entity fields differ");
48
49}

51/**
52 * \brief Checks reading and writing a map data file.
53 */
54void check_map(TestEnvironment& /* env */, const std::string& map_id) {
55
56 MapData map_data;
57 bool success = false;
58
59 // Import the map data file.
60 std::string file_name = "maps/" + map_id + ".dat";
61 std::string imported_map_buffer = QuestFiles::data_file_read(file_name);
62 success = map_data.import_from_buffer(imported_map_buffer, file_name);
63 Debug::check_assertion(success, "Map import failed");
64
65 // Export it.
66 std::string exported_map_buffer;
67 success = map_data.export_to_buffer(exported_map_buffer);
68 Debug::check_assertion(success, "Map export failed");
69
70 // Check that the file is identical after import/export.
71 if (exported_map_buffer != imported_map_buffer) {
72 std::cerr << "Map '" << map_id << "' differs from the original one after export." << std::endl
73 << "*** Original map file:" << std::endl << imported_map_buffer << std::endl
74 << "*** Exported map file:" << std::endl << exported_map_buffer << std::endl;
75 Debug::die("Map '" + map_id + "': exported file differs from the original one");
76 }
77
78 // Then export and import every entity of the map.
79 for (int layer = map_data.get_min_layer(); layer <= map_data.get_max_layer(); ++layer) {
80 for (int j = 0; j < map_data.get_num_entities(layer); ++j) {
81 EntityIndex index = { layer, j };
82 const EntityData& entity = map_data.get_entity(index);
83 check_entity(entity);
84 }
85 }
86}
87
88}
89
90/**
91 * \brief Tests reading and writing map data files.
92 */
93int main(int argc, char** argv) {
94
95 TestEnvironment env(argc, argv);
96
97 const std::map<std::string, std::string>& map_elements =
98 CurrentQuest::get_resources().get_elements(ResourceType::MAP);
99 Debug::check_assertion(!map_elements.empty(), "No maps");
100 for (const auto& kvp : map_elements) {
101 const std::string& map_id = kvp.first;
102 check_map(env, map_id);
103 }
104
105 return 0;
106}
