74/**
75 * \brief Sets the entity to be controlled by this movement object.
76 *
77 * This entity can be nullptr if your movement applies to something else than a map entity.
78 * However, some subclasses of Movement may require a non nullptr entity because they
79 * implement movements that depend on the map content (e.g. to handle the collisions).
80 *
81 * \param entity The entity to control, or nullptr if the movement should not be
82 * attached to a map entity.
83 */
84void Movement::set_entity(Entity* entity) {
85
86 Debug::check_assertion(drawable == nullptr, "This movement is already assigned to a drawable");
87
88 this->entity = entity;
89
90 if (entity == nullptr) {
91 this->xy = { 0, 0 };
92 }
93 else {
94 this->xy = entity->get_xy();
95 notify_movement_changed();
96 }
97 notify_object_controlled();
98}

159/**
160 * \brief Returns the coordinates of the object controlled by this movement.
161 * \return The coordinates of the object controlled by this movement.
162 */
163Point Movement::get_xy() const {
164
165 if (entity != nullptr) {
166 // The object controlled is a map entity.
167 return entity->get_xy();
168 }
169
170 if (drawable != nullptr) {
171 // The object controlled is a drawable.
172 return drawable->get_xy();
173 }
174
175 // The object controlled is a point.
176 return xy;
177}

204/**
205 * \brief Sets the position of the object controlled by this movement.
206 * \param xy The new coordinates.
207 */
208void Movement::set_xy(const Point& xy) {
209
210 if (entity != nullptr) {
211 // The object controlled is a map entity.
212 entity->set_xy(xy);
213 }
214
215 else if (drawable != nullptr) {
216 // The object controlled is a drawable.
217 drawable->set_xy(xy);
218 }
219
220 // The object controlled is a point.
221 this->xy = xy;
222
223 notify_position_changed();
224 last_move_date = System::now();
225}

260/**
261 * \brief Notifies this movement that the coordinates controlled by it
262 * have just been changed.
263 */
264void Movement::notify_position_changed() {
265
266 LuaContext* lua_context = get_lua_context();
267 if (lua_context != nullptr && are_lua_notifications_enabled()) {
268 lua_context->movement_on_position_changed(*this, get_xy());
269 }
270
271 if (entity != nullptr) {
272 if (!entity->is_being_removed()) {
273 entity->notify_position_changed();
274 }
275 }
276}

310/**
311 * \brief Notifies this movement that it has just finished.
312 */
313void Movement::notify_movement_finished() {
314
315 LuaContext* lua_context = get_lua_context();
316 if (lua_context != nullptr && are_lua_notifications_enabled()) {
317 finished_callback_ref.clear_and_call("movement callback");
318 lua_context->movement_on_finished(*this);
319 }
320
321 if (entity != nullptr && !entity->is_being_removed()) {
322 entity->notify_movement_finished();
323 }
324}

374/**
375 * \brief Suspends or resumes the movement.
376 *
377 * This function is called by the entity to suspend or resume its movement.
378 *
379 * \param suspended true to suspend the movement, false to resume it
380 */
381void Movement::set_suspended(bool suspended) {
382
383 if (suspended != this->suspended) {
384 this->suspended = suspended;
385
386 uint32_t now = System::now();
387
388 if (suspended) {
389 // the movement is being suspended
390 when_suspended = now;
391 }
392 }
393}

406/**
407 * \brief Updates the position of the object controlled by this movement.
408 *
409 * This function is called repeatedly.
410 * You can redefine this function.
411 */
412void Movement::update() {
413
414 if (!finished && is_finished()) {
415 finished = true;
416 notify_movement_finished();
417 }
418 else if (finished && !is_finished()) {
419 finished = false;
420 }
421}

423/**
424 * \brief Returns whether the entity would collide with the map
425 * if it was moved a few pixels from its position.
426 *
427 * If the movement is not attached to an entity of a map,
428 * or if obstacles are ignored, false is always returned.
429 *
430 * \param dx x distance between the current position and the position to check
431 * \param dy y distance between the current position and the position to check
432 * \return true if the entity would overlap the map obstacles in this position
433 */
434bool Movement::test_collision_with_obstacles(int dx, int dy) const {
435
436 if (entity == nullptr || current_ignore_obstacles) {
437 return false;
438 }
439
440 Map& map = entity->get_map();
441
442 // place the collision box where we want to check the collisions
443 Rectangle collision_box = entity->get_bounding_box();
444 collision_box.add_xy(dx, dy);
445
446 bool collision = map.test_collision_with_obstacles(entity->get_layer(), collision_box, *entity);
447
448 if (collision) {
449 last_collision_box_on_obstacle = collision_box;
450 }
451
452 return collision;
453}
454
455/**
456 * \brief Returns whether the entity would collide with the map
457 * if it was moved a few pixels from its position.
458 *
459 * If the movement is not attached to an entity of a map,
460 * or if obstacles are ignored, false is always returned.
461 *
462 * \param dxy distance between the current position and the position to check
463 * \return true if the entity would overlap the map obstacles in this position
464 */
465bool Movement::test_collision_with_obstacles(const Point& dxy) const {
466 return test_collision_with_obstacles(dxy.x, dxy.y);
467}
