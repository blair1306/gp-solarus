23/**
24 * \brief Creates a follow movement.
25 * \param entity_followed the entity to follow
26 * \param x x coordinate of where this entity should be placed (relative to the entity followed)
27 * \param y y coordinate of where this entity should be placed (relative to the entity followed)
28 * \param ignore_obstacles true to make the movement ignore obstacles
29 */
30RelativeMovement::RelativeMovement(
31 const EntityPtr& entity_followed,
32 int x,
33 int y,
34 bool ignore_obstacles):
35 RelativeMovement(entity_followed, Point(x,y), ignore_obstacles) {
36}
37
38/**
39 * \brief Creates a relative movement.
40 * \param entity_followed the entity to follow
41 * \param entity_offset coordinate of where this entity should be placed (relative to the entity followed)
42 * \param ignore_obstacles true to make the movement ignore obstacles
43 */
44RelativeMovement::RelativeMovement(
45 const EntityPtr& entity_followed,
46 const Point& entity_offset,
47 bool ignore_obstacles):
48 Movement(ignore_obstacles),
49 entity_followed(entity_followed),
50 entity_offset(entity_offset),
51 finished(false) {
52}
53
54/**
55 * \brief Returns whether the movement is finished.
56 * \return true if there was a collision or the entity followed disappeared
57 */
58bool RelativeMovement::is_finished() const {
59 return finished;
60}
61
62/**
63 * \brief Updates the position.
64 */
65void RelativeMovement::update() {
66
67 if (entity_followed == nullptr) {
68 finished = true;
69 return;
70 }
71
72 if (entity_followed->is_being_removed()) {
73 finished = true;
74 entity_followed = nullptr;
75 }
76 else {
77
78 Point next = entity_followed->get_xy() + entity_offset;
79 Point dnext = next - get_xy();
80
81 if (!are_obstacles_ignored()) {
82
83 if (!finished && (dnext.x != 0 || dnext.y != 0)) {
84
85 if (!test_collision_with_obstacles(dnext)) {
86 set_xy(next);
87 }
88 else {
89 finished = true;
90 notify_obstacle_reached();
91 }
92 }
93 }
94 else {
95 set_xy(next);
96 }
97 }
98}
99
100/**
101 * \brief Returns the coordinates where an object controlled by this movement
102 * should be displayed.
103 * \return the coordinates to use to display the object controlled by this movement
104 */
105Point RelativeMovement::get_displayed_xy() const {
106
107 if (entity_followed == nullptr) {
108 return get_xy();
109 }
110
111 // if the followed entity is displayed at a different position than its real position,
112 // we apply the same difference when displaying this entity
113
114 const Point& followed_xy = entity_followed->get_xy();
115 const Point& followed_displayed_xy = entity_followed->get_displayed_xy();
116
117 Point dxy = followed_displayed_xy - followed_xy;
118
119 return get_xy() + dxy;
120}
