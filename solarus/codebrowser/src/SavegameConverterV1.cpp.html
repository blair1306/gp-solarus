27/**
28 * \brief Creates an old savegame converter with a specified existing file name.
29 * \param file_name Name of the savegame file (must exist),
30 * relative to the savegames directory of the quest.
31 */
32SavegameConverterV1::SavegameConverterV1(const std::string& file_name) {
33
34 Debug::check_assertion(QuestFiles::data_file_exists(file_name),
35 std::string("Cannot convert savegame '") + file_name
36 + "': file does not exist"
37 );
38
39 // Let's load this obsolete savegame.
40 const std::string& buffer = QuestFiles::data_file_read(file_name);
41 Debug::check_assertion(buffer.size() == sizeof(SavedData),
42 std::string("Cannot read savegame file version 1 '")
43 + file_name + "': invalid file size"
44 );
45 std::memcpy(&saved_data, buffer.data(), sizeof(SavedData));
46}
47
48/**
49 * \brief Returns a string value saved.
50 * \param index index of the value to get, between 0 and 63
51 * (see enum StringIndex for their definition).
52 * \return The string value saved at this index.
53 */
54std::string SavegameConverterV1::get_string(int index) {
55 return saved_data.strings[index];
56}
57
58/**
59 * \brief Returns a integer value saved.
60 *
61 * Values between 0 and 1023 are used by the engine (the C++ code)
62 * and values between 1024 and 2047 are available to the map scripts.
63 * \param index index of the value to get, between 0 and 2047
64 * (see enum IntegerIndex for their definition)
65 * \return the integer value saved at this index
66 */
67uint32_t SavegameConverterV1::get_integer(int index) {
68 return saved_data.integers[index];
69}
70
71/**
72 * \brief Returns a boolean value saved.
73 * \param index index of the value to get, between 0 and 32767
74 * \return the boolean value saved at this index
75 */
76bool SavegameConverterV1::get_boolean(int index) {
77
78 uint32_t word = saved_data.booleans[index / 32];
79 return ((word >> (index % 32)) & 0x0001) != 0x0000;
80}

82/**
83 * \brief Converts this savegame v1 into a savegame v2.
84 * \param savegame_v2 The savegame to fill.
85 */
86void SavegameConverterV1::convert_to_v2(Savegame& savegame_v2) {
87
88 // 1. Built-in values.
89 savegame_v2.set_string(Savegame::KEY_STARTING_POINT, get_string(STARTING_POINT));
90 if (!get_string(STARTING_MAP).empty()) {
91 savegame_v2.set_string(Savegame::KEY_STARTING_MAP, get_string(STARTING_MAP));
92 }
93 else {
94 // Older v1 savegames used integers to identify maps.
95 std::ostringstream oss;
96 oss << get_integer(STARTING_MAP_INT);
97 savegame_v2.set_string(Savegame::KEY_STARTING_MAP, oss.str());
98 }
99 savegame_v2.set_integer(Savegame::KEY_CURRENT_LIFE, get_integer(CURRENT_LIFE));
100 savegame_v2.set_integer(Savegame::KEY_CURRENT_MONEY, get_integer(CURRENT_MONEY));
101 savegame_v2.set_integer(Savegame::KEY_CURRENT_MAGIC, get_integer(CURRENT_MAGIC));
102 savegame_v2.set_integer(Savegame::KEY_MAX_LIFE, get_integer(MAX_LIFE));
103 savegame_v2.set_integer(Savegame::KEY_MAX_MONEY, get_integer(MAX_MONEY));
104 savegame_v2.set_integer(Savegame::KEY_MAX_MAGIC, get_integer(MAX_MAGIC));
105 savegame_v2.set_string(Savegame::KEY_ITEM_SLOT_1, get_string(ITEM_SLOT_0));
106 savegame_v2.set_string(Savegame::KEY_ITEM_SLOT_2, get_string(ITEM_SLOT_1));
107 savegame_v2.set_default_keyboard_controls(); // It would be unsafe to get old SDL_Keycode values.
108 savegame_v2.set_string(Savegame::KEY_JOYPAD_ACTION, get_string(JOYPAD_ACTION_KEY));
109 savegame_v2.set_string(Savegame::KEY_JOYPAD_ATTACK, get_string(JOYPAD_SWORD_KEY));
110 savegame_v2.set_string(Savegame::KEY_JOYPAD_ITEM_1, get_string(JOYPAD_ITEM_1_KEY));
111 savegame_v2.set_string(Savegame::KEY_JOYPAD_ITEM_2, get_string(JOYPAD_ITEM_2_KEY));
112 savegame_v2.set_string(Savegame::KEY_JOYPAD_PAUSE, get_string(JOYPAD_PAUSE_KEY));
113 savegame_v2.set_string(Savegame::KEY_JOYPAD_RIGHT, get_string(JOYPAD_RIGHT_KEY));
114 savegame_v2.set_string(Savegame::KEY_JOYPAD_UP, get_string(JOYPAD_UP_KEY));
115 savegame_v2.set_string(Savegame::KEY_JOYPAD_LEFT, get_string(JOYPAD_LEFT_KEY));
116 savegame_v2.set_string(Savegame::KEY_JOYPAD_DOWN, get_string(JOYPAD_DOWN_KEY));
117 savegame_v2.set_integer(Savegame::KEY_ABILITY_TUNIC, get_integer(ABILITY_TUNIC));
118 savegame_v2.set_integer(Savegame::KEY_ABILITY_SWORD, get_integer(ABILITY_SWORD));
119 savegame_v2.set_integer(Savegame::KEY_ABILITY_SHIELD, get_integer(ABILITY_SHIELD));
120 savegame_v2.set_integer(Savegame::KEY_ABILITY_LIFT, get_integer(ABILITY_LIFT));
121 savegame_v2.set_integer(Savegame::KEY_ABILITY_SWIM, get_integer(ABILITY_SWIM));
122 savegame_v2.set_integer(Savegame::KEY_ABILITY_SWORD_KNOWLEDGE, get_integer(ABILITY_SWORD_KNOWLEDGE));
123 savegame_v2.set_integer(Savegame::KEY_ABILITY_DETECT_WEAK_WALLS, get_integer(ABILITY_DETECT_WEAK_WALLS));
124 savegame_v2.set_integer(Savegame::KEY_ABILITY_GET_BACK_FROM_DEATH, get_integer(ABILITY_GET_BACK_FROM_DEATH));
125 savegame_v2.set_integer(Savegame::KEY_ABILITY_RUN, get_integer(ABILITY_RUN));
126
127 // 2. Values that used to be built-in in v1 and become pure data in v2.
128 savegame_v2.set_string("player_name", get_string(PLAYER_NAME));
129 savegame_v2.set_integer("pause_last_submenu", get_integer(PAUSE_LAST_SUBMENU) + 1);
130 savegame_v2.set_integer("pause_inventory_last_item_index", get_integer(INVENTORY_LAST_ITEM_INDEX) + 1);
131
132 for (int i = 0; i < 40; i++) {
133 int index = 200 + i * 10;
134
135 std::ostringstream oss;
136 oss << "dungeon_" << (i + 1);
137 const std::string& dungeon_number = oss.str();
138
139 // Dungeon finished (integer replaced by a boolean).
140 if (get_integer(index) > 0) {
141 savegame_v2.set_boolean(dungeon_number + "_finished", true);
142 }
143
144 // Got the map? (integer replaced by a boolean).
145 ++index;
146 if (get_integer(index) > 0) {
147 savegame_v2.set_boolean(dungeon_number + "_map", true);
148 }
149
150 // Got the compass? (integer replaced by a boolean).
151 ++index;
152 if (get_integer(index) > 0) {
153 savegame_v2.set_boolean(dungeon_number + "_compass", true);
154 }
155
156 // Got the big key? (integer replaced by a boolean).
157 ++index;
158 if (get_integer(index) > 0) {
159 savegame_v2.set_boolean(dungeon_number + "_big_key", true);
160 }
161
162 // Got the boss key? (integer replaced by a boolean).
163 ++index;
164 if (get_integer(index) > 0) {
165 savegame_v2.set_boolean(dungeon_number + "_boss_key", true);
166 }
167
168 // Number of small keys.
169 ++index;
170 int nb_small_keys = get_integer(index);
171 if (nb_small_keys > 0) {
172 savegame_v2.set_integer(dungeon_number + "_small_keys", nb_small_keys);
173 }
174 }
175
176 // 3. Custom values.
177 int i;
178 for (i = 32; i < 64; i++) {
179 const std::string& value = get_string(i);
180 if (!value.empty()) {
181 std::ostringstream oss;
182 oss << "s" << i;
183 savegame_v2.set_string(oss.str(), value);
184 }
185 }
186
187 for (i = 1024; i < 2048; i++) {
188 int value = get_integer(i);
189 if (value != 0) {
190 std::ostringstream oss;
191 oss << "i" << i;
192 savegame_v2.set_integer(oss.str(), value);
193 }
194 }
195
196 for (i = 0; i < 32768; i++) {
197 bool value = get_boolean(i);
198 if (value) {
199 std::ostringstream oss;
200 oss << "b" << i;
201 savegame_v2.set_boolean(oss.str(), value);
202 }
203 }
204}
