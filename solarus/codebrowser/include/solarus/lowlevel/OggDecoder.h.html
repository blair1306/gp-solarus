17#ifndef SOLARUS_OGG_DECODER_H
18#define SOLARUS_OGG_DECODER_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Sound.h"
22#include <memory>
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief Encapsulates the Ogg Vorbis music decoding.
29 */
30class OggDecoder {
31
32 public:
33
34 OggDecoder();
35
36 bool load(std::string&& ogg_data, bool loop);
37 void unload();
38 void decode(ALuint destination_buffer, ALsizei nb_samples);
39
40 private:
41
42 struct OggFileDeleter {
43 void operator()(OggVorbis_File* ogg_file) {
44 ov_clear(ogg_file);
45 delete ogg_file;
46 }
47 };
48 using OggFileUniquePtr = std::unique_ptr<OggVorbis_File, OggFileDeleter>;
49
50 OggFileUniquePtr ogg_file; /**< The file used by the vorbisfile lib. */
51 Sound::SoundFromMemory ogg_mem; /**< The encoded music loaded in memory,
52 * passed to the vorbisfile lib as user data. */
53 vorbis_info* ogg_info; /**< Info about the OGG file. */
54 ogg_int64_t loop_start_pcm; /**< Where to loop to in PCM samples.
55 * -1 means no loop. */
56 ogg_int64_t loop_end_pcm; /**< Where to loop from in PCM samples.
57 * -1 means no loop. */
58
59};
60
61}
62
63#endif
