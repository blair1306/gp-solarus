56class QLayout;
57class QSize;
58
59
60class QLayoutPrivate;
61
62class Q_WIDGETS_EXPORT QLayout : public QObject, public QLayoutItem
63{
64 Q_OBJECT
65 Q_DECLARE_PRIVATE(QLayout)
66
67 Q_ENUMS(SizeConstraint)
68 Q_PROPERTY(int margin READ margin WRITE setMargin)
69 Q_PROPERTY(int spacing READ spacing WRITE setSpacing)
70 Q_PROPERTY(SizeConstraint sizeConstraint READ sizeConstraint WRITE setSizeConstraint)
71public:
72 enum SizeConstraint {
73 SetDefaultConstraint,
74 SetNoConstraint,
75 SetMinimumSize,
76 SetFixedSize,
77 SetMaximumSize,
78 SetMinAndMaxSize
79 };
80
81 QLayout(QWidget *parent);
82 QLayout();
83 ~QLayout();
84
85 int margin() const;
86 int spacing() const;
87
88 void setMargin(int);
89 void setSpacing(int);
90
91 void setContentsMargins(int left, int top, int right, int bottom);
92 void setContentsMargins(const QMargins &margins);
93 void getContentsMargins(int *left, int *top, int *right, int *bottom) const;
94 QMargins contentsMargins() const;
95 QRect contentsRect() const;
96
97 bool setAlignment(QWidget *w, Qt::Alignment alignment);
98 bool setAlignment(QLayout *l, Qt::Alignment alignment);
99#ifdef Q_NO_USING_KEYWORD
100 inline void setAlignment(Qt::Alignment alignment) { QLayoutItem::setAlignment(alignment); }
101#else
102 using QLayoutItem::setAlignment;
103#endif
104
105 void setSizeConstraint(SizeConstraint);
106 SizeConstraint sizeConstraint() const;
107 void setMenuBar(QWidget *w);
108 QWidget *menuBar() const;
109
110 QWidget *parentWidget() const;
111
112 void invalidate();
113 QRect geometry() const;
114 bool activate();
115 void update();
116
117 void addWidget(QWidget *w);
118 virtual void addItem(QLayoutItem *) = 0;
119
120 void removeWidget(QWidget *w);
121 void removeItem(QLayoutItem *);
122
123 Qt::Orientations expandingDirections() const;
124 QSize minimumSize() const;
125 QSize maximumSize() const;
126 virtual void setGeometry(const QRect&);
127 virtual QLayoutItem *itemAt(int index) const = 0;
128 virtual QLayoutItem *takeAt(int index) = 0;
129 virtual int indexOf(QWidget *) const;
130 virtual int count() const = 0;
131 bool isEmpty() const;
132 QSizePolicy::ControlTypes controlTypes() const;
133
134 // ### Qt 6 make this function virtual
135 QLayoutItem *replaceWidget(QWidget *from, QWidget *to, Qt::FindChildOptions options = Qt::FindChildrenRecursively);
136
137 int totalHeightForWidth(int w) const;
138 QSize totalMinimumSize() const;
139 QSize totalMaximumSize() const;
140 QSize totalSizeHint() const;
141 QLayout *layout();
142
143 void setEnabled(bool);
144 bool isEnabled() const;
145
146
147 static QSize closestAcceptableSize(const QWidget *w, const QSize &s);
148
149protected:
150 void widgetEvent(QEvent *);
151 void childEvent(QChildEvent *e);
152 void addChildLayout(QLayout *l);
153 void addChildWidget(QWidget *w);
154 bool adoptLayout(QLayout *layout);
155
156 QRect alignmentRect(const QRect&) const;
157protected:
158 QLayout(QLayoutPrivate &d, QLayout*, QWidget*);
159
160private:
161 Q_DISABLE_COPY(QLayout)
162
163 static void activateRecursiveHelper(QLayoutItem *item);
164
165 friend class QApplicationPrivate;
166 friend class QWidget;
167
168};
