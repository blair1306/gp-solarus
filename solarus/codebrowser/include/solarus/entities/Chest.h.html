17#ifndef SOLARUS_CHEST_H
18#define SOLARUS_CHEST_H
19
20#include "solarus/Common.h"
21#include "solarus/Treasure.h"
22#include "solarus/entities/Entity.h"
23#include <map>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief Represents a chest on a map.
30 *
31 * It can be a normal chest or a big chest.
32 */
33class Chest: public Entity {
34
35 public:
36
37 static constexpr EntityType ThisType = EntityType::CHEST;
38
39 /**
40 * \brief The different possible ways of opening a chest.
41 */
42 enum class OpeningMethod {
43 BY_INTERACTION, /**< Can be opened by pressing the action command. */
44 BY_INTERACTION_IF_SAVEGAME_VARIABLE, /**< Can be opened by pressing the action command, provided that a savegame variable has the correct value. */
45 BY_INTERACTION_IF_ITEM, /**< Can be opened by pressing the action command, provided that the player has a specific equipment item. */
46 };
47
48 Chest(
49 const std::string& name,
50 int layer,
51 const Point& xy,
52 const std::string& sprite_name,
53 const Treasure& treasure);
54
55 virtual EntityType get_type() const override;
56
57 const Treasure& get_treasure() const;
58 void set_treasure(const Treasure& treasure);
59
60 bool is_open() const;
61 void set_open(bool open);
62 bool can_open();
63
64 OpeningMethod get_opening_method() const;
65 void set_opening_method(OpeningMethod opening_method);
66 const std::string& get_opening_condition() const;
67 void set_opening_condition(const std::string& opening_condition);
68 bool is_opening_condition_consumed() const;
69 void set_opening_condition_consumed(bool opening_condition_consumed);
70 const std::string& get_cannot_open_dialog_id() const;
71 void set_cannot_open_dialog_id(const std::string& cannot_open_dialog_id);
72
73 virtual bool is_obstacle_for(Entity& other) override;
74 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
75 virtual void notify_enabled(bool enabled) override;
76 virtual void update() override;
77 virtual bool notify_action_command_pressed() override;
78 virtual void set_suspended(bool suspended) override;
79
80 static const std::map<OpeningMethod, std::string> opening_method_names;
81
82 private:
83
84 Treasure treasure; /**< The treasure placed in this chest (possibly empty, possibly non-obtainable).
85 * Remains unchanged when the chest is open. */
86 bool open; /**< whether the chest is open (but the treasure may not have
87 * been given yet because there is a delay of 500 ms). */
88 bool treasure_given; /**< whether the chest is open and the treasure has been given to the player. */
89 uint32_t treasure_date; /**< Date when the treasure will be given to the player. */
90
91 OpeningMethod opening_method; /**< How this chest can be opened. */
92 std::string opening_condition; /**< Condition required to open the chest: a savegame variable if
93 * the opening mode is \c OPENING_BY_INTERACTION_IF_SAVEGAME_VARIABLE,
94 * or an equipment item name if the opening mode is
95 * \c OPENING_BY_INTERACTION_IF_ITEM. */
96 bool opening_condition_consumed; /**< Indicates that the required savegame variable
97 * (in the case of \c OPENING_BY_INTERACTION_IF_SAVEGAME_VARIABLE)
98 * or the required item
99 * (in the case of \c OPENING_BY_INTERACTION_IF_ITEM)
100 * should be consumed when opening the chesty. */
101 std::string cannot_open_dialog_id; /**< Dialog to show if the chesty cannot be opened,
102 * or an empty string. */
103
104};
105
106}
107
108#endif
