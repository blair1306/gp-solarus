33namespace Solarus {
34
35class CommandsEffects;
36class Equipment;
37class GameCommands;
38class InputEvent;
39class LuaContext;
40class MainLoop;
41class Map;
42class ResourceProvider;
43class Savegame;
44
45/**
46 * \brief Represents the game currently running.
47 *
48 * The game shows the current map and handles all game elements.
49 */
50class SOLARUS_API Game {
51
52 public:
53
54 // creation
55 Game(MainLoop& main_loop, const std::shared_ptr<Savegame>& savegame);
56
57 // no copy operations
58 Game(const Game& game) = delete;
59 Game& operator=(const Game& game) = delete;
60
61 void start();
62 void stop();
63 void restart();
64
65 // global objects
66 MainLoop& get_main_loop();
67 LuaContext& get_lua_context();
68 ResourceProvider& get_resource_provider();
69 const HeroPtr& get_hero();
70 GameCommands& get_commands();
71 const GameCommands& get_commands() const;
72 CommandsEffects& get_commands_effects();
73 Savegame& get_savegame();
74 const Savegame& get_savegame() const;
75 Equipment& get_equipment();
76 const Equipment& get_equipment() const;
77
78 // functions called by the main loop
79 bool notify_input(const InputEvent& event);
80 void update();
81 void draw(const SurfacePtr& dst_surface);
82
83 // game controls
84 void notify_command_pressed(GameCommand command);
85 void notify_command_released(GameCommand command);
86
87 // simulate commands
88 void simulate_command_pressed(GameCommand command);
89 void simulate_command_released(GameCommand command);
90
91 // map
92 bool has_current_map() const;
93 Map& get_current_map();
94 void set_current_map(const std::string& map_id, const std::string& destination_name,
95 Transition::Style transition_style);
96
97 // world
98 bool get_crystal_state() const;
99 void change_crystal_state();
100
101 // current game state
102 bool is_paused() const;
103 bool is_dialog_enabled() const;
104 bool is_playing_transition() const;
105 bool is_showing_game_over() const;
106 bool is_suspended_by_camera() const;
107 bool is_suspended_by_script() const;
108 bool is_suspended() const; // true if at least one of the 5 functions above returns true
109
110 // pause
111 bool can_pause() const;
112 bool can_unpause() const;
113 bool is_pause_allowed() const;
114 void set_pause_allowed(bool pause_allowed);
115 void set_paused(bool paused);
116
117 // dialogs
118 void start_dialog(
119 const std::string& dialog_id,
120 const ScopedLuaRef& info_ref,
121 const ScopedLuaRef& callback_ref
122 );
123 void stop_dialog(const ScopedLuaRef& status_ref);
124
125 // game over
126 void start_game_over();
127 void stop_game_over();
128
129 // Suspend manually.
130 void set_suspended_by_script(bool suspended);
131
132private:
133
134 // main objects
135 MainLoop& main_loop; /**< the main loop object */
136 std::shared_ptr<Savegame>
137 savegame; /**< the game data saved */
138 HeroPtr hero; /**< The hero entity. */
139
140 // current game state (elements currently shown)
141 bool pause_allowed; /**< indicates that the player is allowed to use the pause command */
142 bool paused; /**< indicates that the game is paused */
143 DialogBoxSystem
144 dialog_box; /**< The dialog box manager. */
145 bool showing_game_over; /**< Whether a game-over sequence is currently active. */
146 bool suspended_by_script; /**< Whether the game is manually suspended. */
147 bool started; /**< true if this game is running, false if it is not yet started or being closed. */
148 bool restarting; /**< true if the game will be restarted */
149
150 // controls
151 std::unique_ptr<GameCommands>
152 commands; /**< this object receives the keyboard and joypad events */
153 CommandsEffects
154 commands_effects; /**< current effect associated to the main game keys
155 * (represented on the HUD by the action icon, the objects icons, etc.) */
156
157 // map
158 std::shared_ptr<Map>
159 current_map; /**< the map currently displayed */
160 std::shared_ptr<Map>
161 next_map; /**< the map where the hero is going to; if not nullptr, it means that the hero
162 * is changing from current_map to next_map */
163 SurfacePtr
164 previous_map_surface; /**< a copy of the previous map surface for transition effects that display two maps */
165
166 Transition::Style
167 transition_style; /**< the transition style between the current map and the next one */
168 std::unique_ptr<Transition>
169 transition; /**< the transition currently shown, or nullptr if no transition is playing */
170
171 // world (i.e. the current set of maps)
172 bool crystal_state; /**< indicates that a crystal has been enabled (i.e. the orange blocks are raised) */
173
174 // update functions
175 void update_commands_effects();
176 void update_transitions();
177 void update_gameover_sequence();
178 void notify_map_changed();
179
180};
181
182}
183
184#endif
