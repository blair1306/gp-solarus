36/**
37 * \brief Creates an NPC.
38 * \param game the game
39 * \param name name identifying this NPC
40 * \param layer layer of the entity to create
41 * \param xy Coordinates of the entity to create.
42 * \param subtype the subtype of interaction
43 * \param sprite_name sprite animation set of the entity, or an empty string to create no sprite
44 * \param direction for a generalized NPC: direction for which the interactions are allowed
45 * (0 to 4, or -1 for any direction), for a usual NPC: initial direction of the NPC's sprite
46 * \param behavior_string indicates what happens when the hero interacts with this NPC:
47 * "dialog#XXX" to start the dialog XXX, "map" to call the map script
48 * (with an event_hero_interaction() call) or "item#XXX" to call the script
49 * of item XXX (with an event_hero_interaction() call)
50 */
51Npc::Npc(
52 Game& /* game */,
53 const std::string& name,
54 int layer,
55 const Point& xy,
56 Subtype subtype,
57 const std::string& sprite_name,
58 int direction,
59 const std::string& behavior_string
60):
61 Entity(name, 0, layer, xy, Size(0, 0)),
62 subtype(subtype),
63 behavior(BEHAVIOR_MAP_SCRIPT),
64 traversable(false),
65 dialog_to_show(""),
66 item_name("") {
67
68 set_collision_modes(CollisionMode::COLLISION_FACING | CollisionMode::COLLISION_OVERLAPPING);
69 initialize_sprite(sprite_name, direction);
70 set_size(16, 16);
71 set_origin(8, 13);
72 set_direction(direction);
73
74 // Usual NPCs are displayed like the hero whereas generalized NPCs are
75 // not necessarily people.
76 set_drawn_in_y_order(subtype == USUAL_NPC);
77
78 // behavior
79 if (behavior_string == "map") {
80 behavior = BEHAVIOR_MAP_SCRIPT;
81 }
82 else if (behavior_string.substr(0, 5) == "item#") {
83 behavior = BEHAVIOR_ITEM_SCRIPT;
84 item_name = behavior_string.substr(5);
85 }
86 else if (behavior_string.substr(0, 7) == "dialog#") {
87 behavior = BEHAVIOR_DIALOG;
88 dialog_to_show = behavior_string.substr(7);
89 }
90 else {
91 Debug::die(std::string("Invalid behavior string for NPC '") + name
92 + "': '" + behavior_string + "'");
93 }
94}

104/**
105 * \brief Creates the sprite specified.
106 * \param sprite_name sprite animation set of the entity, or an empty string to create no sprite
107 * \param initial_direction direction of the entity's sprite (ignored if there is no sprite
108 * of if the direction specified is -1)
109 */
110void Npc::initialize_sprite(const std::string& sprite_name, int initial_direction) {
111
112 if (!sprite_name.empty()) {
113 const SpritePtr& sprite = create_sprite(sprite_name);
114 if (initial_direction != -1) {
115 sprite->set_current_direction(initial_direction);
116 }
117 }
118}

210/**
211 * \brief This function is called by the engine when there is a collision with another entity.
212 *
213 * If the entity is the hero, we allow him to interact with this entity.
214 *
215 * \param entity_overlapping the entity overlapping the detector
216 * \param collision_mode the collision mode that detected the collision
217 */
218void Npc::notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) {
219
220 if (collision_mode == COLLISION_FACING && entity_overlapping.is_hero()) {
221
222 Hero& hero = static_cast<Hero&>(entity_overlapping);
223
224 if (get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_NONE
225 && hero.is_free()) {
226
227 if (subtype == USUAL_NPC // the hero can talk to usual NPCs from any direction
228 || get_direction() == -1
229 || hero.is_facing_direction4((get_direction() + 2) % 4)) {
230
231 // show the appropriate action icon
232 get_commands_effects().set_action_key_effect(subtype == USUAL_NPC ?
233 CommandsEffects::ACTION_KEY_SPEAK : CommandsEffects::ACTION_KEY_LOOK);
234 }
235 else if (can_be_lifted() && get_equipment().has_ability(Ability::LIFT)) {
236 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_LIFT);
237 }
238 }
239 }
240 else if (collision_mode == COLLISION_OVERLAPPING && entity_overlapping.get_type() == EntityType::FIRE) {
241
242 if (behavior == BEHAVIOR_ITEM_SCRIPT) {
243 EquipmentItem& item = get_equipment().get_item(item_name);
244 get_lua_context()->item_on_npc_collision_fire(item, *this);
245 }
246 else {
247 get_lua_context()->npc_on_collision_fire(*this);
248 }
249 }
250}

252/**
253 * \copydoc Entity::notify_action_command_pressed
254 */
255bool Npc::notify_action_command_pressed() {
256
257 Hero& hero = get_hero();
258 if (hero.is_free() &&
259 get_commands_effects().get_action_key_effect() != CommandsEffects::ACTION_KEY_NONE
260 ) {
261
262 CommandsEffects::ActionKeyEffect effect = get_commands_effects().get_action_key_effect();
263 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
264
265 SpritePtr sprite = get_sprite();
266
267 // if this is a usual NPC, look towards the hero
268 if (subtype == USUAL_NPC) {
269 int direction = (get_hero().get_animation_direction() + 2) % 4;
270 if (sprite != nullptr) {
271 sprite->set_current_direction(direction);
272 }
273 }
274
275 if (effect != CommandsEffects::ACTION_KEY_LIFT) {
276 // start the normal behavior
277 if (behavior == BEHAVIOR_DIALOG) {
278 get_game().start_dialog(dialog_to_show, ScopedLuaRef(), ScopedLuaRef());
279 }
280 else {
281 call_script_hero_interaction();
282 }
283 return true;
284 }
285 else {
286 // lift the entity
287 if (get_equipment().has_ability(Ability::LIFT)) {
288
289 std::string animation_set_id = "stopped";
290 if (sprite != nullptr) {
291 animation_set_id = sprite->get_animation_set_id();
292 }
293 hero.start_lifting(std::make_shared<CarriedObject>(
294 hero,
295 *this,
296 animation_set_id,
297 "stone",
298 2,
299 0)
300 );
301 Sound::play("lift");
302 remove_from_map();
303 return true;
304 }
305 }
306 }
307 return false;
308}

310/**
311 * \brief Notifies the appropriate script that the hero is interacting with this entity.
312 */
313void Npc::call_script_hero_interaction() {
314
315 if (behavior == BEHAVIOR_MAP_SCRIPT) {
316 get_lua_context()->entity_on_interaction(*this);
317 }
318 else {
319 EquipmentItem& item = get_equipment().get_item(item_name);
320 get_lua_context()->item_on_npc_interaction(item, *this);
321 }
322}

324/**
325 * \brief Notifies this detector that the player is interacting by using an
326 * equipment item.
327 *
328 * This function is called when the player uses an equipment item
329 * while the hero is facing this NPC.
330 *
331 * \param item_used The equipment item used.
332 * \return true if an interaction occured.
333 */
334bool Npc::notify_interaction_with_item(EquipmentItem& item_used) {
335
336 bool interaction_occured;
337 if (behavior == BEHAVIOR_ITEM_SCRIPT) {
338 EquipmentItem& item_to_notify = get_equipment().get_item(item_name);
339 interaction_occured = get_lua_context()->item_on_npc_interaction_item(
340 item_to_notify, *this, item_used
341 );
342 }
343 else {
344 interaction_occured = get_lua_context()->entity_on_interaction_item(
345 *this, item_used
346 );
347 }
348
349 return interaction_occured;
350}

352/**
353 * \brief This function is called when the entity has just moved.
354 *
355 * If it is an NPC, its sprite's direction is updated.
356 */
357void Npc::notify_position_changed() {
358
359 Entity::notify_position_changed();
360
361 if (subtype == USUAL_NPC) {
362
363 const SpritePtr& sprite = get_sprite();
364 if (get_movement() != nullptr) {
365 // The NPC is moving.
366 if (sprite != nullptr) {
367 if (sprite->get_current_animation() != "walking") {
368 sprite->set_current_animation("walking");
369 }
370 int direction4 = get_movement()->get_displayed_direction4();
371 sprite->set_current_direction(direction4);
372 }
373 }
374
375 if (get_hero().get_facing_entity() == this &&
376 get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_SPEAK &&
377 !get_hero().is_facing_point_in(get_bounding_box())) {
378
379 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
380 }
381 }
382}

384/**
385 * \brief This function is called when the movement of the entity is finished.
386 */
387void Npc::notify_movement_finished() {
388
389 Entity::notify_movement_finished();
390
391 if (subtype == USUAL_NPC) {
392 const SpritePtr& sprite = get_sprite();
393 if (sprite != nullptr) {
394 sprite->set_current_animation("stopped");
395 }
396 }

400/**
401 * \brief Returns whether this interactive entity can be lifted.
402 */
403bool Npc::can_be_lifted() const {
404
405 // there is currently no way to specify from the data file of the map
406 // that an NPC can be lifted (nor its weight, damage, sound, etc) so this is hardcoded
407 // TODO: specify the possibility to lift and the weight from Lua
408 // npc:set_weight()?
409
410 const SpritePtr& sprite = get_sprite();
411 return sprite != nullptr && sprite->get_animation_set_id() == "entities/sign";
412}
