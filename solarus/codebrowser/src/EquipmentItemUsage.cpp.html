17#include "solarus/EquipmentItemUsage.h"
18#include "solarus/Game.h"
19#include "solarus/Equipment.h"
20#include "solarus/EquipmentItem.h"

24namespace Solarus {
25
26/**
27 * \brief Creates a new item to be used.
28 * \param game The game.
29 * \param item The item to use.
30 */
31EquipmentItemUsage::EquipmentItemUsage(Game& /* game */, EquipmentItem& item):
32 item(item),
33 variant(item.get_variant()),
34 finished(true) {
35
36}
37
38/**
39 * \brief Returns the equipment item being used.
40 */
41EquipmentItem& EquipmentItemUsage::get_item() {
42 return item;
43}
44
45/**
46 * \brief Starts using this item.
47 */
48void EquipmentItemUsage::start() {
49
50 Debug::check_assertion(variant > 0,
51 std::string("Attempt to use equipment item '") + item.get_name()
52 + "' without having it");
53
54 this->finished = false;
55 item.notify_using();
56}
57
58/**
59 * \brief Updates this item when it is being used.
60 */
61void EquipmentItemUsage::update() {
62}
63
64/**
65 * \brief Returns whether this item has finished to be used.
66 * \return true if this item has finished to be used
67 */
68bool EquipmentItemUsage::is_finished() {
69 return finished;
70}
71
72/**
73 * \brief Indicates that the player has finished using this item.
74 */
75void EquipmentItemUsage::set_finished() {
76 this->finished = true;
77}
