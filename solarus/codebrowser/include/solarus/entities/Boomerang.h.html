17#ifndef SOLARUS_BOOMERANG_H
18#define SOLARUS_BOOMERANG_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/entities/HeroPtr.h"
23#include <memory>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief A boomerang thrown on the map.
30 *
31 * In the current implementation, only the hero is supposed to be able to
32 * throw a boomerang.
33 */
34class Boomerang: public Entity {
35
36 public:
37
38 static constexpr EntityType ThisType = EntityType::BOOMERANG;
39
40 Boomerang(
41 const HeroPtr& hero,
42 int max_distance,
43 int speed,
44 double angle,
45 const std::string& sprite_name
46 );
47
48 EntityType get_type() const override;
49 virtual bool can_be_drawn() const override;
50
51 // obstacles
52 virtual bool is_teletransporter_obstacle(Teletransporter& teletransporter) override;
53 virtual bool is_stream_obstacle(Stream& stream) override;
54 virtual bool is_stairs_obstacle(Stairs& stairs) override;
55 virtual bool is_low_wall_obstacle() const override;
56 virtual bool is_deep_water_obstacle() const override;
57 virtual bool is_hole_obstacle() const override;
58 virtual bool is_lava_obstacle() const override;
59 virtual bool is_prickle_obstacle() const override;
60 virtual bool is_ladder_obstacle() const override;
61 virtual bool is_switch_obstacle(Switch& sw) override;
62 virtual bool is_raised_block_obstacle(CrystalBlock& raised_block) override;
63 virtual bool is_crystal_obstacle(Crystal& crystal) override;
64 virtual bool is_npc_obstacle(Npc& npc) override;
65 virtual bool is_jumper_obstacle(Jumper& jumper, const Rectangle& candidate_position) override;
66
67 // state
68 bool is_going_back() const;
69 void go_back();
70 virtual void update() override;
71
72 // movement
73 virtual void notify_obstacle_reached() override;
74 virtual void notify_movement_finished() override;
75
76 // collisions
77 virtual void notify_collision_with_switch(Switch& sw, CollisionMode collision_mode) override;
78 virtual void notify_collision_with_crystal(Crystal& crystal, CollisionMode collision_mode) override;
79 virtual void notify_collision_with_enemy(Enemy& enemy) override;
80 virtual void notify_attacked_enemy(
81 EnemyAttack attack,
82 Enemy& victim,
83 const Sprite* victim_sprite,
84 EnemyReaction::Reaction& result,
85 bool killed
86 ) override;
87
88 private:
89
90 HeroPtr hero; /**< The hero who shot this boomerang. */
91
92 uint32_t next_sound_date; /**< date when the boomerang sound is played next time */
93
94 bool has_to_go_back; /**< true if the boomerang is about to go back */
95 bool going_back; /**< indicates that the boomerang is going back towards the hero */
96
97 int speed; /**< speed of the movement in pixels per second */
98
99};
100
101}

103#endif
