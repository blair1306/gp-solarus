17#ifndef SOLARUS_NPC_H
18#define SOLARUS_NPC_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/CommandsEffects.h"
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief A non-playing character or a more general interactive entity.
29 *
30 * The entity triggers a message or an event
31 * when the player presses the action key while facing it.
32 *
33 * This class can be used for usual living NPCs who the hero can talk to,
34 * but also for more general non-living, interactive objects that show a
35 * dialog or trigger a script, like signs or torches.
36 * Such customizable interactive objects are called generalized NPCs.
37 *
38 * A usual NPC has some specific collision and displaying properties.
39 * It must have a sprite with four directions, an animation "stopped",
40 * and an animation "walking" if the script starts a movement.
41 * This class handles automatically the sprite's direction and animations
42 * depending on the movement and the interactions with the hero.
43 *
44 * On the contrary, a generalized NPC is just assumed to be a solid
45 * interactive entity, and the map script has to handle explicitly its
46 * animations (if any).
47 */
48class SOLARUS_API Npc: public Entity {
49
50 public:
51
52 static constexpr EntityType ThisType = EntityType::NPC;
53
54 /**
55 * \brief The different kinds of interactions.
56 */
57 enum Subtype {
58 GENERALIZED_NPC, /**< a custom interactive entity (not necessarily a person) */
59 USUAL_NPC /**< a usual NPC who the player can talk to */
60 };
61
62 /**
63 * \brief Type of action done when an interaction occurs.
64 */
65 enum Behavior {
66 BEHAVIOR_DIALOG, /**< start a dialog */
67 BEHAVIOR_MAP_SCRIPT, /**< notify the NPC in Lua */
68 BEHAVIOR_ITEM_SCRIPT /**< notify an equipment item in Lua */
69 };
70
71 Npc(
72 Game& game,
73 const std::string& name,
74 int layer,
75 const Point& xy,
76 Subtype subtype,
77 const std::string& sprite_name,
78 int initial_direction,
79 const std::string& behavior_string
80 );
81
82 EntityType get_type() const override;
83
84 bool is_solid() const;
85 bool is_traversable() const;
86 void set_traversable(bool traversable);
87
88 virtual bool is_obstacle_for(Entity& other) override;
89 virtual bool is_hero_obstacle(Hero& hero) override;
90 virtual bool is_npc_obstacle(Npc& npc) override;
91 virtual bool is_enemy_obstacle(Enemy& enemy) override;
92 virtual bool is_sword_ignored() const override;
93
94 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
95 virtual bool notify_action_command_pressed() override;
96 virtual bool notify_interaction_with_item(EquipmentItem& item) override;
97 virtual void notify_position_changed() override;
98 virtual void notify_movement_finished() override;
99 bool can_be_lifted() const;
100
101 private:
102
103 void initialize_sprite(const std::string& sprite_name, int initial_direction);
104 void call_script_hero_interaction();
105
106 Subtype subtype; /**< subtpype of NPC */
107 Behavior behavior; /**< type of action done when the player interacts with this entity */
108 bool traversable; /**< Whether this NPC can be traversed by other entities
109 * (not that some entities override this setting). */
110 std::string dialog_to_show; /**< dialog to show when an interaction occurs, or an empty string */
111 std::string item_name; /**< name of an equipment item to notify when an interaction occurs */
112
113};
114
115}
116
117#endif
