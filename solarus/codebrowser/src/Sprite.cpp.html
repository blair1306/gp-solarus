59/**
60 * \brief Returns the sprite animation set corresponding to the specified id.
61 *
62 * The animation set may be created if it is new, or just retrieved from
63 * memory if it way already used before.
64 *
65 * \param id id of the animation set
66 * \return the corresponding animation set
67 */
68SpriteAnimationSet& Sprite::get_animation_set(const std::string& id) {
69
70 SpriteAnimationSet* animation_set = nullptr;
71 auto it = all_animation_sets.find(id);
72 if (it != all_animation_sets.end()) {
73 animation_set = it->second;
74 }
75 else {
76 animation_set = new SpriteAnimationSet(id);
77 all_animation_sets[id] = animation_set;
78 }
79
80 Debug::check_assertion(animation_set != nullptr, "No animation set");
81
82 return *animation_set;
83}

266/**
267 * \brief Sets the current animation of the sprite.
268 *
269 * If the sprite is already playing another animation, this animation is interrupted.
270 * If the sprite is already playing the same animation, nothing is done.
271 *
272 * \param animation_name name of the new animation of the sprite
273 */
274void Sprite::set_current_animation(const std::string& animation_name) {
275
276 if (animation_name != this->current_animation_name || !is_animation_started()) {
277
278 this->current_animation_name = animation_name;
279 if (animation_set.has_animation(animation_name)) {
280 this->current_animation = &animation_set.get_animation(animation_name);
281 set_frame_delay(current_animation->get_frame_delay());
282 }
283 else {
284 this->current_animation = nullptr;
285 }
286
287 int old_direction = this->current_direction;
288 if (current_direction < 0
289 || current_direction >= get_nb_directions()) {
290 current_direction = 0;
291 }
292
293 set_current_frame(0, false);
294 set_finished_callback(ScopedLuaRef());
295
296 LuaContext* lua_context = get_lua_context();
297 if (lua_context != nullptr) {
298 lua_context->sprite_on_animation_changed(*this, current_animation_name);
299 if (current_direction != old_direction) {
300 lua_context->sprite_on_direction_changed(*this, current_animation_name, current_direction);
301 }
302 lua_context->sprite_on_frame_changed(*this, current_animation_name, 0);
303 }
304 }
305}

337/**
338 * \brief Sets the current direction of the sprite's animation and restarts the animation.
339 *
340 * If the specified direction is the current direction, nothing is done.
341 *
342 * \param current_direction the current direction
343 */
344void Sprite::set_current_direction(int current_direction) {
345
346 if (current_direction != this->current_direction) {
347
348 if (current_direction < 0
349 || current_direction >= get_nb_directions()) {
350 std::ostringstream oss;
351 oss << "Illegal direction " << current_direction
352 << " for sprite '" << get_animation_set_id()
353 << "' in animation '" << current_animation_name << "'";
354 Debug::error(oss.str());
355 return;
356 }
357
358 this->current_direction = current_direction;
359
360 set_current_frame(0, false);
361
362 LuaContext* lua_context = get_lua_context();
363 if (lua_context != nullptr) {
364 lua_context->sprite_on_direction_changed(*this, current_animation_name, current_direction);
365 lua_context->sprite_on_frame_changed(*this, current_animation_name, 0);
366 }
367 }

391/**
392 * \brief Sets the current frame of the sprite's animation.
393 *
394 * If the animation was finished, it is restarted.
395 * If the animation is suspended, it remains suspended
396 * but the specified frame is drawn.
397 *
398 * \param current_frame The current frame.
399 * \param notify_script \c true to notify the Lua sprite if any.
400 * Don't set notify_script to \c false unless you do the notification
401 * yourself later.
402 */
403void Sprite::set_current_frame(int current_frame, bool notify_script) {
404
405 finished = false;
406 next_frame_date = System::now() + get_frame_delay();
407
408 if (current_frame != this->current_frame) {
409 this->current_frame = current_frame;
410 set_frame_changed(true);
411
412 if (notify_script) {
413 LuaContext* lua_context = get_lua_context();
414 if (lua_context != nullptr) {
415 lua_context->sprite_on_frame_changed(
416 *this, current_animation_name, current_frame);
417 }
418 }
419 }

621/**
622 * \brief Tests whether this sprite's pixels are overlapping another sprite.
623 * \param other Another sprite.
624 * \param x1 X coordinate of this sprite's origin point on the map,
625 * before applying its current movement if any.
626 * \param y1 Y coordinate of this sprite's origin point on the map,
627 * before applying its current movement if any.
628 * \param x2 X coordinate of the other sprite's origin point on the map,
629 * before applying its current movement if any.
630 * \param y2 Y coordinate of the other sprite's origin point on the map,
631 * before applying its current movement if any.
632 * \return \c true if the sprites are overlapping.
633 */
634bool Sprite::test_collision(const Sprite& other, int x1, int y1, int x2, int y2) const {
635
636 if (current_animation == nullptr || other.current_animation == nullptr) {
637 return false;
638 }
639
640 if (!is_animation_started() || !other.is_animation_started()) {
641 // The animation is not running.
642 return false;
643 }
644
645 if (!are_pixel_collisions_enabled()) {
646 Debug::error(std::string("Pixel-precise collisions are not enabled for sprite '") +
647 get_animation_set_id() + "'");
648 return false;
649 }
650
651 if (!other.are_pixel_collisions_enabled()) {
652 Debug::error(std::string("Pixel-precise collisions are not enabled for sprite '") +
653 other.get_animation_set_id() + "'");
654 return false;
655 }
656
657 const SpriteAnimationDirection& direction1 = current_animation->get_direction(current_direction);
658 const Point& origin1 = direction1.get_origin();
659 Point location1 = { x1 - origin1.x, y1 - origin1.y };
660 location1 += get_xy();
661 const PixelBits& pixel_bits1 = direction1.get_pixel_bits(current_frame);
662
663 const SpriteAnimationDirection& direction2 = other.current_animation->get_direction(other.current_direction);
664 const Point& origin2 = direction2.get_origin();
665 Point location2 = { x2 - origin2.x, y2 - origin2.y };
666 location2 += other.get_xy();
667 const PixelBits& pixel_bits2 = direction2.get_pixel_bits(other.current_frame);
668
669 return pixel_bits1.test_collision(pixel_bits2, location1, location2);
670}

672/**
673 * \brief Checks whether the frame has to be changed.
674 *
675 * If the frame changes, next_frame_date is updated.
676 */
677void Sprite::update() {
678
679 Drawable::update();
680
681 if (is_suspended() || paused) {
682 return;
683 }
684
685 LuaContext* lua_context = get_lua_context();
686
687 frame_changed = false;
688 uint32_t now = System::now();
689
690 // Update the current frame.
691 if (synchronize_to == nullptr
692 || current_animation_name != synchronize_to->get_current_animation()
693 || synchronize_to->get_current_direction() > get_nb_directions()
694 || synchronize_to->get_current_frame() > get_nb_frames()) {
695
696 // Update frames normally (with time).
697 while (!finished &&
698 !is_suspended() &&
699 !paused &&
700 get_frame_delay() > 0 &&
701 now >= next_frame_date
702 ) {
703 int next_frame = get_next_frame();
704
705 // Test if the animation is finished.
706 if (next_frame == -1) {
707 finished = true;
708 notify_finished();
709 }
710 else {
711 current_frame = next_frame;
712 uint32_t old_next_frame_date = next_frame_date;
713 next_frame_date += get_frame_delay();
714 if (next_frame_date < old_next_frame_date) {
715 // Overflow. Can happen with data files generated with the editor
716 // before 1.4 (frame_delay is too big).
717 next_frame_date = std::numeric_limits<uint32_t>::max();
718 }
719 }
720 set_frame_changed(true);
721
722 if (lua_context != nullptr) {
723 lua_context->sprite_on_frame_changed(*this, current_animation_name, current_frame);
724 }
725 }
726 }
727 else {
728 // Take the same frame as the other sprite.
729 if (synchronize_to->is_animation_finished()) {
730 finished = true;
731 notify_finished();
732 }
733 else {
734 int other_frame = synchronize_to->get_current_frame();
735 if (other_frame != current_frame) {
736 current_frame = other_frame;
737 next_frame_date = now + get_frame_delay();
738 set_frame_changed(true);
739
740 if (lua_context != nullptr) {
741 lua_context->sprite_on_frame_changed(*this, current_animation_name, current_frame);
742 }
743 }
744 }
745 }
746
747 // Update the special effects.
748 if (is_blinking()) {
749 // The sprite is blinking.
750
751 while (now >= blink_next_change_date) {
752 blink_is_sprite_visible = !blink_is_sprite_visible;
753 blink_next_change_date += blink_delay;
754 }
755 }
756}

758/**
759 * \brief Draws the sprite on a surface, with its current animation,
760 * direction and frame.
761 * \param dst_surface The destination surface.
762 * \param dst_position Coordinates on the destination surface
763 * (the origin will be placed at this position).
764 */
765void Sprite::raw_draw(
766 Surface& dst_surface,
767 const Point& dst_position) {
768
769 if (current_animation == nullptr) {
770 return;
771 }
772
773 if (!is_animation_finished()
774 && (blink_delay == 0 || blink_is_sprite_visible)) {
775
776 get_intermediate_surface().clear();
777 current_animation->draw(get_intermediate_surface(), get_origin(),
778 current_direction, current_frame);
779 get_intermediate_surface().set_blend_mode(get_blend_mode());
780 get_intermediate_surface().draw_region(
781 Rectangle(get_size()),
782 std::static_pointer_cast<Surface>(dst_surface.shared_from_this()),
783 dst_position - get_origin()
784 );
785 }
786}
787
788/**
789 * \brief Draws a subrectangle of the current frame of this sprite.
790 * \param region The subrectangle to draw, relative to the origin point.
791 * It may be bigger than the frame: in this case it will be clipped.
792 * \param dst_surface The destination surface.
793 * \param dst_position Coordinates on the destination surface.
794 * The origin point of the sprite will appear at these coordinates.
795 */
796void Sprite::raw_draw_region(
797 const Rectangle& region,
798 Surface& dst_surface,
799 const Point& dst_position) {
800
801 if (current_animation == nullptr) {
802 return;
803 }
804
805 if (!is_animation_finished()
806 && (blink_delay == 0 || blink_is_sprite_visible)) {
807
808 // Clear the working surface.
809 get_intermediate_surface().clear();
810
811 // Draw the current animation on the working surface.
812 const Point& origin = get_origin();
813 current_animation->draw(
814 get_intermediate_surface(),
815 origin,
816 current_direction,
817 current_frame);
818
819 // If the region is bigger than the current frame, clip it.
820 // Otherwise, more than the current frame could be visible.
821 Rectangle src_position(region);
822 src_position.add_xy(origin);
823 const Size& frame_size = get_size();
824 if (src_position.get_x() < 0) {
825 src_position.set_width(src_position.get_width() + src_position.get_x());
826 src_position.set_x(0);
827 }
828 if (src_position.get_x() + src_position.get_width() > frame_size.width) {
829 src_position.set_width(frame_size.width - src_position.get_x());
830 }
831 if (src_position.get_y() < 0) {
832 src_position.set_height(src_position.get_height() + src_position.get_y());
833 src_position.set_y(0);
834 }
835 if (src_position.get_y() + src_position.get_height() > frame_size.height) {
836 src_position.set_height(frame_size.height - src_position.get_y());
837 }
838
839 if (src_position.get_width() <= 0 || src_position.get_height() <= 0) {
840 // Nothing remains visible.
841 return;
842 }
843
844 // Calculate the destination coordinates.
845 Point dst_position2 = dst_position;
846 dst_position2 += src_position.get_xy(); // Let a space for the part outside the region.
847 dst_position2 -= origin; // Input coordinates were relative to the origin.
848 get_intermediate_surface().set_blend_mode(get_blend_mode());
849 get_intermediate_surface().draw_region(
850 src_position,
851 std::static_pointer_cast<Surface>(dst_surface.shared_from_this()),
852 dst_position2
853 );
854 }
855}

923/**
924 * \brief Performs appropriate notifications when the current animation finishes.
925 */
926void Sprite::notify_finished() {
927
928 LuaContext* lua_context = get_lua_context();
929 if (lua_context != nullptr) {
930 lua_State* l = finished_callback_ref.get_lua_state();
931
932 // Sprite callback.
933 if (!finished_callback_ref.is_empty()) {
934 // The callback may be a function or a string.
935 finished_callback_ref.push();
936 finished_callback_ref.clear();
937 if (lua_isstring(l, -1)) {
938 // Name of a next animation to set.
939 std::string animation = lua_tostring(l, -1);
940 lua_pop(l, 1);
941 set_current_animation(animation);
942 }
943 else {
944 // Function to call.
945 LuaTools::call_function(l, 0, 0, "sprite callback");
946 }
947 }
948
949 // Sprite event.
950 lua_context->sprite_on_animation_finished(*this, current_animation_name);
951 }
