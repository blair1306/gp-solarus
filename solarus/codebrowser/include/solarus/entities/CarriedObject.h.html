17#ifndef SOLARUS_CARRIED_OBJECT_H
18#define SOLARUS_CARRIED_OBJECT_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/SpritePtr.h"
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief An item carried or thrown by the hero.
29 *
30 * Represents an item (pot, bush...) carried by the hero. He can throw it.
31 * While the hero is carrying it, the item is attached to the hero, not to the map.
32 * As soon as he throws it, the item becomes attached to the map and the hero
33 * may lift another item.
34 */
35class CarriedObject: public Entity {
36
37 public:
38
39 static constexpr EntityType ThisType = EntityType::CARRIED_OBJECT;
40
41 /**
42 * Indicates what to do with a carried object.
43 */
44 enum Behavior {
45 BEHAVIOR_THROW, /**< make the hero throw the item */
46 BEHAVIOR_DESTROY, /**< destroy the item silently */
47 BEHAVIOR_KEEP /**< let the hero continue to carry the item */
48 };
49
50 CarriedObject(
51 Hero& hero,
52 const Entity& original_entity,
53 const std::string& animation_set_id,
54 const std::string& destruction_sound_id,
55 int damage_on_enemies,
56 uint32_t explosion_date
57 );
58
59 EntityType get_type() const override;
60 bool is_ground_observer() const override;
61
62 int get_damage_on_enemies() const;
63
64 void set_animation_stopped();
65 void set_animation_walking();
66 void throw_item(int direction);
67 bool is_being_lifted() const;
68 bool is_being_thrown() const;
69 void break_item();
70 void break_item_on_ground();
71 bool is_broken() const;
72 bool can_explode() const;
73
74 void set_suspended(bool suspended) override;
75 void update() override;
76 void draw_on_map() override;
77
78 bool is_teletransporter_obstacle(Teletransporter& teletransporter) override;
79 bool is_stream_obstacle(Stream& stream) override;
80 bool is_stairs_obstacle(Stairs& stairs) override;
81 bool is_low_wall_obstacle() const override;
82 bool is_deep_water_obstacle() const override;
83 bool is_hole_obstacle() const override;
84 bool is_lava_obstacle() const override;
85 bool is_prickle_obstacle() const override;
86 bool is_ladder_obstacle() const override;
87 bool is_switch_obstacle(Switch& sw) override;
88 bool is_raised_block_obstacle(CrystalBlock& raised_block) override;
89 bool is_crystal_obstacle(Crystal& crystal) override;
90 bool is_sensor_obstacle(Sensor& sensor) override;
91 bool is_npc_obstacle(Npc& npc) override;
92 bool is_jumper_obstacle(Jumper& jumper, const Rectangle& candidate_position) override;
93 bool is_enemy_obstacle(Enemy& enemy) override;
94 void notify_obstacle_reached() override;
95 void notify_collision_with_switch(Switch& sw, CollisionMode collision_mode) override;
96 void notify_collision_with_crystal(Crystal& crystal, CollisionMode collision_mode) override;
97 void notify_collision_with_stairs(Stairs& stairs, CollisionMode collision_mode) override;
98 void notify_collision_with_enemy(
99 Enemy& enemy,
100 Sprite& enemy_sprite,
101 Sprite& this_sprite
102 ) override;
103 void notify_attacked_enemy(
104 EnemyAttack attack,
105 Enemy& victim,
106 const Sprite* victim_sprite,
107 EnemyReaction::Reaction& result,
108 bool killed
109 ) override;
110
111 private:
112
113 bool will_explode_soon() const;
114
115 // game data
116 Hero& hero; /**< the hero, who is carrying or throwing this item */
117
118 // state
119 bool is_lifting; /**< indicates that the hero is lifting this item */
120 bool is_throwing; /**< indicates that the item is being thrown */
121 bool is_breaking; /**< indicates that the item is breaking */
122 bool break_one_layer_above; /**< indicates that the item has to get broken
123 * now one layer above its current position */
124 std::string
125 destruction_sound_id; /**< the sound played when the item breaks */
126 int damage_on_enemies; /**< damage for an enemy that receives the item */
127
128 // throwing the item
129 SpritePtr main_sprite; /**< main sprite of the destructible object */
130 SpritePtr shadow_sprite; /**< sprite of the shadow when the item is being thrown */
131 int throwing_direction; /**< direction where the item is thrown (0 to 3) */
132 uint32_t next_down_date; /**< when the item is thrown, date when it move one pixel downwards next time */
133 int item_height; /**< current height where the item is drawn above its shadow */
134 int y_increment; /**< next y change for item_height */
135
136 // explosion of the item
137 uint32_t explosion_date; /**< date when the item explodes (0 if there is no explosion) */
138
139 static const std::string lifting_trajectories[4]; /**< trajectory of the lifting movement for each direction */
140
141};
142
143}
144
145#endif
