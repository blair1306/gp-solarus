17#ifndef SOLARUS_HERO_PLAYER_MOVEMENT_STATE_H
18#define SOLARUS_HERO_PLAYER_MOVEMENT_STATE_H
19
20#include "solarus/Common.h"
21#include "solarus/hero/HeroState.h"
22#include <cstdint>
23#include <memory>
24#include <string>
25
26namespace Solarus {
27
28class PlayerMovement;
29
30/**
31 * \brief State where the hero is controlled by the player.
32 *
33 * Abstract class for all hero states such that the movement is controlled by the player.
34 * This means that the hero can move in the eight directions and the movement
35 * is an instance of PlayerMovement.
36 * These states include FreeState (the normal walk), SwordLoadingState (the sword is
37 * loading), CarryingState (the hero is carrying something) and SwimmingState.
38 */
39class Hero::PlayerMovementState: public HeroState {
40
41 public:
42
43 virtual ~PlayerMovementState();
44
45 virtual void start(const State* previous_state) override;
46 virtual void stop(const State* next_state) override;
47 virtual void set_map(Map& map) override;
48
49 virtual void update() override;
50 virtual void set_suspended(bool suspended) override;
51 virtual void set_animation_stopped();
52 virtual void set_animation_walking();
53 virtual bool can_control_movement() const override;
54 virtual int get_wanted_movement_direction8() const override;
55 virtual void notify_walking_speed_changed() override;
56 virtual void notify_movement_changed() override;
57 virtual void notify_position_changed() override;
58 virtual void notify_layer_changed() override;
59 virtual bool can_be_hurt(Entity* attacker) const override;
60 virtual bool can_pick_treasure(EquipmentItem& item) const override;
61 virtual bool can_take_jumper() const override;
62 virtual void notify_jumper_activated(Jumper& jumper) override;
63
64 protected:
65
66 PlayerMovementState(Hero& hero, const std::string& state_name);
67
68 const std::shared_ptr<PlayerMovement>& get_player_movement();
69 std::shared_ptr<const PlayerMovement> get_player_movement() const;
70
71 private:
72
73 void cancel_jumper();
74
75 std::shared_ptr<PlayerMovement>
76 player_movement; /**< The movement created by this state.
77 * The movement of the hero is also this object,
78 * unless a script decided to change it. */
79 std::shared_ptr<Jumper>
80 current_jumper; /**< The jumper about to be triggered or nullptr */
81 uint32_t jumper_start_date; /**< Date to trigger the jumper
82 * (because a small delay is necessary) */
83
84};
85
86}
87
88#endif
