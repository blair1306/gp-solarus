131/**
132 * \brief Load and parse the Lua file of the requested shader.
133 * \param path The path to the lua file, relative to the data folder.
134 */
135void Shader::load_lua_file(const std::string& path) {
136
137 lua_State* l = luaL_newstate();
138 luaL_openlibs(l); // FIXME don't open the libs
139
140
141 const std::string& buffer = QuestFiles::data_file_read(path);
142 int load_result = luaL_loadbuffer(l, buffer.data(), buffer.size(), path.c_str());
143
144 if (load_result != 0) {
145 // Syntax error in the lua file.
146 Debug::die(std::string("Failed to load ") + path + " : " + lua_tostring(l, -1));
147 }
148 else {
149 // Register the callback and send string parameters to the lua script.
150 register_callback(l);
151 lua_pushstring(l, Video::get_rendering_driver_name().c_str());
152 lua_pushstring(l, shading_language_version.c_str());
153 lua_pushstring(l, sampler_type.c_str());
154
155 if (lua_pcall(l, 3, 0, 0) != 0) {
156
157 // Runtime error.
158 Debug::die(std::string("Failed to parse ") + path + " : " + lua_tostring(l, -1));
159 }
160 }
161
162 lua_close(l);
163}
