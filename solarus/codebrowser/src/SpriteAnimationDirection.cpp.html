26/**
27 * \brief Constructor.
28 * \param frames Position of each frame of the sequence in the image.
29 * \param origin Coordinates of the sprite's origin point.
30 */
31SpriteAnimationDirection::SpriteAnimationDirection(
32 const std::vector<Rectangle>& frames,
33 const Point& origin):
34 frames(frames),
35 origin(origin) {
36
37 Debug::check_assertion(!frames.empty(), "Empty sprite direction");
38}
39
40/**
41 * \brief Returns the size of a frame.
42 * \return The size of a frame.
43 */
44Size SpriteAnimationDirection::get_size() const {
45
46 Debug::check_assertion(get_nb_frames() > 0, "Invalid number of frames");
47 return { frames[0].get_width(), frames[0].get_height() };
48}
49
50/**
51 * \brief Returns the number of frames in this direction.
52 * \return the number of frames
53 */
54int SpriteAnimationDirection::get_nb_frames() const {
55 return frames.size();
56}
57
58/**
59 * \brief Returns the rectangle representing the specified frame on the source image.
60 * \param frame a frame number
61 * \return the rectangle of this frame
62 */
63const Rectangle& SpriteAnimationDirection::get_frame(int frame) const {
64
65 if (frame < 0 || frame >= get_nb_frames()) {
66 std::ostringstream oss;
67 oss << "Invalid frame " << frame
68 << ": this direction has " << get_nb_frames() << " frames";
69 Debug::die(oss.str());
70 }
71 return frames[frame];
72}
73
74/**
75 * \brief Draws a specific frame on the map.
76 * \param dst_surface the surface on which the frame will be drawn
77 * \param dst_position coordinates on the destination surface
78 * (the origin point will be drawn at this position)
79 * \param current_frame the frame to show
80 * \param src_image the image from which the frame is extracted
81 */
82void SpriteAnimationDirection::draw(Surface& dst_surface,
83 const Point& dst_position, int current_frame, Surface& src_image) {
84
85 const Rectangle& current_frame_rect = get_frame(current_frame);
86
87 // Position of the sprite's upper left corner.
88 Point position_top_left = dst_position;
89 position_top_left -= origin;
90
91 src_image.draw_region(
92 current_frame_rect,
93 std::static_pointer_cast<Surface>(dst_surface.shared_from_this()),
94 position_top_left
95 );
96}
97
98/**
99 * \brief Calculates the bit fields representing the non-transparent pixels
100 * of the images in this direction.
101 *
102 * This method has to be called if you want a sprite having this animations
103 * to be able to detect pixel-perfect collisions.
104 * If the pixel-perfect collisions are already enabled, this function does nothing.
105 *
106 * \param src_image the surface containing the animations
107 */
108void SpriteAnimationDirection::enable_pixel_collisions(Surface& src_image) {
109
110 if (!are_pixel_collisions_enabled()) {
111 for (int i = 0; i < get_nb_frames(); i++) {
112 pixel_bits.emplace_back(src_image, frames[i]);
113 }
114 }
115}
116
117/**
118 * \brief Disables the pixel-perfect collision ability of this sprite animation direction.
119 */
120void SpriteAnimationDirection::disable_pixel_collisions() {
121
122 pixel_bits.clear();
123}
124
125/**
126 * \brief Returns whether the pixel-perfect collisions are enabled for this direction.
127 * \return true if the pixel-perfect collisions are enabled
128 */
129bool SpriteAnimationDirection::are_pixel_collisions_enabled() const {
130 return !pixel_bits.empty();
131}
