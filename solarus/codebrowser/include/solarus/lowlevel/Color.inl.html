19
20////////////////////////////////////////////////////////////
21// Point methods
22
23/**
24 * \brief Creates a transparent color with black RGB values.
25 */
26constexpr Color::Color():
27 r(0),
28 g(0),
29 b(0),
30 a(0) {
31}
32
33/**
34 * \brief Creates a color with the specified RGBA values.
35 * \param r The red component (0 to 255)
36 * \param g The green component (0 to 255)
37 * \param b The blue component (0 to 255)
38 * \param a The alpha component (0 to 255, 255 is opaque).
39 */
40constexpr Color::Color(int r, int g, int b, int a):
41 r(r),
42 g(g),
43 b(b),
44 a(a) {
45}
46
47/**
48 * \brief Returns the alpha component of this color.
49 * \return The alpha component. 255 is opaque.
50 */
51constexpr uint8_t Color::get_alpha() const {
52 return a;
53}
54
55/**
56 * \brief Sets the alpha component of this color.
57 * \param alpha The alpha component. 255 is opaque.
58 */
59inline void Color::set_alpha(uint8_t alpha) {
60 this->a = alpha;
61}
62
63/**
64 * \brief Gets the red, green, blue and alpha values of this color.
65 * \param[out] r Red component.
66 * \param[out] g Green component.
67 * \param[out] b Blue component.
68 * \param[out] a Alpha component.
69 */
70inline void Color::get_components(
71 uint8_t& r, uint8_t& g, uint8_t& b, uint8_t& a) const {
72
73 r = this->r;
74 g = this->g;
75 b = this->b;
76 a = this->a;
77}
78
79/**
80 * \brief Sets the red, green, blue and alpha values of this color.
81 * \param r Red component to write.
82 * \param g Green component to write.
83 * \param b Blue component to write.
84 * \param a Alpha component to write.
85 */
86inline void Color::set_components(int r, int g, int b, int a) {
87
88 this->r = r;
89 this->g = g;
90 this->b = b;
91 this->a = a;
92}
93
94/**
95 * \brief Equality operator.
96 * \param lhs A color.
97 * \param rhs Another color.
98 * \return \c true if both colors are equal.
99 */
100constexpr bool operator==(const Color& lhs, const Color& rhs) {
101
102 return lhs.r == rhs.r &&
103 lhs.g == rhs.g &&
104 lhs.b == rhs.b &&
105 lhs.a == rhs.a;
106}
107
108/**
109 * \brief Compare two colors.
110 * \param lhs A color.
111 * \param rhs Another color.
112 * \return \c true if both colors are equal.
113 */
114constexpr bool operator!=(const Color& lhs, const Color& rhs) {
115
116 return lhs.r != rhs.r ||
117 lhs.g != rhs.g ||
118 lhs.b != rhs.b ||
119 lhs.a != rhs.a;
120}
