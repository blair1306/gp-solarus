42#ifndef QRGB_H
43#define QRGB_H
44
45#include <QtCore/qglobal.h>
46#include <QtCore/qprocessordetection.h>
47
48QT_BEGIN_NAMESPACE
49
50
51typedef unsigned int QRgb; // RGB triplet
52
53// non-namespaced Qt global variable
54const Q_DECL_UNUSED QRgb RGB_MASK = 0x00ffffff; // masks RGB values
55
56inline Q_DECL_CONSTEXPR int qRed(QRgb rgb) // get red part of RGB
57{ return ((rgb >> 16) & 0xff); }
58
59inline Q_DECL_CONSTEXPR int qGreen(QRgb rgb) // get green part of RGB
60{ return ((rgb >> 8) & 0xff); }
61
62inline Q_DECL_CONSTEXPR int qBlue(QRgb rgb) // get blue part of RGB
63{ return (rgb & 0xff); }
64
65inline Q_DECL_CONSTEXPR int qAlpha(QRgb rgb) // get alpha part of RGBA
66{ return rgb >> 24; }
67
68inline Q_DECL_CONSTEXPR QRgb qRgb(int r, int g, int b)// set RGB value
69{ return (0xffu << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }
70
71inline Q_DECL_CONSTEXPR QRgb qRgba(int r, int g, int b, int a)// set RGBA value
72{ return ((a & 0xff) << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }
73
74inline Q_DECL_CONSTEXPR int qGray(int r, int g, int b)// convert R,G,B to gray 0..255
75{ return (r*11+g*16+b*5)/32; }
76
77inline Q_DECL_CONSTEXPR int qGray(QRgb rgb) // convert RGB to gray 0..255
78{ return qGray(qRed(rgb), qGreen(rgb), qBlue(rgb)); }
79
80inline Q_DECL_CONSTEXPR bool qIsGray(QRgb rgb)
81{ return qRed(rgb) == qGreen(rgb) && qRed(rgb) == qBlue(rgb); }
82
83
84#if Q_PROCESSOR_WORDSIZE == 8 // 64-bit version
85inline QRgb qPremultiply(QRgb x)
86{
87 const uint a = qAlpha(x);
88 quint64 t = (((quint64(x)) | ((quint64(x)) << 24)) & 0x00ff00ff00ff00ff) * a;
89 t = (t + ((t >> 8) & 0xff00ff00ff00ff) + 0x80008000800080) >> 8;
90 t &= 0x000000ff00ff00ff;
91 return (uint(t)) | (uint(t >> 24)) | (a << 24);
92}
93#else // 32-bit version
94inline QRgb qPremultiply(QRgb x)
95{
96 const uint a = qAlpha(x);
97 uint t = (x & 0xff00ff) * a;
98 t = (t + ((t >> 8) & 0xff00ff) + 0x800080) >> 8;
99 t &= 0xff00ff;
100
101 x = ((x >> 8) & 0xff) * a;
102 x = (x + ((x >> 8) & 0xff) + 0x80);
103 x &= 0xff00;
104 x |= t | (a << 24);
105 return x;
106}
107#endif
108
109Q_GUI_EXPORT extern const uint qt_inv_premul_factor[];
110
111inline QRgb qUnpremultiply(QRgb p)
112{
113 const uint alpha = qAlpha(p);
114 // Alpha 255 and 0 are the two most common values, which makes them beneficial to short-cut.
115 if (alpha == 255)
116 return p;
117 if (alpha == 0)
118 return 0;
119 // (p*(0x00ff00ff/alpha)) >> 16 == (p*255)/alpha for all p and alpha <= 256.
120 const uint invAlpha = qt_inv_premul_factor[alpha];
121 // We add 0x8000 to get even rounding. The rounding also ensures that qPremultiply(qUnpremultiply(p)) == p for all p.
122 return qRgba((qRed(p)*invAlpha + 0x8000)>>16, (qGreen(p)*invAlpha + 0x8000)>>16, (qBlue(p)*invAlpha + 0x8000)>>16, alpha);
123}
