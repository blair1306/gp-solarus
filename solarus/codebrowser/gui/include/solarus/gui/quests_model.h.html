17#ifndef SOLARUS_GUI_QUESTS_MODEL_H
18#define SOLARUS_GUI_QUESTS_MODEL_H
19
20#include "solarus/gui/gui_common.h"
21#include "solarus/QuestProperties.h"
22#include <QAbstractListModel>
23#include <QIcon>
24#include <QMetaType>
25
26namespace SolarusGui {
27
28/**
29 * @brief List of quests added to Solarus.
30 */
31class SOLARUS_GUI_API QuestsModel : public QAbstractListModel {
32
33public:
34
35 /**
36 * @brief Ways to sort the quests in the list. More convenient than columns.
37 */
38 enum class QuestSort {
39 SortByName = 0,
40 SortByAuthor = 1,
41 SortByDate = 2
42 };
43
44 /**
45 * @brief Info of a quest from the list.
46 */
47 struct QuestInfo {
48 QString path; /**< Path to the quest directory. */
49 QString directory_name; /**< Name of the quest directory. */
50 QIcon icon; /**< Icon of the quest. */
51 QPixmap logo; /**< Logo of the quest (recommended: 200x140). */
52 Solarus::QuestProperties
53 properties; /**< All properties from quest.dat. */
54 };
55
56 explicit QuestsModel(QObject* parent = nullptr);
57
58 // QAbstractListModel API
59 int rowCount(const QModelIndex& parent = QModelIndex()) const override;
60 QVariant data(const QModelIndex& index, int role = Qt::DisplayRole) const override;
61
62 // QuestsModel API
63 int path_to_index(const QString& quest_path) const;
64 QString index_to_path(int quest_index) const;
65
66 bool has_quest(const QString& quest_path);
67 bool add_quest(const QString& quest_path);
68 bool remove_quest(int index);
69 QStringList get_paths() const;
70
71 Solarus::QuestProperties get_quest_properties(int quest_index) const;
72
73 const QPixmap& get_quest_default_logo() const;
74 const QPixmap& get_quest_logo(int index) const;
75
76 const QIcon& get_quest_default_icon() const;
77
78 void sort(int column, Qt::SortOrder order = Qt::AscendingOrder) override;
79 void sort(QuestSort sort, Qt::SortOrder order = Qt::AscendingOrder);
80
81private:
82
83 void do_sort(QuestSort sort, Qt::SortOrder order = Qt::AscendingOrder);
84 void load_icon(int quest_index) const;
85
86 mutable std::vector<QuestInfo>
87 quests; /**< Info of each quest in the list. */
88};
89
90}
91
92// Allow to use QuestInfo in QVariant
93Q_DECLARE_METATYPE(SolarusGui::QuestsModel::QuestInfo)
94
95#endif
