32/**
33 * \brief Constructor.
34 * \param hero The hero controlled by this state.
35 * \param source_xy Coordinates of the thing (usually an enemy) that hurts
36 * the hero, used to compute the trajectory of the hero.
37 * If nullptr, the hero won't move.
38 * \param damage Number of life poitns to remove
39 * (this number may be reduced by the tunic or by hero:on_taking_damage()).
40 */
41Hero::HurtState::HurtState(
42 Hero& hero,
43 const Point* source_xy,
44 int damage):
45 HeroState(hero, "hurt"),
46 has_source(source_xy != nullptr),
47 source_xy(has_source ? *source_xy : Point()),
48 damage(damage),
49 end_hurt_date(0) {
50
51}
52
53/**
54 * \brief Starts this state.
55 * \param previous_state The previous state.
56 */
57void Hero::HurtState::start(const State* previous_state) {
58
59 HeroState::start(previous_state);
60
61 Equipment& equipment = get_equipment();
62
63 Sound::play("hero_hurt");
64
65 Hero& hero = get_entity();
66 const uint32_t invincibility_duration = 2000;
67 hero.set_invincible(true, invincibility_duration);
68 get_sprites().set_animation_hurt();
69 get_sprites().blink(invincibility_duration);
70
71 if (has_source) {
72 double angle = Geometry::get_angle(source_xy, hero.get_xy());
73 std::shared_ptr<StraightMovement> movement =
74 std::make_shared<StraightMovement>(false, true);
75 movement->set_max_distance(24);
76 movement->set_speed(120);
77 movement->set_angle(angle);
78 hero.set_movement(movement);
79 }
80 end_hurt_date = System::now() + 200;
81
82 // See if the script customizes how the hero takes damages.
83 bool handled = get_lua_context().hero_on_taking_damage(hero, damage);
84
85 if (!handled && damage != 0) {
86 // No customized damage: perform the default calculation.
87 // The level of the tunic reduces the damage,
88 // but we remove at least 1 life point.
89 int life_points = std::max(1, damage / (equipment.get_ability(Ability::TUNIC)));
90
91 equipment.remove_life(life_points);
92 if (equipment.has_ability(Ability::TUNIC)) {
93 equipment.notify_ability_used(Ability::TUNIC);
94 }
95 }
96}

98/**
99 * \brief Ends this state.
100 * \param next_state the next state
101 */
102void Hero::HurtState::stop(const State* next_state) {
103
104 HeroState::stop(next_state);
105
106 get_entity().clear_movement();
107}
108
109/**
110 * \brief Updates this state.
111 */
112void Hero::HurtState::update() {
113
114 HeroState::update();
115
116 Hero& hero = get_entity();
117 if ((hero.get_movement() != nullptr && hero.get_movement()->is_finished())
118 || System::now() >= end_hurt_date) {
119 // The movement may be finished, or the end date may be reached
120 // when there is an obstacle or when there is no movement at all.
121
122 hero.clear_movement();
123 hero.start_state_from_ground();
124 }
125}

127/**
128 * \brief Notifies this state that the game was just suspended or resumed.
129 * \param suspended true if the game is suspended
130 */
131void Hero::HurtState::set_suspended(bool suspended) {
132
133 HeroState::set_suspended(suspended);
134
135 if (!suspended) {
136 uint32_t diff = System::now() - get_when_suspended();
137 end_hurt_date += diff;
138 }
139}
