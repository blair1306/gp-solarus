55/**
56 * \brief Draws the tile on the map.
57 */
58void Tile::draw_on_map() {
59
60 const CameraPtr& camera = get_map().get_camera();
61 if (camera == nullptr) {
62 return;
63 }
64
65 // Note that the tiles are also optimized for drawing.
66 // This function is called at each frame only if the tile is in an
67 // animated region. Otherwise, tiles are drawn once when loading the map.
68 draw(get_map().get_camera_surface(), camera->get_top_left_xy());
69}
70
71/**
72 * \brief Draws the tile on the specified surface.
73 * \param dst_surface the destination surface
74 * \param viewport coordinates of the top-left corner of dst_surface
75 * relative to the map
76 */
77void Tile::draw(const SurfacePtr& dst_surface, const Point& viewport) {
78
79 Rectangle dst_position(
80 get_top_left_x() - viewport.x,
81 get_top_left_y() - viewport.y,
82 get_width(),
83 get_height()
84 );
85
86 tile_pattern.fill_surface(
87 dst_surface,
88 dst_position,
89 get_map().get_tileset(),
90 viewport
91 );
92}

110/**
111 * \brief Returns whether the pattern is animated.
112 *
113 * Non-animated tiles may be rendered faster by using intermediate surfaces
114 * that are drawn only once.
115 * This function should return false if the tile pattern is always drawn the same way.
116 *
117 * \return true if the pattern of this tile is animated
118 */
119bool Tile::is_animated() const {
120 return tile_pattern.is_animated();
121}
