79/**
80 * \brief Returns the distance between two points.
81 * \param x1 X coordinate of the first point.
82 * \param y1 Y coordinate of the first point.
83 * \param x2 X coordinate of the second point.
84 * \param y2 Y coordinate of the second point.
85 * \return The distance in pixels
86 */
87inline double get_distance(int x1, int y1, int x2, int y2) {
88
89 const int dx = x2 - x1;
90 const int dy = y2 - y1;
91 return std::sqrt((double) (dx * dx + dy * dy));
92}
93
94/**
95 * \brief Returns the square of the distance between two points.
96 * \param x1 X coordinate of the first point.
97 * \param y1 Y coordinate of the first point.
98 * \param x2 X coordinate of the second point.
99 * \param y2 Y coordinate of the second point.
100 * \return Square of the distance.
101 */
102inline int get_distance2(int x1, int y1, int x2, int y2) {
103
104 const int dx = x2 - x1;
105 const int dy = y2 - y1;
106 return dx * dx + dy * dy;
107}
108
109/**
110 * \brief Returns the distance between two points.
111 * \param point1 First point.
112 * \param point2 Second point.
113 * \return The distance in pixels.
114 */
115inline double get_distance(const Point& point1, const Point& point2) {
116
117 return get_distance(point1.x, point1.y, point2.x, point2.y);
118}
119
120/**
121 * \brief Returns the square of the distance between two points.
122 * \param point1 The first point.
123 * \param point2 The second point.
124 * \return The distance in pixels
125 */
126inline int get_distance2(const Point& point1, const Point& point2) {
127
128 return get_distance2(point1.x, point1.y, point2.x, point2.y);
129}
130
131/**
132 * \brief Returns the Manhattan distance of two points, measured in number of pixels.
133 * \param point1 First point
134 * \param point2 Second point
135 * \return The Manhattan distance between these points
136 */
137inline int get_manhattan_distance(const Point& point1, const Point& point2) {
138
139 using std::abs;
140 return abs(point2.x - point1.x) +
141 abs(point2.y - point1.y);
142}
