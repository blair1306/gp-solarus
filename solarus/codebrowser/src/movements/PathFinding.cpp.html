67/**
68 * \brief Tries to find a path between the source point and the target point.
69 * \return the path found, or an empty string if no path was found
70 * (because there is no path or the target is too far)
71 */
72std::string PathFinding::compute_path() {
73
74 if (!target_entity.is_obstacle_for(source_entity)) {
75 // No offset needed.
76 return compute_path(Point());
77 }
78
79 // The target is not traversable: then try to compute a path to somewhere close.
80 const std::vector<Point> offsets = {
81 Point(target_entity.get_width(), 0),
82 Point(0, -target_entity.get_height()),
83 Point(-target_entity.get_width(), 0),
84 Point(0, target_entity.get_height())
85 };
86
87 std::string best_path;
88 size_t minimum_steps = std::numeric_limits<int>::max();
89 for (const Point& offset : offsets) {
90 std::string path = compute_path(offset);
91 if (!path.empty() && path.size() < minimum_steps) {
92 best_path = path;
93 minimum_steps = path.size();
94 }
95 }
96
97 return best_path;
98}

100/**
101 * \brief Tries to find a path from the source point to the target point
102 * plus an offset.
103 * \param offset Translation to add to the target.
104 * \return the path found, or an empty string if no path was found
105 * (because there is no path or the target is too far)
106 */
107std::string PathFinding::compute_path(const Point& offset) {
108
109 Point source = source_entity.get_bounding_box().get_xy();
110 Point target = target_entity.get_bounding_box().get_xy() + offset;
111
112 // std::cout << "will compute a path from " << source << " to " << target << std::endl;
113
114 target.x += 4;
115 target.x += -target.x % 8;
116 target.y += 4;
117 target.y += -target.y % 8;
118 const int target_index = get_square_index(target);
119
120 Debug::check_assertion(target.x % 8 == 0 && target.y % 8 == 0,
121 "Could not snap the target to the map grid");
122
123 const int total_mdistance = Geometry::get_manhattan_distance(source, target);
124 if (total_mdistance > 200 || target_entity.get_layer() != source_entity.get_layer()) {
125 //std::cout << "too far, not computing a path\n";
126 return ""; // too far to compute a path
127 }
128
129 std::string path = "";
130
131 Node starting_node;
132 const int index = get_square_index(source);
133 starting_node.location = source;
134 starting_node.index = index;
135 starting_node.previous_cost = 0;
136 starting_node.heuristic = total_mdistance;
137 starting_node.total_cost = total_mdistance;
138 starting_node.direction = ' ';
139 starting_node.parent_index = -1;
140
141 open_list.clear();
142 closed_list.clear();
143 open_list_indices.clear();
144
145 open_list[index] = starting_node;
146 open_list_indices.push_front(index);
147
148 bool finished = false;
149 while (!finished) {
150
151 // pick the node with the lowest total cost in the open list
152 const int index = open_list_indices.front();
153 Node* current_node = &open_list[index];
154 open_list_indices.pop_front();
155 closed_list[index] = *current_node;
156 open_list.erase(index);
157 current_node = &closed_list[index];
158
159 //std::cout << System::now() << " picking the lowest cost node in the open list ("
160 // << (open_list_indices.size() + 1) << " elements): "
161 // << current_node->location << ", index = " << current_node->index
162 // << ", cost = " << current_node->previous_cost << " + " << current_node->heuristic << "\n";
163
164 if (index == target_index) {
165 //std::cout << "target node was added to the closed list\n";
166 finished = true;
167 path = rebuild_path(current_node);
168 //std::cout << "rebuild done\n";
169 }
170 else {
171 // look at the accessible nodes from it
172 //std::cout << System::now() << " looking for accessible states\n";
173 for (int i = 0; i < 8; i++) {
174
175 Node new_node;
176 const int immediate_cost = (i & 1) ? 11 : 8;
177 new_node.previous_cost = current_node->previous_cost + immediate_cost;
178 new_node.location = current_node->location;
179 new_node.location += neighbours_locations[i];
180 new_node.index = get_square_index(new_node.location);
181 //std::cout << " node in direction " << i << ": index = " << new_node.index << std::endl;
182
183 const bool in_closed_list = (closed_list.find(new_node.index) != closed_list.end());
184 if (!in_closed_list && Geometry::get_manhattan_distance(new_node.location, target) < 200
185 && is_node_transition_valid(*current_node, i)) {
186 //std::cout << " node in direction " << i << " is not in the closed list\n";
187 // not in the closed list: look in the open list
188
189 const bool in_open_list = open_list.find(new_node.index) != open_list.end();
190
191 if (!in_open_list) {
192 // not in the open list: add it
193 new_node.heuristic = Geometry::get_manhattan_distance(new_node.location, target);
194 new_node.total_cost = new_node.previous_cost + new_node.heuristic;
195 new_node.parent_index = index;
196 new_node.direction = '0' + i;
197 open_list[new_node.index] = new_node;
198 add_index_sorted(&open_list[new_node.index]);
199 //std::cout << " node in direction " << i << " is not in the open list, adding it with cost "
200 // << new_node.previous_cost << " (" << current_node->previous_cost << " + "
201 // << immediate_cost << ") + " << new_node.heuristic << " and parent " << new_node.parent_index << "\n";
202 }
203 else {
204 //std::cout << " node in direction " << i << " is already in the open list\n";
205 Node* existing_node = &open_list[new_node.index];
206 // already in the open list: see if the current path is better
207 if (new_node.previous_cost < existing_node->previous_cost) {
208 existing_node->previous_cost = new_node.previous_cost;
209 existing_node->total_cost = existing_node->previous_cost + existing_node->heuristic;
210 existing_node->parent_index = index;
211 open_list_indices.sort();
212 }
213 }
214 }
215 else {
216 //std::cout << "skipping node in direction " << i << ": already in closed list = "
217 // << in_closed_list << ", too far from target = "
218 // << (get_manhattan_distance(new_node.location, target) >= 200) << ", invalid transition = "
219 // << (!is_node_transition_valid(*current_node, i)) << std::endl;
220 }
221 }
222 if (open_list_indices.empty()) {
223 finished = true;
224 }
225 }
226 }
227
228 //std::cout << "path found: " << path << ", open nodes: " << open_list.size() << ", closed nodes: " << closed_list.size() << std::endl;
229 return path;
230}

232/**
233 * \brief Returns the index of the 8*8 square in the map
234 * corresponding to the specified location.
235 * \param location location of a node on the map
236 * \return index of the square corresponding to the top-left part of the location
237 */
238int PathFinding::get_square_index(const Point& location) const {
239
240 const int x8 = location.x / 8;
241 const int y8 = location.y / 8;
242 return y8 * map.get_width8() + x8;
243}

245/**
246 * \brief Compares two nodes according to their total estimated cost.
247 * \param other the other node
248 */
249bool PathFinding::Node::operator<(const Node& other) const {
250 return total_cost < other.total_cost;
251}

253/**
254 * \brief Adds the index of a node to the sorted list of indices of the open
255 * list, making sure that the list remains sorted.
256 * \param node The node.
257 */
258void PathFinding::add_index_sorted(Node* node) {
259
260 bool inserted = false;
261 for (auto it = open_list_indices.begin();
262 it != open_list_indices.end() && !inserted;
263 ++it) {
264 const int index = *it;
265 const Node* current_node = &open_list[index];
266 if (current_node->total_cost >= node->total_cost) {
267 open_list_indices.insert(it, node->index);
268 inserted = true;
269 }
270 }
271
272 if (!inserted) {
273 open_list_indices.push_back(node->index);
274 }
275}

277/**
278 * \brief Builds the string representation of the path found by the algorithm.
279 * \param closed_list The closed list of A*.
280 * \return final_node The final node of the path.
281 */
282std::string PathFinding::rebuild_path(const Node* final_node) {
283
284 const Node* current_node = final_node;
285 std::string path = "";
286 while (current_node->direction != ' ') {
287 path = current_node->direction + path;
288 current_node = &closed_list[current_node->parent_index];
289 //std::cout << "current_node: " << current_node->index << ", path = " << path << std::endl;
290 }
291 return path;
292}

294/**
295 * \brief Returns whether a transition between two nodes is valid, i.e.
296 * whether there is no collision with the map.
297 * \param initial_node the first node
298 * \param direction the direction to take (0 to 7)
299 * \return true if there is no collision for this transition
300 */
301bool PathFinding::is_node_transition_valid(
302 const Node& initial_node, int direction) const {
303
304 Rectangle collision_box = transition_collision_boxes[direction];
305 collision_box.add_xy(initial_node.location);
306
307 return !map.test_collision_with_obstacles(source_entity.get_layer(), collision_box, source_entity);
308}
