37/**
38 * @brief Prints the command-line usage.
39 * @param argc Number of arguments of the command line.
40 * @param argv Command-line arguments.
41 * @return 0 in case of success.
42 */
43int print_help(int argc, char* argv[]) {
44
45 std::string binary_name = (argc > 0) ? argv[0] : "solarus";
46 std::cout << "Usage:\n" <<
47 " " << binary_name << " Opens the Solarus launcher GUI\n" <<
48 " " << binary_name << " -help Prints this help message and exits\n" <<
49 " " << binary_name << " -run [/path/to/quest] Plays a quest (default: " << SOLARUS_DEFAULT_QUEST << ")\n" <<
50 " " << binary_name << " -add /path/to/quest Adds a quest to the launcher's list and exits\n";
51 return 0;
52}

54/**
55 * @brief Runs the Solarus GUI.
56 * @param argc Number of arguments of the command line.
57 * @param argv Command-line arguments.
58 * @return 0 in case of success.
59 */
60int run_gui(int argc, char* argv[]) {
61
62 // Set up the application.
63 QApplication application(argc, argv);
64 application.setApplicationName("solarus");
65 application.setApplicationVersion(SOLARUS_VERSION);
66 application.setOrganizationName("solarus");
67
68 // Set up the translations.
69 QTranslator qt_translator;
70 qt_translator.load("qt_" + QLocale::system().name(),
71 QLibraryInfo::location(QLibraryInfo::TranslationsPath));
72 application.installTranslator(&qt_translator);
73
74 QTranslator translator;
75 translator.load("solarus_" + QLocale::system().name());
76 application.installTranslator(&translator);
77
78 MainWindow window(nullptr);
79 window.initialize_geometry_on_screen();
80 window.show();
81
82 return application.exec();
83}

85/**
86 * @brief Runs a quest like the solarus-run executable does.
87 * @param argc Number of arguments of the command line.
88 * @param argv Command-line arguments.
89 * @return 0 in case of success.
90 */
91int run_quest(int argc, char* argv[]) {
92
93 // Abort the process in case of fatal error.
94 Solarus::Debug::set_abort_on_die(true);
95
96 // Run the Solarus main loop.
97 const Solarus::Arguments args(argc, argv);
98 Solarus::MainLoop(args).run();
99
100 return 0;
101}

103/**
104 * @brief Adds a quest to the list of quest if it is not already there.
105 * @param argc Number of arguments of the command line.
106 * @param argv Command-line arguments.
107 * @return 0 in case of success.
108 */
109int add_quest(int argc, char* argv[]) {
110
111 if (argc <= 2) {
112 std::cerr << "Usage: " << ((argc > 0) ? argv[0] : "solarus") << " -add /path/to/quest" << std::endl;
113 return 1;
114 }
115
116 // Set up the application.
117 QApplication application(argc, argv);
118 application.setApplicationName("solarus");
119 application.setApplicationVersion(SOLARUS_VERSION);
120 application.setOrganizationName("solarus");
121
122 QString new_path = argv[2];
123 const QString& canonical_path = QFileInfo(new_path).canonicalFilePath();
124 if (!QFile(canonical_path + "/data/quest.dat").exists() &&
125 !QFile(canonical_path + "/data.solarus").exists() &&
126 !QFile(canonical_path + "/data.solarus.zip").exists()) {
127 std::cerr << "Not a valid Solarus quest: " << new_path.toStdString() << std::endl;
128 return 1;
129 }
130
131 // Add the new path if not already present.
132 Settings test_settings(QSettings::SystemScope);
133 const bool system_settings_writable = test_settings.isWritable();
134
135 Settings settings(system_settings_writable ? QSettings::SystemScope : QSettings::UserScope);
136
137 QStringList all_paths = settings.value("quests_paths").toStringList();
138 for (const QString& path : all_paths) {
139 if (QFileInfo(path).canonicalFilePath() == canonical_path) {
140 std::cout << "Nothing to do, quest already registered: " << new_path.toStdString() << std::endl;
141 return 0;
142 }
143 }
144
145 if (!settings.isWritable()) {
146 std::cerr << "Settings file is not writable: " << settings.fileName().toStdString() << std::endl;
147 return 1;
148 }
149 std::cout << "Writing settings file " << settings.fileName().toStdString() << std::endl;
150
151 all_paths.prepend(canonical_path);
152 settings.setValue("quests_paths", all_paths);
153 settings.setValue("last_quest", new_path);
154 std::cout << "Quest successfully added to Solarus: " << new_path.toStdString() << std::endl;
155
156 return 0;
157}

163/**
164 * @brief Entry point of Solarus.
165 *
166 * To run the Solarus GUI:
167 * solarus
168 * To directly run a quest (no GUI, similar to solarus-run):
169 * solarus -run quest_path
170 * To silently add a quest to the Solarus GUI quest list and exit immediately
171 * (no GUI is run):
172 * solarus -add quest_path
173 *
174 * @param argc Number of arguments of the command line.
175 * @param argv Command-line arguments.
176 * @return 0 in case of success.
177 */
178int main(int argc, char* argv[]) {
179
180 if (argc > 1) {
181 const QString& arg1 = argv[1];
182 if (arg1 == "-help") {
183 return SolarusGui::print_help(argc, argv);
184 }
185 else if (arg1 == "-run") {
186 // Quest run mode.
187 return SolarusGui::run_quest(argc, argv);
188 }
189 else if (arg1 == "-add") {
190 return SolarusGui::add_quest(argc, argv);
191 }
192 }
193
194 // Solarus GUI mode.
195 return SolarusGui::run_gui(argc, argv);
196}
