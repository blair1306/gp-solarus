17#ifndef SOLARUS_TRANSITION_H
18#define SOLARUS_TRANSITION_H
19
20#include "solarus/Common.h"
21#include "solarus/EnumInfo.h"
22#include <map>
23#include <string>
24
25namespace Solarus {
26
27class Game;
28class Surface;
29
30/**
31 * \brief Abstract class for a transition effect between two phases.
32 *
33 * The transitions may be applied to maps or any surface.
34 */
35class SOLARUS_API Transition {
36
37 public:
38
39 /**
40 * \brief Styles of transitions.
41 */
42 enum class Style {
43 IMMEDIATE = 0, /**< No transition between the two surfaces. */
44 FADE = 1, /**< Fade-out and fade-in effect. */
45 SCROLLING = 2 /**< Scrolling between two maps. */
46 };
47
48 /**
49 * \brief Possible directions of a transition.
50 */
51 enum class Direction {
52 OPENING = 0,
53 CLOSING = 1
54 };
55
56 virtual ~Transition();
57 static Transition* create(
58 Style style,
59 Direction direction,
60 Surface& dst_surface,
61 Game* game = nullptr);
62
63 Game* get_game() const;
64 Direction get_direction() const;
65 void set_previous_surface(Surface* previous_surface);
66 virtual bool needs_previous_surface() const;
67
68 bool is_suspended() const;
69 void set_suspended(bool suspended);
70
71 /**
72 * \brief Starts this transition effect.
73 */
74 virtual void start() = 0;
75
76 /**
77 * \brief Returns whether the transition effect is started.
78 * \return true if the transition effect is started
79 */
80 virtual bool is_started() const = 0;
81
82 /**
83 * \brief Returns whether the transition effect is finished.
84 * \return true if the transition effect is finished
85 */
86 virtual bool is_finished() const = 0;
87
88 /**
89 * \brief Updates this transition effect.
90 */
91 virtual void update() = 0;
92
93 /**
94 * \brief Draws the transition effect on a surface.
95 * \param dst_surface the surface to draw
96 */
97 virtual void draw(Surface& dst_surface) = 0;
98
99 protected:
100
101 explicit Transition(Direction direction);
102
103 Surface* get_previous_surface() const;
104
105 uint32_t get_when_suspended() const;
106
107 /**
108 * \brief Notifies the transition effect that it was just suspended
109 * or resumed.
110 */
111 virtual void notify_suspended(bool suspended) = 0;
112
113 private:
114
115 Game* game; /**< The current game if any (required by some kinds of transitions). */
116 Direction direction; /**< Direction of the transition (in or out). */
117 Surface* previous_surface; /**< During an in transition, this is the surface that was displayed
118 * when the out transition was played. */
119 bool suspended; /**< Indicates that the transition is currently paused. */
120 uint32_t when_suspended; /**< Date when the transition was suspended. */
121
122};
123
124template <>
125struct SOLARUS_API EnumInfoTraits<Transition::Style> {
126 static const std::string pretty_name;
127
128 static const EnumInfo<Transition::Style>::names_type names;
129};
130
131}
132
133#endif
