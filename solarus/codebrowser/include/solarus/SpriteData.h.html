31class Point;
32class Size;

34/**
35 * \brief Stores the properties of a sprite animation direction.
36 */
37class SOLARUS_API SpriteAnimationDirectionData {
38
39 public:
40
41 SpriteAnimationDirectionData();
42 SpriteAnimationDirectionData(const Point& xy, const Size& size);
43 SpriteAnimationDirectionData(
44 const Point& xy, const Size& size, const Point& origin,
45 int num_frames, int num_columns = 1);
46
47 Point get_xy() const;
48 void set_xy(const Point& xy);
49
50 Size get_size() const;
51 void set_size(const Size& size);
52
53 Point get_origin() const;
54 void set_origin(const Point& origin);
55
56 Rectangle get_bounding_box() const;
57
58 bool is_multi_frame() const;
59 int get_num_frames() const;
60 void set_num_frames(int num_frames);
61
62 int get_num_columns() const;
63 void set_num_columns(int num_columns);
64
65 Rectangle get_frame() const;
66 std::vector<Rectangle> get_all_frames() const;
67
68 private:
69
70 Point xy; /**< Coordinates of the first frame. */
71 Size size; /**< Size of frames. */
72 Point origin; /**< Coordinates of the origin of frames. */
73 int num_frames; /**< Number of frames. */
74 int num_columns; /**< Number of columns. */
75
76};

78/**
79 * \brief Stores the properties of a sprite animation.
80 */
81class SOLARUS_API SpriteAnimationData {
82
83 public:
84
85 SpriteAnimationData();
86 SpriteAnimationData(const std::string& src_image,
87 std::deque<SpriteAnimationDirectionData>& directions,
88 uint32_t frame_delay, int loop_on_frame);
89
90 bool src_image_is_tileset() const;
91 std::string get_src_image() const;
92 void set_src_image(const std::string& src_image);
93
94 uint32_t get_frame_delay() const;
95 void set_frame_delay(const uint32_t& frame_delay);
96
97 int get_loop_on_frame() const;
98 void set_loop_on_frame(int loop_on_frame);
99
100 int get_num_directions() const;
101 const SpriteAnimationDirectionData& get_direction(int direction_nb) const;
102 SpriteAnimationDirectionData& get_direction(int direction_nb);
103 const std::deque<SpriteAnimationDirectionData>& get_directions() const;
104 void set_directions(const std::deque<SpriteAnimationDirectionData>& directions);
105 void add_direction(const SpriteAnimationDirectionData& direction);
106 bool remove_direction(int direction_nb);
107 bool move_direction(int direction_nb, int new_direction_nb);
108
109 private:
110
111 std::string src_image; /**< Name of the source image. */
112 uint32_t frame_delay; /**< Delay in milliseconds between each frame. */
113 int loop_on_frame; /**< Number of the frame to loop on. */
114
115 std::deque<SpriteAnimationDirectionData>
116 directions; /**< List of directions. */
117
118};

127 SpriteData();
128
129 int get_num_animations() const;
130 const std::map<std::string, SpriteAnimationData>& get_animations() const;
131 bool has_animation(const std::string& animation_name) const;
132 const SpriteAnimationData& get_animation(
133 const std::string& animation_name) const;
134 SpriteAnimationData& get_animation(const std::string& animation_name);
135 bool add_animation(
136 const std::string& animation_name, const SpriteAnimationData& animation);
137 bool remove_animation(const std::string& animation_name);
138 bool set_animation_name(
139 const std::string& old_animation_name, const std::string& new_animation_name);
140
141 std::string get_default_animation_name() const;
142 bool set_default_animation_name(const std::string& animation_name);
143
144 virtual bool import_from_lua(lua_State* l) override;
145 virtual bool export_to_lua(std::ostream& out) const override;
146
147 private:
148
149 static int l_animation(lua_State* l);
150 void export_animation_to_lua(
151 const std::string &animation_name, std::ostream& out) const;
152
153 std::map<std::string, SpriteAnimationData>
154 animations; /**< The sprite animations indexed by their name. */
155 std::string default_animation_name; /**< Name of the default animation. */
156
157};
