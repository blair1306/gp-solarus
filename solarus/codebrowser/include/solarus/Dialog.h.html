17#ifndef SOLARUS_DIALOG_H
18#define SOLARUS_DIALOG_H
19
20#include "solarus/Common.h"
21#include <string>
22#include <map>
23
24namespace Solarus {
25
26/**
27 * \brief A dialog that can be displayed during the game in the dialog box.
28 */
29class Dialog {
30
31 public:
32
33 Dialog();
34
35 const std::string& get_id() const;
36 void set_id(const std::string& id);
37
38 const std::string& get_text() const;
39 void set_text(const std::string& text);
40
41 const std::map<std::string, std::string>& get_properties() const;
42 bool has_property(const std::string& key) const;
43 const std::string& get_property(const std::string& key) const;
44 void set_property(const std::string& key, const std::string& value);
45
46 private:
47
48 std::string id; /**< Id of this dialog. */
49 std::string text; /**< The whole text of this dialog. */
50 std::map<std::string, std::string>
51 properties; /**< Custom properties of this dialog. */
52};
53
54}
55
56#endif
