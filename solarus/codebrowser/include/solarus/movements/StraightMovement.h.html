17#ifndef SOLARUS_STRAIGHT_MOVEMENT_H
18#define SOLARUS_STRAIGHT_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Point.h"
22#include "solarus/movements/Movement.h"
23#include <cstdint>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief A straight movement represented as a speed vector
30 * whose properties (speed and angle) can be changed.
31 */
32class StraightMovement: public Movement {
33
34 public:
35
36 StraightMovement(bool ignore_obstacles, bool smooth);
37
38 virtual void notify_object_controlled() override;
39 virtual void update() override;
40 virtual void set_suspended(bool suspended) override;
41 bool has_to_move_now() const;
42
43 // speed vector
44 double get_x_speed() const;
45 double get_y_speed() const;
46 double get_speed() const;
47 void set_x_speed(double x_speed);
48 void set_y_speed(double y_speed);
49 void set_speed(double speed);
50 double get_angle() const;
51 void set_angle(double angle);
52 int get_max_distance() const;
53 void set_max_distance(int max_distance);
54 bool is_smooth() const;
55 void set_smooth(bool smooth);
56 virtual int get_displayed_direction4() const override;
57
58 // movement
59 virtual bool is_started() const override;
60 virtual bool is_finished() const override;
61 void set_finished();
62 virtual void stop() override;
63
64 virtual const std::string& get_lua_type_name() const override;
65
66 protected:
67
68 void set_next_move_date_x(uint32_t next_move_date_x);
69 void set_next_move_date_y(uint32_t next_move_date_y);
70
71 void update_smooth_xy();
72 void update_smooth_x();
73 void update_smooth_y();
74
75 void update_non_smooth_xy();
76
77 private:
78
79 // speed vector
80 double angle; /**< angle between the speed vector and the horizontal axis in radians */
81 double x_speed; /**< X speed of the object to move in pixels per second.
82 * 0: stopped
83 * positive value: moving to the right
84 * negative value: moving to the left */
85 double y_speed; /**< Y speed of the object to move in pixels per second.
86 * 0: stopped
87 * positive value: moving downwards
88 * negative value: moving upwards */
89
90 uint32_t next_move_date_x; /**< Date of the next x move in ticks. */
91 uint32_t next_move_date_y; /**< Date of the next y move in ticks. */
92
93 // the following fields are redundant and can be computed from x_speed and y_speed
94 uint32_t x_delay; /**< Delay in ticks between an x move of 1 pixel.
95 * x_delay = 200 / |x_speed| */
96 uint32_t y_delay; /**< Delay in ticks between an y move of 1 pixel.
97 * y_delay = 200 / |y_speed| */
98 int x_move; /**< Number of pixels of the next x move : 0, 1 or -1. */
99 int y_move; /**< Number of pixels of the next y move : 0, 1 or -1. */
100
101 Point initial_xy; /**< Initial position when the movement started
102 * (reset whenever the speed of the angle changes) */
103 int max_distance; /**< After this distance in pixels from the initial position,
104 * or when an obstacle is reached, the movement will stop
105 * (0 means no limit) */
106 bool finished; /**< When max_distance is not zero, indicates
107 * that max_distance or an obstacle is reached */
108 bool smooth; /**< Makes the movement adjust its trajectory
109 * when an obstacle is close */
110
111};
112
113}
114
115#endif
