42#ifndef QICON_H
43#define QICON_H
44
45#include <QtCore/qglobal.h>
46#include <QtCore/qsize.h>
47#include <QtCore/qlist.h>
48#include <QtGui/qpixmap.h>
49
50QT_BEGIN_NAMESPACE
51
52
53class QIconPrivate;
54class QIconEngine;
55
56class Q_GUI_EXPORT QIcon
57{
58public:
59 enum Mode { Normal, Disabled, Active, Selected };
60 enum State { On, Off };
61
62 QIcon();
63 QIcon(const QPixmap &pixmap);
64 QIcon(const QIcon &other);
65#ifdef Q_COMPILER_RVALUE_REFS
66 QIcon(QIcon &&other)
67 :d(0) { qSwap(d, other.d); }
68#endif
69 explicit QIcon(const QString &fileName); // file or resource name
70 explicit QIcon(QIconEngine *engine);
71 ~QIcon();
72 QIcon &operator=(const QIcon &other);
73#ifdef Q_COMPILER_RVALUE_REFS
74 inline QIcon &operator=(QIcon &&other)
75 { qSwap(d, other.d); return *this; }
76#endif
77 inline void swap(QIcon &other) { qSwap(d, other.d); }
78
79 operator QVariant() const;
80
81 QPixmap pixmap(const QSize &size, Mode mode = Normal, State state = Off) const;
82 inline QPixmap pixmap(int w, int h, Mode mode = Normal, State state = Off) const
83 { return pixmap(QSize(w, h), mode, state); }
84 inline QPixmap pixmap(int extent, Mode mode = Normal, State state = Off) const
85 { return pixmap(QSize(extent, extent), mode, state); }
86 QPixmap pixmap(QWindow *window, const QSize &size, Mode mode = Normal, State state = Off) const;
87
88 QSize actualSize(const QSize &size, Mode mode = Normal, State state = Off) const;
89 QSize actualSize(QWindow *window, const QSize &size, Mode mode = Normal, State state = Off) const;
90
91 QString name() const;
92
93 void paint(QPainter *painter, const QRect &rect, Qt::Alignment alignment = Qt::AlignCenter, Mode mode = Normal, State state = Off) const;
94 inline void paint(QPainter *painter, int x, int y, int w, int h, Qt::Alignment alignment = Qt::AlignCenter, Mode mode = Normal, State state = Off) const
95 { paint(painter, QRect(x, y, w, h), alignment, mode, state); }
96
97 bool isNull() const;
98 bool isDetached() const;
99 void detach();
100
101#if QT_DEPRECATED_SINCE(5, 0)
102 QT_DEPRECATED inline int serialNumber() const { return cacheKey() >> 32; }
103#endif
104 qint64 cacheKey() const;
105
106 void addPixmap(const QPixmap &pixmap, Mode mode = Normal, State state = Off);
107 void addFile(const QString &fileName, const QSize &size = QSize(), Mode mode = Normal, State state = Off);
108
109 QList<QSize> availableSizes(Mode mode = Normal, State state = Off) const;
110
111 static QIcon fromTheme(const QString &name, const QIcon &fallback = QIcon());
112 static bool hasThemeIcon(const QString &name);
113
114 static QStringList themeSearchPaths();
115 static void setThemeSearchPaths(const QStringList &searchpath);
116
117 static QString themeName();
118 static void setThemeName(const QString &path);
119
120 Q_DUMMY_COMPARISON_OPERATOR(QIcon)
121
122private:
123 QIconPrivate *d;
124#if !defined(QT_NO_DATASTREAM)
125 friend Q_GUI_EXPORT QDataStream &operator<<(QDataStream &, const QIcon &);
126 friend Q_GUI_EXPORT QDataStream &operator>>(QDataStream &, QIcon &);
127#endif
128
129public:
130 typedef QIconPrivate * DataPtr;
131 inline DataPtr &data_ptr() { return d; }
132};
133
134Q_DECLARE_SHARED(QIcon)
135
136#if !defined(QT_NO_DATASTREAM)
137Q_GUI_EXPORT QDataStream &operator<<(QDataStream &, const QIcon &);
138Q_GUI_EXPORT QDataStream &operator>>(QDataStream &, QIcon &);
139#endif
140
141#ifndef QT_NO_DEBUG_STREAM
142Q_GUI_EXPORT QDebug operator<<(QDebug dbg, const QIcon &);
143#endif
144
145QT_END_NAMESPACE
146
147#endif // QICON_H
