42#ifndef QINPUTMETHOD_H
43#define QINPUTMETHOD_H
44
45#include <QtCore/qobject.h>
46
47QT_BEGIN_NAMESPACE
48
49class QInputMethodPrivate;
50class QWindow;
51class QRectF;
52class QTransform;
53class QInputMethodQueryEvent;
54
55class Q_GUI_EXPORT QInputMethod : public QObject
56{
57 Q_OBJECT
58 Q_DECLARE_PRIVATE(QInputMethod)
59 Q_PROPERTY(QRectF cursorRectangle READ cursorRectangle NOTIFY cursorRectangleChanged)
60 Q_PROPERTY(QRectF keyboardRectangle READ keyboardRectangle NOTIFY keyboardRectangleChanged)
61 Q_PROPERTY(bool visible READ isVisible NOTIFY visibleChanged)
62 Q_PROPERTY(bool animating READ isAnimating NOTIFY animatingChanged)
63 Q_PROPERTY(QLocale locale READ locale NOTIFY localeChanged)
64 Q_PROPERTY(Qt::LayoutDirection inputDirection READ inputDirection NOTIFY inputDirectionChanged)
65
66 Q_ENUMS(Action)
67public:
68 QTransform inputItemTransform() const;
69 void setInputItemTransform(const QTransform &transform);
70
71 QRectF inputItemRectangle() const;
72 void setInputItemRectangle(const QRectF &rect);
73
74 // in window coordinates
75 QRectF cursorRectangle() const; // ### what if we have rotations for the item?
76
77 // keyboard geometry in window coords
78 QRectF keyboardRectangle() const;
79
80 enum Action {
81 Click,
82 ContextMenu
83 };
84
85 bool isVisible() const;
86 void setVisible(bool visible);
87
88 bool isAnimating() const;
89
90 QLocale locale() const;
91 Qt::LayoutDirection inputDirection() const;
92
93 static QVariant queryFocusObject(Qt::InputMethodQuery query, QVariant argument);
94
95public Q_SLOTS:
96 void show();
97 void hide();
98
99 void update(Qt::InputMethodQueries queries);
100 void reset();
101 void commit();
102
103 void invokeAction(Action a, int cursorPosition);
104
105Q_SIGNALS:
106 void cursorRectangleChanged();
107 void keyboardRectangleChanged();
108 void visibleChanged();
109 void animatingChanged();
110 void localeChanged();
111 void inputDirectionChanged(Qt::LayoutDirection newDirection);
112
113private:
114 friend class QGuiApplication;
115 friend class QGuiApplicationPrivate;
116 friend class QPlatformInputContext;
117 QInputMethod();
118 ~QInputMethod();
119};
120
121QT_END_NAMESPACE
122
123#endif
