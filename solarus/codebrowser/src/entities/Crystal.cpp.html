33/**
34 * \brief Creates a new crystal.
35 * \param name name of the entity to create
36 * \param layer layer of the entity to create on the map
37 * \param xy Coordinates of the entity to create.
38 */
39Crystal::Crystal(const std::string& name, int layer, const Point& xy):
40 Entity(name, 0, layer, xy, Size(16, 16)),
41 state(false),
42 next_possible_hit_date(System::now()) {
43
44 set_collision_modes(
45 CollisionMode::COLLISION_SPRITE |
46 CollisionMode::COLLISION_OVERLAPPING |
47 CollisionMode::COLLISION_FACING
48 );
49 set_origin(8, 13);
50 main_sprite = create_sprite("entities/crystal", "main");
51 main_sprite->enable_pixel_collisions();
52 star_sprite = create_sprite("entities/star", "star");
53 star_sprite->enable_pixel_collisions();
54 twinkle();
55}

57/**
58 * \brief Returns the type of entity.
59 * \return the type of entity
60 */
61EntityType Crystal::get_type() const {
62 return ThisType;
63}

65/**
66 * \copydoc Entity::notify_creating
67 */
68void Crystal::notify_creating() {
69
70 Entity::notify_creating();
71
72 bool state = get_game().get_crystal_state();
73 if (state != this->state) {
74
75 this->state = state;
76 main_sprite->set_current_animation(state ? "blue_lowered" : "orange_lowered");
77 }
78}

80/**
81 * \brief Returns whether this entity is an obstacle for another one.
82 * \param other another entity

85bool Crystal::is_obstacle_for(Entity& other) {
86 return other.is_crystal_obstacle(*this);
87}

89/**
90 * \brief This function is called when another entity collides with this crystal.
91 * \param entity_overlapping the other entity
92 * \param collision_mode the collision mode that detected the collision
93 */
94void Crystal::notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) {
95 entity_overlapping.notify_collision_with_crystal(*this, collision_mode);
96}

98/**
99 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
100 */
101void Crystal::notify_collision(Entity& other_entity, Sprite& this_sprite, Sprite& other_sprite) {
102
103 if (&this_sprite == main_sprite.get()) {
104 other_entity.notify_collision_with_crystal(*this, other_sprite);
105 }
106}

108/**
109 * \copydoc Entity::notify_action_command_pressed
110 */
111bool Crystal::notify_action_command_pressed() {
112
113 if (get_hero().is_free() &&
114 get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_LOOK
115 ) {
116 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
117
118 // start a dialog
119 get_game().start_dialog("_crystal", ScopedLuaRef(), ScopedLuaRef());
120 return true;
121 }
122
123 return false;
124}

126/**
127 * \brief Activates the crystal if the delay since the last activation allows it.
128 * \param entity_activating the entity that activates this crystal
129 */
130void Crystal::activate(Entity& entity_activating) {
131
132 bool recently_activated = false;
133 for (Entity* entity: entities_activating) {
134 if (entity == &entity_activating) {
135 recently_activated = true;
136 break;
137 }
138 }
139
140 uint32_t now = System::now();
141 if (!recently_activated || now >= next_possible_hit_date) {
142 Sound::play("switch");
143 get_game().change_crystal_state();
144 next_possible_hit_date = now + 1000;
145 entities_activating.push_back(&entity_activating);
146 }
147}

149/**
150 * \brief Makes a star twinkle on the crystal at a random position.
151 */
152void Crystal::twinkle() {
153
154 const Point star_xy = { Random::get_number(3, 13), Random::get_number(3, 13) };
155 star_sprite->restart_animation();
156 star_sprite->set_xy(star_xy - get_origin());
157}

159/**
160 * \brief Updates the entity.
161 */
162void Crystal::update() {
163
164 if (!is_suspended()) {
165 bool state = get_game().get_crystal_state();
166 if (state != this->state) {
167
168 this->state = state;
169 main_sprite->set_current_animation(state ? "blue_lowered" : "orange_lowered");
170 }
171
172 star_sprite->update();
173 if (star_sprite->is_animation_finished()) {
174 twinkle();
175 }
176
177 uint32_t now = System::now();
178 if (now >= next_possible_hit_date) {
179 entities_activating.clear();
180 }
181 }
182
183 Entity::update();
184}
185
186/**
187 * \brief Suspends or resumes the entity.
188 * \param suspended true to suspend the entity, false to resume it
189 */
190void Crystal::set_suspended(bool suspended) {
191
192 Entity::set_suspended(suspended);
193
194 if (!suspended) {
195 next_possible_hit_date += System::now() - get_when_suspended();
196 }
197}
