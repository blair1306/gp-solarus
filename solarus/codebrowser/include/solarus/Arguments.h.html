17#ifndef SOLARUS_ARGUMENTS_H
18#define SOLARUS_ARGUMENTS_H
19
20#include "solarus/Common.h"
21#include <string>
22#include <vector>
23
24namespace Solarus {
25
26/**
27 * \brief Stores runtime arguments passed to the program.
28 *
29 * Arguments may come from the command-line or be set explicitly.
30 * Provides easy access to find whether a particular option was passed,
31 * and to get the value of options of the form key=value.
32 */
33class SOLARUS_API Arguments {
34
35 public:
36
37 Arguments();
38 Arguments(int argc, char** argv);
39
40 const std::string& get_program_name() const;
41 void set_program_name(const std::string& program_name);
42 const std::vector<std::string>& get_arguments() const;
43 bool has_argument(const std::string& option) const;
44 std::string get_argument_value(const std::string& key) const;
45
46 void add_argument(const std::string& argument);
47 void add_argument(const std::string& key, const std::string& value);
48
49 private:
50
51 std::string program_name;
52 std::vector<std::string> args;
53
54};
55
56}
57
58#endif
