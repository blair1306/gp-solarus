238/**
239 * \brief Returns the pattern with the specified id.
240 * \param pattern_id A tile pattern id. It must exist.
241 * \return The pattern with this id.
242 * The object remains valid until tile patterns are added or removed.
243 */
244const TilePatternData& TilesetData::get_pattern(const std::string& pattern_id) const {
245
246 const auto& it = patterns.find(pattern_id);
247 Debug::check_assertion(it != patterns.end(),
248 std::string("No such tile pattern: '") + pattern_id + "'");

250 return it->second;

271/**
272 * \brief Adds a pattern to the tileset.
273 * \param pattern_id Id of the new pattern.
274 * \param pattern The pattern to add.
275 * \return \c true in case of success.
276 */
277bool TilesetData::add_pattern(
278 const std::string& pattern_id, const TilePatternData& pattern) {
279
280 const auto& result = patterns.emplace(pattern_id, pattern);
281 if (!result.second) {
282 // Insertion failed: the id already exists.
283 return false;
284 }
285
286 return true;
287}
288
289/**
290 * \brief Removes a pattern from the tileset.
291 * \param pattern_id Id of the pattern to remove.
292 * \return \c true in case of success.
293 */
294bool TilesetData::remove_pattern(const std::string& pattern_id) {
295
296 return patterns.erase(pattern_id) > 0;
297}
298
299/**
300 * \brief Changes the id of a pattern in the tileset.
301 * \param old_pattern_id Old id of the pattern.
302 * \param new_pattern_id New id to set.
303 * \return \c true in case of success.
304 * In case of failure, the old pattern is unchanged.
305 */
306bool TilesetData::set_pattern_id(
307 const std::string& old_pattern_id, const std::string& new_pattern_id) {
308
309 if (!pattern_exists(old_pattern_id)) {
310 // No pattern was found with the old id.
311 return false;
312 }
313
314 if (pattern_exists(new_pattern_id)) {
315 // The new id is already used.
316 return false;
317 }
318
319 TilePatternData pattern = get_pattern(old_pattern_id);
320 remove_pattern(old_pattern_id);
321 add_pattern(new_pattern_id, pattern);
322
323 return true;
324}

441/**
442 * \brief Function called by Lua to set the background color of the tileset.
443 *
444 * - Argument 1 (table): background color (must be an array of 3 integers).
445 *
446 * \param l The Lua context that is calling this function.
447 * \return Number of values to return to Lua.
448 */
449int l_background_color(lua_State* l) {
450
451 return LuaTools::exception_boundary_handle(l, [&] {
452 lua_getfield(l, LUA_REGISTRYINDEX, "tileset");
453 TilesetData& tileset = *static_cast<TilesetData*>(lua_touserdata(l, -1));
454 lua_pop(l, 1);
455
456 const Color& background_color = LuaTools::check_color(l, 1);
457
458 tileset.set_background_color(background_color);
459
460 return 0;
461 });

464/**
465 * \brief Function called by Lua to add a tile pattern to the tileset.
466 *
467 * - Argument 1 (table): A table describing the tile pattern to create.
468 *
469 * \param l The Lua context that is calling this function.
470 * \return Number of values to return to Lua.
471 */
472int l_tile_pattern(lua_State* l) {
473
474 return LuaTools::exception_boundary_handle(l, [&] {
475 lua_getfield(l, LUA_REGISTRYINDEX, "tileset");
476 TilesetData& tileset_data = *static_cast<TilesetData*>(lua_touserdata(l, -1));
477 lua_pop(l, 1);
478
479 TilePatternData pattern_data;
480
481 const std::string& id = LuaTools::check_string_field(l, 1, "id");
482
483 const Ground ground = LuaTools::check_enum_field<Ground>(
484 l, 1, "ground"
485 );
486 pattern_data.set_ground(ground);
487
488 const int default_layer = LuaTools::check_int_field(
489 l, 1, "default_layer"
490 );
491 pattern_data.set_default_layer(default_layer);
492
493 const TileScrolling scrolling = LuaTools::opt_enum_field<TileScrolling>(
494 l, 1, "scrolling", TileScrolling::NONE
495 );
496 pattern_data.set_scrolling(scrolling);
497
498 const TilePatternRepeatMode repeat_mode = LuaTools::opt_enum_field<TilePatternRepeatMode>(
499 l, 1, "repeat_mode", TilePatternRepeatMode::ALL
500 );
501 pattern_data.set_repeat_mode(repeat_mode);
502
503 const int width = LuaTools::check_int_field(l, 1, "width");
504 const int height = LuaTools::check_int_field(l, 1, "height");
505 // Start with the maximum number of frames.
506 std::vector<Rectangle> frames(4, Rectangle(0, 0, width, height));
507
508 int i = 0, j = 0;
509 lua_settop(l, 1);
510 lua_getfield(l, 1, "x");
511 if (lua_isnumber(l, 2)) {
512 // Single frame.
513 frames[0].set_x(LuaTools::check_int(l, 2));
514 i = 1;
515 }
516 else {
517 // Multi-frame.
518 lua_pushnil(l);
519 while (lua_next(l, 2) != 0 && i < 4) {
520 frames[i].set_x(LuaTools::check_int(l, 4));
521 ++i;
522 lua_pop(l, 1);
523 }
524 }
525 lua_pop(l, 1);
526 Debug::check_assertion(lua_gettop(l) == 1, "Invalid stack when parsing tile pattern");
527
528 lua_getfield(l, 1, "y");
529 if (lua_isnumber(l, 2)) {
530 // Single frame.
531 frames[0].set_y(LuaTools::check_int(l, 2));
532 j = 1;
533 }
534 else {
535 // Multi-frame.
536 lua_pushnil(l);
537 while (lua_next(l, 2) != 0 && j < 4) {
538 frames[j].set_y(LuaTools::check_int(l, 4));
539 ++j;
540 lua_pop(l, 1);
541 }
542 }
543 lua_pop(l, 1);
544 Debug::check_assertion(lua_gettop(l) == 1, "Invalid stack when parsing tile pattern");
545
546 // Check data.
547 if (i != 1 && i != 3 && i != 4) {
548 LuaTools::arg_error(l, 1, "Invalid number of frames for x");
549 }
550 if (j != 1 && j != 3 && j != 4) {
551 LuaTools::arg_error(l, 1, "Invalid number of frames for y");
552 }
553 if (i != j) {
554 LuaTools::arg_error(l, 1, "The length of x and y must match");
555 }
556 frames.resize(i);
557 pattern_data.set_frames(frames);
558
559 tileset_data.add_pattern(id, pattern_data);
560
561 return 0;
562 });
563}

565/**
566 * \brief Function called by Lua to add a border set to the tileset.
567 *
568 * - Argument 1 (table): A table describing the border set to create.
569 *
570 * \param l The Lua context that is calling this function.
571 * \return Number of values to return to Lua.
572 */
573int l_border_set(lua_State* l) {
574
575 return LuaTools::exception_boundary_handle(l, [&] {
576 lua_getfield(l, LUA_REGISTRYINDEX, "tileset");
577 TilesetData& tileset_data = *static_cast<TilesetData*>(lua_touserdata(l, -1));
578 lua_pop(l, 1);
579
580 BorderSet border_set;
581
582 const std::string& id = LuaTools::check_string_field(l, 1, "id");
583
584 bool inner = LuaTools::opt_boolean_field(l, 1, "inner", false);
585 border_set.set_inner(inner);
586
587 std::map<BorderKind, std::string> patterns;
588 for (int i = 0; i < 12; ++i) {
589 std::ostringstream oss;
590 oss << "pattern_" << i;
591 const std::string& pattern_id = LuaTools::opt_string_field(l, 1, oss.str(), "");
592 if (!pattern_id.empty()) {
593 BorderKind border_kind = static_cast<BorderKind>(i);
594 border_set.set_pattern(border_kind, pattern_id);
595 }
596 }
597
598 tileset_data.add_border_set(id, border_set);
599
600 return 0;
601 });
602}

606/**
607 * \copydoc LuaData::import_from_lua
608 */
609bool TilesetData::import_from_lua(lua_State* l) {
610
611 lua_pushlightuserdata(l, this);
612 lua_setfield(l, LUA_REGISTRYINDEX, "tileset");
613 lua_register(l, "background_color", l_background_color);
614 lua_register(l, "tile_pattern", l_tile_pattern);
615 lua_register(l, "border_set", l_border_set);
616 if (lua_pcall(l, 0, 0, 0) != 0) {
617 Debug::error(std::string("Failed to load tileset: ") + lua_tostring(l, -1));
618 lua_pop(l, 1);
619 return false;
620 }
621
622 return true;
623}

625/**
626 * \copydoc LuaData::export_to_lua
627 */
628bool TilesetData::export_to_lua(std::ostream& out) const {
629
630 // Background color.
631 uint8_t r, g, b, a;
632 background_color.get_components(r, g, b, a);
633 out << "background_color{ "
634 << static_cast<int>(r)
635 << ", "
636 << static_cast<int>(g)
637 << ", "
638 << static_cast<int>(b)
639 << " }\n";
640
641 // Tile patterns.
642 for (const auto& kvp : patterns) {
643 const std::string& id = kvp.first;
644 const TilePatternData& pattern = kvp.second;
645
646 const Rectangle& first_frame = pattern.get_frame();
647 int width = first_frame.get_width();
648 int height = first_frame.get_height();
649 std::ostringstream x;
650 std::ostringstream y;
651 if (!pattern.is_multi_frame()) {
652 x << first_frame.get_x();
653 y << first_frame.get_y();
654 }
655 else {
656 x << "{ ";
657 y << "{ ";
658 bool first = true;
659 for (const Rectangle& frame : pattern.get_frames()) {
660 if (first) {
661 first = false;
662 }
663 else {
664 x << ", ";
665 y << ", ";
666 }
667 x << frame.get_x();
668 y << frame.get_y();
669 }
670 x << " }";
671 y << " }";
672 }
673
674 const std::string& ground_name = enum_to_name(pattern.get_ground());
675 int default_layer = static_cast<int>(pattern.get_default_layer());
676
677 out << "tile_pattern{\n"
678 << " id = \"" << escape_string(id) << "\",\n"
679 << " ground = \"" << ground_name << "\",\n"
680 << " default_layer = " << default_layer << ",\n"
681 << " x = " << x.str() << ",\n"
682 << " y = " << y.str() << ",\n"
683 << " width = " << width << ",\n"
684 << " height = " << height << ",\n";
685 if (pattern.get_scrolling() != TileScrolling::NONE) {
686 const std::string& scolling_name = enum_to_name(pattern.get_scrolling());
687 out << " scrolling = \"" << scolling_name << "\",\n";
688 }
689 if (pattern.get_repeat_mode() != TilePatternRepeatMode::ALL) {
690 const std::string& repeat_mode_name = enum_to_name(pattern.get_repeat_mode());
691 out << " repeat_mode = \"" << repeat_mode_name << "\",\n";
692 }
693 out << "}\n\n";
694 }
695
696 // Border sets.
697 for (const auto& kvp : border_sets) {
698 const std::string& id = kvp.first;
699 const BorderSet& border_set = kvp.second;
700
701 out << "border_set{\n"
702 << " id = \"" << escape_string(id) << "\",\n";
703 if (border_set.is_inner()) {
704 out << " inner = true,\n";
705 }
706 for (int i = 0; i < 12; ++i) {
707 BorderKind border_kind = static_cast<BorderKind>(i);
708 const std::string& pattern_id = border_set.get_pattern(border_kind);
709 if (!pattern_id.empty()) {
710 out << " pattern_" << i << " = \"" << pattern_id << "\",\n";
711 }
712 }
713
714 out << "}\n\n";
715 }
716
717 return true;
718}
