42/**
43 * \brief Represents a map where the game can take place.
44 *
45 * A map is a game scene. It contains various information, including:
46 * - its dimensions,
47 * - the tileset,

49 * - the ground of each 8x8 square,
50 * - the background music.

220 // map state

222 bool started; /**< Whether this map is the current map. */
223 std::string destination_name; /**< Current destination point on the map,

227 * or an empty string to use the one saved. */
228
229 std::unique_ptr<Entities>
230 entities; /**< The entities on the map. */
231 bool suspended; /**< Whether the game is suspended. */

234/**
235 * \brief Tests whether a point is outside the map area.
236 * \param x x of the point to check.
237 * \param y y of the point to check.
238 * \return true if this point is outside the map area.
239 */
240inline bool Map::test_collision_with_border(int x, int y) const {
241
242 return (x < 0 || y < 0 || x >= location.get_width() || y >= location.get_height());
243}
244
245/**
246 * \brief Tests whether a point is outside the map area.
247 * \param point point to check.
248 * \return true if this point is outside the map area.
249 */
250inline bool Map::test_collision_with_border(const Point& point) const {
251
252 return test_collision_with_border(point.x, point.y);
253}
