17#ifndef SOLARUS_PATH_FINDING_MOVEMENT_H
18#define SOLARUS_PATH_FINDING_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityPtr.h"
22#include "solarus/movements/PathMovement.h"
23#include <cstdint>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief Movement for an entity that looks for a path to another entity.
30 *
31 * This movement is typically used by enemies that try to reach the hero.
32 * The entity tries to find a path and to avoid the obstacles on the way.
33 * To this end, the PathFinding class (i.e. an implementation of the A* algorithm) is used.
34 * If the target entity is too far or not reachable, the movement is a random walk.
35 */
36class SOLARUS_API PathFindingMovement: public PathMovement {
37
38 public:
39
40 explicit PathFindingMovement(int speed);
41
42 void set_target(const EntityPtr& target);
43 virtual bool is_finished() const override;
44
45 virtual const std::string& get_lua_type_name() const override;
46
47 protected:
48
49 virtual void update() override;
50 void recompute_movement();
51
52 private:
53
54 EntityPtr target; /**< the entity targeted by this movement (usually the hero) */
55 uint32_t next_recomputation_date;
56
57};
58
59}
60
61#endif
