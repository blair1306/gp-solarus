27 * Name of the Lua table representing the video module.

29const std::string LuaContext::video_module_name = "sol.video";

31/**
32 * \brief Initializes the video features provided to Lua.
33 */
34void LuaContext::register_video_module() {
35
36 static const luaL_Reg functions[] = {
37 { "get_window_title", video_api_get_window_title },
38 { "set_window_title", video_api_set_window_title },
39 { "get_mode", video_api_get_mode },
40 { "set_mode", video_api_set_mode },
41 { "switch_mode", video_api_switch_mode },
42 { "is_mode_supported", video_api_is_mode_supported },
43 { "get_modes", video_api_get_modes },
44 { "is_fullscreen", video_api_is_fullscreen },
45 { "set_fullscreen", video_api_set_fullscreen },
46 { "is_cursor_visible", video_api_is_cursor_visible },
47 { "set_cursor_visible", video_api_set_cursor_visible },
48 { "get_quest_size", video_api_get_quest_size },
49 { "get_window_size", video_api_get_window_size },
50 { "set_window_size", video_api_set_window_size },
51 { "reset_window_size", video_api_reset_window_size },
52 { nullptr, nullptr }
53 };
54 register_functions(video_module_name, functions);
55}

57/**
58 * \brief Implementation of sol.video.get_window_title().
59 * \param l the Lua context that is calling this function
60 * \return number of values to return to Lua
61 */
62int LuaContext::video_api_get_window_title(lua_State *l) {
63
64 return LuaTools::exception_boundary_handle(l, [&] {
65 const std::string& window_title =
66 Video::get_window_title();
67
68 push_string(l, window_title);
69 return 1;
70 });
71}

103/**
104 * \brief Implementation of sol.video.set_mode().
105 * \param l the Lua context that is calling this function
106 * \return Number of values to return to Lua.
107 */
108int LuaContext::video_api_set_mode(lua_State *l) {
109
110 return LuaTools::exception_boundary_handle(l, [&] {
111 std::string mode_name = LuaTools::check_string(l, 1);
112 const VideoMode* mode = Video::get_video_mode_by_name(mode_name);
113
114 if (mode != nullptr && Video::get_video_mode().get_name() != mode_name) {
115 Video::set_video_mode(*mode);
116 }
117
118 return 0;
119 });
120}

136/**
137 * \brief Implementation of sol.video.get_modes().
138 * \param l the Lua context that is calling this function
139 * \return number of values to return to Lua
140 */
141int LuaContext::video_api_get_modes(lua_State* l) {
142
143 return LuaTools::exception_boundary_handle(l, [&] {
144 const std::vector<const VideoMode*>& modes =
145 Video::get_video_modes();
146
147 lua_newtable(l);
148
149 int i = 1;
150 for (const VideoMode* mode: modes) {
151 push_string(l, mode->get_name());
152 lua_rawseti(l, -2, i);
153 ++i;
154 }
155
156 return 1;
157 });
158}

272/**
273 * \brief Implementation of sol.video.set_window_size().
274 * \param l The Lua context that is calling this function.
275 * \return Number of values to return to Lua.
276 */
277int LuaContext::video_api_set_window_size(lua_State* l) {
278
279 return LuaTools::exception_boundary_handle(l, [&] {
280 int width = LuaTools::check_int(l, 1);
281 int height = LuaTools::check_int(l, 2);
282
283 if (width <= 0) {
284 LuaTools::arg_error(l, 1, "Window width must be positive");
285 }
286 if (height <= 0) {
287 LuaTools::arg_error(l, 2, "Window height must be positive");
288 }
289
290 Video::set_window_size(Size(width, height));
291
292 return 0;
293 });
294}
