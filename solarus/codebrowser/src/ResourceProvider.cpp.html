27/**
28 * \brief Provides the tileset with the given id.
29 * \param tileset_id A tileset id.
30 * \return The corresponding tileset.
31 */
32const Tileset& ResourceProvider::get_tileset(const std::string& tileset_id) {
33
34 auto it = tileset_cache.find(tileset_id);
35 if (it != tileset_cache.end()) {
36 return *it->second;
37 }
38
39 it = tileset_cache.emplace(
40 tileset_id,
41 std::unique_ptr<Tileset>(new Tileset(tileset_id))
42 ).first;
43 Tileset& tileset = *it->second;
44 tileset.load();
45 return tileset;
46}
47
48/**
49 * \brief Notifies the resource provider that cached data (if any) is no longer valid.
50 *
51 * This function must be called when a resource element has changed on disk.
52 *
53 * \param resource_type Type of resource that has changed.
54 * \param element_id Resource element that has changed.
55 */
56void ResourceProvider::invalidate_resource_element(
57 ResourceType resource_type,
58 const std::string& element_id) {
59
60 switch (resource_type) {
61
62 case ResourceType::TILESET:
63 tileset_cache.erase(element_id);
64 break;
65
66 default:
67 break;
68 }
69}
