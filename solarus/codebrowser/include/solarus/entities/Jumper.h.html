17#ifndef SOLARUS_JUMPER_H
18#define SOLARUS_JUMPER_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <string>
23
24namespace Solarus {
25
26/**
27 * \brief A sensor that makes the hero jump.
28 *
29 * A jumper is a detector placed on a map.
30 * When the hero touches this detector, he makes a jump with a specified length.
31 */
32class Jumper: public Entity {
33
34 public:
35
36 static constexpr EntityType ThisType = EntityType::JUMPER;
37
38 Jumper(
39 const std::string& name,
40 int layer,
41 const Point& xy,
42 const Size& size,
43 int direction,
44 int jump_length
45 );
46
47 virtual EntityType get_type() const override;
48 virtual bool can_be_drawn() const override;
49 virtual bool is_sword_ignored() const override;
50 virtual bool is_obstacle_for(Entity& other, const Rectangle& candidate_position) override;
51 virtual bool test_collision_custom(Entity& entity) override;
52 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
53
54 int get_jump_length() const;
55 bool is_jump_horizontal() const;
56 bool is_jump_vertical() const;
57 bool is_jump_diagonal() const;
58 bool is_in_jump_position(
59 const Hero& hero,
60 const Rectangle& candidate_position,
61 bool extended_region
62 ) const;
63
64 bool overlaps_jumping_region(
65 const Rectangle& rectangle,
66 bool extended_region
67 ) const;
68
69 private:
70
71 bool is_point_in_diagonal(const Point& point) const;
72 bool is_point_in_extended_diagonal(const Point& point) const;
73
74 int jump_length; /**< length of the jump in pixels (should be a multiple of 8) */
75
76};
77
78}
79
80#endif
