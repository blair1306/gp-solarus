42#ifndef QACTIONGROUP_H
43#define QACTIONGROUP_H
44
45#include <QtWidgets/qaction.h>
46
47QT_BEGIN_NAMESPACE
48
49
50#ifndef QT_NO_ACTION
51
52class QActionGroupPrivate;
53
54class Q_WIDGETS_EXPORT QActionGroup : public QObject
55{
56 Q_OBJECT
57 Q_DECLARE_PRIVATE(QActionGroup)
58
59 Q_PROPERTY(bool exclusive READ isExclusive WRITE setExclusive)
60 Q_PROPERTY(bool enabled READ isEnabled WRITE setEnabled)
61 Q_PROPERTY(bool visible READ isVisible WRITE setVisible)
62
63public:
64 explicit QActionGroup(QObject* parent);
65 ~QActionGroup();
66
67 QAction *addAction(QAction* a);
68 QAction *addAction(const QString &text);
69 QAction *addAction(const QIcon &icon, const QString &text);
70 void removeAction(QAction *a);
71 QList<QAction*> actions() const;
72
73 QAction *checkedAction() const;
74 bool isExclusive() const;
75 bool isEnabled() const;
76 bool isVisible() const;
77
78
79public Q_SLOTS:
80 void setEnabled(bool);
81 inline void setDisabled(bool b) { setEnabled(!b); }
82 void setVisible(bool);
83 void setExclusive(bool);
84
85Q_SIGNALS:
86 void triggered(QAction *);
87 void hovered(QAction *);
88
89private:
90 Q_DISABLE_COPY(QActionGroup)
91 Q_PRIVATE_SLOT(d_func(), void _q_actionTriggered())
92 Q_PRIVATE_SLOT(d_func(), void _q_actionChanged())
93 Q_PRIVATE_SLOT(d_func(), void _q_actionHovered())
94};
95
96#endif // QT_NO_ACTION
97
98QT_END_NAMESPACE
99
100#endif // QACTIONGROUP_H
