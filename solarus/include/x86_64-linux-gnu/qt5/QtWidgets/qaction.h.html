45#include <QtGui/qkeysequence.h>

47#include <QtWidgets/qwidget.h>
48#include <QtCore/qvariant.h>
49#include <QtGui/qicon.h>

51QT_BEGIN_NAMESPACE
52
53
54#ifndef QT_NO_ACTION
55
56class QMenu;
57class QActionGroup;
58class QActionPrivate;
59class QGraphicsWidget;
60
61class Q_WIDGETS_EXPORT QAction : public QObject
62{
63 Q_OBJECT
64 Q_DECLARE_PRIVATE(QAction)
65
66 Q_ENUMS(MenuRole)
67 Q_ENUMS(Priority)
68 Q_PROPERTY(bool checkable READ isCheckable WRITE setCheckable NOTIFY changed)
69 Q_PROPERTY(bool checked READ isChecked WRITE setChecked DESIGNABLE isCheckable NOTIFY toggled)
70 Q_PROPERTY(bool enabled READ isEnabled WRITE setEnabled NOTIFY changed)
71 Q_PROPERTY(QIcon icon READ icon WRITE setIcon NOTIFY changed)
72 Q_PROPERTY(QString text READ text WRITE setText NOTIFY changed)
73 Q_PROPERTY(QString iconText READ iconText WRITE setIconText NOTIFY changed)
74 Q_PROPERTY(QString toolTip READ toolTip WRITE setToolTip NOTIFY changed)
75 Q_PROPERTY(QString statusTip READ statusTip WRITE setStatusTip NOTIFY changed)
76 Q_PROPERTY(QString whatsThis READ whatsThis WRITE setWhatsThis NOTIFY changed)
77 Q_PROPERTY(QFont font READ font WRITE setFont NOTIFY changed)
78#ifndef QT_NO_SHORTCUT
79 Q_PROPERTY(QKeySequence shortcut READ shortcut WRITE setShortcut NOTIFY changed)
80 Q_PROPERTY(Qt::ShortcutContext shortcutContext READ shortcutContext WRITE setShortcutContext NOTIFY changed)
81 Q_PROPERTY(bool autoRepeat READ autoRepeat WRITE setAutoRepeat NOTIFY changed)
82#endif
83 Q_PROPERTY(bool visible READ isVisible WRITE setVisible NOTIFY changed)
84 Q_PROPERTY(MenuRole menuRole READ menuRole WRITE setMenuRole NOTIFY changed)
85 Q_PROPERTY(bool iconVisibleInMenu READ isIconVisibleInMenu WRITE setIconVisibleInMenu NOTIFY changed)
86 Q_PROPERTY(Priority priority READ priority WRITE setPriority)
87
88public:
89 // note this is copied into qplatformmenu.h, which must stay in sync
90 enum MenuRole { NoRole = 0, TextHeuristicRole, ApplicationSpecificRole, AboutQtRole,
91 AboutRole, PreferencesRole, QuitRole };
92 enum Priority { LowPriority = 0,
93 NormalPriority = 128,
94 HighPriority = 256};
95 explicit QAction(QObject* parent);
96 QAction(const QString &text, QObject* parent);
97 QAction(const QIcon &icon, const QString &text, QObject* parent);
98
99 ~QAction();
100
101 void setActionGroup(QActionGroup *group);
102 QActionGroup *actionGroup() const;
103 void setIcon(const QIcon &icon);
104 QIcon icon() const;
105
106 void setText(const QString &text);
107 QString text() const;
108
109 void setIconText(const QString &text);
110 QString iconText() const;
111
112 void setToolTip(const QString &tip);
113 QString toolTip() const;
114
115 void setStatusTip(const QString &statusTip);
116 QString statusTip() const;
117
118 void setWhatsThis(const QString &what);
119 QString whatsThis() const;
120
121 void setPriority(Priority priority);
122 Priority priority() const;
123
124#ifndef QT_NO_MENU
125 QMenu *menu() const;
126 void setMenu(QMenu *menu);
127#endif
128
129 void setSeparator(bool b);
130 bool isSeparator() const;
131
132#ifndef QT_NO_SHORTCUT
133 void setShortcut(const QKeySequence &shortcut);
134 QKeySequence shortcut() const;
135
136 void setShortcuts(const QList<QKeySequence> &shortcuts);
137 void setShortcuts(QKeySequence::StandardKey);
138 QList<QKeySequence> shortcuts() const;
139
140 void setShortcutContext(Qt::ShortcutContext context);
141 Qt::ShortcutContext shortcutContext() const;
142
143 void setAutoRepeat(bool);
144 bool autoRepeat() const;
145#endif
146
147 void setFont(const QFont &font);
148 QFont font() const;
149
150 void setCheckable(bool);
151 bool isCheckable() const;
152
153 QVariant data() const;
154 void setData(const QVariant &var);
155
156 bool isChecked() const;
157
158 bool isEnabled() const;
159
160 bool isVisible() const;
161
162 enum ActionEvent { Trigger, Hover };
163 void activate(ActionEvent event);
164 bool showStatusText(QWidget *widget=0);
165
166 void setMenuRole(MenuRole menuRole);
167 MenuRole menuRole() const;
168
169 void setIconVisibleInMenu(bool visible);
170 bool isIconVisibleInMenu() const;
171
172
173 QWidget *parentWidget() const;
174
175 QList<QWidget *> associatedWidgets() const;
176#ifndef QT_NO_GRAPHICSVIEW
177 QList<QGraphicsWidget *> associatedGraphicsWidgets() const; // ### suboptimal
178#endif
179
180protected:
181 bool event(QEvent *);
182 QAction(QActionPrivate &dd, QObject *parent);
183
184public Q_SLOTS:
185 void trigger() { activate(Trigger); }
186 void hover() { activate(Hover); }
187 void setChecked(bool);
188 void toggle();
189 void setEnabled(bool);
190 inline void setDisabled(bool b) { setEnabled(!b); }
191 void setVisible(bool);
192
193Q_SIGNALS:
194 void changed();
195 void triggered(bool checked = false);
196 void hovered();
197 void toggled(bool);
198
199private:
200 Q_DISABLE_COPY(QAction)
201
202 friend class QGraphicsWidget;
203 friend class QWidget;
204 friend class QActionGroup;
205 friend class QMenu;
206 friend class QMenuPrivate;
207 friend class QMenuBar;
208 friend class QToolButton;
209#ifdef Q_OS_MAC
210 friend void qt_mac_clear_status_text(QAction *action);
211#endif
212};
