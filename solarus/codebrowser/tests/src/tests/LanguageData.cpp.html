27using namespace Solarus;

29namespace {
30
31/**
32 * \brief Checks reading and writing a strings data file.
33 */
34void check_strings(TestEnvironment& /* env */, const std::string& language_id) {
35
36 StringResources string_resources;
37 bool success = false;
38
39 // Import the strings data file.
40 std::string file_name = "languages/" + language_id + "/text/strings.dat";
41 std::string imported_string_buffer = QuestFiles::data_file_read(file_name);
42 success = string_resources.import_from_buffer(imported_string_buffer, file_name);
43 Debug::check_assertion(success, "Strings import failed");
44
45 // Export it.
46 std::string exported_string_buffer;
47 success = string_resources.export_to_buffer(exported_string_buffer);
48 Debug::check_assertion(success, "Strings export failed");
49
50 // Check that the file is identical after import/export.
51 if (exported_string_buffer != imported_string_buffer) {
52 std::cerr << "Strings '" << language_id << "' differs from the original one after export." << std::endl
53 << "*** Original strings file:" << std::endl << imported_string_buffer << std::endl
54 << "*** Exported strings file:" << std::endl << exported_string_buffer << std::endl;
55 Debug::die("Strings '" + language_id + "': exported file differs from the original one");
56 }
57}
58
59/**
60 * \brief Checks reading and writing a dialogs data file.
61 */
62void check_dialogs(TestEnvironment& /* env */, const std::string& language_id) {
63
64 DialogResources dialog_resources;
65 bool success = false;
66
67 // Import the dialogs data file.
68 std::string file_name = "languages/" + language_id + "/text/dialogs.dat";
69 std::string imported_dialog_buffer = QuestFiles::data_file_read(file_name);
70 success = dialog_resources.import_from_quest_file(file_name);
71 Debug::check_assertion(success, "Dialogs import failed");
72
73 // Export it.
74 std::string exported_dialog_buffer;
75 success = dialog_resources.export_to_buffer(exported_dialog_buffer);
76 Debug::check_assertion(success, "Dialogs export failed");
77
78 // Check that the file is identical after import/export.
79 if (exported_dialog_buffer != imported_dialog_buffer) {
80 std::cerr << "Dialogs '" << language_id << "' differs from the original one after export." << std::endl
81 << "*** Original dialogs file:" << std::endl << imported_dialog_buffer << std::endl
82 << "*** Exported dialogs file:" << std::endl << exported_dialog_buffer << std::endl;
83 Debug::die("Dialogs '" + language_id + "': exported file differs from the original one");
84 }
85}
86
87}
88
89/**
90 * \brief Tests reading and writing language data files.
91 */
92int main(int argc, char** argv) {
93
94 TestEnvironment env(argc, argv);
95
96 const std::map<std::string, std::string>& language_elements =
97 CurrentQuest::get_resources().get_elements(ResourceType::LANGUAGE);
98 Debug::check_assertion(!language_elements.empty(), "No languages");
99 for (const auto& kvp : language_elements) {
100 const std::string& language_id = kvp.first;
101 check_strings(env, language_id);
102 check_dialogs(env, language_id);
103 }
104
105 return 0;
106}
