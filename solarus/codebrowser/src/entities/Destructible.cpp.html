41/**
42 * \brief Creates a new destructible item with the specified subtype.
43 * \param name Name identifying the entity on the map or an empty string.
44 * \param layer Layer where to create the entity.
45 * \param xy Coordinates where to create the entity.
46 * \param animation_set_id Sprite animation set id for this destructible object.
47 * \param treasure The treasure contained in this object.
48 * \param modified_ground Ground defined by this entity (usually Ground::WALL).
49 */
50Destructible::Destructible(
51 const std::string& name,
52 int layer,
53 const Point& xy,
54 const std::string& animation_set_id,
55 const Treasure& treasure,
56 Ground modified_ground
57):
58 Entity(name, 0, layer, xy, Size(16, 16)),
59 modified_ground(modified_ground),
60 treasure(treasure),
61 animation_set_id(animation_set_id),
62 destruction_sound_id(),
63 can_be_cut(false),
64 can_explode(false),
65 can_regenerate(false),
66 weight(0),
67 damage_on_enemies(1),
68 is_being_cut(false),
69 regeneration_date(0),
70 is_regenerating(false) {
71
72 set_origin(8, 13);
73 create_sprite(get_animation_set_id());
74
75 update_collision_modes();
76}

240/**
241 * \brief Returns whether this entity is an obstacle for another one.
242 * \param other Another entity.
243 * \return \c true if this entity is an obstacle for the other one.
244 */
245bool Destructible::is_obstacle_for(Entity& other) {
246
247 return get_modified_ground() == Ground::WALL &&
248 !is_being_cut &&
249 !is_waiting_for_regeneration() &&
250 other.is_destructible_obstacle(*this);
251}

253/**
254 * \brief Sets the appropriate collisions modes.
255 *
256 * This depends on whether the object is an obstacle, can be cut
257 * or can explode.
258 */
259void Destructible::update_collision_modes() {
260
261 // Reset previous collision modes.
262 set_collision_modes(0);
263
264 // Sets the new ones.
265 if (get_modified_ground() == Ground::WALL) {
266 // The object is an obstacle.
267 // Set the facing collision mode to allow the hero to look at it.
268 add_collision_mode(CollisionMode::COLLISION_FACING);
269 }
270
271 if (get_can_be_cut()
272 || get_can_explode()) {
273 add_collision_mode(CollisionMode::COLLISION_SPRITE);
274 }
275}

319/**
320 * \brief This function is called when this entity detects a collision with the hero.
321 * \param hero the hero
322 * \param collision_mode the collision mode that detected the collision
323 */
324void Destructible::notify_collision_with_hero(Hero& hero, CollisionMode /* collision_mode */) {
325
326 if (get_weight() != -1
327 && !is_being_cut
328 && !is_waiting_for_regeneration()
329 && !is_regenerating
330 && get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_NONE
331 && hero.is_free()) {
332
333 if (get_equipment().has_ability(Ability::LIFT, get_weight())) {
334 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_LIFT);
335 }
336 else {
337 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_LOOK);
338 }
339 }

342/**
343 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
344 */
345void Destructible::notify_collision(
346 Entity& other_entity,
347 Sprite& /* this_sprite */,
348 Sprite& other_sprite
349) {
350 if (get_can_be_cut()
351 && !is_being_cut
352 && !is_waiting_for_regeneration()
353 && !is_regenerating
354 && other_entity.is_hero()) {
355
356 Hero& hero = static_cast<Hero&>(other_entity);
357 if (other_sprite.get_animation_set_id() == hero.get_hero_sprites().get_sword_sprite_id() &&
358 hero.is_striking_with_sword(*this)) {
359
360 play_destroy_animation();
361 hero.check_position(); // To update the ground under the hero.
362 create_treasure();
363
364 get_lua_context()->destructible_on_cut(*this);
365
366 if (get_can_explode()) {
367 explode();
368 }
369 }
370 }
371
372 // TODO use dynamic dispatch
373 if (other_entity.get_type() == EntityType::EXPLOSION
374 && get_can_explode()
375 && !is_being_cut
376 && !is_waiting_for_regeneration()
377 && !is_regenerating) {
378
379 play_destroy_animation();
380 create_treasure();
381 explode();
382 }
383}

385/**
386 * \copydoc Entity::notify_action_command_pressed
387 */
388bool Destructible::notify_action_command_pressed() {
389
390 CommandsEffects::ActionKeyEffect effect = get_commands_effects().get_action_key_effect();
391
392 if ((effect == CommandsEffects::ACTION_KEY_LIFT || effect == CommandsEffects::ACTION_KEY_LOOK)
393 && get_weight() != -1
394 && !is_being_cut
395 && !is_waiting_for_regeneration()
396 && !is_regenerating) {
397
398 if (get_equipment().has_ability(Ability::LIFT, get_weight())) {
399
400 uint32_t explosion_date = get_can_explode() ? System::now() + 6000 : 0;
401 get_hero().start_lifting(std::make_shared<CarriedObject>(
402 get_hero(),
403 *this,
404 get_animation_set_id(),
405 get_destruction_sound(),
406 get_damage_on_enemies(),
407 explosion_date)
408 );
409
410 // Play the sound.
411 Sound::play("lift");
412
413 // Create the pickable treasure.
414 create_treasure();
415
416 if (!get_can_regenerate()) {
417 // Remove this destructible from the map.
418 remove_from_map();
419 }
420 else {
421 // The item can actually regenerate.
422 play_destroy_animation();
423 }
424
425 // Notify Lua.
426 get_lua_context()->destructible_on_lifting(*this);
427 }
428 else {
429 // Cannot lift the object.
430 get_hero().start_grabbing();
431 get_lua_context()->destructible_on_looked(*this);
432 }
433
434 return true;
435 }
436
437 return false;
438}

440/**
441 * \brief Plays the animation destroy of this item.
442 */
443void Destructible::play_destroy_animation() {
444
445 is_being_cut = true;
446 if (!destruction_sound_id.empty()) {
447 Sound::play(destruction_sound_id);
448 }
449 const SpritePtr& sprite = get_sprite();
450 if (sprite != nullptr) {
451 sprite->set_current_animation("destroy");
452 }
453 if (!is_drawn_in_y_order()) {
454 get_entities().bring_to_front(*this); // Show animation destroy to front.
455 }
456 if (is_ground_modifier()) {
457 update_ground_observers(); // The ground has just disappeared.
458 }
459}

481/**
482 * \brief This function is called by the map when the game is suspended or resumed.
483 * \param suspended true to suspend the entity, false to resume it
484 */
485void Destructible::set_suspended(bool suspended) {
486
487 Entity::set_suspended(suspended); // suspend the animation and the movement
488
489 if (!suspended && regeneration_date != 0) {
490 // Recompute the date.
491 regeneration_date += System::now() - get_when_suspended();
492 }
493}
494
495/**
496 * \brief Updates this entity.
497 */
498void Destructible::update() {
499
500 Entity::update();
501
502 if (is_suspended()) {
503 return;
504 }
505
506 const SpritePtr& sprite = get_sprite();
507
508 if (is_being_cut &&
509 sprite != nullptr &&
510 sprite->is_animation_finished()) {
511
512 if (!get_can_regenerate()) {
513 // Remove this destructible from the map.
514 remove_from_map();
515 }
516 else {
517 is_being_cut = false;
518 regeneration_date = System::now() + 10000;
519 }
520 }
521
522 else if (is_waiting_for_regeneration()
523 && System::now() >= regeneration_date
524 && !overlaps(get_hero())) {
525
526 if (sprite != nullptr) {
527 sprite->set_current_animation("regenerating");
528 }
529 is_regenerating = true;
530 regeneration_date = 0;
531 get_lua_context()->destructible_on_regenerating(*this);
532 }
533 else if (is_regenerating &&
534 sprite != nullptr &&
535 sprite->is_animation_finished()) {
536
537 sprite->set_current_animation("on_ground");
538 is_regenerating = false;
539 }
540}
