53template <class T>
54class QSet
55{
56 typedef QHash<T, QHashDummyValue> Hash;
57
58public:
59 inline QSet() {}
60#ifdef Q_COMPILER_INITIALIZER_LISTS
61 inline QSet(std::initializer_list<T> list)
62 {
63 reserve(list.size());
64 for (typename std::initializer_list<T>::const_iterator it = list.begin(); it != list.end(); ++it)
65 insert(*it);
66 }
67#endif
68 inline QSet(const QSet<T> &other) : q_hash(other.q_hash) {}
69
70 inline QSet<T> &operator=(const QSet<T> &other)
71 { q_hash = other.q_hash; return *this; }
72#ifdef Q_COMPILER_RVALUE_REFS
73 inline QSet(QSet &&other) : q_hash(qMove(other.q_hash)) {}
74 inline QSet<T> &operator=(QSet<T> &&other)
75 { qSwap(q_hash, other.q_hash); return *this; }
76#endif
77 inline void swap(QSet<T> &other) { q_hash.swap(other.q_hash); }
78
79 inline bool operator==(const QSet<T> &other) const
80 { return q_hash == other.q_hash; }
81 inline bool operator!=(const QSet<T> &other) const
82 { return q_hash != other.q_hash; }
83
84 inline int size() const { return q_hash.size(); }
85
86 inline bool isEmpty() const { return q_hash.isEmpty(); }
87
88 inline int capacity() const { return q_hash.capacity(); }
89 inline void reserve(int size);
90 inline void squeeze() { q_hash.squeeze(); }
91
92 inline void detach() { q_hash.detach(); }
93 inline bool isDetached() const { return q_hash.isDetached(); }
94#if QT_SUPPORTS(UNSHARABLE_CONTAINERS)
95 inline void setSharable(bool sharable) { q_hash.setSharable(sharable); }
96#endif
97
98 inline void clear() { q_hash.clear(); }
99
100 inline bool remove(const T &value) { return q_hash.remove(value) != 0; }
101
102 inline bool contains(const T &value) const { return q_hash.contains(value); }
103
104 bool contains(const QSet<T> &set) const;

251template <class T>
252Q_INLINE_TEMPLATE QSet<T> &QSet<T>::unite(const QSet<T> &other)
253{
254 QSet<T> copy(other);
255 typename QSet<T>::const_iterator i = copy.constEnd();
256 while (i != copy.constBegin()) {
257 --i;
258 insert(*i);
259 }
260 return *this;
261}
262
263template <class T>
264Q_INLINE_TEMPLATE QSet<T> &QSet<T>::intersect(const QSet<T> &other)
265{
266 QSet<T> copy1;
267 QSet<T> copy2;
268 if (size() <= other.size()) {
269 copy1 = *this;
270 copy2 = other;
271 } else {
272 copy1 = other;
273 copy2 = *this;
274 *this = copy1;
275 }
276 typename QSet<T>::const_iterator i = copy1.constEnd();
277 while (i != copy1.constBegin()) {
278 --i;
279 if (!copy2.contains(*i))
280 remove(*i);
281 }
282 return *this;
283}

285template <class T>
286Q_INLINE_TEMPLATE QSet<T> &QSet<T>::subtract(const QSet<T> &other)
287{
288 QSet<T> copy1(*this);
289 QSet<T> copy2(other);
290 typename QSet<T>::const_iterator i = copy1.constEnd();
291 while (i != copy1.constBegin()) {
292 --i;
293 if (copy2.contains(*i))
294 remove(*i);
295 }
296 return *this;
297}

299template <class T>
300Q_INLINE_TEMPLATE bool QSet<T>::contains(const QSet<T> &other) const
301{
302 typename QSet<T>::const_iterator i = other.constBegin();
303 while (i != other.constEnd()) {
304 if (!contains(*i))
305 return false;
306 ++i;
307 }
308 return true;
309}

311template <typename T>
312Q_OUTOFLINE_TEMPLATE QList<T> QSet<T>::toList() const
313{
314 QList<T> result;
315 result.reserve(size());
316 typename QSet<T>::const_iterator i = constBegin();
317 while (i != constEnd()) {
318 result.append(*i);
319 ++i;
320 }
321 return result;
322}
323
324template <typename T>
325Q_OUTOFLINE_TEMPLATE QSet<T> QList<T>::toSet() const
326{
327 QSet<T> result;
328 result.reserve(size());
329 for (int i = 0; i < size(); ++i)
330 result.insert(at(i));
331 return result;
332}

348template <typename T>
349class QMutableSetIterator
350{
351 typedef typename QSet<T>::iterator iterator;
352 QSet<T> *c;
353 iterator i, n;
354 inline bool item_exists() const { return c->constEnd() != n; }
355
356public:
357 inline QMutableSetIterator(QSet<T> &container)
358 : c(&container)
359 { c->setSharable(false); i = c->begin(); n = c->end(); }
360 inline ~QMutableSetIterator()
361 { c->setSharable(true); }
362 inline QMutableSetIterator &operator=(QSet<T> &container)
363 { c->setSharable(true); c = &container; c->setSharable(false);
364 i = c->begin(); n = c->end(); return *this; }
365 inline void toFront() { i = c->begin(); n = c->end(); }
366 inline void toBack() { i = c->end(); n = i; }
367 inline bool hasNext() const { return c->constEnd() != i; }
368 inline const T &next() { n = i++; return *n; }
369 inline const T &peekNext() const { return *i; }
370 inline bool hasPrevious() const { return c->constBegin() != i; }
371 inline const T &previous() { n = --i; return *n; }
372 inline const T &peekPrevious() const { iterator p = i; return *--p; }
373 inline void remove()
374 { if (c->constEnd() != n) { i = c->erase(n); n = c->end(); } }
375 inline const T &value() const { Q_ASSERT(item_exists()); return *n; }
376 inline bool findNext(const T &t)
377 { while (c->constEnd() != (n = i)) if (*i++ == t) return true; return false; }
378 inline bool findPrevious(const T &t)
379 { while (c->constBegin() != i) if (*(n = --i) == t) return true;
380 n = c->end(); return false; }
381};
