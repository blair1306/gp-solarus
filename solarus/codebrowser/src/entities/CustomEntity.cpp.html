43/**
44 * \brief Creates a custom entity.
45 * \param game The game.
46 * \param name Name of the entity or an empty string.
47 * \param direction Direction of the entity (0 to 3).
48 * \param layer The layer on the map.
49 * \param xy Coordinates on the map.
50 * \param size Size of the entity.
51 * \param sprite_name Animation set id of a sprite or an empty string.
52 * \param model Model of custom entity or an empty string.
53 */
54CustomEntity::CustomEntity(
55 Game& /* game */,
56 const std::string& name,
57 int direction,
58 int layer,
59 const Point& xy,
60 const Size& size,
61 const std::string& sprite_name,
62 const std::string& model
63):
64 Entity(
65 name, 0, layer, xy, size
66 ),
67 model(model),
68 ground_observer(false),
69 modified_ground(Ground::EMPTY) {
70
71 set_collision_modes(
72 CollisionMode::COLLISION_FACING |
73 CollisionMode::COLLISION_CUSTOM |
74 CollisionMode::COLLISION_SPRITE
75 );
76 set_origin(8, 13);
77
78 if (!sprite_name.empty()) {
79 create_sprite(sprite_name);
80 enable_pixel_collisions();
81 }
82 set_sprites_direction(direction);
83}

114/**
115 * \brief Sets the direction of this custom entity.
116 *
117 * It will be applied to sprites that have such a direction.
118 *
119 * \param direction The direction.
120 */
121void CustomEntity::set_sprites_direction(int direction) {
122
123 set_direction(direction);
124
125 for (const SpritePtr& sprite: get_sprites()) {
126 if (direction >= 0 && direction < sprite->get_nb_directions()) {
127 sprite->set_current_direction(direction);
128 }
129 }
130}

132/**
133 * \brief Returns the info about whether this custom entity can be traversed
134 * by a type of entity.
135 * \param type Type of entity to test.
136 * \return The corresponding traversable property.
137 */
138const CustomEntity::TraversableInfo& CustomEntity::get_traversable_by_entity_info(
139 EntityType type) {
140
141 const auto it = traversable_by_entities_type.find(type);
142 if (it != traversable_by_entities_type.end()) {
143 // This entity type overrides the general setting.
144 return it->second;
145 }
146
147 return traversable_by_entities_general;
148}
149
150/**
151 * \brief Sets whether this custom entity can be traversed by the specified
152 * entity.
153 * \param entity The entity to test.
154 * \return \c true if the entity can traverse this custom entity.
155 */
156bool CustomEntity::is_traversable_by_entity(Entity& entity) {
157
158 // Find the obstacle settings.
159 const TraversableInfo& info = get_traversable_by_entity_info(entity.get_type());
160
161 if (info.is_empty()) {
162 // Nothing was set: make the custom entity traversable by default.
163 return true;
164 }
165
166 return info.is_traversable(*this, entity);
167}

606/**
607 * \brief Returns whether this custom entity can traverse a kind of ground.
608 * \param ground A kind of ground.
609 * \return \c true if this custom entity can traverse this kind of ground.
610 */
611bool CustomEntity::can_traverse_ground(Ground ground) const {
612
613 const auto& it = can_traverse_grounds.find(ground);
614
615 if (it != can_traverse_grounds.end()) {
616 // Return the boolean value that was set.
617 return it->second;
618 }
619
620 // Return a default value.
621 switch (ground) {
622
623 case Ground::EMPTY:
624 case Ground::TRAVERSABLE:
625 case Ground::GRASS:
626 case Ground::ICE:
627 case Ground::LADDER:
628 return true;
629
630 case Ground::WALL:
631 case Ground::LOW_WALL:
632 case Ground::WALL_TOP_RIGHT:
633 case Ground::WALL_TOP_LEFT:
634 case Ground::WALL_BOTTOM_LEFT:
635 case Ground::WALL_BOTTOM_RIGHT:
636 case Ground::WALL_TOP_RIGHT_WATER:
637 case Ground::WALL_TOP_LEFT_WATER:
638 case Ground::WALL_BOTTOM_LEFT_WATER:
639 case Ground::WALL_BOTTOM_RIGHT_WATER:
640 case Ground::DEEP_WATER:
641 case Ground::SHALLOW_WATER:
642 case Ground::HOLE:
643 case Ground::PRICKLE:
644 case Ground::LAVA:
645 return false;
646 }
647
648 return false;
649}

763/**
764 * \brief Registers a function to be called when the specified test detects a
765 * collision.
766 * \param collision_test A built-in collision test.
767 * \param callback_ref Lua ref to a function to call when this collision is
768 * detected.
769 */
770void CustomEntity::add_collision_test(
771 CollisionMode collision_test,
772 const ScopedLuaRef& callback_ref
773) {

778 *get_lua_context(),
779 collision_test,
780 callback_ref
781 );
782
783 check_collision_with_detectors();
784}

820/**
821 * \copydoc Entity::test_collision_custom
822 */
823bool CustomEntity::test_collision_custom(Entity& entity) {
824
825 if (!successful_collision_tests.empty()) {
826 // Avoid reentrant collision tests.
827 return false;
828 }
829
830 bool collision = false;
831
832 const std::vector<CollisionInfo> collision_tests = this->collision_tests;
833 for (const CollisionInfo& info: collision_tests) {
834
835 switch (info.get_built_in_test()) {
836
837 case COLLISION_OVERLAPPING:
838 if (test_collision_rectangle(entity)) {
839 collision = true;
840 successful_collision_tests.push_back(info);
841 }
842 break;
843
844 case COLLISION_CONTAINING:
845 if (test_collision_inside(entity)) {
846 collision = true;
847 successful_collision_tests.push_back(info);
848 }
849 break;
850
851 case COLLISION_ORIGIN:
852 if (test_collision_origin_point(entity)) {
853 collision = true;
854 successful_collision_tests.push_back(info);
855 }
856 break;
857
858 case COLLISION_FACING:
859 if (test_collision_facing_point(entity)) {
860 collision = true;
861 successful_collision_tests.push_back(info);
862 }
863 break;
864
865 case COLLISION_TOUCHING:
866 if (test_collision_touching(entity)) {
867 collision = true;
868 successful_collision_tests.push_back(info);
869 }
870 break;
871
872 case COLLISION_CENTER:
873 if (test_collision_center(entity)) {
874 collision = true;
875 successful_collision_tests.push_back(info);
876 }
877 break;
878
879 case COLLISION_CUSTOM:
880 if (get_lua_context()->do_custom_entity_collision_test_function(
881 info.get_custom_test_ref(), *this, entity)
882 ) {
883 collision = true;
884 successful_collision_tests.push_back(info);
885 }
886 break;
887
888 case COLLISION_SPRITE:
889 // Not handled here.
890 break;
891
892 case COLLISION_NONE:
893 Debug::die("Invalid collision mode");
894 break;
895 }
896 }
897
898 return collision;
899}

901/**
902 * \copydoc Entity::notify_collision(Entity&,CollisionMode)
903 */
904void CustomEntity::notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) {
905
906 if (collision_mode == COLLISION_FACING) {
907 // This collision mode is only useful to set the facing entity, which
908 // is already done by the Entity class.
909 return;
910 }
911
912 // One or several collisions were detected with an another entity.
913 // The collision tests could have been of any kind
914 // (even a custom Lua collision test function),
915 // except COLLISION_SPRITE that is handled separately.
916
917 Debug::check_assertion(collision_mode == COLLISION_CUSTOM,
918 "Unexpected collision mode");
919
920 // There is a collision: execute the callbacks.
921 for (const CollisionInfo& info: successful_collision_tests) {
922 get_lua_context()->do_custom_entity_collision_callback(
923 info.get_callback_ref(), *this, entity_overlapping
924 );
925 }
926
927 successful_collision_tests.clear();
928}

930/**
931 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
932 */
933void CustomEntity::notify_collision(
934 Entity& other_entity,
935 Sprite& this_sprite,
936 Sprite& other_sprite
937) {
938 // A collision was detected with a sprite of another entity.
939 const std::vector<CollisionInfo> collision_tests = this->collision_tests;
940 for (const CollisionInfo& info: collision_tests) {
941
942 if (info.get_built_in_test() == COLLISION_SPRITE) {
943 // Execute the callback.
944 get_lua_context()->do_custom_entity_collision_callback(
945 info.get_callback_ref(),
946 *this,
947 other_entity,
948 this_sprite,
949 other_sprite
950 );
951 }
952 }
953}
