18#define SOLARUS_GUI_CONSOLE_LINE_EDIT_H
19
20#include "solarus/gui/gui_common.h"
21#include <QLineEdit>
22
23class QStringListModel;
24
25namespace SolarusGui {
26
27/**
28 * @brief A line field to execute Lua commands to the quest.
29 *
30 * Manages the history of commands.
31 */
32class SOLARUS_GUI_API ConsoleLineEdit : public QLineEdit {
33 Q_OBJECT
34
35public:
36
37 explicit ConsoleLineEdit(QWidget* parent = nullptr);
38
39public slots:
40
41 void set_history_position(int history_position);
42 void history_previous();
43 void history_next();
44 void command_executed(const QString& command);
45
46protected:
47
48 void keyPressEvent(QKeyEvent* event) override;
49
50private:
51
52 QStringList history; /**< Commands history. */
53 int history_position; /**< Current position in the history, between
54 * 0 and history.size() - 1, or
55 * history.size() if not navigating the history. */
56 QString current_command; /**< Command currently being edited, preserved
57 * while navigating the history. */
58 QStringListModel*
59 completer_model; /**< Model for the completer. */
60
61};
62
63}
