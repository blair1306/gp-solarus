116/**
117 * \brief Implementation of sol.sprite.create().
118 * \param l The Lua context that is calling this function.
119 * \return Number of values to return to Lua.
120 */
121int LuaContext::sprite_api_create(lua_State* l) {
122
123 return LuaTools::exception_boundary_handle(l, [&] {
124 const std::string& animation_set_id = LuaTools::check_string(l, 1);
125
126 // TODO if the file does not exist, make a Lua error instead of an assertion error.
127 SpritePtr sprite = std::make_shared<Sprite>(animation_set_id);
128 get_lua_context(l).add_drawable(sprite);
129
130 push_sprite(l, *sprite);
131 return 1;
132 });
133}

169/**
170 * \brief Implementation of sprite:set_animation().
171 * \param l The Lua context that is calling this function.
172 * \return Number of values to return to Lua.
173 */
174int LuaContext::sprite_api_set_animation(lua_State* l) {
175
176 return LuaTools::exception_boundary_handle(l, [&] {
177 Sprite& sprite = *check_sprite(l, 1);
178 const std::string& animation_name = LuaTools::check_string(l, 2);
179 ScopedLuaRef callback_ref;
180 if (lua_gettop(l) >= 3) {
181 if (!lua_isfunction(l, 3) && !lua_isstring(l, 3)) {
182 LuaTools::type_error(l, 3, "function or string");
183 }
184 callback_ref = LuaTools::create_ref(l, 3);
185 }
186
187 if (!sprite.has_animation(animation_name)) {
188 LuaTools::arg_error(l, 2,
189 std::string("Animation '") + animation_name
190 + "' does not exist in sprite '" + sprite.get_animation_set_id() + "'"
191 );
192 }
193
194 sprite.set_current_animation(animation_name);
195 sprite.set_finished_callback(callback_ref);
196 sprite.restart_animation();
197
198 return 0;
199 });
200}

233/**
234 * \brief Implementation of sprite:set_direction().
235 * \param l The Lua context that is calling this function.
236 * \return Number of values to return to Lua.
237 */
238int LuaContext::sprite_api_set_direction(lua_State* l) {
239
240 return LuaTools::exception_boundary_handle(l, [&] {
241 Sprite& sprite = *check_sprite(l, 1);
242 int direction = LuaTools::check_int(l, 2);
243
244 if (direction < 0 || direction >= sprite.get_nb_directions()) {
245 std::ostringstream oss;
246 oss << "Illegal direction " << direction
247 << " for sprite '" + sprite.get_animation_set_id()
248 << "' in animation '" + sprite.get_current_animation() + "'";
249 LuaTools::arg_error(l, 2, oss.str());
250 }
251 sprite.set_current_direction(direction);
252
253 return 0;
254 });
255}
