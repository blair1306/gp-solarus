38/**
39 * \brief Returns a random integer number in [0, x[ with a uniform distribution.
40 *
41 * This is equivalent to: Random::get_number(0, x).
42 *
43 * \param x The superior bound.
44 * \return a Random integer number in [0, x[.
45 */
46int get_number(unsigned int x) {
47 return get_number(0, x);
48}
49
50/**
51 * \brief Returns a random integer number in [x, y[ with a uniform distribution.
52 * \param x The inferior bound.
53 * \param y The superior bound.
54 * \return A random integer number in [x, y[.
55 */
56int get_number(int x, int y) {
57
58 // Initialize the engine and the distribution
59 //
60 // The engine is not initialized with std::random_device
61 // because not every main platform support non-deterministic
62 // random numbers generation yet.
63 //
64 static std::mt19937 engine(std::time(nullptr));
65 static std::uniform_int_distribution<int> dist{};
66
67 // Type of the parameters of the distribution
68 using param_type = std::uniform_int_distribution<int>::param_type;
69
70 // Get a random number in [x, y[
71 return dist(engine, param_type{x, y - 1});
72}
