31/**
32 * \brief Initializes the audio features provided to Lua.
33 */
34void LuaContext::register_audio_module() {
35
36 static const luaL_Reg functions[] = {
37 { "get_sound_volume", audio_api_get_sound_volume },
38 { "set_sound_volume", audio_api_set_sound_volume },
39 { "play_sound", audio_api_play_sound },
40 { "preload_sounds", audio_api_preload_sounds },
41 { "get_music_volume", audio_api_get_music_volume },
42 { "set_music_volume", audio_api_set_music_volume },
43 { "play_music", audio_api_play_music },
44 { "stop_music", audio_api_stop_music },
45 { "get_music", audio_api_get_music },
46 { "get_music_format", audio_api_get_music_format },
47 { "get_music_num_channels", audio_api_get_music_num_channels },
48 { "get_music_channel_volume", audio_api_get_music_channel_volume },
49 { "set_music_channel_volume", audio_api_set_music_channel_volume },
50 { "get_music_tempo", audio_api_get_music_tempo },
51 { "set_music_tempo", audio_api_set_music_tempo },
52 { nullptr, nullptr }
53 };
54 register_functions(audio_module_name, functions);
55}

60 * \return number of values to return to Lua

62int LuaContext::audio_api_get_sound_volume(lua_State* l) {
63
64 return LuaTools::exception_boundary_handle(l, [&] {

85/**
86 * \brief Implementation of sol.audio.play_sound().
87 * \param l the Lua context that is calling this function
88 * \return number of values to return to Lua
89 */
90int LuaContext::audio_api_play_sound(lua_State* l) {
91
92 return LuaTools::exception_boundary_handle(l, [&] {
93 const std::string& sound_id = LuaTools::check_string(l, 1);
94
95 if (!Sound::exists(sound_id)) {
96 LuaTools::error(l, std::string("No such sound: '") + sound_id + "'");
97 }
98 Sound::play(sound_id);
99
100 return 0;
101 });
102}

146/**
147 * \brief Implementation of sol.audio.play_music().
148 * \param l the Lua context that is calling this function
149 * \return number of values to return to Lua
150 */
151int LuaContext::audio_api_play_music(lua_State* l) {
152
153 return LuaTools::exception_boundary_handle(l, [&] {
154 const std::string& music_id = LuaTools::opt_string(l, 1, "");
155 bool loop = true; // true by default, unless there is a callback.
156 ScopedLuaRef callback_ref;
157 if (lua_gettop(l) >= 2) {
158 if (lua_isboolean(l, 2)) {
159 // There is a loop parameter.
160 loop = lua_toboolean(l, 2);
161 }
162 else {
163 // There is a callback parameter.
164 loop = false; // No loop when there is a callback.
165 callback_ref = LuaTools::check_function(l, 2);
166 }
167 }
168
169 if (music_id.empty()) {
170 // nil music: stop playing any music.
171 Music::stop_playing();
172 }
173 else {
174 if (!Music::exists(music_id)) {
175 // Could not find the specified music.
176 LuaTools::error(l, std::string("No such music: '") + music_id + "'");
177 }
178
179 // Valid music file name.
180 Music::play(music_id, loop, callback_ref);
181 }
182 return 0;
183 });
184}

200/**
201 * \brief Implementation of sol.audio.get_music().
202 * \param l The Lua context that is calling this function.
203 * \return Number of values to return to Lua.
204 */
205int LuaContext::audio_api_get_music(lua_State* l) {
206
207 return LuaTools::exception_boundary_handle(l, [&] {
208 const std::string& music_id = Music::get_current_music_id();
209
210 if (music_id == Music::none) {
211 lua_pushnil(l);
212 }
213 else {
214 push_string(l, music_id);
215 }
216 return 1;
217 });
218}

220/**
221 * \brief Implementation of sol.audio.get_music_format().
222 * \param l The Lua context that is calling this function.
223 * \return Number of values to return to Lua.
224 */
225int LuaContext::audio_api_get_music_format(lua_State* l) {
226
227 return LuaTools::exception_boundary_handle(l, [&] {
228 const Music::Format format = Music::get_format();
229
230 if (format == Music::NO_FORMAT) {
231 // No music is playing.
232 lua_pushnil(l);
233 }
234 else {
235 push_string(l, Music::format_names[format]);
236 }
237 return 1;
238 });
239}

259/**
260 * \brief Implementation of sol.audio.get_music_channel_volume().
261 * \param l The Lua context that is calling this function.
262 * \return Number of values to return to Lua.
263 */
264int LuaContext::audio_api_get_music_channel_volume(lua_State* l) {
265
266 return LuaTools::exception_boundary_handle(l, [&] {
267 int channel = LuaTools::check_int(l, 1);
268
269 if (Music::get_format() != Music::IT) {
270 lua_pushnil(l);
271 }
272 else {
273 if (channel < 0 || channel >= Music::get_num_channels()) {
274 std::ostringstream oss;
275 oss << "Invalid channel number: " << channel;
276 LuaTools::arg_error(l, 1, oss.str());
277 }
278 lua_pushinteger(l, Music::get_channel_volume(channel));
279 }
280 return 1;
281 });
282}

284/**
285 * \brief Implementation of sol.audio.set_music_channel_volume().
286 * \param l The Lua context that is calling this function.
287 * \return Number of values to return to Lua.
288 */
289int LuaContext::audio_api_set_music_channel_volume(lua_State* l) {
290
291 return LuaTools::exception_boundary_handle(l, [&] {
292 int channel = LuaTools::check_int(l, 1);
293 int volume = LuaTools::check_int(l, 2);
294
295 if (Music::get_format() != Music::IT) {
296 lua_pushboolean(l, false);
297 }
298 else {
299 if (channel < 0 || channel >= Music::get_num_channels()) {
300 std::ostringstream oss;
301 oss << "Invalid channel number: " << channel;
302 LuaTools::arg_error(l, 1, oss.str());
303 }
304 Music::set_channel_volume(channel, volume);
305 lua_pushboolean(l, true);
306 }
307 return 1;
308 });
309}
