30/**
31 * \brief Constructor.
32 * \param hero The hero controlled by this state.
33 * \param lifted_item The item to lift.
34 */
35Hero::LiftingState::LiftingState(
36 Hero& hero,
37 const std::shared_ptr<CarriedObject>& lifted_item
38):
39 HeroState(hero, "lifting"),
40 lifted_item(lifted_item) {
41
42 Debug::check_assertion(lifted_item != nullptr, "Missing lifted item");
43}
44
45/**
46 * \brief Starts this state.
47 * \param previous_state the previous state
48 */
49void Hero::LiftingState::start(const State* previous_state) {
50
51 HeroState::start(previous_state);
52
53 // initialize the entity that will be lifted
54 lifted_item->set_map(get_map());
55
56 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_THROW);
57 get_sprites().set_animation_lifting();
58 get_sprites().set_lifted_item(lifted_item);
59 get_entity().set_facing_entity(nullptr);
60
61 get_equipment().notify_ability_used(Ability::LIFT);
62}
63
64/**
65 * \brief Ends this state.
66 * \param next_state the next state
67 */
68void Hero::LiftingState::stop(const State* next_state) {
69
70 HeroState::stop(next_state);
71
72 if (lifted_item != nullptr) {
73
74 get_sprites().set_lifted_item(nullptr);
75
76 // the lifted item is still managed by this state
77 switch (next_state->get_previous_carried_object_behavior()) {
78
79 case CarriedObject::BEHAVIOR_THROW:
80 throw_item();
81 break;
82
83 case CarriedObject::BEHAVIOR_DESTROY:
84 case CarriedObject::BEHAVIOR_KEEP:
85 lifted_item = nullptr;
86 break;
87 }
88 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
89 }
90}
91
92/**
93 * \brief Updates this state.
94 */
95void Hero::LiftingState::update() {
96
97 HeroState::update();
98
99 lifted_item->update();
100
101 // See if the item has finished being lifted.
102 Hero& hero = get_entity();
103 const bool lift_finished = !lifted_item->is_being_lifted() || hero.is_animation_finished();
104 if (!is_suspended() && lift_finished) {
105
106 std::shared_ptr<CarriedObject> carried_object = lifted_item;
107 lifted_item = nullptr; // we do not take care of the carried object from this state anymore
108 hero.set_state(new CarryingState(hero, carried_object));
109 }
110}
111
112/**
113 * \brief Notifies this state that the game was just suspended or resumed.
114 * \param suspended true if the game is suspended
115 */
116void Hero::LiftingState::set_suspended(bool suspended) {
117
118 HeroState::set_suspended(suspended);
119
120 if (lifted_item != nullptr) {
121 lifted_item->set_suspended(suspended);
122 }
123}
124
125/**
126 * \brief Returns whether the hero can be hurt in this state.
127 * \return true if the hero can be hurt in this state
128 * \param attacker an attacker that is trying to hurt the hero
129 * (or nullptr if the source of the attack is not an enemy)
130 */
131bool Hero::LiftingState::can_be_hurt(Entity* /* attacker */) const {
132 return true;
133}
134
135/**
136 * \brief Throws the item that is being lifted.
137 *
138 * This function is called when this state is interrupted by a new state,
139 * e.g. when the hero is hurt while lifting an item.
140 */
141void Hero::LiftingState::throw_item() {
142
143 lifted_item->throw_item(get_sprites().get_animation_direction());
144 get_entities().add_entity(lifted_item);
145 lifted_item = nullptr;
146}
