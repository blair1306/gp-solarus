44#ifndef QGLOBALSTATIC_H

47#include <QtCore/qatomic.h>

49QT_BEGIN_NAMESPACE
50
51namespace QtGlobalStatic {
52enum GuardValues {
53 Destroyed = -2,
54 Initialized = -1,
55 Uninitialized = 0,
56 Initializing = 1
57};
58}
59
60#if defined(QT_NO_THREAD) || (defined(Q_CC_GNU) && !defined(Q_OS_MAC))
61// some compilers support thread-safe statics
62// The IA-64 C++ ABI requires this, so we know that all GCC versions since 3.4
63// support it. C++11 also requires this behavior.
64// Clang and Intel CC masquerade as GCC when compiling on Linux.
65//
66// Apple's libc++abi however uses a global lock for initializing local statics,
67// which will block other threads also trying to initialize a local static
68// until the constructor returns ...
69// We better avoid these kind of problems by using our own locked implementation.
70
71#if defined(Q_OS_UNIX) && defined(Q_CC_INTEL)
72// Work around Intel issue ID 6000058488:
73// local statics inside an inline function inside an anonymous namespace are global
74// symbols (this affects the IA-64 C++ ABI, so OS X and Linux only)
75# define Q_GLOBAL_STATIC_INTERNAL_DECORATION Q_DECL_HIDDEN
76#else
77# define Q_GLOBAL_STATIC_INTERNAL_DECORATION Q_DECL_HIDDEN inline
78#endif
79
80#define Q_GLOBAL_STATIC_INTERNAL(ARGS) \
81 Q_GLOBAL_STATIC_INTERNAL_DECORATION Type *innerFunction() \
82 { \
83 struct HolderBase { \
84 ~HolderBase() Q_DECL_NOTHROW \
85 { if (guard.load() == QtGlobalStatic::Initialized) \
86 guard.store(QtGlobalStatic::Destroyed); } \
87 }; \
88 static struct Holder : public HolderBase { \
89 Type value; \
90 Holder() \
91 Q_DECL_NOEXCEPT_EXPR(noexcept(Type ARGS)) \
92 : value ARGS \
93 { guard.store(QtGlobalStatic::Initialized); } \
94 } holder; \
95 return &holder.value; \
96 }
97#else
98// We don't know if this compiler supports thread-safe global statics
99// so use our own locked implementation
100
101QT_END_NAMESPACE
102#include <QtCore/qmutex.h>
103QT_BEGIN_NAMESPACE
104
105#define Q_GLOBAL_STATIC_INTERNAL(ARGS) \
106 Q_DECL_HIDDEN inline Type *innerFunction() \
107 { \
108 static Type *d; \
109 static QBasicMutex mutex; \
110 int x = guard.loadAcquire(); \
111 if (Q_UNLIKELY(x >= QtGlobalStatic::Uninitialized)) { \
112 QMutexLocker locker(&mutex); \
113 if (guard.load() == QtGlobalStatic::Uninitialized) { \
114 d = new Type ARGS; \
115 static struct Cleanup { \
116 ~Cleanup() { \
117 delete d; \
118 guard.store(QtGlobalStatic::Destroyed); \
119 } \
120 } cleanup; \
121 guard.store(QtGlobalStatic::Initialized); \
122 } \
123 } \
124 return d; \
125 }
126#endif
127
128// this class must be POD, unless the compiler supports thread-safe statics
129template <typename T, T *(&innerFunction)(), QBasicAtomicInt &guard>
130struct QGlobalStatic
131{
132 typedef T Type;
133
134 bool isDestroyed() const { return guard.load() <= QtGlobalStatic::Destroyed; }
135 bool exists() const { return guard.load() == QtGlobalStatic::Initialized; }
136 operator Type *() { if (isDestroyed()) return 0; return innerFunction(); }
137 Type *operator()() { if (isDestroyed()) return 0; return innerFunction(); }
138 Type *operator->()
139 {
140 Q_ASSERT_X(!isDestroyed(), "Q_GLOBAL_STATIC", "The global static was used after being destroyed");
141 return innerFunction();
142 }
143 Type &operator*()
144 {
145 Q_ASSERT_X(!isDestroyed(), "Q_GLOBAL_STATIC", "The global static was used after being destroyed");
146 return *innerFunction();
147 }
148};
149
150#define Q_GLOBAL_STATIC_WITH_ARGS(TYPE, NAME, ARGS) \
151 namespace { namespace Q_QGS_ ## NAME { \
152 typedef TYPE Type; \
153 QBasicAtomicInt guard = Q_BASIC_ATOMIC_INITIALIZER(QtGlobalStatic::Uninitialized); \
154 Q_GLOBAL_STATIC_INTERNAL(ARGS) \
155 } } \
156 static QGlobalStatic<TYPE, \
157 Q_QGS_ ## NAME::innerFunction, \
158 Q_QGS_ ## NAME::guard> NAME;
159
160#define Q_GLOBAL_STATIC(TYPE, NAME) \
161 Q_GLOBAL_STATIC_WITH_ARGS(TYPE, NAME, ())
162
163QT_END_NAMESPACE
