17#ifndef SOLARUS_RECTANGLE_H
18#define SOLARUS_RECTANGLE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Point.h"
22#include "solarus/lowlevel/Size.h"
23#include <SDL_rect.h>
24#include <iosfwd>
25
26namespace Solarus {
27
28/**
29 * \brief Represents a rectangle.
30 *
31 * This class represents a rectangle with its upper-left corner, width and height.
32 * This class can also be used to represent a single point
33 * (then the width and the height are 1).
34 * This class basically encapsulates a library-dependent rectangle object.
35 */
36class Rectangle {
37
38 // low-level classes allowed to manipulate directly the internal SDL rectangle encapsulated
39 friend class Surface;
40 friend class Video;
41
42 public:
43
44 constexpr Rectangle();
45 constexpr Rectangle(int x, int y);
46 explicit constexpr Rectangle(const Point& xy);
47 explicit constexpr Rectangle(const Size& size);
48 constexpr Rectangle(int x, int y, int width, int height);
49 constexpr Rectangle(const Point& xy, const Size& size);
50 constexpr Rectangle(const Point& top_left, const Point& bottom_right);
51
52 int get_x() const;
53 int get_y() const;
54 Point get_xy() const;
55 int get_width() const;
56 int get_height() const;
57 Size get_size() const;
58 bool is_flat() const;
59
60 int get_bottom() const;
61 Point get_bottom_left() const;
62 Point get_bottom_right() const;
63 Point get_center() const;
64 int get_left() const;
65 int get_right() const;
66 int get_top() const;
67 Point get_top_left() const;
68 Point get_top_right() const;
69
70 void set_x(int x);
71 void set_y(int y);
72 void set_width(int width);
73 void set_height(int height);
74
75 void set_xy(int x, int y);
76 void set_xy(const Point& xy);
77 void set_size(int width, int height);
78 void set_size(const Size& other);
79 void set_center(int x, int y);
80 void set_center(const Point& xy);
81
82 void add_x(int dx);
83 void add_y(int dy);
84 void add_width(int dw);
85 void add_height(int dh);
86 void add_xy(int dx, int dy);
87 void add_xy(const Point& other);
88
89 bool contains(int x, int y) const;
90 bool contains(const Point& point) const;
91 bool contains(const Rectangle& other) const;
92 bool overlaps(const Rectangle& other) const;
93
94 Rectangle get_intersection(const Rectangle& other) const;
95
96 Rectangle operator|(const Rectangle& other) const;
97 Rectangle operator&(const Rectangle& other) const;
98 Rectangle& operator|=(const Rectangle& other);
99 Rectangle& operator&=(const Rectangle& other);
100
101 private:
102
103 SDL_Rect* get_internal_rect();
104 const SDL_Rect* get_internal_rect() const;
105
106 SDL_Rect rect; /**< the SDL_Rect encapsulated */
107
108};
109
110bool operator==(const Rectangle& lhs, const Rectangle& rhs);
111bool operator!=(const Rectangle& lhs, const Rectangle& rhs);
112
113SOLARUS_API std::ostream& operator<<(std::ostream& stream, const Rectangle& rectangle);
114
115}

118
119#endif
