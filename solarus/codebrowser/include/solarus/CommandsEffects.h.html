17#ifndef SOLARUS_COMMANDS_EFFECT_H
18#define SOLARUS_COMMANDS_EFFECT_H
19
20#include "solarus/Common.h"
21#include "solarus/EnumInfo.h"
22#include <string>
23#include <vector>
24
25namespace Solarus {
26
27/**
28 * \brief Associate to each game key its current effect on the game.
29 *
30 * This class associates to each game key (the action key,
31 * the sword key, the pause key, and the two equipment item keys)
32 * the current effect of pressing that key.
33 *
34 * When the user presses one of those keys, the engine calls
35 * this class to determine what should happen (e.g. swing the sword,
36 * open a chest, save the game, use an item, etc.).
37 * This class is also used by the HUD to display the five icons
38 * representing those keys.
39 *
40 * TODO make a single enum for all commands
41 */
42class CommandsEffects {
43
44 public:
45
46 /**
47 * \brief What happens when the user presses the Action key.
48 */
49 enum ActionKeyEffect {
50 ACTION_KEY_NONE, /**< nothing happens when the user presses the Action key and the action icon is not shown */
51 ACTION_KEY_NEXT, /**< display the next message in a dialog */
52 ACTION_KEY_LOOK, /**< look at something */
53 ACTION_KEY_OPEN, /**< open a door or a chest */
54 ACTION_KEY_LIFT, /**< lift an object */
55 ACTION_KEY_THROW, /**< throw an object */
56 ACTION_KEY_GRAB, /**< grab an object */
57 ACTION_KEY_SPEAK, /**< speak with an NPC */
58 ACTION_KEY_SWIM /**< swim faster */
59 };
60
61 /**
62 * \brief What happens when the user presses the Attack key.
63 */
64 enum AttackKeyEffect {
65 ATTACK_KEY_NONE, /**< nothing happens when the user presses the Attack key */
66 ATTACK_KEY_SWORD /**< swing the sword (this is the case most of the time) */
67 };
68
69 /**
70 * \brief What happens when the user presses the Pause key.
71 * TODO these values are not used but they should be.
72 */
73 enum PauseKeyEffect {
74 PAUSE_KEY_NONE, /**< nothing happens when the user presses the Pause key */
75 PAUSE_KEY_PAUSE, /**< open the pause menu */
76 PAUSE_KEY_RETURN /**< close the pause menu */
77 };
78
79 CommandsEffects();
80
81 // action key
82
83 ActionKeyEffect get_action_key_effect();
84 void set_action_key_effect(ActionKeyEffect action_key_effect);
85 bool is_action_key_enabled();
86 void set_action_key_enabled(bool enable);
87 void save_action_key_effect();
88 void restore_action_key_effect();
89 bool is_action_key_acting_on_facing_entity();
90
91 // sword key
92
93 AttackKeyEffect get_sword_key_effect();
94 void set_sword_key_effect(AttackKeyEffect sword_key_effect);
95 bool is_sword_key_enabled();
96 void set_sword_key_enabled(bool enable);
97 void save_sword_key_effect();
98 void restore_sword_key_effect();
99
100 // pause key
101
102 PauseKeyEffect get_pause_key_effect();
103 void set_pause_key_effect(PauseKeyEffect pause_key_effect);
104 bool is_pause_key_enabled();
105 void set_pause_key_enabled(bool enable);
106 void save_pause_key_effect();
107 void restore_pause_key_effect();
108
109 // item keys
110 bool are_item_keys_enabled();
111 void set_item_keys_enabled(bool enable);
112
113 // all keys
114 void set_all_keys_enabled(bool enable);
115
116 private:
117
118 ActionKeyEffect action_key_effect; /**< current effect of the action key */
119 ActionKeyEffect action_key_effect_saved; /**< a previous effect of the action key */
120 bool action_key_enabled; /**< indicates that the action key is currently enabled */
121
122 AttackKeyEffect sword_key_effect; /**< current effect of the sword key */
123 AttackKeyEffect sword_key_effect_saved; /**< a previous effect of the sword key */
124 bool sword_key_enabled; /**< indicates that the sword key is currently enabled */
125
126 PauseKeyEffect pause_key_effect; /**< current effect of the pause key */
127 PauseKeyEffect pause_key_effect_saved; /**< a previous effect of the pause key */
128 bool pause_key_enabled; /**< indicates that the pause key is currently enabled */
129
130 bool item_keys_enabled;
131
132};

153};
154
155}

157#endif
