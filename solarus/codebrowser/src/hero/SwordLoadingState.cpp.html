35/**
36 * \brief Constructor.
37 * \param hero The hero controlled by this state.
38 */
39Hero::SwordLoadingState::SwordLoadingState(Hero& hero):
40 PlayerMovementState(hero, "sword loading"),
41 sword_loaded_date(0),
42 sword_loaded(false) {
43
44}

46/**
47 * \brief Starts this state.
48 * \param previous_state the previous state
49 */
50void Hero::SwordLoadingState::start(const State* previous_state) {
51
52 PlayerMovementState::start(previous_state);
53
54 sword_loaded = false;
55 sword_loaded_date = System::now() + 1000;
56}
57
58/**
59 * \brief Updates this state.
60 */
61void Hero::SwordLoadingState::update() {
62
63 PlayerMovementState::update();
64
65 if (is_suspended() || !is_current_state()) {
66 return;
67 }
68
69 uint32_t now = System::now();
70
71 // detect when the sword is loaded (i.e. ready for a spin attack)
72 if (!sword_loaded && now >= sword_loaded_date) {
73 play_load_sound();
74 sword_loaded = true;
75 }
76
77 if (!get_commands().is_command_pressed(GameCommand::ATTACK)) {
78 // the player has just released the sword key
79
80 // stop loading the sword, go to the normal state or make a spin attack
81 Hero& hero = get_entity();
82 if (!sword_loaded) {
83 // the sword was not loaded yet: go to the normal state
84 hero.set_state(new FreeState(hero));
85 }
86 else {
87 // the sword is loaded: release a spin attack
88 hero.set_state(new SpinAttackState(hero));
89 }
90 }
91}
92
93/**
94 * \brief Notifies this state that the game was just suspended or resumed.
95 * \param suspended true if the game is suspended
96 */
97void Hero::SwordLoadingState::set_suspended(bool suspended) {
98
99 PlayerMovementState::set_suspended(suspended);
100
101 if (!suspended) {
102 sword_loaded_date += System::now() - get_when_suspended();
103 }
104}

106/**
107 * \brief Notifies this state that the hero has just failed to change its
108 * position because of obstacles.
109 */
110void Hero::SwordLoadingState::notify_obstacle_reached() {
111
112 PlayerMovementState::notify_obstacle_reached();
113
114 Hero& hero = get_entity();
115 Entity* facing_entity = hero.get_facing_entity();
116
117 if (hero.is_facing_point_on_obstacle() // he is really facing an obstacle
118 && get_wanted_movement_direction8() == get_sprites().get_animation_direction8() // he is trying to move towards the obstacle
119 && (facing_entity == nullptr || !facing_entity->is_sword_ignored())) { // the obstacle allows him to tap with his sword
120
121 hero.set_state(new SwordTappingState(hero));
122 }
123}

125/**
126 * \copydoc Entity::State::notify_attacked_enemy
127 */
128void Hero::SwordLoadingState::notify_attacked_enemy(
129 EnemyAttack attack,
130 Enemy& victim,
131 const Sprite* victim_sprite,
132 EnemyReaction::Reaction& result,
133 bool killed) {
134
135 if (result.type != EnemyReaction::ReactionType::IGNORED && attack == EnemyAttack::SWORD) {
136
137 Hero& hero = get_entity();
138 if (victim.get_push_hero_on_sword()) {
139 // let SwordTappingState do the job so that no player movement interferes
140 State* state = new SwordTappingState(hero);
141 hero.set_state(state);
142 state->notify_attacked_enemy(attack, victim, victim_sprite, result, killed);
143 }
144 else {
145 // after an attack, stop loading the sword
146 hero.set_state(new FreeState(hero));
147 }
148 }
149}

198/**
199 * \brief Plays the sword loading sound.
200 */
201void Hero::SwordLoadingState::play_load_sound() {
202
203 std::ostringstream oss;
204 oss << "sword_spin_attack_load_" << get_equipment().get_ability(Ability::SWORD);
205 std::string custom_sound_name = oss.str();
206 if (Sound::exists(custom_sound_name)) {
207 Sound::play(custom_sound_name); // this particular sword has a custom loading sound effect
208 }
209 else {
210 Sound::play("sword_spin_attack_load");
211 }
212}
