72/**
73 * \brief Creates a rectangle, specifying its properties.
74 *
75 * This is the equivalent of:
76 * Rectangle(top_left.x, top_left.y,
77 * bottom_right.x - top_left.x , bottom_right.y - top_left.y)
78 *
79 * \param top_left coordinates of the top-left corner
80 * \param bottom_right coordinates of the bottom-right corner
81 */
82constexpr Rectangle::Rectangle(const Point& top_left, const Point& bottom_right):
83 Rectangle(top_left.x, top_left.y,
84 bottom_right.x - top_left.x , bottom_right.y - top_left.y)
85{}

371/**
372 * \brief Returns whether the specified point is inside this rectangle.
373 * \param x x coordinate of the point
374 * \param y y coordinate of the point
375 * \return true if the point is in this rectangle
376 */
377inline bool Rectangle::contains(int x, int y) const {
378 return x >= get_x() && x < get_x() + get_width()
379 && y >= get_y() && y < get_y() + get_height();
380}

382/**
383 * \brief Returns whether the specified point is inside this rectangle.
384 * \param point point that may be in this rectangle
385 * \return true if \a point is in this rectangle

387inline bool Rectangle::contains(const Point& point) const {
388 return contains(point.x, point.y);

391/**
392 * \brief Returns whether the specified rectangle is inside this rectangle.
393 * \param other another rectangle
394 * \return true if the specified rectangle is inside this rectangle
395 */
396inline bool Rectangle::contains(const Rectangle& other) const {
397 return contains(other.get_x(), other.get_y())
398 && contains(other.get_x() + other.get_width() - 1, other.get_y() + other.get_height() - 1);
399}

401/**
402 * \brief Returns whether or not another rectangle overlaps this one.
403 * \param other another rectangle
404 * \return true if the two rectangles overlap
405 */
406inline bool Rectangle::overlaps(const Rectangle& other) const {
407
408 int x1 = get_x();
409 int x2 = x1 + get_width();
410 int x3 = other.get_x();
411 int x4 = x3 + other.get_width();
412
413 bool overlap_x = (x3 < x2 && x1 < x4);
414
415 int y1 = get_y();
416 int y2 = y1 + get_height();
417 int y3 = other.get_y();
418 int y4 = y3 + other.get_height();
419
420 bool overlap_y = (y3 < y2 && y1 < y4);
421
422 return overlap_x && overlap_y && !is_flat() && !other.is_flat();
423}

468/**
469 * \brief Intersects this rectangle with another one.
470 * \param other Another rectangle.
471 * \return This rectangle after modification.
472 */
473inline Rectangle& Rectangle::operator&=(const Rectangle& other) {
474
475 Rectangle intersection;
476 const bool intersects = SDL_IntersectRect(
477 this->get_internal_rect(),
478 other.get_internal_rect(),
479 intersection.get_internal_rect()
480 );
481 if (intersects) {
482 *this = intersection;
483 }
484 else {
485 *this = Rectangle(0, 0, 0, 0);
486 }
487 return *this;
488}

500/**
501 * \brief Unites this rectangle with another one.
502 * \param other Another rectangle.
503 * \return This rectangle after modification.
504 */
505inline Rectangle& Rectangle::operator|=(const Rectangle& other) {
506
507 Rectangle result;
508 SDL_UnionRect(
509 this->get_internal_rect(),
510 other.get_internal_rect(),
511 result.get_internal_rect()
512 );
513 *this = result;
514 return *this;
515}

517/**
518 * \brief Compares two rectangles.
519 * \param lhs first rectangle
520 * \param rhs second rectangle
521 * \return true if both rectangles have the same coordinates and size
522 */
523inline bool operator==(const Rectangle& lhs, const Rectangle& rhs) {
524 return lhs.get_xy() == rhs.get_xy()
525 && lhs.get_size() == rhs.get_size();
526}
527
528/**
529 * \brief Compares two rectangles.
530 * \param lhs first rectangle
531 * \param rhs second rectangle
532 * \return true if the rectangles are not equal
533 */
534inline bool operator!=(const Rectangle& lhs, const Rectangle& rhs) {
535 return !(rhs == lhs);
