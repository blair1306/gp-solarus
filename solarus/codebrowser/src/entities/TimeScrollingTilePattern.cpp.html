27/**

33TimeScrollingTilePattern::TimeScrollingTilePattern(Ground ground,
34 const Point& xy, const Size& size):

39/**
40 * \brief Initializes the time-scrolling tile pattern system.
41 */
42void TimeScrollingTilePattern::initialize() {
43 shift = System::now();
44 next_shift_date = System::now();
45}
46
47/**
48 * \brief Cleans the animated time-scrolling tile pattern system.
49 */
50void TimeScrollingTilePattern::quit() {
51 shift = 0;
52 next_shift_date = 0;
53}
54
55/**
56 * \brief Updates all scrolling tiles patterns.
57 *
58 * This function is called repeatedly by the map.
59 */
60void TimeScrollingTilePattern::update() {
61
62 const uint32_t now = System::now();
63
64 while (now >= next_shift_date) {
65 shift++;
66 next_shift_date += 50;
67 }
68}

70/**
71 * \brief Draws the tile image on a surface.
72 * \param dst_surface the surface to draw
73 * \param dst_position position where tile pattern should be drawn on dst_surface
74 * \param tileset the tileset of this tile
75 * \param viewport coordinates of the top-left corner of dst_surface relative
76 * to the map (may be used for scrolling tiles)
77 */
78void TimeScrollingTilePattern::draw(
79 const SurfacePtr& dst_surface,
80 const Point& dst_position,
81 const Tileset& tileset,
82 const Point& /* viewport */
83) const {
84 Rectangle src = position_in_tileset;
85 Point dst = dst_position;
86
87 Point offset; // draw the tile with an offset that depends on the time
88
89 offset.x = src.get_width() - (shift % src.get_width());
90 offset.y = shift % src.get_height();
91
92 src.add_x(offset.x);
93 src.add_width(-offset.x);
94 src.add_y(offset.y);
95 src.add_height(-offset.y);
96 tileset.get_tiles_image()->draw_region(src, dst_surface, dst);
97
98 src = position_in_tileset;
99 dst = dst_position;
100 src.add_y(offset.y);
101 src.add_height(-offset.y);
102 dst.x += src.get_width() - offset.x;
103 src.set_width(offset.x);
104 tileset.get_tiles_image()->draw_region(src, dst_surface, dst);
105
106 src = position_in_tileset;
107 dst = dst_position;
108 src.add_x(offset.x);
109 src.add_width(-offset.x);
110 dst.y += src.get_height() - offset.y;
111 src.set_height(offset.y);
112 tileset.get_tiles_image()->draw_region(src, dst_surface, dst);
113
114 src = position_in_tileset;
115 dst = dst_position;
116 dst.x += src.get_width() - offset.x;
117 src.set_width(offset.x);
118 dst.y += src.get_height() - offset.y;
119 src.set_height(offset.y);
120 tileset.get_tiles_image()->draw_region(src, dst_surface, dst);
121}

123/**
124 * \brief Returns whether this tile pattern is animated, i.e. not always displayed
125 * the same way.
126 *
127 * Non-animated tiles may be rendered faster by using intermediate surfaces
128 * that are drawn only once.
129 *
130 * \return true if this tile pattern is animated
131 */
132bool TimeScrollingTilePattern::is_animated() const {
133 return true;
134}
