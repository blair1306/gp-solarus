28/**
29 * @brief Creates a main window.
30 * @param parent Parent object or nullptr.
31 */
32MainWindow::MainWindow(QWidget* parent) :
33 QMainWindow(parent),
34 quest_runner() {
35
36 // Set up widgets.
37 ui.setupUi(this);
38
39 // Title.
40 update_title();
41
42 // Icon.
43 const QStringList& icon_sizes = { "16", "24", "32", "48", "64", "128", "256" };
44 QIcon icon;
45 for (const QString& size : icon_sizes) {
46 icon.addPixmap(":/images/icon/solarus_launcher_icon_" + size + ".png");
47 }
48 setWindowIcon(icon);
49
50 // Console.
51 ui.console->set_quest_runner(quest_runner);
52 ui.console->set_command_enabled(false);
53 const int console_height = 150;
54 ui.console_splitter->setSizes({ height() - console_height, console_height });
55
56 // Show recent quests.
57 Settings settings;
58 const QStringList& quest_paths = settings.value("quests_paths").toStringList();
59 for (const QString& path : quest_paths) {
60 ui.quests_view->add_quest(path);
61 }
62
63 // Select the last quest played.
64 QString last_quest = settings.value("last_quest").toString();
65 if (!last_quest.isEmpty()) {
66 ui.quests_view->select_quest(last_quest);
67 }
68
69 // Menus.
70 initialize_menus();
71
72 // Actions.
73 ui.action_add_quest->setShortcut(QKeySequence::Open);
74 ui.action_exit->setShortcut(QKeySequence::Quit);
75 ui.add_button->setDefaultAction(ui.action_add_quest);
76 ui.remove_button->setDefaultAction(ui.action_remove_quest);
77
78 // Make connections.
79 connect(ui.quests_view->selectionModel(), SIGNAL(selectionChanged(QItemSelection, QItemSelection)),
80 this, SLOT(selected_quest_changed()));
81 connect(ui.play_button, SIGNAL(clicked()),
82 this, SLOT(on_action_play_quest_triggered()));
83 connect(ui.quests_view, SIGNAL(activated(QModelIndex)),
84 this, SLOT(on_action_play_quest_triggered()));
85 connect(&quest_runner, SIGNAL(running()),
86 this, SLOT(update_run_quest()));
87 connect(&quest_runner, SIGNAL(finished()),
88 this, SLOT(update_run_quest()));
89 connect(ui.console, SIGNAL(setting_changed_in_quest(QString, QVariant)),
90 this, SLOT(setting_changed_in_quest(QString, QVariant)));
91
92 selected_quest_changed();
93 update_run_quest();
94}

96/**
97 * @brief Sets an appropriate size and centers the window on the screen having
98 * the mouse.
99 */
100void MainWindow::initialize_geometry_on_screen() {
101
102 QDesktopWidget* desktop = QApplication::desktop();
103 QRect screen = desktop->screenGeometry(desktop->screenNumber(QCursor::pos()));
104
105 // Center the window on the screen where the mouse is currently.
106 int x = screen.width() / 2 - frameGeometry().width() / 2 + screen.left() - 2;
107 int y = screen.height() / 2 - frameGeometry().height() / 2 + screen.top() - 10;
108
109 move(qMax(0, x), qMax(0, y));
110}

122/**
123 * @brief Sets up the menus of the main window.
124 */
125void MainWindow::initialize_menus() {
126
127 QActionGroup* group = new QActionGroup(this);
128
129 const QMap<QAction*, QString> video_mode_names = {
130 { ui.action_filter_normal, "normal" },
131 { ui.action_filter_scale2x, "scale2x" },
132 { ui.action_filter_hq2x, "hq2x" },
133 { ui.action_filter_hq3x, "hq3x" },
134 { ui.action_filter_hq4x, "hq4x" }
135 };
136
137 for (auto it = video_mode_names.constBegin(); it != video_mode_names.constEnd(); ++it) {
138 QAction* action = it.key();
139 const QString& name = it.value();
140 group->addAction(action);
141 action->setData(name);
142 connect(action, &QAction::triggered, [this, name]() {
143 set_video_mode_requested(name);
144 });
145 }
146
147 // TODO implement the audio menu
148 delete ui.menu_audio;
149 ui.menu_audio = nullptr;
150
151 update_menus();
152}

164/**
165 * @brief Updates the video filter menu with the current settings.
166 */
167void MainWindow::update_filter_menu() {
168
169 Settings settings;
170
171 QString video_mode = settings.value("quest_video_mode", "normal").toString();
172
173 if (video_mode == "normal") {
174 ui.action_filter_normal->setChecked(true);
175 }
176 else if (video_mode == "scale2x") {
177 ui.action_filter_scale2x->setChecked(true);
178 }
179 else if (video_mode == "hq2x") {
180 ui.action_filter_hq2x->setChecked(true);
181 }
182 else if (video_mode == "hq3x") {
183 ui.action_filter_hq3x->setChecked(true);
184 }
185 else if (video_mode == "hq4x") {
186 ui.action_filter_hq4x->setChecked(true);
187 }
188 else {
189 ui.action_filter_normal->setChecked(true);
190 settings.setValue("quest_video_mode", "normal");
191 }
192}

230/**
231 * @brief Function called when the user wants to exit the program.
232 *
233 * A confirmation dialog is shown if a quest is running.
234 *
235 * @return @c false to cancel the closing operation.
236 */
237bool MainWindow::confirm_close() {
238
239 if (!quest_runner.is_started()) {
240 // No quest is running.
241 return true;
242 }
243
244 QMessageBox::StandardButton answer = QMessageBox::warning(
245 nullptr,
246 tr("A quest is playing"),
247 tr("A quest is playing. Do you really want to exit Solarus?"),
248 QMessageBox::Close | QMessageBox::Cancel
249 );
250
251 switch (answer) {
252
253 case QMessageBox::Close:
254 return true;
255
256 case QMessageBox::Cancel:
257 case QMessageBox::Escape:
258 return false;
259
260 default:
261 return false;
262 }
263
264 return true;
265}

267/**
268 * @brief Slot called when the user adds a quest to the quest list.
269 */
270void MainWindow::on_action_add_quest_triggered() {
271
272 QString quest_path = QFileDialog::getExistingDirectory(
273 this,
274 tr("Select quest directory")
275 );
276
277 if (quest_path.isEmpty()) {
278 return;
279 }
280
281 if (ui.quests_view->has_quest(quest_path)) {
282 // Quest already in the list.
283 ui.quests_view->select_quest(quest_path);
284 return;
285 }
286
287 // Add to the quest list view.
288 if (!ui.quests_view->add_quest(quest_path)) {
289 GuiTools::error_dialog("No quest was found in this directory");
290 return;
291 }
292
293 // Remember the quest list.
294 Settings settings;
295 settings.setValue("quests_paths", ui.quests_view->get_paths());
296
297 // Select it.
298 ui.quests_view->select_quest(quest_path);
299}

301/**
302 * @brief Slot called when the user removes a quest to the quest list.
303 */
304void MainWindow::on_action_remove_quest_triggered() {
305
306 int selected_index = ui.quests_view->get_selected_index();
307
308 if (selected_index == -1) {
309 return;
310 }
311
312 // Remove from the quest list view.
313 if (ui.quests_view->remove_quest(selected_index)) {
314
315 // Update the quest list.
316 Settings settings;
317 settings.setValue("quests_paths", ui.quests_view->get_paths());
318 }
319
320 // Select the next one.
321 int num_quests = ui.quests_view->get_num_quests();
322 selected_index = qMin(selected_index, num_quests - 1);
323 ui.quests_view->select_quest(selected_index);
324}

336/**
337 * @brief Slot called when the user triggers the "Play quest" action.
338 */
339void MainWindow::on_action_play_quest_triggered() {
340
341 if (quest_runner.is_started()) {
342 return;
343 }
344
345 QString path = ui.quests_view->get_selected_path();
346 if (path.isEmpty()) {
347 return;
348 }
349
350 // Write system settings to the settings.dat file of the quest.
351 Settings settings;
352 settings.export_to_quest(path);
353
354 // Run the quest.
355 quest_runner.start(path);
356
357 update_run_quest();
358}

374/**
375 * @brief Slot called when the user triggers the "Fullscreen" action.
376 */
377void MainWindow::on_action_fullscreen_triggered() {
378
379 bool fullscreen = ui.action_fullscreen->isChecked();
380
381 Settings settings;
382 bool previous = settings.value("quest_fullscreen", false).toBool();
383 if (fullscreen == previous) {
384 return;
385 }
386
387 settings.setValue("quest_fullscreen", fullscreen);
388
389 if (quest_runner.is_started()) {
390 // Change the setting in the current quest process.
391 QString command = QString("sol.video.set_fullscreen(%1)").arg(fullscreen ? "true" : "false");
392 ui.console->execute_command(command);
393 }
394}

457/**
458 * @brief Slot called when the quest has just started or stopped.
459 */
460void MainWindow::update_run_quest() {
461
462 QString selected_path = ui.quests_view->get_selected_path();
463 bool has_current = !selected_path.isEmpty();
464 bool playing = quest_runner.is_started();
465
466 bool enable_play = has_current && !playing;
467 bool enable_stop = has_current && playing;
468 ui.action_play_quest->setEnabled(enable_play);
469 ui.play_button->setEnabled(enable_play);
470 ui.action_stop_quest->setEnabled(enable_stop);
471
472 ui.menu_zoom->setEnabled(playing);
473}

475/**
476 * @brief Slot called when the selection changes in the quest list.
477 */
478void MainWindow::selected_quest_changed() {
479
480 QString selected_path = ui.quests_view->get_selected_path();
481 bool has_current = !selected_path.isEmpty();
482
483 // Enable/disable buttons.
484 ui.action_remove_quest->setEnabled(has_current);
485 update_run_quest();
486
487 // Save the last selected quest (including if none).
488 Settings settings;
489 settings.setValue("last_quest", selected_path);
490
491 static const QPixmap default_pixmap = QPixmap(":/images/no_logo.png");
492
493 if (!has_current) {
494 ui.quest_info_panel->setEnabled(false);
495
496 ui.quest_title_value->clear();
497 ui.quest_title_value->setVisible(false);
498
499 ui.quest_author_label->setVisible(false);
500 ui.quest_author_value->setVisible(false);
501 ui.quest_author_value->clear();
502
503 ui.quest_version_label->setVisible(false);
504 ui.quest_version_value->setVisible(false);
505 ui.quest_version_value->clear();
506
507 ui.quest_long_description_value->setVisible(false);
508 ui.quest_long_description_value->clear();
509
510 ui.quest_box_label->setPixmap(default_pixmap);
511 }
512 else {
513 ui.quest_info_panel->setEnabled(true);
514
515 const Solarus::QuestProperties properties =
516 ui.quests_view->get_selected_quest_properties();
517
518 ui.quest_box_label->setPixmap(ui.quests_view->get_selected_logo());
519
520 QString title = QString::fromStdString(properties.get_title());
521 ui.quest_title_value->setVisible(true);
522 ui.quest_title_value->setText(title);
523
524 QString author = QString::fromStdString(properties.get_author());
525 if (author.isEmpty()) {
526 ui.quest_author_label->setVisible(false);
527 ui.quest_author_value->setVisible(false);
528 }
529 else {
530 ui.quest_author_label->setVisible(true);
531 ui.quest_author_value->setText(author);
532 ui.quest_author_value->setVisible(true);
533 }
534
535 QString quest_version =
536 QString::fromStdString(properties.get_quest_version());
537 if (quest_version.isEmpty()) {
538 ui.quest_version_label->setVisible(false);
539 ui.quest_version_value->setVisible(false);
540 }
541 else {
542 ui.quest_version_label->setVisible(true);
543 ui.quest_version_value->setText(quest_version);
544 ui.quest_version_value->setVisible(true);
545 }
546
547 QString long_description =
548 QString::fromStdString(properties.get_long_description());
549 ui.quest_long_description_value->setVisible(true);
550 ui.quest_long_description_value->setText(long_description);
551 }

554/**
555 * @brief Slot called when a setting changes in the currently running quest.
556 *
557 * The GUI is updated to reflect and remember the new setting.
558 *
559 * @param key Name of the system setting that has just changed.
560 * @param value The new value.
561 */
562void MainWindow::setting_changed_in_quest(
563 const QString& key, const QVariant& value
564) {
565
566 Settings settings;
567 if (key == "quest_video_mode") {
568 settings.setValue(key, value);
569 update_filter_menu();
570 }
571 else if (key == "quest_fullscreen") {
572 settings.setValue(key, value);
573 update_fullscreen_action();
574 }
575}

577/**
578 * @brief Slot called when the user wants to change the video mode.
579 * @param video_mode_name The new video mode.
580 */
581void MainWindow::set_video_mode_requested(const QString& video_mode_name) {
582
583 Settings settings;
584 const QString& previous = settings.value("quest_video_mode").toString();
585 if (video_mode_name == previous) {
586 return;
587 }
588
589 settings.setValue("quest_video_mode", video_mode_name);
590
591 if (quest_runner.is_started()) {
592 // Change the video mode of the current quest process.
593 QString command = QString("sol.video.set_mode(\"%1\")").arg(video_mode_name);
594 ui.console->execute_command(command);
595 }
596}

598/**
599 * @brief Slot called when the user wants to change the window zoom.
600 * @param zoom The new zoom factor.
601 */
602void MainWindow::set_zoom_requested(int zoom) {
603
604 if (!quest_runner.is_started()) {
605 return;
606 }
607
608 QString command = QString("local w, h = sol.video.get_quest_size(); sol.video.set_window_size(w * %1, h * %2)").arg(zoom).arg(zoom);
609 ui.console->execute_command(command);
610}
