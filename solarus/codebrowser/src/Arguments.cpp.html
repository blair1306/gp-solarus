17#include "solarus/Arguments.h"
18#include <algorithm>
19
20namespace Solarus {
21
22/**
23 * \brief Creates an empty argument object.
24 */
25Arguments::Arguments() {
26}
27
28/**
29 * \brief Creates an argument object from standard main() arguments.
30 * \param argc Number of command-line arguments.
31 * \param argv Command-line arguments. The first one, if any, is the name
32 * of the program.
33 */
34Arguments::Arguments(int argc, char** argv) {
35
36 if (argc >= 1) {
37 program_name = argv[0];
38 }
39 for (int i = 1; i < argc; ++i) {
40 if (argv[i] != nullptr) {
41 args.push_back(argv[i]);
42 }
43 }
44}
45
46/**
47 * \brief Returns the program name, usually the first argument of main().
48 * \return The program name, or an empty string if unknown.
49 */
50const std::string& Arguments::get_program_name() const {
51 return program_name;
52}
53
54/**
55 * \brief Sets the program name of this argument list.
56 * \param program_name The program name.
57 */
58void Arguments::set_program_name(const std::string& program_name) {
59 this->program_name = program_name;
60}
61
62/**
63 * \brief Returns the list of arguments passed to main, except the program name.
64 * \return The list of arguments.
65 */
66const std::vector<std::string>& Arguments::get_arguments() const {
67 return args;
68}
69
70/**
71 * \brief Returns whether the following exact string was passed as an argument.
72 * \param argument The string to test.
73 * \return \c true if an argument is exactly this string.
74 */
75bool Arguments::has_argument(const std::string& argument) const {
76
77 return std::find(args.begin(), args.end(), argument) != args.end();
78}
79
80/**
81 * \brief If there is an argument of the form \c key=value, returns the value.
82 * \param key The key to look for.
83 * \return The value that was passed for this key, or an empty string if
84 * no such argument was passed.
85 */
86std::string Arguments::get_argument_value(const std::string& key) const {
87
88 for (const std::string& arg: args) {
89 if (arg.size() > key.size()
90 && arg.substr(0, key.size() + 1) == key + '=') {
91 return arg.substr(key.size() + 1);
92 }
93 }
94
95 return "";
96}
97
98/**
99 * \brief Adds an argument.
100 * \param argument The string to add.
101 */
102void Arguments::add_argument(const std::string& argument) {
103
104 args.push_back(argument);
105}
106
107/**
108 * \brief Adds an argument as a key and value.
109 *
110 * This is equivalent to <tt>add_argument(key + "=" + value)</tt>.
111 *
112 * \param key Key of the argument to add.
113 * \param value Value of the argument to add.
114 */
115void Arguments::add_argument(const std::string& key, const std::string& value) {
116
117 add_argument(key + "=" + value);
118}
