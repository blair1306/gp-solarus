17#ifndef SOLARUS_GUI_CONSOLE_H
18#define SOLARUS_GUI_CONSOLE_H
19
20#include "solarus/gui/gui_common.h"
21#include "ui_console.h"
22#include <QMap>
23#include <QPointer>
24
25namespace SolarusGui {
26
27class QuestRunner;
28
29/**
30 * @brief Quest execution log viewer and interactive console to execute Lua
31 * commands.
32 */
33class SOLARUS_GUI_API Console : public QWidget {
34 Q_OBJECT
35
36public:
37
38 explicit Console(QWidget* parent = nullptr);
39
40 void set_quest_runner(QuestRunner& quest_runner);
41
42 bool is_command_enabled() const;
43 void set_command_enabled(bool enable);
44 int execute_command(const QString& command);
45
46signals:
47
48 void setting_changed_in_quest(const QString& key, const QVariant& value);
49 void command_result_received(int id, const QString& command, bool success, const QString& result);
50
51private slots:
52
53 void quest_running();
54 void quest_finished();
55 void quest_output_produced(const QStringList& lines);
56 void command_field_activated();
57
58private:
59
60 void parse_output(const QString& line);
61 bool detect_command_result(const QString& log_level, const QString& message);
62 void detect_setting_change(const QString& log_level, const QString& message);
63 QString colorize_output(const QString& log_level, const QString& message);
64
65 QStringList get_quest_lua_commands_from_settings() const;
66 bool apply_settings();
67
68 Ui::Console ui; /**< The widgets. */
69 QPointer<QuestRunner> quest_runner; /**< The quest execution. */
70 QMap<int, QString> pending_commands; /**< Commands for which we are waiting a result. */
71 int output_command_id; /**< Id of the command we are reading the result of
72 * (-1 if none). */
73 QString output_command_result; /**< Partial result of the command. */
74 bool command_enabled; /**< Whether the user can type Lua commands. */
75
76};
77
78}
79
80#endif
