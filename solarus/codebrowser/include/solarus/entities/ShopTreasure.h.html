20#include "solarus/Common.h"
21#include "solarus/Treasure.h"
22#include "solarus/Sprite.h"

28struct lua_State;

30namespace Solarus {
31
32/**
33 * \brief A treasure the hero can buy in a shop.
34 *
35 * The treasure is automatically drawn with its icon,
36 * the amount and the price. When the hero presses the action
37 * key in front of it, a dialog is automatically showed and
38 * the hero can buy the item.
39 */
40class ShopTreasure: public Entity {
41
42 public:
43
44 static constexpr EntityType ThisType = EntityType::SHOP_TREASURE;
45
46 ShopTreasure(
47 const std::string& name,
48 int layer,
49 const Point& xy,
50 const Treasure& treasure,
51 int price,
52 const std::string& font_id,
53 const std::string& dialog_id
54 );
55
56 static std::shared_ptr<ShopTreasure> create(
57 Game& game,
58 const std::string& name,
59 int layer,
60 const Point& xy,
61 const Treasure& treasure,
62 int price,
63 const std::string& font_id,
64 const std::string& dialog_id
65 );
66
67 virtual EntityType get_type() const override;
68
69 const Treasure& get_treasure() const;
70 int get_price() const;
71 const std::string& get_dialog_id() const;
72
73 bool is_sword_ignored() const override;
74 bool is_obstacle_for(Entity& other) override;
75 void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
76 bool notify_action_command_pressed() override;
77
78 void update() override;
79 void draw_on_map() override;
80
81 private:
82
83 // Data.
84 Treasure treasure; /**< The treasure the player can buy. */
85 int price; /**< The treasure's price. */
86 std::string dialog_id; /**< Id of the dialog describing the treasure. */
87
88 // Displaying.
89 SpritePtr treasure_sprite; /**< Sprite of the treasure. */
90 SpritePtr rupee_icon_sprite; /**< The rupee icon near the price. */
91 TextSurface price_digits; /**< The digits that show the price. */
92
93};
94
95}
96
97#endif
