25/**
26 * \brief Creates and starts a timer.
27 * \param duration Duration of the timer in milliseconds.
28 */
29Timer::Timer(uint32_t duration):
30 expiration_date(System::now() + duration),
31 duration(duration),
32 finished(System::now() >= this->expiration_date),
33 suspended_with_map(false),
34 suspended(false),
35 when_suspended(0),
36 next_sound_date(0) {
37
38}
39
40/**
41 * \brief Returns whether a clock sound is played during this timer.
42 * \return true if a clock sound is played.
43 */
44bool Timer::is_with_sound() const {
45 return next_sound_date != 0;
46}
47
48/**
49 * \brief Sets whether a clock sound is played during this timer.
50 * \param with_sound true to play a clock sound during this timer.
51 */
52void Timer::set_with_sound(bool with_sound) {
53 next_sound_date = with_sound ? System::now() : 0;
54}
55
56/**
57 * \brief Returns whether this timer is currently suspended.
58 * \return true if this timer is suspended.
59 */
60bool Timer::is_suspended() const {
61 return suspended;
62}
63
64/**
65 * \brief Suspends or resumes the timer.
66 *
67 * It is okay to call this function when is_suspended_with_map() is true:
68 * this means that you temporarily override the automatic suspending behavior.
69 *
70 * \param suspended true to suspend the timer, false to resume it.
71 */
72void Timer::set_suspended(bool suspended) {
73
74 if (suspended != this->suspended) {
75 this->suspended = suspended;
76
77 uint32_t now = System::now();
78
79 if (suspended) {
80 // the timer is being suspended
81 when_suspended = now;
82 }
83 else {
84 // recalculate the expiration date
85 if (when_suspended != 0) {
86 expiration_date += now - when_suspended;
87 if (is_with_sound()) {
88 next_sound_date += now - when_suspended;
89 }
90 }
91 }
92 }
93}
94
95/**
96 * \brief Returns whether this timer is suspended when the map is suspended.
97 * \return true if this timer is suspended with the map.
98 */
99bool Timer::is_suspended_with_map() const {
100 return suspended_with_map;
101}
102
103/**
104 * \brief Sets whether this timer should be suspended when the map is suspended.
105 * \param suspended_with_map true to suspend this timer with the map.
106 */
107void Timer::set_suspended_with_map(bool suspended_with_map) {
108
109 if (suspended_with_map != this->suspended_with_map) {
110 this->suspended_with_map = suspended_with_map;
111
112 if (is_suspended() && !suspended_with_map) {
113 set_suspended(false);
114 }
115 }
116}

142/**
143 * \brief Sets the expiration date of this timer.
144 * \param expiration_date The expiration date in milliseconds.
145 */
146void Timer::set_expiration_date(uint32_t expiration_date) {
147
148 this->expiration_date = expiration_date;
149 this->finished = System::now() >= this->expiration_date;
150}
151
152/**
153 * \brief Updates the timer.
154 */
155void Timer::update() {
156
157 if (suspended || is_finished()) {
158 return;
159 }
160
161 // check the time
162 uint32_t now = System::now();
163 finished = (now >= expiration_date);
164
165 // play the sound
166 if (is_with_sound() && now >= next_sound_date) {
167
168 uint32_t remaining_time = expiration_date - now;
169 if (remaining_time > 6000) {
170 Sound::play("timer");
171 next_sound_date += 1000;
172 }
173 else {
174 Sound::play("timer_hurry");
175 if (remaining_time > 2000) {
176 next_sound_date += 1000;
177 }
178 else {
179 next_sound_date += 250;
180 }
181 }
182 }
183}
184
185/**
186 * \brief Notifies this timer that the current map is being suspended or resumed.
187 * \param suspended true if the map is suspended, false if it is resumed.
188 */
189void Timer::notify_map_suspended(bool suspended) {
190
191 if (suspended_with_map) {
192 set_suspended(suspended);
193 }
194}
