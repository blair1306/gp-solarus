17#ifndef SOLARUS_PIXEL_MOVEMENT_H
18#define SOLARUS_PIXEL_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/movements/Movement.h"
22#include "solarus/lowlevel/Point.h"
23#include <cstdint>
24#include <list>
25#include <string>
26
27namespace Solarus {
28
29/**
30 * \brief Movement of an entity that follows a predetermined sequence of pixel-precise moves.
31 */
32class SOLARUS_API PixelMovement: public Movement {
33
34 public:
35
36 // creation and destruction
37 PixelMovement(const std::string& trajectory_string, uint32_t delay, bool loop, bool ignore_obstacles);
38
39 // properties
40 const std::list<Point>& get_trajectory() const;
41 void set_trajectory(const std::list<Point>& trajectory);
42 void set_trajectory(const std::string &trajectory_string);
43 uint32_t get_delay() const;
44 void set_delay(uint32_t delay);
45 bool get_loop() const;
46 void set_loop(bool loop);
47
48 // state
49 virtual bool is_started() const override;
50 virtual bool is_finished() const override;
51 int get_length() const;
52
53 virtual void update() override;
54 virtual void set_suspended(bool suspended) override;
55
56 virtual const std::string& get_lua_type_name() const override;
57
58 protected:
59
60 virtual void notify_step_done(int step_index, bool success);
61
62 private:
63
64 void make_next_step();
65 void restart();
66
67 // movement properties
68
69 std::list<Point> trajectory; /**< The trajectory. Each element of the
70 * represents a move in pixels. */
71 std::string trajectory_string; /**< String representation of the trajectory, like "dx1 dy1 dx2 dy2 dx3 dy3 ..." */
72 uint32_t next_move_date; /**< Date of the next move */
73 uint32_t delay; /**< Delay in milliseconds between two translations. */
74 bool loop; /**< Should the trajectory return to the beginning once finished? */
75
76 // current state
77
78 std::list<Point>::iterator
79 trajectory_iterator; /**< Current element of the trajectory. */
80 int nb_steps_done; /**< Number of steps already done in the trajectory */
81 bool finished; /**< Indicates whether the object has reached the end of the trajectory
82 * (only possible when loop is false). */
83
84};
85
86}
87
88#endif
