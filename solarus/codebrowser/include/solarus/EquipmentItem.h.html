17#ifndef SOLARUS_EQUIPMENT_ITEM_H
18#define SOLARUS_EQUIPMENT_ITEM_H
19
20#include "solarus/Common.h"
21#include "solarus/lua/ExportableToLua.h"
22#include "solarus/Ability.h"
23#include <string>
24
25namespace Solarus {
26
27class Equipment;
28class Game;
29class LuaContext;
30class Map;
31class Pickable;
32class Savegame;
33
34/**
35 * \brief An item possibly possessed by the player.
36 *
37 * Its properties are controlled by the script of the item.
38 */
39class EquipmentItem: public ExportableToLua {
40
41 public:
42
43 explicit EquipmentItem(Equipment& equipment);
44
45 Equipment& get_equipment();
46 Game* get_game();
47 Savegame& get_savegame();
48 const Savegame& get_savegame() const;
49 LuaContext& get_lua_context();
50
51 // Properties.
52 const std::string& get_name() const;
53 void set_name(const std::string& name);
54 bool is_saved() const;
55 const std::string& get_savegame_variable() const;
56 void set_savegame_variable(const std::string& savegame_variable);
57 bool has_amount() const;
58 const std::string& get_amount_savegame_variable() const;
59 void set_amount_savegame_variable(const std::string& amount_savegame_variable);
60 bool is_obtainable() const;
61 void set_obtainable(bool obtainable);
62 bool is_assignable() const;
63 void set_assignable(bool assignable);
64 bool get_can_disappear() const;
65 void set_can_disappear(bool can_disappear);
66 bool get_brandish_when_picked() const;
67 void set_brandish_when_picked(bool brandish_when_picked);
68 const std::string& get_sound_when_picked() const;
69 void set_sound_when_picked(const std::string& sound_when_picked);
70 const std::string& get_sound_when_brandished() const;
71 void set_sound_when_brandished(const std::string& sound_when_brandished);
72 const std::string& get_shadow() const;
73 void set_shadow(const std::string& shadow);
74
75 // Current state.
76 void initialize();
77 void exit();
78 void update();
79 void set_suspended(bool suspended);
80 void start();
81 void notify_map_changed(Map& map);
82 void notify_amount_changed(int amount);
83 void notify_variant_changed(int variant);
84 void notify_using();
85 void notify_ability_used(Ability ability);
86 void notify_pickable_appeared(Pickable& pickable);
87
88 int get_variant() const;
89 void set_variant(int variant);
90
91 int get_amount() const;
92 void set_amount(int amount);
93
94 int get_max_amount() const;
95 void set_max_amount(int max_amount);
96
97 virtual const std::string& get_lua_type_name() const override;
98
99 private:
100
101 Equipment& equipment; /**< the equipment object that manages all items */
102 std::string name; /**< name that identifies this item */
103 std::string savegame_variable; /**< savegame variable that stores the possession state */
104 std::string amount_savegame_variable; /**< savegame variable that stores the amount associated to this item
105 * or an empty string if there is no amount */
106 int max_amount; /**< limit of the amount associated to this item, or 0 */
107 bool obtainable; /**< whether the player can receive this item */
108 bool assignable; /**< indicates that this item can be assigned to an item key an then
109 * be used explicitly */
110 bool can_disappear; /**< when the item is dropped on the ground, indicates that it will disappear
111 * after some time */
112 bool brandish_when_picked; /**< when the item is picked on the ground, indicates that the hero
113 * brandishes it */
114 std::string sound_when_picked; /**< sound to play when the item is picked on the ground (or an empty string) */
115 std::string sound_when_brandished; /**< sound to play when the item is brandished (or an empty string) */
116 std::string shadow; /**< animation of the shadow sprite when the item is on the ground (or an empty string) */
117
118};
119
120}
121
122#endif
