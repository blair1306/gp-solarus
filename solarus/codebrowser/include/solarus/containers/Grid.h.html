18#define SOLARUS_GRID_H

27namespace Solarus {
28
29/**
30 * \brief A collection of objects spatially located in a grid.
31 *
32 * In the current implementation, objects cannot move or be resized.
33 */
34template <typename T>
35class Grid {
36
37 public:
38
39 Grid(const Size& grid_size, const Size& cell_size);
40
41 const Size& get_grid_size() const;
42 const Size& get_cell_size() const;
43 size_t get_num_rows() const;
44 size_t get_num_columns() const;
45 size_t get_num_cells() const;
46
47 void clear();
48 void add(const T& element, const Rectangle& bounding_box);
49
50 const std::vector<T>& get_elements(size_t cell_index) const;
51 void get_elements(const Rectangle& where,
52 std::vector<T>& elements) const;
53
54 private:
55
56 const Size grid_size;
57 const Size cell_size;
58 size_t num_rows;
59 size_t num_columns;
60 std::vector<std::vector<T>> elements; /**< Two-dimensional array of cells. */
61
62};
63
64/**
65 * \brief Constructor.
66 * \param grid_size Size of the grid to create.
67 * \param cell_size Size of a cell of the grid. It is allowed that the size of
68 * the grid is not a multiple of the cell size.
69 */
70template <typename T>
71Grid<T>::Grid(const Size& grid_size, const Size& cell_size):
72 grid_size(grid_size),
73 cell_size(cell_size),
74 num_rows(0),
75 num_columns(0) {
76
77 Debug::check_assertion(grid_size.width > 0 && grid_size.height > 0,
78 "Invalid grid size");
79 Debug::check_assertion(cell_size.width > 0 && cell_size.height > 0,
80 "Invalid cell size");
81
82 num_rows = grid_size.height / cell_size.height;
83 if (grid_size.height % cell_size.height != 0) {
84 ++num_rows;
85 }
86
87 num_columns = grid_size.width / cell_size.width;
88 if (grid_size.width % cell_size.width != 0) {
89 ++num_columns;
90 }
91 elements.resize(num_rows * num_columns);
92}
93
94/**
95 * \brief Returns the size of the grid.
96 * \return The size of the grid.
97 */
98template <typename T>
99const Size& Grid<T>::get_grid_size() const {
100 return grid_size;
101}
102
103/**
104 * \brief Returns the size of a cell of the grid.
105 * \return The size of a cell.
106 */
107template <typename T>
108const Size& Grid<T>::get_cell_size() const {
109 return cell_size;
110}
111
112/**
113 * \brief Returns the number of rows in the grid.
114 * \return The number of rows.
115 */
116template <typename T>
117size_t Grid<T>::get_num_rows() const {
118 return num_rows;
119}
120
121/**
122 * \brief Returns the number of columns in the grid.
123 * \return The number of columns.
124 */
125template <typename T>
126size_t Grid<T>::get_num_columns() const {
127 return num_columns;
128}
129
130/**
131 * \brief Returns the total number of cells in the grid.
132 * \return The number of cells.
133 */
134template <typename T>
135size_t Grid<T>::get_num_cells() const {
136 return elements.size();
137}
138
139/**
140 * \brief Returns all elements in the specified cell.
141 * \param cell_index Index of a cell in the grid.
142 * \return The elements in this cell.
143 */
144template <typename T>
145const std::vector<T>& Grid<T>::get_elements(size_t cell_index) const {
146
147 Debug::check_assertion(cell_index < get_num_cells(),
148 "Invalid index");
149
150 return elements[cell_index];
151}
152
153/**
154 * \brief Returns all elements in the specified rectangle.
155 *
156 * Elements that are in several cells are returned only once.
157 *
158 * \param where The area to get.
159 * \param[out] elements The vector to fill.
160 */
161template <typename T>
162void Grid<T>::get_elements(
163 const Rectangle& where,
164 std::vector<T>& elements) const {
165
166 const size_t row1 = where.get_y() / num_rows;
167 const size_t row2 = where.get_y() + where.get_height() / num_rows;
168 const size_t column1 = where.get_x() / num_columns;
169 const size_t column2 = where.get_x() + where.get_width() / num_columns;
170
171 if (row1 > row2 || column1 > column2) {
172 // No cell.
173 return;
174 }
175
176 std::set<T> elements_added;
177 for (size_t i = row1; i <= row2; ++i) {
178 if (i >= num_rows) {
179 continue;
180 }
181
182 for (size_t j = column1; j <= column2; ++j) {
183 if (j >= num_columns) {
184 continue;
185 }
186
187 const std::vector<T>& in_cell = this->elements[i * num_columns + j];
188 for (const T& element: in_cell) {
189 if (elements_added.find(element) == elements_added.end()) {
190 elements_added.insert(element);
191 elements.push_back(element);
192 }
193 }
194 }
195 }
196}
197
198/**
199 * \brief Removes all elements in the grid.
200 */
201template <typename T>
202void Grid<T>::clear() {
203
204 elements.clear();
205 elements.resize(num_rows * num_columns);
206}
207
208/**
209 * \brief Adds an element in the grid.
210 *
211 * The element will be added to all cells it overlaps.
212 *
213 * \param element The element to add.
214 * \param bounding_box Bounding box of the element.
215 */
216template <typename T>
217void Grid<T>::add(const T& element, const Rectangle& bounding_box) {
218
219 const int row1 = bounding_box.get_y() / cell_size.height;
220 const int row2 = (bounding_box.get_y() + bounding_box.get_height()) / cell_size.height;
221 const int column1 = bounding_box.get_x() / cell_size.width;
222 const int column2 = (bounding_box.get_x() + bounding_box.get_width()) / cell_size.width;
223
224 if (row1 > row2 || column1 > column2) {
225 // No cell.
226 return;
227 }
228
229 for (int i = row1; i <= row2; ++i) {
230 if (i < 0 || i >= (int) num_rows) {
231 continue;
232 }
233
234 for (int j = column1; j <= column2; ++j) {
235 if (j < 0 || j >= (int) num_columns) {
236 continue;
237 }
238
239 elements[i * num_columns + j].push_back(element);
240 }
241 }
242}
243
244}
