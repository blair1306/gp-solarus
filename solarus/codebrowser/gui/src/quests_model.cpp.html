36/**
37 * @brief Returns the number of quests.
38 * @param parent Parent index.
39 * @return The number of quests.
40 */
41int QuestsModel::rowCount(const QModelIndex& parent) const {
42
43 Q_UNUSED(parent);
44 return quests.size();
45}

47/**
48 * @brief Returns the data of an item for the given role.
49 * @param index Index of the item to get.
50 * @param role Kind of data to get.
51 * @return The data for this item and this role.
52 */
53QVariant QuestsModel::data(const QModelIndex& index, int role) const {
54
55 if (index.row() < 0 || index.row() >= rowCount()) {
56 return QVariant();
57 }
58
59 switch (role) {
60
61 case Qt::ItemDataRole::DisplayRole:
62 load_icon(index.row());
63 return QVariant::fromValue(quests.at(index.row()));
64
65 case Qt::ItemDataRole::ToolTipRole:
66 return QString::fromStdString(
67 quests.at(index.row()).properties.get_title());
68
69 default:
70 return QVariant();
71 }
72}

74/**
75 * @brief Finds the index of a quest in the list.
76 * @param quest_path Path of the quest to look for.
77 * @return The quest index, or -1 if it is not in the list.
78 */
79int QuestsModel::path_to_index(const QString& quest_path) const {
80
81 for (size_t i = 0; i < quests.size(); ++i) {
82 if (quests[i].path == quest_path) {
83 return i;
84 }
85 }
86
87 return -1;
88}

90/**
91 * @brief Returns the quest path at the given index.
92 * @param quest_index Index of the quest to get.
93 * @return The path, or an empty string if the index is invalid.
94 */
95QString QuestsModel::index_to_path(int quest_index) const {
96
97 if (quest_index < 0 || quest_index >= (int) quests.size()) {
98 return QString();
99 }
100
101 return quests[quest_index].path;
102}

104/**
105 * @brief Returns whether the model contains the given quest path.
106 * @param quest_path Quest path to test.
107 * @return @c true if it is in the model.
108 */
109bool QuestsModel::has_quest(const QString& quest_path) {
110
111 return path_to_index(quest_path) != -1;
112}

114/**
115 * @brief Adds a quest to the model.
116 * @param quest_path Path of the quest to add.
117 * @return @c true if it was added, @c false if there is no such quest
118 * or if it is already there.
119 */
120bool QuestsModel::add_quest(const QString& quest_path) {
121
122 if (has_quest(quest_path)) {
123 return false;
124 }
125
126 QuestInfo info;
127
128 // Open the quest to get its quest.dat file.
129 QStringList arguments = QApplication::arguments();
130 QString program_name = arguments.isEmpty() ? QString() : arguments.first();
131 if (!Solarus::QuestFiles::open_quest(program_name.toStdString(),
132 quest_path.toStdString())) {
133 Solarus::QuestFiles::close_quest();
134 return false;
135 }
136 info.properties = Solarus::CurrentQuest::get_properties();
137 Solarus::QuestFiles::close_quest();
138
139 const int num_quests = rowCount();
140 beginInsertRows(QModelIndex(), num_quests, num_quests);
141
142 info.path = quest_path;
143 info.directory_name = quest_path.section('/', -1, -1, QString::SectionSkipEmpty);
144 quests.push_back(info);
145
146 endInsertRows();
147
148 return true;
149}
150
151/**
152 * @brief Removes a quest from this model.
153 * @param quest_index Index of the quest to remove.
154 * @return @c true if the quest was removed, @c false if there is no quest
155 * with this index.
156 */
157bool QuestsModel::remove_quest(int quest_index) {
158
159 if (quest_index < 0 || quest_index > rowCount()) {
160 return false;
161 }
162
163 beginRemoveRows(QModelIndex(), quest_index, quest_index);
164 quests.erase(quests.begin() + quest_index);
165 endRemoveRows();
166 return true;
167}

207/**
208 * @brief Returns the quest logo.
209 * @param quest_index Index of the quest to get.
210 * @return The quest logo.
211 */
212const QPixmap& QuestsModel::get_quest_logo(int quest_index) const {
213
214 if (quest_index < 0 || quest_index > rowCount()) {
215 return get_quest_default_logo();
216 }
217
218 QuestInfo& quest = quests[quest_index];
219 if (quest.logo.isNull()) {
220 // Lazily load the logo.
221 quest.logo = get_quest_default_logo();
222
223 QStringList arguments = QApplication::arguments();
224 QString program_name = arguments.isEmpty() ? QString() : arguments.first();
225 if (Solarus::QuestFiles::open_quest(program_name.toStdString(),
226 quest.path.toStdString())) {
227 std::string file_name = "logos/logo.png";
228 if (Solarus::QuestFiles::data_file_exists(file_name)) {
229 std::string buffer = Solarus::QuestFiles::data_file_read(file_name);
230 QPixmap pixmap;
231 if (pixmap.loadFromData((const uchar*) buffer.data(), (uint) buffer.size())) {
232 quests[quest_index].logo = pixmap;
233 }
234 }
235 }
236 Solarus::QuestFiles::close_quest();
237 }
238
239 return quest.logo;
240}

253/**
254 * @brief Loads the icon of the given quest.
255 * @param quest_index Index of a quest.
256 */
257void QuestsModel::load_icon(int quest_index) const {
258
259 if (quest_index < 0 || quest_index > rowCount()) {
260 return;
261 }
262
263 QuestInfo& quest = quests[quest_index];
264 QIcon& icon = quest.icon;
265 if (!icon.isNull()) {
266 // Already loaded.
267 return;
268 }
269
270 QStringList arguments = QApplication::arguments();
271 QString program_name = arguments.isEmpty() ? QString() : arguments.first();
272 if (!Solarus::QuestFiles::open_quest(program_name.toStdString(),
273 quest.path.toStdString())) {
274 Solarus::QuestFiles::close_quest();
275 icon = get_quest_default_icon();
276 return;
277 }
278
279 QStringList file_names = {
280 "logos/icon_16.png",
281 "logos/icon_24.png",
282 "logos/icon_32.png",
283 "logos/icon_48.png",
284 "logos/icon_64.png",
285 "logos/icon_128.png",
286 "logos/icon_256.png",
287 "logos/icon_512.png",
288 "logos/icon_1024.png",
289 };
290 for (const QString& file_name : file_names) {
291 if (Solarus::QuestFiles::data_file_exists(file_name.toStdString())) {
292 std::string buffer = Solarus::QuestFiles::data_file_read(file_name.toStdString());
293 QPixmap pixmap;
294 if (!pixmap.loadFromData((const uchar*) buffer.data(), (uint) buffer.size())) {
295 continue;
296 }
297 icon.addPixmap(pixmap);
298 }
299 }
300 Solarus::QuestFiles::close_quest();
301
302 if (icon.isNull()) {
303 icon = get_quest_default_icon();
304 }
305}

317/**
318 * @brief Sort the quests and notify the view
319 * @param column int value of a QuestSort value, column to sort the quests by
320 * @param order Order to the quests in the list by
321 */
322void QuestsModel::sort(int column, Qt::SortOrder order) {
323
324 do_sort(static_cast<QuestSort>(column), order);
325}
326
327/**
328 * @brief Do the actual sort, without notifying the view
329 * @param sortType Way to sort the quests in the list by
330 * @param order Order to sort the quests in the list by
331 */
332void QuestsModel::do_sort(QuestSort sortType, Qt::SortOrder order) {
333
334 std::sort(quests.begin(), quests.end(),
335 [sortType, order](const QuestInfo& a, const QuestInfo& b) {
336 const bool ascending = order == Qt::AscendingOrder;
337 switch (sortType) {
338 case QuestSort::SortByAuthor:
339 return ascending ? a.properties.get_author() < b.properties.get_author()
340 : a.properties.get_author() > b.properties.get_author();
341 case QuestSort::SortByDate:
342 return ascending ? a.properties.get_release_date() < b.properties.get_release_date()
343 : a.properties.get_release_date() > b.properties.get_release_date();
344 case QuestSort::SortByName:
345 default:
346 return ascending ? a.properties.get_title() < b.properties.get_title()
347 : a.properties.get_title() > b.properties.get_title();
348 }
349 });
350}
