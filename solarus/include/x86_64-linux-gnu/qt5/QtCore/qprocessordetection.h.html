49/*
50 This file uses preprocessor #defines to set various Q_PROCESSOR_* #defines
51 based on the following patterns:
52
53 Q_PROCESSOR_{FAMILY}
54 Q_PROCESSOR_{FAMILY}_{VARIANT}
55 Q_PROCESSOR_{FAMILY}_{REVISION}
56
57 The first is always defined. Defines for the various revisions/variants are
58 optional and usually dependent on how the compiler was invoked. Variants
59 that are a superset of another should have a define for the superset.
60
61 In addition to the procesor family, variants, and revisions, we also set
62 Q_BYTE_ORDER appropriately for the target processor. For bi-endian
63 processors, we try to auto-detect the byte order using the __BIG_ENDIAN__,
64 __LITTLE_ENDIAN__, or __BYTE_ORDER__ preprocessor macros.
65
66 Note: when adding support for new processors, be sure to update
67 config.tests/arch/arch.cpp to ensure that configure can detect the target
68 and host architectures.
69*/
70
71/* Machine byte-order, reuse preprocessor provided macros when available */
72#if defined(__ORDER_BIG_ENDIAN__)
73# define Q_BIG_ENDIAN __ORDER_BIG_ENDIAN__
74#else
75# define Q_BIG_ENDIAN 4321
76#endif
77#if defined(__ORDER_LITTLE_ENDIAN__)
78# define Q_LITTLE_ENDIAN __ORDER_LITTLE_ENDIAN__
79#else
80# define Q_LITTLE_ENDIAN 1234
81#endif

84 Alpha family, no revisions or variants
85
86 Alpha is bi-endian, use endianness auto-detection implemented below.

88// #elif defined(__alpha__) || defined(_M_ALPHA)
89// # define Q_PROCESSOR_ALPHA
90// Q_BYTE_ORDER not defined, use endianness auto-detection

92/*
93 ARM family, known revisions: V5, V6, V7, V8
94
95 ARM is bi-endian, detect using __ARMEL__ or __ARMEB__, falling back to
96 auto-detection implemented below.
97*/
98#if defined(__arm__) || defined(__TARGET_ARCH_ARM) || defined(_M_ARM) || defined(__arm64__)
99# define Q_PROCESSOR_ARM
100# if defined(__arm64__)
101# define Q_PROCESSOR_ARM_64
102# else
103# define Q_PROCESSOR_ARM_32
104# endif
105# if defined(__ARM64_ARCH_8__)
106# define Q_PROCESSOR_ARM_V8
107# define Q_PROCESSOR_ARM_V7
108# define Q_PROCESSOR_ARM_V6
109# define Q_PROCESSOR_ARM_V5
110# elif defined(__ARM_ARCH_7__) \
111 || defined(__ARM_ARCH_7A__) \
112 || defined(__ARM_ARCH_7R__) \
113 || defined(__ARM_ARCH_7M__) \
114 || defined(__ARM_ARCH_7S__) \
115 || defined(_ARM_ARCH_7) \
116 || (defined(__TARGET_ARCH_ARM) && __TARGET_ARCH_ARM-0 >= 7) \
117 || (defined(_M_ARM) && _M_ARM-0 >= 7)
118# define Q_PROCESSOR_ARM_V7
119# define Q_PROCESSOR_ARM_V6
120# define Q_PROCESSOR_ARM_V5
121# elif defined(__ARM_ARCH_6__) \
122 || defined(__ARM_ARCH_6J__) \
123 || defined(__ARM_ARCH_6T2__) \
124 || defined(__ARM_ARCH_6Z__) \
125 || defined(__ARM_ARCH_6K__) \
126 || defined(__ARM_ARCH_6ZK__) \
127 || defined(__ARM_ARCH_6M__) \
128 || (defined(__TARGET_ARCH_ARM) && __TARGET_ARCH_ARM-0 >= 6) \
129 || (defined(_M_ARM) && _M_ARM-0 >= 6)
130# define Q_PROCESSOR_ARM_V6
131# define Q_PROCESSOR_ARM_V5
132# elif defined(__ARM_ARCH_5TEJ__) \
133 || defined(__ARM_ARCH_5TE__) \
134 || (defined(__TARGET_ARCH_ARM) && __TARGET_ARCH_ARM-0 >= 5) \
135 || (defined(_M_ARM) && _M_ARM-0 >= 5)
136# define Q_PROCESSOR_ARM_V5
137# endif
138# if defined(__ARMEL__)
139# define Q_BYTE_ORDER Q_LITTLE_ENDIAN
140# elif defined(__ARMEB__)
141# define Q_BYTE_ORDER Q_BIG_ENDIAN
142# else
143// Q_BYTE_ORDER not defined, use endianness auto-detection
144#endif

164/*
165 X86 family, known variants: 32- and 64-bit
166
167 X86 is little-endian.
168*/
169#elif defined(__i386) || defined(__i386__) || defined(_M_IX86)
170# define Q_PROCESSOR_X86_32
171# define Q_BYTE_ORDER Q_LITTLE_ENDIAN
172# define Q_PROCESSOR_WORDSIZE 4
173
174/*
175 * We define Q_PROCESSOR_X86 == 6 for anything above a equivalent or better
176 * than a Pentium Pro (the processor whose architecture was called P6) or an
177 * Athlon.
178 *
179 * All processors since the Pentium III and the Athlon 4 have SSE support, so
180 * we use that to detect. That leaves the original Athlon, Pentium Pro and
181 * Pentium II.
182 */
183
184# if defined(_M_IX86)
185# define Q_PROCESSOR_X86 (_M_IX86/100)
186# elif defined(__i686__) || defined(__athlon__) || defined(__SSE__)
187# define Q_PROCESSOR_X86 6
188# elif defined(__i586__) || defined(__k6__)
189# define Q_PROCESSOR_X86 5
190# elif defined(__i486__)
191# define Q_PROCESSOR_X86 4
192# else
193# define Q_PROCESSOR_X86 3
194# endif
195
196#elif defined(__x86_64) || defined(__x86_64__) || defined(__amd64) || defined(_M_X64)
197# define Q_PROCESSOR_X86 6
198# define Q_PROCESSOR_X86_64
199# define Q_BYTE_ORDER Q_LITTLE_ENDIAN
200# define Q_PROCESSOR_WORDSIZE 8
201
202/*
203 Itanium (IA-64) family, no revisions or variants
204
205 Itanium is bi-endian, use endianness auto-detection implemented below.
206*/
207#elif defined(__ia64) || defined(__ia64__) || defined(_M_IA64)
208# define Q_PROCESSOR_IA64
209# define Q_PROCESSOR_WORDSIZE 8
210// Q_BYTE_ORDER not defined, use endianness auto-detection
211
212/*
213 MIPS family, known revisions: I, II, III, IV, 32, 64
214
215 MIPS is bi-endian, use endianness auto-detection implemented below.
216*/
217#elif defined(__mips) || defined(__mips__) || defined(_M_MRX000)
218# define Q_PROCESSOR_MIPS
219# if defined(_MIPS_ARCH_MIPS1) || (defined(__mips) && __mips - 0 >= 1)
220# define Q_PROCESSOR_MIPS_I
221# endif
222# if defined(_MIPS_ARCH_MIPS2) || (defined(__mips) && __mips - 0 >= 2)
223# define Q_PROCESSOR_MIPS_II
224# endif
225# if defined(_MIPS_ARCH_MIPS32) || defined(__mips32)
226# define Q_PROCESSOR_MIPS_32
227# endif
228# if defined(_MIPS_ARCH_MIPS3) || (defined(__mips) && __mips - 0 >= 3)
229# define Q_PROCESSOR_MIPS_III
230# endif
231# if defined(_MIPS_ARCH_MIPS4) || (defined(__mips) && __mips - 0 >= 4)
232# define Q_PROCESSOR_MIPS_IV
233# endif
234# if defined(_MIPS_ARCH_MIPS5) || (defined(__mips) && __mips - 0 >= 5)
235# define Q_PROCESSOR_MIPS_V
236# endif
237# if defined(_MIPS_ARCH_MIPS64) || defined(__mips64)
238# define Q_PROCESSOR_MIPS_64
239# endif
240# if defined(__MIPSEL__)
241# define Q_BYTE_ORDER Q_LITTLE_ENDIAN
242# elif defined(__MIPSEB__)
243# define Q_BYTE_ORDER Q_BIG_ENDIAN
244# else
245// Q_BYTE_ORDER not defined, use endianness auto-detection
246# endif

265# endif

273#elif defined(__s390__)

276# define Q_PROCESSOR_S390_X

292/*
293 SPARC family, optional revision: V9
294
295 SPARC is big-endian only prior to V9, while V9 is bi-endian with big-endian
296 as the default byte order. Assume all SPARC systems are big-endian.
297*/
298#elif defined(__sparc__)
299# define Q_PROCESSOR_SPARC
300# if defined(__sparc_v9__)
301# define Q_PROCESSOR_SPARC_V9
302# endif
303# if defined(__sparc64__)
304# define Q_PROCESSOR_SPARC_64
305# endif
306# define Q_BYTE_ORDER Q_BIG_ENDIAN

310/*
311 NOTE:
312 GCC 4.6 added __BYTE_ORDER__, __ORDER_BIG_ENDIAN__, __ORDER_LITTLE_ENDIAN__
313 and __ORDER_PDP_ENDIAN__ in SVN r165881. If you are using GCC 4.6 or newer,
314 this code will properly detect your target byte order; if you are not, and
315 the __LITTLE_ENDIAN__ or __BIG_ENDIAN__ macros are not defined, then this
316 code will fail to detect the target byte order.
317*/
318// Some processors support either endian format, try to detect which we are using.
319#if !defined(Q_BYTE_ORDER)
320# if defined(__BYTE_ORDER__) && (__BYTE_ORDER__ == Q_BIG_ENDIAN || __BYTE_ORDER__ == Q_LITTLE_ENDIAN)
321// Reuse __BYTE_ORDER__ as-is, since our Q_*_ENDIAN #defines match the preprocessor defaults
322# define Q_BYTE_ORDER __BYTE_ORDER__
323# elif defined(__BIG_ENDIAN__) || defined(_big_endian__) || defined(_BIG_ENDIAN)
324# define Q_BYTE_ORDER Q_BIG_ENDIAN
325# elif defined(__LITTLE_ENDIAN__) || defined(_little_endian__) || defined(_LITTLE_ENDIAN) \
326 || defined(_WIN32_WCE) || defined(WINAPI_FAMILY) // Windows CE is always little-endian according to MSDN.
327# define Q_BYTE_ORDER Q_LITTLE_ENDIAN
328# else
329# error "Unable to determine byte order!"
330# endif
331#endif
332
333/*
334 Define Q_PROCESSOR_WORDSIZE to be the size of the machine's word (usually,
335 the size of the register). On some architectures where a pointer could be
336 smaller than the register, the macro is defined above.
337
338 Falls back to QT_POINTER_SIZE if not set explicitly for the platform.
339*/
340#ifndef Q_PROCESSOR_WORDSIZE
341# ifdef __SIZEOF_POINTER__
342 /* GCC & friends define this */
343# define Q_PROCESSOR_WORDSIZE __SIZEOF_POINTER__
344# elif defined(_LP64) || defined(__LP64__) || defined(WIN64) || defined(_WIN64)
345# define Q_PROCESSOR_WORDSIZE 8
346# else
347# define Q_PROCESSOR_WORDSIZE QT_POINTER_SIZE
348# endif
349#endif
