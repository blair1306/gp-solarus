107/*****************************************************************************
108 QSize inline functions

110
111Q_DECL_CONSTEXPR inline QSize::QSize() : wd(-1), ht(-1) {}
112
113Q_DECL_CONSTEXPR inline QSize::QSize(int w, int h) : wd(w), ht(h) {}
114
115Q_DECL_CONSTEXPR inline bool QSize::isNull() const
116{ return wd==0 && ht==0; }
117
118Q_DECL_CONSTEXPR inline bool QSize::isEmpty() const
119{ return wd<1 || ht<1; }
120
121Q_DECL_CONSTEXPR inline bool QSize::isValid() const
122{ return wd>=0 && ht>=0; }
123
124Q_DECL_CONSTEXPR inline int QSize::width() const
125{ return wd; }
126
127Q_DECL_CONSTEXPR inline int QSize::height() const
128{ return ht; }
129
130inline void QSize::setWidth(int w)
131{ wd = w; }
132
133inline void QSize::setHeight(int h)
134{ ht = h; }
135
136Q_DECL_CONSTEXPR inline QSize QSize::transposed() const
137{ return QSize(ht, wd); }
138
139inline void QSize::scale(int w, int h, Qt::AspectRatioMode mode)
140{ scale(QSize(w, h), mode); }
141
142inline void QSize::scale(const QSize &s, Qt::AspectRatioMode mode)
143{ *this = scaled(s, mode); }
144
145inline QSize QSize::scaled(int w, int h, Qt::AspectRatioMode mode) const
146{ return scaled(QSize(w, h), mode); }
147
148inline int &QSize::rwidth()
149{ return wd; }
150
151inline int &QSize::rheight()
152{ return ht; }
153
154inline QSize &QSize::operator+=(const QSize &s)
155{ wd+=s.wd; ht+=s.ht; return *this; }
156
157inline QSize &QSize::operator-=(const QSize &s)
158{ wd-=s.wd; ht-=s.ht; return *this; }
159
160inline QSize &QSize::operator*=(qreal c)
161{ wd = qRound(wd*c); ht = qRound(ht*c); return *this; }
162
163Q_DECL_CONSTEXPR inline bool operator==(const QSize &s1, const QSize &s2)
164{ return s1.wd == s2.wd && s1.ht == s2.ht; }
165
166Q_DECL_CONSTEXPR inline bool operator!=(const QSize &s1, const QSize &s2)
167{ return s1.wd != s2.wd || s1.ht != s2.ht; }
168
169Q_DECL_CONSTEXPR inline const QSize operator+(const QSize & s1, const QSize & s2)
170{ return QSize(s1.wd+s2.wd, s1.ht+s2.ht); }
171
172Q_DECL_CONSTEXPR inline const QSize operator-(const QSize &s1, const QSize &s2)
173{ return QSize(s1.wd-s2.wd, s1.ht-s2.ht); }
174
175Q_DECL_CONSTEXPR inline const QSize operator*(const QSize &s, qreal c)
176{ return QSize(qRound(s.wd*c), qRound(s.ht*c)); }
177
178Q_DECL_CONSTEXPR inline const QSize operator*(qreal c, const QSize &s)
179{ return QSize(qRound(s.wd*c), qRound(s.ht*c)); }
180
181inline QSize &QSize::operator/=(qreal c)
182{
183 Q_ASSERT(!qFuzzyIsNull(c));
184 wd = qRound(wd/c); ht = qRound(ht/c);
185 return *this;
186}
187
188inline const QSize operator/(const QSize &s, qreal c)
189{
190 Q_ASSERT(!qFuzzyIsNull(c));
191 return QSize(qRound(s.wd/c), qRound(s.ht/c));
192}
193
194Q_DECL_CONSTEXPR inline QSize QSize::expandedTo(const QSize & otherSize) const
195{
196 return QSize(qMax(wd,otherSize.wd), qMax(ht,otherSize.ht));
197}
198
199Q_DECL_CONSTEXPR inline QSize QSize::boundedTo(const QSize & otherSize) const
200{
201 return QSize(qMin(wd,otherSize.wd), qMin(ht,otherSize.ht));
202}

278Q_DECL_CONSTEXPR inline QSizeF::QSizeF(qreal w, qreal h) : wd(w), ht(h) {}

280inline bool QSizeF::isNull() const
281{ return qIsNull(wd) && qIsNull(ht); }
282
283Q_DECL_CONSTEXPR inline bool QSizeF::isEmpty() const
284{ return wd <= 0. || ht <= 0.; }
285
286Q_DECL_CONSTEXPR inline bool QSizeF::isValid() const
287{ return wd >= 0. && ht >= 0.; }
288
289Q_DECL_CONSTEXPR inline qreal QSizeF::width() const
290{ return wd; }
291
292Q_DECL_CONSTEXPR inline qreal QSizeF::height() const

295inline void QSizeF::setWidth(qreal w)
296{ wd = w; }
297
298inline void QSizeF::setHeight(qreal h)
299{ ht = h; }
300
301Q_DECL_CONSTEXPR inline QSizeF QSizeF::transposed() const
302{ return QSizeF(ht, wd); }
303
304inline void QSizeF::scale(qreal w, qreal h, Qt::AspectRatioMode mode)
305{ scale(QSizeF(w, h), mode); }
306
307inline void QSizeF::scale(const QSizeF &s, Qt::AspectRatioMode mode)
308{ *this = scaled(s, mode); }
309
310inline QSizeF QSizeF::scaled(qreal w, qreal h, Qt::AspectRatioMode mode) const
311{ return scaled(QSizeF(w, h), mode); }
312
313inline qreal &QSizeF::rwidth()
314{ return wd; }
315
316inline qreal &QSizeF::rheight()
317{ return ht; }
318
319inline QSizeF &QSizeF::operator+=(const QSizeF &s)
320{ wd += s.wd; ht += s.ht; return *this; }
321
322inline QSizeF &QSizeF::operator-=(const QSizeF &s)
323{ wd -= s.wd; ht -= s.ht; return *this; }
324
325inline QSizeF &QSizeF::operator*=(qreal c)
326{ wd *= c; ht *= c; return *this; }
327
328Q_DECL_CONSTEXPR inline bool operator==(const QSizeF &s1, const QSizeF &s2)
329{ return qFuzzyCompare(s1.wd, s2.wd) && qFuzzyCompare(s1.ht, s2.ht); }
330
331Q_DECL_CONSTEXPR inline bool operator!=(const QSizeF &s1, const QSizeF &s2)
332{ return !qFuzzyCompare(s1.wd, s2.wd) || !qFuzzyCompare(s1.ht, s2.ht); }
333
334Q_DECL_CONSTEXPR inline const QSizeF operator+(const QSizeF & s1, const QSizeF & s2)
335{ return QSizeF(s1.wd+s2.wd, s1.ht+s2.ht); }
336
337Q_DECL_CONSTEXPR inline const QSizeF operator-(const QSizeF &s1, const QSizeF &s2)
338{ return QSizeF(s1.wd-s2.wd, s1.ht-s2.ht); }
339
340Q_DECL_CONSTEXPR inline const QSizeF operator*(const QSizeF &s, qreal c)
341{ return QSizeF(s.wd*c, s.ht*c); }
342
343Q_DECL_CONSTEXPR inline const QSizeF operator*(qreal c, const QSizeF &s)
344{ return QSizeF(s.wd*c, s.ht*c); }
345
346inline QSizeF &QSizeF::operator/=(qreal c)
347{
348 Q_ASSERT(!qFuzzyIsNull(c));
349 wd = wd/c; ht = ht/c;
350 return *this;
351}
352
353inline const QSizeF operator/(const QSizeF &s, qreal c)
354{
355 Q_ASSERT(!qFuzzyIsNull(c));
356 return QSizeF(s.wd/c, s.ht/c);
357}
358
359Q_DECL_CONSTEXPR inline QSizeF QSizeF::expandedTo(const QSizeF & otherSize) const
360{
361 return QSizeF(qMax(wd,otherSize.wd), qMax(ht,otherSize.ht));
362}
363
364Q_DECL_CONSTEXPR inline QSizeF QSizeF::boundedTo(const QSizeF & otherSize) const
365{
366 return QSizeF(qMin(wd,otherSize.wd), qMin(ht,otherSize.ht));
367}
368
369Q_DECL_CONSTEXPR inline QSize QSizeF::toSize() const
370{
371 return QSize(qRound(wd), qRound(ht));
372}
