17#ifndef SOLARUS_ENTITY_STATE_H
18#define SOLARUS_ENTITY_STATE_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/CarriedObject.h"
22#include "solarus/entities/Hero.h"
23#include <cstdint>
24#include <memory>
25#include <string>
26
27namespace Solarus {
28
29/**
30 * \brief Abstract base class for a state of an entity.
31 *
32 * This class contains all entity-related functions that depend on its state
33 * and provide a default implementation for them.
34 * Most of them are almost empty here because they depend on the state.
35 * Redefine for each state the functions that you need to implement or change.
36 */
37class Entity::State {
38
39 public:
40
41 // creation and destruction
42 virtual ~State();
43 const std::string& get_name() const;
44 virtual void start(const State* previous_state);
45 virtual void stop(const State* next_state);
46
47 // game loop
48 virtual void update();
49 virtual void draw_on_map();
50 bool is_suspended() const;
51 virtual void set_suspended(bool suspended);
52 uint32_t get_when_suspended() const;
53 void notify_command_pressed(GameCommand command);
54 void notify_command_released(GameCommand command);
55 virtual void notify_action_command_pressed();
56 virtual void notify_action_command_released();
57 virtual void notify_attack_command_pressed();
58 virtual void notify_attack_command_released();
59 virtual void notify_direction_command_pressed(int direction4);
60 virtual void notify_direction_command_released(int direction4);
61 virtual void notify_item_command_pressed(int slot);
62 virtual void notify_item_command_released(int slot);
63
64 // game
65 virtual void set_map(Map& map);
66 virtual bool can_start_gameover_sequence() const;
67
68 // sprites
69 virtual bool is_visible() const;
70 virtual bool is_direction_locked() const;
71
72 // movement
73 virtual bool can_control_movement() const;
74 virtual int get_wanted_movement_direction8() const;
75 virtual void notify_walking_speed_changed();
76 virtual void notify_layer_changed();
77 virtual void notify_movement_changed();
78 virtual void notify_movement_finished();
79 virtual void notify_obstacle_reached();
80 virtual void notify_position_changed();
81
82 // ground
83 virtual bool can_avoid_deep_water() const;
84 virtual bool can_avoid_hole() const;
85 virtual bool can_avoid_ice() const;
86 virtual bool can_avoid_lava() const;
87 virtual bool can_avoid_prickle() const;
88 virtual bool is_touching_ground() const;
89 virtual bool can_come_from_bad_ground() const;
90 virtual void notify_ground_changed();
91
92 // obstacles and collisions
93 virtual bool are_collisions_ignored() const;
94 virtual bool is_shallow_water_obstacle() const;
95 virtual bool is_deep_water_obstacle() const;
96 virtual bool is_hole_obstacle() const;
97 virtual bool is_lava_obstacle() const;
98 virtual bool is_prickle_obstacle() const;
99 virtual bool is_ladder_obstacle() const;
100 virtual bool is_teletransporter_obstacle(const Teletransporter& teletransporter) const;
101 virtual bool can_avoid_teletransporter() const;
102 virtual bool is_teletransporter_delayed() const;
103 virtual bool is_stream_obstacle(const Stream& stream) const;
104 virtual bool can_avoid_stream(const Stream& stream) const;
105 virtual bool can_persist_on_stream(const Stream& stream) const;
106 virtual bool is_stairs_obstacle(const Stairs& stairs) const;
107 virtual bool is_sensor_obstacle(const Sensor& sensor) const;
108 virtual bool is_jumper_obstacle(const Jumper& jumper, const Rectangle& candidate_position) const;
109 virtual bool is_separator_obstacle(const Separator& separator) const;
110 virtual bool can_avoid_sensor() const;
111 virtual bool can_avoid_explosion() const;
112 virtual bool can_avoid_switch() const;
113
114 // enemies
115 virtual void notify_attacked_enemy(
116 EnemyAttack attack,
117 Enemy& victim,
118 const Sprite* victim_sprite,
119 EnemyReaction::Reaction& result,
120 bool killed);
121 virtual int get_sword_damage_factor() const;
122 virtual bool can_be_hurt(Entity* attacker) const;
123
124 // state specific
125 virtual bool is_free() const;
126 virtual bool is_using_item() const;
127 virtual EquipmentItemUsage& get_item_being_used();
128 virtual bool is_brandishing_treasure() const;
129 virtual bool is_grabbing_or_pulling() const;
130 virtual bool is_moving_grabbed_entity() const;
131 virtual void notify_grabbed_entity_collision();
132 virtual bool is_cutting_with_sword(Entity& entity);
133 virtual bool can_start_sword() const;
134 virtual bool can_pick_treasure(EquipmentItem& item) const;
135 virtual bool can_use_shield() const;
136 virtual bool can_start_item(EquipmentItem& item) const;
137 virtual bool can_take_stairs() const;
138 virtual bool can_sword_hit_crystal() const;
139 virtual bool can_take_jumper() const;
140 virtual void notify_jumper_activated(Jumper& jumper);
141 bool is_carrying_item() const;
142 virtual std::shared_ptr<CarriedObject> get_carried_object() const;
143 virtual CarriedObject::Behavior get_previous_carried_object_behavior() const;
144
145 protected:
146
147 State(Entity& entity, const std::string& state_name);
148
149 bool is_current_state() const;
150 bool is_stopping() const;
151
152 // access to various game objects
153 LuaContext& get_lua_context();
154 Entities& get_entities();
155 Game& get_game();
156 const Game& get_game() const;
157 Map& get_map();
158 Equipment& get_equipment();
159 const Equipment& get_equipment() const;
160 CommandsEffects& get_commands_effects();
161 GameCommands& get_commands();
162 const GameCommands& get_commands() const;
163 virtual Entity& get_entity();
164 virtual const Entity& get_entity() const;
165
166 template<typename T>
167 T& get_entity();
168 template<typename T>
169 const T& get_entity() const;
170
171 private:
172
173 Entity& entity; /**< The entity controlled by this state. */
174 bool suspended; /**< Whether this state is suspended. */
175 uint32_t when_suspended; /**< When this state was suspended. */
176 Map* map; /**< The current map (it may change during this state). */
177 const std::string name; /**< Name describing this state. */
178 bool stopping; /**< Indicates that this state is being stopped. */
179
180};
181
182/**
183 * \brief Returns the entity of this state cast to an entity type.
184 * \return The entity.
185 */
186template<typename T>
187T& Entity::State::get_entity() {
188 return static_cast<T&>(get_entity());
189}
190
191/**
192 * \brief Returns the entity of this state cast to an entity type.
193 * \return The entity.
194 */
195template<typename T>
196const T& Entity::State::get_entity() const {
197 return static_cast<const T&>(get_entity());
198}
199
200}
201
202#endif
