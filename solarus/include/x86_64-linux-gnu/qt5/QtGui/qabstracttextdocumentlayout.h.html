42#ifndef QABSTRACTTEXTDOCUMENTLAYOUT_H
43#define QABSTRACTTEXTDOCUMENTLAYOUT_H
44
45#include <QtCore/qobject.h>
46#include <QtGui/qtextlayout.h>
47#include <QtGui/qtextdocument.h>
48#include <QtGui/qtextcursor.h>
49#include <QtGui/qpalette.h>
50
51QT_BEGIN_NAMESPACE
52
53
54class QAbstractTextDocumentLayoutPrivate;
55class QTextBlock;
56class QTextObjectInterface;
57class QTextFrame;
58
59class Q_GUI_EXPORT QAbstractTextDocumentLayout : public QObject
60{
61 Q_OBJECT
62 Q_DECLARE_PRIVATE(QAbstractTextDocumentLayout)
63
64public:
65 explicit QAbstractTextDocumentLayout(QTextDocument *doc);
66 ~QAbstractTextDocumentLayout();
67
68 struct Selection
69 {
70 QTextCursor cursor;
71 QTextCharFormat format;
72 };
73
74 struct PaintContext
75 {
76 PaintContext()
77 : cursorPosition(-1)
78 {}
79 int cursorPosition;
80 QPalette palette;
81 QRectF clip;
82 QVector<Selection> selections;
83 };
84
85 virtual void draw(QPainter *painter, const PaintContext &context) = 0;
86 virtual int hitTest(const QPointF &point, Qt::HitTestAccuracy accuracy) const = 0;
87 QString anchorAt(const QPointF& pos) const;
88
89 virtual int pageCount() const = 0;
90 virtual QSizeF documentSize() const = 0;
91
92 virtual QRectF frameBoundingRect(QTextFrame *frame) const = 0;
93 virtual QRectF blockBoundingRect(const QTextBlock &block) const = 0;
94
95 void setPaintDevice(QPaintDevice *device);
96 QPaintDevice *paintDevice() const;
97
98 QTextDocument *document() const;
99
100 void registerHandler(int objectType, QObject *component);
101 void unregisterHandler(int objectType, QObject *component = 0);
102 QTextObjectInterface *handlerForObject(int objectType) const;
103
104Q_SIGNALS:
105 void update(const QRectF & = QRectF(0., 0., 1000000000., 1000000000.));
106 void updateBlock(const QTextBlock &block);
107 void documentSizeChanged(const QSizeF &newSize);
108 void pageCountChanged(int newPages);
109
110protected:
111 QAbstractTextDocumentLayout(QAbstractTextDocumentLayoutPrivate &, QTextDocument *);
112
113 virtual void documentChanged(int from, int charsRemoved, int charsAdded) = 0;
114
115 virtual void resizeInlineObject(QTextInlineObject item, int posInDocument, const QTextFormat &format);
116 virtual void positionInlineObject(QTextInlineObject item, int posInDocument, const QTextFormat &format);
117 virtual void drawInlineObject(QPainter *painter, const QRectF &rect, QTextInlineObject object, int posInDocument, const QTextFormat &format);
118
119 int formatIndex(int pos);
120 QTextCharFormat format(int pos);
121
122private:
123 friend class QWidgetTextControl;
124 friend class QTextDocument;
125 friend class QTextDocumentPrivate;
126 friend class QTextEngine;
127 friend class QTextLayout;
128 friend class QTextLine;
129 Q_PRIVATE_SLOT(d_func(), void _q_handlerDestroyed(QObject *obj))
130 Q_PRIVATE_SLOT(d_func(), int _q_dynamicPageCountSlot())
131 Q_PRIVATE_SLOT(d_func(), QSizeF _q_dynamicDocumentSizeSlot())
132};
133
134class Q_GUI_EXPORT QTextObjectInterface
135{
136public:
137 virtual ~QTextObjectInterface() {}
138 virtual QSizeF intrinsicSize(QTextDocument *doc, int posInDocument, const QTextFormat &format) = 0;
139 virtual void drawObject(QPainter *painter, const QRectF &rect, QTextDocument *doc, int posInDocument, const QTextFormat &format) = 0;
140};
141
142Q_DECLARE_INTERFACE(QTextObjectInterface, "org.qt-project.Qt.QTextObjectInterface")
143
144QT_END_NAMESPACE
