65 };
66

69 { "get_horizontal_alignment", text_surface_api_get_horizontal_alignment },
70 { "set_horizontal_alignment", text_surface_api_set_horizontal_alignment },
71 { "get_vertical_alignment", text_surface_api_get_vertical_alignment },
72 { "set_vertical_alignment", text_surface_api_set_vertical_alignment },
73 { "get_font", text_surface_api_get_font },
74 { "set_font", text_surface_api_set_font },
75 { "get_rendering_mode", text_surface_api_get_rendering_mode },
76 { "set_rendering_mode", text_surface_api_set_rendering_mode },
77 { "get_color", text_surface_api_get_color },
78 { "set_color", text_surface_api_set_color },
79 { "get_font_size", text_surface_api_get_font_size },
80 { "set_font_size", text_surface_api_set_font_size },
81 { "get_text", text_surface_api_get_text },
82 { "set_text", text_surface_api_set_text },
83 { "set_text_key", text_surface_api_set_text_key },
84 { "get_size", text_surface_api_get_size },
85 { "draw", drawable_api_draw },
86 { "draw_region", drawable_api_draw_region },
87 { "get_blend_mode", drawable_api_get_blend_mode },
88 { "set_blend_mode", drawable_api_set_blend_mode },
89 { "fade_in", drawable_api_fade_in },
90 { "fade_out", drawable_api_fade_out },
91 { "get_xy", drawable_api_get_xy },
92 { "set_xy", drawable_api_set_xy },
93 { "get_movement", drawable_api_get_movement },
94 { "stop_movement", drawable_api_stop_movement },
95 { nullptr, nullptr }
96 };
97
98 static const luaL_Reg metamethods[] = {
99 { "__gc", drawable_meta_gc },
100 { nullptr, nullptr }
101 };
102
103 register_type(text_surface_module_name, functions, methods, metamethods);
104}

106/**
107 * \brief Returns whether a value is a userdata of type text surface.
108 * \param l A Lua context.
109 * \param index An index in the stack.
110 * \return true if the value at this index is a text surface.
111 */
112bool LuaContext::is_text_surface(lua_State* l, int index) {
113 return is_userdata(l, index, text_surface_module_name);
114}
115
116/**
117 * \brief Checks that the userdata at the specified index of the stack is a
118 * text surface and returns it.
119 * \param l a Lua context
120 * \param index an index in the stack
121 * \return the text surface
122 */
123std::shared_ptr<TextSurface> LuaContext::check_text_surface(lua_State* l, int index) {
124 return std::static_pointer_cast<TextSurface>(check_userdata(
125 l, index, text_surface_module_name
126 ));
127}

138/**
139 * \brief Implementation of sol.text_surface.create().
140 * \param l The Lua context that is calling this function.
141 * \return Number of values to return to Lua.
142 */
143int LuaContext::text_surface_api_create(lua_State* l) {
144
145 return LuaTools::exception_boundary_handle(l, [&] {
146 std::shared_ptr<TextSurface> text_surface =
147 std::make_shared<TextSurface>(0, 0);
148
149 if (lua_gettop(l) > 0) {
150 LuaTools::check_type(l, 1, LUA_TTABLE);
151
152 const std::string& font_id = LuaTools::opt_string_field(
153 l, 1, "font", FontResource::get_default_font_id()
154 );
155 RenderingMode rendering_mode =
156 LuaTools::opt_enum_field<RenderingMode>(
157 l, 1, "rendering_mode", rendering_mode_names, RenderingMode::SOLID
158 );
159 HorizontalAlignment horizontal_alignment =
160 LuaTools::opt_enum_field<HorizontalAlignment>(
161 l, 1, "horizontal_alignment", horizontal_alignment_names, HorizontalAlignment::LEFT
162 );
163 VerticalAlignment vertical_alignment =
164 LuaTools::opt_enum_field<VerticalAlignment>(
165 l, 1, "vertical_alignment", vertical_alignment_names, VerticalAlignment::MIDDLE
166 );
167 const Color& color = LuaTools::opt_color_field(l, 1, "color", Color::white);
168 int font_size = LuaTools::opt_int_field(l, 1, "font_size", TextSurface::default_font_size);
169 const std::string& text = LuaTools::opt_string_field(l, 1, "text", "");
170 const std::string& text_key = LuaTools::opt_string_field(l, 1, "text_key", "");
171
172 if (!FontResource::exists(font_id)) {
173 LuaTools::error(l, std::string("No such font: '") + font_id + "'");
174 }
175 text_surface->set_font(font_id);
176 text_surface->set_rendering_mode(rendering_mode);
177 text_surface->set_horizontal_alignment(horizontal_alignment);
178 text_surface->set_vertical_alignment(vertical_alignment);
179 text_surface->set_text_color(color);
180 text_surface->set_font_size(font_size);
181
182 if (!text.empty()) {
183 text_surface->set_text(text);
184 }
185 else if (!text_key.empty()) {
186 if (!CurrentQuest::string_exists(text_key)) {
187 LuaTools::error(l, std::string("No value with key '") + text_key
188 + "' in strings.dat for language '"
189 + CurrentQuest::get_language() + "'"
190 );
191 }
192 text_surface->set_text(CurrentQuest::get_string(text_key));
193 }
194 }
195 get_lua_context(l).add_drawable(text_surface);
196
197 push_text_surface(l, *text_surface);
198 return 1;
199 });
200}

202/**
203 * \brief Implementation of text_surface:get_horizontal_alignment().
204 * \param l the Lua context that is calling this function
205 * \return number of values to return to Lua
206 */
207int LuaContext::text_surface_api_get_horizontal_alignment(lua_State* l) {
208
209 return LuaTools::exception_boundary_handle(l, [&] {
210 const TextSurface& text_surface = *check_text_surface(l, 1);
211
212 HorizontalAlignment alignment = text_surface.get_horizontal_alignment();
213
214 push_string(l, horizontal_alignment_names.find(alignment)->second);
215 return 1;
216 });
217}
