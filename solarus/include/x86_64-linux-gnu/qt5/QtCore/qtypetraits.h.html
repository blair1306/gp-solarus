130// integral_constant, defined in tr1, is a wrapper for an integer
131// value. We don't really need this generality; we could get away
132// with hardcoding the integer type to bool. We use the fully
133// general integer_constant for compatibility with tr1.
134
135template<class T, T v>
136struct integral_constant {
137 static const T value = v;
138 typedef T value_type;
139 typedef integral_constant<T, v> type;
140};
141
142template <class T, T v> const T integral_constant<T, v>::value;
143
144
145// Abbreviations: true_type and false_type are structs that represent boolean
146// true and false values. Also define the boost::mpl versions of those names,
147// true_ and false_.
148typedef integral_constant<bool, true> true_type;
149typedef integral_constant<bool, false> false_type;
150typedef true_type true_;
151typedef false_type false_;
152
153// if_ is a templatized conditional statement.
154// if_<cond, A, B> is a compile time evaluation of cond.
155// if_<>::type contains A if cond is true, B otherwise.
156template<bool cond, typename A, typename B>
157struct if_{
158 typedef A type;
159};
160
161template<typename A, typename B>
162struct if_<false, A, B> {
163 typedef B type;
164};

167// type_equals_ is a template type comparator, similar to Loki IsSameType.
168// type_equals_<A, B>::value is true iff "A" is the same type as "B".
169//
170// New code should prefer base::is_same, defined in base/type_traits.h.
171// It is functionally identical, but is_same is the standard spelling.
172template<typename A, typename B>
173struct type_equals_ : public false_ {
174};
175
176template<typename A>
177struct type_equals_<A, A> : public true_ {
178};
179
180// and_ is a template && operator.
181// and_<A, B>::value evaluates "A::value && B::value".
182template<typename A, typename B>
183struct and_ : public integral_constant<bool, (A::value && B::value)> {
184};
185
186// or_ is a template || operator.
187// or_<A, B>::value evaluates "A::value || B::value".
188template<typename A, typename B>
189struct or_ : public integral_constant<bool, (A::value || B::value)> {
190};

294// Specified by TR1 [4.5.1] primary type categories.
295
296// Implementation note:
297//
298// Each type is either void, integral, floating point, array, pointer,
299// reference, member object pointer, member function pointer, enum,
300// union or class. Out of these, only integral, floating point, reference,
301// class and enum types are potentially convertible to int. Therefore,
302// if a type is not a reference, integral, floating point or class and
303// is convertible to int, it's a enum. Adding cv-qualification to a type
304// does not change whether it's an enum.
305//
306// Is-convertible-to-int check is done only if all other checks pass,
307// because it can't be used with some types (e.g. void or classes with
308// inaccessible conversion operators).
309template <class T> struct is_enum
310 : internal::is_enum_impl<
311 is_same<T, void>::value ||
312 is_integral<T>::value ||
313 is_floating_point<T>::value ||
314 is_reference<T>::value ||
315 internal::is_class_or_union<T>::value,
316 T> { };
317
318template <class T> struct is_enum<const T> : is_enum<T> { };
319template <class T> struct is_enum<volatile T> : is_enum<T> { };
320template <class T> struct is_enum<const volatile T> : is_enum<T> { };
321
322#endif

330// We will assume it's false except for arithmetic types, enumerations,
331// pointers and cv-qualified versions thereof. Note that std::pair<T,U>
332// is not a POD even if T and U are PODs.
333template <class T> struct is_pod
334 : integral_constant<bool, (is_integral<T>::value ||
335 is_floating_point<T>::value ||

338 is_enum<T>::value ||

435// This class is an implementation detail for is_convertible, and you
436// don't need to know how it works to use is_convertible. For those
437// who care: we declare two different functions, one whose argument is
438// of type To and one with a variadic argument list. We give them
439// return types of different size, so we can use sizeof to trick the
440// compiler into telling us which function it would have chosen if we
441// had called it with an argument of type From. See Alexandrescu's
442// _Modern C++ Design_ for more details on this sort of trick.
443
444template <typename From, typename To>
445struct ConvertHelper {
446 static small_ Test(To);
447 static big_ Test(...);
448 static From Create();
449};
450} // namespace internal
451
452// Inherits from true_type if From is convertible to To, false_type otherwise.
453template <typename From, typename To>
454struct is_convertible
455 : integral_constant<bool,
456 sizeof(internal::ConvertHelper<From, To>::Test(
457 internal::ConvertHelper<From, To>::Create()))
458 == sizeof(small_)> {
459};
460#endif
