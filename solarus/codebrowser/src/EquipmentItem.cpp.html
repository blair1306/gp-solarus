382/**
383 * \brief Returns the possession state of this item.
384 *
385 * The item must be saved.
386 *
387 * \return The possessed variant of this item.
388 */
389int EquipmentItem::get_variant() const {
390
391 Debug::check_assertion(is_saved(),
392 std::string("The item '") + get_name() + "' is not saved");
393
394 return get_savegame().get_integer(get_savegame_variable());
395}
396
397/**
398 * \brief Sets the possession state of this item.
399 *
400 * This function only makes sense if the item is saved.
401 *
402 * \param variant The possessed variant of this item.
403 */
404void EquipmentItem::set_variant(int variant) {
405
406 Debug::check_assertion(is_saved(),
407 std::string("The item '") + get_name() + "' is not saved");
408
409 // Set the possession state in the savegame.
410 get_savegame().set_integer(get_savegame_variable(), variant);
411
412 // If we are removing the item, unassign it.
413 if (variant == 0) {
414 int slot = equipment.get_item_slot(*this);
415 if (slot != 0) {
416 equipment.set_item_assigned(slot, nullptr);
417 }
418 }
419
420 // Notify the script.
421 notify_variant_changed(variant);
422}

424/**
425 * \brief Returns the current value of the amount associated to this item.
426 * \return The player's current amount of this item.
427 */
428int EquipmentItem::get_amount() const {
429
430 Debug::check_assertion(has_amount(),
431 std::string("The item '") + get_name() + "' has no amount");
432
433 return get_savegame().get_integer(get_amount_savegame_variable());
434}
435
436/**
437 * \brief Sets the current value of the amount associated to this item.
438 * \param amount The player's new amount of this item.
439 */
440void EquipmentItem::set_amount(int amount) {
441
442 Debug::check_assertion(has_amount(),
443 std::string("The item '") + get_name() + "' has no amount");
444
445 amount = std::max(0, std::min(get_max_amount(), amount));
446 get_savegame().set_integer(get_amount_savegame_variable(), amount);
447
448 notify_amount_changed(amount);
449}
450
451/**
452 * \brief Returns the maximum value of the amount associated to this item.
453 * \return The maximum amount of this item.
454 */
455int EquipmentItem::get_max_amount() const {
456
457 Debug::check_assertion(has_amount(),
458 std::string("The item '") + get_name() + "' has no amount");
459
460 return max_amount;
461}
462
463/**
464 * \brief Sets the maximum value of the amount associated to this item.
465 * \param max_amount The maximum amount of this item.
466 */
467void EquipmentItem::set_max_amount(int max_amount) {
468
469 Debug::check_assertion(has_amount(),
470 std::string("The item '") + get_name() + "' has no amount");
471
472 this->max_amount = max_amount;
473
474 // If the max amount is reduced, make sure the current amount does not exceed
475 // the new maximum.
476 if (get_amount() > max_amount) {
477 set_amount(max_amount);
478 }
479}
