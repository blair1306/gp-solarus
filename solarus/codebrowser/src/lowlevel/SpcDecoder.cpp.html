17#include "solarus/lowlevel/SpcDecoder.h"
18#include "solarus/lowlevel/Debug.h"
19#include <string>
20
21namespace Solarus {
22
23/**
24 * \brief Creates an SPC decoder.
25 */
26SpcDecoder::SpcDecoder():
27 snes_spc_manager(spc_new()),
28 snes_spc_filter(spc_filter_new()) {
29}
30
31/**
32 * \brief Loads an SPC file from the memory.
33 * \param sound_data The memory area to read.
34 * \param sound_size Size of the memory area in bytes.
35 */
36void SpcDecoder::load(int16_t* sound_data, size_t sound_size) {
37
38 // Load the SPC data into the SPC library.
39 spc_load_spc(snes_spc_manager.get(), (short int*) sound_data, sound_size);
40 spc_clear_echo(snes_spc_manager.get());
41 spc_filter_clear(snes_spc_filter.get());
42}
43
44
45/**
46 * \brief Decodes a chunk of the previously loaded SPC data into PCM data.
47 * \param decoded_data pointer to where you want the decoded data to be written.
48 * \param nb_samples Number of samples to write.
49 */
50void SpcDecoder::decode(int16_t* decoded_data, int nb_samples) {
51
52 // Decode from the SPC data the specified number of PCM samples.
53
54 const char* err = spc_play(snes_spc_manager.get(), nb_samples, (short int*) decoded_data);
55 if (err != nullptr) {
56 Debug::die(std::string("Failed to decode SPC data: ") + err);
57 }
58 spc_filter_run(snes_spc_filter.get(), (short int*) decoded_data, nb_samples);
59}
