29/**
30 * \brief A map entity fully controlled by Lua.
31 *
32 * A custom entity may have a model, which is the name of a Lua script
33 * that controls it. This is useful to make several custom entities with the
34 * same behavior more easily.
35 */

172 /**
173 * \brief Stores whether a custom entity can be traversed by or can traverse
174 * other entities.
175 */
176 class TraversableInfo {
177
178 public:
179
180 TraversableInfo();
181 TraversableInfo(
182 LuaContext& lua_context,
183 bool traversable
184 );
185 TraversableInfo(
186 LuaContext& lua_context,
187 const ScopedLuaRef& traversable_test_ref
188 );
189
190 bool is_empty() const;
191 bool is_traversable(
192 CustomEntity& current_entity,
193 Entity& other_entity
194 ) const;
195
196 private:
197
198 LuaContext* lua_context; /**< The Lua world.
199 * nullptr means no info. */
200 ScopedLuaRef
201 traversable_test_ref; /**< Lua ref to a boolean function
202 * that decides, or LUA_REFNIL. */
203 bool traversable; /**< Traversable property (unused if
204 * there is a Lua function). */
205
206 };
207
208 /**
209 * \brief Stores a callback to be executed when the specified test
210 * detects a collision.
211 */
212 class CollisionInfo {
213
214 public:
215
216 CollisionInfo();
217 CollisionInfo(
218 LuaContext& lua_context,
219 CollisionMode built_in_test,
220 const ScopedLuaRef& callback_ref
221 );
222 CollisionInfo(
223 LuaContext& lua_context,
224 const ScopedLuaRef& custom_test_ref,
225 const ScopedLuaRef& callback_ref
226 );
227
228 CollisionMode get_built_in_test() const;
229 const ScopedLuaRef& get_custom_test_ref() const;
230 const ScopedLuaRef& get_callback_ref() const;
231
232 private:
233
234 LuaContext* lua_context; /**< The Lua world.
235 * nullptr means no info. */
236 CollisionMode built_in_test; /**< A built-in collision test
237 * or COLLISION_CUSTOM. */
238 ScopedLuaRef custom_test_ref; /**< Ref to a custom collision test
239 * or LUA_REFNIL. */
240 ScopedLuaRef callback_ref; /**< Ref to the function to called when
241 * a collision is detected. */
242
243 };
244

249 void notify_collision_from(Entity& other_entity, Sprite& other_sprite, Sprite& this_sprite);

251 void update_ground_observer();

254
255 // Obstacles.
256
257 TraversableInfo traversable_by_entities_general; /**< Whether entities can traverse me by default or nullptr. */
258 std::map<EntityType, TraversableInfo>
259 traversable_by_entities_type; /**< Whether entities of a type can traverse me. */
260
261 TraversableInfo can_traverse_entities_general; /**< Whether I can traverse entities by default or nullptr. */
262 std::map<EntityType, TraversableInfo>
263 can_traverse_entities_type; /**< Whether I can traverse entities of a type. */
264 std::map<Ground, bool> can_traverse_grounds; /**< Whether I can traverse each kind of ground. */
265
266 // Collisions.
267
268 std::vector<CollisionInfo>
269 collision_tests; /**< The collision tests to perform. */
270 std::vector<CollisionInfo>
271 successful_collision_tests; /**< Collision test that detected
272 * collisions other than
273 * COLLISION_SPRITE. */
274
275 bool ground_observer; /**< Whether this custom entity is a ground observer. */
276 Ground modified_ground; /**< The ground defined by this custom
277 * entity or GROUND_EMPTY. */
278};
