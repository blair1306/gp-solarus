42#ifndef QABSTRACTITEMDELEGATE_H
43#define QABSTRACTITEMDELEGATE_H
44
45#include <QtCore/qobject.h>
46#include <QtWidgets/qstyleoption.h>
47
48QT_BEGIN_NAMESPACE
49
50
51#ifndef QT_NO_ITEMVIEWS
52
53class QPainter;
54class QModelIndex;
55class QAbstractItemModel;
56class QAbstractItemView;
57class QHelpEvent;
58
59class Q_WIDGETS_EXPORT QAbstractItemDelegate : public QObject
60{
61 Q_OBJECT
62
63public:
64
65 enum EndEditHint {
66 NoHint,
67 EditNextItem,
68 EditPreviousItem,
69 SubmitModelCache,
70 RevertModelCache
71 };
72
73 explicit QAbstractItemDelegate(QObject *parent = 0);
74 virtual ~QAbstractItemDelegate();
75
76 // painting
77 virtual void paint(QPainter *painter,
78 const QStyleOptionViewItem &option,
79 const QModelIndex &index) const = 0;
80
81 virtual QSize sizeHint(const QStyleOptionViewItem &option,
82 const QModelIndex &index) const = 0;
83
84 // editing
85 virtual QWidget *createEditor(QWidget *parent,
86 const QStyleOptionViewItem &option,
87 const QModelIndex &index) const;
88
89 virtual void destroyEditor(QWidget *editor, const QModelIndex &index) const;
90
91 virtual void setEditorData(QWidget *editor, const QModelIndex &index) const;
92
93 virtual void setModelData(QWidget *editor,
94 QAbstractItemModel *model,
95 const QModelIndex &index) const;
96
97 virtual void updateEditorGeometry(QWidget *editor,
98 const QStyleOptionViewItem &option,
99 const QModelIndex &index) const;
100
101 // for non-widget editors
102 virtual bool editorEvent(QEvent *event,
103 QAbstractItemModel *model,
104 const QStyleOptionViewItem &option,
105 const QModelIndex &index);
106
107 static QString elidedText(const QFontMetrics &fontMetrics, int width,
108 Qt::TextElideMode mode, const QString &text);
109
110 virtual bool helpEvent(QHelpEvent *event,
111 QAbstractItemView *view,
112 const QStyleOptionViewItem &option,
113 const QModelIndex &index);
114
115 virtual QVector<int> paintingRoles() const;
116
117Q_SIGNALS:
118 void commitData(QWidget *editor);
119 void closeEditor(QWidget *editor, QAbstractItemDelegate::EndEditHint hint = NoHint);
120 void sizeHintChanged(const QModelIndex &);
121
122protected:
123 QAbstractItemDelegate(QObjectPrivate &, QObject *parent = 0);
124private:
125 Q_DISABLE_COPY(QAbstractItemDelegate)
126};
127
128#endif // QT_NO_ITEMVIEWS
129
130QT_END_NAMESPACE
131
132#endif // QABSTRACTITEMDELEGATE_H
