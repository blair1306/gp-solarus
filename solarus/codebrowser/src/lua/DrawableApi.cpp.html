33/**
34 * \brief Returns whether a value is a userdata of a type.
35 * \param l A Lua context.
36 * \param index An index in the stack.
37 * \return true if the value at this index is a drawable.
38 */
39bool LuaContext::is_drawable(lua_State* l, int index) {
40 return is_surface(l, index)
41 || is_text_surface(l, index)
42 || is_sprite(l, index);
43}
44
45/**
46 * \brief Check that the userdata at the specified index is a drawable
47 * object (surface, text surface of sprite) and returns it.
48 * \param l a Lua context
49 * \param index an index in the stack
50 * \return The drawable.
51 */
52DrawablePtr LuaContext::check_drawable(lua_State* l, int index) {
53
54 if (is_drawable(l, index)) {
55 const ExportableToLuaPtr& userdata = *(static_cast<ExportableToLuaPtr*>(
56 lua_touserdata(l, index)
57 ));
58 return std::static_pointer_cast<Drawable>(userdata);
59 }
60 else {
61 LuaTools::type_error(l, index, "drawable");
62 throw;
63 }
64}

66/**
67 * \brief Returns whether a drawable object was created by this script.
68 * \param drawable A drawable object.
69 * \return true if the drawable object was created by this script.
70 */
71bool LuaContext::has_drawable(const DrawablePtr& drawable) {
72
73 return drawables.find(drawable) != drawables.end();
74}
75
76/**
77 * \brief Registers a drawable object created by Lua.
78 * \param drawable a drawable object
79 */
80void LuaContext::add_drawable(const DrawablePtr& drawable) {
81
82 Debug::check_assertion(!has_drawable(drawable),
83 "This drawable object is already registered");
84
85 drawables.insert(drawable);
86}
87
88/**
89 * \brief Unregisters a drawable object created by Lua.
90 * \param drawable a drawable object
91 */
92void LuaContext::remove_drawable(const DrawablePtr& drawable) {
93
94 Debug::check_assertion(has_drawable(drawable),
95 "This drawable object was not created by Lua");
96
97 drawables_to_remove.insert(drawable);
98}
99
100/**
101 * \brief Cleanups all drawable objects registered.
102 */
103void LuaContext::destroy_drawables() {
104
105 drawables.clear();
106 drawables_to_remove.clear();
107}
108
109/**
110 * \brief Updates all drawable objects created by this script.
111 */
112void LuaContext::update_drawables() {
113
114 // Update all drawables.
115 for (const DrawablePtr& drawable: drawables) {
116 if (has_drawable(drawable)) {
117 drawable->update();
118 }
119 }
120
121 // Remove the ones that should be removed.
122 for (const DrawablePtr& drawable: drawables_to_remove) {
123 drawables.erase(drawable);
124 }
125 drawables_to_remove.clear();
126}

128/**
129 * \brief Implementation of drawable:draw().
130 * \param l the Lua context that is calling this function
131 * \return number of values to return to Lua
132 */
133int LuaContext::drawable_api_draw(lua_State* l) {
134
135 return LuaTools::exception_boundary_handle(l, [&] {
136 Drawable& drawable = *check_drawable(l, 1);
137 SurfacePtr dst_surface = check_surface(l, 2);
138 int x = LuaTools::opt_int(l, 3, 0);
139 int y = LuaTools::opt_int(l, 4, 0);
140 drawable.draw(dst_surface, x, y);
141
142 return 0;
143 });
144}

146/**
147 * \brief Implementation of drawable:draw_region().
148 * \param l the Lua context that is calling this function
149 * \return number of values to return to Lua
150 */
151int LuaContext::drawable_api_draw_region(lua_State* l) {
152
153 return LuaTools::exception_boundary_handle(l, [&] {
154 Drawable& drawable = *check_drawable(l, 1);
155 Rectangle region = {
156 LuaTools::check_int(l, 2),
157 LuaTools::check_int(l, 3),
158 LuaTools::check_int(l, 4),
159 LuaTools::check_int(l, 5)
160 };
161 SurfacePtr dst_surface = check_surface(l, 6);
162 Point dst_position = {
163 LuaTools::opt_int(l, 7, 0),
164 LuaTools::opt_int(l, 8, 0)
165 };
166 drawable.draw_region(region, dst_surface, dst_position);
167
168 return 0;
169 });
170}

207/**
208 * \brief Implementation of drawable:fade_in().
209 * \param l the Lua context that is calling this function
210 * \return number of values to return to Lua
211 */
212int LuaContext::drawable_api_fade_in(lua_State* l) {
213
214 return LuaTools::exception_boundary_handle(l, [&] {
215 uint32_t delay = 20;
216 ScopedLuaRef callback_ref;
217
218 Drawable& drawable = *check_drawable(l, 1);
219
220 if (lua_gettop(l) >= 2) {
221 // the second argument can be the delay or the callback
222 int index = 2;
223 if (lua_isnumber(l, index)) {
224 delay = lua_tonumber(l, index);
225 index++;
226 }
227 // the next argument (if any) is the callback
228 callback_ref = LuaTools::opt_function(l, index);
229 }
230
231 TransitionFade* transition(new TransitionFade(
232 Transition::Direction::OPENING,
233 drawable.get_transition_surface()
234 ));
235 transition->clear_color();
236 transition->set_delay(delay);
237 drawable.start_transition(
238 std::unique_ptr<Transition>(transition),
239 callback_ref
240 );
241
242 return 0;
243 });
244}

285/**
286 * \brief Implementation of drawable:get_xy().
287 * \param l The Lua context that is calling this function.
288 * \return Number of values to return to Lua.
289 */
290int LuaContext::drawable_api_get_xy(lua_State* l) {
291
292 return LuaTools::exception_boundary_handle(l, [&] {
293 Drawable& drawable = *check_drawable(l, 1);
294
295 lua_pushinteger(l, drawable.get_xy().x);
296 lua_pushinteger(l, drawable.get_xy().y);
297 return 2;
298 });
299}

319/**
320 * \brief Implementation of drawable:get_movement().
321 * \param l The Lua context that is calling this function.
322 * \return Number of values to return to Lua.
323 */
324int LuaContext::drawable_api_get_movement(lua_State* l) {
325
326 return LuaTools::exception_boundary_handle(l, [&] {
327 Drawable& drawable = *check_drawable(l, 1);
328 std::shared_ptr<Movement> movement = drawable.get_movement();
329 if (movement == nullptr) {
330 lua_pushnil(l);
331 }
332 else {
333 push_movement(l, *movement);
334 }
335
336 return 1;
337 });
338}

356/**
357 * \brief Finalizer of types sprite, surface and text surface.
358 * \param l The Lua context that is calling this function.
359 * \return Number of values to return to Lua.
360 */
361int LuaContext::drawable_meta_gc(lua_State* l) {
362
363 return LuaTools::exception_boundary_handle(l, [&] {
364 LuaContext& lua_context = get_lua_context(l);
365 DrawablePtr drawable = check_drawable(l, 1);
366
367 if (lua_context.has_drawable(drawable)) {
368 // This drawable was created from Lua.
369 lua_context.remove_drawable(drawable);
370 }
371 userdata_meta_gc(l);
372
373 return 0;
374 });
375}
