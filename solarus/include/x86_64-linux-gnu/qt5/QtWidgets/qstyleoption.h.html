666template <typename T>
667T qstyleoption_cast(const QStyleOption *opt)
668{
669 if (opt && opt->version >= static_cast<T>(0)->Version && (opt->type == static_cast<T>(0)->Type
670 || int(static_cast<T>(0)->Type) == QStyleOption::SO_Default
671 || (int(static_cast<T>(0)->Type) == QStyleOption::SO_Complex
672 && opt->type > QStyleOption::SO_Complex)))
673 return static_cast<T>(opt);
674 return 0;
675}
