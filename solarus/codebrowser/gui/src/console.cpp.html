27/**
28 * @brief Wraps a line of plain text in html color tags.
29 * @param line A plain text line.
30 * @param color The color to set, with "#rrggbb" syntax.
31 * @return The HTML colorized line.
32 */
33QString colorize(const QString& line, const QString& color) {
34
35 return QString("<span style=\"color: %1\">%2</span>").arg(color, line.toHtmlEscaped());
36}

38const QRegularExpression output_regexp("^\\[Solarus\\] \\[(\\d+)\\] (\\w*): (.+)$");
39const QRegularExpression output_command_result_begin_regexp("^====== Begin Lua command #(\\d+) ======$");
40const QRegularExpression output_command_result_end_regexp("^====== End Lua command #(\\d+): (\\w+) ======$");
41const QRegularExpression output_simplify_console_error_regexp("In Lua command: \\[string \".*\"\\]:\\d+: ");
42const QRegularExpression output_setting_video_mode_regexp("^Video mode: (\\w+)$");
43const QRegularExpression output_setting_fullscreen_regexp("^Fullscreen: (\\w+)$");
44

79/**
80 * @brief Connects this console to a quest runner.
81 * @param quest_runner The quest runner.
82 */
83void Console::set_quest_runner(QuestRunner& quest_runner) {
84
85 this->quest_runner = &quest_runner;
86
87 connect(ui.command_field, SIGNAL(returnPressed()),
88 this, SLOT(command_field_activated()));
89
90 connect(&quest_runner, SIGNAL(running()),
91 this, SLOT(quest_running()));
92 connect(&quest_runner, SIGNAL(finished()),
93 this, SLOT(quest_finished()));
94 connect(&quest_runner, SIGNAL(output_produced(QStringList)),
95 this, SLOT(quest_output_produced(QStringList)));
96
97}

130/**
131 * @brief Slot called when the user wants to execute a Lua instruction from the console.
132 */
133void Console::command_field_activated() {
134
135 if (quest_runner == nullptr) {
136 return;
137 }
138
139 if (!quest_runner->is_running()) {
140 return;
141 }
142
143 const QString& command = ui.command_field->text();
144
145 execute_command(command);
146 ui.command_field->command_executed(command);
147}

149/**
150 * @brief Executes some Lua code in the quest process and logs it into the console.
151 *
152 * Use the command_result_received() signal to be notified of the result if you
153 * need to.
154 *
155 * @param command The Lua code.
156 * @return The id of the command executed, or -1 if it could not be sent
157 * to the process.
158 */
159int Console::execute_command(const QString& command) {
160
161 if (quest_runner == nullptr) {
162 return -1;
163 }
164
165 if (!quest_runner->is_started()) {
166 return -1;
167 }
168
169 if (command.isEmpty()) {
170 return -1;
171 }
172
173 int command_id = quest_runner->execute_command(command);
174 pending_commands[command_id] = command;
175
176 return command_id;
177}

179/**
180 * @brief Parses a Solarus output line and handles it.
181 * @param line The output line.
182 */
183void Console::parse_output(const QString& line) {
184
185 if (line.isEmpty()) {
186 return;
187 }
188
189 QString log_level;
190 QString message = line;
191 QRegularExpressionMatch match_result = output_regexp.match(line);
192
193 if (match_result.hasMatch()) {
194 // 4 captures expected: full line, time, log level, message.
195 QStringList captures = match_result.capturedTexts();
196 if (captures.size() != 4) {
197 ui.log_view->appendHtml(line.toHtmlEscaped());
198 return;
199 }
200
201 log_level = captures[2];
202 message = captures[3];
203 }
204
205 if (!log_level.isEmpty() && message.isEmpty()) {
206 // Solarus produced an empty message just to flush its stdout.
207 return;
208 }
209
210 // Detect technical delimiters of commands output but don't show them.
211 if (detect_command_result(log_level, message)) {
212 return;
213 }
214
215 if (log_level.isEmpty()) {
216 // Not a line from Solarus, probably one from the quest.
217 ui.log_view->appendHtml(line.toHtmlEscaped());
218 return;
219 }
220
221 // Report system setting changes.
222 detect_setting_change(log_level, message);
223
224 if (log_level == "Error") {
225 // Clean specific error messages.
226 message.remove(output_simplify_console_error_regexp);
227 }
228
229 // Add color.
230 QString line_html = colorize_output(log_level, message);
231 if (line_html.isEmpty()) {
232 return;
233 }
234
235 ui.log_view->appendHtml(line_html);
236}

238/**
239 * @brief Detects output messages that are the result of a command that was
240 * send from the console.
241 * @param log_level The Solarus log level of the line.
242 * @param message The rest of the message.
243 * @return @c true if the message is a command result delimiter and was consumed.
244 */
245bool Console::detect_command_result(
246 const QString& log_level,
247 const QString& message) {
248
249 QRegularExpressionMatch match_result;
250
251 // Detect the beginning of a console command result.
252 match_result = output_command_result_begin_regexp.match(message);
253 if (log_level == "Info" && match_result.lastCapturedIndex() == 1) {
254 // Start of a command result.
255 if (output_command_id != -1) {
256 qWarning() << "Beginning of a command result inside another command result";
257 }
258
259 output_command_id = match_result.captured(1).toInt();
260 output_command_result = QString();
261
262 // Show the command in the log view.
263 // We show the command only when receiving its results,
264 // to make sure it is displayed just before its results.
265 QString command = pending_commands.take(output_command_id);
266 ui.log_view->appendHtml(QString("> %1").arg(command).toHtmlEscaped());
267
268 return true;
269 }
270
271 // Detect the end of a console command result.
272 match_result = output_command_result_end_regexp.match(message);
273 if (log_level == "Info" && match_result.lastCapturedIndex() == 2) {
274 // End of a command result.
275 if (output_command_id == -1) {
276 qWarning() << "End of a command result without beginning";
277 return false;
278 }
279 int id = match_result.captured(1).toInt();
280 bool success = (match_result.captured(2) == "success");
281 QString command = pending_commands.take(output_command_id);
282 QString result = output_command_result;
283
284 if (id != output_command_id) {
285 qWarning() << "Unmatched command delimiters";
286 }
287
288 pending_commands.remove(output_command_id);
289 output_command_id = -1;
290 output_command_result.clear();
291
292 emit command_result_received(id, command, success, result);
293
294 return true;
295 }
296
297 // Process the current result between delimiters.
298 if (output_command_id != -1) {
299 // We are inside the delimiters.
300 output_command_result += message;
301 return false; // Let the console colorize and show the text normally.
302 }
303
304 return false;
305}

307/**
308 * @brief Emits the signal system_setting_changed if an output message
309 * indicates that a setting has just changed.
310 * @param log_level The Solarus log level of the line.
311 * @param message The rest of the message.
312 */
313void Console::detect_setting_change(
314 const QString& log_level,
315 const QString& message) {
316
317 if (log_level != "Info") {
318 return;
319 }
320
321 QRegularExpressionMatch match_result;
322
323 match_result = output_setting_video_mode_regexp.match(message);
324 if (match_result.lastCapturedIndex() == 1) {
325 QVariant value = match_result.captured(1);
326 emit setting_changed_in_quest("quest_video_mode", value);
327 return;
328 }
329
330 match_result = output_setting_fullscreen_regexp.match(message);
331 if (match_result.lastCapturedIndex() == 1) {
332 QVariant value = (match_result.captured(1) == "yes");
333 emit setting_changed_in_quest("quest_fullscreen", value);
334 return;
335 }
336}

338/**
339 * @brief Returns a colorized version of a Solarus output line.
340 *
341 * Colors may be added.
342 *
343 * @param log_level The Solarus log level of the line.
344 * @param message The rest of the message.
345 * @return The HTML decorated line.
346 */
347QString Console::colorize_output(const QString& log_level, const QString& message) {
348
349 if (message.isEmpty()) {
350 return message;
351 }
352
353 QString decorated_line = log_level + ": " + message;
354
355 // Colorize warnings and errors.
356 if (log_level == "Debug") {
357 decorated_line = colorize(decorated_line, "#808080");
358 }
359 else if (log_level == "Info") {
360 decorated_line = colorize(decorated_line, "#0000ff");
361 }
362 else if (log_level == "Warning") {
363 decorated_line = colorize(decorated_line, "#b05000");
364 }
365 else if (log_level == "Error") {
366 decorated_line = colorize(decorated_line, "#ff0000");
367 }
368 else if (log_level == "Fatal") {
369 decorated_line = colorize(decorated_line, "#ff0000");
370 }
371 else {
372 // Unknown log level.
373 decorated_line = message;
374 }
375
376 return decorated_line;
377}

379/**
380 * @brief Returns Lua commands that set the given settings in the Solarus API.
381 *
382 * System settings like audio volume and the video mode are returned as
383 * Lua commands.
384 * This function allows to apply to a running quest process
385 * what the user chose in the GUI.
386 */
387QStringList Console::get_quest_lua_commands_from_settings() const {
388
389 Settings settings;
390 QStringList commands;
391 QVariant video_mode = settings.value("quest_video_mode");
392 if (video_mode.isValid()) {
393 commands << QString("sol.video.set_mode(\"%1\")").
394 arg(video_mode.toString());
395 }
396
397 QVariant fullscreen = settings.value("quest_fullscreen");
398 if (fullscreen.isValid()) {
399 commands << QString("sol.video.set_fullscreen(%1)").
400 arg(fullscreen.toBool() ? "true" : "false");
401 }
402
403 QVariant sound_volume = settings.value("quest_sound_volume");
404 if (sound_volume.isValid()) {
405 commands << QString("sol.audio.set_sound_volume(%1)").
406 arg(sound_volume.toInt());
407 }
408
409 QVariant music_volume = settings.value("quest_music_volume");
410 if (music_volume.isValid()) {
411 commands << QString("sol.audio.set_music_volume(%1)").
412 arg(music_volume.toInt());
413 }
414
415 QVariant language = settings.value("quest_language");
416 if (language.isValid()) {
417 commands << QString("sol.language.set_language(\"%1\")").
418 arg(language.toString());
419 }
420
421 return commands;
422}

424/**
425 * @brief Applies current settings of the GUI to the quest process.
426 *
427 * Settings are applied as Lua commands.
428 *
429 * @param settings The settings.
430 * @return @c true if settings commands could be sent to the process
431 * (even if they produce an error).
432 */
433bool Console::apply_settings() {
434
435 if (quest_runner == nullptr || !quest_runner->is_running()) {
436 return false;
437 }
438
439 const QStringList& commands = get_quest_lua_commands_from_settings();
440 bool success = true;
441 for (const QString& command : commands) {
442 success = execute_command(command) && success;
443 }
444 return success;
445}
