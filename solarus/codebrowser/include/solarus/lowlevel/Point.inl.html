18namespace Solarus {
19
20////////////////////////////////////////////////////////////
21// Point methods
22
23/**
24 * \brief Creates a point with coordinates <tt>(0,0)</tt>.
25 */
26constexpr Point::Point()
27 = default;
28
29/**
30 * \brief Creates a point with the specified coordinates.
31 * \param x The x coordinate.
32 * \param y The y coordinate.
33 */
34constexpr Point::Point(int x, int y):
35 x(x),
36 y(y) {
37}
38
39/**
40 * \brief Adds another point to this one.
41 * \param other Another point.
42 * \return The current point after modification.
43 */
44inline Point& Point::operator+=(const Point& other) {
45 x += other.x;
46 y += other.y;
47 return *this;
48}
49
50/**
51 * \brief Subtracts another point from this one.
52 * \param other Another point.
53 * \return The current point after modification.
54 */
55inline Point& Point::operator-=(const Point& other) {
56 x -= other.x;
57 y -= other.y;
58 return *this;
59}
60
61/**
62 * \brief Multiplies this point by a value.
63 * \param factor The multiplication factor.
64 * \return The current point after modification.
65 */
66inline Point& Point::operator*=(int factor) {
67 x *= factor;
68 y *= factor;
69 return *this;
70}
71
72/**
73 * \brief Multiplies this point by a value.
74 * \param divisor The divisor.
75 * \return The current point after modification.
76 */
77inline Point& Point::operator/=(int divisor) {
78 x /= divisor;
79 y /= divisor;
80 return *this;
81}
82
83////////////////////////////////////////////////////////////
84// Free functions
85
86/**
87 * \brief Returns whether two points are equal.
88 * \param lhs A point.
89 * \param rhs Another point.
90 * \return \c true if the points are equal.
91 */
92constexpr bool operator==(const Point& lhs, const Point& rhs) {
93 return lhs.x == rhs.x && lhs.y == rhs.y;
94}
95
96/**
97 * \brief Returns whether two points are different.
98 * \param lhs A point.
99 * \param rhs Another point.
100 * \return \c true if the points are not equal.
101 */
102constexpr bool operator!=(const Point& lhs, const Point& rhs) {
103 return !(lhs == rhs);
104}
105
106/**
107 * \brief Unary plus operator.
108 * \param point A point.
109 * \return A copy of the point.
110 */
111constexpr Point operator+(const Point& point) {
112 return point;
113}
114
115/**
116 * \brief Unary minus operator.
117 * \param point A point.
118 * \return A point with opposite coordinates.
119 */
120constexpr Point operator-(const Point& point) {
121 return { -point.x, -point.y };
122}
123
124/**
125 * \brief Adds two points.
126 * \param lhs A point.
127 * \param rhs Another point.
128 * \return A point with the sum of their coordinates.
129 */
130constexpr Point operator+(const Point& lhs, const Point& rhs) {
131 return { lhs.x + rhs.x, lhs.y + rhs.y };
132}
133
134/**
135 * \brief Subtracts two points.
136 * \param lhs A point.
137 * \param rhs The point to subtract.
138 * \return A point with the difference of their coordinates.
139 */
140constexpr Point operator-(const Point& lhs, const Point& rhs) {
141 return { lhs.x - rhs.x, lhs.y - rhs.y };
142}
143
144/**
145 * \brief Multiples a point by a number.
146 * \param point A point.
147 * \param factor The factor.
148 * \return A point with coordinates multiplied by the factor.
149 */
150constexpr Point operator*(const Point& point, int factor) {
151 return { point.x * factor, point.y * factor };
152}
153
154/**
155 * \brief Multiples a point by a number.
156 * \param factor The factor.
157 * \param point A point.
158 * \return A point with coordinates multiplied by the factor.
159 */
160constexpr Point operator*(int factor, const Point& point) {
161 return point * factor;
162}
163
164/**
165 * \brief Divides a point by a number.
166 * \param point A point.
167 * \param divisor The divisor.
168 * \return A point with coordinates divided by the divisor.
169 */
170constexpr Point operator/(const Point& point, int divisor) {
171 return { point.x / divisor, point.y / divisor };
172}
