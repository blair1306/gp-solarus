44/**
45 * \brief Creates a game.
46 * \param main_loop The Solarus root object.
47 * \param savegame The saved data of this game.
48 */
49Game::Game(MainLoop& main_loop, const std::shared_ptr<Savegame>& savegame):
50 main_loop(main_loop),
51 savegame(savegame),
52 pause_allowed(true),
53 paused(false),
54 dialog_box(*this),
55 showing_game_over(false),
56 suspended_by_script(false),
57 started(false),
58 restarting(false),
59 commands_effects(),
60 current_map(nullptr),
61 next_map(nullptr),
62 previous_map_surface(nullptr),
63 transition_style(Transition::Style::IMMEDIATE),
64 transition(nullptr),
65 crystal_state(false) {
66
67 // notify objects
68 savegame->set_game(this);
69
70 // initialize members
71 commands = std::unique_ptr<GameCommands>(new GameCommands(*this));
72 hero = std::make_shared<Hero>(get_equipment());
73 update_commands_effects();
74
75 // Maybe we are restarting after a game-over sequence.
76 if (get_equipment().get_life() <= 0) {
77 get_equipment().restore_all_life();
78 }
79
80 // Launch the starting map.
81 std::string starting_map_id = savegame->get_string(Savegame::KEY_STARTING_MAP);
82 std::string starting_destination_name = savegame->get_string(Savegame::KEY_STARTING_POINT);
83
84 bool valid_map_saved = false;
85 if (!starting_map_id.empty()) {
86
87 if (CurrentQuest::resource_exists(ResourceType::MAP, starting_map_id)) {
88 // We are okay: the savegame file refers to an existing map.
89 valid_map_saved = true;
90 }
91 else {
92 // The savegame refers to a map that no longer exists.
93 // Maybe the quest is in an intermediate development phase.
94 // Show an error and fallback to the default map.
95 Debug::error(std::string("The savegame refers to a non-existing map: '") + starting_map_id + "'");
96 }
97 }
98
99 if (!valid_map_saved) {
100 // When no valid starting map is set, use the first one declared in the
101 // resource list file.
102 const std::map<std::string, std::string>& maps =
103 CurrentQuest::get_resources(ResourceType::MAP);
104 if (maps.empty()) {
105 Debug::die("This quest has no map");
106 }
107 starting_map_id = maps.begin()->first;
108 starting_destination_name = ""; // Default destination.
109 }
110
111 set_current_map(starting_map_id, starting_destination_name, Transition::Style::FADE);
112}

114/**
115 * \brief Starts this game.
116 *
117 * Does nothing if the game is already started.
118 */
119void Game::start() {
120
121 if (started) {
122 return;
123 }
124
125 started = true;
126 get_savegame().notify_game_started();
127 get_lua_context().game_on_started(*this);
128}
129
130/**
131 * \brief Ends this game.
132 *
133 * Does nothing if the game is not started.
134 */
135void Game::stop() {
136
137 if (!started) {
138 return;
139 }
140
141 if (current_map != nullptr) {
142
143 if (hero->is_on_map()) {
144 // The hero was initialized.
145 hero->notify_being_removed();
146 }
147
148 if (current_map->is_started()) {
149 current_map->leave();
150 }
151 if (current_map->is_loaded()) {
152 current_map->unload();
153 }
154 }
155
156 get_lua_context().game_on_finished(*this);
157 get_savegame().notify_game_finished();
158 get_savegame().set_game(nullptr);
159
160 Music::stop_playing();
161
162 started = false;
163}

259/**
260 * \brief This function is called when a low-level input event occurs during the game.
261 * \param event the event to handle
262 * \return \c true if the event was handled and should stop being propagated.
263 */
264bool Game::notify_input(const InputEvent& event) {
265
266 if (current_map != nullptr && current_map->is_loaded()) {
267 bool handled = get_lua_context().game_on_input(*this, event);
268 if (!handled) {
269 handled = current_map->notify_input(event);
270 if (!handled) {
271 // Built-in behavior:
272 // the GameCommands object will transform the low-level input event into
273 // a high-level game command event (i.e. a command_pressed event or
274 // a command_released event).
275 commands->notify_input(event);
276 }
277 }
278 }
279 return true;
280}

282/**
283 * \brief This function is called when a game command is pressed.
284 * \param command A game command.
285 */
286void Game::notify_command_pressed(GameCommand command) {
287
288 // Is a built-in dialog box being shown?
289 if (is_dialog_enabled()) {
290 if (dialog_box.notify_command_pressed(command)) {
291 return;
292 }
293 }
294
295 // See if the game script handles the command.
296 if (get_lua_context().game_on_command_pressed(*this, command)) {
297 return;
298 }
299
300 // See if the map script handled the command.
301 if (get_lua_context().map_on_command_pressed(get_current_map(), command)) {
302 return;
303 }
304
305 // Lua scripts did not override the command: do the built-in behavior.
306 if (command == GameCommand::PAUSE) {
307 if (is_paused()) {
308 if (can_unpause()) {
309 set_paused(false);
310 }
311 }
312 else {
313 if (can_pause()) {
314 set_paused(true);
315 }
316 }
317 }
318
319 else if (!is_suspended()) {
320 // When the game is not suspended, all other commands apply to the hero.
321 hero->notify_command_pressed(command);
322 }
323}
324
325/**
326 * \brief This function is called when a game command is released.
327 * \param command A game command.
328 */
329void Game::notify_command_released(GameCommand command) {
330
331 bool handled = get_lua_context().game_on_command_released(*this, command);
332
333 if (!handled) {
334
335 handled = get_lua_context().map_on_command_released(get_current_map(), command);
336
337 if (!handled) {
338 // The Lua script did not override the command: do the built-in behavior.
339
340 if (!is_suspended()) {
341 // When the game is not suspended, the command apply to the hero.
342 hero->notify_command_released(command);
343 }
344 }
345 }
346}

348/**
349 * \brief Updates the game elements.
350 *
351 * Updates the map, the equipment, the HUD, etc.
352 */
353void Game::update() {
354
355 // Update the transitions between maps.
356 update_transitions();
357
358 if (restarting || !started) {
359 return;
360 }
361
362 // Update the map.
363 current_map->update();
364
365 // Call game:on_update() in Lua.
366 get_lua_context().game_on_update(*this);
367
368 // Update the equipment and HUD.
369 get_equipment().update();
370 update_commands_effects();
371}

373/**
374 * \brief Handles the transitions.
375 *
376 * This functions changes the map when needed and plays the
377 * transitions between the two maps. This function is called
378 * by the update() function.
379 * Note that the two maps can actually be the same.
380 */
381void Game::update_transitions() {
382
383 if (transition != nullptr) {
384 transition->update();
385 }
386
387 // if the map has just changed, close the current map if any and play an out transition
388 if (next_map != nullptr && transition == nullptr) { // the map has changed (i.e. set_current_map has been called)
389
390 if (current_map == nullptr) { // special case: no map was playing, so we don't have any out transition to do
391 current_map = next_map;
392 next_map = nullptr;
393 }
394 else { // normal case: stop the control and play an out transition before leaving the current map
395 transition = std::unique_ptr<Transition>(Transition::create(
396 transition_style,
397 Transition::Direction::CLOSING,
398 *current_map->get_camera_surface(),
399 this
400 ));
401 transition->start();
402 }
403 }
404
405 Rectangle previous_map_location = current_map->get_location();
406
407 // if a transition was playing and has just been finished
408 if (transition != nullptr && transition->is_finished()) {
409
410 Transition::Direction transition_direction = transition->get_direction();
411 bool needs_previous_surface = transition->needs_previous_surface();
412 transition = nullptr;
413
414 MainLoop& main_loop = get_main_loop();
415 if (restarting) {
416 stop();
417 main_loop.set_game(new Game(main_loop, savegame));
418 this->savegame = nullptr; // The new game is the owner.
419 }
420 else if (transition_direction == Transition::Direction::CLOSING) {
421
422 bool world_changed = next_map != current_map &&
423 (!next_map->has_world() || next_map->get_world() != current_map->get_world());
424
425 if (world_changed) {
426 // Reset the crystal blocks.
427 crystal_state = false;
428 }
429
430 // Determine the destination to use and its name.
431 std::string destination_name = next_map->get_destination_name();
432 bool special_destination = destination_name == "_same" ||
433 destination_name.substr(0,5) == "_side";
434 StartingLocationMode starting_location_mode = StartingLocationMode::NO;
435 if (!special_destination) {
436 EntityPtr destination;
437 if (destination_name.empty()) {
438 std::shared_ptr<Destination> default_destination = next_map->get_entities().get_default_destination();
439 if (default_destination != nullptr) {
440 destination = default_destination;
441 destination_name = destination->get_name();
442 }
443 }
444 else {
445 destination = next_map->get_entities().find_entity(destination_name);
446 }
447 if (destination != nullptr && destination->get_type() == EntityType::DESTINATION) {
448 starting_location_mode =
449 std::static_pointer_cast<Destination>(destination)->get_starting_location_mode();
450 }
451 }
452
453 bool save_starting_location = false;
454 switch (starting_location_mode) {
455
456 case StartingLocationMode::YES:
457 save_starting_location = true;
458 break;
459
460 case StartingLocationMode::NO:
461 save_starting_location = false;
462 break;
463
464 case StartingLocationMode::WHEN_WORLD_CHANGES:
465 save_starting_location = world_changed;
466 break;
467 }
468
469 // Save the location if needed, except if this is a special destination.
470 if (save_starting_location && !special_destination) {
471 get_savegame().set_string(Savegame::KEY_STARTING_MAP, next_map->get_id());
472 get_savegame().set_string(Savegame::KEY_STARTING_POINT, destination_name);
473 }
474
475 if (next_map == current_map) {
476 // same map
477 hero->place_on_destination(*current_map, previous_map_location);
478 transition = std::unique_ptr<Transition>(Transition::create(
479 transition_style,
480 Transition::Direction::OPENING,
481 *current_map->get_camera_surface(),
482 this
483 ));
484 transition->start();
485 next_map = nullptr;
486 }
487 else {
488
489 // change the map
490 current_map->leave();
491
492 // before closing the map, draw it on a backup surface for transition effects
493 // that want to display both maps at the same time
494 if (needs_previous_surface && current_map->get_camera() != nullptr) {
495 previous_map_surface = Surface::create(
496 current_map->get_camera()->get_size()
497 );
498 current_map->draw();
499 current_map->get_camera_surface()->draw(previous_map_surface);
500 }
501
502 // set the next map
503 current_map->unload();
504
505 current_map = next_map;
506 next_map = nullptr;
507 }
508 }
509 else {
510 current_map->notify_opening_transition_finished();
511 previous_map_surface = nullptr;
512 }
513 }
514
515 // if a map has just been set as the current map, start it and play the in transition
516 if (started && !current_map->is_started()) {
517 Debug::check_assertion(current_map->is_loaded(), "This map is not loaded");
518 transition = std::unique_ptr<Transition>(Transition::create(
519 transition_style,
520 Transition::Direction::OPENING,
521 *current_map->get_camera_surface(),
522 this
523 ));
524
525 if (previous_map_surface != nullptr) {
526 // some transition effects need to display both maps simultaneously
527 transition->set_previous_surface(previous_map_surface.get());
528 }
529
530 set_suspended_by_script(false);
531
532 hero->place_on_destination(*current_map, previous_map_location);
533 transition->start();
534 current_map->start();
535 notify_map_changed();
536 }
537}

539/**
540 * \brief Makes sure the keys effects are coherent with the hero's equipment and abilities.
541 */
542void Game::update_commands_effects() {
543
544 // when the game is paused or a dialog box is shown, the sword key is not the usual one
545 if (is_paused() || is_dialog_enabled()) {
546 return; // if the game is interrupted for some other reason (e.g. a transition), let the normal sword icon
547 }
548
549 // make sure the sword key is coherent with having a sword
550 if (get_equipment().has_ability(Ability::SWORD)
551 && commands_effects.get_sword_key_effect() != CommandsEffects::ATTACK_KEY_SWORD) {
552
553 commands_effects.set_sword_key_effect(CommandsEffects::ATTACK_KEY_SWORD);
554 }
555 else if (!get_equipment().has_ability(Ability::SWORD)
556 && commands_effects.get_sword_key_effect() == CommandsEffects::ATTACK_KEY_SWORD) {
557
558 commands_effects.set_sword_key_effect(CommandsEffects::ATTACK_KEY_NONE);
559 }
560}

562/**
563 * \brief Draws the game.
564 * \param dst_surface The surface where the game will be drawn.
565 */
566void Game::draw(const SurfacePtr& dst_surface) {
567
568 if (current_map == nullptr) {
569 // Nothing to do. The game is not fully initialized yet.
570 return;
571 }
572
573 // Draw the map.
574 if (current_map->is_loaded()) {
575 dst_surface->fill_with_color(current_map->get_tileset().get_background_color());
576 current_map->draw();
577 const CameraPtr& camera = current_map->get_camera();
578 if (camera != nullptr) {
579 const SurfacePtr& camera_surface = camera->get_surface();
580 if (transition != nullptr) {
581 transition->draw(*camera_surface);
582 }
583 camera_surface->draw(dst_surface, camera->get_position_on_screen());
584 }
585
586 // Draw the built-in dialog box if any.
587 if (is_dialog_enabled()) {
588 dialog_box.draw(dst_surface);
589 }
590 }
591
592 get_lua_context().game_on_draw(*this, dst_surface);
593}

615/**
616 * \brief Changes the current map.
617 *
618 * Call this function when you want the hero to go to another map.
619 *
620 * \param map_id Id of the map to launch. It must exist.
621 * \param destination_name Name of the destination point you want to use.
622 * An empty string means the default one.
623 * You can also use "_same" to keep the hero's coordinates, or
624 * "_side0", "_side1", "_side2" or "_side3"
625 * to place the hero on a side of the map.
626 * \param transition_style Type of transition between the two maps.
627 */
628void Game::set_current_map(
629 const std::string& map_id,
630 const std::string& destination_name,
631 Transition::Style transition_style) {
632
633 if (current_map != nullptr) {
634 // stop the hero's movement
635 hero->reset_movement();
636 }
637
638 // prepare the next map
639 if (current_map == nullptr || map_id != current_map->get_id()) {
640 // another map
641 next_map = std::make_shared<Map>(map_id);
642 next_map->load(*this);
643 next_map->check_suspended();
644 }
645 else {
646 // same map
647 next_map = current_map;
648 }
649
650 if (current_map != nullptr) {
651 current_map->check_suspended();
652 }
653
654 next_map->set_destination(destination_name);
655 this->transition_style = transition_style;
656}

704/**
705 * \brief Returns whether the game is suspended.
706 *
707 * This is true in the following cases:
708 * - the game is paused,
709 * - or a dialog a being dispayed,
710 * - or a transition between two maps is playing,

712 * - or the camera is moving,
713 * - or a script explicitly suspended the game.

715 * \return true if the game is suspended

718
719 return current_map == nullptr ||
720 is_paused() ||

722 is_playing_transition() ||
723 is_showing_game_over() ||
724 is_suspended_by_camera() ||

773/**
774 * \brief Returns whether the player can currently pause the game.
775 *
776 * He can pause the game if the pause command is enabled
777 * and if his life is greater than zero.

823/**
824 * \brief Pauses or resumes the game.
825 * \param paused true to pause the game, false to resume it.
826 */
827void Game::set_paused(bool paused) {
828
829 if (paused != is_paused()) {
830
831 this->paused = paused;
832 if (paused) {
833 commands_effects.save_action_key_effect();
834 commands_effects.set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
835 commands_effects.save_sword_key_effect();
836 commands_effects.set_sword_key_effect(CommandsEffects::ATTACK_KEY_NONE);
837 commands_effects.set_pause_key_effect(CommandsEffects::PAUSE_KEY_RETURN);
838 get_lua_context().game_on_paused(*this);
839 }
840 else {
841 get_lua_context().game_on_unpaused(*this);
842 commands_effects.restore_action_key_effect();
843 commands_effects.restore_sword_key_effect();
844 commands_effects.set_pause_key_effect(CommandsEffects::PAUSE_KEY_PAUSE);
845 }
846 }

906/**
907 * \brief Launches the game-over sequence.
908 */
909void Game::start_game_over() {
910
911 Debug::check_assertion(!is_showing_game_over(),
912 "The game-over sequence is already active");
913
914 showing_game_over = true;
915
916 if (!get_lua_context().game_on_game_over_started(*this)) {
917 // The script does not define a game-over sequence:
918 // then, the built-in behavior is to restart the game.
919 restart();
920 stop_game_over();
921 }
922}

924/**
925 * \brief Cancels the current game-over sequence.
926 */
927void Game::stop_game_over() {
928
929 Debug::check_assertion(is_showing_game_over(),
930 "The game-over sequence is not running");
931
932 get_lua_context().game_on_game_over_finished(*this);
933 showing_game_over = false;
934 if (!restarting && !get_main_loop().is_resetting()) {
935 // The hero gets back to life.
936 current_map->check_suspended(); // To unsuspend the hero before making him blink.
937 hero->notify_game_over_finished();
938 }
