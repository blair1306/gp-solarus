44/**
45 * \brief Starts this state.
46 * \param previous_state the previous state
47 */
48void Hero::SpinAttackState::start(const State* previous_state) {
49
50 HeroState::start(previous_state);
51
52 // play the sound
53 play_spin_attack_sound();
54
55 // start the animation
56 Hero& hero = get_entity();
57 if (get_equipment().has_ability(Ability::SWORD_KNOWLEDGE)) {
58 get_sprites().set_animation_super_spin_attack();
59 std::shared_ptr<CircleMovement> movement =
60 std::make_shared<CircleMovement>(false);
61 movement->set_center(hero.get_xy());
62 movement->set_radius_speed(128);
63 movement->set_radius(24);
64 movement->set_angle_speed(540);
65 movement->set_max_rotations(3);
66 movement->set_clockwise(true);
67 hero.set_movement(movement);
68 }
69 else {
70 get_sprites().set_animation_spin_attack();
71 }
72}
73
74/**
75 * \brief Ends this state.
76 * \param next_state the next state
77 */
78void Hero::SpinAttackState::stop(const State* next_state) {
79
80 HeroState::stop(next_state);
81
82 Hero& hero = get_entity();
83 if (hero.get_movement() != nullptr) {
84 // stop the movement of being pushed by an enemy after hitting him
85 hero.clear_movement();
86 }
87}
88
89/**
90 * \brief Updates this state.
91 */
92void Hero::SpinAttackState::update() {
93
94 // check the animation
95 Hero& hero = get_entity();
96 if (get_sprites().is_animation_finished()) {
97 hero.set_state(new FreeState(hero));
98 }
99
100 // check the movement if any
101 if (hero.get_movement() != nullptr && hero.get_movement()->is_finished()) {
102 hero.clear_movement();
103
104 if (!being_pushed) {
105 // end of a super spin attack
106 hero.set_state(new FreeState(hero));
107 }
108 }
109}

157/**
158 * \brief Plays the sword loading sound.
159 */
160void Hero::SpinAttackState::play_spin_attack_sound() {
161
162 std::ostringstream oss;
163 oss << "sword_spin_attack_release_" << get_equipment().get_ability(Ability::SWORD);
164 std::string custom_sound_name = oss.str();
165 if (Sound::exists(custom_sound_name)) {
166 Sound::play(custom_sound_name); // this particular sword has a spin attack sound effect
167 }
168 else {
169 Sound::play("sword_spin_attack_release");
170 }
171}

226/**
227 * \brief Notifies this state that the hero has just failed to change its
228 * position because of obstacles.
229 */
230void Hero::SpinAttackState::notify_obstacle_reached() {
231
232 // the hero reached an obstacle while being pushed after hitting an enemy
233 // or making a super spin attack
234 get_entity().clear_movement();
235
236 if (!being_pushed) {
237 // obstacle while making a super spin attack: finish with a normal spin attack
238 get_sprites().set_animation_spin_attack();
239 }
240}

242/**
243 * \copydoc Entity::State::notify_attacked_enemy
244 */
245void Hero::SpinAttackState::notify_attacked_enemy(
246 EnemyAttack attack,
247 Enemy& victim,
248 const Sprite* victim_sprite,
249 EnemyReaction::Reaction& result,
250 bool /* killed */) {
251
252 Hero& hero = get_entity();
253 if (result.type != EnemyReaction::ReactionType::IGNORED && attack == EnemyAttack::SWORD) {
254
255 if (victim.get_push_hero_on_sword()) {
256
257 if (hero.get_movement() != nullptr) {
258 // interrupting a super spin attack: finish with a normal one
259 hero.clear_movement();
260 get_sprites().set_animation_spin_attack();
261 }
262
263 being_pushed = true;
264 double angle = victim.get_angle(hero, victim_sprite, nullptr);
265 std::shared_ptr<StraightMovement> movement =
266 std::make_shared<StraightMovement>(false, true);
267 movement->set_max_distance(24);
268 movement->set_speed(120);
269 movement->set_angle(angle);
270 hero.set_movement(movement);
271 }
272 }
