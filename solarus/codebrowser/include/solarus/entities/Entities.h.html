227/**
228 * \brief Returns the ground property of tiles at the specified point.
229 *
230 * Only static tiles are considered here (not the dynamic entities).
231 * Use get_ground() instead to also take into account dynamic entities that
232 * may change the ground, like dynamic tiles and destructible entities.
233 *
234 * This function assumes that the parameters are correct: for performance
235 * reasons, no check is done here.
236 *
237 * \param layer Layer of the point.
238 * \param x X coordinate of the point.
239 * \param y Y coordinate of the point.
240 * \return The ground of the highest tile at this place.
241 */
242inline Ground Entities::get_tile_ground(int layer, int x, int y) const {
243
244 return tiles_ground.at(layer)[(y >> 3) * map_width8 + (x >> 3)];
245}

256/**
257 * \brief Returns all entities of a type.
258 * \return All entities of the type.
259 */
260template<typename T>
261std::set<std::shared_ptr<const T>> Entities::get_entities_by_type() const {
262
263 std::set<std::shared_ptr<const T>> result;
264
265 const EntityType type = T::ThisType;
266 const auto& it = entities_by_type.find(type);
267 if (it == entities_by_type.end()) {
268 return result;
269 }
270
271 for (const auto& kvp : it->second) {
272 for (const ConstEntityPtr& entity : kvp.second) {
273 result.insert(std::static_pointer_cast<const T>(entity));
274 }
275 }
276 return result;
277}

302/**
303 * \brief Returns all entities of a type on the given layer.
304 * \param layer The layer to get entities from.
305 * \return All entities of the type on this layer.
306 */
307template<typename T>
308std::set<std::shared_ptr<const T>> Entities::get_entities_by_type(int layer) const {
309
310 std::set<std::shared_ptr<const T>> result;
311
312 const EntityType type = T::ThisType;
313 const auto& it = entities_by_type.find(type);
314 if (it == entities_by_type.end()) {
315 return result;
316 }
317
318 const ByLayer<EntitySet>& sets = it->second;
319 const auto& layer_it = sets.find(layer);
320 if (layer_it == sets.end()) {
321 return result;
322 }
323 for (const EntityPtr& entity : layer_it->second) {
324 result.insert(std::static_pointer_cast<const T>(entity));
325 }
326 return result;
327}
