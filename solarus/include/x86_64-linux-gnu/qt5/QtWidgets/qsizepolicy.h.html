42#ifndef QSIZEPOLICY_H
43#define QSIZEPOLICY_H
44
45#include <QtCore/qobject.h>
46
47QT_BEGIN_NAMESPACE
48
49
50class QVariant;
51
52class Q_WIDGETS_EXPORT QSizePolicy
53{
54 Q_GADGET
55 Q_ENUMS(Policy)
56
57public:
58 enum PolicyFlag {
59 GrowFlag = 1,
60 ExpandFlag = 2,
61 ShrinkFlag = 4,
62 IgnoreFlag = 8
63 };
64
65 enum Policy {
66 Fixed = 0,
67 Minimum = GrowFlag,
68 Maximum = ShrinkFlag,
69 Preferred = GrowFlag | ShrinkFlag,
70 MinimumExpanding = GrowFlag | ExpandFlag,
71 Expanding = GrowFlag | ShrinkFlag | ExpandFlag,
72 Ignored = ShrinkFlag | GrowFlag | IgnoreFlag
73 };
74
75 enum ControlType {
76 DefaultType = 0x00000001,
77 ButtonBox = 0x00000002,
78 CheckBox = 0x00000004,
79 ComboBox = 0x00000008,
80 Frame = 0x00000010,
81 GroupBox = 0x00000020,
82 Label = 0x00000040,
83 Line = 0x00000080,
84 LineEdit = 0x00000100,
85 PushButton = 0x00000200,
86 RadioButton = 0x00000400,
87 Slider = 0x00000800,
88 SpinBox = 0x00001000,
89 TabWidget = 0x00002000,
90 ToolButton = 0x00004000
91 };
92 Q_DECLARE_FLAGS(ControlTypes, ControlType)
93
94 QSizePolicy() : data(0) { }
95
96 // ### Qt 5: merge these two constructors (with type == DefaultType)
97 QSizePolicy(Policy horizontal, Policy vertical, ControlType type = DefaultType)
98 : data(0) {
99 bits.horPolicy = horizontal;
100 bits.verPolicy = vertical;
101 setControlType(type);
102 }
103 Policy horizontalPolicy() const { return static_cast<Policy>(bits.horPolicy); }
104 Policy verticalPolicy() const { return static_cast<Policy>(bits.verPolicy); }
105 ControlType controlType() const;
106
107 void setHorizontalPolicy(Policy d) { bits.horPolicy = d; }
108 void setVerticalPolicy(Policy d) { bits.verPolicy = d; }
109 void setControlType(ControlType type);
110
111 Qt::Orientations expandingDirections() const {
112 Qt::Orientations result;
113 if (verticalPolicy() & ExpandFlag)
114 result |= Qt::Vertical;
115 if (horizontalPolicy() & ExpandFlag)
116 result |= Qt::Horizontal;
117 return result;
118 }
119
120 void setHeightForWidth(bool b) { bits.hfw = b; }
121 bool hasHeightForWidth() const { return bits.hfw; }
122 void setWidthForHeight(bool b) { bits.wfh = b; }
123 bool hasWidthForHeight() const { return bits.wfh; }
124
125 bool operator==(const QSizePolicy& s) const { return data == s.data; }
126 bool operator!=(const QSizePolicy& s) const { return data != s.data; }
127 operator QVariant() const; // implemented in qlayoutitem.cpp
128
129 int horizontalStretch() const { return static_cast<int>(bits.horStretch); }
130 int verticalStretch() const { return static_cast<int>(bits.verStretch); }
131 void setHorizontalStretch(int stretchFactor) { bits.horStretch = static_cast<quint32>(qBound(0, stretchFactor, 255)); }
132 void setVerticalStretch(int stretchFactor) { bits.verStretch = static_cast<quint32>(qBound(0, stretchFactor, 255)); }
133
134 bool retainSizeWhenHidden() const { return bits.retainSizeWhenHidden; }
135 void setRetainSizeWhenHidden(bool retainSize) { bits.retainSizeWhenHidden = retainSize; }
136
137 void transpose();
138
139
140private:
141#ifndef QT_NO_DATASTREAM
142 friend Q_WIDGETS_EXPORT QDataStream &operator<<(QDataStream &, const QSizePolicy &);
143 friend Q_WIDGETS_EXPORT QDataStream &operator>>(QDataStream &, QSizePolicy &);
144#endif
145 QSizePolicy(int i) : data(i) { }
146
147 struct Bits {
148 quint32 horStretch : 8;
149 quint32 verStretch : 8;
150 quint32 horPolicy : 4;
151 quint32 verPolicy : 4;
152 quint32 ctype : 5;
153 quint32 hfw : 1;
154 quint32 wfh : 1;
155 quint32 retainSizeWhenHidden : 1;
156 };
157 union {
158 Bits bits;
159 quint32 data;
160 };
161};
162
163Q_DECLARE_OPERATORS_FOR_FLAGS(QSizePolicy::ControlTypes)
164
165#ifndef QT_NO_DATASTREAM
166// implemented in qlayout.cpp
167Q_WIDGETS_EXPORT QDataStream &operator<<(QDataStream &, const QSizePolicy &);
168Q_WIDGETS_EXPORT QDataStream &operator>>(QDataStream &, QSizePolicy &);
169#endif
170
171#ifndef QT_NO_DEBUG_STREAM
172Q_WIDGETS_EXPORT QDebug operator<<(QDebug dbg, const QSizePolicy &);
173#endif
174
175inline void QSizePolicy::transpose() {
176 Policy hData = horizontalPolicy();
177 Policy vData = verticalPolicy();
178 int hStretch = horizontalStretch();
179 int vStretch = verticalStretch();
180 setHorizontalPolicy(vData);
181 setVerticalPolicy(hData);
182 setHorizontalStretch(vStretch);
183 setVerticalStretch(hStretch);
184}
