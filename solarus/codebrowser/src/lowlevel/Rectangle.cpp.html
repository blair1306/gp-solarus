17#include "solarus/lowlevel/Rectangle.h"
18#include <ostream>
19
20namespace Solarus {
21
22/**
23 * \brief Prints a rectangle to an output stream.
24 * \param stream The stream.
25 * \param rectangle A rectangle.
26 * \return The output stream.
27 */
28std::ostream& operator<<(std::ostream& stream, const Rectangle& rectangle) {
29
30 stream << rectangle.get_xy() << "x" << rectangle.get_size();
31 return stream;
32}
