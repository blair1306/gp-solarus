26/**
27 * \brief Constructor.
28 * \param map The map. Its size must be known.
29 * \param layer The layer to represent.

31NonAnimatedRegions::NonAnimatedRegions(Map& map, int layer):
32 map(map),
33 layer(layer),
34 non_animated_tiles(map.get_size(), Size(512, 256)) {

36}

38/**
39 * \brief Adds a tile to the list of tiles.
40 */
41void NonAnimatedRegions::add_tile(const TileInfo& tile) {
42
43 Debug::check_assertion(optimized_tiles_surfaces.empty(),
44 "Tile regions are already built");
45 Debug::check_assertion(tile.layer == layer, "Wrong layer for add tile");
46
47 tiles.push_back(tile);
48}

50/**
51 * \brief Determines which rectangles are animated to allow drawing all non-animated
52 * rectangles of tiles only once.
53 * \param[out] rejected_tiles The list of tiles that are in animated regions.
54 * They include all animated tiles plus the static tiles overlapping them.
55 * You will have to redraw these tiles at each frame.
56 */
57void NonAnimatedRegions::build(std::vector<TileInfo>& rejected_tiles) {
58
59 Debug::check_assertion(optimized_tiles_surfaces.empty(),
60 "Tile regions are already built");
61
62 const int map_width8 = map.get_width8();
63 const int map_height8 = map.get_height8();
64
65 // Initialize the are_squares_animated booleans to false.
66 for (int i = 0; i < map_width8 * map_height8; ++i) {
67 are_squares_animated.push_back(false);
68 }
69
70 // Create the surfaces where all non-animated tiles will be drawn.
71 optimized_tiles_surfaces.resize(non_animated_tiles.get_num_cells());
72
73 // Mark animated 8x8 squares of the map.
74 for (size_t i = 0; i < tiles.size(); ++i) {
75 const TileInfo& tile = tiles[i];
76 if (tile.pattern->is_animated()) {
77 // Animated tile: mark its region as non-optimizable
78 // (otherwise, a non-animated tile above an animated one would screw us).
79
80 int tile_x8 = tile.box.get_x() / 8;
81 int tile_y8 = tile.box.get_y() / 8;
82 int tile_width8 = tile.box.get_width() / 8;
83 int tile_height8 = tile.box.get_height() / 8;
84
85 for (int i = 0; i < tile_height8; i++) {
86 for (int j = 0; j < tile_width8; j++) {
87
88 const int x8 = tile_x8 + j;
89 const int y8 = tile_y8 + i;
90 if (x8 >= 0 && x8 < map_width8 && y8 >= 0 && y8 < map_height8) {
91 int index = y8 * map_width8 + x8;
92 are_squares_animated[index] = true;
93 }
94 }
95 }
96 }
97 }
98
99 // Build the list of animated tiles and tiles overlapping them.
100 for (const TileInfo& tile: tiles) {
101 if (!tile.pattern->is_animated()) {
102 non_animated_tiles.add(tile, tile.box);
103 if (overlaps_animated_tile(tile)) {
104 rejected_tiles.push_back(tile);
105 }
106 }
107 else {
108 rejected_tiles.push_back(tile);
109 }
110 }
111
112 // No need to keep all tiles at this point.
113 // Just keep the non-animated ones to draw them lazily.
114 tiles.clear();
115}

117/**
118 * \brief Clears previous drawings because the tileset has changed.
119 */
120void NonAnimatedRegions::notify_tileset_changed() {
121
122 for (unsigned i = 0; i < non_animated_tiles.get_num_cells(); ++i) {
123 optimized_tiles_surfaces[i] = nullptr;
124 }
125 // Everything will be redrawn when necessary.
126}

128/**
129 * \brief Returns whether a tile is overlapping an animated other tile.
130 * \param tile The tile to check.
131 * \return \c true if this tile is overlapping an animated tile.
132 */
133bool NonAnimatedRegions::overlaps_animated_tile(const TileInfo& tile) const {
134
135 if (tile.layer != layer) {
136 return false;
137 }
138
139 int tile_x8 = tile.box.get_x() / 8;
140 int tile_y8 = tile.box.get_y() / 8;
141 int tile_width8 = tile.box.get_width() / 8;
142 int tile_height8 = tile.box.get_height() / 8;
143
144 for (int i = 0; i < tile_height8; i++) {
145 for (int j = 0; j < tile_width8; j++) {
146
147 int x8 = tile_x8 + j;
148 int y8 = tile_y8 + i;
149 if (x8 >= 0 && x8 < map.get_width8() && y8 >= 0 && y8 < map.get_height8()) {
150
151 int index = y8 * map.get_width8() + x8;
152 if (are_squares_animated[index]) {
153 return true;
154 }
155 }
156 }
157 }
158 return false;
159}

161/**
162 * \brief Draws a layer of non-animated regions of tiles on the current map.
163 */
164void NonAnimatedRegions::draw_on_map() {
165
166 const CameraPtr& camera = map.get_camera();
167 if (camera == nullptr) {
168 return;
169 }
170
171 // Check all grid cells that overlap the camera.
172 const int num_rows = non_animated_tiles.get_num_rows();
173 const int num_columns = non_animated_tiles.get_num_columns();
174 const Size& cell_size = non_animated_tiles.get_cell_size();
175 const Rectangle& camera_position = camera->get_bounding_box();
176
177 const int row1 = camera_position.get_y() / cell_size.height;
178 const int row2 = (camera_position.get_y() + camera_position.get_height()) / cell_size.height;
179 const int column1 = camera_position.get_x() / cell_size.width;
180 const int column2 = (camera_position.get_x() + camera_position.get_width()) / cell_size.width;
181
182 if (row1 > row2 || column1 > column2) {
183 // No cell.
184 return;
185 }
186
187 for (int i = row1; i <= row2; ++i) {
188 if (i < 0 || i >= num_rows) {
189 continue;
190 }
191
192 for (int j = column1; j <= column2; ++j) {
193 if (j < 0 || j >= num_columns) {
194 continue;
195 }
196
197 // Make sure this cell is built.
198 int cell_index = i * num_columns + j;
199 if (optimized_tiles_surfaces[cell_index] == nullptr) {
200 // Lazily build the cell.
201 build_cell(cell_index);
202 }
203
204 const Point cell_xy = {
205 j * cell_size.width,
206 i * cell_size.height
207 };
208
209 const Point dst_position = cell_xy - camera_position.get_xy();
210 optimized_tiles_surfaces[cell_index]->draw(
211 map.get_camera_surface(), dst_position
212 );
213 }
214 }
215}

217/**
218 * \brief Draws all non-animated tiles of a cell on its surface.
219 * \param cell_index Index of the cell to draw.
220 */
221void NonAnimatedRegions::build_cell(int cell_index) {
222
223 Debug::check_assertion(
224 cell_index >= 0 && (size_t) cell_index < non_animated_tiles.get_num_cells(),
225 "Wrong cell index"
226 );
227 Debug::check_assertion(optimized_tiles_surfaces[cell_index] == nullptr,
228 "This cell is already built"
229 );
230
231 const int row = cell_index / non_animated_tiles.get_num_columns();
232 const int column = cell_index % non_animated_tiles.get_num_columns();
233
234 // Position of this cell on the map.
235 const Size cell_size = non_animated_tiles.get_cell_size();
236 const Point cell_xy = {
237 column * cell_size.width,
238 row * cell_size.height
239 };
240
241 SurfacePtr cell_surface = Surface::create(cell_size);
242 optimized_tiles_surfaces[cell_index] = cell_surface;
243 // Let this surface as a software destination because it is built only
244 // once (here) and never changes later.
245
246 const std::vector<TileInfo>& tiles_in_cell =
247 non_animated_tiles.get_elements(cell_index);
248 for (const TileInfo& tile: tiles_in_cell) {
249
250 Rectangle dst_position(
251 tile.box.get_x() - cell_xy.x,
252 tile.box.get_y() - cell_xy.y,
253 tile.box.get_width(),
254 tile.box.get_height()
255 );
256
257 tile.pattern->fill_surface(
258 cell_surface,
259 dst_position,
260 map.get_tileset(),
261 cell_xy
262 );
263 }
264
265 // Remember that non-animated tiles are drawn after animated ones.
266 // We may have drawn too much.
267 // We have to make sure we don't exceed the non-animated regions.
268 // Erase 8x8 squares that contain animated tiles.
269 for (int y = cell_xy.y; y < cell_xy.y + cell_size.height; y += 8) {
270 if (y >= map.get_height()) { // The last cell might exceed the map border.
271 continue;
272 }
273 for (int x = cell_xy.x; x < cell_xy.x + cell_size.width; x += 8) {
274 if (x >= map.get_width()) {
275 continue;
276 }
277
278 int square_index = (y / 8) * map.get_width8() + (x / 8);
279
280 if (are_squares_animated[square_index]) {
281 Rectangle animated_square(
282 x - cell_xy.x,
283 y - cell_xy.y,
284 8,
285 8
286 );
287 cell_surface->clear(animated_square);
288 }
289 }
290 }
