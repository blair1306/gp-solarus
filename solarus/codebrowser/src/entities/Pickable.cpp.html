39/**
40 * \brief Creates a pickable item with the specified subtype.
41 * \param name Name identifying the entity on the map or an empty string.
42 * \param layer layer of the pickable item to create on the map
43 * \param xy Coordinates of the pickable item to create.
44 * \param treasure the treasure to give when the item is picked
45 */
46Pickable::Pickable(
47 const std::string& name,
48 int layer,
49 const Point& xy,
50 const Treasure& treasure
51):
52 Entity(name, 0, layer, xy, Size(0, 0)),
53 treasure(treasure),
54 given_to_player(false),
55 shadow_sprite(),
56 falling_height(FALLING_NONE),
57 will_disappear(false),
58 shadow_xy(xy),
59 appear_date(System::now()),
60 allow_pick_date(0),
61 can_be_picked(true),
62 blink_date(0),
63 disappear_date(0),
64 entity_followed(nullptr) {
65
66 set_collision_modes(CollisionMode::COLLISION_OVERLAPPING | CollisionMode::COLLISION_SPRITE);
67}

77/**
78 * \brief Creates a pickable item with the specified subtype.
79 *
80 * This method acts like a constructor, except that it can return nullptr in several cases:
81 * - the treasure is saved and the player already has it,
82 * or:
83 * - the treasure is empty,
84 * or:
85 * - the item cannot be obtained by the hero yet,
86 * or:
87 * - the animation of the item is missing in sprite 'entities/items'.
88 *
89 * \param game the current game
90 * \param name Name identifying the entity on the map or an empty string.
91 * \param layer layer of the pickable treasure to create on the map
92 * \param xy Coordinates of the pickable treasure to create
93 * \param treasure the treasure to give
94 * \param falling_height to make the treasure fall when it appears
95 * \param force_persistent true to make the treasure stay forever
96 * (otherwise, the properties of the treasure
97 * decide if it disappears after some time)
98 * \return the pickable item created, or nullptr
99 */
100std::shared_ptr<Pickable> Pickable::create(
101 Game& /* game */,
102 const std::string& name,
103 int layer,
104 const Point& xy,
105 Treasure treasure,
106 FallingHeight falling_height,
107 bool force_persistent
108) {
109 treasure.ensure_obtainable();
110
111 // Don't create anything if there is no treasure to give.
112 if (treasure.is_found() || treasure.is_empty()) {
113 return nullptr;
114 }
115
116 std::shared_ptr<Pickable> pickable = std::make_shared<Pickable>(
117 name, layer, xy, treasure
118 );
119
120 // Set the item properties.
121 pickable->falling_height = falling_height;
122 pickable->will_disappear = !force_persistent && treasure.get_item().get_can_disappear();
123
124 // Initialize the pickable item.
125 if (!pickable->initialize_sprites()) {
126 return nullptr; // No valid sprite: don't create the pickable.
127 }
128 pickable->initialize_movement();
129
130 return pickable;
131}

140/**
141 * \brief Creates the sprite of this pickable treasure.
142 *
143 * Pickable treasures are represented with two sprites:
144 * the treasure itself and, for some items, a shadow.
145 *
146 * \return \c true in case of success, \c false if the animation corresponding
147 * to the treasure is missing.
148 */
149bool Pickable::initialize_sprites() {
150
151 EquipmentItem& item = treasure.get_item();
152
153 // Shadow sprite first, because below the treasure sprite.
154 shadow_sprite = nullptr;
155 const std::string& shadow_animation = item.get_shadow();
156
157 bool has_shadow = false;
158 if (!shadow_animation.empty()) {
159 shadow_sprite = create_sprite("entities/shadow", "shadow");
160 has_shadow = shadow_sprite->has_animation(shadow_animation);
161 }
162
163 if (!has_shadow) {
164 // No shadow or no such shadow animation.
165 shadow_sprite = nullptr;
166 }
167 else {
168 shadow_sprite->set_current_animation(shadow_animation);
169 }
170
171 // Main sprite.
172 const std::string item_name = treasure.get_item_name();
173 item_sprite = create_sprite("entities/items", "treasure");
174 set_default_sprite_name("treasure");
175
176 if( !item_sprite->has_animation(item_name)) {
177 std::ostringstream oss;
178 oss << "Cannot create pickable treasure '" << item_name
179 << "': Sprite 'entities/items' has no animation '"
180 << item_name << "'";
181 Debug::error(oss.str());
182 return false;
183 }
184
185 item_sprite->set_current_animation(item_name);
186 int direction = treasure.get_variant() - 1;
187 if (direction < 0 || direction >= item_sprite->get_nb_directions()) {
188 std::ostringstream oss;
189 oss << "Pickable treasure '" << item_name
190 << "' has variant " << treasure.get_variant()
191 << " but sprite 'entities/items' only has "
192 << item_sprite->get_nb_directions() << " variant(s) in animation '"
193 << item_name << "'";
194 Debug::error(oss.str());
195 direction = 0; // Fallback.
196 }
197 item_sprite->set_current_direction(direction);
198
199 enable_pixel_collisions();
200
201 // Set the origin point and the size of the entity.
202 set_size(16, 16);
203 set_origin(8, 13);
204
205 uint32_t now = System::now();
206
207 if (falling_height != FALLING_NONE) {
208 allow_pick_date = now + 700; // The player will be allowed to take the item after 0.7 seconds.
209 can_be_picked = false;
210 }
211 else {
212 can_be_picked = true;
213 }
214
215 // Initialize the item removal.
216 if (will_disappear) {
217 blink_date = now + 8000; // The item blinks after 8s.
218 disappear_date = now + 10000; // The item disappears after 10s.
219 }
220
221 return true;
222}

224/**
225 * \copydoc Entity::notify_created
226 */
227void Pickable::notify_created() {
228
229 Entity::notify_created();
230
231 update_ground_below();
232 notify_ground_below_changed(); // Necessary if on empty ground.
233
234 // This entity and the map are now both ready. Notify the Lua item.
235 EquipmentItem& item = get_equipment().get_item(treasure.get_item_name());
236 item.notify_pickable_appeared(*this);
237}

290/**
291 * \brief This function is called by the engine when an entity overlaps the pickable item.
292 *
293 * If the entity is the player, we give him the item, and the map is notified
294 * to destroy it.
295 * \param entity_overlapping the entity overlapping the detector
296 * \param collision_mode the collision mode that detected the collision
297 */
298void Pickable::notify_collision(Entity& entity_overlapping, CollisionMode /* collision_mode */) {
299
300 if (entity_overlapping.is_hero()) {
301 try_give_item_to_player();
302 }
303 else if (entity_followed == nullptr) {
304
305 EntityPtr shared_entity_overlapping =
306 std::static_pointer_cast<Entity>(entity_overlapping.shared_from_this());
307 if (entity_overlapping.get_type() == EntityType::BOOMERANG) {
308 Boomerang& boomerang = static_cast<Boomerang&>(entity_overlapping);
309 if (!boomerang.is_going_back()) {
310 boomerang.go_back();
311 }
312 entity_followed = shared_entity_overlapping;
313 }
314 else if (entity_overlapping.get_type() == EntityType::HOOKSHOT) {
315 Hookshot& hookshot = static_cast<Hookshot&>(entity_overlapping);
316 if (!hookshot.is_going_back()) {
317 hookshot.go_back();
318 }
319 entity_followed = shared_entity_overlapping;
320 }
321
322 if (entity_followed != nullptr) {
323 clear_movement();
324 set_movement(std::make_shared<RelativeMovement>(
325 entity_followed, 0, 0, true
326 ));
327 falling_height = FALLING_NONE;
328 set_blinking(false);
329 }
330 }

333/**
334 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
335 */
336void Pickable::notify_collision(
337 Entity& other_entity,
338 Sprite& this_sprite,
339 Sprite& other_sprite
340) {
341 if (&this_sprite != item_sprite.get()) {
342 // Ignore collisions with the shadow.
343 return;
344 }
345
346 // Taking the item with the sword.
347 if (other_entity.is_hero()) {
348 Hero& hero = static_cast<Hero&>(other_entity);
349 if (other_sprite.get_animation_set_id() == hero.get_hero_sprites().get_sword_sprite_id()) {
350 try_give_item_to_player();
351 }
352 }
353}

355/**
356 * \brief Reacts to the ground of the pickable.
357 *
358 * It is removed if it is on water, lava or a hole.
359 * It goes to the lower layer if the ground is empty.
360 */
361void Pickable::check_bad_ground() {
362
363 if (is_being_removed()) {
364 // Be silent if the pickable was already removed by a script.
365 return;
366 }
367
368 if (get_entity_followed() != nullptr) {
369 // We are attached to a hookshot or boomerang: don't fall.
370 return;
371 }
372
373 if (get_y() < shadow_xy.y) {
374 // The pickable is above the ground for now, let it fall first.
375 return;
376 }
377
378 if (get_movement() != nullptr && !get_movement()->is_finished()) {
379 // The falling movement is not finished yet.
380 return;
381 }
382
383 if (System::now() <= appear_date + 200) {
384 // The pickable appeared very recently, let the user see it for
385 // a short time at least.
386 return;
387 }
388
389 Ground ground = get_ground_below();
390 switch (ground) {
391
392 case Ground::EMPTY:
393 {
394 // Fall to a lower layer.
395 int layer = get_layer();
396 if (layer > 0) {
397 --layer;
398 get_entities().set_entity_layer(*this, layer);
399 }
400 }
401 break;
402
403 case Ground::HOLE:
404 {
405 Sound::play("jump");
406 remove_from_map();
407 }
408 break;
409
410 case Ground::DEEP_WATER:
411 case Ground::LAVA:
412 {
413 Sound::play("splash");
414 remove_from_map();
415 }
416 break;
417
418 default:
419 break;
420 }
421}

423/**
424 * \brief Gives the item to the player.
425 */
426void Pickable::try_give_item_to_player() {
427
428 EquipmentItem& item = treasure.get_item();
429
430 if (!can_be_picked
431 || given_to_player
432 || get_game().is_dialog_enabled()
433 || !get_hero().can_pick_treasure(item)) {
434 return;
435 }
436
437 given_to_player = true;
438
439 remove_from_map();
440
441 // play the sound
442 const std::string& sound_id = item.get_sound_when_picked();
443 if (!sound_id.empty()) {
444 Sound::play(sound_id);
445 }
446
447 // give the item
448 if (item.get_brandish_when_picked()) {
449 // The treasure is brandished.
450 // on_obtained() will be called after the dialog.
451 get_hero().start_treasure(treasure, ScopedLuaRef());
452 }
453 else {
454 treasure.give_to_player();
455
456 // Call on_obtained() immediately since the treasure is not brandished.
457 get_lua_context()->item_on_obtained(item, treasure);
458 get_lua_context()->map_on_obtained_treasure(get_map(), treasure);
459 }
460}

462/**
463 * \brief Sets whether the pickable treasure is blinking.
464 * \param blinking true to make it blink, false to make it stop blinking
465 */
466void Pickable::set_blinking(bool blinking) {
467
468 uint32_t blink_delay = blinking ? 75 : 0;
469
470 if (item_sprite != nullptr) {
471 item_sprite->set_blinking(blink_delay);
472 }
473
474 if (shadow_sprite != nullptr) {
475 shadow_sprite->set_blinking(blink_delay);
476 }
477}

479/**
480 * \brief This function is called by the map when the game is suspended or resumed.
481 *
482 * This is a redefinition of Entity::set_suspended() to suspend the timer
483 * which makes the pickable item disappear after a few seconds.
484 *
485 * \param suspended true to suspend the entity, false to resume it
486 */
487void Pickable::set_suspended(bool suspended) {
488
489 Entity::set_suspended(suspended); // suspend the animation and the movement
490
491 if (shadow_sprite != nullptr) {
492 shadow_sprite->set_suspended(suspended);
493 }
494
495 if (!suspended) {
496 // suspend the timers
497
498 uint32_t now = System::now();
499
500 if (!can_be_picked && get_when_suspended() != 0) {
501 allow_pick_date = now + (allow_pick_date - get_when_suspended());
502 }
503
504 if (will_disappear) {
505
506 // the game is being resumed
507 // recalculate the blinking date and the disappearing date
508 if (get_when_suspended() != 0) {
509 blink_date = now + (blink_date - get_when_suspended());
510 disappear_date = now + (disappear_date - get_when_suspended());
511 }
512 }
513 }
514}

516/**
517 * \brief Updates the pickable item.
518 *
519 * This function is called repeatedly by the map.
520 * This is a redefinition of Entity::update() to make
521 * the item blink and then disappear after an amount of time.
522 */
523void Pickable::update() {
524
525 // update the animations and the movement
526 Entity::update();
527
528 // update the shadow
529 if (shadow_sprite != nullptr) {
530 shadow_sprite->update();
531 }
532
533 shadow_xy.x = get_x();
534 if (!is_falling()) {
535 shadow_xy.y = get_y();
536 }
537
538 if (entity_followed != nullptr && entity_followed->is_being_removed()) {
539
540 if (entity_followed->get_type() == EntityType::BOOMERANG ||
541 entity_followed->get_type() == EntityType::HOOKSHOT) {
542 // The pickable may have been dropped by the boomerang/hookshot
543 // not exactly on the hero so let's fix this.
544 if (get_distance(get_hero()) < 16) {
545 try_give_item_to_player();
546 }
547 }
548 entity_followed = nullptr;
549 }
550
551 check_bad_ground();
552
553 if (!is_suspended()) {
554
555 // check the timer
556 uint32_t now = System::now();
557
558 // wait 0.7 second before allowing the hero to take the item
559 if (!can_be_picked && now >= allow_pick_date) {
560 can_be_picked = true;
561 falling_height = FALLING_NONE;
562 get_hero().check_collision_with_detectors();
563 }
564 else {
565 // make the item blink and then disappear
566 if (will_disappear) {
567
568 if (now >= blink_date && !item_sprite->is_blinking() && entity_followed == nullptr) {
569 set_blinking(true);
570 }
571
572 if (now >= disappear_date) {
573 remove_from_map();
574 }
575 }
576 }
577 }
578}
