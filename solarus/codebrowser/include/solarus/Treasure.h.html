17#ifndef SOLARUS_TREASURE_H
18#define SOLARUS_TREASURE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/SurfacePtr.h"
22#include "solarus/SpritePtr.h"
23#include <string>
24
25namespace Solarus {
26
27class EquipmentItem;
28class Game;
29class Sprite;
30
31/**
32 * \brief Represents an item that the hero brandishes when he receives it.
33 *
34 * Treasures include all items the hero can get in chest, buy in a shop,
35 * pick on the ground or get from scripts (e.g. when talking to an NPC).
36 *
37 * A treasure is represented as the following values:
38 * - the item name: a string identifying the nature of the treasure,
39 * - the variant: an integer indicating the variant of this item (some items
40 * may have several variants),
41 * - a savegame variable: name of the saved boolean variable that indicates
42 * whether the player has found this treasure (or an empty string if the treasure
43 * is not saved).
44 */
45class Treasure {
46
47 public:
48
49 Treasure(Game& game, const std::string& item_name, int variant,
50 const std::string& savegame_variable);
51
52 Game& get_game() const;
53 EquipmentItem& get_item() const;
54 const std::string& get_item_name() const;
55 int get_variant() const;
56 const std::string& get_savegame_variable() const;
57 bool is_saved() const;
58 bool is_empty() const;
59
60 bool is_obtainable() const;
61 void ensure_obtainable();
62 void check_obtainable() const;
63 void give_to_player() const;
64 bool is_found() const;
65
66 SpritePtr create_sprite() const;
67
68 private:
69
70 Game* game; /**< the current game */
71 std::string item_name; /**< content of the treasure (or an empty string) */
72 int variant; /**< variant of this content */
73 std::string savegame_variable; /**< name of the savegame boolean variable corresponding to this treasure,
74 * or an empty string if the treasure state is not saved */
75
76};
77
78}
79
80#endif
