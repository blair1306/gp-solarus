18namespace Solarus {

20////////////////////////////////////////////////////////////
21// Size methods
22
23/**
24 * \brief Creates the size <tt>(0,0)</tt>.
25 */
26constexpr Size::Size()
27 = default;
28
29/**
30 * \brief Creates a size with the specified dimensions.
31 * \param width The width.
32 * \param height The height.
33 */
34constexpr Size::Size(int width, int height):
35 width(width),
36 height(height) {
37}
38
39/**
40 * \brief Adds another size to this one.
41 * \param other Another size.
42 * \return The current size after modification.
43 */
44inline Size& Size::operator+=(const Size& other) {
45 width += other.width;
46 height += other.height;
47 return *this;
48}
49
50/**
51 * \brief Subtracts another size from this one.
52 * \param other The size to subtract.
53 * \return The current size after modification.
54 */
55inline Size& Size::operator-=(const Size& other) {
56 width -= other.width;
57 height -= other.height;
58 return *this;
59}
60
61/**
62 * \brief Multiplies this size by a value.
63 * \param factor The multiplication factor.
64 * \return The current size after modification.
65 */
66inline Size& Size::operator*=(int factor) {
67 width *= factor;
68 height *= factor;
69 return *this;
70}
71
72/**
73 * \brief Divides this size by a value.
74 * \param divisor The divisor.
75 * \return The current size after modification.
76 */
77inline Size& Size::operator/=(int divisor) {
78 width /= divisor;
79 height /= divisor;
80 return *this;
81}
82
83/**
84 * \brief Returns whether this size is flat.
85 * \return \c true if the width or the height are zero.
86 */
87constexpr bool Size::is_flat() const {
88 return width == 0 || height == 0;
89}
90
91/**
92 * \brief Returns whether this width and height are the same.
93 * \return \c true if the width and height are equal.
94 */
95constexpr bool Size::is_square() const {
96 return width == height;
97}
98
99////////////////////////////////////////////////////////////
100// Free functions
101
102/**
103 * \brief Returns whether two sizes are equal.
104 * \param lhs A size.
105 * \param rhs Another size.
106 * \return \c true if the sizes are equal.
107 */
108constexpr bool operator==(const Size& lhs, const Size& rhs) {
109 return lhs.width == rhs.width && lhs.height == rhs.height;
110}
111
112/**
113 * \brief Returns whether two sizes are different.
114 * \param lhs A size.
115 * \param rhs Another size.
116 * \return \c true if the sizes are not equal.
117 */
118constexpr bool operator!=(const Size& lhs, const Size& rhs) {
119 return !(lhs == rhs);
120}
121
122/**
123 * \brief Adds two sizes.
124 * \param lhs A size.
125 * \param rhs Another size.
126 * \return The sum of both sizes.
127 */
128constexpr Size operator+(const Size& lhs, const Size& rhs) {
129 return { lhs.width + rhs.width, lhs.height + rhs.height };
130}
131
132/**
133 * \brief Substracts two sizes.
134 * \param lhs A size.
135 * \param rhs The size to subtract.
136 * \return The difference of both sizes.
137 */
138constexpr Size operator-(const Size& lhs, const Size& rhs) {
139 return { lhs.width - rhs.width, lhs.height - rhs.height };
140}
141
142/**
143 * \brief Multiplies a size by a value.
144 * \param size A size.
145 * \param factor The multiplication factor.
146 * \return A size with width and height multiplied by the factor.
147 */
148constexpr Size operator*(const Size& size, int factor) {
149 return { size.width * factor, size.height * factor };
150}
151
152/**
153 * \brief Multiplies a size by a value.
154 * \param factor The multiplication factor.
155 * \param size A size.
156 * \return A size with width and height multiplied by the factor.
157 */
158constexpr Size operator*(int factor, const Size& size) {
159 return size * factor;
160}
161
162/**
163 * \brief Divides a size by a value.
164 * \param size A size.
165 * \param divisor The divisor.
166 * \return A size with width and height divided by the divisor.
167 */
168constexpr Size operator/(const Size& size, int divisor) {
169 return { size.width / divisor, size.height / divisor };
170}
