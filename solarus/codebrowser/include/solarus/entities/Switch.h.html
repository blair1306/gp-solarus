17#ifndef SOLARUS_SWITCH_H
18#define SOLARUS_SWITCH_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <map>
23#include <string>
24
25namespace Solarus {
26
27class Arrow;
28
29/**
30 * \brief A button that the hero can trigger.
31 *
32 * A switch can be triggered by walking onto it
33 * or by using weapons on it, depending on its subtype
34 * Some walkable switches require a block to be activated.
35 * Some walkable switches become disabled when the hero or the block leave it.
36 * A switch can have a sprite or be invisible.
37 */
38class Switch: public Entity {
39
40 public:
41
42 static constexpr EntityType ThisType = EntityType::SWITCH;
43
44 /**
45 * Subtypes of switches.
46 */
47 enum class Subtype {
48 WALKABLE, /**< A button the hero can walk on. */
49 ARROW_TARGET, /**< A switch that can be triggered by shooting an
50 * arrow on it. */
51 SOLID /**< A non-traversable switch that can be triggered
52 * with the sword or a projectile. */
53 };
54
55 Switch(
56 const std::string& name,
57 int layer,
58 const Point& xy,
59 Subtype subtype,
60 const std::string& sprite_name,
61 const std::string& sound_id,
62 bool needs_block,
63 bool inactivate_when_leaving
64 );
65
66 virtual EntityType get_type() const override;
67 virtual bool is_obstacle_for(Entity& other) override;
68
69 bool is_walkable() const;
70 bool is_arrow_target() const;
71 bool is_solid() const;
72 bool is_activated() const;
73 void activate();
74 void set_activated(bool enabled);
75 bool is_locked() const;
76 void set_locked(bool locked);
77
78 void try_activate(Hero& hero);
79 void try_activate(Block& block);
80 void try_activate(Arrow& arrow);
81 void try_activate();
82
83 virtual void update() override;
84 virtual bool test_collision_custom(Entity& entity) override;
85 virtual void notify_collision(
86 Entity& entity_overlapping,
87 CollisionMode collision_mode
88 ) override;
89 virtual void notify_collision(
90 Entity& other_entity,
91 Sprite& this_sprite,
92 Sprite& other_sprite
93 ) override;
94
95 static const std::map<Subtype, std::string> subtype_names; /**< Lua names of the Subtype enum. */
96
97 private:
98
99 Subtype subtype; /**< Subtype of switch. */
100 const std::string sound_id; /**< Sound to play when activating this switch. */
101 bool activated; /**< Whether this switch is currently on. */
102 bool locked; /**< Whether this switch cannot be activated or inactivated by other entities for now. */
103
104 // the following fields are used only for walkable switches
105 bool needs_block; /**< Whether a block or a statue is required to activate this walkable switch. */
106 bool inactivate_when_leaving; /**< Whether this walkable switch becomes inactivated when the hero or the block leaves it. */
107 Entity* entity_overlapping; /**< The entity currently on this walkable switch (as arrows may be destroyed at any moment). */
108 bool entity_overlapping_still_present; /**< To detect when the entity overlapping leaves the switch. */
109
110};
111
112}
113
114#endif
