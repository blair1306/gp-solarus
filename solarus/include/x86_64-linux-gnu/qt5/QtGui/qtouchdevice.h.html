42#ifndef QTOUCHDEVICE_H
43#define QTOUCHDEVICE_H
44
45#include <QtCore/qobject.h>
46
47QT_BEGIN_NAMESPACE
48
49
50class QTouchDevicePrivate;
51
52class Q_GUI_EXPORT QTouchDevice
53{
54public:
55 enum DeviceType {
56 TouchScreen,
57 TouchPad
58 };
59
60 enum CapabilityFlag {
61 Position = 0x0001,
62 Area = 0x0002,
63 Pressure = 0x0004,
64 Velocity = 0x0008,
65 RawPositions = 0x0010,
66 NormalizedPosition = 0x0020
67 };
68 Q_DECLARE_FLAGS(Capabilities, CapabilityFlag)
69
70 QTouchDevice();
71 ~QTouchDevice();
72
73 static QList<const QTouchDevice *> devices();
74
75 QString name() const;
76 DeviceType type() const;
77 Capabilities capabilities() const;
78 int maximumTouchPoints() const;
79
80 void setName(const QString &name);
81 void setType(DeviceType devType);
82 void setCapabilities(Capabilities caps);
83 void setMaximumTouchPoints(int max);
84
85private:
86 QTouchDevicePrivate *d;
87};
88
89Q_DECLARE_OPERATORS_FOR_FLAGS(QTouchDevice::Capabilities)
90
91QT_END_NAMESPACE
92
93#endif // QTOUCHDEVICE_H
