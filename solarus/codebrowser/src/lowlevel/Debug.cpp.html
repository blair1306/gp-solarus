18#include "solarus/lowlevel/Logger.h"

23namespace Solarus {
24namespace Debug {
25
26namespace {
27
28 bool die_on_error = false;
29 bool show_popup_on_die = true;
30 bool abort_on_die = false;
31
32}
33
34/**
35 * \brief Sets whether errors are fatal.
36 * \param die If \c true, calling Debug::error() will stop Solarus.
37 * The default is \c false.
38 */
39SOLARUS_API void set_die_on_error(bool die) {
40 die_on_error = die;
41}
42
43/**
44 * \brief Sets whether a dialog should pop when Solarus dies.
45 * \param show Whether a dialog should pop when Solarus dies.
46 * The default is \c true.
47 */
48SOLARUS_API void set_show_popup_on_die(bool show) {
49 show_popup_on_die = show;
50}
51
52/**
53 * \brief Sets whether the process should abort when Solarus dies.
54 *
55 * \param abort Whether std::abort should be called.
56 * If \c false, the main loop will stop and a SolarusFatal exception will
57 * be thrown.
58 * This should be preferred if Solarus is used as a library.
59 * The default is \c false.
60 */
61SOLARUS_API void set_abort_on_die(bool abort) {
62 abort_on_die = abort;
63}
64
65/**
66 * \brief Prints "Warning: " and a message on both stdout and error.txt.
67 * \param message The warning message to print.
68 */
69SOLARUS_API void warning(const std::string& message) {
70
71 Logger::warning(message);
72}
73
74/**
75 * \brief Prints "Error: " and a message on both stdout and error.txt.
76 *
77 * Use this function for non fatal errors such as errors in quest data files.
78 * Stops Solarus if set_die_on_error(true) was called.
79 *
80 * \param message The error message to print.
81 */
82SOLARUS_API void error(const std::string& message) {
83
84 if (die_on_error) {
85 // Errors are fatal.
86 die(message);
87 }
88
89 Logger::error(message);
90}
91
92/**
93 * \brief Like check_assertion(bool, const std::string&), but avoids
94 * a useless conversion to std::string when the assertion is true.
95 */
96void SOLARUS_API check_assertion(bool assertion, const char* error_message) {
97
98 // Don't build the std::string when the assertion succeeds.
99 if (!assertion) {
100 die(error_message);
101 }
102}
103
104/**
105 * \brief Stops Solarus if the specified assertion is \c false.
106 * \param assertion The assertion to check.
107 * \param error_message Error message to show in case of failure.
108 */
109void SOLARUS_API check_assertion(bool assertion, const std::string& error_message) {
110
111 if (!assertion) {
112 die(error_message);
113 }
114}
115
116/**
117 * \brief Stops Solarus on a fatal error.
118 *
119 * This function is equivalent to check_assertion(false, error_message).
120 * The error message is printed on both stdout and error.txt.
121 *
122 * \param error_message The error message to report.
123 */
124void SOLARUS_API die(const std::string& error_message) {
125
126 Logger::fatal(error_message);
127
128 if (show_popup_on_die) {
129 SDL_ShowSimpleMessageBox(
130 SDL_MESSAGEBOX_ERROR,
131 "Error",
132 error_message.c_str(),
133 nullptr
134 );
135 }
136
137 if (abort_on_die) {
138 std::abort();
139 }
140 else {
141 throw SolarusFatal(error_message);
142 }
143}
