17#ifndef SOLARUS_DOOR_H
18#define SOLARUS_DOOR_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <map>
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief A door that may be open or closed.
29 *
30 * It can be opened by one of the following methods depending on your choice:
31 * - by pressing the action key,
32 * - by pressing the action key when a specific savegame variable is set,
33 * - by pressing the action key when the player has a specific equipment item,
34 * - with an explosion,
35 * - or only explicitly by a Lua script.
36 *
37 * The state of a door can be saved.
38 * However, transitional states (opening and closing) are not saved:
39 * opening is saved as open and closing is saved as closed.
40 *
41 * TODO For opening method OPENING_BY_INTERACTION_IF_ITEM, allow to interact
42 * with the item command.
43 */
44class Door: public Entity {
45
46 public:
47
48 static constexpr EntityType ThisType = EntityType::DOOR;
49
50 /**
51 * \brief The different possible ways of opening a door.
52 *
53 * Note that any kind door can always be opened manually from Lua.
54 */
55 enum class OpeningMethod {
56 NONE, /**< Can only be opened from Lua. */
57 BY_INTERACTION, /**< Can be opened by pressing the action command. */
58 BY_INTERACTION_IF_SAVEGAME_VARIABLE, /**< Can be opened by pressing the action command, provided that a savegame variable has the correct value. */
59 BY_INTERACTION_IF_ITEM, /**< Can be opened by pressing the action command, provided that the player has a specific equipment item. */
60 BY_EXPLOSION /**< Can be opened by an explosion. */
61 };
62
63 Door(Game& game,
64 const std::string& name,
65 int layer,
66 const Point& xy,
67 int direction,
68 const std::string& sprite_name,
69 const std::string& savegame_variable
70 );
71
72 virtual EntityType get_type() const override;
73
74 virtual bool is_obstacle_for(Entity& other) override;
75 virtual void set_suspended(bool suspended) override;
76 virtual void update() override;
77 virtual void draw_on_map() override;
78 virtual std::string get_sword_tapping_sound() override;
79 virtual bool notify_action_command_pressed() override;
80 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
81 virtual void notify_collision(Entity& entity, Sprite& this_sprite, Sprite& other_sprite) override;
82 virtual void notify_collision_with_explosion(Explosion& explosion, Sprite& sprite_overlapping) override;
83
84 // Properties.
85 bool is_saved() const;
86 const std::string& get_savegame_variable() const;
87 OpeningMethod get_opening_method() const;
88 void set_opening_method(OpeningMethod opening_method);
89 bool is_interaction_required() const;
90 const std::string& get_opening_condition() const;
91 void set_opening_condition(const std::string& opening_condition);
92 bool is_opening_condition_consumed() const;
93 void set_opening_condition_consumed(bool opening_condition_consumed);
94 const std::string& get_cannot_open_dialog_id() const;
95 void set_cannot_open_dialog_id(const std::string& cannot_open_dialog_id);
96
97 // State.
98 bool can_open() const;
99 bool is_open() const;
100 bool is_opening() const;
101 bool is_closed() const;
102 bool is_closing() const;
103 bool is_changing() const;
104 void open();
105 void close();
106 void set_open(bool open);
107
108 static const std::map<OpeningMethod, std::string> opening_method_names;
109
110 private:
111
112 /**
113 * Possible states of a door.
114 */
115 enum State {
116 OPEN,
117 OPENING,
118 CLOSED,
119 CLOSING
120 };
121
122 void set_opening();
123 void set_closing();
124 void update_dynamic_tiles();
125 void consume_opening_condition();
126
127 // Properties.
128 const std::string savegame_variable; /**< Boolean variable that saves the door state. */
129 OpeningMethod opening_method; /**< How this door can be opened. */
130 std::string opening_condition; /**< Condition required to open the door: a savegame variable if
131 * the opening mode is \c OPENING_BY_INTERACTION_IF_SAVEGAME_VARIABLE,
132 * or an equipment item name if the opening mode is
133 * \c OPENING_BY_INTERACTION_IF_ITEM. */
134 bool opening_condition_consumed; /**< Indicates that the required savegame variable
135 * (in the case of \c OPENING_BY_INTERACTION_IF_SAVEGAME_VARIABLE)
136 * or the required item
137 * (in the case of \c OPENING_BY_INTERACTION_IF_ITEM)
138 * should be consumed when opening the door. */
139 std::string cannot_open_dialog_id; /**< Dialog to show if the door cannot be opened,
140 * or an empty string. */
141
142 // State.
143 State state; /**< State of the door: open, opening, closed or closing. */
144 bool initialized; /**< \c true if update() was called at least once. */
145 uint32_t next_hint_sound_date; /**< If the player has the ability to detect weak walls,
146 * indicates when a hint sound is played next time. */
147
148};
149
150}
151
152#endif
