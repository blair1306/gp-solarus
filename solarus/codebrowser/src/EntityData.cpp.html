274/**
275 * \brief Implementation of all entity data creation functions.
276 * \param l The Lua context that is calling this function.
277 * \return Number of values to return to Lua.
278 */
279int l_create_entity_data(lua_State* l) {
280
281 return LuaTools::exception_boundary_handle(l, [&] {
282
283 // Retrieve the entity data instance to build.
284 lua_getfield(l, LUA_REGISTRYINDEX, "entity");
285 EntityData& entity = *static_cast<EntityData*>(lua_touserdata(l, -1));
286 lua_pop(l, 1);
287
288 // Get the type of entity, it should have been passed as an upvalue.
289 EntityType type = LuaTools::check_enum<EntityType>(
290 l, lua_upvalueindex(1)
291 );
292
293 entity = EntityData::check_entity_data(l, 1, type);
294
295 return 0;
296 });
297}

358/**
359 * \brief Returns whether two values are equal.
360 */
361bool FieldValue::operator==(const FieldValue& other) const {
362
363 return value_type == other.value_type &&
364 string_value == other.string_value &&
365 int_value == other.int_value;
366}

371bool FieldValue::operator!=(const FieldValue& other) const {
372
373 return !(*this == other);

688/**
689 * \brief Returns whether a field is optional.
690 * \param key Key of the field to check.
691 * \return \c true if the field exists and is optional.
692 */
693bool EntityData::is_field_optional(const std::string& key) const {
694
695 const EntityTypeDescription& type_description = entity_type_descriptions.at(type);
696 for (const EntityFieldDescription& field_description : type_description) {
697 if (field_description.key == key) {
698 return field_description.optional == OptionalFlag::OPTIONAL;
699 }
700 }
701
702 return false;
703}
704
705/**
706 * \brief Returns whether a field is unset.
707 *
708 * A field can only be unset if it is optional.
709 *
710 * \param key Key of the field to check.
711 * \return \c true if the field exists, is optional and is equal to its
712 * default value.
713 */
714bool EntityData::is_field_unset(const std::string& key) const {
715
716 const EntityTypeDescription& type_description = entity_type_descriptions.at(type);
717 for (const EntityFieldDescription& field_description : type_description) {
718 if (field_description.key != key) {
719 continue;
720 }
721
722 if (field_description.optional != OptionalFlag::OPTIONAL) {
723 // Mandatory field: always set.
724 return false;
725 }
726
727 const FieldValue& default_value = field_description.default_value;
728 switch (default_value.value_type) {
729
730 case EntityFieldType::STRING:
731 return get_string(key) == default_value.string_value;
732
733 case EntityFieldType::INTEGER:
734 return get_integer(key) == default_value.int_value;
735
736 case EntityFieldType::BOOLEAN:
737 return get_boolean(key) == (default_value.int_value != 0);
738
739 case EntityFieldType::NIL:
740 Debug::die("Nil entity field");
741 break;
742 }
743 }
744
745 // The field does not exist.
746 return false;
747}

749/**
750 * \brief Reads an EntityData from the Lua table at the specified index.
751 * \param l A Lua state.
752 * \param index Index of the table in the Lua stack.
753 * \param type Type of entity to read.
754 * \return The EntityData created.
755 */
756EntityData EntityData::check_entity_data(lua_State* l, int index, EntityType type) {
757
758 LuaTools::check_type(l, index, LUA_TTABLE);
759 const std::string& name = LuaTools::opt_string_field(l, index, "name", "");
760 int layer = LuaTools::check_int_field(l, index, "layer");
761 int x = LuaTools::check_int_field(l, index, "x");
762 int y = LuaTools::check_int_field(l, index, "y");
763
764 EntityData entity(type);
765 entity.set_name(name);
766 entity.set_layer(layer);
767 entity.set_xy({ x, y });
768
769 // Additional fields for this type.
770 const EntityTypeDescription& type_description = entity_type_descriptions.at(type);
771 for (const EntityFieldDescription& field_description : type_description) {
772
773 const std::string& key = field_description.key;
774 const FieldValue& default_value = field_description.default_value;
775 switch (default_value.value_type) {
776
777 case EntityFieldType::STRING:
778 {
779 std::string value;
780 if (field_description.optional == OptionalFlag::OPTIONAL) {
781 value = LuaTools::opt_string_field(l, index, key, default_value.string_value);
782 }
783 else {
784 value = LuaTools::check_string_field(l, index, key);
785 }
786 entity.set_string(key, value);
787 break;
788 }
789
790 case EntityFieldType::INTEGER:
791 {
792 int value;
793 if (field_description.optional == OptionalFlag::OPTIONAL) {
794 value = LuaTools::opt_int_field(l, index, key, default_value.int_value);
795 }
796 else {
797 value = LuaTools::check_int_field(l, index, key);
798 }
799 entity.set_integer(key, value);
800 break;
801 }
802
803 case EntityFieldType::BOOLEAN:
804 {
805 bool value;
806 if (field_description.optional == OptionalFlag::OPTIONAL) {
807 value = LuaTools::opt_boolean_field(l, index, key, default_value.int_value != 0);
808 }
809 else {
810 value = LuaTools::check_boolean_field(l, index, key);
811 }
812 entity.set_boolean(key, value);
813 break;
814 }
815
816 case EntityFieldType::NIL:
817 Debug::die("Nil entity field");
818 break;
819 }
820 }
821
822 return entity;
823}

828 * Note: entities whose type is not in this map can only be created

840/**
841 * \copydoc LuaData::import_from_lua
842 */
843bool EntityData::import_from_lua(lua_State* l) {
844
845 lua_pushlightuserdata(l, this);
846 lua_setfield(l, LUA_REGISTRYINDEX, "entity");
847 for (const auto& kvp : EnumInfoTraits<EntityType>::names) {
848 EntityType type = kvp.first;
849 const std::string& type_name = kvp.second;
850 if (!EntityTypeInfo::can_be_stored_in_map_file(type)) {
851 continue;
852 }
853 std::string function_name = type_name;
854 lua_pushstring(l, function_name.c_str());
855 lua_pushcclosure(l, l_create_entity_data, 1);
856 lua_setglobal(l, function_name.c_str());
857 }
858
859 if (lua_pcall(l, 0, 0, 0) != 0) {
860 Debug::error(std::string("Failed to load entity: ") + lua_tostring(l, -1));
861 lua_pop(l, 1);
862 return false;
863 }
864
865 return true;
866}

868/**
869 * \copydoc LuaData::export_to_lua
870 */
871bool EntityData::export_to_lua(std::ostream& out) const {
872
873 // Entity type.
874 out << get_type_name() << "{\n";
875
876 // Name.
877 if (has_name()) {
878 out << " name = \"" << escape_string(get_name()) << "\",\n";
879 }
880
881 // Position on the map.
882 out << " layer = " << get_layer() << ",\n"
883 << " x = " << get_xy().x << ",\n"
884 << " y = " << get_xy().y << ",\n";
885
886 // Properties specific to a type of entity.
887 const EntityTypeDescription& type_description = entity_type_descriptions.at(get_type());
888
889 for (const EntityFieldDescription& field_description : type_description) {
890
891 const std::string& key = field_description.key;
892 const bool optional = field_description.optional == OptionalFlag::OPTIONAL;
893 switch (field_description.default_value.value_type) {
894
895 case EntityFieldType::STRING:
896 {
897 const std::string& value = get_string(key);
898 const std::string& default_value = field_description.default_value.string_value;
899 if (optional && value == default_value) {
900 // No need to write the value.
901 continue;
902 }
903 out << " " << key << " = \"" << escape_string(value) << "\",\n";
904 break;
905 }
906
907 case EntityFieldType::INTEGER:
908 {
909 int value = get_integer(key);
910 int default_value = field_description.default_value.int_value;
911 if (optional && value == default_value) {
912 // No need to write the value.
913 continue;
914 }
915 out << " " << key << " = " << value << ",\n";
916 break;
917 }
918
919 case EntityFieldType::BOOLEAN:
920 {
921 bool value = get_boolean(key);
922 bool default_value = field_description.default_value.int_value != 0;
923 if (optional && value == default_value) {
924 // No need to write the value.
925 continue;
926 }
927 out << " " << key << " = " << (value ? "true" : "false") << ",\n";
928
929 break;
930 }
931
932 case EntityFieldType::NIL:
933 Debug::die("Nil entity field");
934 break;
935 }
936 }
937 out << "}\n\n";
938
939 return true;
940}
