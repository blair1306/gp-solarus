85 // creation and destruction
86 Savegame(MainLoop& main_loop, const std::string& file_name);

88 // file state
89 bool is_empty() const;
90 void initialize();

93
94 // data
95 bool is_string(const std::string& key) const;
96 std::string get_string(const std::string& key) const;
97 void set_string(const std::string& key, const std::string& value);
98 bool is_integer(const std::string& key) const;
99 int get_integer(const std::string& key) const;
100 void set_integer(const std::string& key, int value);
101 bool is_boolean(const std::string& key) const;
102 bool get_boolean(const std::string& key) const;
103 void set_boolean(const std::string& key, bool value);
104 void unset(const std::string& key);
105
106 void set_initial_values();
107 void set_default_keyboard_controls();
108 void set_default_joypad_controls();
109
110 // unsaved data
111 MainLoop& get_main_loop();
112 LuaContext& get_lua_context();
113 Equipment& get_equipment();
114 const Equipment& get_equipment() const;
115 Game* get_game();
116 void set_game(Game* game);
117 void notify_game_started();
118 void notify_game_finished();
119
120 virtual const std::string& get_lua_type_name() const override;
121
122 private:
123
124 struct SavedValue {
125
126 enum {
127 VALUE_STRING,
128 VALUE_INTEGER,
129 VALUE_BOOLEAN
130 } type;
131
132 std::string string_data;
133 int int_data; // Also used for boolean
134 };
135
136 std::map<std::string, SavedValue> saved_values;
137
138 bool empty;
139 std::string file_name; /**< Savegame file name relative to the quest write directory. */
140 MainLoop& main_loop;
141 Equipment equipment;
142 Game* game; /**< nullptr if this savegame is not currently running */
143
144 void import_from_file();
145 static int l_newindex(lua_State* l);
146
147};
