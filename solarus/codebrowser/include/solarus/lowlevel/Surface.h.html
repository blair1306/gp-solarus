17#ifndef SOLARUS_SURFACE_H
18#define SOLARUS_SURFACE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/PixelBits.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/Drawable.h"
24#include <SDL.h>
25#include <SDL_image.h>
26#include <cstdint>
27#include <memory>
28#include <string>
29#include <vector>
30
31namespace Solarus {
32
33class Color;
34class PixelFilter;
35class Size;
36class Surface;
37
38/**
39 * \brief Represents a graphic surface.
40 *
41 * A surface is a rectangle of pixels.
42 * A surface can be drawn or blitted on another surface.
43 * This class basically encapsulates a library-dependent surface object.
44 */
45class Surface: public Drawable {
46
47 // low-level classes allowed to manipulate directly the internal SDL surface encapsulated
48 friend class TextSurface;
49 friend class PixelBits;
50
51 public:
52
53 /**
54 * \brief The base directory to use when opening image files.
55 */
56 enum ImageDirectory {
57 DIR_DATA, /**< the root directory of the data package */
58 DIR_SPRITES, /**< the sprites subdirectory of the data package (default) */
59 DIR_LANGUAGE /**< the language-specific image directory of the data package, for the current language */
60 };
61
62 Surface(int width, int height);
63 explicit Surface(SDL_Surface* internal_surface);
64
65 // Surfaces should only created with std::make_shared.
66 // This is what create() functions do, so you should call them rather than
67 // constructors.
68 // This is because they are always reference-counted with shared_ptr
69 // internally for drawing.
70 static SurfacePtr create(int width, int height);
71 static SurfacePtr create(const Size& size);
72 static SurfacePtr create(const std::string& file_name,
73 ImageDirectory base_directory = DIR_SPRITES);
74
75 int get_width() const;
76 int get_height() const;
77 virtual Size get_size() const override;
78
79 bool is_software_destination() const;
80 void set_software_destination(bool software_destination);
81
82 void clear();
83 void clear(const Rectangle& where);
84 void fill_with_color(const Color& color);
85 void fill_with_color(const Color& color, const Rectangle& where);
86 uint8_t get_opacity() const;
87 void set_opacity(uint8_t opacity);
88
89 std::string get_pixels() const;
90
91 void apply_pixel_filter(const PixelFilter& pixel_filter, Surface& dst_surface);
92
93 void render(SDL_Renderer* renderer);
94
95 virtual const std::string& get_lua_type_name() const override;
96
97 protected:
98
99 // Implementation from Drawable.
100 virtual void raw_draw(
101 Surface& dst_surface,
102 const Point& dst_position
103 ) override;
104 virtual void raw_draw_region(
105 const Rectangle& region,
106 Surface& dst_surface,
107 const Point& dst_position
108 ) override;
109 virtual void draw_transition(Transition& transition) override;
110 virtual Surface& get_transition_surface() override;
111
112 private:
113
114 class SubSurfaceNode;
115 using SubSurfaceNodePtr = std::shared_ptr<Surface::SubSurfaceNode>;
116
117 struct SDL_Surface_Deleter {
118 void operator()(SDL_Surface* sdl_surface) {
119 SDL_FreeSurface(sdl_surface);
120 }
121 };
122 using SDL_Surface_UniquePtr = std::unique_ptr<SDL_Surface, SDL_Surface_Deleter>;
123
124 struct SDL_Texture_Deleter {
125 void operator()(SDL_Texture* sdl_texture) {
126 SDL_DestroyTexture(sdl_texture);
127 }
128 };
129 using SDL_Texture_UniquePtr = std::unique_ptr<SDL_Texture, SDL_Texture_Deleter>;
130
131 uint32_t get_pixel(int index) const;
132 bool is_pixel_transparent(int index) const;
133 uint32_t get_color_value(const Color& color) const;
134 SDL_BlendMode get_sdl_blend_mode() const;
135
136 static SDL_Surface* get_surface_from_file(
137 const std::string& file_name,
138 ImageDirectory base_directory);
139
140 void create_software_surface();
141 void convert_software_surface();
142 void create_texture_from_surface();
143 void add_subsurface(const SurfacePtr& src_surface, const Rectangle& region, const Point& dst_position);
144 void clear_subsurfaces();
145 void render(
146 SDL_Renderer* renderer,
147 const Rectangle& src_rect,
148 const Rectangle& dst_rect,
149 const Rectangle& clip_rect,
150 uint8_t opacity,
151 const std::vector<SubSurfaceNodePtr>& subsurfaces
152 );
153
154 std::vector<SubSurfaceNodePtr>
155 subsurfaces; /**< Source Subsurfaces not in the tree yet */
156
157 bool software_destination; /**< Whether this surface should be modified on software side
158 * (and therefore immediately) when used as a destination */
159 SDL_Surface_UniquePtr
160 internal_surface; /**< The SDL_Surface encapsulated, if any. */
161 SDL_Texture_UniquePtr
162 internal_texture; /**< The SDL_Texture encapsulated, if any. */
163 std::unique_ptr<Color>
164 internal_color; /**< The background color to use, if any. */
165 bool is_rendered; /**< Whether the current surface has been rendered.
166 * Set to false when drawing a surface on this one. */
167 uint8_t opacity; /**< Opacity (0: transparent, 255: opaque). */
168 int width, height; /**< Size of the texture, avoid to use SDL_QueryTexture. */
169
170};
171
172}
173
174#endif
