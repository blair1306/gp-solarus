179/**
180 * \brief Returns all frames of this direction.
181 * \return All frames.
182 */
183std::vector<Rectangle> SpriteAnimationDirectionData::get_all_frames() const {
184
185 std::vector<Rectangle> frames;
186 int frame_number = 0;
187
188 int num_rows = num_frames / num_columns;
189 if (num_frames % num_columns != 0) {
190 num_rows++;
191 }
192
193 for (int row = 0; row < num_rows && frame_number < num_frames; ++row) {
194 for (int col = 0; col < num_columns && frame_number < num_frames; ++col) {
195
196 frames.emplace_back(
197 xy.x + col * size.width,
198 xy.y + row * size.height,
199 size.width,
200 size.height
201 );
202 ++frame_number;
203 }
204 }
205
206 return frames;
207}

362/**
363 * \brief Removes a direction from this animation.
364 * \param direction_nb Number of the direction to remove.
365 * \return \c true in case of success.
366 */
367bool SpriteAnimationData::remove_direction(int direction_nb) {
368
369 int size = directions.size();
370 if (direction_nb >= size) {
371 return false;
372 }
373
374 directions.erase(directions.begin() + direction_nb);
375 return true;
376}
377
378/**
379 * @brief Moves a direction of this animation to a specified position.
380 * @param direction_nb Number of direction to move.
381 * @param new_direction_nb The new position of the direction.
382 * @return \c true in case of success.
383 */
384bool SpriteAnimationData::move_direction(int direction_nb, int new_direction_nb) {
385
386 int size = directions.size();
387
388 if (direction_nb < 0 || direction_nb >= size) {
389 return false;
390 }
391
392 new_direction_nb = std::min(std::max(new_direction_nb, 0), size - 1);
393
394 if (direction_nb == new_direction_nb) {
395 return true;
396 }
397
398 // insert to new position
399 auto new_pos = directions.begin() + new_direction_nb;
400 if (direction_nb < new_direction_nb) {
401 new_pos++;
402 }
403 directions.insert(new_pos, directions[direction_nb]);
404
405 // remove from last position
406 auto last_pos = directions.begin() + direction_nb;
407 if (direction_nb > new_direction_nb) {
408 last_pos++;
409 }
410 directions.erase(last_pos);
411
412 return true;
413}

482/**
483 * \brief Adds an animation to this sprite.
484 * \param animation_name Name of the animation to add.
485 * \param animation The animation to add.
486 * \return \c true in case of success.
487 */
488bool SpriteData::add_animation(
489 const std::string& animation_name, const SpriteAnimationData& animation) {
490
491 const auto& result = animations.emplace(animation_name, animation);
492 if (!result.second) {
493 // Insertion failed: the name already exists.
494 return false;
495 }
496
497 if (animations.size() == 1) {
498 default_animation_name = animation_name;
499 }
500
501 return true;
502}
503
504/**
505 * \brief Removes an animation from this sprite.
506 * \param animation_name Name of the animation to remove.
507 * \return \c true in case of success.
508 */
509bool SpriteData::remove_animation(const std::string& animation_name) {
510
511 bool removed = animations.erase(animation_name) > 0;
512 if (removed && default_animation_name == animation_name) {
513 if (animations.size() > 0) {
514 const auto& it = animations.cbegin();
515 default_animation_name = it->first;
516 } else {
517 default_animation_name = "";
518 }
519 }
520 return removed;
521}

523/**
524 * \brief Changes the name of an animation in the sprite.
525 * \param old_animation_name Old name of the animation.
526 * \param new_animation_name New name to set.
527 * \return \c true in case of success.
528 * In case of failure, the old animation is unchanged.
529 */
530bool SpriteData::set_animation_name(
531 const std::string& old_animation_name,
532 const std::string& new_animation_name) {
533
534 if (!has_animation(old_animation_name)) {
535 // No animation was found with the old name.
536 return false;
537 }
538
539 if (has_animation(new_animation_name)) {
540 // The new name is already used.
541 return false;
542 }
543
544 if (default_animation_name == old_animation_name) {
545 default_animation_name = new_animation_name;
546 }
547
548 SpriteAnimationData animation = get_animation(old_animation_name);
549 remove_animation(old_animation_name);
550 add_animation(new_animation_name, animation);
551
552 return true;
553}

579/**
580 * \brief Function called by the Lua data file to define an animation.
581 *
582 * - Argument 1 (table): properties of the animation.
583 *
584 * \param l the Lua context that is calling this function
585 * \return Number of values to return to Lua.
586 */
587int SpriteData::l_animation(lua_State* l) {
588
589 return LuaTools::exception_boundary_handle(l, [&] {
590 lua_getfield(l, LUA_REGISTRYINDEX, "sprite");
591 SpriteData* sprite = static_cast<SpriteData*>(
592 lua_touserdata(l, -1));
593 lua_pop(l, 1);
594
595 LuaTools::check_type(l, 1, LUA_TTABLE);
596
597 std::string animation_name = LuaTools::check_string_field(l, 1, "name");
598 std::string src_image = LuaTools::check_string_field(l, 1, "src_image");
599 uint32_t frame_delay = (uint32_t) LuaTools::opt_int_field(l, 1, "frame_delay", 0);
600 int frame_to_loop_on = LuaTools::opt_int_field(l, 1, "frame_to_loop_on", -1);
601
602 if (frame_to_loop_on < -1) {
603 LuaTools::arg_error(l, 1,
604 "Bad field 'frame_to_loop_on' (must be a positive number or -1)"
605 );
606 }
607
608 lua_settop(l, 1);
609 lua_getfield(l, 1, "directions");
610 if (lua_type(l, 2) != LUA_TTABLE) {
611 LuaTools::arg_error(l, 1,
612 std::string("Bad field 'directions' (table expected, got ")
613 + luaL_typename(l, -1) + ")");
614 }
615
616 // Traverse the directions table.
617 std::deque<SpriteAnimationDirectionData> directions;
618 int i = 1;
619 lua_rawgeti(l, -1, i);
620 while (!lua_isnil(l, -1)) {
621 ++i;
622
623 if (lua_type(l, -1) != LUA_TTABLE) {
624 LuaTools::arg_error(l, 1,
625 std::string("Bad field 'directions' (expected table, got ")
626 + luaL_typename(l, -1)
627 );
628 }
629
630 int x = LuaTools::check_int_field(l, -1, "x");
631 int y = LuaTools::check_int_field(l, -1, "y");
632 int frame_width = LuaTools::check_int_field(l, -1, "frame_width");
633 int frame_height = LuaTools::check_int_field(l, -1, "frame_height");
634 int origin_x = LuaTools::opt_int_field(l, -1, "origin_x", 0);
635 int origin_y = LuaTools::opt_int_field(l, -1, "origin_y", 0);
636 int num_frames = LuaTools::opt_int_field(l, -1, "num_frames", 1);
637 int num_columns = LuaTools::opt_int_field(l, -1, "num_columns", num_frames);
638
639 if (num_columns < 1 || num_columns > num_frames) {
640 LuaTools::arg_error(l, 1,
641 "Bad field 'num_columns': must be between 1 and the number of frames");
642 }
643
644 if (frame_to_loop_on >= num_frames) {
645 LuaTools::arg_error(l, 1,
646 "Bad field 'frame_to_loop_on': exceeds the number of frames");
647 }
648
649 lua_pop(l, 1);
650 lua_rawgeti(l, -1, i);
651
652 directions.emplace_back(
653 Point(x, y), Size(frame_width, frame_height),
654 Point(origin_x, origin_y), num_frames, num_columns);
655 }
656
657 if (sprite->animations.find(animation_name) != sprite->animations.end()) {
658 LuaTools::error(l, std::string("Duplicate animation '") + animation_name);
659 }
660
661 sprite->animations.emplace(animation_name,
662 SpriteAnimationData(src_image, directions, frame_delay, frame_to_loop_on)
663 );
664
665 // Set the first animation as the default one.
666 if (sprite->animations.size() == 1) {
667 sprite->default_animation_name = animation_name;
668 }
669
670 return 0;
671 });
672}

674/**
675 * \copydoc LuaData::import_from_lua
676 */
677bool SpriteData::import_from_lua(lua_State* l) {
678
679 lua_pushlightuserdata(l, this);
680 lua_setfield(l, LUA_REGISTRYINDEX, "sprite");
681 lua_register(l, "animation", l_animation);
682 if (lua_pcall(l, 0, 0, 0) != 0) {
683 Debug::error(std::string("Failed to load sprite: ") + lua_tostring(l, -1));
684 lua_pop(l, 1);
685 return false;
686 }
687
688 return true;
689}

691/**
692 * \copydoc LuaData::export_to_lua
693 */
694bool SpriteData::export_to_lua(std::ostream& out) const {
695
696 export_animation_to_lua(default_animation_name, out);
697
698 for (const auto& kvp : animations) {
699 const std::string& name = kvp.first;
700 if (name != default_animation_name) {
701 export_animation_to_lua(name, out);
702 }
703 }
704 return true;
705}

707/**
708 * \brief Saves an animation data as Lua into a stream.
709 * \param animation_name The name of animation to save.
710 * \param out The stream to write.
711 */
712void SpriteData::export_animation_to_lua(
713 const std::string& animation_name, std::ostream& out) const {
714
715 if (!has_animation(animation_name)) {
716 return;
717 }
718
719 SpriteAnimationData animation = animations.at(animation_name);
720
721 out << "animation{\n"
722 << " name = \"" << escape_string(animation_name) << "\",\n"
723 << " src_image = \"" << escape_string(animation.get_src_image()) << "\",\n";
724
725 if (animation.get_frame_delay() > 0) {
726 out << " frame_delay = " << animation.get_frame_delay() << ",\n";
727 if (animation.get_loop_on_frame() >= 0) {
728 out << " frame_to_loop_on = " << animation.get_loop_on_frame() << ",\n";
729 }
730 }
731
732 out << " directions = {\n";
733
734 for (const auto& direction: animation.get_directions()) {
735
736 Point xy = direction.get_xy();
737 Size size = direction.get_size();
738 Point origin = direction.get_origin();
739 int num_frames = direction.get_num_frames();
740 int num_columns = direction.get_num_columns();
741
742 out << " { "
743 << "x = " << xy.x << ", "
744 << "y = " << xy.y << ", "
745 << "frame_width = " << size.width << ", "
746 << "frame_height = " << size.height << ", "
747 << "origin_x = " << origin.x << ", "
748 << "origin_y = " << origin.y;
749
750 if (num_frames > 1) {
751 out << ", num_frames = " << num_frames;
752 if (num_columns > 0 && num_columns < num_frames) {
753 out << ", num_columns = " << num_columns;
754 }
755 }
756
757 out << " },\n";
758 }
759 out << " },\n";
760 out << "}\n";
761}
