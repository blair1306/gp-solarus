17#include "solarus/lua/LuaException.h"
18
19namespace Solarus {
20
21/**
22 * \brief Creates a Lua exception with the specified error message.
23 * \param l The Lua state where the error occurred.
24 * \param error_message The error message.
25 */
26LuaException::LuaException(
27 lua_State* l,
28 const std::string& error_message
29):
30 l(l),
31 error_message(error_message) {
32}
33
34/**
35 * \brief Returns The Lua context where the error occurred.
36 * \return The Lua context.
37 */
38lua_State* LuaException::get_lua_state() const {
39 return l;
40}
41
42/**
43 * \brief Returns the error message.
44 * \return The error message of the exception.
45 */
46const char* LuaException::what() const throw() {
47 return error_message.c_str();
48}
49
50}
