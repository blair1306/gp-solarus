74/**
75 * \copydoc Entity::State::notify_attack_command_pressed
76 */
77void HeroState::notify_attack_command_pressed() {
78 Hero& hero = get_entity();
79
80 if (!hero.is_suspended()
81 && get_commands_effects().get_sword_key_effect() == CommandsEffects::ATTACK_KEY_SWORD
82 && hero.can_start_sword()) {
83
84 hero.start_sword();
85 }
86}

88/**
89 * \brief Notifies this state that an item command was just pressed.
90 * \param slot The slot activated (1 or 2).
91 */
92void HeroState::notify_item_command_pressed(int slot) {
93 Hero& hero = get_entity();
94
95 EquipmentItem* item = get_equipment().get_item_assigned(slot);
96
97 if (item != nullptr && hero.can_start_item(*item)) {
98 hero.start_item(*item);
99 }
100}

102/**
103 * \brief Returns whether a jumper is considered as an obstacle in this state
104 * for the hero at the specified position.
105 * \param jumper A jumper.
106 * \param candidate_position Position of the hero to test.
107 * \return \c true if the jumper is an obstacle in this state with this
108 * hero position.
109 */
110bool HeroState::is_jumper_obstacle(
111 const Jumper& jumper, const Rectangle& candidate_position) const {
112 const Hero& hero = get_entity();
113
114 if (jumper.overlaps_jumping_region(hero.get_bounding_box(), false)) {
115 // The hero already overlaps the active part of the jumper.
116 // This is authorized if he arrived from another direction
117 // and thus did not activate it.
118 // This can be used to leave water pools for example.
119 return false;
120 }
121
122 if (!jumper.overlaps_jumping_region(candidate_position, false)) {
123 // The candidate position is in the inactive region: always accept that.
124 return false;
125 }
126
127 if (!can_take_jumper()) {
128 // If jumpers cannot be used in this state, consider their active region
129 // as obstacles and their inactive region as traversable.
130 // The active region should be an obstacle.
131 return true;
132 }
133
134 // At this point, we know that the jumper can be activated.
135
136 const bool hero_in_jump_position =
137 jumper.is_in_jump_position(hero, hero.get_bounding_box(), false);
138 const bool candidate_in_jump_position =
139 jumper.is_in_jump_position(hero, candidate_position, false);
140
141 if (candidate_in_jump_position) {
142 // Wants to move to a valid jump position: accept.
143 return false;
144 }
145
146 if (hero_in_jump_position) {
147 // If the hero is already correctly placed (ready to jump),
148 // make the jumper obstacle so that the player has to move in the
149 // jumper's direction during a small delay before jumping.
150 // This also prevents the hero to get inside the jumper's active region.
151 return true;
152 }
153
154 const bool candidate_in_extended_jump_position =
155 jumper.is_in_jump_position(hero, candidate_position, true);
156
157 if (candidate_in_extended_jump_position) {
158 // Wants to get inside the active region from an end of the jumper:
159 // don't accept this.
160 return true;
161 }
162
163 if (!jumper.is_jump_diagonal() &&
164 hero.is_moving_towards(jumper.get_direction() / 2)) {
165 // Special case: make the jumper traversable so
166 // that the smooth movement can slide to it.
167 return false;
168 }
169
170 if (!jumper.is_jump_diagonal() &&
171 get_name() == "swimming" && // TODO use inheritance instead
172 hero.is_moving_towards(((jumper.get_direction() / 2) + 2) % 4)
173 ) {
174 // Other special case: trying to enter the jumper the reverse way while
175 // swimming: we accept this to allow the hero to leave water pools.
176 // TODO I'm not sure if this behavior is really a good idea.
177 // This may change in a future version.
178 return false;
179 }
180
181 return true;
182}
