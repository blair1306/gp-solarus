124typedef struct luaL_Buffer {
125 char *p; /* current position in buffer */
126 int lvl; /* number of strings in the stack (level) */
127 lua_State *L;
128 char buffer[LUAL_BUFFERSIZE];
129} luaL_Buffer;
130
131#define luaL_addchar(B,c) \
132 ((void)((B)->p < ((B)->buffer+LUAL_BUFFERSIZE) || luaL_prepbuffer(B)), \
133 (*(B)->p++ = (char)(c)))
134
135/* compatibility only */
136#define luaL_putchar(B,c) luaL_addchar(B,c)
137
138#define luaL_addsize(B,n) ((B)->p += (n))
