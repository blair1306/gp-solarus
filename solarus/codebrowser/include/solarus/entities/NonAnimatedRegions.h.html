17#ifndef SOLARUS_NON_ANIMATED_REGIONS_H
18#define SOLARUS_NON_ANIMATED_REGIONS_H
19
20#include "solarus/Common.h"
21#include "solarus/containers/Grid.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/entities/TileInfo.h"
24#include <vector>
25
26namespace Solarus {
27
28class Map;
29
30/**
31 * \brief Manages the tiles that are in non-animated regions.
32 *
33 * A non-animated region is a rectangle of the map that contains no animated
34 * tile. The tiles in such rectangles of the map can be pre-drawn once for all
35 * on an intermediate surface for performance. Furthermore, this intermediate
36 * surface is drawn lazily when the camera moves.
37 */
38class NonAnimatedRegions {
39
40 public:
41
42 NonAnimatedRegions(Map& map, int layer);
43
44 void add_tile(const TileInfo& tile);
45 void build(std::vector<TileInfo>& rejected_tiles);
46 void notify_tileset_changed();
47 void draw_on_map();
48
49 private:
50
51 bool overlaps_animated_tile(const TileInfo& tile) const;
52 void build_cell(int cell_index);
53
54 Map& map; /**< The map. */
55 int layer; /**< Layer of the map managed by this object. */
56 std::vector<TileInfo>
57 tiles; /**< All tiles contained in this layer and candidates to
58 * be optimized. This list is cleared after build() is called. */
59 std::vector<bool> are_squares_animated; /**< Whether each 8x8 square of the map has animated tiles. */
60
61 // Handle the lazy drawing.
62 Grid<TileInfo>
63 non_animated_tiles; /**< All non-animated tiles. Stored in a grid so that
64 * we can quickly find the ones to draw lazily later when the
65 * camera moves. */
66 std::vector<SurfacePtr>
67 optimized_tiles_surfaces; /**< All non-animated tiles are drawn here once for all
68 * for performance. Each cell of the grid has a surface
69 * or nullptr before it is drawn. */
70
71};
72
73}
74
75#endif
