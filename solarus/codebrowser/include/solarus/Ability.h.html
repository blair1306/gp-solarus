17#ifndef SOLARUS_ABILITY_H
18#define SOLARUS_ABILITY_H
19
20namespace Solarus {
21
22/**
23 * \brief The different built-in abilities managed by the engine.
24 */
25enum class Ability {
26
27 TUNIC, /**< Tunic of the hero. Determines his sprite. */
28 SWORD, /**< Sword of the hero. 0 means no sword, a
29 * greater values determine the sword sprite. */
30 SWORD_KNOWLEDGE, /**< Allows to do the super spin attack. */
31 SHIELD, /**< Allows to stop attacks from enemies. */
32 LIFT, /**< Lifting destructible objects. */
33 SWIM, /**< Swimming in deep water. */
34 JUMP_OVER_WATER, /**< Automatically jumping when arriving in water
35 * without the SWIM ability. */
36 RUN, /**< Running faster. */
37 DETECT_WEAK_WALLS, /**< Be notified of weak walls nearby. */
38};
39
40}
41
42#endif
