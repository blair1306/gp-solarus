465 // force compile error, prevent QVariant(bool) to be called
466 inline QVariant(void *) Q_DECL_EQ_DELETE;
467 // QVariant::Type is marked as \obsolete, but we don't want to
468 // provide a constructor from its intended replacement,
469 // QMetaType::Type, instead, because the idea behind these
470 // constructors is flawed in the first place. But we also don't
471 // want QVariant(QMetaType::String) to compile and falsely be an
472 // int variant, so delete this constructor:
473 QVariant(QMetaType::Type) Q_DECL_EQ_DELETE;
474
475 // These constructors don't create QVariants of the type associcated
476 // with the enum, as expected, but they would create a QVariant of
477 // type int with the value of the enum value.
478 // Use QVariant v = QColor(Qt::red) instead of QVariant v = Qt::red for
479 // example.
480 QVariant(Qt::GlobalColor) Q_DECL_EQ_DELETE;
481 QVariant(Qt::BrushStyle) Q_DECL_EQ_DELETE;
482 QVariant(Qt::PenStyle) Q_DECL_EQ_DELETE;
483 QVariant(Qt::CursorShape) Q_DECL_EQ_DELETE;

503template <typename T>
504inline void qVariantSetValue(QVariant &v, const T &t)
505{
506 //if possible we reuse the current QVariant private
507 const uint type = qMetaTypeId<T>();
508 QVariant::Private &d = v.data_ptr();
509 if (v.isDetached() && (type == d.type || (type <= uint(QVariant::Char) && d.type <= uint(QVariant::Char)))) {
510 d.type = type;
511 d.is_null = false;
512 T *old = reinterpret_cast<T*>(d.is_shared ? d.data.shared->ptr : &d.data.ptr);
513 if (QTypeInfo<T>::isComplex)
514 old->~T();
515 new (old) T(t); //call the copy constructor
516 } else {
517 v = QVariant(type, &t, QTypeInfo<T>::isPointer);
518 }
519}
520
521template <>
522inline void qVariantSetValue<QVariant>(QVariant &v, const QVariant &t)
523{
524 v = t;
525}

677 template<typename T>
678 struct QVariantValueHelper : TreatAsQObjectBeforeMetaType<QVariantValueHelper<T>, T, const QVariant &, T>
679 {
680 static T metaType(const QVariant &v)
681 {
682 const int vid = qMetaTypeId<T>();
683 if (vid == v.userType())
684 return *reinterpret_cast<const T *>(v.constData());
685 T t;
686 if (v.convert(vid, &t))
687 return t;
688 return T();
689 }
690#ifndef QT_NO_QOBJECT
691 static T object(const QVariant &v)
692 {
693 return qobject_cast<T>(QMetaType::typeFlags(v.userType()) & QMetaType::PointerToQObject
694 ? v.d.data.o
695 : QVariantValueHelper::metaType(v));
696 }
697#endif
698 };

715 }

717 }

733 template<>
734 struct QVariantValueHelperInterface<QVariantList>
735 {
736 static QVariantList invoke(const QVariant &v)
