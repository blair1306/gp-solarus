38/**
39 * \brief Creates an arrow.
40 * \param hero The hero.
41 */
42Arrow::Arrow(const Hero& hero):
43 Entity("", 0, hero.get_layer(), Point(0, 0), Size(0, 0)),
44 hero(hero) {
45
46 // initialize the entity
47 int direction = hero.get_animation_direction();
48 const SpritePtr& sprite = create_sprite("entities/arrow");
49 sprite->enable_pixel_collisions();
50 sprite->set_current_direction(direction);
51 set_drawn_in_y_order(true);
52
53 if (direction % 2 == 0) {
54 // Horizontal.
55 set_size(16, 8);
56 set_origin(8, 4);
57 }
58 else {
59 // Vertical.
60 set_size(8, 16);
61 set_origin(4, 8);
62 }
63
64 set_xy(hero.get_center_point());
65 notify_position_changed();
66
67 std::string path = " ";
68 path[0] = '0' + (direction * 2);
69 set_movement(std::make_shared<PathMovement>(
70 path, 192, true, false, false
71 ));
72
73 disappear_date = System::now() + 10000;
74 stop_now = false;
75 entity_reached = nullptr;
76}

204/**
205 * \brief Updates this entity.
206 */
207void Arrow::update() {
208
209 Entity::update();
210
211 if (is_suspended()) {
212 return;
213 }
214
215 uint32_t now = System::now();
216
217 // stop the movement if necessary (i.e. stop() was called)
218 if (stop_now) {
219 clear_movement();
220 stop_now = false;
221
222 if (entity_reached != nullptr) {
223 // the arrow just hit an entity (typically an enemy) and this entity may have a movement
224 Point dxy = get_xy() - entity_reached->get_xy();
225 set_movement(std::make_shared<RelativeMovement>(
226 entity_reached, dxy.x, dxy.y, true
227 ));
228 }
229 }
230
231 if (entity_reached != nullptr) {
232
233 // see if the entity reached is still valid
234 if (is_stopped()) {
235 // the arrow is stopped because the entity that was reached just disappeared
236 disappear_date = now;
237 }
238 else if (entity_reached->get_type() == EntityType::DESTRUCTIBLE && !entity_reached->is_obstacle_for(*this)) {
239 disappear_date = now;
240 }
241 else if (entity_reached->get_type() == EntityType::ENEMY &&
242 (std::static_pointer_cast<Enemy>(entity_reached)->is_dying())) {
243 // the enemy is dying
244 disappear_date = now;
245 }
246 }
247
248 // see if the arrow just hit a wall or an entity
249 bool reached_obstacle = false;
250
251 const SpritePtr& sprite = get_sprite();
252 if (sprite != nullptr && sprite->get_current_animation() != "reached_obstacle") {
253
254 if (entity_reached != nullptr) {
255 // the arrow was just attached to an entity
256 reached_obstacle = true;
257 }
258 else if (is_stopped()) {
259
260 if (has_reached_map_border()) {
261 // the map border was reached: destroy the arrow
262 disappear_date = now;
263 }
264 else {
265 // the arrow has just hit another obstacle
266 reached_obstacle = true;
267 }
268 }
269 }
270
271 if (reached_obstacle) {
272 // an obstacle or an entity was just reached
273 disappear_date = now + 1500;
274 if (sprite != nullptr) {
275 sprite->set_current_animation("reached_obstacle");
276 }
277 Sound::play("arrow_hit");
278
279 if (entity_reached == nullptr) {
280 clear_movement();
281 }
282 check_collision_with_detectors();
283 }
284
285 // destroy the arrow when disappear_date is reached
286 if (now >= disappear_date) {
287 remove_from_map();
288 }
289}

291/**
292 * \brief This function is called by the map when the game is suspended or resumed.
293 * \param suspended true to suspend the entity, false to resume it
294 */
295void Arrow::set_suspended(bool suspended) {
296
297 Entity::set_suspended(suspended); // suspend the movement
298
299 if (!suspended) {
300 // recalculate the timer
301 disappear_date += System::now() - get_when_suspended();
302 }
303}

312/**

316bool Arrow::is_stopped() const {

318}

320/**
321 * \brief Returns whether the arrow is currently flying.
322 * \return true if the arrow was shot and has not reached a target yet
323 */
324bool Arrow::is_flying() const {
325 return !is_stopped() && entity_reached == nullptr;
326}

328/**
329 * \brief Stops the arrow movement and attaches the arrow to an entity that was just reached.
330 * \param entity_reached the entity that was reached
331 */
332void Arrow::attach_to(Entity& entity_reached) {
333
334 Debug::check_assertion(this->entity_reached == nullptr,
335 "This arrow is already attached to an entity");
336
337 this->entity_reached = std::static_pointer_cast<Entity>(
338 entity_reached.shared_from_this()
339 );
340 stop_now = true;
341}

343/**
344 * \brief This function is called when a switch detects a collision with this entity.
345 * \param sw the switch
346 * \param collision_mode the collision mode that detected the event
347 */
348void Arrow::notify_collision_with_switch(Switch& sw, CollisionMode /* collision_mode */) {
349
350 if (entity_reached != nullptr) {
351 return;
352 }
353
354 if (sw.is_arrow_target() && is_stopped()) {
355 sw.try_activate(*this);
356 attach_to(sw);
357 }
358 else if (sw.is_solid() && is_flying()) {
359 sw.try_activate();
360 attach_to(sw);
361 }
362}
363
364/**
365 * \brief This function is called when a crystal detects a collision with this entity.
366 * \param crystal the crystal
367 * \param collision_mode the collision mode that detected the event
368 */
369void Arrow::notify_collision_with_crystal(Crystal& crystal, CollisionMode collision_mode) {
370
371 if (collision_mode == COLLISION_OVERLAPPING && is_flying()) {
372
373 crystal.activate(*this);
374 attach_to(crystal);
375 }
376}

378/**
379 * \brief This function is called when a destructible item detects a non-pixel perfect collision with this entity.
380 * \param destructible the destructible item
381 * \param collision_mode the collision mode that detected the event
382 */
383void Arrow::notify_collision_with_destructible(
384 Destructible& destructible, CollisionMode /* collision_mode */) {
385
386 if (destructible.is_obstacle_for(*this) && is_flying()) {
387
388 if (destructible.get_can_explode()) {
389 destructible.explode();
390 remove_from_map();
391 }
392 else {
393 attach_to(destructible);
394 }
395 }
396}

398/**
399 * \brief This function is called when an enemy's sprite collides with a sprite of this entity.
400 * \param enemy the enemy
401 * \param enemy_sprite the enemy's sprite that overlaps this entity
402 * \param this_sprite the arrow sprite
403 */
404void Arrow::notify_collision_with_enemy(
405 Enemy& enemy, Sprite& enemy_sprite, Sprite& /* this_sprite */) {
406
407 if (!overlaps(hero) && is_flying()) {
408 enemy.try_hurt(EnemyAttack::ARROW, *this, &enemy_sprite);
409 }
410}

412/**
413 * \copydoc Entity::notify_attacked_enemy
414 */
415void Arrow::notify_attacked_enemy(
416 EnemyAttack /* attack */,
417 Enemy& victim,
418 const Sprite* /* victim_sprite */,
419 EnemyReaction::Reaction& result,
420 bool killed) {
421
422 if (result.type == EnemyReaction::ReactionType::PROTECTED) {
423 stop();
424 attach_to(victim);
425 }
426 else if (result.type != EnemyReaction::ReactionType::IGNORED) {
427 if (killed) {
428 remove_from_map();
429 }
430 else {
431 attach_to(victim);
432 }
433 }
434}

436/**
437 * \brief Returns whether the arrow has just hit the map border.
438 * \return true if the arrow has just hit the map border
439 */
440bool Arrow::has_reached_map_border() {
441
442 const SpritePtr& sprite = get_sprite();
443 if (sprite != nullptr && sprite->get_current_animation() != "flying") {
444 return false;
445 }
446
447 if (get_movement() == nullptr) {
448 return false;
449 }
450
451 return get_map().test_collision_with_border(get_movement()->get_last_collision_box_on_obstacle());
452}
