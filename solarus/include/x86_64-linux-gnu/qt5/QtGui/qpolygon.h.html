62 inline explicit QPolygon(int size);

65 QPolygon(const QRect &r, bool closed=false);

69 operator QVariant() const;
70
71 void translate(int dx, int dy);
72 void translate(const QPoint &offset);

74 QPolygon translated(int dx, int dy) const;

77 QRect boundingRect() const;
78
79 void point(int i, int *x, int *y) const;
80 QPoint point(int i) const;

111 Misc. QPolygon functions

113
114inline void QPolygon::setPoint(int index, const QPoint &pt)
115{ (*this)[index] = pt; }
116
117inline void QPolygon::setPoint(int index, int x, int y)
118{ (*this)[index] = QPoint(x, y); }
119
120inline QPoint QPolygon::point(int index) const
121{ return at(index); }
122
123inline void QPolygon::translate(const QPoint &offset)
124{ translate(offset.x(), offset.y()); }
125
126inline QPolygon QPolygon::translated(const QPoint &offset) const
127{ return translated(offset.x(), offset.y()); }
