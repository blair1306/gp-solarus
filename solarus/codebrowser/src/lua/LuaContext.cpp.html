97/**
98 * \brief Initializes Lua.
99 */
100void LuaContext::initialize() {
101
102 // Create an execution context.
103 l = luaL_newstate();
104 lua_atpanic(l, l_panic);
105 luaL_openlibs(l);
106
107 print_lua_version();
108
109 // Associate this LuaContext object to the lua_State pointer.
110 lua_contexts[l] = this;
111
112 // Create a table that will keep track of all userdata.
113 // --
114 lua_newtable(l);
115 // all_udata
116 lua_newtable(l);
117 // all_udata meta
118 lua_pushstring(l, "v");
119 // all_udata meta "v"
120 lua_setfield(l, -2, "__mode");
121 // all_udata meta
122 lua_setmetatable(l, -2);
123 // all_udata
124 lua_setfield(l, LUA_REGISTRYINDEX, "sol.all_userdata");
125 // --
126
127 // Allow userdata to be indexable if they want.
128 lua_newtable(l);
129 // udata_tables
130 lua_setfield(l, LUA_REGISTRYINDEX, "sol.userdata_tables");
131 // --
132
133 // Create the sol table that will contain the whole Solarus API.
134 lua_newtable(l);
135 lua_setglobal(l, "sol");
136
137 // Register the C++ functions and types accessible by Lua.
138 register_modules();
139
140 // Make require() able to load Lua files even from the
141 // data.solarus or data.solarus.zip archive.
142 // --
143 lua_getglobal(l, "sol");
144 // -- sol
145 lua_pushcfunction(l, l_loader);
146 // -- sol loader
147 lua_setfield(l, -2, "loader");
148 // -- sol
149 luaL_dostring(l, "table.insert(package.loaders, 2, sol.loader)");
150 // -- sol
151 lua_pushnil(l);
152 // -- sol nil
153 lua_setfield(l, -2, "loader");
154 // -- sol
155 lua_pop(l, 1);
156 // --
157
158 // Make sure that stdout gets flushed when Lua scripts output new lines.
159 // This is not always the case by default.
160 luaL_dostring(l, "io.stdout:setvbuf(\"line\")");
161
162 Debug::check_assertion(lua_gettop(l) == 0, "Non-empty Lua stack after initialization");
163
164 // Execute the main file.
165 do_file_if_exists(l, "main");
166
167 Debug::check_assertion(lua_gettop(l) == 0, "Non-empty Lua Lua stack after running main.lua");
168
169 main_on_started();
170}

172/**
173 * \brief Cleans Lua.
174 */
175void LuaContext::exit() {
176
177 if (l != nullptr) {
178 // Call sol.main.on_finished() if it exists.
179 main_on_finished();
180
181 // Destroy unfinished objects.
182 destroy_menus();
183 destroy_timers();
184 destroy_drawables();
185 userdata_close_lua();
186
187 // Finalize Lua.
188 lua_close(l);
189 lua_contexts.erase(l);
190 l = nullptr;
191 }
192}

244/**
245 * \brief Notifies Lua that a map has just been started.
246 *
247 * The Lua file of this map is automatically loaded.
248 *
249 * \param map The map started.
250 * \param destination The destination point used if it's a normal one,
251 * nullptr otherwise.
252 */
253void LuaContext::run_map(Map& map, Destination* destination) {
254
255 // Compute the file name, depending on the id of the map.
256 std::string file_name = std::string("maps/") + map.get_id();
257
258 // Load the map's code.
259 bool load_success = load_file(l, file_name);
260 // map_fun
261
262 // Set a special environment to access map entities like global variables.
263 lua_newtable(l);
264 // map_fun env
265 lua_newtable(l);
266 // map_fun env env_mt
267 push_map(l, map);
268 // map_fun env env_mt map
269 // Set our special __index function that gets entities on-demand.
270 lua_pushcclosure(l, l_get_map_entity_or_global, 1);
271 // map_fun env env_mt __index
272 lua_setfield(l, -2, "__index");
273 // map_fun env env_mt
274 // We are changing the environment, so we need to also define __newindex
275 // with its usual setting (the global table).
276 lua_pushvalue(l, LUA_GLOBALSINDEX);
277 // map_fun env env_mt _G
278 lua_setfield(l, -2, "__newindex");
279 // map_fun env env_mt
280 lua_setmetatable(l, -2);
281 // map_fun env
282 lua_setfenv(l, -2);
283 // map_fun
284
285 if (load_success) {
286 // Run the map's code with the map userdata as parameter.
287 push_map(l, map);
288 call_function(1, 0, file_name.c_str());
289 }
290
291 // Call the map:on_started() callback.
292 map_on_started(map, destination);
293}

402 * \param game The game.
403 * \param dialog The dialog that was active.
404 * \param callback_ref Lua ref of the function to call, or an empty ref.
405 * \param status_ref Lua ref to a status value to pass to the callback.
406 * "skipped" means that the dialog was canceled by the user.
407 */
408void LuaContext::notify_dialog_finished(
409 Game& game,
410 const Dialog& dialog,
411 const ScopedLuaRef& callback_ref,
412 const ScopedLuaRef& status_ref) {
413
414 game_on_dialog_finished(game, dialog);
415
416 // Execute the callback after game:on_dialog_finished()
417 // because the callback may start another dialog.
418 if (!callback_ref.is_empty()) {
419 push_ref(l, callback_ref);
420 if (!status_ref.is_empty()) {
421 push_ref(l, status_ref);
422 }
423 else {
424 // No status.
425 lua_pushnil(l);
426 }
427 call_function(1, 0, "dialog callback");
428 }

483/**
484 * \brief Returns whether a userdata has an entry with the specified key.
485 *
486 * Userdata can have entries like tables thanks to special __index and
487 * __newindex metamethods.
488 *
489 * Version with const char*, better for performance if you don't have an
490 * std::string representation of the key.
491 *
492 * \param userdata A userdata.
493 * \param key String key to test.
494 * \return \c true if this key exists on the userdata.
495 */
496bool LuaContext::userdata_has_field(
497 const ExportableToLua& userdata, const char* key) const {
498
499 // TODO since this function now also checks the metatable, check that
500 // doing the work below instead of just returning true is still useful
501 // for performance.
502 // If not, kill this function.
503
504 // First check the metatable of the type.
505 if (userdata_has_metafield(userdata, key)) {
506 return true;
507 }
508
509 // Check the userdata itself then.
510 if (!userdata.is_with_lua_table()) {
511 return false;
512 }
513
514 const auto& it = userdata_fields.find(&userdata);
515 if (it == userdata_fields.end()) {
516 return false;
517 }
518
519 return it->second.find(key) != it->second.end();
520}

556/**
557 * \brief Returns whether the metatable of a userdata has the specified field.
558 * \param userdata A userdata.
559 * \param key String key to test.
560 * \return \c true if this key exists on the userdata's metatable.
561 */
562bool LuaContext::userdata_has_metafield(
563 const ExportableToLua& userdata, const char* key) const {
564
565 // We avoid to push the userdata for performance.
566 // Maybe the userdata does not even exist in the Lua side.
567 // ...
568 luaL_getmetatable(l, userdata.get_lua_type_name().c_str());
569 // ... meta
570 lua_pushstring(l, key);
571 // ... meta key
572 lua_rawget(l, -2);
573 // ... meta field/nil
574 const bool found = !lua_isnil(l, -1);
575 lua_pop(l, 2);
576 // ...
577 return found;
578}

595/**
596 * \brief Gets a method of an object.
597 *
598 * If the method exists, the method and the object are both pushed
599 * so that you can call the method immediately with the object as first parameter.
600 * If the method is not found, the stack is left unchanged.
601 *
602 * \param index Index of the object in the stack.
603 * \param function_name Name of the function to find in the object.
604 * This is not an const std::string& but a const char* on purpose to avoid
605 * costly conversions as this function is called very often.
606 *
607 * \return true if the function was found.
608 */
609bool LuaContext::find_method(int index, const char* function_name) {
610
611 index = LuaTools::get_positive_index(l, index);
612 // ... object ...
613 lua_getfield(l, index, function_name);
614 // ... object ... method/?
615
616 bool exists = lua_isfunction(l, -1);
617 if (exists) {
618 // ... object ... method
619 lua_pushvalue(l, index);
620 // ... object ... method object
621 }
622 else {
623 // Restore the stack.
624 lua_pop(l, 1);
625 // ... object ...
626 }
627
628 return exists;
629}

657/**
658 * \brief Opens a script if it exists and lets it on top of the stack as a
659 * function.
660 *
661 * If the file does not exist or has a syntax error,
662 * the stack is left intact and false is returned.
663 *
664 * \param l A Lua state.
665 * \param script_name File name of the script with or without extension,
666 * relative to the data directory.
667 * \return true if the file exists and was loaded.
668 */
669bool LuaContext::load_file(lua_State* l, const std::string& script_name) {
670
671 // Determine the file name (possibly adding ".lua").
672 std::string file_name(script_name);
673
674 if (!QuestFiles::data_file_exists(file_name)) {
675 std::ostringstream oss;
676 oss << script_name << ".lua";
677 file_name = oss.str();
678 }
679
680 if (!QuestFiles::data_file_exists(file_name)) {
681 // No error message: this is not an error.
682 return false;
683 }
684
685 // Load the file.
686 // "@" tells Lua that the name is a file name, which is useful for better error messages.
687 const std::string& buffer = QuestFiles::data_file_read(file_name);
688 int result = luaL_loadbuffer(l, buffer.data(), buffer.size(), ("@" + file_name).c_str());
689
690 if (result != 0) {
691 Debug::error(std::string("Failed to load script '")
692 + script_name + "': " + lua_tostring(l, -1));
693 lua_pop(l, 1);
694 return false;
695 }
696 return true;
697}

739/**
740 * \brief Prints on a line the content of the Lua stack for debugging purposes.
741 * \param l A Lua state.
742 */
743void LuaContext::print_stack(lua_State* l) {
744
745 int i;
746 int top = lua_gettop(l);
747
748 std::ostringstream oss;
749 for (i = 1; i <= top; i++) {
750
751 int type = lua_type(l, i);
752 switch (type) {
753
754 case LUA_TSTRING:
755 oss << "\"" << lua_tostring(l, i) << "\"";
756 break;
757
758 case LUA_TBOOLEAN:
759 oss << (lua_toboolean(l, i) ? "true" : "false");
760 break;
761
762 case LUA_TNUMBER:
763 oss << lua_tonumber(l, i);
764 break;
765
766 case LUA_TLIGHTUSERDATA:
767 oss << "lightuserdata:" << lua_touserdata(l, i);
768 break;
769
770 case LUA_TUSERDATA:
771 {

773 lua_touserdata(l, i)));
774 const std::string& lua_type_name = userdata->get_lua_type_name();
775 oss << lua_type_name.substr(lua_type_name.find_last_of('.') + 1);
776 break;
777 }
778
779 default:
780 oss << lua_typename(l, type);
781 break;
782
783 }
784 oss << " ";
785 }
786 Logger::debug(oss.str());

845/**
846 * \brief Defines some C++ functions into a new Lua userdata type.
847 * \param module_name name of the table that will contain the functions
848 * (e.g. "sol.game"). It may already exist or not.
849 * This string will also identify the type.
850 * \param functions List of functions to define in the module table or nullptr.
851 * Must end with {nullptr, nullptr}.
852 * \param methods List of methods to define in the type or nullptr.
853 * Must end with {nullptr, nullptr}.
854 * \param metamethods List of metamethods to define in the metatable of the
855 * type or nullptr.
856 * Must end with {nullptr, nullptr}.
857 */
858void LuaContext::register_type(
859 const std::string& module_name,
860 const luaL_Reg* functions,
861 const luaL_Reg* methods,
862 const luaL_Reg* metamethods
863) {
864
865 // Check that this type does not already exist.
866 luaL_getmetatable(l, module_name.c_str());
867 Debug::check_assertion(lua_isnil(l, -1),
868 std::string("Type ") + module_name + " already exists");
869 lua_pop(l, 1);
870
871 // Make sure we create the table.
872 const luaL_Reg empty[] = {
873 { nullptr, nullptr }
874 };
875 luaL_register(l, module_name.c_str(), empty);
876 // module
877
878 // Add the functions to the module.
879 if (functions != nullptr) {
880 luaL_register(l, nullptr, functions);
881 // module
882 }
883 lua_pop(l, 1);
884 // --
885
886 // Create the metatable for the type, add it to the Lua registry.
887 luaL_newmetatable(l, module_name.c_str());
888 // meta
889
890 // Store a metafield __solarus_type with the module name.
891 lua_pushstring(l, module_name.c_str());
892 // meta type_name
893 lua_setfield(l, -2, "__solarus_type");
894 // meta
895
896 // Add the methods to the metatable.
897 if (methods != nullptr) {
898 luaL_register(l, nullptr, methods);
899 }
900 // meta
901
902 // Add the metamethods to the metatable.
903 if (metamethods != nullptr) {
904 luaL_register(l, nullptr, metamethods);
905 // meta
906 }
907
908 // make metatable.__index = metatable,
909 // unless if __index is already defined
910 lua_getfield(l, -1, "__index");
911 // meta __index/nil
912 lua_pushvalue(l, -2);
913 // meta __index/nil meta
914 if (lua_isnil(l, -2)) {
915 // meta nil meta
916 lua_setfield(l, -3, "__index");
917 // meta nil
918 }
919 lua_settop(l, 0);
920 // --
921}

981/**
982 * \brief Pushes the Lua equivalent of a C++ object onto the stack.
983 * \param l A Lua context.
984 * \param userdata A userdata. It must live as a std::shared_ptr somewhere:
985 * typically, it should have been stored in a std::shared_ptr at creation time.
986 */
987void LuaContext::push_userdata(lua_State* l, ExportableToLua& userdata) {
988
989 // See if this userdata already exists.
990 lua_getfield(l, LUA_REGISTRYINDEX, "sol.all_userdata");
991 // ... all_udata
992 lua_pushlightuserdata(l, &userdata);
993 // ... all_udata lightudata
994 lua_gettable(l, -2);
995 // ... all_udata udata/nil
996 if (!lua_isnil(l, -1)) {
997 // ... all_udata udata
998 // The userdata already exists in the Lua world.
999 lua_remove(l, -2);
1000 // ... udata
1001 }
1002 else {
1003 // Create a new userdata.
1004
1005 if (!userdata.is_known_to_lua()) {
1006 // This is the first time we create a Lua userdata for this object.
1007 userdata.set_known_to_lua(true);
1008 userdata.set_lua_context(&get_lua_context(l));
1009 }
1010
1011 // ... all_udata nil
1012 lua_pop(l, 1);
1013 // ... all_udata
1014 lua_pushlightuserdata(l, &userdata);
1015 // ... all_udata lightudata
1016
1017 // Find the existing shared_ptr from the raw pointer.
1018 ExportableToLuaPtr shared_userdata;
1019 try {
1020 shared_userdata = userdata.shared_from_this();
1021 }
1022 catch (const std::bad_weak_ptr& ex) {
1023 // No existing shared_ptr. This is probably because you forgot to
1024 // store your object in a shared_ptr at creation time.
1025 Debug::die(
1026 std::string("No living shared_ptr for ") + userdata.get_lua_type_name()
1027 );
1028 }
1029
1030 ExportableToLuaPtr* block_address = static_cast<ExportableToLuaPtr*>(
1031 lua_newuserdata(l, sizeof(ExportableToLuaPtr))
1032 );
1033 // Manually construct a shared_ptr in the block allocated by Lua.
1034 new (block_address) ExportableToLuaPtr(shared_userdata);
1035 // ... all_udata lightudata udata
1036 luaL_getmetatable(l, userdata.get_lua_type_name().c_str());
1037 // ... all_udata lightudata udata mt
1038
1039 Debug::execute_if_debug([&] {
1040 Debug::check_assertion(!lua_isnil(l, -1),
1041 std::string("Userdata of type '" + userdata.get_lua_type_name()
1042 + "' has no metatable, this is a memory leak"));
1043
1044 lua_getfield(l, -1, "__gc");
1045 // ... all_udata lightudata udata mt gc
1046 Debug::check_assertion(lua_isfunction(l, -1),
1047 std::string("Userdata of type '") + userdata.get_lua_type_name()
1048 + "' must have the __gc function LuaContext::userdata_meta_gc");
1049 // ... all_udata lightudata udata mt gc
1050 lua_pop(l, 1);
1051 // ... all_udata lightudata udata mt
1052 });
1053
1054 lua_setmetatable(l, -2);
1055 // ... all_udata lightudata udata
1056 // Keep track of our new userdata.
1057 lua_pushvalue(l, -1);
1058 // ... all_udata lightudata udata udata
1059 lua_insert(l, -4);
1060 // ... udata all_udata lightudata udata
1061 lua_settable(l, -3);
1062 // ... udata all_udata
1063 lua_pop(l, 1);
1064 // ... udata
1065 }
1066}

1162/**
1163 * \brief Finalizer of a userdata type.
1164 * \param l A Lua state.
1165 * \return Number of values to return to Lua.
1166 */
1167int LuaContext::userdata_meta_gc(lua_State* l) {
1168
1169 ExportableToLuaPtr* userdata =
1170 static_cast<ExportableToLuaPtr*>(lua_touserdata(l, 1));
1171
1172 // Note that the full userdata disappears from Lua but it may come back later!
1173 // So we need to keep its table if the refcount is not zero.
1174 // The full userdata is destroyed but the light userdata and its table persist.
1175 // Its table will be destroyed from ~ExportableToLua().
1176
1177 // We don't need to remove the entry from sol.all_userdata
1178 // because it is already done: that table is weak on its values and the
1179 // value was the full userdata.
1180
1181 // Manually destroy the shared_ptr allocated for Lua.
1182 userdata->~shared_ptr<ExportableToLua>();
1183
1184 return 0;
1185}

1222 *
1223 * This must be done when Lua is about to be closed,
1224 * so that they can continue to live normally in C++.

1245/**
1246 * \brief Implementation of __newindex that allows userdata to be like tables.
1247 *
1248 * Lua code can make "object[key] = value" if object is a userdata with this
1249 * __newindex metamethod.
1250 *
1251 * This metamethod must be used with its corresponding __index
1252 * metamethod (see userdata_meta_index_as_table).
1253 *
1254 * \param l The Lua context that is calling this function.
1255 * \return Number of values to return to Lua.
1256 */
1257int LuaContext::userdata_meta_newindex_as_table(lua_State* l) {
1258
1259 LuaTools::check_type(l, 1, LUA_TUSERDATA);
1260 LuaTools::check_any(l, 2);
1261 LuaTools::check_any(l, 3);
1262
1263 const ExportableToLuaPtr& userdata =
1264 *(static_cast<ExportableToLuaPtr*>(lua_touserdata(l, 1)));
1265
1266 // The user wants to make udata[key] = value but udata is a userdata.
1267 // So what we make instead is udata_tables[udata][key] = value.
1268 // This redirection is totally transparent from the Lua side.
1269
1270 lua_getfield(l, LUA_REGISTRYINDEX, "sol.userdata_tables");
1271 // ... udata_tables
1272
1273 if (!userdata->is_with_lua_table()) {
1274 // Create the userdata table if it does not exist yet.
1275
1276 userdata->set_with_lua_table(true);
1277 // ... udata_tables
1278 lua_newtable(l);
1279 // ... udata_tables udata_table
1280 lua_pushlightuserdata(l, userdata.get());
1281 // ... udata_tables udata_table lightudata
1282 lua_pushvalue(l, -2);
1283 // ... udata_tables udata_table lightudata udata_table
1284 lua_settable(l, -4);
1285 // ... udata_tables udata_table
1286 }
1287 else {
1288 // The userdata table already exists.
1289 lua_pushlightuserdata(l, userdata.get());
1290 // ... udata_tables lightudata
1291 lua_gettable(l, -2);
1292 // ... udata_tables udata_table
1293 }
1294 Debug::check_assertion(!lua_isnil(l, -1), "Missing userdata table");
1295 lua_pushvalue(l, 2);
1296 // ... udata_tables udata_table key
1297 lua_pushvalue(l, 3);
1298 // ... udata_tables udata_table key value
1299 lua_settable(l, -3);
1300 // ... udata_tables udata_table
1301
1302 if (lua_isstring(l, 2)) {
1303 if (!lua_isnil(l, 3)) {
1304 // Add the key to the list of existing strings keys on this userdata.
1305 get_lua_context(l).userdata_fields[userdata.get()].insert(lua_tostring(l, 2));
1306 }
1307 else {
1308 // Assigning nil: remove the key from the list.
1309 get_lua_context(l).userdata_fields[userdata.get()].erase(lua_tostring(l, 2));
1310 }
1311 }
1312
1313 return 0;
1314}

1316/**
1317 * \brief Implementation of __index that allows userdata to be like tables.
1318 *
1319 * Lua code can get "object[key]" for an arbitrary key previously set.
1320 *
1321 * This metamethod must be used with its corresponding __newindex
1322 * metamethod (see userdata_meta_newindex_as_table).
1323 *
1324 * \brief Implementation of __index for the type game.
1325 * \param l The Lua context that is calling this function.
1326 * \return Number of values to return to Lua.
1327 */
1328int LuaContext::userdata_meta_index_as_table(lua_State* l) {
1329
1330 /* The user wants to make udata[key] but udata is a userdata.
1331 * So what we retrieve instead is udata_tables[udata][key].
1332 * This redirection is totally transparent from the Lua side.
1333 * If udata_tables[udata][key] does not exist, we fall back
1334 * to the userdata __index metamethod.
1335 */
1336
1337 LuaTools::check_type(l, 1, LUA_TUSERDATA);
1338 LuaTools::check_any(l, 2);
1339
1340 const ExportableToLuaPtr& userdata =
1341 *(static_cast<ExportableToLuaPtr*>(lua_touserdata(l, 1)));
1342 LuaContext& lua_context = get_lua_context(l);
1343
1344 // If the userdata actually has a table, lookup this table, unless we already
1345 // know that we won't find it (because we know all the existing string keys).
1346 if (userdata->is_with_lua_table() &&
1347 (!lua_isstring(l, 2) || lua_context.userdata_has_field(*userdata, lua_tostring(l, 2)))) {
1348
1349 lua_getfield(l, LUA_REGISTRYINDEX, "sol.userdata_tables");
1350 // ... udata_tables
1351 lua_pushlightuserdata(l, userdata.get());
1352 // ... udata_tables lightudata
1353 // Lookup the key in the table, without metamethods.
1354 lua_rawget(l, -2);
1355 // ... udata_tables udata_table/nil
1356 if (!lua_isnil(l, -1)) {
1357 lua_pushvalue(l, 2);
1358 // ... udata_tables udata_table key
1359 lua_gettable(l, -2);
1360 // ... udata_tables udata_table value/nil
1361 if (!lua_isnil(l, -1)) {
1362 // Found it!
1363 return 1;
1364 }
1365 }
1366 }
1367
1368 // Not in the table. See in the metatable
1369 // (just like when metatable.__index = metatable).
1370
1371 lua_pushvalue(l, 1);
1372 // ... udata
1373 lua_getmetatable(l, -1);
1374 // ... udata meta
1375 Debug::check_assertion(!lua_isnil(l, -1), "Missing userdata metatable");
1376 lua_pushvalue(l, 2);
1377 // ... udata meta key
1378 lua_gettable(l, -2);
1379 // ... udata meta key value/nil
1380 return 1;
1381}

1538 * \param event The input event to forward.
1539 * \return \c true if the event was handled and should stop being propagated.

1541bool LuaContext::on_input(const InputEvent& event) {
1542

1544 bool handled = false;
1545 if (event.is_keyboard_event()) {
1546 // Keyboard.
1547 if (event.is_keyboard_key_pressed()) {
1548 handled = on_key_pressed(event) || handled;
1549 }
1550 else if (event.is_keyboard_key_released()) {
1551 handled = on_key_released(event) || handled;
1552 }
1553 }
1554 else if (event.is_character_pressed()) {
1555 handled = on_character_pressed(event) || handled;
1556 }
1557 else if (event.is_joypad_event()) {
1558 // Joypad.
1559 if (event.is_joypad_button_pressed()) {
1560 handled = on_joypad_button_pressed(event) || handled;
1561 }
1562 else if (event.is_joypad_button_released()) {
1563 handled = on_joypad_button_released(event) || handled;
1564 }
1565 else if (event.is_joypad_axis_moved()) {
1566 handled = on_joypad_axis_moved(event) || handled;
1567 }
1568 else if (event.is_joypad_hat_moved()) {
1569 handled = on_joypad_hat_moved(event) || handled;
1570 }
1571 }
1572 else if (event.is_mouse_event()) {
1573 // Mouse.
1574 if (event.is_mouse_button_pressed()) {
1575 handled = on_mouse_button_pressed(event) || handled;
1576 }
1577 else if (event.is_mouse_button_released()) {
1578 handled = on_mouse_button_released(event) || handled;
1579 }
1580 }
1581 else if (event.is_finger_event()) {
1582 // Touch finger.
1583 if (event.is_finger_pressed()) {
1584 handled = on_finger_pressed(event) || handled;
1585 }
1586 else if (event.is_finger_released()) {
1587 handled = on_finger_released(event) || handled;
1588 }
1589 else if (event.is_finger_moved()) {
1590 handled = on_finger_moved(event) || handled;
1591 }
1592 }
1593
1594 return handled;
1595}
