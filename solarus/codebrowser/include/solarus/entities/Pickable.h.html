17#ifndef SOLARUS_PICKABLE_H
18#define SOLARUS_PICKABLE_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include "solarus/entities/EntityPtr.h"
23#include "solarus/lowlevel/Point.h"
24#include "solarus/movements/FallingHeight.h"
25#include "solarus/SpritePtr.h"
26#include "solarus/Treasure.h"
27#include <memory>
28#include <string>
29
30namespace Solarus {
31
32/**
33 * \brief An treasure placed on the ground and that the hero can take.
34 */
35class Pickable: public Entity {
36
37 public:
38
39 static constexpr EntityType ThisType = EntityType::PICKABLE;
40
41 // creation and destruction
42 Pickable(
43 const std::string& name,
44 int layer,
45 const Point& xy,
46 const Treasure& treasure
47 );
48
49 static std::shared_ptr<Pickable> create(
50 Game& game,
51 const std::string& name,
52 int layer,
53 const Point& xy,
54 Treasure treasure,
55 FallingHeight falling_height,
56 bool force_persistent
57 );
58
59 EntityType get_type() const override;
60 bool is_ground_observer() const override;
61
62 void notify_created() override;
63
64 // properties
65 bool is_falling() const;
66 FallingHeight get_falling_height() const;
67 const Treasure& get_treasure() const;
68
69 // item state
70 void set_suspended(bool suspended) override;
71 EntityPtr get_entity_followed();
72
73 bool is_stream_obstacle(Stream& stream) override;
74
75 void notify_collision(
76 Entity& entity_overlapping,
77 CollisionMode collision_mode
78 ) override;
79 void notify_collision(
80 Entity& other_entity,
81 Sprite& this_sprite,
82 Sprite& other_sprite
83 ) override;
84 void update() override;
85 void draw_on_map() override;
86
87 private:
88
89 // initialization
90 bool initialize_sprites();
91 void initialize_movement();
92
93 // item
94 void try_give_item_to_player();
95 void set_blinking(bool blinking);
96
97 void check_bad_ground();
98
99 Treasure treasure; /**< the treasure obtained when the player picks this item */
100 bool given_to_player; /**< indicates that the treasure was given to the player */
101
102 SpritePtr item_sprite; /**< Sprite of the treasure. */
103 SpritePtr shadow_sprite; /**< Sprite of the shadow (if any). */
104 FallingHeight falling_height; /**< indicates whether the item is falling when it appears */
105 bool will_disappear; /**< indicates that the item will disappear after an amount of time
106 * (only possible for items not saved) */
107
108 // current state
109 Point shadow_xy; /**< coordinates of the shadow (which does not move while the item does) */
110
111 uint32_t appear_date; /**< date when the item is created */
112 uint32_t allow_pick_date; /**< date when the item can be picked */
113 bool can_be_picked; /**< indicates that the item can be picked now (i.e. allow_picked_date is past) */
114 uint32_t blink_date; /**< date when the item starts blinking */
115 uint32_t disappear_date; /**< date when the item disappears */
116 EntityPtr entity_followed; /**< an entity this item is attached to (e.g. a boomerang or a hookshot) */
117};
118
119}
120
121#endif
