51
52class QMenuBarPrivate;
53class QStyleOptionMenuItem;
54class QWindowsStyle;
55class QPlatformMenuBar;
56
57class Q_WIDGETS_EXPORT QMenuBar : public QWidget
58{
59 Q_OBJECT
60
61 Q_PROPERTY(bool defaultUp READ isDefaultUp WRITE setDefaultUp)
62 Q_PROPERTY(bool nativeMenuBar READ isNativeMenuBar WRITE setNativeMenuBar)
63
64public:
65 explicit QMenuBar(QWidget *parent = 0);
66 ~QMenuBar();
67
68#ifdef Q_NO_USING_KEYWORD
69 void addAction(QAction *action) { QWidget::addAction(action); }
70#else
71 using QWidget::addAction;
72#endif
73 QAction *addAction(const QString &text);
74 QAction *addAction(const QString &text, const QObject *receiver, const char* member);
75
76 QAction *addMenu(QMenu *menu);
77 QMenu *addMenu(const QString &title);
78 QMenu *addMenu(const QIcon &icon, const QString &title);
79
80
81 QAction *addSeparator();
82 QAction *insertSeparator(QAction *before);
83
84 QAction *insertMenu(QAction *before, QMenu *menu);
85
86 void clear();
87
88 QAction *activeAction() const;
89 void setActiveAction(QAction *action);
90
91 void setDefaultUp(bool);
92 bool isDefaultUp() const;
93
94 QSize sizeHint() const;
95 QSize minimumSizeHint() const;
96 int heightForWidth(int) const;
97
98 QRect actionGeometry(QAction *) const;
99 QAction *actionAt(const QPoint &) const;
100
101 void setCornerWidget(QWidget *w, Qt::Corner corner = Qt::TopRightCorner);
102 QWidget *cornerWidget(Qt::Corner corner = Qt::TopRightCorner) const;
103
104#ifdef Q_OS_WINCE
105 void setDefaultAction(QAction *);
106 QAction *defaultAction() const;
107
108 static void wceCommands(uint command);
109 static void wceRefresh();
110#endif
111#ifdef Q_OS_OSX
112 NSMenu* toNSMenu();
113#endif
114
115 bool isNativeMenuBar() const;
116 void setNativeMenuBar(bool nativeMenuBar);
117 QPlatformMenuBar *platformMenuBar();
118public Q_SLOTS:
119 virtual void setVisible(bool visible);
120
121Q_SIGNALS:
122 void triggered(QAction *action);
123 void hovered(QAction *action);
124
125protected:
126 void changeEvent(QEvent *);
127 void keyPressEvent(QKeyEvent *);
128 void mouseReleaseEvent(QMouseEvent *);
129 void mousePressEvent(QMouseEvent *);
130 void mouseMoveEvent(QMouseEvent *);
131 void leaveEvent(QEvent *);
132 void paintEvent(QPaintEvent *);
133 void resizeEvent(QResizeEvent *);
134 void actionEvent(QActionEvent *);
135 void focusOutEvent(QFocusEvent *);
136 void focusInEvent(QFocusEvent *);
137 void timerEvent(QTimerEvent *);
138 bool eventFilter(QObject *, QEvent *);
139 bool event(QEvent *);
140 void initStyleOption(QStyleOptionMenuItem *option, const QAction *action) const;
141
142private:
143 Q_DECLARE_PRIVATE(QMenuBar)
144 Q_DISABLE_COPY(QMenuBar)
145 Q_PRIVATE_SLOT(d_func(), void _q_actionTriggered())
146 Q_PRIVATE_SLOT(d_func(), void _q_actionHovered())
147 Q_PRIVATE_SLOT(d_func(), void _q_internalShortcutActivated(int))
148 Q_PRIVATE_SLOT(d_func(), void _q_updateLayout())
149
150#ifdef Q_OS_WINCE
151 Q_PRIVATE_SLOT(d_func(), void _q_updateDefaultAction())
152#endif
153
154 friend class QMenu;
155 friend class QMenuPrivate;
156 friend class QWindowsStyle;
157};

159#endif // QT_NO_MENUBAR
