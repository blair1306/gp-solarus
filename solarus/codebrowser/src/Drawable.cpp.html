57/**
58 * \brief Applies a movement to this object.
59 *
60 * Any previous movement is stopped.
61 *
62 * \param movement The movement to apply.
63 */
64void Drawable::start_movement(const std::shared_ptr<Movement>& movement) {
65
66 stop_movement();
67 this->movement = movement;
68 movement->set_drawable(this);
69
70 movement->set_suspended(is_suspended());
71}
72
73/**
74 * \brief Stops the movement applied to the object, if any.
75 *
76 * The movement is deleted unless the owner script uses it elsewhere.
77 */
78void Drawable::stop_movement() {
79
80 movement = nullptr;
81}

108/**
109 * \brief Starts a transition effect on this object.
110 *
111 * Any previous transition is stopped.
112 *
113 * \param transition The transition to start.
114 * \param callback_ref A Lua registry ref to the function to call when
115 * the transition finishes, or an empty ref.
116 */
117void Drawable::start_transition(
118 std::unique_ptr<Transition> transition,
119 const ScopedLuaRef& callback_ref
120) {
121 stop_transition();
122
123 this->transition = std::move(transition);
124 this->transition_callback_ref = callback_ref;
125 this->transition->start();
126 this->transition->set_suspended(is_suspended());
127}
128
129/**
130 * \brief Stops the transition effect applied to this object, if any.
131 *
132 * The transition is deleted and the Lua callback (if any) is canceled.
133 */
134void Drawable::stop_transition() {
135
136 transition = nullptr;
137 transition_callback_ref.clear();
138}

140/**
141 * \brief Returns the current transition of this drawable object.
142 * \return The object's transition, or nullptr if there is no transition.
143 */
144Transition* Drawable::get_transition() {
145 return transition.get();
146}

148/**
149 * \brief Updates this object.
150 *
151 * This function is called repeatedly.
152 * You can redefine it for your needs.
153 */
154void Drawable::update() {
155
156 if (transition != nullptr) {
157 transition->update();
158 if (transition->is_finished()) {
159
160 transition = nullptr;
161
162 if (!transition_callback_ref.is_empty()) {
163 // Note that this callback may create a new transition right now.
164 transition_callback_ref.clear_and_call("transition callback");
165 }
166 }
167 }
168
169 if (movement != nullptr) {
170 movement->update();
171 if (movement != nullptr && movement->is_finished()) {
172 stop_movement();
173 }
174 }
175}

177/**
178 * \brief Returns whether this drawable is suspended.
179 * \return \c true if this drawable is suspended.
180 */
181bool Drawable::is_suspended() const {
182 return suspended;
183}
184
185/**
186 * \brief Suspends or resumes this drawable.
187 * \param suspended \c true to suspend this drawable, \c false to resume it.
188 */
189void Drawable::set_suspended(bool suspended) {
190
191 if (suspended == this->suspended) {
192 return;
193 }
194
195 this->suspended = suspended;
196
197 // Suspend or resume the transition effect and the movement if any.
198 if (transition != nullptr) {
199 transition->set_suspended(suspended);
200 }
201
202 if (movement != nullptr) {
203 movement->set_suspended(suspended);
204 }
205}

216/**

219 * \param x x coordinate of where to draw

222void Drawable::draw(const SurfacePtr& dst_surface, int x, int y) {
223
224 draw(dst_surface, Point(x, y));
225}
226
227/**
228 * \brief Draws this object, applying dynamic effects.
229 * \param dst_surface the destination surface
230 * \param dst_position position on this surface
231 * (will be added to the position obtained by previous movements)
232 */
233void Drawable::draw(const SurfacePtr& dst_surface,
234 const Point& dst_position) {
235
236 if (transition != nullptr) {
237 draw_transition(*transition);
238 }
239
240 raw_draw(*dst_surface, dst_position + xy);
241}
242
243/**
244 * \brief Draws a subrectangle of this object, applying dynamic effects.

248void Drawable::draw_region(

251
252 draw_region(region, dst_surface, Point(0, 0));
253}
254
255/**
256 * \brief Draws a subrectangle of this object, applying dynamic effects.
257 * \param region The rectangle to draw in this object.
258 * \param dst_surface The destination surface
259 * \param dst_position Position on this surface
260 * (will be added to the position obtained by previous movements).
261 */
262void Drawable::draw_region(
263 const Rectangle& region,
264 const SurfacePtr& dst_surface,
265 const Point& dst_position) {
266
267 if (transition != nullptr) {
268 draw_transition(*transition);
269 }
270
271 raw_draw_region(region, *dst_surface, dst_position + xy);
272}
