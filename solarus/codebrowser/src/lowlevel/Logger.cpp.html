17#include "solarus/lowlevel/Logger.h"
18#include "solarus/lowlevel/System.h"
19#include <fstream>
20#include <iostream>
21
22namespace Solarus {
23
24namespace Logger {
25
26namespace {
27
28 const std::string error_log_file_name = "error.txt";
29 std::ofstream error_log_file;
30
31 /**
32 * \brief Returns the error log file.
33 *
34 * Opens it the first time this function is called.
35 */
36 std::ofstream& get_error_log_file() {
37
38 if (!error_log_file.is_open()) {
39 error_log_file.open(error_log_file_name.c_str());
40 }
41 return error_log_file;
42 }
43}
44
45/**
46 * \brief Logs a message on the given output stream.
47 *
48 * The message is prepended by "[Solarus] [t] " where t is the current
49 * simulated time.
50 *
51 * \param message The message to log.
52 * \param out The output stream.
53 */
54SOLARUS_API void print(const std::string& message, std::ostream& out) {
55
56 uint32_t simulated_time = System::now();
57 out << "[Solarus] [" << simulated_time << "] " << message << std::endl;
58}
59
60/**
61 * \brief Logs a debug message on stdout.
62 * \param message The message to log.
63 */
64SOLARUS_API void debug(const std::string& message) {
65
66 print("Debug: " + message);
67}
68
69/**
70 * \brief Logs an information message on stdout.
71 * \param message The message to log.
72 */
73SOLARUS_API void info(const std::string& message) {
74
75 print("Info: " + message);
76}
77
78/**
79 * \brief Logs a warning message on stdout and error.txt.
80 * \param message The message to log.
81 */
82SOLARUS_API void warning(const std::string& message) {
83
84 std::string prefixed_message = "Warning: " + message;
85 print(prefixed_message);
86 print(prefixed_message, get_error_log_file());
87}
88

98}
