29/**
30 * \brief Constructor.
31 * \param name Name of the teletransporter.
32 * \param layer Layer of the teletransporter.
33 * \param xy Coordinates where to create the entity.
34 * \param size Size of the teletransporter's rectangle.
35 * \param sprite_name Sprite animation set id to use, or an empty string.
36 * \param sound_id Sound to play when using the teletransporter,
37 * or an empty string.
38 * \param transition_style Style of transition between the two maps.
39 * \param destination_map_id Id of the destination map.
40 * \param destination_name Location on the destination map,
41 * or "_same" to keep the hero's coordinates,
42 * or "_side" to place the hero on the appropriate side of the map.
43 * An empty string means the default destination entity of the map.
44 */
45Teletransporter::Teletransporter(
46 const std::string& name,
47 int layer,
48 const Point& xy,
49 const Size& size,
50 const std::string& sprite_name,
51 const std::string& sound_id,
52 Transition::Style transition_style,
53 const std::string& destination_map_id,
54 const std::string& destination_name):
55
56 Entity(name, 0, layer, xy, size),
57 sound_id(sound_id),
58 transition_style(transition_style),
59 destination_map_id(destination_map_id),
60 destination_name(destination_name),
61 destination_side(-1),
62 transition_direction(0),
63 transporting_hero(false) {
64
65 set_collision_modes(CollisionMode::COLLISION_CUSTOM);
66
67 if (!sprite_name.empty()) {
68 create_sprite(sprite_name);
69 }
70}

72/**
73 * \copydoc Entity::notify_creating
74 */
75void Teletransporter::notify_creating() {
76
77 Entity::notify_creating();
78
79 int x = get_x();
80 int y = get_y();
81
82 // Compute the destination side in case the destination name is "_side"
83 // or becomes it later.
84 if (get_width() == 16 && x == -16) {
85 destination_side = 0;
86 }
87 else if (get_width() == 16 && x == get_map().get_width()) {
88 destination_side = 2;
89 }
90 else if (get_height() == 16 && y == -16) {
91 destination_side = 3;
92 }
93 else if (get_height() == 16 && y == get_map().get_height()) {
94 destination_side = 1;
95 }
96
97 if (destination_side != -1) {
98 set_layer_independent_collisions(true);
99 transition_direction = (destination_side + 2) % 4;
100 }
101}

211/**
212 * \brief Returns whether an entity's collides with this entity.
213 * \param entity an entity
214 * \return true if the entity's collides with this entity
215 */
216bool Teletransporter::test_collision_custom(Entity& entity) {
217
218 bool collision = false;
219 bool normal_case = true;
220
221 // specific collision tests for some situations
222 if (entity.is_hero()) {
223
224 Hero& hero = static_cast<Hero&>(entity);
225 if (is_on_map_side()) {
226 // scrolling towards an adjacent map
227 const Point& touching_point = hero.get_touching_point(transition_direction);
228 collision = hero.is_moving_towards(transition_direction)
229 && overlaps(touching_point);
230 normal_case = false;
231 }
232
233 else if (!get_map().test_collision_with_border(get_center_point()) &&
234 hero.get_ground_below() == Ground::HOLE) {
235 // falling into a hole
236 collision = overlaps(hero.get_ground_point());
237 normal_case = false;
238 }
239 }
240
241 // normal case
242 if (normal_case) {
243 const Rectangle& entity_rectangle = entity.get_bounding_box();
244 int x1 = entity_rectangle.get_x() + 4;
245 int x2 = x1 + entity_rectangle.get_width() - 9;
246 int y1 = entity_rectangle.get_y() + 4;
247 int y2 = y1 + entity_rectangle.get_height() - 9;
248
249 collision = overlaps(x1, y1) && overlaps(x2, y1) &&
250 overlaps(x1, y2) && overlaps(x2, y2);
251 }
252
253 if (!collision && !is_on_map_side()) {
254 transporting_hero = false;
255 }
256
257 return collision;
258}

270/**
271 * \brief Makes the teletransporter move the hero to the destination.
272 * \param hero the hero
273 */
274void Teletransporter::transport_hero(Hero& hero) {
275
276 if (transporting_hero) {
277 // already done
278 return;
279 }
280
281 std::string name = destination_name;
282 int hero_x = hero.get_x();
283 int hero_y = hero.get_y();
284
285 if (is_on_map_side()) {
286
287 // special destination point: side of the map
288 // we determine the appropriate side based on the teletransporter's position;
289 // we also place the hero on the old map so that its position corresponds to the new map
290
291 switch (destination_side) {
292
293 case 0:
294 name += '0'; // scroll to the west
295 hero_x = 0;
296 break;
297
298 case 1:
299 name += '1'; // scroll to the south
300 hero_y = get_map().get_height() + 5;
301 break;
302
303 case 2:
304 name += '2'; // scroll to the east
305 hero_x = get_map().get_width();
306 break;
307
308 case 3:
309 name += '3'; // scroll to the north
310 hero_y = 5;
311 break;
312
313 default:
314 Debug::die(std::string("Bad destination side for teletransporter '")
315 + get_name() + "'");
316 }
317 }
318
319 transporting_hero = true;
320
321 get_lua_context()->teletransporter_on_activated(*this);
322
323 if (!sound_id.empty()) {
324 Sound::play(sound_id);
325 }
326
327 get_game().set_current_map(destination_map_id, name, transition_style);
328 hero.set_xy(hero_x, hero_y);
329}
