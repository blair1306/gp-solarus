57class Q_WIDGETS_EXPORT QBoxLayout : public QLayout
58{
59 Q_OBJECT
60 Q_DECLARE_PRIVATE(QBoxLayout)
61public:
62 enum Direction { LeftToRight, RightToLeft, TopToBottom, BottomToTop,
63 Down = TopToBottom, Up = BottomToTop };
64
65 explicit QBoxLayout(Direction, QWidget *parent = 0);
66
67 ~QBoxLayout();
68
69 Direction direction() const;
70 void setDirection(Direction);
71
72 void addSpacing(int size);
73 void addStretch(int stretch = 0);
74 void addSpacerItem(QSpacerItem *spacerItem);
75 void addWidget(QWidget *, int stretch = 0, Qt::Alignment alignment = 0);
76 void addLayout(QLayout *layout, int stretch = 0);
77 void addStrut(int);
78 void addItem(QLayoutItem *);
79
80 void insertSpacing(int index, int size);
81 void insertStretch(int index, int stretch = 0);
82 void insertSpacerItem(int index, QSpacerItem *spacerItem);
83 void insertWidget(int index, QWidget *widget, int stretch = 0, Qt::Alignment alignment = 0);
84 void insertLayout(int index, QLayout *layout, int stretch = 0);
85 void insertItem(int index, QLayoutItem *);
86
87 int spacing() const;
88 void setSpacing(int spacing);
89
90 bool setStretchFactor(QWidget *w, int stretch);
91 bool setStretchFactor(QLayout *l, int stretch);
92 void setStretch(int index, int stretch);
93 int stretch(int index) const;
94
95 QSize sizeHint() const;
96 QSize minimumSize() const;
97 QSize maximumSize() const;
98
99 bool hasHeightForWidth() const;
100 int heightForWidth(int) const;
101 int minimumHeightForWidth(int) const;
102
103 Qt::Orientations expandingDirections() const;
104 void invalidate();
105 QLayoutItem *itemAt(int) const;
106 QLayoutItem *takeAt(int);
107 int count() const;
108 void setGeometry(const QRect&);
109
110private:
111 Q_DISABLE_COPY(QBoxLayout)
112};
