28/**
29 * \brief Creates a scrolling transition effect.
30 * \param direction direction of the transition (opening or closing)
31 */
32TransitionScrolling::TransitionScrolling(Transition::Direction direction):
33 Transition(direction),
34 both_maps_surface(nullptr),
35 scrolling_direction(0),
36 next_scroll_date(0),
37 dx(0),
38 dy(0) {
39
40}
41
42/**
43 * \brief Returns where the previous map should be blitted on
44 * both_maps_surface, for the specified scrolling direction.
45 * \param scrolling_direction The scrolling direction (0 to 3).
46 */
47Rectangle TransitionScrolling::get_previous_map_dst_position(
48 int scrolling_direction) {
49
50 const Surface* previous_map_surface = get_previous_surface();
51 Debug::check_assertion(previous_map_surface != nullptr, "Missing previous surface for scrolling");
52 const Size& camera_size = previous_map_surface->get_size();
53
54 Rectangle dst_position(0, 0);
55 if (scrolling_direction == 1) {
56 // Scroll to the north.
57 dst_position.set_y(camera_size.height);
58 }
59 else if (scrolling_direction == 2) {
60 // Scroll to the west.
61 dst_position.set_x(camera_size.width);
62 }
63
64 return dst_position;
65}

67/**
68 * \brief Starts this transition effect.
69 */
70void TransitionScrolling::start() {
71
72 if (get_direction() == Direction::CLOSING) {
73 return;
74 }
75
76 const Game* game = get_game();
77 Debug::check_assertion(game != nullptr, "Missing game for scrolling transition");
78 const Surface* previous_map_surface = get_previous_surface();
79 Debug::check_assertion(previous_map_surface != nullptr, "Missing previous surface for scrolling");
80
81 const Map& map = get_game()->get_current_map();
82
83 // get the scrolling direction
84 scrolling_direction = (map.get_destination_side() + 2) % 4;
85
86 const int scrolling_step = 5;
87
88 const Size& camera_size = previous_map_surface->get_size();
89 int width = camera_size.width;
90 int height = camera_size.height;
91 if (scrolling_direction % 2 == 0) {
92 // right or left
93 width *= 2;
94 dx = (scrolling_direction == 0) ? scrolling_step : -scrolling_step;
95 }
96 else {
97 height *= 2;
98 dy = (scrolling_direction == 3) ? scrolling_step : -scrolling_step;
99 }
100
101 // create a surface with the two maps
102 both_maps_surface = Surface::create(width, height);
103
104 // set the blitting rectangles
105
106 previous_map_dst_position = get_previous_map_dst_position(scrolling_direction);
107 current_map_dst_position = get_previous_map_dst_position((scrolling_direction + 2) % 4);
108 current_scrolling_position = previous_map_dst_position;
109 current_scrolling_position.set_size(camera_size);
110
111 next_scroll_date = System::now();
112}

114/**
115 * \brief Returns whether this transition effect needs the previous surface.
116 * \return true
117 */
118bool TransitionScrolling::needs_previous_surface() const {
119 return true;
120}
121
122/**
123 * \brief Returns whether the transition effect is started.
124 * \return false
125 */
126bool TransitionScrolling::is_started() const {
127 return !is_finished();
128}
129
130/**
131 * \brief Returns whether the transition effect is finished.
132 * \return true
133 */
134bool TransitionScrolling::is_finished() const {
135
136 if (get_direction() == Direction::CLOSING) {
137 return true;
138 }
139
140 return current_scrolling_position.get_x() == current_map_dst_position.get_x()
141 && current_scrolling_position.get_y() == current_map_dst_position.get_y();
142}
143
144/**
145 * \brief Makes a scrolling step.
146 */
147void TransitionScrolling::scroll() {
148
149 // Add dx,dy to current_scrolling_position, but don't exceed the final position.
150
151 if (dx > 0) {
152 current_scrolling_position.set_x(std::min(
153 current_scrolling_position.get_x() + dx,
154 current_map_dst_position.get_x())
155 );
156 }
157 else {
158 current_scrolling_position.set_x(std::max(
159 current_scrolling_position.get_x() + dx,
160 current_map_dst_position.get_x())
161 );
162 }
163
164 if (dy > 0) {
165 current_scrolling_position.set_y(std::min(
166 current_scrolling_position.get_y() + dy,
167 current_map_dst_position.get_y())
168 );
169 }
170 else {
171 current_scrolling_position.set_y(std::max(
172 current_scrolling_position.get_y() + dy,
173 current_map_dst_position.get_y())
174 );
175 }
176}

178/**
179 * \brief Notifies the transition effect that it was just suspended
180 * or resumed.
181 * \param suspended true if suspended, false if resumed.
182 */
183void TransitionScrolling::notify_suspended(bool suspended) {
184
185 if (!suspended) {
186 next_scroll_date += System::now() - get_when_suspended();
187 }
188}
189
190/**
191 * \brief Updates this transition effect.
192 *
193 * This function is called repeatedly while the transition exists.
194 */
195void TransitionScrolling::update() {
196
197 if (!is_started() || is_suspended()) {
198 return;
199 }
200
201 uint32_t now = System::now();
202 while (now >= next_scroll_date && !is_finished()) {
203 scroll();
204 next_scroll_date += 10;
205 }
206}
207
208/**
209 * \brief Draws the transition effect on a surface.
210 * \param dst_surface the surface to draw
211 */
212void TransitionScrolling::draw(Surface& dst_surface) {
213
214 if (get_direction() == Direction::CLOSING) {
215 return;
216 }
217
218 Surface* previous_surface = get_previous_surface();
219 Debug::check_assertion(previous_surface != nullptr,
220 "No previous surface defined for scrolling");
221
222 // draw the old map
223 previous_surface->draw(both_maps_surface, previous_map_dst_position.get_xy());
224
225 // draw the new map
226 dst_surface.draw(both_maps_surface, current_map_dst_position.get_xy());
227
228 // blit both surfaces
229 both_maps_surface->draw_region(
230 current_scrolling_position,
231 std::static_pointer_cast<Surface>(dst_surface.shared_from_this())
232 );
233}
