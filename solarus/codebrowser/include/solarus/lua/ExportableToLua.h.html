17#ifndef SOLARUS_EXPORTABLE_TO_LUA_H
18#define SOLARUS_EXPORTABLE_TO_LUA_H
19
20#include "solarus/Common.h"
21#include <memory>
22#include <string>
23
24namespace Solarus {
25
26class LuaContext;
27
28/**
29 * \brief Interface of a C++ type that can also exist as a Lua userdata.
30 *
31 * Use std::shared_ptr to safely share the object between C++ and Lua,
32 * no matter which world stops using it first.
33 */
34class ExportableToLua:
35 public std::enable_shared_from_this<ExportableToLua> {
36
37 public:
38
39 ExportableToLua();
40 virtual ~ExportableToLua();
41
42 LuaContext* get_lua_context() const;
43 void set_lua_context(LuaContext* lua_context);
44 bool is_known_to_lua() const;
45 void set_known_to_lua(bool known_to_lua);
46 bool is_with_lua_table() const;
47 void set_with_lua_table(bool with_lua_table);
48
49 /**
50 * \brief Returns the name identifying this type in Lua.
51 * \return the name identifying this type in Lua
52 */
53 virtual const std::string& get_lua_type_name() const = 0;
54
55 private:
56
57 LuaContext* lua_context; /**< The Solarus Lua API, or nullptr if
58 * not currently exported to Lua. */
59
60 bool known_to_lua; /**< Whether this object was exported to Lua
61 * at least once. */
62 bool with_lua_table; /**< Whether a Lua table was created to make
63 * this userdata indexable like a table. */
64
65};
66
67}
68
69#endif
