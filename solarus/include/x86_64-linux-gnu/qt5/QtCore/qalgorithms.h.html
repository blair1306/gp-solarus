226template <typename RandomAccessIterator, typename T>
227QT_DEPRECATED_X("Use std::lower_bound") Q_OUTOFLINE_TEMPLATE RandomAccessIterator qLowerBound(RandomAccessIterator begin, RandomAccessIterator end, const T &value)
228{
229 // Implementation is duplicated from QAlgorithmsPrivate to keep existing code
230 // compiling. We have to allow using *begin and value with different types,
231 // and then implementing operator< for those types.
232 RandomAccessIterator middle;
233 int n = end - begin;
234 int half;
235
236 while (n > 0) {
237 half = n >> 1;
238 middle = begin + half;
239 if (*middle < value) {
240 begin = middle + 1;
241 n -= half + 1;
242 } else {
243 n = half;
244 }
245 }
246 return begin;
247}

342template <typename RandomAccessIterator, typename T, typename LessThan>
343QT_DEPRECATED_X("Use std::sort") Q_OUTOFLINE_TEMPLATE void qSortHelper(RandomAccessIterator start, RandomAccessIterator end, const T &t, LessThan lessThan)
344{
345top:
346 int span = int(end - start);
347 if (span < 2)
348 return;
349
350 --end;
351 RandomAccessIterator low = start, high = end - 1;
352 RandomAccessIterator pivot = start + span / 2;
353
354 if (lessThan(*end, *start))
355 qSwap(*end, *start);
356 if (span == 2)
357 return;
358
359 if (lessThan(*pivot, *start))
360 qSwap(*pivot, *start);
361 if (lessThan(*end, *pivot))
362 qSwap(*end, *pivot);
363 if (span == 3)
364 return;
365
366 qSwap(*pivot, *end);
367
368 while (low < high) {
369 while (low < high && lessThan(*low, *end))
370 ++low;
371
372 while (high > low && lessThan(*end, *high))
373 --high;
374
375 if (low < high) {
376 qSwap(*low, *high);
377 ++low;
378 --high;
379 } else {
380 break;
381 }
382 }
383
384 if (lessThan(*low, *end))
385 ++low;
386
387 qSwap(*end, *low);
388 qSortHelper(start, low, t, lessThan);
389
390 start = low + 1;
391 ++end;
392 goto top;
393}

417template <typename RandomAccessIterator, typename T, typename LessThan>
418QT_DEPRECATED_X("Use std::merge") Q_OUTOFLINE_TEMPLATE void qMerge(RandomAccessIterator begin, RandomAccessIterator pivot, RandomAccessIterator end, T &t, LessThan lessThan)
419{
420 const int len1 = pivot - begin;
421 const int len2 = end - pivot;
422
423 if (len1 == 0 || len2 == 0)
424 return;
425
426 if (len1 + len2 == 2) {
427 if (lessThan(*(begin + 1), *(begin)))
428 qSwap(*begin, *(begin + 1));
429 return;
430 }
431
432 RandomAccessIterator firstCut;
433 RandomAccessIterator secondCut;
434 int len2Half;
435 if (len1 > len2) {
436 const int len1Half = len1 / 2;
437 firstCut = begin + len1Half;
438 secondCut = qLowerBound(pivot, end, *firstCut, lessThan);
439 len2Half = secondCut - pivot;
440 } else {
441 len2Half = len2 / 2;
442 secondCut = pivot + len2Half;
443 firstCut = qUpperBound(begin, pivot, *secondCut, lessThan);
444 }
445
446 qRotate(firstCut, pivot, secondCut);
447 const RandomAccessIterator newPivot = firstCut + len2Half;
448 qMerge(begin, firstCut, newPivot, t, lessThan);
449 qMerge(newPivot, secondCut, end, t, lessThan);
450}

452template <typename RandomAccessIterator, typename T, typename LessThan>
453QT_DEPRECATED_X("Use std::stable_sort") Q_OUTOFLINE_TEMPLATE void qStableSortHelper(RandomAccessIterator begin, RandomAccessIterator end, const T &t, LessThan lessThan)
454{
455 const int span = end - begin;
456 if (span < 2)
457 return;
458
459 const RandomAccessIterator middle = begin + span / 2;
460 qStableSortHelper(begin, middle, t, lessThan);
461 qStableSortHelper(middle, end, t, lessThan);
462 qMerge(begin, middle, end, t, lessThan);
463}

535Q_DECL_CONSTEXPR inline uint qPopulationCount(quint32 v)
536{
537#ifdef QALGORITHMS_USE_BUILTIN_POPCOUNT
538 return __builtin_popcount(v);
539#else
540 // See http://graphics.stanford.edu/~seander/bithacks.html#CountBitsSetParallel
541 return
542 (((v ) & 0xfff) * Q_UINT64_C(0x1001001001001) & Q_UINT64_C(0x84210842108421)) % 0x1f +
543 (((v >> 12) & 0xfff) * Q_UINT64_C(0x1001001001001) & Q_UINT64_C(0x84210842108421)) % 0x1f +
544 (((v >> 24) & 0xfff) * Q_UINT64_C(0x1001001001001) & Q_UINT64_C(0x84210842108421)) % 0x1f;
545#endif
546}
