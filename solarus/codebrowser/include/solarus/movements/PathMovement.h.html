17#ifndef SOLARUS_PATH_MOVEMENT_H
18#define SOLARUS_PATH_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/movements/PixelMovement.h"
22#include <cstdint>
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief Movement of an entity that follows a predetermined path.
29 *
30 * The path is a succession of basic moves, where each basic move
31 * is an 8-pixel trajectory towards one of the 8 main directions.
32 * The movement may or may not stop on obstacles.
33 * This movement can be used only on a map entity.
34 *
35 * A "path" refers to a succession of small 8-pixel trajectories.
36 * The notion of path is specific to PathMovement and its subclasses
37 * (like PathFindingMovement and RandomPathMovement).
38 * A "trajectory" is a move of 8 pixels in the sense of the PixelMovement
39 * class.
40 * A path is composed of several trajectories. The notion of trajectory is
41 * hidden from the public interface of PathMovement.
42 *
43 * TODO don't inherit PixelMovement
44 */
45class SOLARUS_API PathMovement: public PixelMovement {
46
47 public:
48
49 PathMovement(const std::string& path, int speed, bool loop, bool ignore_obstacles, bool snap_to_grid);
50
51 virtual void notify_object_controlled() override;
52 virtual void update() override;
53 virtual void set_suspended(bool suspended) override;
54 virtual bool is_finished() const override;
55
56 const std::string& get_path() const;
57 void set_path(const std::string& path);
58 int get_speed() const;
59 void set_speed(int speed);
60 bool get_loop() const;
61 void set_loop(bool loop);
62 bool get_snap_to_grid() const;
63 void set_snap_to_grid(bool snap_to_grid);
64
65 Point get_xy_change() const;
66 int get_current_direction() const;
67 int get_total_distance_covered() const;
68 virtual int get_displayed_direction4() const override;
69
70 static std::string create_random_path();
71
72 virtual const std::string& get_lua_type_name() const override;
73
74 protected:
75
76 virtual void notify_step_done(int step_index, bool success) override;
77
78 private:
79
80 static uint32_t speed_to_delay(int speed, int direction);
81
82 void start_next_elementary_move();
83 bool is_current_elementary_move_finished() const;
84 void restart();
85
86 void snap();
87 void set_snapping_trajectory(const Point& src, const Point& dst);
88
89 std::string initial_path; /**< the path: each character is a direction ('0' to '7')
90 * and corresponds to a trajectory of 8 pixels (performed by PixelMovement) */
91 std::string remaining_path; /**< the remaining part of the path */
92 int current_direction; /**< current element in the path (0 to 7) */
93 int total_distance_covered; /**< total number of pixels covered (each element of the path counts for 8) */
94 bool stopped_by_obstacle; /**< true if the movement was stopped by an obstacle */
95
96 int speed; /**< the movement speed in pixels per second (corrected for diagonal moves) */
97 bool loop; /**< should the path restart from the beginning once finished? */
98
99 // snapping
100 bool snap_to_grid; /**< indicates that the entity must be aligned to the grid before moving */
101 bool snapping; /**< indicates that the entity is currently being aligned to the grid */
102 uint32_t stop_snapping_date; /**< date when we stop trying to snap the entity if it is unsuccessful */
103
104 static const std::string
105 elementary_moves[]; /**< 8 pixel trajectory (in the PixelMovement sense) for each direction (0 to 7) */
106
107};
108
109}
110
111#endif
