27/**
28 * \brief Creates a fade-in or fade-out transition effect.
29 * \param direction direction of the transition effect (opening or closing)
30 * \param dst_surface The destination surface that will receive this
31 * transition.
32 */
33TransitionFade::TransitionFade(Direction direction, Surface& dst_surface):
34 Transition(direction),
35 finished(false),
36 alpha(-1),
37 next_frame_date(0),
38 dst_surface(&dst_surface),
39 colored(true),
40 transition_color(Color::black) {
41
42 if (direction == Direction::CLOSING) {
43 alpha_start = 256;
44 alpha_limit = 0;
45 alpha_increment = -8;
46 }
47 else {
48 alpha_start = 0;
49 alpha_limit = 256;
50 alpha_increment = 8;
51 }
52
53 set_delay(20);
54}
55
56/**
57 * \brief Sets the delay between two frames.
58 *
59 * The default delay is 20 ms.
60 *
61 * \param delay the new delay in milliseconds
62 */
63void TransitionFade::set_delay(uint32_t delay) {
64 this->delay = delay;
65}
66
67/**
68 * \brief Starts this transition effect.
69 */
70void TransitionFade::start() {
71 alpha = alpha_start;
72 next_frame_date = System::now();
73}
74
75/**
76 * \brief Returns whether this fade transition uses a foreground color.
77 * \return \c true if there is a foreground color,
78 * \c false if the transition changes the opacity of the destination surface
79 * (only possible for sotware destination surfaces).
80 */
81bool TransitionFade::is_colored() const {
82 return colored;
83}
84
85/**
86 * \brief Returns the foreground color of this fade transition.
87 *
88 * This function only makes sense when is_colored() is \c true.
89 *
90 * \return The color of the transition.
91 */
92const Color& TransitionFade::get_color() const {
93 return transition_color;
94}
95
96/**
97 * \brief Sets the foreground color of this fade transition.
98 *
99 * The default color is black.
100 *
101 * \param color The color of the transition.
102 */
103void TransitionFade::set_color(const Color& color) {
104
105 transition_color = color;
106 colored = true;
107}
108
109/**
110 * \brief Sets no foreground color for this fade transition.
111 *
112 * The transition effect will then change the opacity of the destination
113 * surface. This is only possible for software destination surfaces, so make
114 * sure you know what you are doing.
115 */
116void TransitionFade::clear_color() {
117
118 colored = false;
119}
120
121/**
122 * \brief Returns whether the transition effect is started and not finished yet.
123 * \return true if the transition effect is started
124 */
125bool TransitionFade::is_started() const {
126
127 return alpha != -1 && !is_finished();
128}
129
130/**
131 * \brief Returns whether the transition effect is finished.
132 * \return true if the transition effect is finished
133 */
134bool TransitionFade::is_finished() const {
135
136 return finished;
137}
138
139/**
140 * \brief Notifies the transition effect that it was just suspended
141 * or resumed.
142 * \param suspended true if suspended, false if resumed.
143 */
144void TransitionFade::notify_suspended(bool suspended) {
145
146 if (!suspended) {
147 next_frame_date += System::now() - get_when_suspended();
148 }
149}
150
151/**
152 * \brief Updates this transition effect.
153 *
154 * This function should be called repeatedly while the transition exists.
155 */
156void TransitionFade::update() {
157
158 if (!is_started() || is_suspended()) {
159 return;
160 }
161
162 uint32_t now = System::now();
163
164 // update the transition effect if needed
165 while (now >= next_frame_date && !finished) {
166 alpha += alpha_increment;
167 next_frame_date += delay; // 20 ms between two frame updates
168
169 finished = (alpha == alpha_limit);
170 }
171}
172
173/**
174 * \brief Draws the transition effect on a surface.
175 * \param dst_surface The destination surface.
176 */
177void TransitionFade::draw(Surface& dst_surface) {
178
179 // Draw the transition effect on the surface.
180 int alpha_impl = std::min(alpha, 255);
181
182 if (!colored) {
183 // Set the opacity on the surface.
184 // Only possible for software destinations.
185 Debug::check_assertion(dst_surface.is_software_destination()
186 || !Video::is_acceleration_enabled(),
187 "Cannot apply fade transition: this surface is in read-only mode");
188 dst_surface.set_opacity(alpha_impl);
189 }
190 else {
191 // Add a colored foreground surface with the appropriate opacity.
192 uint8_t r, g, b, a;
193 transition_color.get_components(r, g, b, a);
194 // A full opaque transition corresponds to a foreground with full alpha.
195 Color fade_color(r, g, b, 255 - std::min(alpha_impl, (int) a));
196
197 dst_surface.fill_with_color(fade_color);
198 }
199
200 this->dst_surface = &dst_surface;
201}
