186/**
187 * \brief Initializes the input event manager.
188 */
189void InputEvent::initialize() {
190
191 initialized = true;
192
193 // Initialize text events.
194 SDL_StartTextInput();
195
196 // Initialize the joypad.
197 set_joypad_enabled(true);
198}
199
200/**
201 * \brief Quits the input event manager.
202 */
203void InputEvent::quit() {
204
205 if (joystick != nullptr) {
206 SDL_JoystickClose(joystick);
207 }
208 SDL_StopTextInput();
209
210 joypad_enabled = false;
211 joystick = nullptr;
212 repeat_keyboard = false;
213 keys_pressed.clear();
214 joypad_axis_state.clear();
215 initialized = false;
216}

219 * \brief Returns whether the input event manager is initialized.
220 */
221bool InputEvent::is_initialized() {
222
223 return initialized;
224}

235/**
236 * \brief Returns the first event from the event queue, or nullptr
237 * if there is no event.
238 * \return The current event to handle.
239 * If it is invalid, the event was suppressed but there may be more events
240 * in the queue.
241 * Returns nullptr if there is no more event in the queue.
242 */
243std::unique_ptr<InputEvent> InputEvent::get_event() {
244
245 InputEvent* result = nullptr;
246 SDL_Event internal_event;
247 if (SDL_PollEvent(&internal_event)) {
248
249 // If this is a joypad axis event
250 if (internal_event.type == SDL_JOYAXISMOTION) {
251 // Determine the current state of the axis
252 int axis = internal_event.jaxis.axis;
253 int value = internal_event.jaxis.value;
254 int joystick_deadzone = 8000;
255 if (axis == 0) { // X axis
256 int x_dir = 0;
257 if (value < -joystick_deadzone) {
258 // Left of dead zone
259 x_dir = -1;
260 } else if(value > joystick_deadzone) {
261 // Right of dead zone
262 x_dir = 1;
263 } else {
264 x_dir = 0;
265 }
266 joypad_axis_state[axis] = x_dir;
267 } else if (axis == 1) { // Y axis
268 int y_dir = 0;
269 if (value < -joystick_deadzone) {
270 // Below dead zone
271 y_dir = -1;
272 } else if(value > joystick_deadzone) {
273 // Above dead zone
274 y_dir = 1;
275 } else {
276 y_dir = 0;
277 }
278 joypad_axis_state[axis] = y_dir;
279 }
280 }
281
282 // Check if keyboard events are correct.
283 // For some reason, when running Solarus from a Qt application
284 // (which is not recommended)
285 // multiple SDL_KEYUP events are generated when a key remains pressed
286 // (Qt/SDL conflict). This fixes most problems but not all of them.
287 else if (internal_event.type == SDL_KEYDOWN) {
288 SDL_Keycode key = internal_event.key.keysym.sym;
289 if (!keys_pressed.insert(key).second) {
290 // Already known as pressed: mark repeated.
291 internal_event.key.repeat = 1;
292 }
293 }
294 else if (internal_event.type == SDL_KEYUP) {
295 SDL_Keycode key = internal_event.key.keysym.sym;
296 if (keys_pressed.erase(key) == 0) {
297 // Already known as not pressed: mark repeated.
298 internal_event.key.repeat = 1;
299 }
300 }
301
302 // Always return a Solarus event if an SDL event occurred, so that
303 // multiple SDL events in the same frame are all treated.
304 result = new InputEvent(internal_event);
305 }
306
307 return std::unique_ptr<InputEvent>(result);
308}

421 * \param finger_id A finger ID.

424bool InputEvent::is_finger_down(int finger_id) {

435/**
436 * \brief Returns the state of a joypad axis.
437 * \param axis Index of a joypad axis.
438 * \return The state of that axis:
439 * -1 (left or up), 0 (centered) or 1 (right or down).
440 */
441int InputEvent::get_joypad_axis_state(int axis) {
442
443 if (joystick == nullptr) {
444 return 0;
445 }
446
447 int state = SDL_JoystickGetAxis(joystick, axis);
448
449 int result;
450 if (std::abs(state) < 10000) {
451 result = 0;
452 }
453 else {
454 result = (state > 0) ? 1 : -1;
455 }
456
457 return result;
458}

460/**
461 * \brief Returns the direction of a joypad hat.
462 * \param hat Index of a joypad hat.
463 * \return The direction of that hat (0 to 7, or -1 if centered).
464 */
465int InputEvent::get_joypad_hat_direction(int hat) {
466
467 if (joystick == nullptr) {
468 return -1;
469 }
470
471 int state = SDL_JoystickGetHat(joystick, hat);
472 int result = -1;
473
474 switch (state) {
475
476 case SDL_HAT_RIGHT:
477 result = 0;
478 break;
479
480 case SDL_HAT_RIGHTUP:
481 result = 1;
482 break;
483
484 case SDL_HAT_UP:
485 result = 2;
486 break;
487
488 case SDL_HAT_LEFTUP:
489 result = 3;
490 break;
491
492 case SDL_HAT_LEFT:
493 result = 4;
494 break;
495
496 case SDL_HAT_LEFTDOWN:
497 result = 5;
498 break;
499
500 case SDL_HAT_DOWN:
501 result = 6;
502 break;
503
504 case SDL_HAT_RIGHTDOWN:
505 result = 7;
506 break;
507
508 }
509
510 return result;
511}

513/**
514 * \brief Gets the x and y position of the mouse.
515 * Values are in quest size coordinates.
516 * \return The mouse position in quest coordinates.
517 */
518Point InputEvent::get_global_mouse_position() {
519
520 int x, y;
521
522 SDL_GetMouseState(&x, &y);
523
524 return Video::window_to_quest_coordinates(Point(x, y));
525}
526
527/**
528 * \brief Gets the x and y position of the finger.
529 * Values are in quest size coordinates.
530 * \param finger_id The ID of the finger.
531 * \param[out] finger_xy The x and y position of the finger in quest coordinates.
532 * \return \c false if the finger is not pressed.
533 */
534bool InputEvent::get_global_finger_position(int finger_id, Point& finger_xy) {
535
536 SDL_Finger* finger;
537
538 for (int i = 0; i < SDL_GetNumTouchDevices(); ++i) {
539 finger = SDL_GetTouchFinger(SDL_GetTouchDevice(i), finger_id);
540
541 if (finger != NULL) {
542 const Size window_size = Video::get_window_size();
543 const int x = finger->x * static_cast<float>(window_size.width);
544 const int y = finger->y * static_cast<float>(window_size.height);
545
546 finger_xy = Video::window_to_quest_coordinates(Point(x, y));
547 return true;
548 }
549 }
550
551 return false;
552}

554/**
555 * \brief Gets the pressure of the finger.
556 * \param finger_id The ID of the finger.
557 * \param[out] finger_pressure The pressure of the finger
558 * \return \c false if the finger is not pressed.
559 */
560bool InputEvent::get_global_finger_pressure(int finger_id, float& finger_pressure) {
561
562 SDL_Finger* finger;
563
564 for (int i = 0; i < SDL_GetNumTouchDevices(); ++i) {
565 finger = SDL_GetTouchFinger(SDL_GetTouchDevice(i), finger_id);
566
567 if (finger != NULL) {
568 finger_pressure = finger->pressure;
569 return true;
570 }
571 }
572
573 return false;
574}

643/**
644 * \brief Returns whether this event is a keyboard event
645 * corresponding to pressing a key.
646 * \return true if this is a key pressed event

648bool InputEvent::is_keyboard_key_pressed() const {
649

652}

654/**
655 * \brief Returns whether this event is a keyboard event
656 * corresponding to pressing a specific key.
657 * \param key the key to test
658 * \return true if this event corresponds to pressing that key
659 */
660bool InputEvent::is_keyboard_key_pressed(KeyboardKey key) const {
661
662 return is_keyboard_key_pressed()
663 && get_keyboard_key() == key;
664}
665
666/**
667 * \brief Returns whether this event is a keyboard event
668 * corresponding to pressing one of the specified keys.
669 * \param keys an array of the keys to test, terminated by KEY_NONE
670 * \return true if this event corresponds to pressing one of those keys
671 */
672bool InputEvent::is_keyboard_key_pressed(const KeyboardKey* keys) const {
673
674 while (*keys != KeyboardKey::NONE) {
675
676 if (is_keyboard_key_pressed(*keys)) {
677 return true;
678 }
679 keys++;
680 }
681
682 return false;
683}

897/**
898 * \brief Enables or disables joypad support.
899 *
900 * Joypad support may be enabled even without any joypad plugged.
901 *
902 * \param joypad_enabled true to enable joypad support, false to disable it.
903 */
904void InputEvent::set_joypad_enabled(bool joypad_enabled) {
905
906 if (joypad_enabled != is_joypad_enabled()) {
907
908 InputEvent::joypad_enabled = joypad_enabled;
909
910 if (joystick != nullptr) {
911 SDL_JoystickClose(joystick);
912 joystick = nullptr;
913 joypad_axis_state.clear();
914 }
915
916 if (joypad_enabled && SDL_NumJoysticks() > 0) {
917 SDL_InitSubSystem(SDL_INIT_JOYSTICK);
918 joystick = SDL_JoystickOpen(0);
919 joypad_axis_state.assign(SDL_JoystickNumAxes(joystick), 0);
920 }
921 else {
922 SDL_JoystickEventState(SDL_IGNORE);
923 SDL_QuitSubSystem(SDL_INIT_JOYSTICK);
924 }
925
926 Logger::info(std::string("Joypad support enabled: ") + (joypad_enabled ? "true" : "false"));
927 }
928}

1106 break;
1107

1109 result = 6;
1110 break;

1343/**
1344 * \brief Returns the direction that was pressed or released
1345 * during this keyboard or joypad event.
1346 *
1347 * If this is not a keyboard or joypad event, or if
1348 * the control was not a direction, -1 is returned.
1349 *
1350 * \return the direction (0 to 7) or -1 if there is no direction
1351 */
1352int InputEvent::get_direction() const {
1353
1354 int result = -1;
1355
1356 if (is_keyboard_direction_key_pressed()) {
1357
1358 switch (get_keyboard_key()) {
1359
1360 case KeyboardKey::RIGHT:
1361 result = 0;
1362 break;
1363
1364 case KeyboardKey::UP:
1365 result = 2;
1366 break;
1367
1368 case KeyboardKey::LEFT:
1369 result = 4;
1370 break;
1371
1372 case KeyboardKey::DOWN:
1373 result = 6;
1374 break;
1375
1376 default:
1377 break;
1378 }
1379 }
1380 else if (is_joypad_axis_moved() && !is_joypad_axis_centered()) {
1381
1382 if (get_joypad_axis() % 2 == 0) {
1383 // we assume the axis is horizontal
1384 result = (get_joypad_axis_state() > 0) ? 0 : 4;
1385 }
1386 else {
1387 // we assume the axis is vertical
1388 result = (get_joypad_axis_state() > 0) ? 6 : 2;
1389 }
1390 }
1391 else if (is_joypad_hat_moved()) {
1392 result = get_joypad_hat_direction();
1393 }
1394
1395 return result;
1396}
