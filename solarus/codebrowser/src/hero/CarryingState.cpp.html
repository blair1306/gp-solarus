29/**
30 * \brief Constructor.
31 * \param hero The hero controlled by this state.
32 * \param carried_object The item to carry.
33 */
34Hero::CarryingState::CarryingState(
35 Hero& hero, const std::shared_ptr<CarriedObject>& carried_object
36):
37 PlayerMovementState(hero, "carrying"),
38 carried_object(carried_object) {
39
40 Debug::check_assertion(carried_object != nullptr, "Missing carried object");
41}
42
43/**
44 * \brief Starts this state.
45 * \param previous_state the previous state
46 */
47void Hero::CarryingState::start(const State* previous_state) {
48
49 PlayerMovementState::start(previous_state);
50
51 if (is_current_state()) {
52 get_sprites().set_lifted_item(carried_object);
53
54 // action icon "throw"
55 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_THROW);
56 }
57}
58
59/**
60 * \brief Stops this state.
61 * \param next_state the next state
62 */
63void Hero::CarryingState::stop(const State* next_state) {
64
65 PlayerMovementState::stop(next_state);
66
67 get_sprites().set_lifted_item(nullptr);
68 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
69
70 if (carried_object != nullptr) {
71
72 switch (next_state->get_previous_carried_object_behavior()) {
73
74 case CarriedObject::BEHAVIOR_THROW:
75 throw_item();
76 break;
77
78 case CarriedObject::BEHAVIOR_DESTROY:
79 case CarriedObject::BEHAVIOR_KEEP:
80 carried_object = nullptr;
81 break;
82
83 default:
84 Debug::die("Invalid carried object behavior");
85 }
86 }
87}

89/**
90 * \brief Changes the map.
91 * \param map the new map
92 */
93void Hero::CarryingState::set_map(Map& map) {
94
95 PlayerMovementState::set_map(map);
96
97 // the hero may go to another map while carrying an item
98 if (carried_object != nullptr) {
99 carried_object->set_map(map);
100 }
101}

103/**
104 * \copydoc Hero::PlayerMovementState::notify_position_changed
105 */
106void Hero::CarryingState::notify_position_changed() {
107
108 PlayerMovementState::notify_position_changed();
109
110 if (carried_object != nullptr) {
111 // Update the position of the carried object too.
112 carried_object->update();
113 }
114}

116/**
117 * \brief Notifies this state that the layer has changed.
118 */
119void Hero::CarryingState::notify_layer_changed() {
120
121 PlayerMovementState::notify_layer_changed();
122
123 if (carried_object != nullptr) {
124 carried_object->set_layer(get_entity().get_layer());
125 }
126}

128/**
129 * \brief Notifies this state that the game was just suspended or resumed.
130 * \param suspended true if the game is suspended
131 */
132void Hero::CarryingState::set_suspended(bool suspended) {
133
134 PlayerMovementState::set_suspended(suspended);
135
136 if (carried_object != nullptr) {
137 carried_object->set_suspended(suspended);
138 }
139}
140
141/**
142 * \brief Updates this state.
143 */
144void Hero::CarryingState::update() {
145
146 PlayerMovementState::update();
147
148 // The state may have just been changed for example by a jumper.
149 if (is_current_state()) {
150 carried_object->update();
151
152 if (!is_suspended()) {
153
154 if (carried_object->is_broken()) {
155 carried_object = nullptr;
156 Hero& hero = get_entity();
157 hero.set_state(new FreeState(hero));
158 }
159 }
160 }
161}
162
163/**
164 * \brief Notifies this state that the action command was just pressed.
165 */
166void Hero::CarryingState::notify_action_command_pressed() {
167
168 if (get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_THROW) {
169 throw_item();
170 Hero& hero = get_entity();
171 hero.set_state(new FreeState(hero));
172 }
173}
174
175/**
176 * \brief Throws the item carried.
177 *
178 * This function is called when the player presses the action key
179 * or when another state becomes the current state.
180 */
181void Hero::CarryingState::throw_item() {
182
183 carried_object->throw_item(get_sprites().get_animation_direction());
184 get_entities().add_entity(carried_object);
185 carried_object = nullptr;
186}
