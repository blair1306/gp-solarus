31/**
32 * \brief A list of entities on a layer of the map.
33 *
34 * The entity at index 0 is the most to the back.
35 * Tiles are always before all other entities in the list.
36 */
37struct EntityDataList {
38 std::deque<EntityData> entities; /**< All entities of a layer, with tiles first. */
39 int num_tiles = 0; /**< Number of tiles in the list.
40 * Dynamic entities are after this index. */
41};
42
43/**
44 * \brief The layer and index of an entity in the map.
45 *
46 * This is how entities are identified on a map.
47 * There is also an additional index on the name for
48 * entities that have a name.
49 */
50struct EntityIndex {
51
52 public:
53
54 /**
55 * \brief Creates an invalid index.
56 */
57 EntityIndex():
58 layer(0),
59 order(-1) {
60 }
61
62 /**
63 * \brief Creates an index.
64 * \param layer The layer.
65 * \param order The order in that layer.
66 */
67 EntityIndex(int layer, int order):
68 layer(layer),
69 order(order) {
70 }
71
72 /**
73 * \brief Returns whether this is a valid index.
74 */
75 bool is_valid() const {
76 return order != -1;
77 }
78
79 bool operator==(const EntityIndex& other) const {
80 return other.layer == layer &&
81 other.order == order;
82 }
83
84 bool operator!=(const EntityIndex& other) const {
85 return !(*this == other);
86 }
87
88 bool operator<=(const EntityIndex& other) const {
89 if (layer < other.layer) {
90 return true;
91 }
92 if (layer > other.layer) {
93 return false;
94 }
95 return order <= other.order;
96 }
97
98 bool operator<(const EntityIndex& other) const {
99 if (layer < other.layer) {
100 return true;
101 }
102 if (layer > other.layer) {
103 return false;
104 }
105 return order < other.order;
106 }
107
108 bool operator>=(const EntityIndex& other) const {
109 if (layer > other.layer) {
110 return true;
111 }
112 if (layer < other.layer) {
113 return false;
114 }
115 return order >= other.order;
116 }
117
118 bool operator>(const EntityIndex& other) const {
119 if (layer > other.layer) {
120 return true;
121 }
122 if (layer < other.layer) {
123 return false;
124 }
125 return order > other.order;
126 }
127
128 int layer; /**< Layer of the entity on the map. */
129 int order; /**< Z order of the entity in that layer.
130 * -1 means an invalid index. */
131
132};

143 Size get_size() const;

145 Point get_location() const;
146 void set_location(const Point& location);
147 int get_min_layer() const;
148 void set_min_layer(int min_layer);
149 int get_max_layer() const;
150 void set_max_layer(int max_layer);
151 bool is_valid_layer(int layer) const;
152 bool has_world() const;

154 void set_world(const std::string& world);
155 bool has_floor() const;
156 int get_floor() const;
157 void set_floor(int floor);
158 const std::string& get_tileset_id() const;
159 void set_tileset_id(const std::string& tileset_id);
160 bool has_music() const;
161 const std::string& get_music_id() const;
162 void set_music_id(const std::string& music_id);
163
164 int get_num_entities() const;
165 int get_num_entities(int layer) const;
166 int get_num_tiles(int layer) const;
167 int get_num_dynamic_entities(int layer) const;
168
169 EntityIndex set_entity_layer(const EntityIndex& src_index, int dst_layer);
170 void set_entity_order(const EntityIndex& src_index, int dst_order);
171 EntityIndex bring_entity_to_front(const EntityIndex& index);
172 EntityIndex bring_entity_to_back(const EntityIndex& index);
173
174 bool entity_exists(const EntityIndex& index) const;
175 const EntityData& get_entity(const EntityIndex& index) const;
176 EntityData& get_entity(const EntityIndex& index);

179 EntityData* get_entity_by_name(const std::string& name);

181 const std::map<std::string, EntityIndex>& get_named_entities_indexes() const;

184 EntityIndex add_entity(const EntityData& entity);

190

192
193 private:
194
195 const std::deque<EntityData>& get_entities(int layer) const;
196 std::deque<EntityData>& get_entities(int layer);
197
198 int min_layer; /**< Lowest layer of the map (0 or less). */
199 int max_layer; /**< Highest layer of the map (0 or more). */
200 Size size; /**< Size of the map in pixels. */
201 std::string world; /**< World of the map or an empty string. */
202 Point location; /**< Coordinates of the upper-left corner of the map in its world. */
203 int floor; /**< Floor of the map or NO_FLOOR. */
204 std::string tileset_id; /**< Tileset to use as skin for the map. */
205 std::string music_id; /**< Background music id or "none" or "same". */
206
207 std::map<int, EntityDataList>
208 entities; /**< The entities on each layer. */
209 std::map<std::string, EntityIndex>
210 named_entities; /**< Entities indexed by their name. */
211
212};
