32/**
33 * \brief Represents an enemy.
34 *
35 * This class stores the attack and defense properties of the enemy.
36 * Every enemy sprite must have at least the following animations
37 * and with the four main directions:
38 * - "walking": used when the enemy is in its normal state
39 * - "hurt": used when it was just hurt
40 * If the enemy can be immobilized (i.e. it is sensible to attacks that immobilize it),
41 * the following animations are also required:
42 * - "immobilized": used while the enemy is immobilized
43 * - "shaking": used when the immobilized enemy is about to get back to its normal state.
44 * The animation of its sprites automatically switches between those animations
45 * depending on its current movement and the attacks it is subject to.
46 * Additional animations may be defined.
47 */
48class Enemy: public Entity {
49
50 public:
51
52 static constexpr EntityType ThisType = EntityType::ENEMY;
53
54 /**
55 * \brief Defines the style of sounds and animations played when an enemy
56 * is hurt or killed.
57 */
58 enum class HurtStyle {
59 NORMAL, /**< "enemy_hurt" (and if necessary "enemy_killed") is played */
60 MONSTER, /**< "monster_hurt" (and if necessary "enemy_killed") is played */
61 BOSS, /**< "boss_hurt" or "boss_killed" is played and explosions are created */
62 };
63
64 /**
65 * \brief Defines how an enemy behaves with obstacles.
66 */
67 enum class ObstacleBehavior {
68 NORMAL, /**< the enemy is on the ground and stops on normal obstacles */
69 FLYING, /**< the enemy ignores holes, lava, water, etc. */
70 SWIMMING /**< the enemy can move in water */
71 };

211 // enemy characteristics
212 std::string breed; /**< breed of the enemy (determines its sprites and behavior) */
213
214 int damage_on_hero; /**< number of heart quarters the player loses when he gets hurt by this enemy;
215 * this number is divided depending on the hero's tunic number (default: 1) */
216 int life; /**< number of health points of the enemy (default: 1) */
217 HurtStyle hurt_style; /**< style of sounds and animations when this enemy gets hurt
218 * (default: HURT_NORMAL) */
219 bool pushed_back_when_hurt; /**< indicates whether the enemy is pushed back when it gets hurt by the hero
220 * (default: true) */
221 bool push_hero_on_sword; /**< indicates whether the hero is pushed back when he hurts the enemy with his
222 * sword (default: false) */
223 bool can_hurt_hero_running; /**< indicates that the enemy can attack the hero even when the hero is running */
224 int minimum_shield_needed; /**< shield number needed by the hero to avoid the attack of this enemy,
225 * or 0 to make the attack unavoidable (default: 0) */
226 std::map<EnemyAttack, EnemyReaction>
227 attack_reactions; /**< how the enemy reacts to each attack
228 * (by default, it depends on the attacks) */
229 std::string savegame_variable; /**< name of the boolean variable indicating whether this enemy is killed,
230 * or an empty string if it is not saved */
231 bool traversable; /**< Whether this enemy can be traversed by other entities. */
232 ObstacleBehavior obstacle_behavior; /**< behavior with obstacles */
233
234 // enemy state
235 bool being_hurt; /**< indicates that the enemy is being hurt */
236 uint32_t stop_hurt_date; /**< date when the enemy stops being hurt */
237 bool invulnerable; /**< indicates that the enemy cannot be hurt for now */
238 uint32_t vulnerable_again_date; /**< date when the enemy can be hurt again */
239 bool can_attack; /**< indicates that the enemy can currently attack the hero */
240 uint32_t can_attack_again_date; /**< date when the enemy can attack again (0 means never) */
241 bool immobilized; /**< indicates that the enemy is currently immobilized */
242 uint32_t start_shaking_date; /**< date when the enemy shakes */
243 uint32_t end_shaking_date; /**< date when the enemy stops shaking and walks again */
244 bool dying_animation_started; /**< whether the dying animation was started */
245
246 Treasure treasure; /**< pickable item that appears when this enemy gets killed */
247
248 // boss or mini-boss
249 bool exploding; /**< indicates that the boss is dying and some explosions are triggered on him */
250 int nb_explosions; /**< number of explosions already played */
251 uint32_t next_explosion_date; /**< date of the next explosion */
252
253};
