37/**
38 * \brief Reads the resource list, the quest properties and stores them.
39 */
40void initialize() {
41
42 // Read the quest resource list file.
43 QuestResources& resources = get_resources();
44 resources.import_from_quest_file("project_db.dat");
45
46 // Read the quest properties file.
47 QuestProperties& properties = get_properties();
48
49 const std::string file_name("quest.dat");
50 lua_State* l = luaL_newstate();
51 const std::string& buffer = QuestFiles::data_file_read(file_name);
52 int load_result = luaL_loadbuffer(l, buffer.data(), buffer.size(), file_name.c_str());
53
54 if (load_result != 0) {
55 // Syntax error in quest.dat.
56 // Loading quest.dat failed.
57 // There may be a syntax error, or this is a quest for Solarus 0.9.
58 // There was no version number at that time.
59
60 const std::string error_message = lua_tostring(l, -1);
61 lua_close(l);
62
63 if (std::string(buffer).find("[info]")) {
64 // Quest format of Solarus 0.9.
65 Debug::die(std::string("This quest is made for Solarus 0.9 but you are running Solarus ")
66 + SOLARUS_VERSION);
67 }
68 else {
69 Debug::die(std::string("Failed to load quest.dat: ") + error_message);
70 }
71 }
72
73 // Normal case.
74 properties.import_from_lua(l);
75 lua_close(l);
76
77 initialized = true;
78}

80/**
81 * \brief Clears the loaded quest resource list.
82 */
83void quit() {
84
85 get_resources().clear();
86 get_strings().clear();
87 get_dialogs().clear();
88
89 initialized = false;
90}

125/**
126 * \brief Returns whether there exists an element with the specified id.
127 * \param resource_type A type of resource.
128 * \param id The id to look for.
129 * \return \c true if there resource_exists an element with the specified id in this
130 * resource type.
131 */
132bool resource_exists(ResourceType resource_type, const std::string& id) {
133
134 return get_resources().exists(resource_type, id);
135}
136
137/**
138 * \brief Returns the list of element IDs of the specified resource type.
139 * \param resource_type A type of resource.
140 * \return The IDs of all declared element of this type, in their declaration
141 * order.
142 */
143const std::map<std::string, std::string>&
144get_resources(ResourceType resource_type) {
145
146 return get_resources().get_elements(resource_type);
147}

149/**
150 * \brief Returns whether a language exists for the current quest.
151 * \param language_code Code of the language to test.
152 * \return \c true if this language exists.

154bool has_language(const std::string& language_code) {

156 return resource_exists(ResourceType::LANGUAGE, language_code);

159/**
160 * \brief Sets the current language.
161 *
162 * The language-specific data will be loaded from the directory of this language.
163 * This function must be called before the first language-specific file is loaded.
164 *
165 * \param language_code Code of the language to set.
166 */
167void set_language(const std::string& language_code) {
168
169 Debug::check_assertion(has_language(language_code),
170 std::string("No such language: '") + language_code + "'");
171
172 get_language() = language_code;
173
174 // Read the quest string list file.
175 get_strings().clear();
176 get_strings().import_from_quest_file("text/strings.dat", true);
177
178 // Read the quest dialog list file.
179 DialogResources resources;
180 std::map<std::string, Dialog>& dialogs = get_dialogs();
181
182 bool success = resources.import_from_quest_file("text/dialogs.dat", true);
183
184 // Create dialogs.
185 dialogs.clear();
186 if (success) {
187 for (const auto& kvp : resources.get_dialogs()) {
188
189 const std::string& id = kvp.first;
190 const DialogData& data = kvp.second;
191
192 Dialog dialog;
193 dialog.set_id(id);
194 dialog.set_text(data.get_text());
195
196 for (const auto& pkvp : data.get_properties()) {
197 dialog.set_property(pkvp.first, pkvp.second);
198 }
199
200 dialogs.emplace(id, dialog);
201 }
202 }
203
204 Logger::info(std::string("Language: ") + language_code);
205}

207/**
208 * \brief Returns the current language.
209 *
210 * The language-specific data are be loaded from the directory of this language.
211 *
212 * \return code of the language, or an empty string if no language is set
213 */
214std::string& get_language() {
215
216 // The language code must be in a function to avoid static initialization
217 // order problems.
218 static std::string language_code;
219 return language_code;
220}
221
222/**
223 * \brief Returns the user-friendly name of a language for this quest.
224 * \param language_code Code of a language.
225 * \return Name of this language of an empty string.
226 */
227std::string get_language_name(const std::string& language_code) {
228
229 const std::map<std::string, std::string>& languages =
230 get_resources(ResourceType::LANGUAGE);
231
232 const auto& it = languages.find(language_code);
233 if (it != languages.end()) {
234 return it->second;
235 }
236
237 return "";
238}

252/**
253 * \brief Returns whether a string exists in the language-specific file
254 * "text/strings.dat" for the current language.

263/**
264 * \brief Returns a string stored in the language-specific file
265 * "text/strings.dat" for the current language.
266 * \param key Id of the string to retrieve. It must exist.

269const std::string& get_string(const std::string& key) {

274/**
275 * \brief Returns the dialog list of the current quest.
276 * \return The current quest dialog list.
277 */
278std::map<std::string, Dialog>& get_dialogs() {
279
280 // The dialog objects must be in a function to avoid static initialization

282 static std::map<std::string, Dialog> dialogs;
283 return dialogs;
284}
285
286/**
287 * \brief Returns whether the specified dialog exists.
288 * \param dialog_id Id of the dialog to test.
289 * \return true if such a dialog exists.
290 */
291bool dialog_exists(const std::string& dialog_id) {
292
293 const auto& dialogs = get_dialogs();
294 return dialogs.find(dialog_id) != dialogs.end();
295}
296
297/**
298 * \brief Returns a dialog stored in the language-specific file
299 * "text/dialogs.dat".
300 * \param dialog_id id of the dialog to retrieve
301 * \return the corresponding localized dialog
302 */
303const Dialog& get_dialog(const std::string& dialog_id) {
304
305 Debug::check_assertion(dialog_exists(dialog_id), std::string(
306 "No such dialog: '") + dialog_id + "'");
307 return get_dialogs()[dialog_id];
308}
