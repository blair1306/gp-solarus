29/**
30 * \brief Creates an explosion.
31 * \param name Name identifying the entity on the map or an empty string.
32 * \param layer layer of the explosion
33 * \param xy coordinates of the center of the explosion
34 * \param with_damage true to hurt the hero and the enemies
35 */
36Explosion::Explosion(const std::string& name, int layer,
37 const Point& xy, bool with_damage):
38 Entity(name, 0, layer, xy, Size(48, 48)) {
39
40 set_collision_modes(CollisionMode::COLLISION_SPRITE | CollisionMode::COLLISION_OVERLAPPING);
41
42 // initialize the entity
43 const SpritePtr& sprite = create_sprite("entities/explosion");
44 sprite->enable_pixel_collisions();
45
46 if (with_damage) {
47 set_size(48, 48);
48 set_origin(24, 24);
49 }
50}

52/**
53 * \brief Returns the type of entity.
54 * \return the type of entity
55 */
56EntityType Explosion::get_type() const {
57 return ThisType;
58}
59
60/**
61 * \brief Updates this entity.
62 */
63void Explosion::update() {
64
65 Entity::update();
66
67 const SpritePtr& sprite = get_sprite();
68 if (sprite != nullptr && sprite->is_animation_finished()) {
69 remove_from_map();
70 }
71}

73/**
74 * \brief Notifies this entity that the frame of one of its sprites has just changed.
75 * \param sprite the sprite
76 * \param animation the current animation
77 * \param frame the new frame
78 */
79void Explosion::notify_sprite_frame_changed(Sprite& /* sprite */, const std::string& /* animation */, int frame) {
80
81 if (frame == 1) {
82 // also detect non-pixel precise collisions
83 check_collision_with_detectors();
84 }
85}

87/**
88 * \copydoc Entity::notify_collision(Entity&, Sprite&, Sprite&)
89 */
90void Explosion::notify_collision(
91 Entity& other_entity,
92 Sprite& /* this_sprite */,
93 Sprite& other_sprite
94) {
95 other_entity.notify_collision_with_explosion(*this, other_sprite);
96}
97
98/**
99 * \brief This function is called when a the sprite of a switch
100 * detects a pixel-precise collision with a sprite of this entity.
101 * \param sw the switch
102 * \param sprite_overlapping the sprite of the current entity that collides with the switch
103 */
104void Explosion::notify_collision_with_switch(Switch& sw, Sprite& /* sprite_overlapping */) {
105
106 sw.try_activate();
107}
108
109/**
110 * \brief This function is called when a the sprite of a crystal
111 * detects a pixel-precise collision with a sprite of this entity.
112 * \param crystal the crystal
113 * \param sprite_overlapping the sprite of the current entity that collides with the crystal
114 */
115void Explosion::notify_collision_with_crystal(Crystal& crystal, Sprite& /* sprite_overlapping */) {
116
117 crystal.activate(*this);
118}
119
120/**
121 * \brief This function is called when a sensor detects a collision with this entity.
122 * \param sensor a sensor
123 * \param collision_mode the collision mode that detected the collision
124 */
125void Explosion::notify_collision_with_sensor(Sensor& sensor, CollisionMode collision_mode) {
126
127 if (collision_mode == COLLISION_OVERLAPPING) {
128 sensor.notify_collision_with_explosion(*this, collision_mode);
129 }
130}
131
132/**
133 * \brief This function is called when an enemy's sprite collides with a sprite of this entity
134 * \param enemy the enemy
135 * \param enemy_sprite the enemy's sprite that overlaps a sprite of this entity
136 * \param this_sprite this entity's sprite that overlaps the enemy's sprite
137 */
138void Explosion::notify_collision_with_enemy(Enemy& enemy, Sprite& enemy_sprite, Sprite& /* this_sprite */) {
139
140 try_attack_enemy(enemy, enemy_sprite);
141}
142
143/**
144 * \brief Attacks the specified enemy if possible.
145 *
146 * This function is called by this explosion when it detects an enemy, or by an enemy who detects this explosion.
147 *
148 * \param enemy the enemy to attack
149 * \param enemy_sprite the enemy's sprite detected by the explosion
150 */
151void Explosion::try_attack_enemy(Enemy& enemy, Sprite& enemy_sprite) {
152
153 // see if the enemy was already hurt by this explosion
154 auto it = std::find(victims.begin(), victims.end(), &enemy);
155 if (it == victims.end()) {
156 enemy.try_hurt(EnemyAttack::EXPLOSION, *this, &enemy_sprite);
157 }
158}
159
160/**
161 * \copydoc Entity::notify_attacked_enemy
162 */
163void Explosion::notify_attacked_enemy(
164 EnemyAttack /* attack */,
165 Enemy& victim,
166 const Sprite* /* victim_sprite */,
167 EnemyReaction::Reaction& result,
168 bool /* killed */) {
169
170 if (result.type != EnemyReaction::ReactionType::IGNORED) {
171 victims.push_back(&victim);
172 }
173}
