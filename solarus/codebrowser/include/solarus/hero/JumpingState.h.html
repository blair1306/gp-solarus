20#include "solarus/hero/HeroState.h"
21#include <memory>
22
23namespace Solarus {
24
25class JumpMovement;
26
27/**
28 * \brief The state "Jumping" of the hero.
29 */
30class Hero::JumpingState: public HeroState {
31
32 public:
33
34 JumpingState(
35 Hero& hero,
36 int direction8,
37 int distance,
38 bool ignore_obstacles,
39 bool with_sound
40 );
41
42 virtual void start(const State* previous_state) override;
43 virtual void stop(const State* next_state) override;
44 virtual void set_map(Map& map) override;
45 virtual void update() override;
46 virtual void set_suspended(bool suspended) override;
47 virtual void notify_layer_changed() override;
48 virtual int get_wanted_movement_direction8() const override;
49 virtual bool can_start_gameover_sequence() const override;
50 virtual bool is_touching_ground() const override;
51 virtual bool is_stairs_obstacle(const Stairs& stairs) const override;
52 virtual bool is_sensor_obstacle(const Sensor& sensor) const override;
53 virtual bool is_separator_obstacle(const Separator& separator) const override;
54 virtual bool can_avoid_deep_water() const override;
55 virtual bool can_avoid_hole() const override;
56 virtual bool can_avoid_ice() const override;
57 virtual bool can_avoid_lava() const override;
58 virtual bool can_avoid_prickle() const override;
59 virtual bool can_avoid_teletransporter() const override;
60 virtual bool can_avoid_stream(const Stream& stream) const override;
61 virtual bool can_avoid_sensor() const override;
62 virtual bool can_avoid_switch() const override;
63 virtual bool can_be_hurt(Entity* attacker) const override;
64 virtual std::shared_ptr<CarriedObject> get_carried_object() const override;
65 virtual CarriedObject::Behavior get_previous_carried_object_behavior() const override;
66
67 private:
68
69 std::shared_ptr<JumpMovement>
70 movement; /**< the movement applied to the hero */
71 int direction8; /**< direction of the jump (0 to 7) */
72 bool with_sound; /**< indicates that a jump sound is played */
73 std::shared_ptr<CarriedObject>
74 carried_object; /**< an item carried by the hero while making
75 * this jump, or nullptr */
76
77};
78
79}
