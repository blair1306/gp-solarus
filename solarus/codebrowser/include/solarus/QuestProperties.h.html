18#ifndef SOLARUS_QUEST_PROPERTIES_H
19#define SOLARUS_QUEST_PROPERTIES_H
20
21#include "solarus/Common.h"
22#include "solarus/lowlevel/Size.h"
23#include "solarus/lua/LuaData.h"
24#include <iosfwd>
25#include <string>
26
27struct lua_State;
28
29namespace Solarus {
30
31class MainLoop;
32
33/**
34 * \brief This class stores properties of a quest (the quest.dat file).
35 *
36 * These properties are general information about a quest.
37 * They include the Solarus compatibility version,
38 * the writing directory for savegames of this quest,
39 * and some metadata describing the quest.
40 */
41class SOLARUS_API QuestProperties : public LuaData {
42
43 public:
44
45 QuestProperties();
46
47 bool import_from_lua(lua_State* l) override;
48 bool export_to_lua(std::ostream& out) const override;
49
50 std::string get_solarus_version() const;
51 void set_solarus_version(const std::string& solarus_version);
52 std::string get_quest_write_dir() const;
53 void set_quest_write_dir(const std::string& quest_write_dir);
54 std::string get_title() const;
55 void set_title(const std::string& title);
56 std::string get_short_description() const;
57 void set_short_description(const std::string& short_description);
58 std::string get_long_description() const;
59 void set_long_description(const std::string& long_description);
60 std::string get_author() const;
61 void set_author(const std::string& author);
62 std::string get_quest_version() const;
63 void set_quest_version(const std::string& quest_version);
64 std::string get_release_date() const;
65 void set_release_date(const std::string& release_date);
66 std::string get_website() const;
67 void set_website(const std::string& website);
68 Size get_normal_quest_size() const;
69 void set_normal_quest_size(const Size& normal_quest_size);
70 Size get_min_quest_size() const;
71 void set_min_quest_size(const Size& min_quest_size);
72 Size get_max_quest_size() const;
73 void set_max_quest_size(const Size& max_quest_size);
74
75 private:
76
77 std::string solarus_version; /**< Version of Solarus the quest is
78 * compatible with. */
79 std::string quest_write_dir; /**< Directory where to save files of
80 * the quest, relative to
81 * FileTools::get_base_write_dir(). */
82 std::string title; /**< Title of the quest. */
83 std::string short_description; /**< One-line description of the quest. */
84 std::string long_description; /**< Longer description of the quest. */
85 std::string author; /**< Who created the quest. */
86 std::string quest_version; /**< Version of the quest. */
87 std::string release_date; /**< Date of the last quest release
88 * (YYYY-MM-DD). */
89 std::string website; /**< URL of the quest website. */
90 Size normal_quest_size; /**< Default quest size. */
91 Size min_quest_size; /**< Minimum quest size. */
92 Size max_quest_size; /**< Maximum quest size. */
93
94};
95
96}
97
98#endif
