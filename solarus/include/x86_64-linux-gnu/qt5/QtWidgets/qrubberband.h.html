42#ifndef QRUBBERBAND_H
43#define QRUBBERBAND_H
44
45#include <QtWidgets/qwidget.h>
46
47QT_BEGIN_NAMESPACE
48
49
50#ifndef QT_NO_RUBBERBAND
51
52class QRubberBandPrivate;
53class QStyleOptionRubberBand;
54
55class Q_WIDGETS_EXPORT QRubberBand : public QWidget
56{
57 Q_OBJECT
58
59public:
60 enum Shape { Line, Rectangle };
61 explicit QRubberBand(Shape, QWidget * =0);
62 ~QRubberBand();
63
64 Shape shape() const;
65
66 void setGeometry(const QRect &r);
67
68 inline void setGeometry(int x, int y, int w, int h);
69 inline void move(int x, int y);
70 inline void move(const QPoint &p)
71 { move(p.x(), p.y()); }
72 inline void resize(int w, int h)
73 { setGeometry(geometry().x(), geometry().y(), w, h); }
74 inline void resize(const QSize &s)
75 { resize(s.width(), s.height()); }
76
77protected:
78 bool event(QEvent *e);
79 void paintEvent(QPaintEvent *);
80 void changeEvent(QEvent *);
81 void showEvent(QShowEvent *);
82 void resizeEvent(QResizeEvent *);
83 void moveEvent(QMoveEvent *);
84 void initStyleOption(QStyleOptionRubberBand *option) const;
85
86private:
87 Q_DECLARE_PRIVATE(QRubberBand)
88};
89
90inline void QRubberBand::setGeometry(int ax, int ay, int aw, int ah)
91{ setGeometry(QRect(ax, ay, aw, ah)); }
92inline void QRubberBand::move(int ax, int ay)
93{ setGeometry(ax, ay, width(), height()); }
94
95#endif // QT_NO_RUBBERBAND
96
97QT_END_NAMESPACE
98
99#endif // QRUBBERBAND_H
