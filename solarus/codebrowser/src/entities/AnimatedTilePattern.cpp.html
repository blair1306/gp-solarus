25/**
26 * \brief Interval in millisecond between two frames of an animation.
27 */
28static constexpr uint32_t TILE_FRAME_INTERVAL = 250;
29
30/**
31 * \brief Gives the current frame depending on the sequence type and the frame counter.
32 *
33 * This array associates the current frame (0, 1 or 2) to
34 * the sequence type and the frame counter (0 to 11).
35 */
36static constexpr short frames[2][12] = {
37 {0, 1, 2, 0, 1, 2, 0, 1, 2, 0, 1, 2}, // sequence 0-1-2
38 {0, 1, 2, 1, 0, 1, 2, 1, 0, 1, 2, 1}, // sequence 0-1-2-1
39};
40
41int AnimatedTilePattern::frame_counter = 0;
42
43int AnimatedTilePattern::current_frames[3] = {0, 0, 0};
44
45uint32_t AnimatedTilePattern::next_frame_date = 0;
46
47/**
48 * \brief Constructor.
49 * \param ground Kind of ground of the tile pattern.
50 * \param sequence animation sequence type
51 * \param size Size of the pattern (the same for each frame of the animation)
52 * \param x1 x position of the first frame in the tileset
53 * \param y1 y position of the first frame in the tileset
54 * \param x2 x position of the second frame in the tileset
55 * \param y2 y position of the second frame in the tileset
56 * \param x3 x position of the third frame in the tileset
57 * \param y3 y position of the third frame in the tileset
58 * \param parallax true to also set parallax scrolling to the tile pattern
59 */
60AnimatedTilePattern::AnimatedTilePattern(Ground ground,
61 AnimationSequence sequence,
62 const Size& size,
63 int x1, int y1,
64 int x2, int y2,
65 int x3, int y3,
66 bool parallax):
67 TilePattern(ground, size),
68 sequence(sequence),
69 parallax(parallax) {
70
71 this->position_in_tileset[0] = Rectangle(x1, y1);
72 this->position_in_tileset[1] = Rectangle(x2, y2);
73 this->position_in_tileset[2] = Rectangle(x3, y3);
74
75 for (int i = 0; i < 3; i++) {
76 this->position_in_tileset[i].set_size(size);
77 }
78}

80/**
81 * \brief Initializes the animated tile pattern system.
82 */
83void AnimatedTilePattern::initialize() {
84 next_frame_date = System::now();
85 frame_counter = 0;
86 for (int i = 0; i < 3; ++i) {
87 current_frames[i] = 0;
88 }
89}
90
91/**
92 * \brief Cleans the animated tile pattern system.
93 */
94void AnimatedTilePattern::quit() {
95 next_frame_date = 0;
96 frame_counter = 0;
97 for (int i = 0; i < 3; ++i) {
98 current_frames[i] = 0;
99 }
100}
101
102/**
103 * \brief Updates the current frame of all tiles.
104 *
105 * This function is called repeatedly by the map.
106 */
107void AnimatedTilePattern::update() {
108
109 uint32_t now = System::now();
110
111 while (now >= next_frame_date) {
112
113 frame_counter = (frame_counter + 1) % 12;
114 current_frames[1] = frames[0][frame_counter];
115 current_frames[2] = frames[1][frame_counter];
116
117 next_frame_date += TILE_FRAME_INTERVAL; // the frame changes every 250 ms
118 }
119}

121/**
122 * \brief Draws the tile image on a surface.
123 * \param dst_surface the surface to draw
124 * \param dst_position position where tile pattern should be drawn on dst_surface
125 * \param tileset the tileset of this tile
126 * \param viewport coordinates of the top-left corner of dst_surface relative
127 * to the map (may be used for scrolling tiles)
128 */
129void AnimatedTilePattern::draw(
130 const SurfacePtr& dst_surface,
131 const Point& dst_position,
132 const Tileset& tileset,
133 const Point& viewport
134) const {
135 const SurfacePtr& tileset_image = tileset.get_tiles_image();
136 const Rectangle& src = position_in_tileset[current_frames[sequence]];
137 Point dst = dst_position;
138
139 if (parallax) {
140 dst += viewport / ParallaxScrollingTilePattern::ratio;
141 }
142
143 tileset_image->draw_region(src, dst_surface, dst);
144}
145
146/**
147 * \brief Returns whether tiles having this tile pattern are drawn at their
148 * position.
149 *
150 * Usually, this function returns true, and when it is the case, draw() is
151 * called only for tiles that are located in the current viewport.
152 *
153 * However, some tile patterns may want to be drawn even when they are not
154 * in the viewport, typically to make an illusion of movement like parallax
155 * scrolling.
156 *
157 * \return true to if this tile pattern is always drawn at its coordinates
158 */
159bool AnimatedTilePattern::is_drawn_at_its_position() const {
160 return !parallax;
161}
