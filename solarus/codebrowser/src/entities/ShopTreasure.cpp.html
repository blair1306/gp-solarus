36/**
37 * \brief Creates a new shop treasure with the specified treasure and price.
38 * \param name The name identifying this entity.
39 * \param layer Layer of the entity to create.
40 * \param xy Coordinates of the entity to create.
41 * \param treasure The treasure that the hero can buy.
42 * \param price The treasure's price.
43 * \param font_id Id of the font to use to display the price or an empty
44 * string for the default font.
45 * \param dialog_id Id of the dialog describing the treasure when the hero
46 * watches it.
47 */
48ShopTreasure::ShopTreasure(
49 const std::string& name,
50 int layer,
51 const Point& xy,
52 const Treasure& treasure,
53 int price,
54 const std::string& font_id,
55 const std::string& dialog_id):
56 Entity(name, 0, layer, xy, Size(32, 32)),
57 treasure(treasure),
58 price(price),
59 dialog_id(dialog_id),
60 treasure_sprite(treasure.create_sprite()),
61 rupee_icon_sprite(std::make_shared<Sprite>("entities/rupee_icon")),
62 price_digits(0, 0, TextSurface::HorizontalAlignment::LEFT, TextSurface::VerticalAlignment::TOP) {
63
64 set_collision_modes(CollisionMode::COLLISION_FACING);
65
66 std::ostringstream oss;
67 oss << price;
68 price_digits.set_text(oss.str());
69 if (FontResource::exists(font_id)) {
70 price_digits.set_font(font_id);
71 }
72}

74/**
75 * \brief Creates a new shop treasure with the specified treasure and price.
76 * \param game The current game.
77 * \param name The name identifying this entity.
78 * \param layer Layer of the entity to create.
79 * \param xy Coordinates of the entity to create.
80 * \param treasure The treasure that the hero can buy.
81 * \param price The treasure's price.
82 * \param font_id Id of the font to use to display the price.
83 * \param dialog_id Id of the dialog describing the treasure when the hero watches it.
84 * \return The shop treasure created, or nullptr if it is already bought or if it
85 * is not obtainable.
86 */
87std::shared_ptr<ShopTreasure> ShopTreasure::create(
88 Game& /* game */,
89 const std::string& name,
90 int layer,
91 const Point& xy,
92 const Treasure& treasure,
93 int price,
94 const std::string& font_id,
95 const std::string& dialog_id
96) {
97 // See if the item is not already bought and is obtainable.
98 if (treasure.is_found() || !treasure.is_obtainable()) {
99 return nullptr;
100 }
101
102 return std::make_shared<ShopTreasure>(
103 name, layer, xy, treasure, price, font_id, dialog_id
104 );
105}

160/**
161 * \brief This function is called by the engine when an entity overlaps the shop item.
162 *
163 * If the entity is the hero, we allow him to buy the item.
164 *
165 * \param entity_overlapping the entity overlapping the detector
166 * \param collision_mode the collision mode that detected the collision
167 */
168void ShopTreasure::notify_collision(
169 Entity& entity_overlapping, CollisionMode /* collision_mode */) {
170
171 if (entity_overlapping.is_hero() && !get_game().is_suspended()) {
172
173 Hero& hero = static_cast<Hero&>(entity_overlapping);
174
175 if (get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_NONE
176 && hero.is_free()) {
177
178 // we show the 'look' icon
179 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_LOOK);
180 }
181 }
182}
183
184/**
185 * \copydoc Entity::notify_action_command_pressed
186 */
187bool ShopTreasure::notify_action_command_pressed() {
188
189 if (get_hero().is_free()
190 && get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_LOOK) {
191
192 get_lua_context()->notify_shop_treasure_interaction(*this);
193 return true;
194 }
195
196 return false;
197}

210/**
211 * \brief Draws the entity on the map.
212 */
213void ShopTreasure::draw_on_map() {
214
215 const CameraPtr& camera = get_map().get_camera();
216 if (camera == nullptr) {
217 return;
218 }
219
220 const SurfacePtr& map_surface = get_map().get_camera_surface();
221 int x = get_x();
222 int y = get_y();
223
224 // draw the treasure
225 treasure_sprite->draw(map_surface,
226 x + 16 - camera->get_top_left_x(),
227 y + 13 - camera->get_top_left_y()
228 );
229
230 // also draw the price
231 price_digits.draw(map_surface,
232 x + 12 - camera->get_top_left_x(),
233 y + 21 - camera->get_top_left_y());
234 rupee_icon_sprite->draw(map_surface,
235 x - camera->get_top_left_x(),
236 y + 22 - camera->get_top_left_y());
237}
