72 QT_DEPRECATED QImageTextKeyLang(const char* k, const char* l) : key(k), lang(l) { }
73 QT_DEPRECATED QImageTextKeyLang() { }

75 QByteArray key;

77

83 { return !operator==(other); }

363inline QString QImage::text(const char* key, const char* lang) const
364{
365 if (!d)
366 return QString();
367 QString k = QString::fromLatin1(key);
368 if (lang && *lang)
369 k += QLatin1Char('/') + QString::fromLatin1(lang);
370 return text(k);
371}
372
373inline QList<QImageTextKeyLang> QImage::textList() const
374{
375 QList<QImageTextKeyLang> imageTextKeys;
376 if (!d)
377 return imageTextKeys;
378 QStringList keys = textKeys();
379 for (int i = 0; i < keys.size(); ++i) {
380 int index = keys.at(i).indexOf(QLatin1Char('/'));
381 if (index > 0) {
382 QImageTextKeyLang tkl(true);
383 tkl.key = keys.at(i).left(index).toLatin1();
384 tkl.lang = keys.at(i).mid(index+1).toLatin1();
385 imageTextKeys += tkl;
386 }
387 }
388
389 return imageTextKeys;
390}
391
392inline QStringList QImage::textLanguages() const
393{
394 if (!d)
395 return QStringList();
396 QStringList keys = textKeys();
397 QStringList languages;
398 for (int i = 0; i < keys.size(); ++i) {
399 int index = keys.at(i).indexOf(QLatin1Char('/'));
400 if (index > 0)
401 languages += keys.at(i).mid(index+1);
402 }
403
404 return languages;
405}

407inline QString QImage::text(const QImageTextKeyLang&kl) const
408{
409 if (!d)
410 return QString();
411 QString k = QString::fromLatin1(kl.key.constData());
412 if (!kl.lang.isEmpty())
413 k += QLatin1Char('/') + QString::fromLatin1(kl.lang.constData());
414 return text(k);
415}

417inline void QImage::setText(const char* key, const char* lang, const QString &s)
418{
419 if (!d)
420 return;
421 detach();
422
423 // In case detach() ran out of memory
424 if (!d)
425 return;
426
427 QString k = QString::fromLatin1(key);
428 if (lang && *lang)
429 k += QLatin1Char('/') + QString::fromLatin1(lang);
430 setText(k, s);
431}
