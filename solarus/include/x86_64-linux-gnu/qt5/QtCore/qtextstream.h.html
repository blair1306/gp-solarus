206typedef QTextStream & (*QTextStreamFunction)(QTextStream &);// manipulator function
207typedef void (QTextStream::*QTSMFI)(int); // manipulator w/int argument
208typedef void (QTextStream::*QTSMFC)(QChar); // manipulator w/QChar argument
209
210
211class Q_CORE_EXPORT QTextStreamManipulator
212{
213public:
214 QTextStreamManipulator(QTSMFI m, int a) { mf = m; mc = 0; arg = a; }
215 QTextStreamManipulator(QTSMFC m, QChar c) { mf = 0; mc = m; ch = c; arg = -1; }
216 void exec(QTextStream &s) { if (mf) { (s.*mf)(arg); } else { (s.*mc)(ch); } }
217
218private:
219 QTSMFI mf; // QTextStream member function
220 QTSMFC mc; // QTextStream member function
221 int arg; // member function argument
222 QChar ch;
223};
224
225inline QTextStream &operator>>(QTextStream &s, QTextStreamFunction f)
226{ return (*f)(s); }
227
228inline QTextStream &operator<<(QTextStream &s, QTextStreamFunction f)
229{ return (*f)(s); }
230
231inline QTextStream &operator<<(QTextStream &s, QTextStreamManipulator m)
232{ m.exec(s); return s; }
