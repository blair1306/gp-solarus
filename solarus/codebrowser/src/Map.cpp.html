78/**
79 * \brief Changes the tileset used to draw this map.
80 *
81 * The new tileset must be compatible with the previous one,
82 * i.e. every tile of the previous tileset must exist in the new one
83 * and have the same properties.
84 *
85 * \param tileset_id Id of the new tileset.
86 */
87void Map::set_tileset(const std::string& tileset_id) {
88
89 ResourceProvider& resource_provider = get_game().get_resource_provider();
90 tileset = &resource_provider.get_tileset(tileset_id);
91 get_entities().notify_tileset_changed();
92 this->tileset_id = tileset_id;
93 build_background_surface();
94}

273/**
274 * \brief Loads the map into a game.
275 *
276 * Reads the description file of the map.
277 *
278 * \param game the game
279 */
280void Map::load(Game& game) {
281
282 background_surface = Surface::create(
283 Video::get_quest_size()
284 );
285
286 // Read the map data file.
287 MapData data;
288 const std::string& file_name = std::string("maps/") + get_id() + ".dat";
289 bool success = data.import_from_quest_file(file_name);
290
291 if (!success) {
292 Debug::die("Failed to load map data file '" + file_name + "'");
293 }
294
295 // Initialize the map from the data just read.
296 this->game = &game;
297 ResourceProvider& resource_provider = game.get_resource_provider();
298 location.set_xy(data.get_location());
299 location.set_size(data.get_size());
300 width8 = data.get_size().width / 8;
301 height8 = data.get_size().height / 8;
302 min_layer = data.get_min_layer();
303 max_layer = data.get_max_layer();
304 music_id = data.get_music_id();
305 set_world(data.get_world());
306 set_floor(data.get_floor());
307 tileset_id = data.get_tileset_id();
308 tileset = &resource_provider.get_tileset(tileset_id);
309 entities = std::unique_ptr<Entities>(new Entities(game, *this));
310 entities->create_entities(data);
311
312 build_background_surface();
313 build_foreground_surface();
314
315 loaded = true;
316}

340/**
341 * \brief Sets the current destination point of the map.
342 * \param destination_name Name of the destination point you want to use.
343 * An empty string means the default one.
344 * You can also use "_same" to keep the hero's coordinates, or
345 * "_side0", "_side1", "_side2" or "_side3"
346 * to place the hero on a side of the map.
347 */
348void Map::set_destination(const std::string& destination_name) {
349
350 this->destination_name = destination_name;
351}
352
353/**
354 * \brief Returns the destination point name set by the last call to
355 * set_destination().
356 * \return The name of the destination point previously set,
357 * possibly an empty string (meaning the default one).
358 */
359const std::string& Map::get_destination_name() const {
360 return destination_name;
361}
362
363/**
364 * \brief Returns the destination point specified by the last call to
365 * set_destination().
366 *
367 * If the destination point was set to a special value
368 * ("_same", "_side0", "_side1", "_side2" or "_side3"), returns nullptr.
369 *
370 * If the destination name is empty, returns the default destination if any,
371 * or nullptr.
372 *
373 * Otherwise, if there is no destination entity with this name on the map,
374 * prints an error message and returns the default destination if any or nullptr.
375 *
376 * \return The destination point previously set, or nullptr.
377 */
378Destination* Map::get_destination() {
379
380 if (destination_name == "_same"
381 || destination_name.substr(0,5) == "_side") {
382 return nullptr;
383 }
384
385 Debug::check_assertion(is_loaded(), "This map is not loaded");
386
387 std::shared_ptr<Destination> destination;
388 if (!destination_name.empty()) {
389 // Use the destination whose name was specified.
390 const EntityPtr& entity = get_entities().find_entity(destination_name);
391
392 if (entity == nullptr || entity->get_type() != EntityType::DESTINATION) {
393 Debug::error(
394 std::string("Map '") + get_id() + "': No such destination: '"
395 + destination_name + "'"
396 );
397 // Perhaps the game was saved with a destination that no longer exists
398 // or whose name was changed during the development of the quest.
399 // This is not a fatal error: we will use the default destination
400 // instead. It is up to quest makers to avoid this situation once their
401 // quest is released.
402 }
403 else {
404 destination = std::static_pointer_cast<Destination>(entity);
405 }
406 }
407
408 if (destination == nullptr) {
409 // No valid destination name was set: use the default one if any.
410 destination = get_entities().get_default_destination();
411 }
412
413 return destination.get();
414}
415
416/**
417 * \brief When the destination point is a side of the map,
418 * returns this side.
419 * \return the destination side (0 to 3), or -1 if the destination point is not a side
420 */
421int Map::get_destination_side() const {
422
423 if (destination_name.substr(0,5) == "_side") {
424 int destination_side = destination_name[5] - '0';
425 return destination_side;
426 }
427 return -1;
428}

511/**
512 * \brief Draws the map with all its entities on the screen.
513 */
514void Map::draw() {
515
516 if (!is_loaded()) {
517 return;
518 }
519
520 const SurfacePtr& camera_surface = get_camera_surface();
521
522 if (camera_surface == nullptr) {
523 return;
524 }
525
526 // background
527 draw_background(camera_surface);
528
529 // draw all entities (including the hero)
530 entities->draw();
531
532 // foreground
533 draw_foreground(camera_surface);
534
535 // Lua
536 get_lua_context().map_on_draw(*this, camera_surface);
537}

560/**
561 * \brief Builds a surface with black bars when the map is smaller than the
562 * camera size.
563 */
564void Map::build_foreground_surface() {
565
566 foreground_surface = nullptr;
567
568 const CameraPtr& camera = get_camera();
569 if (camera != nullptr) {
570 return;
571 }
572
573 const Size& camera_size = camera->get_size();
574 const int camera_width = camera_size.width;
575 const int camera_height = camera_size.height;
576
577 // If the map is too small for the camera, add black bars outside the map.
578 const int map_width = get_width();
579 const int map_height = get_height();
580
581 if (map_width >= camera_width
582 && map_height >= camera_height) {
583 // Nothing to do.
584 return;
585 }
586
587 foreground_surface = Surface::create(camera_size);
588 // Keep this surface as software destination because it is built only once.
589
590 if (map_width < camera_width) {
591 int bar_width = (camera_width - map_width) / 2;
592 Rectangle dst_position(0, 0, bar_width, camera_height);
593 foreground_surface->fill_with_color(Color::black, dst_position);
594 dst_position.set_x(bar_width + map_width);
595 foreground_surface->fill_with_color(Color::black, dst_position);
596 }
597
598 if (map_height < camera_height) {
599 int bar_height = (camera_height - map_height) / 2;
600 Rectangle dst_position(0, 0, camera_width, bar_height);
601 foreground_surface->fill_with_color(Color::black, dst_position);
602 dst_position.set_y(bar_height + map_height);
603 foreground_surface->fill_with_color(Color::black, dst_position);
604 }
605}

628/**
629 * \brief Draws a drawable object on the camera surface.
630 * \param drawable The drawable object to draw.
631 * \param x X coordinate of the drawable's origin point in the map.
632 * \param y Y coordinate of the drawable's origin point in the map.
633 */
634void Map::draw_visual(Drawable& drawable, int x, int y) {
635
636 // the position is given in the map coordinate system:
637 // convert it to the visible surface coordinate system
638 const CameraPtr& camera = get_camera();
639 if (camera == nullptr) {
640 return;
641 }
642 const SurfacePtr& camera_surface = camera->get_surface();
643 drawable.draw(camera_surface,
644 x - camera->get_top_left_x(),
645 y - camera->get_top_left_y()
646 );
647}
648
649/**
650 * \brief Draws a drawable object on a restricted area of the camera surface.
651 * \param drawable The drawable object to draw.
652 * \param x X coordinate of the drawable's origin point in the map.
653 * \param y Y coordinate of the drawable's origin point in the map.
654 * \param clipping_area Rectangle of the map where the drawing will be
655 * restricted. A flat rectangle means no restriction.
656 */
657void Map::draw_visual(Drawable& drawable, int x, int y,
658 const Rectangle& clipping_area) {
659
660 if (clipping_area.is_flat()) {
661 // No clipping area.
662 draw_visual(drawable, x, y);
663 return;
664 }
665
666 const CameraPtr& camera = get_camera();
667 if (camera == nullptr) {
668 return;
669 }
670 const SurfacePtr& camera_surface = camera->get_surface();
671
672 const Rectangle region_in_frame(
673 clipping_area.get_x() - x,
674 clipping_area.get_y() - y,
675 clipping_area.get_width(),
676 clipping_area.get_height()
677 );
678 const Point dst_position = {
679 x - camera->get_top_left_x(),
680 y - camera->get_top_left_y()
681 };
682 drawable.draw_region(
683 region_in_frame,
684 camera_surface,
685 dst_position
686 );
687}

743/**
744 * \brief Tests whether a rectangle has overlaps the outside part of the map area.
745 * \param collision_box the rectangle to check
746 * \return true if a point of the rectangle is outside the map area
747 */
748bool Map::test_collision_with_border(const Rectangle& collision_box) const {
749
750 return collision_box.get_x() < 0 || collision_box.get_x() + collision_box.get_width() >= get_width()
751 || collision_box.get_y() < 0 || collision_box.get_y() + collision_box.get_height() >= get_height();
752}

754/**
755 * \brief Tests whether a point collides with the ground of the map.
756 *
757 * The ground is the terrain of the point. It is defined by the tiles and
758 * by the presence of entities that may change it
759 * (like dynamic tiles and destructible items).
760 *
761 * This method also returns true if the point is outside the map.
762 *
763 * \param layer Layer of the point.
764 * \param x X of the point in pixels.
765 * \param y Y of the point in pixels.
766 * \param entity_to_check The entity to check (used to decide what grounds are
767 * considered as obstacle).
768 * \param [out] found_diagonal_wall \c true if the ground under this point was
769 * a diagonal wall, unchanged otherwise.
770 * Your algorithm may decide to check more points if there is a diagonal wall.
771 * \return \c true if this point is on an obstacle.
772 */
773bool Map::test_collision_with_ground(
774 int layer,
775 int x,
776 int y,
777 const Entity& entity_to_check,
778 bool& found_diagonal_wall) const {
779
780 bool on_obstacle = false;
781 int x_in_tile, y_in_tile;
782
783 // If the point is outside the map, this is an obstacle.
784 if (test_collision_with_border(x, y)) {
785 return true;
786 }
787
788 // Get the ground property under this point.
789 Ground ground = get_ground(layer, x, y, &entity_to_check);
790 switch (ground) {
791
792 case Ground::EMPTY:
793 case Ground::TRAVERSABLE:
794 case Ground::GRASS:
795 case Ground::ICE:
796 case Ground::WALL:
797 case Ground::LOW_WALL:
798 case Ground::SHALLOW_WATER:
799 case Ground::DEEP_WATER:
800 case Ground::HOLE:
801 case Ground::LAVA:
802 case Ground::PRICKLE:
803 case Ground::LADDER:
804 on_obstacle = entity_to_check.is_ground_obstacle(ground);
805 break;
806
807 case Ground::WALL_TOP_RIGHT:
808 case Ground::WALL_TOP_RIGHT_WATER:
809 // The upper right half of the square is an obstacle
810 // so we have to test the position of the point in the square.
811 x_in_tile = x & 7;
812 y_in_tile = y & 7;
813 on_obstacle = y_in_tile <= x_in_tile;
814 found_diagonal_wall = true;
815 break;
816
817 case Ground::WALL_TOP_LEFT:
818 case Ground::WALL_TOP_LEFT_WATER:
819 // Same thing.
820 x_in_tile = x & 7;
821 y_in_tile = y & 7;
822 on_obstacle = y_in_tile <= 7 - x_in_tile;
823 found_diagonal_wall = true;
824 break;
825
826 case Ground::WALL_BOTTOM_LEFT:
827 case Ground::WALL_BOTTOM_LEFT_WATER:
828 x_in_tile = x & 7;
829 y_in_tile = y & 7;
830 on_obstacle = y_in_tile >= x_in_tile;
831 found_diagonal_wall = true;
832 break;
833
834 case Ground::WALL_BOTTOM_RIGHT:
835 case Ground::WALL_BOTTOM_RIGHT_WATER:
836 x_in_tile = x & 7;
837 y_in_tile = y & 7;
838 on_obstacle = y_in_tile >= 7 - x_in_tile;
839 found_diagonal_wall = true;
840 break;
841
842 }
843
844 return on_obstacle;
845}

849 * \param layer The layer.
850 * \param collision_box The rectangle to check.
851 * \param entity_to_check The entity to check (used to decide what is
852 * considered as obstacle).

855bool Map::test_collision_with_entities(
856 int layer,
857 const Rectangle& collision_box,
858 Entity& entity_to_check) {
859
860 if (!is_loaded()) {
861 return false;
862 }
863
864 EntityVector entities_nearby;
865 get_entities().get_entities_in_rectangle(collision_box, entities_nearby);
866 for (const EntityPtr& entity_nearby: entities_nearby) {
867

869 (entity_nearby->get_layer() == layer || entity_nearby->has_layer_independent_collisions()) &&
870 entity_nearby->is_obstacle_for(entity_to_check, collision_box) &&

881/**
882 * \brief Tests whether a rectangle collides with the map obstacles.
883 * \param layer Layer of the rectangle in the map.
884 * \param collision_box The rectangle to check (its dimensions should be
885 * multiples of 8).
886 * \param entity_to_check The entity to check (used to decide what is
887 * considered as obstacle).
888 * \return \c true if the rectangle is overlapping an obstacle.
889 */
890bool Map::test_collision_with_obstacles(
891 int layer,
892 const Rectangle& collision_box,
893 Entity& entity_to_check) {
894
895 // This function is called very often.
896 // For performance reasons, we only check the border of the of the collision box.
897
898 // Collisions with the terrain
899 // (i.e., tiles and dynamic entities that may change it).
900 const int x1 = collision_box.get_x();
901 const int x2 = x1 + collision_box.get_width() - 1;
902 const int y1 = collision_box.get_y();
903 const int y2 = y1 + collision_box.get_height() - 1;
904
905 // First, only check the terrain of both extremities of each 8-pixel
906 // segment of the border.
907 // This is enough for all terrains (except diagonal ones, see below)
908 // because the tested collision box makes at least 8x8 pixels.
909 bool found_diagonal_wall = false;
910 for (int x = x1; x <= x2; x += 8) {
911 if (test_collision_with_ground(layer, x, y1, entity_to_check, found_diagonal_wall)
912 || test_collision_with_ground(layer, x, y2, entity_to_check, found_diagonal_wall)
913 || test_collision_with_ground(layer, x + 7, y1, entity_to_check, found_diagonal_wall)
914 || test_collision_with_ground(layer, x + 7, y2, entity_to_check, found_diagonal_wall)) {
915 return true;
916 }
917 }
918
919 for (int y = y1; y <= y2; y += 8) {
920 if (test_collision_with_ground(layer, x1, y, entity_to_check, found_diagonal_wall)
921 || test_collision_with_ground(layer, x2, y, entity_to_check, found_diagonal_wall)
922 || test_collision_with_ground(layer, x1, y + 7, entity_to_check, found_diagonal_wall)
923 || test_collision_with_ground(layer, x2, y + 7, entity_to_check, found_diagonal_wall)) {
924 return true;
925 }
926 }
927
928 // Is a full check of the border needed?
929 // This is costly, but hopefully, we seldom need it.
930 if (found_diagonal_wall) {
931 // A diagonal wall ground was involved in the terrain check.
932 // In this case, we need to test all points of the border of the collision
933 // box. Otherwise, walls with sharp angles like 'V' become
934 // partially traversable.
935 for (int x = x1; x <= x2; ++x) {
936 if (test_collision_with_ground(layer, x, y1, entity_to_check, found_diagonal_wall)
937 || test_collision_with_ground(layer, x, y2, entity_to_check, found_diagonal_wall)) {
938 return true;
939 }
940 }
941
942 for (int y = y1; y <= y2; ++y) {
943 if (test_collision_with_ground(layer, x1, y, entity_to_check, found_diagonal_wall)
944 || test_collision_with_ground(layer, x2, y, entity_to_check, found_diagonal_wall)) {
945 return true;
946 }
947 }
948 }
949
950 // No collision with the terrain: check collisions with dynamic entities.
951 return test_collision_with_entities(layer, collision_box, entity_to_check);
952}

957 * \param x X coordinate of the point to check.
958 * \param y Y coordinate of the point to check.

1002 *
1003 * Only the borders of the rectangle are checked.
1004 *
1005 * \param layer The layer.
1006 * \param collision_box The rectangle to test.
1007 * \return \c true if there is at least one empty tile in this rectangle.

1009bool Map::has_empty_ground(int layer, const Rectangle& collision_box) const {
1010
1011 bool empty_tile = false;
1012
1013 // We just check the borders of the collision box.
1014 int y1 = collision_box.get_y();
1015 int y2 = y1 + collision_box.get_height() - 1;
1016 int x1 = collision_box.get_x();
1017 int x2 = x1 + collision_box.get_width() - 1;
1018
1019 for (int x = x1; x <= x2 && !empty_tile; x++) {
1020 empty_tile = get_ground(layer, x, y1, nullptr) == Ground::EMPTY
1021 || get_ground(layer, x, y2, nullptr) == Ground::EMPTY;
1022 }
1023
1024 for (int y = y1; y <= y2 && !empty_tile; y++) {
1025 empty_tile = get_ground(layer, x1, y, nullptr) == Ground::EMPTY
1026 || get_ground(layer, x2, y, nullptr) == Ground::EMPTY;
1027 }
1028
1029 return empty_tile;

1061/**
1062 * \brief Returns the ground at the specified point.
1063 *
1064 * Static tiles and dynamic entities are all taken into account here.
1065 *
1066 * \param layer Layer of the point.
1067 * \param xy Coordinates of the point.
1068 * \param entity_to_check The entity you want to know the ground of (if any).
1069 * Used to make sure that the entity's own modified ground does not count.
1070 * \return The ground at this place.
1071 */
1072Ground Map::get_ground(
1073 int layer,
1074 const Point& xy,
1075 const Entity* entity_to_check
1076) const {
1077
1078 if (!is_loaded()) {
1079 return Ground::EMPTY;
1080 }
1081
1082 if (test_collision_with_border(xy)) {
1083 // Outside the map bounds.
1084 return Ground::EMPTY;
1085 }
1086
1087 // See if a dynamic entity changes the ground.
1088 const Rectangle box(xy, Size(1, 1));
1089 ConstEntityVector entities_nearby;
1090 get_entities().get_entities_in_rectangle_sorted(box, entities_nearby);
1091
1092 const auto& rend = entities_nearby.rend();
1093 for (auto it = entities_nearby.rbegin(); it != rend; ++it) {
1094 const Entity& entity_nearby = *(*it);
1095
1096 const Ground ground = entity_nearby.get_modified_ground();
1097
1098 if (&entity_nearby == entity_to_check) {
1099 // Skip the entity itself.
1100 continue;
1101 }
1102 // TODO also skip entities above?
1103
1104 if (ground == Ground::EMPTY) {
1105 // The entity has no influence on the ground.
1106 continue;
1107 }
1108
1109 if (entity_nearby.overlaps(xy) &&
1110 entity_nearby.get_layer() == layer &&
1111 entity_nearby.is_enabled() &&
1112 !entity_nearby.is_being_removed()
1113 ) {
1114 return get_ground_from_entity(entity_nearby, xy);
1115 }
1116 }
1117
1118 // Otherwise, return the ground defined by static tiles (this is very fast).
1119 return entities->get_tile_ground(layer, xy.x, xy.y);
1120}

1144/**
1145 * \brief Returns the modified ground of an entity at the specified point.
1146 *
1147 * The point is assumed to overlap the entity.
1148 *
1149 * The point only matters if the ground is diagonal.
1150 * When an entity has a diagonal ground, only 8x8 squares on the diagonal of
1151 * that entity are diagonal grounds, other ones are resolved Ground::WALL on a
1152 * side and Ground::TRAVERSABLE or Ground::DEEP_WATER on the other side.
1153 *
1154 * A current limitation is that entities with diagonal grounds must be aligned
1155 * to the 8x8 grid to work correctly.
1156 *
1157 * \param entity Entity whose modified ground to get.
1158 * \param xy Coordinates of the point.
1159 * \return The ground at this place.
1160 */
1161Ground Map::get_ground_from_entity(const Entity& entity, const Point& xy) const {
1162
1163 const Ground ground = entity.get_modified_ground();
1164 if (!GroundInfo::is_ground_diagonal(ground)) {
1165 // Easy case.
1166 return ground;
1167 }
1168
1169 const bool square = entity.get_width() == entity.get_height();
1170 if (!square) {
1171 // Malformed entity.
1172 return Ground::TRAVERSABLE;
1173 }
1174 if (!entity.is_aligned_to_grid() || entity.get_width() % 8 != 0) {
1175 // Not supported yet.
1176 return Ground::TRAVERSABLE;
1177 }
1178
1179 if (entity.get_width() == 8) {
1180 // The entity only occupies one 8x8 square: the ground is then fully diagonal.
1181 return ground;
1182 }
1183
1184 const Point point_in_entity = xy - entity.get_top_left_xy();
1185 const Point square_in_entity = point_in_entity / 8;
1186
1187 const int num_squares = entity.get_width() / 8;
1188 int sum = 0;
1189
1190 switch (ground) {
1191
1192 // If the top right corner of the tile is an obstacle,
1193 // then the top right 8x8 squares are Ground::WALL, the bottom left 8x8
1194 // squares are Ground::TRAVERSABLE or Ground::DEEP_WATER and the 8x8 squares
1195 // on the diagonal are Ground::WALL_TOP_RIGHT.
1196 case Ground::WALL_TOP_RIGHT:
1197 case Ground::WALL_TOP_RIGHT_WATER:
1198
1199 if (square_in_entity.x == square_in_entity.y) {
1200 // 8x8 square on the diagonal.
1201 return ground;
1202 }
1203 if (square_in_entity.x > square_in_entity.y) {
1204 // 8x8 square in the wall part.
1205 return Ground::WALL;
1206 }
1207 // 8x8 square in the non-wall part.
1208 return ground == Ground::WALL_TOP_RIGHT ? Ground::TRAVERSABLE : Ground::DEEP_WATER;
1209
1210 case Ground::WALL_TOP_LEFT:
1211 case Ground::WALL_TOP_LEFT_WATER:
1212
1213 sum = square_in_entity.x + square_in_entity.y;
1214 if (sum == num_squares - 1) {
1215 // 8x8 square on the diagonal.
1216 return ground;
1217 }
1218 if (sum < num_squares - 1) {
1219 // 8x8 square in the wall part.
1220 return Ground::WALL;
1221 }
1222 // 8x8 square in the non-wall part.
1223 return ground == Ground::WALL_TOP_LEFT ? Ground::TRAVERSABLE : Ground::DEEP_WATER;
1224
1225 case Ground::WALL_BOTTOM_LEFT:
1226 case Ground::WALL_BOTTOM_LEFT_WATER:
1227
1228 if (square_in_entity.x == square_in_entity.y) {
1229 // 8x8 square on the diagonal.
1230 return ground;
1231 }
1232 if (square_in_entity.x < square_in_entity.y) {
1233 // 8x8 square in the wall part.
1234 return Ground::WALL;
1235 }
1236 // 8x8 square in the non-wall part.
1237 return ground == Ground::WALL_BOTTOM_LEFT ? Ground::TRAVERSABLE : Ground::DEEP_WATER;
1238
1239 case Ground::WALL_BOTTOM_RIGHT:
1240 case Ground::WALL_BOTTOM_RIGHT_WATER:
1241
1242 sum = square_in_entity.x + square_in_entity.y;
1243 if (sum == num_squares - 1) {
1244 // 8x8 square on the diagonal.
1245 return ground;
1246 }
1247 if (sum > num_squares - 1) {
1248 // 8x8 square in the wall part.
1249 return Ground::WALL;
1250 }
1251 // 8x8 square in the non-wall part.
1252 return ground == Ground::WALL_BOTTOM_RIGHT ? Ground::TRAVERSABLE : Ground::DEEP_WATER;
1253
1254 default:
1255 return ground;
1256 }
1257}

1263 * when this entity has just moved on the map, or when a detector
1264 * wants to check this entity.
