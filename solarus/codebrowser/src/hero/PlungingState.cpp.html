32Hero::PlungingState::PlungingState(Hero& hero):

34

37/**
38 * \brief Starts this state.
39 * \param previous_state the previous state
40 */
41void Hero::PlungingState::start(const State* previous_state) {
42
43 HeroState::start(previous_state);
44
45 if (get_entity().get_ground_below() == Ground::DEEP_WATER) {
46 get_sprites().set_animation("plunging_water");
47 }
48 else {
49 get_sprites().set_animation("plunging_lava");
50 }
51 Sound::play("splash");
52}
53
54/**
55 * \brief Updates this state.
56 */
57void Hero::PlungingState::update() {
58
59 HeroState::update();
60
61 if (get_sprites().is_animation_finished()) {
62
63 Hero& hero = get_entity();
64 int drown = 0;
65 if (hero.get_ground_below() == Ground::DEEP_WATER) {
66
67 if (get_equipment().has_ability(Ability::SWIM)) {
68 hero.set_state(new SwimmingState(hero));
69 }
70 else {
71 drown = 1;
72 }
73 }
74 else if (hero.get_ground_below() == Ground::LAVA) {
75 drown = 4;
76 }
77 else {
78 hero.set_state(new FreeState(hero));
79 }
80
81 if (drown > 0) {
82 get_equipment().remove_life(drown);
83 hero.set_state(new BackToSolidGroundState(hero, true, 300));
84 }
85 }
86}
