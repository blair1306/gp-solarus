370template <typename T>
371Q_INLINE_TEMPLATE void QList<T>::node_construct(Node *n, const T &t)
372{
373 if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic) n->v = new T(t);
374 else if (QTypeInfo<T>::isComplex) new (n) T(t);
375#if (defined(__GNUC__) || defined(__INTEL_COMPILER) || defined(__IBMCPP__)) && !defined(__OPTIMIZE__)
376 // This violates pointer aliasing rules, but it is known to be safe (and silent)
377 // in unoptimized GCC builds (-fno-strict-aliasing). The other compilers which
378 // set the same define are assumed to be safe.
379 else *reinterpret_cast<T*>(n) = t;
380#else
381 // This is always safe, but penaltizes unoptimized builds a lot.
382 else ::memcpy(n, static_cast<const void *>(&t), sizeof(T));
383#endif
384}

386template <typename T>
387Q_INLINE_TEMPLATE void QList<T>::node_destruct(Node *n)
388{
389 if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic) delete reinterpret_cast<T*>(n->v);
390 else if (QTypeInfo<T>::isComplex) reinterpret_cast<T*>(n)->~T();
391}

393template <typename T>
394Q_INLINE_TEMPLATE void QList<T>::node_copy(Node *from, Node *to, Node *src)
395{
396 Node *current = from;
397 if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic) {
398 QT_TRY {
399 while(current != to) {
400 current->v = new T(*reinterpret_cast<T*>(src->v));
401 ++current;
402 ++src;
403 }
404 } QT_CATCH(...) {
405 while (current-- != from)
406 delete reinterpret_cast<T*>(current->v);
407 QT_RETHROW;
408 }
409
410 } else if (QTypeInfo<T>::isComplex) {
411 QT_TRY {
412 while(current != to) {
413 new (current) T(*reinterpret_cast<T*>(src));
414 ++current;
415 ++src;
416 }
417 } QT_CATCH(...) {
418 while (current-- != from)
419 (reinterpret_cast<T*>(current))->~T();
420 QT_RETHROW;
421 }
422 } else {
423 if (src != from && to - from > 0)
424 memcpy(from, src, (to - from) * sizeof(Node));
425 }
426}

428template <typename T>
429Q_INLINE_TEMPLATE void QList<T>::node_destruct(Node *from, Node *to)
430{
431 if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic)
432 while(from != to) --to, delete reinterpret_cast<T*>(to->v);
433 else if (QTypeInfo<T>::isComplex)
434 while (from != to) --to, reinterpret_cast<T*>(to)->~T();
435}

446template <typename T>
447inline typename QList<T>::iterator QList<T>::insert(iterator before, const T &t)
448{
449 Q_ASSERT_X(isValidIterator(before), "QList::insert", "The specified iterator argument 'before' is invalid");
450
451 int iBefore = int(before.i - reinterpret_cast<Node *>(p.begin()));
452 Node *n = 0;
453 if (d->ref.isShared())
454 n = detach_helper_grow(iBefore, 1);
455 else
456 n = reinterpret_cast<Node *>(p.insert(iBefore));
457 QT_TRY {
458 node_construct(n, t);
459 } QT_CATCH(...) {
460 p.remove(iBefore);
461 QT_RETHROW;
462 }
463 return n;
464}
465template <typename T>
466inline typename QList<T>::iterator QList<T>::erase(iterator it)
467{
468 Q_ASSERT_X(isValidIterator(it), "QList::erase", "The specified iterator argument 'it' is invalid");
469 if (d->ref.isShared()) {
470 int offset = int(it.i - reinterpret_cast<Node *>(p.begin()));
471 it = begin(); // implies detach()
472 it += offset;
473 }
474 node_destruct(it.i);
475 return reinterpret_cast<Node *>(p.erase(reinterpret_cast<void**>(it.i)));
476}

516template <typename T>
517Q_OUTOFLINE_TEMPLATE void QList<T>::append(const T &t)
518{
519 if (d->ref.isShared()) {
520 Node *n = detach_helper_grow(INT_MAX, 1);
521 QT_TRY {
522 node_construct(n, t);
523 } QT_CATCH(...) {
524 --d->end;
525 QT_RETHROW;
526 }
527 } else {
528 if (QTypeInfo<T>::isLarge || QTypeInfo<T>::isStatic) {
529 Node *n = reinterpret_cast<Node *>(p.append());
530 QT_TRY {
531 node_construct(n, t);
532 } QT_CATCH(...) {
533 --d->end;
534 QT_RETHROW;
535 }
536 } else {
537 Node *n, copy;
538 node_construct(&copy, t); // t might be a reference to an object in the array
539 QT_TRY {
540 n = reinterpret_cast<Node *>(p.append());;
541 } QT_CATCH(...) {
542 node_destruct(&copy);
543 QT_RETHROW;
544 }
545 *n = copy;
546 }
547 }

626template <typename T>
627inline void QList<T>::swap(int i, int j)
628{
629 Q_ASSERT_X(i >= 0 && i < p.size() && j >= 0 && j < p.size(),
630 "QList<T>::swap", "index out of range");
631 detach();
632 void *t = d->array[d->begin + i];
633 d->array[d->begin + i] = d->array[d->begin + j];
634 d->array[d->begin + j] = t;
635}

646template<typename T>
647Q_OUTOFLINE_TEMPLATE QList<T> QList<T>::mid(int pos, int alength) const
648{
649 if (alength < 0 || pos > size() - alength)
650 alength = size() - pos;
651 if (pos == 0 && alength == size())
652 return *this;
653 QList<T> cpy;
654 if (alength <= 0)
655 return cpy;
656 cpy.reserve(alength);
657 cpy.d->end = alength;
658 QT_TRY {
659 cpy.node_copy(reinterpret_cast<Node *>(cpy.p.begin()),
660 reinterpret_cast<Node *>(cpy.p.end()),
661 reinterpret_cast<Node *>(p.begin() + pos));
662 } QT_CATCH(...) {
663 // restore the old end
664 cpy.d->end = 0;
665 QT_RETHROW;
666 }
667 return cpy;
668}

685template <typename T>
686Q_OUTOFLINE_TEMPLATE typename QList<T>::Node *QList<T>::detach_helper_grow(int i, int c)
687{
688 Node *n = reinterpret_cast<Node *>(p.begin());
689 QListData::Data *x = p.detach_grow(&i, c);
690 QT_TRY {
691 node_copy(reinterpret_cast<Node *>(p.begin()),
692 reinterpret_cast<Node *>(p.begin() + i), n);
693 } QT_CATCH(...) {
694 p.dispose();
695 d = x;
696 QT_RETHROW;
697 }
698 QT_TRY {
699 node_copy(reinterpret_cast<Node *>(p.begin() + i + c),
700 reinterpret_cast<Node *>(p.end()), n + i);
701 } QT_CATCH(...) {
702 node_destruct(reinterpret_cast<Node *>(p.begin()),
703 reinterpret_cast<Node *>(p.begin() + i));
704 p.dispose();
705 d = x;
706 QT_RETHROW;
707 }
708
709 if (!x->ref.deref())
710 dealloc(x);
711
712 return reinterpret_cast<Node *>(p.begin() + i);
713}

763template <typename T>
764Q_OUTOFLINE_TEMPLATE bool QList<T>::operator==(const QList<T> &l) const
765{

767 return false;
768 if (d == l.d)
769 return true;
770 Node *i = reinterpret_cast<Node *>(p.end());
771 Node *b = reinterpret_cast<Node *>(p.begin());
772 Node *li = reinterpret_cast<Node *>(l.p.end());
773 while (i != b) {
774 --i; --li;
775 if (!(i->t() == li->t()))

796template <typename T>
797Q_OUTOFLINE_TEMPLATE int QList<T>::removeAll(const T &_t)
798{
799 int index = indexOf(_t);
800 if (index == -1)
801 return 0;
802
803 const T t = _t;
804 detach();
805
806 Node *i = reinterpret_cast<Node *>(p.at(index));
807 Node *e = reinterpret_cast<Node *>(p.end());
808 Node *n = i;
809 node_destruct(i);
810 while (++i != e) {
811 if (i->t() == t)
812 node_destruct(i);
813 else
814 *n++ = *i;
815 }
816
817 int removedCount = e - n;
818 d->end -= removedCount;
819 return removedCount;
820}

833template <typename T>
834Q_OUTOFLINE_TEMPLATE typename QList<T>::iterator QList<T>::erase(typename QList<T>::iterator afirst,
835 typename QList<T>::iterator alast)
836{
837 Q_ASSERT_X(isValidIterator(afirst), "QList::erase", "The specified iterator argument 'afirst' is invalid");
838 Q_ASSERT_X(isValidIterator(alast), "QList::erase", "The specified iterator argument 'alast' is invalid");
839
840 if (d->ref.isShared()) {
841 // ### A block is erased and a detach is needed. We should shrink and only copy relevant items.
842 int offsetfirst = int(afirst.i - reinterpret_cast<Node *>(p.begin()));
843 int offsetlast = int(alast.i - reinterpret_cast<Node *>(p.begin()));
844 afirst = begin(); // implies detach()
845 alast = afirst;
846 afirst += offsetfirst;
847 alast += offsetlast;
848 }
849
850 for (Node *n = afirst.i; n < alast.i; ++n)
851 node_destruct(n);
852 int idx = afirst - begin();
853 p.remove(idx, alast - afirst);
854 return begin() + idx;
855}

857template <typename T>
858Q_OUTOFLINE_TEMPLATE QList<T> &QList<T>::operator+=(const QList<T> &l)
859{
860 if (!l.isEmpty()) {
861 if (isEmpty()) {
862 *this = l;
863 } else {
864 Node *n = (d->ref.isShared())
865 ? detach_helper_grow(INT_MAX, l.size())
866 : reinterpret_cast<Node *>(p.append(l.p));
867 QT_TRY {
868 node_copy(n, reinterpret_cast<Node *>(p.end()),
869 reinterpret_cast<Node *>(l.p.begin()));
870 } QT_CATCH(...) {
871 // restore the old end
872 d->end -= int(reinterpret_cast<Node *>(p.end()) - n);
873 QT_RETHROW;
874 }
875 }
876 }
877 return *this;
878}

887Q_OUTOFLINE_TEMPLATE int QList<T>::indexOf(const T &t, int from) const
888{
889 if (from < 0)
890 from = qMax(from + p.size(), 0);
891 if (from < p.size()) {
892 Node *n = reinterpret_cast<Node *>(p.at(from -1));
893 Node *e = reinterpret_cast<Node *>(p.end());
894 while (++n != e)
895 if (n->t() == t)
896 return int(n - reinterpret_cast<Node *>(p.begin()));
897 }

903{
