27namespace Solarus {
28
29class Hero;
30class Map;
31class Entities;
32
33/**
34 * \brief Wraps a Solarus main loop and provides features to easily make tests.
35 *
36 * Contrary to the traditional main loop, the test environment does not
37 * automatically advance simulated time.
38 * The simulation is paused at a specific cycle and will only advance if you
39 * decide so.
40 *
41 * If whatever you want to test can be tested in the same cycle, you don't
42 * need to simulate more cycles.
43 * If you want to test timed objects like timers or movements,
44 * you can make the simulation advance by manual steps.
45 * This will also be much faster than real time.
46 */
47class TestEnvironment {
48
49 public:
50
51 TestEnvironment(int argc, char** argv);
52
53 // Get important objects.
54 const Arguments& get_arguments() const;
55 MainLoop& get_main_loop();
56 Game& get_game();
57 Map& get_map();
58 Entities& get_entities();
59 Hero& get_hero();
60
61 void run_map(const std::string& map_id);
62
63 // Creating entities.
64 template<typename T>
65 std::shared_ptr<T> make_entity(
66 const Point& xy = Point(0, 0),
67 int layer = 0
68 );
69
70 // Simulation.
71 uint32_t now();
72 void step();
73
74 private:
75
76 Arguments arguments;
77 MainLoop main_loop;
78 std::string map_id;
79};
80
81}
