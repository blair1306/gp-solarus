27/**
28 * \brief Constructor.
29 *
30 * It is called by the subclasses.
31 *
32 * \param ground Kind of ground.
33 * \param size Size of the pattern in pixels (must be a multiple of 8).
34 */
35TilePattern::TilePattern(Ground ground, const Size& size):
36 ground(ground),
37 size(size) {
38
39 // Check the width and the height.
40 if (size.width <= 0 || size.height <= 0
41 || size.width % 8 != 0 || size.height % 8 != 0) {
42 std::ostringstream oss;
43 oss << "Invalid tile pattern: the size is ("
44 << size.width << "x" << size.height <<
45 ") but should be positive and multiple of 8 pixels";
46 Debug::die(oss.str());
47 }
48}

114/**
115 * \brief Returns whether this tile pattern is animated, i.e. not always drawn
116 * the same way.
117 *
118 * Non-animated tiles may be rendered faster by using intermediate surfaces
119 * that are drawn only once.
120 * This function should return false if the tile pattern is always drawn the same way.
121 * Returns true by default.
122 *
123 * \return true if this tile pattern is animated
124 */
125bool TilePattern::is_animated() const {
126 return true;
127}
128
129/**
130 * \brief Returns whether tiles having this tile pattern are drawn at their
131 * position.
132 *
133 * Usually, this function returns true, and when it is the case, draw() is
134 * called only for tiles that are located in the current viewport.
135 *
136 * However, some tile patterns may want to be drawn even when they are not
137 * in the viewport, typically to make an illusion of movement like parallax
138 * scrolling.
139 *
140 * \return true if tiles having this pattern are drawn where they are placed
141 */
142bool TilePattern::is_drawn_at_its_position() const {
143 return true;
144}
145
146/**
147 * \brief Fills a rectangle by repeating this tile pattern.
148 * \param dst_surface The destination surface.
149 * \param dst_position Coordinates of the rectangle to fill in \c dst_surface.
150 * \param tileset The tileset to use.
151 * \param viewport Coordinates of the top-left corner of \c dst_surface
152 * relative to the map (may be used for scrolling tiles).
153 */
154void TilePattern::fill_surface(
155 const SurfacePtr& dst_surface,
156 const Rectangle& dst_position,
157 const Tileset& tileset,
158 const Point& viewport
159) const {
160 Point dst;
161
162 int limit_x = dst_position.get_x() + dst_position.get_width();
163 int limit_y = dst_position.get_y() + dst_position.get_height();
164
165 for (int y = dst_position.get_y();
166 y < limit_y;
167 y += get_height()) {
168
169 if ((y <= dst_surface->get_height() && y + get_height() > 0)
170 || !is_drawn_at_its_position()) {
171 dst.y = y;
172
173 for (int x = dst_position.get_x();
174 x < limit_x;
175 x += get_width()) {
176
177 if ((x <= dst_surface->get_width() && x + get_width() > 0)
178 || !is_drawn_at_its_position()) {
179 dst.x = x;
180 draw(dst_surface, dst, tileset, viewport);
181 }
182 }
183 }
184 }
