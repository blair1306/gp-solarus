17#ifndef SOLARUS_MOVEMENT_H
18#define SOLARUS_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Rectangle.h"
22#include "solarus/lua/ExportableToLua.h"
23#include "solarus/lua/ScopedLuaRef.h"
24#include <cstdint>
25#include <string>
26
27namespace Solarus {
28
29class Drawable;
30class Entity;
31
32/**
33 * \brief Abstract class for representing a movement.
34 *
35 * This is the parent class of all kinds of movement.
36 * An instance of Movement can be applied to a map entity during the game,
37 * to a drawable object or to some arbitrary coordinates.
38 *
39 * TODO: generalize this and allow to apply a movement to any Lua object that
40 * has a set_xy method()?
41 */
42class SOLARUS_API Movement: public ExportableToLua {
43
44 public:
45
46 virtual ~Movement();
47
48 // object controlled
49 Entity* get_entity() const;
50 void set_entity(Entity* entity);
51 Drawable* get_drawable() const;
52 void set_drawable(Drawable* drawable);
53 virtual void notify_object_controlled();
54
55 // update
56 virtual void update(); // called repeatedly
57 bool is_suspended() const;
58 virtual void set_suspended(bool suspended);
59
60 // position
61 int get_x() const;
62 int get_y() const;
63 Point get_xy() const;
64 void set_x(int x);
65 void set_y(int y);
66 void set_xy(int x, int y);
67 void set_xy(const Point& xy);
68 void translate_x(int dx);
69 void translate_y(int dy);
70 void translate_xy(int dx, int dy);
71 void translate_xy(const Point& dxy);
72 virtual void notify_position_changed(); // called whenever x or y is changed
73 virtual void notify_obstacle_reached();
74 virtual void notify_movement_changed();
75 virtual void notify_movement_finished();
76
77 // movement
78 bool is_stopped() const;
79 virtual bool is_started() const;
80 virtual void stop();
81 virtual bool is_finished() const;
82
83 // obstacles
84 bool test_collision_with_obstacles(int dx, int dy) const;
85 bool test_collision_with_obstacles(const Point& dxy) const;
86 const Rectangle& get_last_collision_box_on_obstacle() const;
87 bool are_obstacles_ignored() const;
88 void set_ignore_obstacles(bool ignore_obstacles);
89 void restore_default_ignore_obstacles();
90
91 // displaying moving objects
92 virtual int get_displayed_direction4() const;
93 virtual Point get_displayed_xy() const;
94
95 // Lua
96 const ScopedLuaRef& get_finished_callback() const;
97 void set_finished_callback(const ScopedLuaRef& finished_callback_ref);
98 bool are_lua_notifications_enabled() const;
99 void set_lua_notifications_enabled(bool lua_notifications_enabled);
100 virtual const std::string& get_lua_type_name() const override;
101
102 protected:
103
104 Movement();
105 explicit Movement(bool ignore_obstacles);
106
107 // suspended
108 uint32_t get_when_suspended() const;
109
110 // obstacles (only when the movement is applied to an entity)
111 void set_default_ignore_obstacles(bool ignore_obstacles);
112
113 private:
114
115 // Object to move (can be an entity, a drawable or a point).
116 Entity* entity; /**< The entity controlled by this movement. */
117 Drawable* drawable; /**< The drawable controlled by this movement. */
118 Point xy; /**< Coordinates of the point controlled by this movement. */
119
120 uint32_t last_move_date; /**< Date of the last x or y move. */
121 bool finished; /**< true if is_finished() returns true. */
122 bool lua_notifications_enabled; /**< Whether Lua events and callbacks should be called for this movement. */
123
124 // suspended
125 bool suspended; /**< Indicates whether the movement is suspended. */
126 uint32_t when_suspended; /**< Indicates when the movement was suspended. */
127
128 // obstacles (only when the movement is applied to an entity)
129 mutable Rectangle
130 last_collision_box_on_obstacle; /**< Copy of the entity's bounding box of the last call
131 * to test_collision_with_map() returning true. */
132
133 bool default_ignore_obstacles; /**< Indicates that this movement normally ignores obstacles. */
134 bool current_ignore_obstacles; /**< Indicates that this movement currently ignores obstacles. */
135
136 ScopedLuaRef finished_callback_ref; /**< Lua ref to a function to call when this movement finishes. */
137
138};
139
140}
141
142#endif
