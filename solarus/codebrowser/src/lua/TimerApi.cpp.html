32/**
33 * Name of the Lua table representing the timer module.
34 */
35const std::string LuaContext::timer_module_name = "sol.timer";
36
37/**
38 * \brief Initializes the timer features provided to Lua.
39 */
40void LuaContext::register_timer_module() {
41
42 // Functions of sol.timer.
43 static const luaL_Reg functions[] = {
44 { "start", timer_api_start },
45 { "stop_all", timer_api_stop_all },
46 { nullptr, nullptr }
47 };
48
49 // Methods of the timer type.
50 static const luaL_Reg methods[] = {
51 { "stop", timer_api_stop },
52 { "is_with_sound", timer_api_is_with_sound },
53 { "set_with_sound", timer_api_set_with_sound },
54 { "is_suspended", timer_api_is_suspended },
55 { "set_suspended", timer_api_set_suspended },
56 { "is_suspended_with_map", timer_api_is_suspended_with_map },
57 { "set_suspended_with_map", timer_api_set_suspended_with_map },
58 { "get_remaining_time", timer_api_get_remaining_time },
59 { "set_remaining_time", timer_api_set_remaining_time },
60 { nullptr, nullptr }
61 };
62
63 static const luaL_Reg metamethods[] = {
64 { "__gc", userdata_meta_gc },
65 { nullptr, nullptr }
66 };
67
68 register_type(timer_module_name, functions, methods, metamethods);
69}

71/**

74 * \param index An index in the stack.
75 * \return true if the value at this index is a timer.

77bool LuaContext::is_timer(lua_State* l, int index) {

81/**

88TimerPtr LuaContext::check_timer(lua_State* l, int index) {

103/**
104 * \brief Registers a timer into a context (table or a userdata).
105 * \param timer A timer.
106 * \param context_index Index of the table or userdata in the stack.
107 * \param callback_ref Lua ref to the function to call when the timer finishes.
108 */
109void LuaContext::add_timer(
110 const TimerPtr& timer,
111 int context_index,
112 const ScopedLuaRef& callback_ref
113) {
114 const void* context;
115 if (lua_type(l, context_index) == LUA_TUSERDATA) {
116 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
117 lua_touserdata(l, context_index)
118 );
119 context = userdata->get();
120 }
121 else {
122 context = lua_topointer(l, context_index);
123 }
124
125 Debug::execute_if_debug([&] {
126 // Sanity check: check the uniqueness of the ref.
127 for (const auto& kvp: timers) {
128 if (kvp.second.callback_ref.get() == callback_ref.get()) {
129 std::ostringstream oss;
130 oss << "Callback ref " << callback_ref.get()
131 << " is already used by a timer (duplicate luaL_unref?)";
132 Debug::die(oss.str());
133 }
134 }
135 });
136
137 Debug::check_assertion(timers.find(timer) == timers.end(),
138 "Duplicate timer in the system");
139
140 timers[timer].callback_ref = callback_ref;
141 timers[timer].context = context;
142
143 Game* game = main_loop.get_game();
144 if (game != nullptr) {
145 // We are during a game: depending on the timer's context,
146 // suspend the timer or not.
147 if (is_map(l, context_index)
148 || is_entity(l, context_index)
149 || is_item(l, context_index)) {
150
151 bool initially_suspended = false;
152
153 // By default, we want the timer to be automatically suspended when a
154 // camera movement, a dialog or the pause menu starts.
155 if (!is_entity(l, context_index)) {
156 // The timer normally gets suspended/resumed with the map.
157 timer->set_suspended_with_map(true);
158
159 // But in the initial state, we override that rule.
160 // We initially suspend the timer only during a dialog.
161 // In particular, we don't want to suspend timers created during a
162 // camera movement.
163 // This would be very painful for users.
164 initially_suspended = game->is_dialog_enabled();
165 }
166 else {
167 // Entities are more complex: they also get suspended when disabled.
168 // Therefore, they don't simply follow
169 // the map suspended state.
170 EntityPtr entity = check_entity(l, context_index);
171 initially_suspended = entity->is_suspended() || !entity->is_enabled();
172 }
173
174 timer->set_suspended(initially_suspended);
175 }
176 }
177}

179/**
180 * \brief Unregisters a timer associated to a context.
181 *
182 * This function can be called safely even while iterating on the timer list.
183 *
184 * \param timer A timer.
185 */
186void LuaContext::remove_timer(const TimerPtr& timer) {
187 if (timers.find(timer) != timers.end()) {
188 timers[timer].callback_ref.clear();
189 timers_to_remove.push_back(timer);
190 }
191}
192
193/**
194 * \brief Unregisters all timers associated to a context.
195 *
196 * This function can be called safely even while iterating on the timer list.
197 *
198 * \param context_index Index of a table or userdata containing timers.
199 */
200void LuaContext::remove_timers(int context_index) {
201
202 const void* context;
203 if (lua_type(l, context_index) == LUA_TUSERDATA) {
204 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
205 lua_touserdata(l, context_index));
206 context = userdata->get();
207 }
208 else {
209 context = lua_topointer(l, context_index);
210 }
211
212 for (auto& kvp: timers) {
213 const TimerPtr& timer = kvp.first;
214 if (kvp.second.context == context) {
215 kvp.second.callback_ref.clear();
216 timers_to_remove.push_back(timer);
217 }
218 }
219}

224void LuaContext::destroy_timers() {
225 timers.clear();

228/**
229 * \brief Updates all timers currently running for this script.
230 */
231void LuaContext::update_timers() {
232
233 // Update all timers.
234 for (const auto& kvp: timers) {
235
236 const TimerPtr& timer = kvp.first;
237 const ScopedLuaRef& callback_ref = kvp.second.callback_ref;
238 if (!callback_ref.is_empty()) {
239 // The timer is not being removed: update it.
240 timer->update();
241 if (timer->is_finished()) {
242 do_timer_callback(timer);
243 }
244 }
245 }
246
247 // Destroy the ones that should be removed.
248 for (const TimerPtr& timer: timers_to_remove) {
249
250 const auto& it = timers.find(timer);
251 if (it != timers.end()) {
252 timers.erase(it);
253
254 Debug::check_assertion(timers.find(timer) == timers.end(),
255 "Failed to remove timer");
256 }
257 }
258 timers_to_remove.clear();
259}

261/**
262 * \brief This function is called when the game (if any) is being suspended
263 * or resumed.
264 * \param suspended true if the game is suspended, false if it is resumed.
265 */
266void LuaContext::notify_timers_map_suspended(bool suspended) {
267
268 for (const auto& kvp: timers) {
269 const TimerPtr& timer = kvp.first;
270 if (timer->is_suspended_with_map()) {
271 timer->notify_map_suspended(suspended);
272 }
273 }
274}

276/**
277 * \brief Suspends or resumes the timers attached to a map entity.
278 *
279 * This is independent from the Timer::is_suspended_with_map() property.
280 *
281 * \param entity A map entity.
282 * \param suspended \c true to suspend its timers, \c false to resume them.
283 */
284void LuaContext::set_entity_timers_suspended(
285 Entity& entity, bool suspended
286) {
287
288 for (const auto& kvp: timers) {
289 const TimerPtr& timer = kvp.first;
290 if (kvp.second.context == &entity) {
291 timer->set_suspended(suspended);
292 }
293 }
294}

296/**
297 * \brief Executes the callback of a timer.
298 *
299 * Then, if the callback returns \c true, the timer is rescheduled,
300 * otherwise it is discarded.
301 *
302 * Does nothing if the timer is already finished.
303 *
304 * \param timer The timer to execute.
305 */
306void LuaContext::do_timer_callback(const TimerPtr& timer) {
307
308 Debug::check_assertion(timer->is_finished(), "This timer is still running");
309
310 auto it = timers.find(timer);
311 if (it != timers.end() &&
312 !it->second.callback_ref.is_empty()) {
313 ScopedLuaRef& callback_ref = it->second.callback_ref;
314 push_ref(l, callback_ref);
315 const bool success = call_function(0, 1, "timer callback");
316
317 bool repeat = false;
318 if (success) {
319 repeat = lua_isboolean(l, -1) && lua_toboolean(l, -1);
320 lua_pop(l, 1);
321 }
322
323 if (repeat) {
324 // The callback returned true: reschedule the timer.
325 timer->set_expiration_date(timer->get_expiration_date() + timer->get_initial_duration());
326 if (timer->is_finished()) {
327 // Already finished: this is possible if the duration is smaller than
328 // the main loop stepsize.
329 do_timer_callback(timer);
330 }
331 }
332 else {
333 callback_ref.clear();
334 timers_to_remove.push_back(timer);
335 }
336 }
337}

339/**
340 * \brief Implementation of sol.timer.start().
341 * \param l the Lua context that is calling this function
342 * \return number of values to return to Lua
343 */
344int LuaContext::timer_api_start(lua_State *l) {
345
346 return LuaTools::exception_boundary_handle(l, [&] {
347 // Parameters: [context] delay callback.
348 LuaContext& lua_context = get_lua_context(l);
349
350 if (lua_type(l, 1) != LUA_TNUMBER) {
351 // The first parameter is the context.
352 if (lua_type(l, 1) != LUA_TTABLE
353 && lua_type(l, 1) != LUA_TUSERDATA) {
354 LuaTools::type_error(l, 1, "table or userdata");
355 }
356 }
357 else {
358 // No context specified: set a default context:
359 // - during a game: the current map,
360 // - outside a game: sol.main.
361
362 Game* game = lua_context.get_main_loop().get_game();
363 if (game != nullptr && game->has_current_map()) {
364 push_map(l, game->get_current_map());
365 }
366 else {
367 LuaContext::push_main(l);
368 }
369
370 lua_insert(l, 1);
371 }
372 // Now the first parameter is the context.
373
374 uint32_t delay = uint32_t(LuaTools::check_int(l, 2));
375 const ScopedLuaRef& callback_ref = LuaTools::check_function(l, 3);
376
377 // Create the timer.
378 TimerPtr timer = std::make_shared<Timer>(delay);
379 lua_context.add_timer(timer, 1, callback_ref);
380
381 if (delay == 0) {
382 // The delay is zero: call the function right now.
383 lua_context.do_timer_callback(timer);
384 }
385
386 push_timer(l, timer);
387
388 return 1;
389 });
390}

392/**
393 * \brief Implementation of timer:stop().
394 * \param l the Lua context that is calling this function
395 * \return number of values to return to Lua
396 */
397int LuaContext::timer_api_stop(lua_State* l) {
398
399 return LuaTools::exception_boundary_handle(l, [&] {
400 LuaContext& lua_context = get_lua_context(l);
401 const TimerPtr& timer = check_timer(l, 1);
402 lua_context.remove_timer(timer);
403
404 return 0;
405 });
406}

531/**
532 * \brief Implementation of timer:get_remaining_time().
533 * \param l The Lua context that is calling this function.
534 * \return Number of values to return to Lua.
535 */
536int LuaContext::timer_api_get_remaining_time(lua_State* l) {
537
538 return LuaTools::exception_boundary_handle(l, [&] {
539 const TimerPtr& timer = check_timer(l, 1);
540
541 LuaContext& lua_context = get_lua_context(l);
542 const auto it = lua_context.timers.find(timer);
543 if (it == lua_context.timers.end() ||
544 it->second.callback_ref.is_empty()) {
545 // This timer is already finished or was canceled.
546 lua_pushinteger(l, 0);
547 }
548 else {
549 int remaining_time = (int) timer->get_expiration_date() - (int) System::now();
550 if (remaining_time < 0) {
551 remaining_time = 0;
552 }
553 lua_pushinteger(l, remaining_time);
554 }
555 return 1;
556 });
557}

559/**
560 * \brief Implementation of timer:set_remaining_time().
561 * \param l The Lua context that is calling this function.
562 * \return Number of values to return to Lua.
563 */
564int LuaContext::timer_api_set_remaining_time(lua_State* l) {
565
566 return LuaTools::exception_boundary_handle(l, [&] {
567 const TimerPtr& timer = check_timer(l, 1);
568 uint32_t remaining_time = LuaTools::check_int(l, 2);
569
570 LuaContext& lua_context = get_lua_context(l);
571 const auto it = lua_context.timers.find(timer);
572 if (it != lua_context.timers.end() &&
573 !it->second.callback_ref.is_empty()) {
574 // The timer is still active.
575 const uint32_t now = System::now();
576 const uint32_t expiration_date = now + remaining_time;
577 timer->set_expiration_date(expiration_date);
578 if (now >= expiration_date) {
579 // Execute the callback now.
580 lua_context.do_timer_callback(timer);
581 }
582 }
583
584 return 0;
585 });
586}
