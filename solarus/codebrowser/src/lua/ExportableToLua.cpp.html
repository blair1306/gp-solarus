17#include "solarus/lowlevel/Debug.h"
18#include "solarus/lua/ExportableToLua.h"
19#include "solarus/lua/LuaContext.h"
20
21namespace Solarus {
22
23/**
24 * \brief Creates an object exportable to Lua.
25 */
26ExportableToLua::ExportableToLua():
27 lua_context(nullptr),
28 known_to_lua(false),
29 with_lua_table(false) {
30
31}
32
33/**
34 * \brief Destroys this exportable object.
35 */
36ExportableToLua::~ExportableToLua() {
37
38 if (lua_context != nullptr) {
39 lua_context->notify_userdata_destroyed(*this);
40 }
41}
42
43/**
44 * \brief Returns the Solarus Lua API.
45 * \return The Lua context, or nullptr if Lua is not enabled for this object.
46 */
47LuaContext* ExportableToLua::get_lua_context() const {
48 return lua_context;
49}
50
51/**
52 * \brief Sets the Solarus Lua API.
53 * \param lua_context The Lua context, or nullptr to disable Lua for this object.
54 */
55void ExportableToLua::set_lua_context(LuaContext* lua_context) {
56 this->lua_context = lua_context;
57}
58
59/**
60 * \brief Returns whether this object was exported in the Lua side.
61 *
62 * This stays \c true even if the object was collected from Lua.
63 *
64 * \return \c true if this object was exported in the Lua side at least once.
65 */
66bool ExportableToLua::is_known_to_lua() const {
67 return known_to_lua;
68}
69
70/**
71 * \brief Sets whether this object was exported in the Lua side.
72 * \param known_to_lua \c true if this object was exported in the Lua side
73 * at least once.
74 */
75void ExportableToLua::set_known_to_lua(bool known_to_lua) {
76 this->known_to_lua = known_to_lua;
77}
78
79/**
80 * \brief Returns whether this userdata has an associated table in Lua.
81 *
82 * This is the case if __newindex was called at least once for the userdata.
83 * It might be \c true even if the userdata does not exist anymore in Lua.
84 *
85 * \return \c true if there is a Lua table for this userdata.
86 */
87bool ExportableToLua::is_with_lua_table() const {
88 return with_lua_table;
89}
90
91/**
92 * \brief Sets whether this userdata has an associated table in Lua.
93 * \param with_lua_table \c true if there is a Lua table for this userdata.
94 */
95void ExportableToLua::set_with_lua_table(bool with_lua_table) {
96 this->with_lua_table = with_lua_table;
97}
98
99}
