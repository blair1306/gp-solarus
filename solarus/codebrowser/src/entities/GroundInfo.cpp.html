47/**
48 * \brief Returns whether a ground value represents a diagonal obstacle.
49 * \return ground The ground to test.
50 * \retrun \c true if this is diagonal ground.
51 */
52bool GroundInfo::is_ground_diagonal(Ground ground) {
53
54 switch (ground) {
55
56 case Ground::WALL_TOP_RIGHT:
57 case Ground::WALL_TOP_LEFT:
58 case Ground::WALL_BOTTOM_LEFT:
59 case Ground::WALL_BOTTOM_RIGHT:
60 case Ground::WALL_TOP_RIGHT_WATER:
61 case Ground::WALL_TOP_LEFT_WATER:
62 case Ground::WALL_BOTTOM_LEFT_WATER:
63 case Ground::WALL_BOTTOM_RIGHT_WATER:
64 return true;
65
66 case Ground::EMPTY:
67 case Ground::TRAVERSABLE:
68 case Ground::WALL:
69 case Ground::LOW_WALL:
70 case Ground::DEEP_WATER:
71 case Ground::SHALLOW_WATER:
72 case Ground::GRASS:
73 case Ground::HOLE:
74 case Ground::ICE:
75 case Ground::LADDER:
76 case Ground::PRICKLE:
77 case Ground::LAVA:
78 return false;
79
80 }
81
82 return false;
83}
