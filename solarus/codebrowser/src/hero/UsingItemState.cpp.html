24/**
25 * \brief Constructor.
26 * \param hero The hero controlled by this state.
27 * \param item The equipment item to use.
28 */
29Hero::UsingItemState::UsingItemState(
30 Hero& hero,
31 EquipmentItem& item):
32
33 HeroState(hero, "using item"),
34 item_usage(hero.get_game(), item) {
35
36}
37
38/**
39 * \brief Starts this state.
40 * \param previous_state The previous state.
41 */
42void Hero::UsingItemState::start(const State* previous_state) {
43
44 HeroState::start(previous_state);
45
46 bool interaction = false;
47 Entity* facing_entity = get_entity().get_facing_entity();
48 if (facing_entity != nullptr && !facing_entity->is_being_removed()) {
49 // Maybe the facing entity (e.g. an NPC) accepts an interaction with this
50 // particular item.
51 interaction = facing_entity->notify_interaction_with_item(item_usage.get_item());
52 }
53
54 if (!interaction) {
55 // No interaction occurred with the facing entity: use the item normally.
56 item_usage.start();
57 }
58}
59
60/**
61 * \brief Updates this state.
62 */
63void Hero::UsingItemState::update() {
64
65 HeroState::update();
66
67 item_usage.update();
68 if (item_usage.is_finished() && is_current_state()) {
69 // if the state was not modified by the item, return to the normal state
70 Hero& hero = get_entity();
71 hero.set_state(new FreeState(hero));
72 }
73}
74
75/**
76 * \brief Returns whether the hero is using an equipment item in this state.
77 * \return true if the hero is using an equipment item.
78 */
79bool Hero::UsingItemState::is_using_item() const {
80 return true;
81}
82
83/**
84 * \brief When the hero is using an equipment item, returns that item.
85 * \return The current equipment item.
86 */
87EquipmentItemUsage& Hero::UsingItemState::get_item_being_used() {
88 return item_usage;
89}
90
91/**
92 * \copydoc Entity::State::can_avoid_stream
93 */
94bool Hero::UsingItemState::can_avoid_stream(const Stream& stream) const {
95 // If the hero can use items on this stream, allow to stay in UsingItemState.
96 return stream.get_allow_item();
97}
