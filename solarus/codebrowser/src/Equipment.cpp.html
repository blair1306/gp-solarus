90/**
91 * \brief This function is be called repeatedly by the game.
92 *
93 * Most of the time, there is nothing to update in this class.
94 * The only elements updated here are some dynamic things
95 * such as equipment-related animations or dialogs.
96 */
97void Equipment::update() {
98
99 Game* game = savegame.get_game();
100 if (game == nullptr) {
101 // Nothing dynamic when there is no game.
102 return;
103 }
104
105 // Check if the game is suspended.
106 bool game_suspended = game->is_suspended();
107 if (suspended != game_suspended) {
108 set_suspended(game_suspended);
109 }
110
111 // Update item scripts.
112 for (const auto& kvp: items) {
113 EquipmentItem& item = *kvp.second;
114 item.update();
115 }
116}

118/**
119 * \brief This function is called when the game is suspended or resumed.
120 * \param suspended true if the game is suspended, false if it is resumed
121 */
122void Equipment::set_suspended(bool suspended) {
123
124 this->suspended = suspended;
125
126 // Notify the item scripts.
127 for (const auto& kvp: items) {
128 EquipmentItem& item = *kvp.second;
129 item.set_suspended(suspended);
130 }
131}

143 * \param max_money the player's maximum amount number of money
144 */
145void Equipment::set_max_money(int max_money) {
146
147 Debug::check_assertion(max_money >= 0, "Invalid money amount to add");
148
149 savegame.set_integer(Savegame::KEY_MAX_MONEY, max_money);
150
151 // If the max money is reduced, make sure the current money does not exceed
152 // the new maximum.
153 if (get_money() > get_max_money()) {
154 set_money(max_money);
155 }
156}

166/**
167 * \brief Sets the player's current amount of money.
168 *
169 * If the amount is lower than zero, it is replaced by zero.
170 * If the amount is greater than get_max_money(), it is replaced by that value.
171 *
172 * \param money The player's new amount of money.
173 */
174void Equipment::set_money(int money) {
175
176 money = std::max(0, std::min(get_max_money(), money));
177 savegame.set_integer(Savegame::KEY_CURRENT_MONEY, money);
178}

383/**
384 * \brief Runs the Lua script of each equipment item.
385 */
386void Equipment::load_items() {
387
388 // Create each equipment item declared in project_db.dat.
389 const std::map<std::string, std::string>& item_elements =
390 CurrentQuest::get_resources(ResourceType::ITEM);
391 for (const auto& kvp: item_elements) {
392 const std::string& item_id = kvp.first;
393 std::shared_ptr<EquipmentItem> item = std::make_shared<EquipmentItem>(*this);
394 item->set_name(item_id);
395 items[item_id] = item;
396 }
397
398 // Load the item scripts.
399 for (const auto& kvp: items) {
400 EquipmentItem& item = *kvp.second;
401 item.initialize();
402 }
403
404 // Start the items once they all exist.
405 for (const auto& kvp: items) {
406 EquipmentItem& item = *kvp.second;
407 item.start();
408 }
409}

447/**
448 * \brief Returns the item currently assigned to a slot.
449 * \param slot Slot of the item to get (1 or 2).
450 * \return The item currently assigned to this slot or nullptr.
451 */
452EquipmentItem* Equipment::get_item_assigned(int slot) {
453
454 // TODO don't hardcode item slots
455
456 Debug::check_assertion(slot >= 1 && slot <= 2,
457 "Invalid item slot");
458
459 char savegame_variable[] = "_item_slot_0";
460 savegame_variable[11] = '0' + slot;
461 const std::string& item_name = savegame.get_string(savegame_variable);
462
463 EquipmentItem* item = nullptr;
464 if (!item_name.empty()) {
465 item = &get_item(item_name);
466 }
467 return item;
468}
469
470/**
471 * \brief Returns the item currently assigned to a slot.
472 * \param slot Slot of the item to get (1 or 2).
473 * \return The item currently assigned to this slot or nullptr.
474 */
475const EquipmentItem* Equipment::get_item_assigned(int slot) const {
476
477 Debug::check_assertion(slot >= 1 && slot <= 2,
478 "Invalid item slot");
479
480 std::ostringstream oss;
481 oss << "_item_slot_" << slot;
482 const std::string& item_name = savegame.get_string(oss.str());
483
484 const EquipmentItem* item = nullptr;
485 if (!item_name.empty()) {
486 item = &get_item(item_name);
487 }
488 return item;
489}

491/**
492 * \brief Assigns an item to a slot.
493 *
494 * The program exits with an error message if the specified item
495 * cannot be assigned or if the player does not have it.
496 *
497 * \param slot Slot to set (1 or 2).
498 * \param item The item to assign to this slot or nullptr to empty the slot.
499 */
500void Equipment::set_item_assigned(int slot, EquipmentItem* item) {
501
502 Debug::check_assertion(slot >= 1 && slot <= 2,
503 "Invalid item slot");
504
505 std::ostringstream oss;
506 oss << "_item_slot_" << slot;
507
508 if (item != nullptr) {
509 Debug::check_assertion(item->get_variant() > 0,
510 std::string("Cannot assign item '") + item->get_name()
511 + "' because the player does not have it");
512 Debug::check_assertion(item->is_assignable(),
513 std::string("The item '") + item->get_name()
514 + "' cannot be assigned");
515 savegame.set_string(oss.str(), item->get_name());
516 }
517 else {
518 savegame.set_string(oss.str(), "");
519 }
520}
521
522/**
523 * \brief Returns the slot (1 or 2) where the specified item is currently assigned.
524 * \param item The item to find.
525 * \return The slot of this item, or 0 if this item is not assigned
526 */
527int Equipment::get_item_slot(const EquipmentItem& item) const {
528
529 for (int i = 1; i <= 2; ++i) {
530 const EquipmentItem* assigned_item = get_item_assigned(i);
531 if (assigned_item != nullptr && assigned_item->get_name() == item.get_name()) {
532 return i;
533 }
534 }
535
536 return 0;
537}

541/**
542 * \brief Returns the index of the savegame variable that stores the specified ability.
543 * \param ability An ability.
544 * \return Name of the boolean savegame variable that stores this ability.
545 */
546std::string Equipment::get_ability_savegame_variable(Ability ability) const {
547
548 switch (ability) {
549
550 case Ability::TUNIC:
551 return Savegame::KEY_ABILITY_TUNIC;
552
553 case Ability::SWORD:
554 return Savegame::KEY_ABILITY_SWORD;
555
556 case Ability::SWORD_KNOWLEDGE:
557 return Savegame::KEY_ABILITY_SWORD_KNOWLEDGE;
558
559 case Ability::SHIELD:
560 return Savegame::KEY_ABILITY_SHIELD;
561
562 case Ability::LIFT:
563 return Savegame::KEY_ABILITY_LIFT;
564
565 case Ability::SWIM:
566 return Savegame::KEY_ABILITY_SWIM;
567
568 case Ability::JUMP_OVER_WATER:
569 return Savegame::KEY_ABILITY_JUMP_OVER_WATER;
570
571 case Ability::RUN:
572 return Savegame::KEY_ABILITY_RUN;
573
574 case Ability::DETECT_WEAK_WALLS:
575 return Savegame::KEY_ABILITY_DETECT_WEAK_WALLS;
576 }
577
578 Debug::die("Invalid ability");
579}

591/**
592 * \brief Returns the level of the specified ability.
593 * \param ability The ability to get.
594 * \return The level of this ability.
595 */
596int Equipment::get_ability(Ability ability) const {
597 return savegame.get_integer(get_ability_savegame_variable(ability));
598}

600/**
601 * \brief Sets the level of the specified ability.
602 * \param ability The ability to set.
603 * \param level The level of this ability.
604 */
605void Equipment::set_ability(Ability ability, int level) {
606
607 savegame.set_integer(get_ability_savegame_variable(ability), level);
608
609 Game* game = get_game();
610 if (game != nullptr) {
611 if (ability == Ability::TUNIC ||
612 ability == Ability::SWORD ||
613 ability == Ability::SHIELD) {
614 // The hero's sprites may depend on these abilities.
615 game->get_hero()->rebuild_equipment();
616 }
617 }
618}
