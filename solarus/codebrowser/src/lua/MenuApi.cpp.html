31/**
32 * \brief Initializes the menu features provided to Lua.
33 */
34void LuaContext::register_menu_module() {
35
36 // Functions of sol.menu.
37 static const luaL_Reg functions[] = {
38 { "start", menu_api_start },
39 { "stop", menu_api_stop },
40 { "stop_all", menu_api_stop_all },
41 { "is_started", menu_api_is_started },
42 { nullptr, nullptr }
43 };
44
45 register_functions(menu_module_name, functions);
46}

48/**
49 * \brief Registers a menu into a context (table or a userdata).
50 *
51 * This function can be called safely even while iterating on the menus list.
52 *
53 * \param menu_ref Lua ref of the menu to add.
54 * \param context_index Index of the table or userdata in the stack.
55 * \param on_top \c true to place this menu on top of existing one in the
56 * same context, \c false to place it behind.
57 */
58void LuaContext::add_menu(
59 const ScopedLuaRef& menu_ref,
60 int context_index,
61 bool on_top
62) {
63 const void* context;
64 if (lua_type(l, context_index) == LUA_TUSERDATA) {
65 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
66 lua_touserdata(l, context_index));
67 context = userdata->get();
68 }
69 else {
70 context = lua_topointer(l, context_index);
71 }
72
73 if (on_top) {
74 menus.emplace_back(menu_ref, context);
75 }
76 else {
77 menus.emplace_front(menu_ref, context);
78 }
79
80 menu_on_started(menu_ref);
81}

83/**
84 * \brief Unregisters all menus associated to a context.
85 *
86 * This function can be called safely even while iterating on the menus list.
87 *
88 * \param context_index Index of a table or userdata containing menus.
89 */
90void LuaContext::remove_menus(int context_index) {
91
92 const void* context;
93 if (lua_type(l, context_index) == LUA_TUSERDATA) {
94 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
95 lua_touserdata(l, context_index));
96 context = userdata->get();
97 }
98 else {
99 context = lua_topointer(l, context_index);
100 }
101
102 // Some menu:on_finished() callbacks may create menus themselves,
103 // and we don't want those new menus to get removed.
104 for (LuaMenuData& menu: menus) {
105 menu.recently_added = false;
106 }
107
108 for (LuaMenuData& menu: menus) {
109 ScopedLuaRef menu_ref = menu.ref;
110 if (menu.context == context && !menu.recently_added) {
111 menu.ref.clear();
112 menu.context = nullptr;
113 menu_on_finished(menu_ref);
114 }
115 }
116}

118/**
119 * \brief Unregisters all existing menus.
120 *
121 * This function can be called safely even while iterating on the menus list.
122 */
123void LuaContext::remove_menus() {
124
125 // Some menu:on_finished() callbacks may create menus themselves,
126 // and we don't want those new menus to get removed.
127 for (LuaMenuData& menu: menus) {
128 menu.recently_added = false;
129 }
130
131 for (LuaMenuData& menu: menus) {
132
133 if (!menu.recently_added) {
134 ScopedLuaRef menu_ref = menu.ref;
135 if (!menu_ref.is_empty()) {
136 menu.ref.clear();
137 menu.context = nullptr;
138 menu_on_finished(menu_ref);
139 }
140 }
141 }
142}

144/**
145 * \brief Destroys immediately all existing menus.
146 */
147void LuaContext::destroy_menus() {
148
149 menus.clear();
150}
151
152/**
153 * \brief Checks all menus and removes the ones that have to be removed.
154 *
155 * Note that the on_update() is called by the context of each menu, not
156 * by this function.
157 */
158void LuaContext::update_menus() {
159
160 // Destroy the ones that should be removed.
161 for (auto it = menus.begin();
162 it != menus.end();
163 // No ++it (elements may be removed while traversing).
164 ) {
165 it->recently_added = false;
166 if (it->ref.is_empty()) {
167 // Empty ref on a menu means that we should remove.
168 // In this case, context must also be nullptr.
169 Debug::check_assertion(it->context == nullptr, "Menu with context and no ref");
170 it = menus.erase(it);
171 }
172 else {
173 ++it;
174 }
175 }
176}

178/**
179 * \brief Implementation of sol.menu.start().
180 * \param l The Lua context that is calling this function.
181 * \return Number of values to return to Lua.
182 */
183int LuaContext::menu_api_start(lua_State *l) {
184
185 return LuaTools::exception_boundary_handle(l, [&] {
186 // Parameters: context table.
187 if (lua_type(l, 1) != LUA_TTABLE
188 && lua_type(l, 1) != LUA_TUSERDATA) {
189 LuaTools::type_error(l, 1, "table or userdata");
190 }
191 LuaTools::check_type(l, 2, LUA_TTABLE);
192 bool on_top = LuaTools::opt_boolean(l, 3, true);
193 lua_settop(l, 2);
194
195 LuaContext& lua_context = get_lua_context(l);
196 const ScopedLuaRef& menu_ref = lua_context.create_ref();
197 lua_context.add_menu(menu_ref, 1, on_top);
198
199 return 0;
200 });
201}

203/**
204 * \brief Implementation of sol.menu.stop().
205 * \param l the Lua context that is calling this function
206 * \return number of values to return to Lua
207 */
208int LuaContext::menu_api_stop(lua_State* l) {
209
210 return LuaTools::exception_boundary_handle(l, [&] {
211 LuaContext& lua_context = get_lua_context(l);
212
213 LuaTools::check_type(l, 1, LUA_TTABLE);
214
215 std::list<LuaMenuData>& menus = lua_context.menus;
216 for (LuaMenuData& menu: menus) {
217 push_ref(l, menu.ref);
218 if (lua_equal(l, 1, -1)) {
219 ScopedLuaRef menu_ref = menu.ref; // Don't erase it immediately since we may be iterating over menus.
220 menu.ref.clear();
221 menu.context = nullptr;
222 lua_context.menu_on_finished(menu_ref);
223 lua_pop(l, 1);
224 break;
225 }
226 lua_pop(l, 1);
227 }
228
229 return 0;
230 });
231}

233/**
234 * \brief Implementation of sol.menu.stop_all().
235 * \param l the Lua context that is calling this function
236 * \return number of values to return to Lua
237 */
238int LuaContext::menu_api_stop_all(lua_State* l) {
239
240 return LuaTools::exception_boundary_handle(l, [&] {
241 if (lua_type(l, 1) != LUA_TTABLE
242 && lua_type(l, 1) != LUA_TUSERDATA) {
243 LuaTools::type_error(l, 1, "table, game or map");
244 }
245
246 get_lua_context(l).remove_menus(1);
247
248 return 0;
249 });
250}

252/**
253 * \brief Implementation of sol.menu.is_started().
254 * \param l the Lua context that is calling this function
255 * \return number of values to return to Lua
256 */
257int LuaContext::menu_api_is_started(lua_State* l) {
258
259 return LuaTools::exception_boundary_handle(l, [&] {
260 LuaContext& lua_context = get_lua_context(l);
261
262 LuaTools::check_type(l, 1, LUA_TTABLE);
263
264 bool found = false;
265 std::list<LuaMenuData>& menus = lua_context.menus;
266 for (LuaMenuData& menu: menus) {
267 push_ref(l, menu.ref);
268 found = lua_equal(l, 1, -1);
269 lua_pop(l, 1);
270
271 if (found) {
272 break;
273 }
274 }
275
276 lua_pushboolean(l, found);
277
278 return 1;
279 });
280}

293/**
294 * \brief Calls the on_finished() method of a Lua menu.
295 * \param menu_ref A reference to the menu object.
296 */
297void LuaContext::menu_on_finished(const ScopedLuaRef& menu_ref) {
298
299 push_ref(l, menu_ref);
300 remove_menus(-1); // First, stop children menus if any.
301 on_finished();
302 remove_timers(-1); // Stop timers associated to this menu.
303 lua_pop(l, 1);
304}

333/**
334 * \brief Calls an input callback method of a Lua menu.
335 * \param menu_ref A reference to the menu object.
336 * \param event The input event to forward.
337 * \return \c true if the event was handled and should stop being propagated.
338 */
339bool LuaContext::menu_on_input(
340 const ScopedLuaRef& menu_ref,
341 const InputEvent& event
342) {
343 // Get the Lua menu.
344 push_ref(l, menu_ref);
345
346 // Send the event to children menus first.
347 bool handled = menus_on_input(-1, event);
348
349 if (!handled) {
350 // Sent the event to this menu.
351 handled = on_input(event);
352 }
353
354 // Remove the menu from the stack.
355 lua_pop(l, 1);
356
357 return handled;
358}

410/**
411 * \brief Calls the on_update() method of the menus associated to a context.
412 * \param context_index Index of an object with menus.
413 */
414void LuaContext::menus_on_update(int context_index) {
415
416 const void* context;
417 if (lua_type(l, context_index) == LUA_TUSERDATA) {
418 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
419 lua_touserdata(l, context_index));
420 context = userdata->get();
421 }
422 else {
423 context = lua_topointer(l, context_index);
424 }
425
426 for (LuaMenuData& menu: menus) {
427 if (menu.context == context) {
428 menu_on_update(menu.ref);
429 }
430 }

457/**
458 * \brief Calls the on_input() method of the menus associated to a context.
459 * \param context_index Index of an object with menus.
460 * \param event The input event to handle.
461 * \return \c true if the event was handled and should stop being propagated.
462 */
463bool LuaContext::menus_on_input(int context_index, const InputEvent& event) {
464
465 const void* context;
466 if (lua_type(l, context_index) == LUA_TUSERDATA) {
467 ExportableToLuaPtr* userdata = static_cast<ExportableToLuaPtr*>(
468 lua_touserdata(l, context_index));
469 context = userdata->get();
470 }
471 else {
472 context = lua_topointer(l, context_index);
473 }
474
475 bool handled = false;
476 std::list<LuaMenuData>::reverse_iterator it;
477 for (it = menus.rbegin(); it != menus.rend() && !handled; ++it) {
478 const ScopedLuaRef& menu_ref = it->ref;
479 if (it->context == context) {
480 handled = menu_on_input(menu_ref, event);
481 }
482 }
483
484 return handled;
485}
