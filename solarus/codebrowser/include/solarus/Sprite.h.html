17#ifndef SOLARUS_SPRITE_H
18#define SOLARUS_SPRITE_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/NonAnimatedRegions.h"
22#include "solarus/lua/ScopedLuaRef.h"
23#include "solarus/Drawable.h"
24#include "solarus/SpritePtr.h"
25#include <map>
26#include <string>
27
28namespace Solarus {
29
30class Size;
31class SpriteAnimation;
32class SpriteAnimationSet;
33class Tileset;
34
35/**
36 * \brief Represents an animated sprite.
37 *
38 * A sprite is represented as a set of animations
39 * (i.e. an instance of SpriteAnimationSet),
40 * a current animation,
41 * a current direction and a current frame.
42 * Several sprites can have the same animation set (i.e. they share
43 * the same SpriteAnimationSet object).
44 *
45 * A sprite can be drawn directly on a surface, or it can
46 * be attached to a map entity.
47 */
48class Sprite: public Drawable {
49
50 public:
51
52 // initialization
53 static void initialize();
54 static void quit();
55
56 // creation and destruction
57 explicit Sprite(const std::string& id);
58
59 void set_tileset(const Tileset& tileset);
60
61 // animation set
62 const std::string& get_animation_set_id() const;
63 const SpriteAnimationSet& get_animation_set() const;
64 void enable_pixel_collisions();
65 bool are_pixel_collisions_enabled() const;
66
67 // size and origin point
68 virtual Size get_size() const override;
69 const Size& get_max_size() const;
70 virtual Point get_origin() const override;
71 const Rectangle& get_max_bounding_box() const;
72
73 // animation state
74 const std::string& get_current_animation() const;
75 void set_current_animation(const std::string& animation_name);
76 bool has_animation(const std::string& animation_name) const;
77 int get_current_direction() const;
78 int get_nb_directions() const;
79 void set_current_direction(int current_direction);
80 int get_nb_frames() const;
81 int get_current_frame() const;
82 void set_current_frame(int current_frame, bool notify_script = true);
83 Rectangle get_current_frame_rectangle() const;
84 uint32_t get_frame_delay() const;
85 void set_frame_delay(uint32_t frame_delay);
86 uint32_t get_next_frame_date() const;
87 void set_synchronized_to(const SpritePtr& other);
88
89 bool is_animation_started() const;
90 void start_animation();
91 void restart_animation();
92 void stop_animation();
93
94 virtual void set_suspended(bool suspended) override;
95 void set_ignore_suspend(bool ignore_suspend);
96 bool is_paused() const;
97 void set_paused(bool suspended);
98 bool is_animation_looping() const;
99 bool is_animation_finished() const;
100 bool is_last_frame_reached() const;
101 bool has_frame_changed() const;
102
103 // effects
104 bool is_blinking() const;
105 void set_blinking(uint32_t blink_delay);
106
107 // collisions
108 bool test_collision(const Sprite& other, int x1, int y1, int x2, int y2) const;
109
110 // update and draw
111 virtual void update() override;
112 virtual void raw_draw(Surface& dst_surface, const Point& dst_position) override;
113 virtual void raw_draw_region(const Rectangle& region,
114 Surface& dst_surface, const Point& dst_position) override;
115 virtual void draw_transition(Transition& transition) override;
116 virtual Surface& get_transition_surface() override;
117
118 // Lua
119 const ScopedLuaRef& get_finished_callback() const;
120 void set_finished_callback(const ScopedLuaRef& finished_callback_ref);
121 virtual const std::string& get_lua_type_name() const override;
122
123 private:
124
125 static SpriteAnimationSet& get_animation_set(const std::string& id);
126 int get_next_frame() const;
127 Surface& get_intermediate_surface() const ;
128 void set_frame_changed(bool frame_changed);
129 void notify_finished();
130
131 // animation set
132 static std::map<std::string, SpriteAnimationSet*> all_animation_sets;
133 const std::string animation_set_id; /**< id of this sprite's animation set */
134 SpriteAnimationSet& animation_set; /**< animation set of this sprite */
135
136 // current state of the sprite
137
138 std::string current_animation_name; /**< name of the current animation */
139 SpriteAnimation* current_animation; /**< the current animation or nullptr if the sprite sheet has no animation */
140 int current_direction; /**< current direction of the animation (the first one is number 0);
141 * it can be different from the movement direction
142 * of the entity, because sometimes a sprite can
143 * go backwards. */
144 int current_frame; /**< current frame of the animation (the first one is number 0) */
145 bool frame_changed; /**< indicates that the frame has just changed */
146
147 uint32_t frame_delay; /**< delay between two frames in milliseconds */
148 uint32_t next_frame_date; /**< date of the next frame */
149
150 bool ignore_suspend; /**< true to continue playing the animation even when the game is suspended */
151 bool paused; /**< true if the animation is paused */
152 bool finished; /**< true if the animation has been stopped because the last frame is finished */
153 SpritePtr
154 synchronize_to; /**< another sprite to synchronize the frame to
155 * when they have the same animation name (or nullptr) */
156
157 // effects
158 mutable SurfacePtr
159 intermediate_surface; /**< an intermediate surface used to show transitions and other effects */
160 uint32_t blink_delay; /**< blink delay of the sprite, or zero if the sprite is not blinking */
161 bool blink_is_sprite_visible; /**< when blinking, true if the sprite is visible or false if it is invisible */
162 uint32_t blink_next_change_date; /**< date of the next change when blinking: visible or not */
163
164 ScopedLuaRef
165 finished_callback_ref; /**< Lua ref to an action to do when this movement finishes.
166 * Automatically cleared when executed. */
167
168};
169
170}
171
172#endif
