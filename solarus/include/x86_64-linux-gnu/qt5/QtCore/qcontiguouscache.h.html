54struct Q_CORE_EXPORT QContiguousCacheData
55{
56 QBasicAtomicInt ref;
57 int alloc;
58 int count;
59 int start;
60 int offset;
61 uint sharable : 1;
62 uint reserved : 31;
63
64 // total is 24 bytes (HP-UX aCC: 40 bytes)
65 // the next entry is already aligned to 8 bytes
66 // there will be an 8 byte gap here if T requires 16-byte alignment
67 // (such as long double on 64-bit platforms, __int128, __float128)
68
69 static QContiguousCacheData *allocateData(int size, int alignment);
70 static void freeData(QContiguousCacheData *data);
71
72#ifdef QT_QCONTIGUOUSCACHE_DEBUG
73 void dump() const;
74#endif
75};
76
77template <typename T>
78struct QContiguousCacheTypedData: private QContiguousCacheData
79{
80 // private inheritance to avoid aliasing warningss
81 T array[1];
82
83 static inline void freeData(QContiguousCacheTypedData *data) { QContiguousCacheData::freeData(data); }
84};

86template<typename T>
87class QContiguousCache {
88 typedef QContiguousCacheTypedData<T> Data;
89 union { QContiguousCacheData *d; QContiguousCacheTypedData<T> *p; };
90public:
91 // STL compatibility
92 typedef T value_type;
93 typedef value_type* pointer;
94 typedef const value_type* const_pointer;
95 typedef value_type& reference;
96 typedef const value_type& const_reference;
97 typedef qptrdiff difference_type;
98 typedef int size_type;
99
100 explicit QContiguousCache(int capacity = 0);
101 QContiguousCache(const QContiguousCache<T> &v) : d(v.d) { d->ref.ref(); if (!d->sharable) detach_helper(); }
102
103 inline ~QContiguousCache() { if (!d) return; if (!d->ref.deref()) freeData(p); }
104
105 inline void detach() { if (d->ref.load() != 1) detach_helper(); }
106 inline bool isDetached() const { return d->ref.load() == 1; }
107#if QT_SUPPORTS(UNSHARABLE_CONTAINERS)
108 inline void setSharable(bool sharable) { if (!sharable) detach(); d->sharable = sharable; }
109#endif
110
111 QContiguousCache<T> &operator=(const QContiguousCache<T> &other);
112#ifdef Q_COMPILER_RVALUE_REFS
113 inline QContiguousCache<T> &operator=(QContiguousCache<T> &&other)
114 { qSwap(d, other.d); return *this; }
115#endif
116 inline void swap(QContiguousCache<T> &other) { qSwap(d, other.d); }
117 bool operator==(const QContiguousCache<T> &other) const;
118 inline bool operator!=(const QContiguousCache<T> &other) const { return !(*this == other); }
119
120 inline int capacity() const {return d->alloc; }
121 inline int count() const { return d->count; }
122 inline int size() const { return d->count; }
123
124 inline bool isEmpty() const { return d->count == 0; }
125 inline bool isFull() const { return d->count == d->alloc; }
126 inline int available() const { return d->alloc - d->count; }
127
128 void clear();
129 void setCapacity(int size);
130
131 const T &at(int pos) const;
132 T &operator[](int i);
133 const T &operator[](int i) const;
134
135 void append(const T &value);
136 void prepend(const T &value);
137 void insert(int pos, const T &value);
138
139 inline bool containsIndex(int pos) const { return pos >= d->offset && pos - d->offset < d->count; }
140 inline int firstIndex() const { return d->offset; }
141 inline int lastIndex() const { return d->offset + d->count - 1; }
142
143 inline const T &first() const { Q_ASSERT(!isEmpty()); return p->array[d->start]; }
144 inline const T &last() const { Q_ASSERT(!isEmpty()); return p->array[(d->start + d->count -1) % d->alloc]; }
145 inline T &first() { Q_ASSERT(!isEmpty()); detach(); return p->array[d->start]; }
146 inline T &last() { Q_ASSERT(!isEmpty()); detach(); return p->array[(d->start + d->count -1) % d->alloc]; }
147
148 void removeFirst();
149 T takeFirst();
150 void removeLast();
151 T takeLast();
152
153 inline bool areIndexesValid() const
154 { return d->offset >= 0 && d->offset < INT_MAX - d->count && (d->offset % d->alloc) == d->start; }
155
156 inline void normalizeIndexes() { d->offset = d->start; }
157
158#ifdef QT_QCONTIGUOUSCACHE_DEBUG
159 void dump() const { p->dump(); }
160#endif
161private:
162 void detach_helper();
163
164 QContiguousCacheData *allocateData(int aalloc);
165 void freeData(Data *x);
166 int sizeOfTypedData() {
167 // this is more or less the same as sizeof(Data), except that it doesn't
168 // count the padding at the end
169 return reinterpret_cast<const char *>(&(reinterpret_cast<const Data *>(this))->array[1]) - reinterpret_cast<const char *>(this);
170 }
171 int alignOfTypedData() const
172 {
173 return qMax<int>(sizeof(void*), Q_ALIGNOF(Data));
174 }
175};

177template <typename T>
178void QContiguousCache<T>::detach_helper()
179{
180 union { QContiguousCacheData *d; QContiguousCacheTypedData<T> *p; } x;
181
182 x.d = allocateData(d->alloc);
183 x.d->ref.store(1);
184 x.d->count = d->count;
185 x.d->start = d->start;
186 x.d->offset = d->offset;
187 x.d->alloc = d->alloc;
188 x.d->sharable = true;
189 x.d->reserved = 0;
190
191 T *dest = x.p->array + x.d->start;
192 T *src = p->array + d->start;
193 int oldcount = x.d->count;
194 while (oldcount--) {
195 if (QTypeInfo<T>::isComplex) {
196 new (dest) T(*src);
197 } else {
198 *dest = *src;
199 }
200 dest++;
201 if (dest == x.p->array + x.d->alloc)
202 dest = x.p->array;
203 src++;
204 if (src == p->array + d->alloc)
205 src = p->array;
206 }
207
208 if (!d->ref.deref())
209 freeData(p);
210 d = x.d;
211}

213template <typename T>
214void QContiguousCache<T>::setCapacity(int asize)
215{
216 if (asize == d->alloc)
217 return;
218 detach();
219 union { QContiguousCacheData *d; QContiguousCacheTypedData<T> *p; } x;
220 x.d = allocateData(asize);
221 x.d->alloc = asize;
222 x.d->count = qMin(d->count, asize);
223 x.d->offset = d->offset + d->count - x.d->count;
224 if(asize)
225 x.d->start = x.d->offset % x.d->alloc;
226 else
227 x.d->start = 0;
228
229 int oldcount = x.d->count;
230 if(oldcount)
231 {
232 T *dest = x.p->array + (x.d->start + x.d->count-1) % x.d->alloc;
233 T *src = p->array + (d->start + d->count-1) % d->alloc;
234 while (oldcount--) {
235 if (QTypeInfo<T>::isComplex) {
236 new (dest) T(*src);
237 } else {
238 *dest = *src;
239 }
240 if (dest == x.p->array)
241 dest = x.p->array + x.d->alloc;
242 dest--;
243 if (src == p->array)
244 src = p->array + d->alloc;
245 src--;
246 }
247 }
248 /* free old */
249 freeData(p);
250 d = x.d;
251}

253template <typename T>
254void QContiguousCache<T>::clear()
255{
256 if (d->ref.load() == 1) {
257 if (QTypeInfo<T>::isComplex) {
258 int oldcount = d->count;
259 T * i = p->array + d->start;
260 T * e = p->array + d->alloc;
261 while (oldcount--) {
262 i->~T();
263 i++;
264 if (i == e)
265 i = p->array;
266 }
267 }
268 d->count = d->start = d->offset = 0;
269 } else {
270 union { QContiguousCacheData *d; QContiguousCacheTypedData<T> *p; } x;
271 x.d = allocateData(d->alloc);
272 x.d->ref.store(1);
273 x.d->alloc = d->alloc;
274 x.d->count = x.d->start = x.d->offset = 0;
275 x.d->sharable = true;
276 if (!d->ref.deref()) freeData(p);
277 d = x.d;
278 }
279}

281template <typename T>
282inline QContiguousCacheData *QContiguousCache<T>::allocateData(int aalloc)
283{
284 return QContiguousCacheData::allocateData(sizeOfTypedData() + (aalloc - 1) * sizeof(T), alignOfTypedData());
285}

287template <typename T>
288QContiguousCache<T>::QContiguousCache(int cap)
289{
290 d = allocateData(cap);
291 d->ref.store(1);
292 d->alloc = cap;
293 d->count = d->start = d->offset = 0;
294 d->sharable = true;
295}

297template <typename T>
298QContiguousCache<T> &QContiguousCache<T>::operator=(const QContiguousCache<T> &other)
299{
300 other.d->ref.ref();
301 if (!d->ref.deref())
302 freeData(d);
303 d = other.d;
304 if (!d->sharable)
305 detach_helper();
306 return *this;
307}

309template <typename T>
310bool QContiguousCache<T>::operator==(const QContiguousCache<T> &other) const
311{
312 if (other.d == d)
313 return true;
314 if (other.d->start != d->start
315 || other.d->count != d->count
316 || other.d->offset != d->offset
317 || other.d->alloc != d->alloc)
318 return false;
319 for (int i = firstIndex(); i <= lastIndex(); ++i)
320 if (!(at(i) == other.at(i)))
321 return false;
322 return true;
323}

325template <typename T>
326void QContiguousCache<T>::freeData(Data *x)
327{
328 if (QTypeInfo<T>::isComplex) {
329 int oldcount = d->count;
330 T * i = p->array + d->start;
331 T * e = p->array + d->alloc;
332 while (oldcount--) {
333 i->~T();
334 i++;
335 if (i == e)
336 i = p->array;
337 }
338 }
339 x->freeData(x);
340}
341template <typename T>
342void QContiguousCache<T>::append(const T &value)
343{
344 if (!d->alloc)
345 return; // zero capacity
346 detach();
347 if (QTypeInfo<T>::isComplex) {
348 if (d->count == d->alloc)
349 (p->array + (d->start+d->count) % d->alloc)->~T();
350 new (p->array + (d->start+d->count) % d->alloc) T(value);
351 } else {
352 p->array[(d->start+d->count) % d->alloc] = value;
353 }
354
355 if (d->count == d->alloc) {
356 d->start++;
357 d->start %= d->alloc;
358 d->offset++;
359 } else {
360 d->count++;
361 }
362}
363
364template<typename T>
365void QContiguousCache<T>::prepend(const T &value)
366{
367 if (!d->alloc)
368 return; // zero capacity
369 detach();
370 if (d->start)
371 d->start--;
372 else
373 d->start = d->alloc-1;
374 d->offset--;
375
376 if (d->count != d->alloc)
377 d->count++;
378 else
379 if (d->count == d->alloc)
380 (p->array + d->start)->~T();
381
382 if (QTypeInfo<T>::isComplex)
383 new (p->array + d->start) T(value);
384 else
385 p->array[d->start] = value;
386}

388template<typename T>
389void QContiguousCache<T>::insert(int pos, const T &value)
390{
391 Q_ASSERT_X(pos >= 0 && pos < INT_MAX, "QContiguousCache<T>::insert", "index out of range");
392 if (!d->alloc)
393 return; // zero capacity
394 detach();
395 if (containsIndex(pos)) {
396 if (QTypeInfo<T>::isComplex) {
397 (p->array + pos % d->alloc)->~T();
398 new (p->array + pos % d->alloc) T(value);
399 } else {
400 p->array[pos % d->alloc] = value;
401 }
402 } else if (pos == d->offset-1)
403 prepend(value);
404 else if (pos == d->offset+d->count)
405 append(value);
406 else {
407 // we don't leave gaps.
408 clear();
409 d->offset = pos;
410 d->start = pos % d->alloc;
411 d->count = 1;
412 if (QTypeInfo<T>::isComplex)
413 new (p->array + d->start) T(value);
414 else
415 p->array[d->start] = value;
416 }
417}

419template <typename T>
420inline const T &QContiguousCache<T>::at(int pos) const
421{ Q_ASSERT_X(pos >= d->offset && pos - d->offset < d->count, "QContiguousCache<T>::at", "index out of range"); return p->array[pos % d->alloc]; }
422template <typename T>
423inline const T &QContiguousCache<T>::operator[](int pos) const
424{ Q_ASSERT_X(pos >= d->offset && pos - d->offset < d->count, "QContiguousCache<T>::at", "index out of range"); return p->array[pos % d->alloc]; }
425
426template <typename T>
427inline T &QContiguousCache<T>::operator[](int pos)
428{
429 detach();
430 if (!containsIndex(pos))
431 insert(pos, T());
432 return p->array[pos % d->alloc];
433}

435template <typename T>
436inline void QContiguousCache<T>::removeFirst()
437{
438 Q_ASSERT(d->count > 0);
439 detach();
440 d->count--;
441 if (QTypeInfo<T>::isComplex)
442 (p->array + d->start)->~T();
443 d->start = (d->start + 1) % d->alloc;
444 d->offset++;
445}
446
447template <typename T>
448inline void QContiguousCache<T>::removeLast()
449{
450 Q_ASSERT(d->count > 0);
451 detach();
452 d->count--;
453 if (QTypeInfo<T>::isComplex)
454 (p->array + (d->start + d->count) % d->alloc)->~T();
455}

457template <typename T>
458inline T QContiguousCache<T>::takeFirst()
459{ T t = first(); removeFirst(); return t; }
460
461template <typename T>
462inline T QContiguousCache<T>::takeLast()
463{ T t = last(); removeLast(); return t; }
