48/**
49 * \brief Returns a string value if any.
50 * \param key Key of the entry to get.
51 * \return The value if any and a boolean indicating if it exists.
52 */
53std::pair<std::string, bool> Settings::get_string(const std::string& key) {
54
55 const auto& it = entries.find(key);
56 if (it == entries.end()) {
57 return std::make_pair("", false);
58 }
59
60 return std::make_pair(it->second, true);
61}
62
63/**
64 * \brief Returns an integer value if any.
65 * \param key Key of the entry to get.
66 * \return The value if any and a boolean indicating if it exists.
67 */
68std::pair<int, bool> Settings::get_integer(const std::string& key) {
69
70 const auto& it = entries.find(key);
71 if (it == entries.end()) {
72 return std::make_pair(0, false);
73 }
74
75 int value = 0;
76 const std::string string_value = it->second;
77 std::istringstream iss(string_value);
78 iss >> value;
79 return std::make_pair(value, true);
80}
81
82/**
83 * \brief Returns a boolean value if any.
84 * \param key Key of the entry to get.
85 * \return The value if any and a boolean indicating if it exists.
86 */
87std::pair<bool, bool> Settings::get_boolean(const std::string& key) {
88
89 const auto& it = entries.find(key);
90 if (it == entries.end()) {
91 return std::make_pair(false, false);
92 }
93
94 return std::make_pair(it->second == "true", true);
95}

144/**
145 * \brief Attempts to load the built-in settings from a file.
146 * \param file_name Settings file to read, relative to the quest write directory.
147 * \return \c true if settings were successfully loaded and applied.
148 */
149bool Settings::load(const std::string& file_name) {
150
151 const std::string& quest_write_dir = QuestFiles::get_quest_write_dir();
152 if (quest_write_dir.empty()) {
153 Debug::error("Cannot load settings: no quest write directory was specified in quest.dat");
154 return false;
155 }
156
157 if (!QuestFiles::data_file_exists(file_name)) {
158 Debug::error("Cannot find settings file '" + file_name + "'");
159 return false;
160 }
161
162 // Read the settings as a Lua data file.
163 lua_State* l = luaL_newstate();
164 const std::string& buffer = QuestFiles::data_file_read(file_name);
165 int load_result = luaL_loadbuffer(l, buffer.data(), buffer.size(), file_name.c_str());
166
167 if (load_result != 0 || lua_pcall(l, 0, 0, 0) != 0) {
168 Debug::error(std::string("Cannot read settings file '")
169 + file_name + "': " + lua_tostring(l, -1)
170 );
171 lua_pop(l, 1);
172 lua_close(l);
173 return false;
174 }
175
176 // The syntax of the settings file is valid. Read its values now.
177 entries.clear();
178
179 // Video mode.
180 lua_getglobal(l, "video_mode");
181 if (lua_isstring(l, 1)) {
182 const std::string& mode_name = lua_tostring(l, 1);
183 if (!mode_name.empty()) {
184 entries[key_video_mode] = mode_name;
185 }
186 }
187 lua_pop(l, 1);
188
189 // Fullscreen.
190 lua_getglobal(l, "fullscreen");
191 if (lua_isboolean(l, 1)) {
192 bool fullscreen = lua_toboolean(l, 1);
193 entries[key_fullscreen] = fullscreen ? "true" : "false";
194 }
195 lua_pop(l, 1);
196
197 // Sound volume.
198 lua_getglobal(l, "sound_volume");
199 if (lua_isnumber(l, 1)) {
200 const std::string& sound_volume = lua_tostring(l, 1);
201 entries[key_sound_volume] = sound_volume;
202 }
203 lua_pop(l, 1);
204
205 // Music volume.
206 lua_getglobal(l, "music_volume");
207 if (lua_isnumber(l, 1)) {
208 const std::string& music_volume = lua_tostring(l, 1);
209 entries[key_music_volume] = music_volume;
210 }
211 lua_pop(l, 1);
212
213 // Language.
214 lua_getglobal(l, "language");
215 if (lua_isstring(l, 1)) {
216 const std::string& language = lua_tostring(l, 1);
217 entries[key_language] = language;
218 }
219 lua_pop(l, 1);
220
221 // Joystick.
222 lua_getglobal(l, "joypad_enabled");
223 if (lua_isboolean(l, 1)) {
224 bool joypad_enabled = lua_toboolean(l, 1);
225 entries[key_joypad_enabled] = joypad_enabled ? "true" : "false";
226 }
227 lua_pop(l, 1);
228
229 lua_close(l);
230
231 return true;
232}

234/**
235 * \brief Attempts to save the built-in settings to a file.
236 * \param file_name Settings file to write, relative to the quest write directory.
237 * \return \c true if settings were successfully saved.
238 */
239bool Settings::save(const std::string& file_name) {
240
241 const std::string& quest_write_dir = QuestFiles::get_quest_write_dir();
242 if (quest_write_dir.empty()) {
243 Debug::error("Cannot save settings: no quest write directory was specified in quest.dat");
244 }
245
246 std::ostringstream oss;
247
248 auto it = entries.find(key_video_mode);
249 if (it != entries.end()) {
250 oss << it->first << " = \"" << it->second << "\"\n";
251 }
252
253 it = entries.find(key_fullscreen);
254 if (it != entries.end()) {
255 oss << it->first << " = " << it->second << "\n";
256 }
257
258 it = entries.find(key_sound_volume);
259 if (it != entries.end()) {
260 oss << it->first << " = " << it->second << "\n";
261 }
262
263 it = entries.find(key_music_volume);
264 if (it != entries.end()) {
265 oss << it->first << " = " << it->second << "\n";
266 }
267
268 it = entries.find(key_language);
269 if (it != entries.end()) {
270 oss << it->first << " = \"" << it->second << "\"\n";
271 }
272
273 it = entries.find(key_joypad_enabled);
274 if (it != entries.end()) {
275 oss << it->first << " = " << it->second << "\n";
276 }
277
278 const std::string& text = oss.str();
279 QuestFiles::data_file_save(file_name, text);
280 return true;
281}

283/**
284 * \brief Sets these settings to the ones of the currently running quest.
285 */
286void Settings::set_from_quest() {
287
288 clear();
289
290 if (Video::is_initialized()) {
291 set_string(key_video_mode, Video::get_video_mode().get_name());
292 set_boolean(key_fullscreen, Video::is_fullscreen());
293 }
294 if (Sound::is_initialized()) {
295 set_integer(key_sound_volume, Sound::get_volume());
296 set_integer(key_music_volume, Music::get_volume());
297 }
298 if (InputEvent::is_initialized()) {
299 set_boolean(key_joypad_enabled, InputEvent::is_joypad_enabled());
300 }
301 if (CurrentQuest::is_initialized()) {
302 if (!CurrentQuest::get_language().empty()) {
303 set_string(key_language, CurrentQuest::get_language());
304 }
305 }
306}
307
308/**
309 * \brief Applies these settings to the currently running quest.
310 */
311void Settings::apply_to_quest() {
312
313 if (Video::is_initialized()) {
314 // Video mode.
315 auto video_mode_name = get_string(key_video_mode);
316 if (video_mode_name.second) {
317 const VideoMode* video_mode = Video::get_video_mode_by_name(video_mode_name.first);
318 if (video_mode != nullptr) {
319 Video::set_video_mode(*video_mode);
320 }
321 }
322
323 // Fullscreen.
324 auto fullscreen = get_boolean(key_fullscreen);
325 if (fullscreen.second) {
326 Video::set_fullscreen(fullscreen.first);
327 }
328 }
329
330 if (Sound::is_initialized()) {
331 // Sound volume.
332 auto sound_volume = get_integer(key_sound_volume);
333 if (sound_volume.second) {
334 Sound::set_volume(sound_volume.first);
335 }
336
337 // Music volume.
338 auto music_volume = get_integer(key_music_volume);
339 if (music_volume.second) {
340 Music::set_volume(music_volume.first);
341 }
342}
343
344 if (CurrentQuest::is_initialized()) {
345 // Language.
346 auto language = get_string(key_language);
347 if (language.second) {
348 if (CurrentQuest::has_language(language.first)) {
349 CurrentQuest::set_language(language.first);
350 }
351 }
352 }
353
354 if (InputEvent::is_initialized()) {
355 // Joystick.
356 auto joypad_enabled = get_boolean(key_joypad_enabled);
357 if (joypad_enabled.second) {
358 InputEvent::set_joypad_enabled(joypad_enabled.first);
359 }
360 }
