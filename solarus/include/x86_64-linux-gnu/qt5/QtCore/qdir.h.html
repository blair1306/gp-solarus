53class QDirIterator;

56class Q_CORE_EXPORT QDir
57{
58public:
59 enum Filter { Dirs = 0x001,
60 Files = 0x002,
61 Drives = 0x004,
62 NoSymLinks = 0x008,
63 AllEntries = Dirs | Files | Drives,
64 TypeMask = 0x00f,
65
66 Readable = 0x010,
67 Writable = 0x020,
68 Executable = 0x040,
69 PermissionMask = 0x070,
70
71 Modified = 0x080,
72 Hidden = 0x100,
73 System = 0x200,
74
75 AccessMask = 0x3F0,
76
77 AllDirs = 0x400,
78 CaseSensitive = 0x800,
79 NoDot = 0x2000,
80 NoDotDot = 0x4000,
81 NoDotAndDotDot = NoDot | NoDotDot,
82
83 NoFilter = -1
84 };
85 Q_DECLARE_FLAGS(Filters, Filter)
86
87 enum SortFlag { Name = 0x00,
88 Time = 0x01,
89 Size = 0x02,
90 Unsorted = 0x03,
91 SortByMask = 0x03,
92
93 DirsFirst = 0x04,
94 Reversed = 0x08,
95 IgnoreCase = 0x10,
96 DirsLast = 0x20,
97 LocaleAware = 0x40,
98 Type = 0x80,
99 NoSort = -1
100 };
101 Q_DECLARE_FLAGS(SortFlags, SortFlag)
102
103 QDir(const QDir &);
104 QDir(const QString &path = QString());
105 QDir(const QString &path, const QString &nameFilter,
106 SortFlags sort = SortFlags(Name | IgnoreCase), Filters filter = AllEntries);
107 ~QDir();
108
109 QDir &operator=(const QDir &);
110 QDir &operator=(const QString &path);

114#endif

116 inline void swap(QDir &other)

119 void setPath(const QString &path);

121 QString absolutePath() const;

124 static void addResourceSearchPath(const QString &path);

133 QString relativeFilePath(const QString &fileName) const;

136 static QString fromNativeSeparators(const QString &pathName);

138 bool cd(const QString &dirName);

140
141 QStringList nameFilters() const;

145 void setFilter(Filters filter);

221 }
