17#include "solarus/entities/Wall.h"
18#include "solarus/entities/Hero.h"
19#include "solarus/lowlevel/QuestFiles.h"
20
21namespace Solarus {
22
23/**
24 * \brief Constructor.
25 * \param name name of the wall to create
26 * \param layer layer of the Wall to create
27 * \param xy Coordinates where to create the wall.
28 * \param size Size of the Wall to create.
29 * \param stops_hero true to make this entity an obstacle for the hero
30 * \param stops_enemies true to make this entity an obstacle for enemies
31 * \param stops_npcs true to make this entity an obstacle for NPCs
32 * \param stops_blocks true to make this entity an obstacle for blocks
33 * \param stops_projectiles true to make this entity an obstacle for thrown
34 * items, arrows, hookshots and boomerangs.
35 */
36Wall::Wall(
37 const std::string& name,
38 int layer,
39 const Point& xy,
40 const Size& size,
41 bool stops_hero,
42 bool stops_enemies,
43 bool stops_npcs,
44 bool stops_blocks,
45 bool stops_projectiles
46):
47 Entity(name, 0, layer, xy, size) {
48
49 if (stops_hero) {
50 entity_types_stopped.insert(EntityType::HERO);
51 }
52 if (stops_enemies) {
53 entity_types_stopped.insert(EntityType::ENEMY);
54 }
55 if (stops_npcs) {
56 entity_types_stopped.insert(EntityType::NPC);
57 }
58 if (stops_blocks) {
59 entity_types_stopped.insert(EntityType::BLOCK);
60 }
61 if (stops_projectiles) {
62 entity_types_stopped.insert(EntityType::CARRIED_OBJECT);
63 entity_types_stopped.insert(EntityType::ARROW);
64 entity_types_stopped.insert(EntityType::HOOKSHOT);
65 entity_types_stopped.insert(EntityType::BOOMERANG);
66 }
67}
68
69/**
70 * \brief Returns the type of entity.
71 * \return the type of entity
72 */
73EntityType Wall::get_type() const {
74 return ThisType;
75}
76
77/**
78 * \brief Returns whether entities of this type can be drawn.
79 * \return true if this type of entity can be drawn
80 */
81bool Wall::can_be_drawn() const {
82 return false;
83}
84
85/**
86 * \brief Returns whether this entity is an obstacle for another one
87 * when it is enabled.
88 * \param other another entity
89 * \return true if this entity is an obstacle for the other one
90 */
91bool Wall::is_obstacle_for(Entity& other) {
92
93 const auto it = entity_types_stopped.find(other.get_type());
94 return it != entity_types_stopped.end();
95}
