17#ifndef SOLARUS_STAIRS_H
18#define SOLARUS_STAIRS_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <string>
23
24namespace Solarus {
25
26/**
27 * \brief Some stairs that trigger a special animation on the hero.
28 *
29 * A stairs entity is a special terrain where the hero walks
30 * either before going to another floor
31 * or to change its layer inside a single floor.
32 *
33 * For stairs inside a single floor,
34 * only one stairs entity has to be placed and the collisions will be
35 * detected on both layers.
36 * This entity has to be placed on the lowest of both layers.
37 *
38 * For stairs that send the hero to another floor (i.e. another map),
39 * there must be three entities:
40 * - a teletransporter (that make the hero go to the next map),
41 * - a stairs entity (that make the hero take the stairs animation),
42 * - a destination point on the next map.
43 */
44class Stairs: public Entity {
45
46 public:
47
48 static constexpr EntityType ThisType = EntityType::STAIRS;
49
50 /**
51 * \brief Subtypes of stairs.
52 */
53 enum Subtype {
54 SPIRAL_UPSTAIRS, /**< going to the floor upstairs (spiral staircase) */
55 SPIRAL_DOWNSTAIRS, /**< going to the floor downstairs (spiral staircase) */
56 STRAIGHT_UPSTAIRS, /**< going to the floor upstairs (straight staircase) */
57 STRAIGHT_DOWNSTAIRS, /**< going to the floor downstairs (straight staircase) */
58 INSIDE_FLOOR /**< small stairs inside a single floor, changing the hero's layer */
59 };
60
61 /**
62 * \brief Indicates the way an entity is walking on stairs.
63 * - For stairs inside a single floor, 0 means that the hero is going upstairs
64 * and 1 means that he is going downstairs.
65 * - For stairs between two different floors, 0 means that the hero is on the stairs
66 * of the old map and 1 means that he is on the stairs on the new map.
67 */
68 enum Way {
69 NORMAL_WAY, /**< for stairs inside a single floor, indicates that the entity is going upstairs;
70 * for stairs between two floors, indicates that the entity is leaving this floor */
71 REVERSE_WAY /**< for stairs inside a single floor, indicates that the entity is going downstairs;
72 * for stairs between two floors, indicates that the entity is arriving on this floor */
73 };
74
75 Stairs(
76 const std::string& name,
77 int layer,
78 const Point& xy,
79 int direction,
80 Subtype subtype
81 );
82
83 virtual EntityType get_type() const override;
84 virtual bool can_be_drawn() const override;
85 bool can_change_ground() const;
86 Ground get_ground() const;
87
88 virtual void notify_creating() override;
89 bool is_inside_floor() const;
90 virtual bool is_sword_ignored() const override;
91 virtual bool is_obstacle_for(Entity& other) override;
92 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
93 virtual void notify_enabled(bool enabled) override;
94
95 int get_movement_direction(Way way) const;
96 int get_animation_direction(Way way) const;
97 void play_sound(Way way) const;
98 std::string get_path(Way way) const;
99 Rectangle get_clipping_rectangle(Way way) const;
100
101 private:
102
103 void update_dynamic_tiles();
104
105 Subtype subtype; /**< subtype of stairs */
106
107};
108
109}
110
111#endif
