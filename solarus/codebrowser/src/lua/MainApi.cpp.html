30/**
31 * Name of the Lua table representing the main module of Solarus.
32 */
33const std::string LuaContext::main_module_name = "sol.main";
34
35/**
36 * \brief Initializes the main features provided to Lua.
37 */
38void LuaContext::register_main_module() {
39
40 static const luaL_Reg functions[] = {
41 { "get_solarus_version", main_api_get_solarus_version },
42 { "get_quest_version", main_api_get_quest_version },
43 { "get_quest_format", main_api_get_quest_format },
44 { "load_file", main_api_load_file },
45 { "do_file", main_api_do_file },
46 { "reset", main_api_reset },
47 { "exit", main_api_exit },
48 { "get_elapsed_time", main_api_get_elapsed_time },
49 { "get_quest_write_dir", main_api_get_quest_write_dir },
50 { "set_quest_write_dir", main_api_set_quest_write_dir },
51 { "load_settings", main_api_load_settings },
52 { "save_settings", main_api_save_settings },
53 { "get_distance", main_api_get_distance },
54 { "get_angle", main_api_get_angle },
55 { "get_type", main_api_get_type },
56 { "get_metatable", main_api_get_metatable },
57 { "get_os", main_api_get_os },
58 { nullptr, nullptr }
59 };
60
61 register_functions(main_module_name, functions);
62
63 // Store sol.main in the registry to access it safely
64 // from C++ (and also slightly faster).
65 // After that, the engine will never rely on the existence of a global
66 // value called "sol". The user can therefore do whatever he wants, including
67 // renaming the sol global table to something else in the unlikely case where
68 // another Lua library called "sol" is required, or if he simply does not
69 // like the name "sol".
70
71 // ...
72 lua_getglobal(l, "sol");
73 // ... sol
74 lua_getfield(l, -1, "main");
75 // ... sol main
76 lua_setfield(l, LUA_REGISTRYINDEX, main_module_name.c_str());
77 // ... sol
78 lua_pop(l, 1);
79 // ...
80}

82/**
83 * \brief Pushes the sol.main table onto the stack.
84 * \param l A Lua state.
85 */
86void LuaContext::push_main(lua_State* l) {
87
88 lua_getfield(l, LUA_REGISTRYINDEX, main_module_name.c_str());
89}

91/**
92 * \brief Implementation of sol.main.get_solarus_version().
93 * \param l The Lua context that is calling this function.
94 * \return Number of values to return to Lua.
95 */
96int LuaContext::main_api_get_solarus_version(lua_State* l) {
97
98 return LuaTools::exception_boundary_handle(l, [&] {
99 const std::string& solarus_version = SOLARUS_VERSION;
100
101 push_string(l, solarus_version);
102 return 1;
103 });
104}

254/**
255 * \brief Implementation of sol.main.load_settings().
256 * \param l the Lua context that is calling this function
257 * \return number of values to return to Lua
258 */
259int LuaContext::main_api_load_settings(lua_State* l) {
260
261 return LuaTools::exception_boundary_handle(l, [&] {
262 std::string file_name = LuaTools::opt_string(l, 1, "settings.dat");
263
264 if (QuestFiles::get_quest_write_dir().empty()) {
265 LuaTools::error(l, "Cannot load settings: no write directory was specified in quest.dat");
266 }
267
268 bool success = false;
269 if (QuestFiles::data_file_exists(file_name)) {
270 Settings settings;
271 success = settings.load(file_name);
272 if (success) {
273 settings.apply_to_quest();
274 }
275 }
276
277 lua_pushboolean(l, success);
278 return 1;
279 });
280}

305/**
306 * \brief Implementation of sol.main.get_distance().
307 * \param l the Lua context that is calling this function
308 * \return number of values to return to Lua
309 */
310int LuaContext::main_api_get_distance(lua_State* l) {
311
312 return LuaTools::exception_boundary_handle(l, [&] {
313 int x1 = LuaTools::check_int(l, 1);
314 int y1 = LuaTools::check_int(l, 2);
315 int x2 = LuaTools::check_int(l, 3);
316 int y2 = LuaTools::check_int(l, 4);
317
318 int distance = (int) Geometry::get_distance(x1, y1, x2, y2);
319
320 lua_pushinteger(l, distance);
321 return 1;
322 });
323}

345/**
346 * \brief Implementation of sol.main.get_type().
347 * \param l The Lua context that is calling this function.
348 * \return Number of values to return to Lua.
349 */
350int LuaContext::main_api_get_type(lua_State* l) {
351
352 return LuaTools::exception_boundary_handle(l, [&] {
353
354 luaL_checkany(l, 1);
355 std::string module_name;
356 if (!is_solarus_userdata(l, 1, module_name)) {
357 // Return the same thing as the usual Lua type() function.
358 std::string type_name = luaL_typename(l, 1);
359 push_string(l, type_name);
360 }
361 else {
362 // Remove the "sol." prefix.
363 push_string(l, module_name.substr(4));
364 }
365
366 return 1;
367 });
368}

398/**
399 * \brief Calls sol.main.on_started() if it exists.
400 *
401 * This function is called when the engine requests Lua to show an
402 * initial screen, i.e. at the beginning of the program
403 * or when the program is reset.
404 */
405void LuaContext::main_on_started() {
406
407 push_main(l);
408 on_started();
409 lua_pop(l, 1);
410}
411
412/**
413 * \brief Calls sol.main.on_finished() if it exists.
414 *
415 * This function is called when the program is reset or stopped.
416 */
417void LuaContext::main_on_finished() {
418
419 push_main(l);
420 on_finished();
421 remove_timers(-1); // Stop timers associated to sol.main.
422 remove_menus(-1); // Stop menus associated to sol.main.
423 lua_pop(l, 1);
424}

451/**
452 * \brief Notifies Lua that an input event has just occurred.
453 *
454 * The appropriate callback in sol.main is triggered if it exists.
455 *
456 * \param event The input event to handle.
457 * \return \c true if the event was handled and should stop being propagated.
458 */
459bool LuaContext::main_on_input(const InputEvent& event) {
460
461 push_main(l);
462 bool handled = on_input(event);
463 if (!handled) {
464 handled = menus_on_input(-1, event);
465 }
466 lua_pop(l, 1);
467 return handled;
468}
