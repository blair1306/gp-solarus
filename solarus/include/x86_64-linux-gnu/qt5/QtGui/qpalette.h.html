47#include <QtGui/qbrush.h>

49QT_BEGIN_NAMESPACE

52class QPalettePrivate;
53class QVariant;
54
55class Q_GUI_EXPORT QPalette
56{
57 Q_GADGET
58 Q_ENUMS(ColorGroup ColorRole)
59public:
60 QPalette();
61 QPalette(const QColor &button);
62 QPalette(Qt::GlobalColor button);
63 QPalette(const QColor &button, const QColor &window);
64 QPalette(const QBrush &windowText, const QBrush &button, const QBrush &light,
65 const QBrush &dark, const QBrush &mid, const QBrush &text,
66 const QBrush &bright_text, const QBrush &base, const QBrush &window);
67 QPalette(const QColor &windowText, const QColor &window, const QColor &light,
68 const QColor &dark, const QColor &mid, const QColor &text, const QColor &base);
69 QPalette(const QPalette &palette);
70 ~QPalette();
71 QPalette &operator=(const QPalette &palette);
72#ifdef Q_COMPILER_RVALUE_REFS
73 inline QPalette &operator=(QPalette &&other)
74 {
75 for_faster_swapping_dont_use = other.for_faster_swapping_dont_use;
76 qSwap(d, other.d); return *this;
77 }
78#endif
79
80 void swap(QPalette &other) {
81 qSwap(d, other.d);
82 qSwap(for_faster_swapping_dont_use, other.for_faster_swapping_dont_use);
83 }
84
85 operator QVariant() const;
86
87 // Do not change the order, the serialization format depends on it
88 enum ColorGroup { Active, Disabled, Inactive, NColorGroups, Current, All, Normal = Active };
89 enum ColorRole { WindowText, Button, Light, Midlight, Dark, Mid,
90 Text, BrightText, ButtonText, Base, Window, Shadow,
91 Highlight, HighlightedText,
92 Link, LinkVisited,
93 AlternateBase,
94 NoRole,
95 ToolTipBase, ToolTipText,
96 NColorRoles = ToolTipText + 1,
97 Foreground = WindowText, Background = Window
98 };
99
100 inline ColorGroup currentColorGroup() const { return static_cast<ColorGroup>(data.current_group); }
101 inline void setCurrentColorGroup(ColorGroup cg) { data.current_group = cg; }
102
103 inline const QColor &color(ColorGroup cg, ColorRole cr) const
104 { return brush(cg, cr).color(); }
105 const QBrush &brush(ColorGroup cg, ColorRole cr) const;
106 inline void setColor(ColorGroup cg, ColorRole cr, const QColor &color);
107 inline void setColor(ColorRole cr, const QColor &color);
108 inline void setBrush(ColorRole cr, const QBrush &brush);
109 bool isBrushSet(ColorGroup cg, ColorRole cr) const;
110 void setBrush(ColorGroup cg, ColorRole cr, const QBrush &brush);
111 void setColorGroup(ColorGroup cr, const QBrush &windowText, const QBrush &button,
112 const QBrush &light, const QBrush &dark, const QBrush &mid,
113 const QBrush &text, const QBrush &bright_text, const QBrush &base,
114 const QBrush &window);
115 bool isEqual(ColorGroup cr1, ColorGroup cr2) const;
116
117 inline const QColor &color(ColorRole cr) const { return color(Current, cr); }
118 inline const QBrush &brush(ColorRole cr) const { return brush(Current, cr); }
119 inline const QBrush &foreground() const { return brush(WindowText); }
120 inline const QBrush &windowText() const { return brush(WindowText); }
121 inline const QBrush &button() const { return brush(Button); }
122 inline const QBrush &light() const { return brush(Light); }
123 inline const QBrush &dark() const { return brush(Dark); }
124 inline const QBrush &mid() const { return brush(Mid); }
125 inline const QBrush &text() const { return brush(Text); }
126 inline const QBrush &base() const { return brush(Base); }
127 inline const QBrush &alternateBase() const { return brush(AlternateBase); }
128 inline const QBrush &toolTipBase() const { return brush(ToolTipBase); }
129 inline const QBrush &toolTipText() const { return brush(ToolTipText); }
130 inline const QBrush &background() const { return brush(Window); }
131 inline const QBrush &window() const { return brush(Window); }
132 inline const QBrush &midlight() const { return brush(Midlight); }
133 inline const QBrush &brightText() const { return brush(BrightText); }
134 inline const QBrush &buttonText() const { return brush(ButtonText); }
135 inline const QBrush &shadow() const { return brush(Shadow); }
136 inline const QBrush &highlight() const { return brush(Highlight); }
137 inline const QBrush &highlightedText() const { return brush(HighlightedText); }
138 inline const QBrush &link() const { return brush(Link); }
139 inline const QBrush &linkVisited() const { return brush(LinkVisited); }
140
141 bool operator==(const QPalette &p) const;
142 inline bool operator!=(const QPalette &p) const { return !(operator==(p)); }
143 bool isCopyOf(const QPalette &p) const;
144
145#if QT_DEPRECATED_SINCE(5, 0)
146 QT_DEPRECATED inline int serialNumber() const { return cacheKey() >> 32; }
147#endif
148 qint64 cacheKey() const;
149
150 QPalette resolve(const QPalette &) const;
151 inline uint resolve() const { return data.resolve_mask; }
152 inline void resolve(uint mask) { data.resolve_mask = mask; }
153
154private:
155 void setColorGroup(ColorGroup cr, const QBrush &windowText, const QBrush &button,
156 const QBrush &light, const QBrush &dark, const QBrush &mid,
157 const QBrush &text, const QBrush &bright_text,
158 const QBrush &base, const QBrush &alternate_base,
159 const QBrush &window, const QBrush &midlight,
160 const QBrush &button_text, const QBrush &shadow,
161 const QBrush &highlight, const QBrush &highlighted_text,
162 const QBrush &link, const QBrush &link_visited);
163 void setColorGroup(ColorGroup cr, const QBrush &windowText, const QBrush &button,
164 const QBrush &light, const QBrush &dark, const QBrush &mid,
165 const QBrush &text, const QBrush &bright_text,
166 const QBrush &base, const QBrush &alternate_base,
167 const QBrush &window, const QBrush &midlight,
168 const QBrush &button_text, const QBrush &shadow,
169 const QBrush &highlight, const QBrush &highlighted_text,
170 const QBrush &link, const QBrush &link_visited,
171 const QBrush &toolTipBase, const QBrush &toolTipText);
172 void init();
173 void detach();
174
175 QPalettePrivate *d;
176 struct Data {
177 uint current_group : 4;
178 uint resolve_mask : 28;
179 };
180 union {
181 Data data;
182 quint32 for_faster_swapping_dont_use;
183 };
184 friend Q_GUI_EXPORT QDataStream &operator<<(QDataStream &s, const QPalette &p);
185};

187Q_DECLARE_SHARED(QPalette)

189inline void QPalette::setColor(ColorGroup acg, ColorRole acr,
190 const QColor &acolor)
191{ setBrush(acg, acr, QBrush(acolor)); }
192inline void QPalette::setColor(ColorRole acr, const QColor &acolor)
193{ setColor(All, acr, acolor); }
194inline void QPalette::setBrush(ColorRole acr, const QBrush &abrush)
195{ setBrush(All, acr, abrush); }
