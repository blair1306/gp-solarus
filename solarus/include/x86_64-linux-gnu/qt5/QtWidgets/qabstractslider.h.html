42#ifndef QABSTRACTSLIDER_H
43#define QABSTRACTSLIDER_H
44
45#include <QtWidgets/qwidget.h>
46
47QT_BEGIN_NAMESPACE
48
49
50class QAbstractSliderPrivate;
51
52class Q_WIDGETS_EXPORT QAbstractSlider : public QWidget
53{
54 Q_OBJECT
55
56 Q_PROPERTY(int minimum READ minimum WRITE setMinimum)
57 Q_PROPERTY(int maximum READ maximum WRITE setMaximum)
58 Q_PROPERTY(int singleStep READ singleStep WRITE setSingleStep)
59 Q_PROPERTY(int pageStep READ pageStep WRITE setPageStep)
60 Q_PROPERTY(int value READ value WRITE setValue NOTIFY valueChanged USER true)
61 Q_PROPERTY(int sliderPosition READ sliderPosition WRITE setSliderPosition NOTIFY sliderMoved)
62 Q_PROPERTY(bool tracking READ hasTracking WRITE setTracking)
63 Q_PROPERTY(Qt::Orientation orientation READ orientation WRITE setOrientation)
64 Q_PROPERTY(bool invertedAppearance READ invertedAppearance WRITE setInvertedAppearance)
65 Q_PROPERTY(bool invertedControls READ invertedControls WRITE setInvertedControls)
66 Q_PROPERTY(bool sliderDown READ isSliderDown WRITE setSliderDown DESIGNABLE false)
67
68public:
69 explicit QAbstractSlider(QWidget *parent=0);
70 ~QAbstractSlider();
71
72 Qt::Orientation orientation() const;
73
74 void setMinimum(int);
75 int minimum() const;
76
77 void setMaximum(int);
78 int maximum() const;
79
80 void setSingleStep(int);
81 int singleStep() const;
82
83 void setPageStep(int);
84 int pageStep() const;
85
86 void setTracking(bool enable);
87 bool hasTracking() const;
88
89 void setSliderDown(bool);
90 bool isSliderDown() const;
91
92 void setSliderPosition(int);
93 int sliderPosition() const;
94
95 void setInvertedAppearance(bool);
96 bool invertedAppearance() const;
97
98 void setInvertedControls(bool);
99 bool invertedControls() const;
100
101 enum SliderAction {
102 SliderNoAction,
103 SliderSingleStepAdd,
104 SliderSingleStepSub,
105 SliderPageStepAdd,
106 SliderPageStepSub,
107 SliderToMinimum,
108 SliderToMaximum,
109 SliderMove
110 };
111
112 int value() const;
113
114 void triggerAction(SliderAction action);
115
116public Q_SLOTS:
117 void setValue(int);
118 void setOrientation(Qt::Orientation);
119 void setRange(int min, int max);
120
121Q_SIGNALS:
122 void valueChanged(int value);
123
124 void sliderPressed();
125 void sliderMoved(int position);
126 void sliderReleased();
127
128 void rangeChanged(int min, int max);
129
130 void actionTriggered(int action);
131
132protected:
133 bool event(QEvent *e);
134
135 void setRepeatAction(SliderAction action, int thresholdTime = 500, int repeatTime = 50);
136 SliderAction repeatAction() const;
137
138 enum SliderChange {
139 SliderRangeChange,
140 SliderOrientationChange,
141 SliderStepsChange,
142 SliderValueChange
143 };
144 virtual void sliderChange(SliderChange change);
145
146 void keyPressEvent(QKeyEvent *ev);
147 void timerEvent(QTimerEvent *);
148#ifndef QT_NO_WHEELEVENT
149 void wheelEvent(QWheelEvent *e);
150#endif
151 void changeEvent(QEvent *e);
152
153
154protected:
155 QAbstractSlider(QAbstractSliderPrivate &dd, QWidget *parent=0);
156
157private:
158 Q_DISABLE_COPY(QAbstractSlider)
159 Q_DECLARE_PRIVATE(QAbstractSlider)
160};
161
162QT_END_NAMESPACE
163
164#endif // QABSTRACTSLIDER_H
