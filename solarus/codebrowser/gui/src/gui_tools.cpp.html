17#include "solarus/gui/gui_tools.h"
18#include <QMessageBox>

20namespace SolarusGui {
21
22namespace GuiTools {
23
24/**
25 * @brief Shows a modal dialog box with an information message.
26 * @param message The message to show.
27 */
28void SOLARUS_GUI_API information_dialog(const QString& message) {
29
30 QMessageBox messageBox;
31 messageBox.setIcon(QMessageBox::Information);
32 messageBox.setText(message);
33 messageBox.setWindowTitle(QMessageBox::tr("Information"));
34 messageBox.exec();
35}
36
