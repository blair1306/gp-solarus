130Q_DECLARE_TYPEINFO(QVector2D, Q_MOVABLE_TYPE);

132Q_DECL_CONSTEXPR inline QVector2D::QVector2D() : xp(0.0f), yp(0.0f) {}
133
134Q_DECL_CONSTEXPR inline QVector2D::QVector2D(float xpos, float ypos) : xp(xpos), yp(ypos) {}
135
136Q_DECL_CONSTEXPR inline QVector2D::QVector2D(const QPoint& point) : xp(point.x()), yp(point.y()) {}
137
138Q_DECL_CONSTEXPR inline QVector2D::QVector2D(const QPointF& point) : xp(point.x()), yp(point.y()) {}
139
140inline bool QVector2D::isNull() const
141{
142 return qIsNull(xp) && qIsNull(yp);
143}
144
145Q_DECL_CONSTEXPR inline float QVector2D::x() const { return xp; }
146Q_DECL_CONSTEXPR inline float QVector2D::y() const { return yp; }
147
148inline void QVector2D::setX(float aX) { xp = aX; }
149inline void QVector2D::setY(float aY) { yp = aY; }
150
151inline float &QVector2D::operator[](int i)
152{
153 Q_ASSERT(uint(i) < 2u);
154 return *(&xp + i);
155}
156
157inline float QVector2D::operator[](int i) const
158{
159 Q_ASSERT(uint(i) < 2u);
160 return *(&xp + i);
161}
162
163inline QVector2D &QVector2D::operator+=(const QVector2D &vector)
164{
165 xp += vector.xp;
166 yp += vector.yp;
167 return *this;
168}
169
170inline QVector2D &QVector2D::operator-=(const QVector2D &vector)
171{
172 xp -= vector.xp;
173 yp -= vector.yp;
174 return *this;
175}
176
177inline QVector2D &QVector2D::operator*=(float factor)
178{
179 xp *= factor;
180 yp *= factor;
181 return *this;
182}
183
184inline QVector2D &QVector2D::operator*=(const QVector2D &vector)
185{
186 xp *= vector.xp;
187 yp *= vector.yp;
188 return *this;
189}
190
191inline QVector2D &QVector2D::operator/=(float divisor)
192{
193 xp /= divisor;
194 yp /= divisor;
195 return *this;
196}
197
198Q_DECL_CONSTEXPR inline bool operator==(const QVector2D &v1, const QVector2D &v2)
199{
200 return v1.xp == v2.xp && v1.yp == v2.yp;
201}
202
203Q_DECL_CONSTEXPR inline bool operator!=(const QVector2D &v1, const QVector2D &v2)
204{
205 return v1.xp != v2.xp || v1.yp != v2.yp;
206}
207
208Q_DECL_CONSTEXPR inline const QVector2D operator+(const QVector2D &v1, const QVector2D &v2)
209{
210 return QVector2D(v1.xp + v2.xp, v1.yp + v2.yp);
211}
212
213Q_DECL_CONSTEXPR inline const QVector2D operator-(const QVector2D &v1, const QVector2D &v2)
214{
215 return QVector2D(v1.xp - v2.xp, v1.yp - v2.yp);
216}
217
218Q_DECL_CONSTEXPR inline const QVector2D operator*(float factor, const QVector2D &vector)
219{
220 return QVector2D(vector.xp * factor, vector.yp * factor);
221}
222
223Q_DECL_CONSTEXPR inline const QVector2D operator*(const QVector2D &vector, float factor)
224{
225 return QVector2D(vector.xp * factor, vector.yp * factor);
226}
227
228Q_DECL_CONSTEXPR inline const QVector2D operator*(const QVector2D &v1, const QVector2D &v2)
229{
230 return QVector2D(v1.xp * v2.xp, v1.yp * v2.yp);
231}
232
233Q_DECL_CONSTEXPR inline const QVector2D operator-(const QVector2D &vector)
234{
235 return QVector2D(-vector.xp, -vector.yp);
236}
237
238Q_DECL_CONSTEXPR inline const QVector2D operator/(const QVector2D &vector, float divisor)
239{
240 return QVector2D(vector.xp / divisor, vector.yp / divisor);
241}
242
243Q_DECL_CONSTEXPR inline bool qFuzzyCompare(const QVector2D& v1, const QVector2D& v2)
244{
245 return qFuzzyCompare(v1.xp, v2.xp) && qFuzzyCompare(v1.yp, v2.yp);
246}
247
248Q_DECL_CONSTEXPR inline QPoint QVector2D::toPoint() const
249{
250 return QPoint(qRound(xp), qRound(yp));
251}
252
253Q_DECL_CONSTEXPR inline QPointF QVector2D::toPointF() const
254{
255 return QPointF(qreal(xp), qreal(yp));
256}
