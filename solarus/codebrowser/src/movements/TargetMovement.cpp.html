36/**
37 * \brief Creates a new target movement toward an entity or a fixed point.
38 * \param target_entity The entity to target or nullptr.
39 * \param x X of the target point, or X offset in the case of an entity.
40 * \param y Y of the target point, or Y offset in the case of an entity.
41 * \param moving_speed Speed of the movement when not stopped.
42 * \param ignore_obstacles \c true to ignore obstacles (if on a map).
43 */
44TargetMovement::TargetMovement(
45 const EntityPtr& target_entity,
46 int x,
47 int y,
48 int moving_speed,
49 bool ignore_obstacles):
50 TargetMovement(target_entity, Point(x,y), moving_speed, ignore_obstacles) {
51}
52/**
53 * \brief Creates a new target movement toward an entity or a fixed point.
54 * \param target_entity The entity to target or nullptr.
55 * \param target the target point, or offset in the case of an entity.
56 * \param moving_speed Speed of the movement when not stopped.
57 * \param ignore_obstacles \c true to ignore obstacles (if on a map).
58 */
59TargetMovement::TargetMovement(
60 const EntityPtr& target_entity,
61 const Point& target,
62 int moving_speed,
63 bool ignore_obstacles):
64
65 StraightMovement(ignore_obstacles, true),
66 target(target),
67 target_entity(target_entity),
68 entity_offset(target),
69 sign_x(0),
70 sign_y(0),
71 moving_speed(moving_speed),
72 next_recomputation_date(System::now()),
73 finished(false) {
74}

76/**
77 * \brief Notifies this movement that the object it controls has changed.
78 */
79void TargetMovement::notify_object_controlled() {
80
81 StraightMovement::notify_object_controlled();
82
83 // Coordinates have changed: compute a new trajectory.
84 recompute_movement();
85}

87/**
88 * \brief Changes the target of this movement.
89 * \param target_entity The entity to target or nullptr.
90 * \param xy The target point, or offset in the case of an entity.
91 */
92void TargetMovement::set_target(
93 const EntityPtr& target_entity,
94 const Point& xy
95) {
96 this->target_entity = target_entity;
97
98 if (this->target_entity != nullptr) {
99 this->entity_offset = xy;
100 }
101 else {
102 this->target = xy;
103 }
104
105 recompute_movement();
106 next_recomputation_date = System::now() + recomputation_delay;
107}

117/**

121void TargetMovement::set_moving_speed(int moving_speed) {
122 this->moving_speed = moving_speed;
123 recompute_movement();

126/**
127 * \brief Updates the movement.
128 */
129void TargetMovement::update() {
130
131 if (target_entity != nullptr && target_entity->is_being_removed()) {
132 set_target(nullptr, target);
133 }
134
135 if (System::now() >= next_recomputation_date) {
136 recompute_movement();
137 next_recomputation_date += recomputation_delay;
138 }
139
140 // see if the target is reached
141 Point dxy = target - get_xy();
142 if (dxy.x * sign_x <= 0 && dxy.y * sign_y <= 0) {
143 if (!test_collision_with_obstacles(dxy)) {
144 set_xy(target); // Because the target movement may have not been very precise.
145 stop();
146 finished = true;
147 }
148 }
149
150 StraightMovement::update();
151}
152
153/**
154 * \brief Calculates the direction and the speed of the movement
155 * depending on the target.
156 */
157void TargetMovement::recompute_movement() {
158
159 if (target_entity != nullptr) {
160 // the target may be a moving entity
161 target = target_entity->get_xy() + entity_offset;
162 }
163
164 if (get_xy() != target) {
165 finished = false;
166
167 double angle = Geometry::get_angle(get_xy(), target);
168
169 Point dxy = target - get_xy();
170 sign_x = (dxy.x >= 0) ? 1 : -1;
171 sign_y = (dxy.y >= 0) ? 1 : -1;
172
173 if (std::abs(angle - get_angle()) > 1E-6 || get_speed() < 1E-6) {
174 // The angle has changed or the movement was stopped.
175 set_speed(moving_speed);
176 set_angle(angle);
177 set_max_distance((int) Geometry::get_distance(get_xy(), target));
178 }
179 }
180}

182/**
183 * \brief Returns whether the movement is finished, i.e.
184 * whether the target was reached.
185 */
186bool TargetMovement::is_finished() const {
187 return finished;
188}
