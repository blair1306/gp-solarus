53/**
54 * \brief Adds a new tile pattern to this tileset.
55 *
56 * This function is called while loading the tileset.
57 *
58 * \param id Id of this tile pattern.
59 * \param pattern_data Properties of the tile pattern to create.
60 */
61void Tileset::add_tile_pattern(
62 const std::string& id,
63 const TilePatternData& pattern_data
64) {
65 // Create the tile pattern.
66 TilePattern* tile_pattern = nullptr;
67
68 const std::vector<Rectangle>& frames = pattern_data.get_frames();
69 const TileScrolling scrolling = pattern_data.get_scrolling();
70 const Ground ground = pattern_data.get_ground();
71
72 if (frames.size() == 1) {
73 // Single frame.
74 const Rectangle& frame = frames[0];
75 const Size& size = frame.get_size();
76
77 // Diagonal obstacle: check that the tile is square.
78 if (GroundInfo::is_ground_diagonal(ground)) {
79 if (!size.is_square()) {
80 Debug::error("Invalid tile pattern '" + id + ": a tile pattern with a diagonal wall must be square");
81 }
82 }
83
84 switch (scrolling) {
85
86 case TileScrolling::NONE:
87 tile_pattern = new SimpleTilePattern(
88 ground, frame.get_xy(), size
89 );
90 break;
91
92 case TileScrolling::PARALLAX:
93 tile_pattern = new ParallaxScrollingTilePattern(
94 ground, frame.get_xy(), size
95 );
96 break;
97
98 case TileScrolling::SELF:
99 tile_pattern = new SelfScrollingTilePattern(
100 ground, frame.get_xy(), size
101 );
102 break;
103 }
104 }
105 else {
106 // Multi-frame.
107 if (scrolling == TileScrolling::SELF) {
108 Debug::error("Multi-frame is not supported for self-scrolling tiles");
109 return;
110 }
111
112 bool parallax = scrolling == TileScrolling::PARALLAX;
113 AnimatedTilePattern::AnimationSequence sequence = (frames.size() == 3) ?
114 AnimatedTilePattern::ANIMATION_SEQUENCE_012 : AnimatedTilePattern::ANIMATION_SEQUENCE_0121;
115 tile_pattern = new AnimatedTilePattern(
116 ground,
117 sequence,
118 frames[0].get_size(),
119 frames[0].get_x(),
120 frames[0].get_y(),
121 frames[1].get_x(),
122 frames[1].get_y(),
123 frames[2].get_x(),
124 frames[2].get_y(),
125 parallax
126 );
127 }
128
129 tile_patterns.emplace(id, std::unique_ptr<TilePattern>(tile_pattern));
130}

132/**
133 * \brief Loads the tileset from its file by creating all tile patterns.
134 */
135void Tileset::load() {
136
137 // Load the tileset data file.
138 std::string file_name = std::string("tilesets/") + id + ".dat";
139 TilesetData data;
140 bool success = data.import_from_quest_file(file_name);
141 if (success) {
142 // Get the imported data.
143 this->background_color = data.get_background_color();
144 for (const auto& kvp : data.get_patterns()) {
145 add_tile_pattern(kvp.first, kvp.second);
146 }
147 }
148
149 // Load the tileset images.
150 file_name = std::string("tilesets/") + id + ".tiles.png";
151 tiles_image = Surface::create(file_name, Surface::DIR_DATA);
152 if (tiles_image == nullptr) {
153 Debug::error(std::string("Missing tiles image for tileset '") + id + "': " + file_name);
154 tiles_image = Surface::create(16, 16);
155 }
156
157 file_name = std::string("tilesets/") + id + ".entities.png";
158 entities_image = Surface::create(file_name, Surface::DIR_DATA);
159}

161/**
162 * \brief Destroys the tile patterns and frees the memory used
163 * by the tileset image.
164 */
165void Tileset::unload() {
166
167 tile_patterns.clear();
168 tiles_image = nullptr;
169 entities_image = nullptr;
170}

196/**
197 * \brief Returns the image containing the tileset-dependent sprites for this tileset.

203}
204
205/**
206 * \brief Returns a tile pattern from this tileset.
207 * \param id Id of the tile pattern to get.
208 * \return The tile pattern with this id.
209 */
210const TilePattern& Tileset::get_tile_pattern(const std::string& id) const {
211
212 const auto& it = tile_patterns.find(id);
213 if (it == tile_patterns.end()) {
214 std::ostringstream oss;
215 oss << "No such tile pattern in tileset '" << get_id() << "': " << id;
216 Debug::die(oss.str());
217 }
218 return *it->second;
219}
220
221/**
222 * \brief Changes the tiles images, the entities images and the background
223 * color of this tileset.
224 * \param other_id Id of another tileset whose images and background color
225 * will be copied into this tileset.
226 */
227void Tileset::set_images(const std::string& other_id) {
228
229 // Load the new tileset to take its images.
230 Tileset tmp_tileset(other_id);
231 tmp_tileset.load();
232
233 tiles_image = tmp_tileset.get_tiles_image();
234 entities_image = tmp_tileset.get_entities_image();
235
236 background_color = tmp_tileset.get_background_color();
237}
