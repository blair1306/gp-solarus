17#ifndef SOLARUS_RELATIVE_MOVEMENT_H
18#define SOLARUS_RELATIVE_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityPtr.h"
22#include "solarus/movements/Movement.h"
23
24namespace Solarus {
25
26/**
27 * \brief Movement of an entity attached to another entity.
28 *
29 * At each moment, the entity is placed at a position
30 * relative to the entity followed.
31 * If the movement is sensible to obstacles, it becomes finished as
32 * soon as the entity cannot be placed at the requested position.
33 */
34class RelativeMovement: public Movement {
35
36 public:
37
38 RelativeMovement(
39 const EntityPtr& entity_followed,
40 int x,
41 int y,
42 bool ignore_obstacles
43 );
44
45 RelativeMovement(
46 const EntityPtr& entity_followed,
47 const Point& entity_offset,
48 bool ignore_obstacles
49 );
50
51 virtual bool is_finished() const override;
52 virtual Point get_displayed_xy() const override;
53
54 protected:
55
56 virtual void update() override;
57
58 private:
59
60 EntityPtr entity_followed; /**< The entity followed by this movement or nullptr. */
61 const Point entity_offset; /**< coordinate of where this entity should be placed,
62 * relative to the entity followed */
63
64 bool finished; /**< indicates that the movement is stopped because of obstacles */
65
66};
67
68}
69
70#endif
