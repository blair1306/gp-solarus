33/**
34 * \brief Constructor.
35 */
36EnemyReaction::EnemyReaction() {
37
38 set_default_reaction();
39}
40
41/**
42 * \brief Resets this reaction to its default value.
43 */
44void EnemyReaction::set_default_reaction() {
45
46 general_reaction.type = ReactionType::IGNORED;
47 general_reaction.life_lost = 0;
48 sprite_reactions.clear();
49}
50
51/**
52 * \brief Sets how the enemy reacts when the sprite that receives the attack
53 * has no specific reaction.
54 * \param reaction the reaction to set
55 * \param life_lost number of life points to remove (only for reaction HURT)
56 */
57void EnemyReaction::set_general_reaction(ReactionType reaction, int life_lost) {
58
59 general_reaction.type = reaction;
60 if (reaction == ReactionType::HURT) {
61 general_reaction.life_lost = life_lost;
62 }
63}
64
65/**
66 * \brief Makes the enemy reacts differently when the attack is received by a particular sprite.
67 * \param sprite a sprite of the enemy (if nullptr, this is equivalent to set_general_reaction)
68 * \param reaction the reaction to make when this specific sprites receives an attack
69 * \param life_lost number of life points to remove (only for reaction HURT)
70 */
71void EnemyReaction::set_sprite_reaction(const Sprite* sprite, ReactionType reaction, int life_lost) {
72
73 if (sprite == nullptr) {
74 set_general_reaction(reaction, life_lost);
75 }
76 else {
77 sprite_reactions[sprite].type = reaction;
78 sprite_reactions[sprite].life_lost = life_lost;
79 }
80}
81
82/**
83 * \brief Returns the reaction to an attack on a sprite.
84 * \param sprite the sprite that receives the attack
85 * \return the corresponding reaction
86 */
87const EnemyReaction::Reaction& EnemyReaction::get_reaction(
88 const Sprite* sprite) const {
89
90 if (sprite != nullptr) {
91 auto it = sprite_reactions.find(sprite);
92 if (it != sprite_reactions.end()) {
93 return it->second;
94 }
95 }
96 return general_reaction;
97}
