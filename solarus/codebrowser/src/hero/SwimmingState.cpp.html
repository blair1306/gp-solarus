27/**
28 * \brief Constructor.
29 * \param hero The hero controlled by this state.
30 */
31Hero::SwimmingState::SwimmingState(Hero& hero):
32 PlayerMovementState(hero, "swimming"),
33 fast_swimming(false),
34 end_fast_swim_date(0) {
35
36}
37
38/**
39 * \brief Starts this state.
40 * \param previous_state the previous state
41 */
42void Hero::SwimmingState::start(const State* previous_state) {
43
44 PlayerMovementState::start(previous_state);
45
46 get_equipment().notify_ability_used(Ability::SWIM);
47 get_entity().set_walking_speed(get_slow_swimming_speed());
48 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_SWIM);
49}
50
51/**
52 * \brief Stops this state.
53 * \param next_state the next state
54 */
55void Hero::SwimmingState::stop(const State* next_state) {
56
57 PlayerMovementState::stop(next_state);
58
59 Hero& hero = get_entity();
60 hero.set_walking_speed(hero.get_normal_walking_speed());
61 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
62}
63
64/**
65 * \brief Updates this state.
66 */
67void Hero::SwimmingState::update() {
68
69 PlayerMovementState::update();
70
71 if (is_suspended() || !is_current_state()) {
72 return;
73 }
74
75 Hero& hero = get_entity();
76 if (hero.get_ground_below() != Ground::DEEP_WATER) {
77 hero.set_state(new FreeState(hero));
78 }
79 else if (fast_swimming && System::now() >= end_fast_swim_date) {
80 fast_swimming = false;
81 hero.set_walking_speed(get_slow_swimming_speed());
82
83 if (get_wanted_movement_direction8() != -1) {
84 set_animation_walking();
85 }
86 else {
87 set_animation_stopped();
88 }
89 }
90}
91
92/**
93 * \brief Notifies this state that the game was just suspended or resumed.
94 * \param suspended true if the game is suspended
95 */
96void Hero::SwimmingState::set_suspended(bool suspended) {
97
98 PlayerMovementState::set_suspended(suspended);
99
100 if (!is_suspended() && fast_swimming) {
101 end_fast_swim_date += System::now() - get_when_suspended();
102 }
103}
104
105/**
106 * Gives the sprites the animation stopped corresponding to this state.
107 */
108void Hero::SwimmingState::set_animation_stopped() {
109
110 if (!fast_swimming) {
111 get_sprites().set_animation_stopped_swimming();
112 }
113}
114
115/**
116 * Gives the sprites the animation walking corresponding to this state.
117 */
118void Hero::SwimmingState::set_animation_walking() {
119
120 if (!fast_swimming) {
121 get_sprites().set_animation_swimming_slow();
122 }
123}
124
125/**
126 * \brief Notifies this state that the action command was just pressed.
127 */
128void Hero::SwimmingState::notify_action_command_pressed() {
129 try_swim_faster();
130}
131/**
132 * \brief Notifies this state that the attack command was just pressed.
133 */
134void Hero::SwimmingState::notify_attack_command_pressed() {
135 try_swim_faster();
136}
137
138/**
139 * \brief Makes the hero swim faster for a while if possible.
140 */
141void Hero::SwimmingState::try_swim_faster() {
142
143 if (!fast_swimming) {
144 fast_swimming = true;
145 get_entity().set_walking_speed(get_fast_swimming_speed());
146 get_sprites().set_animation_swimming_fast();
147 Sound::play("swim");
148 end_fast_swim_date = System::now() + 600;
149 }
150}
151
152/**
153 * \brief Returns the normal swimming speed.
154 * \return the swimming speed in pixels per second
155 */
156int Hero::SwimmingState::get_slow_swimming_speed() const {
157 return get_entity().get_normal_walking_speed() / 2;
158}
159
160/**
161 * \brief Returns the faster swimming speed.
162 * \return the faster swimming speed in pixels per second
163 */
164int Hero::SwimmingState::get_fast_swimming_speed() const {
165 return get_entity().get_normal_walking_speed();
166}

168/**
169 * \brief Returns whether some stairs are considered as obstacle in this state.
170 * \param stairs some stairs
171 * \return true if the stairs are obstacle in this state
172 */
173bool Hero::SwimmingState::is_stairs_obstacle(const Stairs& /* stairs */) const {
174 return false;
175}
176
177/**
178 * \brief Returns whether the hero can pick a treasure in this state.
179 * \param item The equipment item to obtain.
180 * \return true if the hero can pick that treasure in this state.
181 */
182bool Hero::SwimmingState::can_pick_treasure(EquipmentItem& /* item */) const {
183 return true;
184}
