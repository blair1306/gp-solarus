17#ifndef SOLARUS_FALLING_HEIGHT_H
18#define SOLARUS_FALLING_HEIGHT_H
19
20namespace Solarus {
21
22/**
23 * \brief Indicates from which height an entity falls.
24 *
25 * This enumeration is used by the FallingOnFloorMovement class,
26 * typically to make pickable items fall on the ground
27 * when an enemy is killed or when a pot is lifted.
28 */
29enum FallingHeight {
30 FALLING_NONE = 0,
31 FALLING_LOW = 1,
32 FALLING_MEDIUM = 2,
33 FALLING_HIGH = 3
34};

36}
