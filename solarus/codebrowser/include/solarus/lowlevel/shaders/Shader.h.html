17#ifndef SOLARUS_SHADER_H
18#define SOLARUS_SHADER_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Debug.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/lua/LuaContext.h"
24#include "solarus/lua/LuaTools.h"
25#include <string>
26
27namespace Solarus {
28
29/**
30 * \brief Represents a shader for a driver and sampler-independant uses.
31 */
32class Shader {
33
34 public:
35
36 explicit Shader(const std::string& shader_name);
37 virtual ~Shader();
38
39 static void set_shading_language_version(const std::string& version);
40 static const std::string& get_sampler_type();
41 static void reset_time();
42
43 const std::string& get_name();
44 double get_default_window_scale();
45 bool is_valid();
46
47 virtual void render(const SurfacePtr& quest_surface) const;
48
49 protected:
50
51 void load(const std::string& shader_name);
52 virtual void register_callback(lua_State* l);
53
54 static std::string shading_language_version; /**< The version of the shading language. */
55 static std::string sampler_type; /**< The sampler type of the shader. */
56 static int display_time; /**< Time since the current shader is displayed (without interruptions). */
57
58 std::string shader_name; /**< The name of the shader. */
59 double default_window_scale; /**< Default scale of the window when the shader is being active,
60 * compared to the normal quest size. */
61 bool is_shader_valid; /**< False if the engine shader context is explicitly set as not compatible with the shader script. */
62
63 private:
64
65 void load_lua_file(const std::string& path);
66};
67
68}
69
70#endif
