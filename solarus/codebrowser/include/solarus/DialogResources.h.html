17#ifndef SOLARUS_DIALOG_RESOURCES_H
18#define SOLARUS_DIALOG_RESOURCES_H
19
20#include "solarus/Common.h"
21#include "solarus/lua/LuaData.h"
22#include <map>
23
24namespace Solarus {
25
26/**
27 * \brief Stores the properties of a dialog.
28 */
29class SOLARUS_API DialogData {
30
31 public:
32
33 DialogData();
34
35 const std::string& get_text() const;
36 void set_text(const std::string& text);
37
38 const std::map<std::string, std::string>& get_properties() const;
39 bool has_property(const std::string& key) const;
40 const std::string& get_property(const std::string& key) const;
41 void set_property(const std::string& key, const std::string& value);
42 bool remove_property(const std::string& key);
43
44 private:
45
46 std::string text; /**< The whole text of this dialog. */
47 std::map<std::string, std::string>
48 properties; /**< Custom properties of this dialog. */
49};
50
51/**
52 * \brief Stores the content of a dialog data file.
53 */
54class SOLARUS_API DialogResources : public LuaData {
55
56 public:
57
58 DialogResources();
59
60 void clear();
61
62 const std::map<std::string, DialogData>& get_dialogs() const;
63 bool has_dialog(const std::string& dialog_id) const;
64 const DialogData& get_dialog(const std::string& dialog_id) const;
65 DialogData& get_dialog(const std::string& dialog_id);
66 bool add_dialog(const std::string& dialog_id, const DialogData& dialog);
67 bool remove_dialog(const std::string& dialog_id);
68 bool set_dialog_id(
69 const std::string& old_dialog_id, const std::string& new_dialog_id);
70
71 virtual bool import_from_lua(lua_State* l) override;
72 virtual bool export_to_lua(std::ostream& out) const override;
73
74 private:
75
76 static int l_dialog(lua_State* l);
77
78 std::map<std::string, DialogData> dialogs;
79};
80
81}
