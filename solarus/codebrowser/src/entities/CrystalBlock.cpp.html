29/**
30 * \brief Creates a new crystal block.
31 * \param game The current game.
32 * \param name Name identifying the entity on the map or an empty string.
33 * \param layer Layer of the entity to create on the map.
34 * \param xy Coordinates of the entity to create.
35 * \param size Size of the block (the pattern can be repeated).
36 * \param subtype subtype of raised block
37 */
38CrystalBlock::CrystalBlock(Game& game, const std::string& name,
39 int layer, const Point& xy, const Size& size, Subtype subtype):
40 Entity(name, 0, layer, xy, size),
41 subtype(subtype) {
42
43 set_collision_modes(CollisionMode::COLLISION_OVERLAPPING);
44 Sprite& sprite = *create_sprite("entities/crystal_block");
45
46 // Don't pause the sprite animation when the crystal block is far from the
47 // camera. Otherwise it looks weird if the players comes back using a
48 // teletransporter.
49 sprite.set_ignore_suspend(true);
50
51 this->orange_raised = game.get_crystal_state();
52
53 if (subtype == ORANGE) {
54 sprite.set_current_animation(orange_raised ? "orange_raised" : "orange_lowered");
55 }
56 else {
57 sprite.set_current_animation(orange_raised ? "blue_lowered" : "blue_raised");
58 }
59 sprite.set_current_frame(sprite.get_nb_frames() - 1); // to avoid the animations at the map beginning
60}

70/**
71 * \brief Returns whether this block is raised.
72 * \return true if this block is raised
73 */
74bool CrystalBlock::is_raised() const {
75
76 return (subtype == ORANGE && orange_raised) ||
77 (subtype == BLUE && !orange_raised);
78}

80/**
81 * \brief Returns whether this entity is an obstacle for another one.
82 * \param other another entity
83 * \return true if this entity is an obstacle for the other one
84 */
85bool CrystalBlock::is_obstacle_for(Entity& other) {
86
87 // if this block is lowered, it is obviously not an obstacle
88 if (!is_raised()) {
89 return false;
90 }
91
92 // if the block is raised, only some specific entities may overlap it
93 return other.is_raised_block_obstacle(*this);
94}

96/**
97 * \brief This function is called when another entity overlaps this crystal block.
98 * \param entity_overlapping the other entity
99 * \param collision_mode the collision mode that detected the collision
100 */
101void CrystalBlock::notify_collision(Entity& entity_overlapping, CollisionMode /* collision_mode */) {
102
103 if (entity_overlapping.is_hero() && is_raised()) {
104
105 // see if we have to make him fall
106
107 Hero& hero = static_cast<Hero&>(entity_overlapping);
108 if (hero.can_control_movement()) {
109
110 Rectangle collision_box = hero.get_bounding_box();
111 int x1 = get_top_left_x();
112 int x2 = x1 + get_width();
113 int y1 = get_top_left_y();
114 int y2 = y1 + get_height();
115 int jump_direction = 0;
116 int jump_length = 0;
117 bool jumped = false;
118
119 const Point& hero_center = hero.get_center_point();
120
121 if (hero_center.y < y1) {
122 // fall to the north
123 collision_box.set_y(y1 - 16);
124 jump_direction = 2;
125 jump_length = hero.get_top_left_y() + 16 - y1;
126 jumped = try_jump(hero, collision_box, jump_direction, jump_length);
127 }
128 else if (hero_center.y >= y2) {
129 // fall to the south
130 collision_box.set_y(y2);
131 jump_direction = 6;
132 jump_length = y2 - hero.get_top_left_y();
133 jumped = try_jump(hero, collision_box, jump_direction, jump_length);
134 }
135
136 if (!jumped) {
137 if (hero_center.x >= x2) {
138 // fall to the east
139 collision_box.set_x(x2);
140 jump_direction = 0;
141 jump_length = x2 - hero.get_top_left_x();
142 try_jump(hero, collision_box, jump_direction, jump_length);
143 }
144 else if (hero_center.x < x1) {
145 // fall to the west
146 collision_box.set_x(x1 - 16);
147 jump_direction = 4;
148 jump_length = hero.get_top_left_x() + 16 - x1;
149 try_jump(hero, collision_box, jump_direction, jump_length);
150 }
151 }
152 }
153 }
154}

156/**
157 * \brief Makes the hero leave this raised crystal block if possible.
158 * \param hero the hero
159 * \param collision_box destination wanted for the jump
160 * \param jump_direction direction of the jump
161 * \param jump_length length of the jump
162 * \return true if the jump was done, i.e. if the collision box was overlapping
163 * no obstacle and no raised crystal block.
164 */
165bool CrystalBlock::try_jump(Hero& hero, const Rectangle& collision_box,
166 int jump_direction, int jump_length) {
167
168 // jump if there is no collision and no other raised crystal blocks
169 if (!get_map().test_collision_with_obstacles(get_layer(), collision_box, hero)
170 && !get_entities().overlaps_raised_blocks(get_layer(), collision_box)) {
171
172 hero.start_jumping(jump_direction, jump_length, true, false);
173 Sound::play("hero_lands");
174 return true;
175 }
176
177 return false;
178}

180/**
181 * \brief Updates the entity.
182 */
183void CrystalBlock::update() {
184
185 const SpritePtr& sprite = get_sprite();
186
187 // see if the state has to be changed
188 bool orange_raised = get_game().get_crystal_state();
189 if (orange_raised != this->orange_raised) {
190
191 this->orange_raised = orange_raised;
192
193 if (sprite != nullptr) {
194
195 if (subtype == ORANGE) {
196 sprite->set_current_animation(orange_raised ? "orange_raised" : "orange_lowered");
197 }
198 else {
199 sprite->set_current_animation(orange_raised ? "blue_lowered" : "blue_raised");
200 }
201 }
202 }
203
204 Entity::update();
205}

207/**
208 * \brief Draws the entity on the map.
209 *
210 * This is a redefinition of Entity::draw_on_map to repeat the block pattern.
211 */
212void CrystalBlock::draw_on_map() {
213
214 const SpritePtr& sprite = get_sprite();
215 if (sprite == nullptr) {
216 return;
217 }
218
219 int x1 = get_top_left_x();
220 int y1 = get_top_left_y();
221 int x2 = x1 + get_width();
222 int y2 = y1 + get_height();
223
224 for (int y = y1; y < y2; y += 16) {
225 for (int x = x1; x < x2; x += 16) {
226 get_map().draw_visual(*sprite, x, y);
227 }
228 }
229}
