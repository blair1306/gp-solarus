17#ifndef SOLARUS_VIDEO_MODE_H
18#define SOLARUS_VIDEO_MODE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Size.h"
22#include <memory>
23#include <string>
24
25namespace Solarus {
26
27class PixelFilter;
28class Shader;
29
30/**
31 * \brief Represents a method to display the quest content on the screen.
32 *
33 * The video mode may include a scaling algorithm.
34 */
35class VideoMode {
36
37 public:
38
39 VideoMode(
40 const std::string& name,
41 const Size& initial_window_size,
42 std::unique_ptr<PixelFilter> software_filter,
43 std::unique_ptr<Shader> hardware_filter
44 );
45
46 const std::string& get_name() const;
47 const Size& get_initial_window_size() const;
48 const PixelFilter* get_software_filter() const;
49 const Shader* get_hardware_filter() const;
50
51 private:
52
53 std::string name; /**< Lua name of this video mode. */
54 Size initial_window_size; /**< Default size of the window when
55 * selecting this video mode. */
56
57 // Pixel filter (in CPU or GPU).
58 std::unique_ptr<PixelFilter> software_filter; /**< Software scaling pixel filter to use or nullptr. */
59 std::unique_ptr<Shader> hardware_filter; /**< Scaling shader to use or nullptr. */
60
61};
62
63}
64
65#endif
