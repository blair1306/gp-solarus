28/**
29 * \brief Creates a pixel bits object.
30 * \param surface The surface where the image is.
31 * \param image_position Position of the image on this surface.
32 */
33PixelBits::PixelBits(const Surface& surface, const Rectangle& image_position):
34 width(0),
35 height(0),
36 nb_integers_per_row(0),
37 bits() {
38
39 // Create a list of boolean values representing the transparency of each pixel.
40 // This list is implemented as bit fields.
41
42 Debug::check_assertion(surface.internal_surface != nullptr,
43 "Attempt to read a surface that doesn't have pixel buffer in RAM.");
44
45 // Clip the rectangle passed as parameter.
46 const Rectangle clipped_image_position(
47 image_position.get_intersection(Rectangle(surface.get_size()))
48 );
49
50 if (clipped_image_position.is_flat()) {
51 return;
52 }
53
54 width = clipped_image_position.get_width();
55 height = clipped_image_position.get_height();
56
57 nb_integers_per_row = width >> 5; // width / 32
58 if ((width & 31) != 0) { // width % 32 != 0
59 nb_integers_per_row++;
60 }
61
62 int pixel_index = clipped_image_position.get_y() * surface.get_width() + clipped_image_position.get_x();
63
64 bits.resize(height);
65 for (int i = 0; i < height; ++i) {
66 bits[i].resize(nb_integers_per_row);
67
68 // Fill the bits for this row, using nb_integers_per_row sequences of 32 bits.
69 int k = -1;
70 uint32_t mask = 0x00000000; // Current bit in the sequence of 32 bits.
71 for (int j = 0; j < width; ++j) {
72 if (mask == 0x00000000) {
73 // Time for a new sequence of 32 bits.
74 k++;
75 mask = 0x80000000;
76 bits[i][k] = 0x00000000; // Initialize the sequence to transparent.
77 }
78
79 // If the pixel is opaque.
80 if (!surface.is_pixel_transparent(pixel_index)) {
81 bits[i][k] |= mask;
82 }
83
84 mask >>= 1;
85 ++pixel_index;
86 }
87 pixel_index += surface.get_width() - width;
88 }
89}

91/**
92 * \brief Detects whether the image represented by these pixel bits is
93 * overlapping another image.
94 * \param other The other image.
95 * \param location1 Position of the upper-left corner of this image on the map.
96 * \param location2 Position of the upper-left corner of the other image.
97 * \return \c true if there is a collision.
98 */
99bool PixelBits::test_collision(
100 const PixelBits& other,
101 const Point& location1,
102 const Point& location2
103) const {
104 const bool debug_pixel_collisions = false;
105
106 if (bits.empty()) {
107 // No image.
108 return false;
109 }
110
111 // Compute both bounding boxes.
112 const Rectangle bounding_box1(location1.x, location1.y, width, height);
113 const Rectangle bounding_box2(location2.x, location2.y, other.width, other.height);
114
115 // Check collision between the two bounding boxes.
116 if (!bounding_box1.overlaps(bounding_box2)) {
117 return false;
118 }
119
120 if (debug_pixel_collisions) {
121 std::cout << System::now() << "\n bounding box collision\n";
122 std::cout << "rect1 = " << bounding_box1 << "\n";
123 std::cout << "rect2 = " << bounding_box2 << "\n";
124 print();
125 other.print();
126 }
127
128 // Compute the intersection between both rectangles.
129 const int intersection_x = std::max(bounding_box1.get_x(), bounding_box2.get_x());
130 const int intersection_y = std::max(bounding_box1.get_y(), bounding_box2.get_y());
131 const Rectangle intersection(
132 intersection_x,
133 intersection_y,
134 std::min(bounding_box1.get_x() + bounding_box1.get_width(),
135 bounding_box2.get_x() + bounding_box2.get_width()) - intersection_x,
136 std::min(bounding_box1.get_y() + bounding_box1.get_height(),
137 bounding_box2.get_y() + bounding_box2.get_height()) - intersection_y);
138
139 if (debug_pixel_collisions) {
140 std::cout << "intersection: " << intersection << "\n";
141 }
142
143 // compute the relative position of the intersection rectangle for each bounding_box
144 Point offset1 = intersection.get_xy() - bounding_box1.get_xy();
145 Point offset2 = intersection.get_xy() - bounding_box2.get_xy();
146
147 if (debug_pixel_collisions) {
148 std::cout << "offset1.x = " << offset1.x << ", offset1.y = " << offset1.y;
149 std::cout << ", offset2.x = " << offset2.x << ", offset2.y = " << offset2.y << std::endl;
150 }
151
152 // For each row of the intersection, we will call row 'a' the row coming from the right bounding box
153 // and row 'b' the one coming from the left bounding box.
154
155 std::vector<std::vector<uint32_t>>::const_iterator rows_a; // for each row: array of masks of the right bounding box
156 std::vector<std::vector<uint32_t>>::const_iterator rows_b; // for each row: array of masks of the left bounding box
157
158 int nb_masks_per_row_a; // number of masks on row a that are in the intersection (row b may have one more mask)
159 int nb_masks_per_row_b; // number of masks on row b that are in the intersection
160 int nb_unused_masks_row_b; // number of unused masks on row b (i.e. before the intersection)
161 int nb_unused_bits_row_b; // number of unused bits on the first used mask of row b
162 int nb_used_bits_row_b; // number of bits used on the first used mask of row b
163
164 // make sure row a starts after row b
165 if (bounding_box1.get_x() > bounding_box2.get_x()) {
166 rows_a = this->bits.begin() + offset1.y;
167 rows_b = other.bits.begin() + offset2.y;
168 nb_unused_masks_row_b = offset2.x >> 5;
169 nb_unused_bits_row_b = offset2.x & 31;
170 }
171 else {
172 rows_a = other.bits.begin() + offset2.y;
173 rows_b = this->bits.begin() + offset1.y;
174 nb_unused_masks_row_b = offset1.x >> 5;
175 nb_unused_bits_row_b = offset1.x & 31;
176 }
177 nb_used_bits_row_b = 32 - nb_unused_bits_row_b;
178
179 // compute the number of masks in the intersection on row a
180 nb_masks_per_row_a = intersection.get_width() >> 5;
181 if ((intersection.get_width() & 31) != 0) {
182 nb_masks_per_row_a++;
183 }
184
185 // compute the number of masks in the intersection on row b
186 nb_masks_per_row_b = (intersection.get_width() + nb_unused_bits_row_b) >> 5;
187 if (((intersection.get_width() + nb_unused_bits_row_b) & 31) != 0) {
188 nb_masks_per_row_b++;
189 }
190
191 // check the collisions each row of the intersection rectangle
192 for (int i = 0; i < intersection.get_height(); ++i) {
193
194 // current row
195 const std::vector<uint32_t>& bits_a = *rows_a;
196 const std::vector<uint32_t>& bits_b = *rows_b;
197
198 ++rows_a;
199 ++rows_b;
200
201 if (debug_pixel_collisions) {
202 std::cout << "*** checking row " << i << " of the intersection rectangle\n";
203 }
204
205 // Check each mask.
206 for (int j = 0; j < nb_masks_per_row_a; ++j) {
207
208 // We compare the left of a with the right part of b,
209 // and the right part of a with the right part of the next b if any.
210
211 uint32_t mask_a = bits_a[j];
212 uint32_t mask_b = bits_b[j + nb_unused_masks_row_b];
213 uint32_t mask_a_left = mask_a >> nb_unused_bits_row_b;
214 uint32_t next_mask_b_left = 0x00000000;
215 if (j + 1 < nb_masks_per_row_a || // Not the last a-mask.
216 nb_masks_per_row_b > nb_masks_per_row_a // Last a-mask but there exists one more b-mask.
217 ) {
218 // We are still inside the intersection: next_mask_b_left exists.
219 next_mask_b_left = bits_b[j + nb_unused_masks_row_b + 1] >> nb_used_bits_row_b;
220 }
221
222 if (debug_pixel_collisions) {
223 std::cout << "mask_a = ";
224 print_mask(mask_a);
225 std::cout << ", mask b = ";
226 print_mask(mask_b);
227 std::cout << "\n";
228 }
229
230 if (((mask_a_left & mask_b) | (mask_a & next_mask_b_left)) != 0x00000000) {
231 return true;
232 }
233 }
234 }
235
236 return false;
237}

239/**
240 * \brief Prints an ASCII representation of the pixels (for debugging purposes only).
241 */
242void PixelBits::print() const {
243
244 std::cout << "frame size is " << width << " x " << height << std::endl;
245 for (int i = 0; i < height; i++) {
246 int k = -1;
247 uint32_t mask = 0x00000000;
248 for (int j = 0; j < width; j++) {
249
250 if (mask == 0x00000000) {
251 k++;
252 mask = 0x80000000;
253 }
254
255 if (bits[i][k] & mask) {
256 std::cout << "X";
257 }
258 else {
259 std::cout << ".";
260 }
261
262 mask >>= 1;
263 }
264 std::cout << std::endl;
265 }
266}

268/**
269 * \brief Prints an ASCII representation of a 32-bit mask (for debugging purposes only).
270 */
271void PixelBits::print_mask(uint32_t mask) const {
272
273 for (int i = 0; i < 32; i++) {
274 std::cout << (((mask & 0x80000000) != 0x00000000) ? "X" : ".");
275 mask <<= 1;
276 }
277}
