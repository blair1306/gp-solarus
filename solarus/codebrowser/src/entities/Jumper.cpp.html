24/**
25 * \brief Creates a jumper.
26 * \param name Name identifying this entity.
27 * \param layer layer of the entity on the map
28 * \param xy Coordinates of the top-left corner of the entity's rectangle.
29 * \param size Size of the entity's rectangle.
30 * \param direction direction of the jump (0 to 7 as the jump may be diagonal)
31 * \param jump_length length of the jump in pixels (usually a multiple of 8)
32 */
33Jumper::Jumper(const std::string& name,
34 int layer,
35 const Point& xy,
36 const Size& size,
37 int direction,
38 int jump_length):
39 Entity(name, 0, layer, xy, size),
40 jump_length(jump_length) {
41
42 // Facing point detection is necessary to avoid sword tapping.
43 set_collision_modes(CollisionMode::COLLISION_CUSTOM | COLLISION_FACING);
44
45 set_direction(direction);
46
47 // check the size
48 if (direction % 2 != 0) {
49 Debug::check_assertion(size.is_square(), "This jumper has a diagonal direction but is not square");
50 }
51 else {
52 if (direction % 4 == 0) {
53 Debug::check_assertion(size.width == 8, "This jumper is horizontal but its height is not 8");
54 }
55 else {
56 Debug::check_assertion(size.height == 8, "This jumper is vertical but its width is not 8");
57 }
58 }
59 // check the jump length
60 Debug::check_assertion(jump_length >= 16, "The jump length of this jumper is lower than 16");
61}

88/**
89 * \brief Returns whether the hero is correctly placed to
90 * start jumping with this jumper.
91 * \param hero The hero.
92 * \param candidate_position The candidate bounding box.
93 * \param extended_region Whether you want to consider the extended jumper
94 * beyond its bounding box, or to keep it restricted to its bounding box.
95 * \return \c true if the hero is correctly placed to start a jump.
96 */
97bool Jumper::is_in_jump_position(
98 const Hero& hero,
99 const Rectangle& candidate_position,
100 bool extended_region) const {
101
102 if (overlaps_jumping_region(candidate_position, extended_region)) {
103 // Overlapping the active region: cannot start a jump from there.
104 return false;
105 }
106
107 const int direction8 = get_direction();
108 const int expected_hero_direction4 = direction8 / 2;
109
110 if (is_jump_diagonal()) {
111 // Diagonal case: The sensor's shape is a diagonal bar.
112
113 // The player should move toward one of both directions.
114 if (!hero.is_moving_towards(expected_hero_direction4) &&
115 !hero.is_moving_towards((expected_hero_direction4 + 1) % 4)) {
116 return false;
117 }
118
119 // Test if the appropriate corner of the hero crosses the diagonal.
120 Point corner = {
121 candidate_position.get_x() - 1,
122 candidate_position.get_y() - 1
123 };
124 if (direction8 == 1 || direction8 == 7) {
125 // Right-up or right-down.
126 corner.x += candidate_position.get_width() + 1;
127 }
128 if (direction8 == 5 || direction8 == 7) {
129 // Left-down or right-down.
130 corner.y += candidate_position.get_height() + 1;
131 }
132
133 return extended_region ?
134 is_point_in_extended_diagonal(corner) :
135 is_point_in_diagonal(corner);
136 }
137
138 // Non-diagonal case: the sensor has one of the four main directions.
139 // Its shape is exactly its rectangle.
140
141 // The player should move toward the jumper's direction.
142 if (!hero.is_moving_towards(expected_hero_direction4)) {
143 return false;
144 }
145
146 Point facing_point;
147
148 switch (expected_hero_direction4) {
149
150 // right
151 case 0:
152 facing_point = {
153 candidate_position.get_x() + 16,
154 candidate_position.get_y() + 8
155 };
156 break;
157
158 // up
159 case 1:
160 facing_point = {
161 candidate_position.get_x() + 8,
162 candidate_position.get_y() - 1
163 };
164 break;
165
166 // left
167 case 2:
168 facing_point = {
169 candidate_position.get_x() - 1,
170 candidate_position.get_y() + 8
171 };
172 break;
173
174 // down
175 case 3:
176 facing_point = {
177 candidate_position.get_x() + 8,
178 candidate_position.get_y() + 16
179 };
180 break;
181
182 default:
183 Debug::die("Invalid direction");
184 }
185
186 if (is_jump_horizontal()) {
187 if (extended_region) {
188 // Are we inside the extended strip?
189 return facing_point.x >= get_top_left_x() &&
190 facing_point.x < get_top_left_x() + get_width();
191 }
192 else {
193 // Are we inside the strip and the bounding box?
194 return overlaps(facing_point.x, facing_point.y - 8)
195 && overlaps(facing_point.x, facing_point.y + 7);
196 }
197 }
198 else {
199 // Same thing for a vertical jump.
200 if (extended_region) {
201 // Are we inside the extended strip?
202 return facing_point.y >= get_top_left_y() &&
203 facing_point.y < get_top_left_y() + get_height();
204 }
205 else {
206 return overlaps(facing_point.x - 8, facing_point.y) &&
207 overlaps(facing_point.x + 7, facing_point.y);
208 }
209 }
210}

212/**
213 * \brief Returns whether an entity's collides with this jumper.
214 *
215 * The result depends on the sensor's shape.
216 *
217 * \param entity the entity
218 * \return true if the entity's collides with this jumper
219 */
220bool Jumper::test_collision_custom(Entity& entity) {
221
222 if (!entity.is_hero()) {
223 return false;
224 }
225
226 return is_in_jump_position(
227 static_cast<Hero&>(entity), entity.get_bounding_box(), false
228 );
229}

239/**
240 * \brief Returns whether this jumper makes horizontal jumps.
241 * \return \c true if this jumper makes horizontal jumps.
242 */
243bool Jumper::is_jump_horizontal() const {
244
245 return get_direction() == 0 || get_direction() == 4;
246}

248/**
249 * \brief Returns whether this jumper makes vertical jumps.
250 * \return \c true if this jumper makes vertical jumps.
251 */
252bool Jumper::is_jump_vertical() const {
253
254 return get_direction() == 2 || get_direction() == 6;
255}

258/**
259 * \brief Returns whether the direction of this jumper is diagonal.
260 * \return \c true if this jumper is a diagonal one.
261 */
262bool Jumper::is_jump_diagonal() const {
263
264 return get_direction() % 2 != 0;

267/**
268 * \brief Returns whether a point is in the 8-pixel active strip
269 * of this diagonal jumper.
270 *
271 * This function only make sense for a diagonal jumper.
272 *
273 * \param point The point to check.
274 * \return \c true if this point is overlapping the jumper
275 */
276bool Jumper::is_point_in_diagonal(const Point& point) const {
277
278 return overlaps(point) &&
279 is_point_in_extended_diagonal(point);
280}
281
282/**
283 * \brief Returns whether a point is in the 8-pixel active strip
284 * of this diagonal jumper or in the extension of that strip.
285 *
286 * This function only make sense for a diagonal jumper.
287 *
288 * \param point The point to check.
289 * \return \c true if this point is overlapping the jumper
290 */
291bool Jumper::is_point_in_extended_diagonal(const Point& point) const {
292
293 const int x = point.x - this->get_x();
294 const int y = point.y - this->get_y();
295 const int width = get_width();
296
297 switch (get_direction()) {
298
299 case 1:
300 return (x >= y) && (x < y + 8);
301
302 case 3:
303 return (x + y <= width) && (x + y > width - 8);
304
305 case 5:
306 return (y >= x) && (y < x + 8);
307
308 case 7:
309 return (x + y >= width) && (x + y < width + 8);
310
311 default:
312 Debug::die("Invalid direction of jumper");
313 }
314
315 return false;
316}

318/**
319 * \brief Returns whether a rectangle overlaps the active region of the jumper.
320 *
321 * For a horizontal or vertical jumper, the active region is the whole
322 * bounding box, that is, an 8-pixel thick horizontal or vertical strip.
323 * For a diagonal jumper, the active region is an 8-pixel thick diagonal strip.
324 *
325 * \param rectangle A rectangle.
326 * \param extended_region Whether you want to consider the extended jumper
327 * beyond its bounding box, or to keep it restricted to its bounding box.
328 * \return \c true if the rectangle overlaps the active region of the jumper.
329 */
330bool Jumper::overlaps_jumping_region(const Rectangle& rectangle, bool /* extended_region */) const {
331
332 if (!is_jump_diagonal()) {
333 return overlaps(rectangle);
334 }
335
336 // Check the 4 corners of the rectangle.
337 Point xy = rectangle.get_xy();
338 if (is_point_in_diagonal(xy)) {
339 return true;
340 }
341
342 xy.x += rectangle.get_width() - 1;
343 if (is_point_in_diagonal(xy)) {
344 return true;
345 }
346
347 xy.y += rectangle.get_height() - 1;
348 if (is_point_in_diagonal(xy)) {
349 return true;
350 }
351
352 xy.x = rectangle.get_x();
353 if (is_point_in_diagonal(xy)) {
354 return true;
355 }
356
357 // Check the two ends of the diagonal.
358 if (get_direction() == 1 || get_direction() == 5) {
359 if (rectangle.contains(get_top_left_x(), get_top_left_y())) {
360 return true;
361 }
362 if (rectangle.contains(
363 get_top_left_x() + get_width() - 1,
364 get_top_left_y() + get_height() - 1)) {
365 return true;
366 }
367 }
368 else {
369 if (rectangle.contains(
370 get_top_left_x() + get_width() - 1, get_top_left_y())) {
371 return true;
372 }
373 if (rectangle.contains(
374 get_top_left_x(), get_top_left_y() + get_height() - 1)) {
375 return true;
376 }
377 }
378
379 return false;
380}
