347inline void QRect::moveCenter(const QPoint &p)
348{
349 int w = x2 - x1;
350 int h = y2 - y1;
351 x1 = p.x() - w/2;
352 y1 = p.y() - h/2;
353 x2 = x1 + w;
354 y2 = y1 + h;
355}

362 *ah = y2 - y1 + 1;
363}
