73/**
74 * \brief Abstract class for all objects placed on a map.
75 *
76 * Example of entities include tiles, enemies, the hero,
77 * non-playing characters, doors, chests, etc.
78 * Each entity has:
79 * - a bounding box that represents its position on the map (a rectangle),
80 * - a layer on the map,
81 * - an origin point, relative to the rectangle's top-left corner.
82 * Some entities can also have a name, a movement and some sprites.
83 */

434 int layer; /**< Layer of the entity on the map.
435 * The layer is constant for the tiles and can change for the hero and the dynamic entities. */
436
437 Rectangle bounding_box; /**< This rectangle represents the position of the entity of the map and is
438 * used for the collision tests. It corresponds to the bounding box of the entity.
439 * It can be different from the sprite's rectangle of the entity.
440 * For example, the hero's bounding box is a 16*16 rectangle, but its sprite may be
441 * a 24*32 rectangle. */
442
443 Ground ground_below; /**< Kind of ground under this entity: grass, shallow water, etc.
444 * Only used by entities sensible to their ground. */
445
446 Point origin; /**< Coordinates of the origin point of the entity,
447 * relative to the top-left corner of its rectangle.
448 * Remember that when you call get_x() and get_y(), you get the coordinates
449 * of the origin point on the map, not the coordinates of the rectangle's
450 * top-left corner.
451 * This is useful because the top-left corner of the entity's bounding box does
452 * not represent the actual entity's coordinates and does not match necessarily
453 * the sprite's rectangle. */
454

457 std::string name; /**< Name of the entity or an empty string.

460 int direction; /**< direction of the entity, not used for all kinds of entities */

521/**
522 * \brief Returns whether or not this entity's bounding box overlaps a specified rectangle.
523 * \param rectangle the rectangle to check
524 * \return true if this entity's bounding box overlaps the specified rectangle
525 */
526inline bool Entity::overlaps(const Rectangle& rectangle) const {
527 return bounding_box.overlaps(rectangle);
528}
529
530/**
531 * \brief Returns whether or not a point overlaps this entity's bounding box.
532 * \param point point to check
533 * \return true if the point is in this entity's bounding box
534 */
535inline bool Entity::overlaps(const Point& point) const {
536 return bounding_box.contains(point);
537}
538
539/**
540 * \brief Returns whether or not a point overlaps this entity's bounding box.
541 * \param x x coordinate of the point to check
542 * \param y y coordinate of the point to check
543 * \return true if the point is in this entity's bounding box
544 */
545inline bool Entity::overlaps(int x, int y) const {
546 return bounding_box.contains(x, y);
547}
548
549/**
550 * \brief Returns whether or not this entity's bounding box overlaps
551 * another entity's bounding box.
552 * \param other another entity
553 * \return true if this entity's bounding box overlaps the other entity's bounding box
554 */
555inline bool Entity::overlaps(const Entity& other) const {
556 return overlaps(other.get_bounding_box());
557}
