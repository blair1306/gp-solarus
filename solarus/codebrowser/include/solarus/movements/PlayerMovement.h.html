17#ifndef SOLARUS_PLAYER_MOVEMENT_H
18#define SOLARUS_PLAYER_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/movements/StraightMovement.h"
22
23namespace Solarus {
24
25/**
26 * \brief Movement of an entity controlled by the player.
27 *
28 * This movement is an 8-direction movement typically used by the hero.
29 * This class basically converts the combination of directional keys pressed
30 * (from the class GameControls) into a movement.
31 */
32class PlayerMovement: public StraightMovement {
33
34 public:
35
36 // creation and destruction
37 explicit PlayerMovement(int speed);
38
39 virtual void update() override;
40
41 int get_moving_speed() const;
42 void set_moving_speed(int moving_speed);
43 int get_wanted_direction8() const;
44 void compute_movement();
45
46 protected:
47
48 void set_wanted_direction();
49
50 int moving_speed; /**< Speed of the entity when it is moving. */
51 int direction8; /**< Current direction of the movement (0 to 7),
52 * as defined by the directional commands
53 * currently pressed by the player (when the
54 * movement allows them) or -1. */
55 bool blocked_by_stream; /**< Whether on a blocking stream. */
56
57};
58
59}
60
61#endif
