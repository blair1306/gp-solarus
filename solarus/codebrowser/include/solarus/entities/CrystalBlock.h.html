17#ifndef SOLARUS_CRYSTAL_BLOCK_H
18#define SOLARUS_CRYSTAL_BLOCK_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <string>
23
24namespace Solarus {
25
26/**
27 * \brief A block in the ground that can be raised or lowered with the help of a crystal.
28 *
29 * When the block is raised, the hero cannot traverse it. However, if the hero is
30 * on raised blocks, he can walk on them until he jumps back to the normal ground.
31 */
32class CrystalBlock: public Entity {
33
34 public:
35
36 static constexpr EntityType ThisType = EntityType::CRYSTAL_BLOCK;
37
38 /**
39 * Subtypes of raised blocks.
40 */
41 enum Subtype {
42 ORANGE = 0, /**< an orange block */
43 BLUE = 1 /**< a blue block */
44 };
45
46 CrystalBlock(Game& game, const std::string& name,
47 int layer, const Point& xy, const Size& size, Subtype subtype);
48
49 virtual EntityType get_type() const override;
50
51 virtual bool is_obstacle_for(Entity& other) override;
52 virtual void notify_collision(Entity& entity_overlapping, CollisionMode collision_mode) override;
53
54 virtual void update() override;
55 virtual void draw_on_map() override;
56
57 bool is_raised() const;
58
59 private:
60
61 Subtype subtype; /**< indicates whether this is an orange or a blue block */
62 bool orange_raised; /**< true if the orange blocks are raised,
63 * false if the blue blocks are raised */
64
65 bool try_jump(Hero& hero, const Rectangle& collision_box,
66 int jump_direction, int jump_length);
67
68};
69
70}
71
72#endif
