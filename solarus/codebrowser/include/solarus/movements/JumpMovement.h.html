26/**
27 * \brief Movement of an entity that jumps towards a direction.

32
33 public:
34
35 JumpMovement(int direction8, int distance, int speed, bool ignore_obstacles);

37 int get_direction8() const;

41 int get_speed() const;
42 void set_speed(int speed);
43
44 virtual int get_displayed_direction4() const override;
45 virtual Point get_displayed_xy() const override;
46
47 virtual const std::string& get_lua_type_name() const override;
48
49 protected:
50
51 virtual void notify_step_done(int step_index, bool success) override;
52
53 private:
54
55 void restart();
56
57 static const std::string basic_trajectories[]; /**< one-pixel trajectory of each direction */
58
59 // properties
60 int direction8; /**< direction of the jump (0 to 7) */
61 int distance; /**< jump length in pixels */
62 int speed; /**< speed in pixels per second */

64 // state
65 int jump_height; /**< current height of the object while jumping */
66
67};
