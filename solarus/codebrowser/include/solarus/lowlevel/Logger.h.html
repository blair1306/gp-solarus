20#include "solarus/Common.h"

24namespace Solarus {
25
26/**
27 * \brief Provides features for logging output and errors.
28 *
29 * Messages logged are all prepended by "[Solarus] [t] " where t is the current
30 * simulated time.
31 * This allows to better distinguish messages from the engine and messages
32 * from the quest.
33 */
34namespace Logger {
35
36SOLARUS_API void print(const std::string& message, std::ostream& out = std::cout);
37
38SOLARUS_API void debug(const std::string& message);
39SOLARUS_API void info(const std::string& message);
40SOLARUS_API void warning(const std::string& message);
41SOLARUS_API void error(const std::string& message);
42SOLARUS_API void fatal(const std::string& message);
43
44} // namespace Logger
45
46} // namespace Solarus
47
48#endif
