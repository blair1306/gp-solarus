17#ifndef SOLARUS_CIRCLE_MOVEMENT_H
18#define SOLARUS_CIRCLE_MOVEMENT_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityPtr.h"
22#include "solarus/movements/Movement.h"
23#include <cstdint>
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief Movement that makes circles around a center.
30 *
31 * This movement is used by entities that makes a circle around a center point or another entity.
32 */
33class CircleMovement: public Movement {
34
35 public:
36
37 // creation and destruction
38 explicit CircleMovement(bool ignore_obstacles);
39
40 // state
41 virtual void update() override;
42 virtual void set_suspended(bool suspended) override;
43 void start();
44 virtual void stop() override;
45 virtual bool is_started() const override;
46 virtual bool is_finished() const override;
47
48 // properties
49 void set_center(const Point& center_point);
50 void set_center(
51 const EntityPtr& center_entity,
52 int x = 0,
53 int y = 0
54 );
55 int get_radius() const;
56 void set_radius(int radius);
57 int get_radius_speed() const;
58 void set_radius_speed(int radius_speed);
59 int get_angle_speed() const;
60 void set_angle_speed(int angle_speed);
61 double get_initial_angle() const;
62 void set_initial_angle(int initial_angle);
63 bool is_clockwise() const;
64 void set_clockwise(bool clockwise);
65 uint32_t get_duration() const;
66 void set_duration(uint32_t duration);
67 int get_max_rotations() const;
68 void set_max_rotations(int max_rotations);
69 uint32_t get_loop() const;
70 void set_loop(uint32_t delay);
71
72 virtual const std::string& get_lua_type_name() const override;
73
74 private:
75
76 void recompute_position();
77
78 // center of the circle
79 EntityPtr center_entity; /**< the entity to make circles around (nullptr if only a point is used) */
80 Point center_point; /**< absolute coordinates of the center if only a point is used,
81 * or coordinates relative to the center entity otherwise */
82
83 // angle
84 int current_angle; /**< current angle in the circle in degrees */
85 int initial_angle; /**< the first circle starts from this angle in degrees */
86 int angle_increment; /**< number of degrees to add when the angle changes (1 or -1) */
87 uint32_t next_angle_change_date; /**< date when the angle changes */
88 uint32_t angle_change_delay; /**< if not zero, time interval between two angle changes in milliseconds */
89
90 // radius
91 int current_radius; /**< current radius of the circle in pixels */
92 int wanted_radius; /**< the current radius changes gradually towards this wanted value */
93 int previous_radius; /**< radius before the movement stops */
94 int radius_increment; /**< number of pixels to add when the radius is changing (1 or -1) */
95 uint32_t next_radius_change_date; /**< date of the next radius change */
96 uint32_t radius_change_delay; /**< if not zero, time interval between two radius changes */
97
98 // stop after an amount of time
99 uint32_t duration; /**< if not zero, the movement will stop after this delay */
100 uint32_t end_movement_date; /**< date when the movement stops */
101
102 // stop after a number of rotations
103 int max_rotations; /**< if not zero, the movement will stop after this number of rotations are done */
104 int nb_rotations; /**< number of complete circles already done */
105
106 // restart when stopped
107 uint32_t loop_delay; /**< if not zero, when the movement finishes, it will start again after this delay */
108 uint32_t restart_date; /**< date when the movement restarts */
109
110};
111
112}
113
114#endif
