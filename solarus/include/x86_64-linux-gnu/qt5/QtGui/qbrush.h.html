160inline Qt::BrushStyle QBrush::style() const { return d->style; }
161inline const QColor &QBrush::color() const { return d->color; }

172typedef QPair<qreal, QColor> QGradientStop;
173typedef QVector<QGradientStop> QGradientStops;

175class Q_GUI_EXPORT QGradient
176{
177 Q_GADGET
178 Q_ENUMS(Type Spread CoordinateMode)
179public:
180 enum Type {
181 LinearGradient,
182 RadialGradient,
183 ConicalGradient,
184 NoGradient
185 };
186
187 enum Spread {
188 PadSpread,
189 ReflectSpread,
190 RepeatSpread
191 };
192
193 enum CoordinateMode {
194 LogicalMode,
195 StretchToDeviceMode,
196 ObjectBoundingMode
197 };
198
199 enum InterpolationMode {
200 ColorInterpolation,
201 ComponentInterpolation
202 };
203
204 QGradient();
205
206 Type type() const { return m_type; }
207
208 inline void setSpread(Spread spread);
209 Spread spread() const { return m_spread; }
210
211 void setColorAt(qreal pos, const QColor &color);
212
213 void setStops(const QGradientStops &stops);
214 QGradientStops stops() const;
215
216 CoordinateMode coordinateMode() const;
217 void setCoordinateMode(CoordinateMode mode);
218
219 InterpolationMode interpolationMode() const;
220 void setInterpolationMode(InterpolationMode mode);
221
222 bool operator==(const QGradient &gradient) const;
223 inline bool operator!=(const QGradient &other) const
224 { return !operator==(other); }
225
226private:
227 friend class QLinearGradient;
228 friend class QRadialGradient;
229 friend class QConicalGradient;
230 friend class QBrush;
231
232 Type m_type;
233 Spread m_spread;
234 QGradientStops m_stops;
235 union {
236 struct {
237 qreal x1, y1, x2, y2;
238 } linear;
239 struct {
240 qreal cx, cy, fx, fy, cradius;
241 } radial;
242 struct {
243 qreal cx, cy, angle;
244 } conical;
245 } m_data;
246 void *dummy;
247};
248
249inline void QGradient::setSpread(Spread aspread)
250{ m_spread = aspread; }
