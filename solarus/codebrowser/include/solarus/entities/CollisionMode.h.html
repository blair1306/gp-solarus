17#ifndef SOLARUS_COLLISION_MODE_H
18#define SOLARUS_COLLISION_MODE_H
19
20namespace Solarus {
21
22/**
23 * \brief Collisions modes an entity can detect.
24 */
25enum CollisionMode {
26 COLLISION_NONE = 0x0000, /**< No collision will be detected
27 * (the detector doesn't detect any entity). */
28 COLLISION_OVERLAPPING = 0x0001, /**< Collision if the entity's rectangle
29 * overlaps the detector's rectangle. */
30 COLLISION_CONTAINING = 0x0002, /**< Collision if the entity's rectangle
31 * is inside the detector's rectangle. */
32 COLLISION_ORIGIN = 0x0004, /**< Collision if the entity's origin point
33 * is inside the detector's rectangle. */
34 COLLISION_FACING = 0x0008, /**< Collision if the entity's facing point corresponding to
35 * its current direction is inside the detector's rectangle. */
36 COLLISION_TOUCHING = 0x0010, /**< Collision if any entity's facing point (i.e. in any of
37 * the four main directions) is inside the detector's rectangle. */
38 COLLISION_CENTER = 0x0020, /**< Collision if the entity's center point
39 * is inside the detector's rectangle. */
40 COLLISION_SPRITE = 0x0040, /**< Collision if an entity's sprite has pixels
41 * overlapping pixels of the detector's sprite. */
42 COLLISION_CUSTOM = 0x0080 /**< Custom collision function, defined by a subclass of Entity. */
43};
44
45}
46
47#endif
