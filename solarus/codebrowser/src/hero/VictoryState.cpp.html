29/**
30 * \brief Constructor.
31 * \param hero The hero controlled by this state.
32 * \param callback_ref Lua ref to a function to call when the
33 * victory sequence finishes (possibly an empty ref).
34 */
35Hero::VictoryState::VictoryState(Hero& hero, const ScopedLuaRef& callback_ref):
36 HeroState(hero, "victory"),
37 end_victory_date(0),
38 finished(false),
39 callback_ref(callback_ref) {
40
41}
42
43/**
44 * \brief Starts this state.
45 * \param previous_state the previous state
46 */
47void Hero::VictoryState::start(const State* previous_state) {
48
49 HeroState::start(previous_state);
50
51 get_sprites().set_animation_victory();
52 get_sprites().set_ignore_suspend(true);
53 Sound::play("victory");
54
55 // compute the date when the victory state is considered as finished,
56 // but the game may be currently suspended
57 uint32_t start_victory_date = is_suspended() ?
58 get_when_suspended() : System::now();
59 end_victory_date = start_victory_date + 1500;
60}
61
62/**
63 * \brief Stops this state.
64 * \param next_state the next state
65 */
66void Hero::VictoryState::stop(const State* next_state) {
67
68 HeroState::stop(next_state);
69 get_sprites().set_ignore_suspend(false);
70 callback_ref.clear();
71}
72
73/**
74 * \brief Updates this state.
75 */
76void Hero::VictoryState::update() {
77
78 HeroState::update();
79
80 if (!finished && System::now() >= end_victory_date) {
81 finished = true;
82 if (!callback_ref.is_empty()) {
83 // The behavior is defined by Lua.
84 callback_ref.clear_and_call("hero victory callback");
85 }
86 else {
87 // By default, get back to the normal state.
88 Hero& hero = get_entity();
89 hero.set_state(new FreeState(hero));
90 }
91 }
92}
93
94/**
95 * \brief Notifies this state that the game was just suspended or resumed.
96 * \param suspended true if the game is suspended
97 */
98void Hero::VictoryState::set_suspended(bool suspended) {
99
100 HeroState::set_suspended(suspended);
101
102 if (!suspended) {
103 end_victory_date += System::now() - get_when_suspended();
104 }
105}
