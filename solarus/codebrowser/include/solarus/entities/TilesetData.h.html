34/**
35 * \brief Kind of scrolling applied to a tile pattern.
36 */
37enum class TileScrolling {
38 NONE, /**< No scrolling. */
39 PARALLAX, /**< Parallax scrolling. */
40 SELF /**< Scrolling on itself. */
41};
42
43template <>
44struct SOLARUS_API EnumInfoTraits<TileScrolling> {
45 static const std::string pretty_name;
46
47 static const EnumInfo<TileScrolling>::names_type names;
48};
49
50/**
51 * \brief Kind of scrolling applied to a tile pattern.
52 */
53enum class TilePatternRepeatMode {
54 ALL, /**< Repeatable in both directions. */
55 HORIZONTAL, /**< Repeatable only horizontally. */
56 VERTICAL, /**< Repeatable only vertically. */
57 NONE /**< Not repeatable. */
58};
59
60template <>
61struct SOLARUS_API EnumInfoTraits<TilePatternRepeatMode> {
62 static const std::string pretty_name;
63
64 static const EnumInfo<TilePatternRepeatMode>::names_type names;
65};
66
67/**
68 * \brief Stores the properties of a tile pattern.
69 */
70class SOLARUS_API TilePatternData {
71
72 public:
73
74 TilePatternData();
75 explicit TilePatternData(const Rectangle& frame);
76
77 Ground get_ground() const;
78 void set_ground(Ground ground);
79
80 int get_default_layer() const;
81 void set_default_layer(int default_layer);
82
83 TileScrolling get_scrolling() const;
84 void set_scrolling(TileScrolling scrolling);
85
86 TilePatternRepeatMode get_repeat_mode() const;
87 void set_repeat_mode(TilePatternRepeatMode repeat_mode);
88
89 bool is_multi_frame() const;
90 int get_num_frames() const;
91 Rectangle get_frame() const;
92 void set_frame(const Rectangle& frame);
93
94 const std::vector<Rectangle>& get_frames() const;
95 void set_frames(const std::vector<Rectangle>& frames);
96
97 private:
98
99 Ground ground; /**< Terrain of this pattern. */
100 int default_layer; /**< Initial layer when creating a tile. */
101 TileScrolling scrolling; /**< Kind of scrolling if any. */
102 TilePatternRepeatMode repeat_mode; /**< How this patterns intends to be repeated. */
103 std::vector<Rectangle> frames; /**< Coordinates of the pattern's frame(s).
104 * - 1 element: one frame (no animation).
105 * - 3 elements: three frames, animated with 0-1-2-0.
106 * - 4 elements: three frames, animated with 0-1-2-1-0. */
107
108};

110/**
111 * \brief Stores the content of a tileset data file.
112 */
113class SOLARUS_API TilesetData : public LuaData {
114
115 public:
116
117 TilesetData();
118
119 Color get_background_color() const;
120 void set_background_color(const Color& background_color);
121
122 int get_num_patterns() const;
123 const std::map<std::string, TilePatternData>& get_patterns() const;
124 bool pattern_exists(const std::string& pattern_id) const;
125 const TilePatternData& get_pattern(const std::string& pattern_id) const;
126 TilePatternData& get_pattern(const std::string& pattern_id);
127 bool add_pattern(
128 const std::string& pattern_id, const TilePatternData& pattern);
129 bool remove_pattern(const std::string& pattern_id);
130 bool set_pattern_id(
131 const std::string& old_pattern_id, const std::string& new_pattern_id);
132
133 int get_num_border_sets() const;
134 const std::map<std::string, BorderSet>& get_border_sets() const;
135 bool border_set_exists(const std::string& border_set_id) const;
136 const BorderSet& get_border_set(const std::string& border_set_id) const;
137 BorderSet& get_border_set(const std::string& border_set_id);
138 bool add_border_set(
139 const std::string& border_set_id, const BorderSet& border_set);
140 bool remove_border_set(const std::string& border_set_id);
141 bool set_border_set_id(
142 const std::string& old_border_set_id, const std::string& new_border_set_id);
143
144 virtual bool import_from_lua(lua_State* l) override;
145 virtual bool export_to_lua(std::ostream& out) const override;
146
147 private:
148
149 Color background_color; /**< Background color of the tileset. */
150 std::map<std::string, TilePatternData>
151 patterns; /**< The tile patterns indexed by their id. */
152 std::map<std::string, BorderSet>
153 border_sets; /**< The border sets indexes by their id. */
154
155};
