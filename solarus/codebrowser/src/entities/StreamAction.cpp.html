27/**
28 * \brief Creates a stream action.
29 * \param stream The stream applied.
30 * \param entity_moved Entity the stream is applied to.
31 */
32StreamAction::StreamAction(Stream& stream, Entity& entity_moved):
33 stream(std::static_pointer_cast<Stream>(stream.shared_from_this())),
34 entity_moved(std::static_pointer_cast<Entity>(entity_moved.shared_from_this())),
35 active(true),
36 suspended(false),
37 when_suspended(0),
38 target(0, 0),
39 next_move_date(0),
40 delay(0) {
41
42 recompute_movement();
43
44 next_move_date = System::now();
45}

72/**
73 * \brief Updates the direction of the movement to the target.
74 *
75 * This function should be called periodically.
76 */
77void StreamAction::recompute_movement() {
78
79 if (!is_active()) {
80 return;
81 }
82
83 // Compute the direction of the movement and its target point.
84 const int direction8 = stream->get_direction();
85 const Point& xy = Entity::direction_to_xy_move(direction8);
86 const int dx = xy.x;
87 const int dy = xy.y;
88 const int speed = stream->get_speed();
89
90 if (speed > 0) {
91 delay = (uint32_t) (1000 / speed);
92 }
93 else {
94 // Stream of speed 0: inactive.
95 delay = 0;
96 }
97
98 if (stream->get_allow_movement()) {
99 // Don't center the entity on non-blocking streams.
100 target = entity_moved->get_xy();
101 }
102 else {
103 target = stream->get_xy();
104 }
105
106 // Stop 16 pixels after the stream.
107 if (dx != 0) {
108 // Horizontal stream.
109 target.x += dx > 0 ? 16 : -16;
110 }
111
112 if (dy != 0) {
113 // Vertical stream.
114 target.y += dy > 0 ? 16 : -16;
115 }
116
117 if (stream->get_direction() % 2 != 0) {
118 if (target != entity_moved->get_xy()) {
119 // Adjust the speed to the diagonal movement.
120 delay = (uint32_t) (delay * std::sqrt(2));
121 }
122 }
123
124 const SpritePtr& sprite = stream->get_sprite();
125 if (sprite != nullptr &&
126 sprite->get_nb_frames() > 1 &&
127 delay == sprite->get_frame_delay() &&
128 sprite->get_next_frame_date() >= System::now() &&
129 next_move_date < sprite->get_next_frame_date()) {
130 // The stream and its sprite happen to have exactly the same speed:
131 // adjust the delay to synchronize them perfectly
132 // to avoid flickering.
133 next_move_date = sprite->get_next_frame_date();
134 }
135}

137/**
138 * \brief Called repeatedly by the main loop.
139 * Updates the effect of the stream on the entity.
140 */
141void StreamAction::update() {
142
143 // If the action is already disabled, do nothing.
144 if (!is_active()) {
145 return;
146 }
147
148 // First check that the stream and the entity still exist
149 // and are enabled.
150 if (stream->is_being_removed()) {
151 stream = nullptr;
152 active = false;
153 return;
154 }
155
156 if (!stream->is_enabled()) {
157 active = false;
158 return;
159 }
160
161 if (entity_moved->is_being_removed()) {
162 entity_moved = nullptr;
163 active = false;
164 return;
165 }
166
167 if (!entity_moved->is_enabled()) {
168 active = false;
169 return;
170 }
171
172 // Stop the stream action if the entity escapes a non-blocking stream.
173 const Point& ground_point = entity_moved->get_ground_point();
174 if (
175 stream->get_allow_movement() &&
176 !stream->overlaps(ground_point) // We are no longer on the stream.
177 ) {
178 // Blocking streams are more special.
179 // The hero cannot escape them so we don't need this.
180 // Also, diagonal blocking streams continue
181 // to move the entity even when it does not overlap anymore.
182 // This is needed to have precise
183 // exact diagonal movements of 16 pixels in stream mazes.
184
185 if (entity_moved->get_distance(target) > 12) {
186 // This last test is to avoid stopping a stream when being close to the target.
187 // Indeed, in the last pixels before the target, the entity's ground
188 // point is no longer on the stream. We continue anyway until the target.
189 active = false;
190 return;
191 }
192 }
193
194 if (is_suspended()) {
195 return;
196 }
197
198 if (stream->get_speed() <= 0) {
199 return;
200 }
201
202 // Update the position.
203 recompute_movement();
204 while (
205 System::now() >= next_move_date &&
206 is_active()
207 ) {
208 next_move_date += delay;
209
210 int dx = 0;
211 int dy = 0;
212 if (target.x > entity_moved->get_x()) {
213 dx = 1;
214 }
215 else if (target.x < entity_moved->get_x()) {
216 dx = -1;
217 }
218 if (target.y > entity_moved->get_y()) {
219 dy = 1;
220 }
221 else if (target.y < entity_moved->get_y()) {
222 dy = -1;
223 }
224
225 if (test_obstacles(dx, dy)) {
226 bool collision = true;
227 // Collision with an obstacle: try only X or only Y.
228 if (dx != 0 && dy != 0) {
229 if (!test_obstacles(dx, 0)) {
230 dy = 0;
231 collision = false;
232 }
233 else if (!test_obstacles(0, dy)) {
234 dx = 0;
235 collision = false;
236 }
237 }
238
239 if (collision) {
240 if (!stream->get_allow_movement()) {
241 // Stop the stream if it was a blocking one.
242 active = false;
243 }
244 break;
245 }
246 }
247
248 entity_moved->set_xy(entity_moved->get_x() + dx, entity_moved->get_y() + dy);
249 entity_moved->notify_position_changed();
250
251 // See if the entity has come outside the stream,
252 // in other words, if the movement is finished.
253 if (has_reached_target()) {
254 // The target point is reached: stop the stream.
255 active = false;
256 }
257 recompute_movement();
258 }
259}

261/**
262 * \brief Returns whether the entity moved has finished to follow the stream.
263 * \return \c true if the target point is reached.
264 */
265bool StreamAction::has_reached_target() const {
266
267 return entity_moved->get_xy() == target;
268}

278/**
279 * \brief Suspends or resumes this stream action.
280 * \param suspended \c true to suspend, \c false to resume.
281 */
282void StreamAction::set_suspended(bool suspended) {
283
284 this->suspended = suspended;
285 if (suspended) {
286 when_suspended = System::now();
287 }
288 else {
289 if (when_suspended != 0) {
290 next_move_date += System::now() - when_suspended;
291 }
292 }
293}
294
295/**
296 * \brief Returns whether an obstacle blocks a candidate move.
297 * \param dx X component of the translation to test.
298 * \param dy Y component of the translation to test.
299 * \return \c true if the entity cannot currently move of dx,dy because of an
300 * obstacle.
301 */
302bool StreamAction::test_obstacles(int dx, int dy) {
303
304 if (entity_moved->get_movement() != nullptr &&
305 entity_moved->get_movement()->are_obstacles_ignored()) {
306 // This entity currently ignores obstacles.
307 return false;
308 }
309
310 Rectangle collision_box = entity_moved->get_bounding_box();
311 collision_box.add_xy(dx, dy);
312
313 Map& map = entity_moved->get_map();
314 return map.test_collision_with_obstacles(
315 entity_moved->get_layer(),
316 collision_box,
317 *entity_moved
318 );
319
320}
