17#ifndef SOLARUS_HERO_HURT_STATE_H
18#define SOLARUS_HERO_HURT_STATE_H
19
20#include "solarus/hero/HeroState.h"
21#include <cstdint>
22
23namespace Solarus {
24
25/**
26 * \brief The state "Hurt" of the hero.
27 */
28class Hero::HurtState: public HeroState {
29
30 public:
31
32 HurtState(Hero& hero,
33 const Point* source_xy,
34 int damage);
35
36 virtual void start(const State* previous_state) override;
37 virtual void stop(const State* next_state) override;
38 virtual void update() override;
39 virtual void set_suspended(bool suspended) override;
40 virtual bool can_start_gameover_sequence() const override;
41 virtual bool is_touching_ground() const override;
42 virtual bool is_teletransporter_obstacle(const Teletransporter& teletransporter) const override;
43 virtual bool is_stream_obstacle(const Stream& stream) const override;
44 virtual bool is_sensor_obstacle(const Sensor& sensor) const override;
45 virtual bool is_separator_obstacle(const Separator& separator) const override;
46 virtual bool can_be_hurt(Entity* attacker) const override;
47 virtual bool can_avoid_switch() const override;
48 virtual bool can_avoid_ice() const override;
49
50 private:
51
52 bool has_source; /**< Whether there is a source_xy rectangle.
53 * If no, the hero does not move. */
54 const Point source_xy; /**< Coordinates of whatever is hurting the
55 * hero (usually an enemy). */
56 int damage; /**< Number of life points to remove
57 * (this number may be reduced by the tunic
58 * or by hero:on_taking_damage()). */
59 uint32_t end_hurt_date; /**< Date when the state ends. */
60
61};
62
63}
64
65#endif
