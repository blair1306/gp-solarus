58class Q_GUI_EXPORT QMatrix // 2D transform matrix
59{
60public:
61 inline explicit QMatrix(Qt::Initialization) {}
62 QMatrix();
63 QMatrix(qreal m11, qreal m12, qreal m21, qreal m22,
64 qreal dx, qreal dy);
65 QMatrix(const QMatrix &matrix);
66
67 void setMatrix(qreal m11, qreal m12, qreal m21, qreal m22,
68 qreal dx, qreal dy);
69
70 qreal m11() const { return _m11; }
71 qreal m12() const { return _m12; }
72 qreal m21() const { return _m21; }
73 qreal m22() const { return _m22; }
74 qreal dx() const { return _dx; }
75 qreal dy() const { return _dy; }
76
77 void map(int x, int y, int *tx, int *ty) const;
78 void map(qreal x, qreal y, qreal *tx, qreal *ty) const;
79 QRect mapRect(const QRect &) const;
80 QRectF mapRect(const QRectF &) const;
81
82 QPoint map(const QPoint &p) const;
83 QPointF map(const QPointF&p) const;
84 QLine map(const QLine &l) const;
85 QLineF map(const QLineF &l) const;
86 QPolygonF map(const QPolygonF &a) const;
87 QPolygon map(const QPolygon &a) const;
88 QRegion map(const QRegion &r) const;
89 QPainterPath map(const QPainterPath &p) const;
90 QPolygon mapToPolygon(const QRect &r) const;
91
92 void reset();
93 inline bool isIdentity() const;
94
95 QMatrix &translate(qreal dx, qreal dy);
96 QMatrix &scale(qreal sx, qreal sy);
97 QMatrix &shear(qreal sh, qreal sv);
98 QMatrix &rotate(qreal a);
99
100 bool isInvertible() const { return !qFuzzyIsNull(_m11*_m22 - _m12*_m21); }
101 qreal determinant() const { return _m11*_m22 - _m12*_m21; }
102
103 QMatrix inverted(bool *invertible = 0) const;
104
105 bool operator==(const QMatrix &) const;
106 bool operator!=(const QMatrix &) const;
107
108 QMatrix &operator*=(const QMatrix &);
109 QMatrix operator*(const QMatrix &o) const;
110
111 QMatrix &operator=(const QMatrix &);
112
113 operator QVariant() const;
114
115private:
116 inline QMatrix(bool)
117 : _m11(1.)
118 , _m12(0.)
119 , _m21(0.)
120 , _m22(1.)
121 , _dx(0.)
122 , _dy(0.) {}
123 inline QMatrix(qreal am11, qreal am12, qreal am21, qreal am22, qreal adx, qreal ady, bool)
124 : _m11(am11)
125 , _m12(am12)
126 , _m21(am21)
127 , _m22(am22)
128 , _dx(adx)
129 , _dy(ady) {}
130 friend class QTransform;
131 qreal _m11, _m12;
132 qreal _m21, _m22;
133 qreal _dx, _dy;
134};

138inline QPoint operator*(const QPoint &p, const QMatrix &m)
139{ return m.map(p); }
140inline QPointF operator*(const QPointF &p, const QMatrix &m)

142inline QLineF operator*(const QLineF &l, const QMatrix &m)
143{ return m.map(l); }
144inline QLine operator*(const QLine &l, const QMatrix &m)
145{ return m.map(l); }
146inline QPolygon operator *(const QPolygon &a, const QMatrix &m)
147{ return m.map(a); }
148inline QPolygonF operator *(const QPolygonF &a, const QMatrix &m)
149{ return m.map(a); }
150inline QRegion operator *(const QRegion &r, const QMatrix &m)
151{ return m.map(r); }
152Q_GUI_EXPORT QPainterPath operator *(const QPainterPath &p, const QMatrix &m);
153
154inline bool QMatrix::isIdentity() const
155{
156 return qFuzzyIsNull(_m11 - 1) && qFuzzyIsNull(_m22 - 1) && qFuzzyIsNull(_m12)
157 && qFuzzyIsNull(_m21) && qFuzzyIsNull(_dx) && qFuzzyIsNull(_dy);
158}
159
160inline bool qFuzzyCompare(const QMatrix& m1, const QMatrix& m2)
161{
162 return qFuzzyCompare(m1.m11(), m2.m11())
163 && qFuzzyCompare(m1.m12(), m2.m12())
164 && qFuzzyCompare(m1.m21(), m2.m21())
165 && qFuzzyCompare(m1.m22(), m2.m22())
166 && qFuzzyCompare(m1.dx(), m2.dx())
167 && qFuzzyCompare(m1.dy(), m2.dy());
168}
