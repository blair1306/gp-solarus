31/**
32 * \brief Loads the animations of a sprite from a file.
33 * \param id Id of the sprite animation set to load
34 * (name of a sprite definition file, without the ".dat" extension).
35 */
36SpriteAnimationSet::SpriteAnimationSet(const std::string& id):
37 id(id) {
38
39 load();
40}
41
42/**
43 * \brief Attempts to load this animation set from its file.
44 */
45void SpriteAnimationSet::load() {
46
47 Debug::check_assertion(animations.empty(),
48 "Animation set already loaded");
49
50 // Load the sprite data file.
51 std::string file_name = std::string("sprites/") + id + ".dat";
52 SpriteData data;
53 bool success = data.import_from_quest_file(file_name);
54 if (success) {
55 // Get the imported data.
56 default_animation_name = data.get_default_animation_name();
57 for (const auto& kvp : data.get_animations()) {
58 add_animation(kvp.first, kvp.second);
59 }
60 }
61}
62
63/**
64 * \brief Adds a new animation to this animation set.
65 *
66 * This function is called while loading the animation set.
67 *
68 * \param animation_name Name of this animation.
69 * \param animation_data Properties of the animation to create.
70 */
71void SpriteAnimationSet::add_animation(
72 const std::string& animation_name,
73 const SpriteAnimationData& animation_data) {
74
75 std::string src_image = animation_data.get_src_image();
76 uint32_t frame_delay = animation_data.get_frame_delay();
77 int frame_to_loop_on = animation_data.get_loop_on_frame();
78 std::vector<SpriteAnimationDirection> directions;
79
80 // Create directions
81 for (const SpriteAnimationDirectionData& direction: animation_data.get_directions()) {
82
83 Size size = direction.get_size();
84 max_size.width = std::max(size.width, max_size.width);
85 max_size.height = std::max(size.height, max_size.height);
86 max_bounding_box |= direction.get_bounding_box();
87
88 directions.emplace_back(direction.get_all_frames(), direction.get_origin());
89 }
90
91 animations.emplace(
92 animation_name,
93 SpriteAnimation(src_image, directions, frame_delay, frame_to_loop_on)
94 );
95}

97/**
98 * \brief When the sprite is displayed on a map, sets the tileset.
99 *
100 * This function must be called if this sprite image depends on the map's tileset.
101 *
102 * \param tileset The tileset.
103 */
104void SpriteAnimationSet::set_tileset(const Tileset& tileset) {
105
106 for (auto& kvp: animations) {
107 kvp.second.set_tileset(tileset);
108 }
109}

111/**
112 * \brief Returns whether this animation set has an animation with the specified name.
113 * \param animation_name an animation name
114 * \return true if this animation exists
115 */
116bool SpriteAnimationSet::has_animation(
117 const std::string& animation_name) const {
118 return animations.find(animation_name) != animations.end();
119}
120
121/**
122 * \brief Returns an animation.
123 * \param animation_name Name of the animation to get.
124 * \return The specified animation.
125 */
126const SpriteAnimation& SpriteAnimationSet::get_animation(
127 const std::string& animation_name) const {
128
129 Debug::check_assertion(has_animation(animation_name),
130 std::string("No animation '") + animation_name
131 + "' in animation set '" + id + "'"
132 );
133
134 return animations.find(animation_name)->second;
135}

137/**
138 * \brief Returns an animation.
139 * \param animation_name Name of the animation to get.
140 * \return The specified animation.
141 */
142SpriteAnimation& SpriteAnimationSet::get_animation(
143 const std::string& animation_name) {
144
145 Debug::check_assertion(has_animation(animation_name),
146 std::string("No animation '") + animation_name
147 + "' in animation set '" + id + "'"
148 );
149
150 return animations.find(animation_name)->second;
151}

153/**
154 * \brief Returns the name of the default animation, i.e. the first one.
155 * \return the name of the default animation
156 */
157const std::string& SpriteAnimationSet::get_default_animation() const {
158 return default_animation_name;
159}

161/**
162 * \brief Enables the pixel-perfect collision detection for these animations.
163 */
164void SpriteAnimationSet::enable_pixel_collisions() {
165
166 if (!are_pixel_collisions_enabled()) {
167
168 for (auto& kvp: animations) {
169 kvp.second.enable_pixel_collisions();
170 }
171 }
172}
173
174/**
175 * \brief Returns whether the pixel-perfect collisions are enabled for these animations.
176 * \return true if the pixel-perfect collisions are enabled
177 */
178bool SpriteAnimationSet::are_pixel_collisions_enabled() const {
179
180 if (animations.empty()) {
181 return false;
182 }
183
184 return animations.begin()->second.are_pixel_collisions_enabled();
185}

187/**
188 * \brief Returns a size big enough to contain any frame of this animation set.
189 * \return The maximum size of a frame in this animation set.
190 */
191const Size& SpriteAnimationSet::get_max_size() const {
192 return max_size;
193}
194
195/**
196 * \brief Returns a rectangle big enough to contain a frame of any animation
197 * and direction.
198 *
199 * The rectangle is anchored to the origin point of the sprite,
200 * so it has has negative top and left coordinates.
201 *
202 * \return The maximum frame size.
203 */
204const Rectangle& SpriteAnimationSet::get_max_bounding_box() const {
205 return max_bounding_box;
206}
