30/**
31 * \brief Checks that a path computed corresponds to the expected result.
32 */
33void test_path_to_hero(TestEnvironment& env, Entity& entity) {
34
35 Hero& hero = env.get_hero();
36
37 entity.set_top_left_xy(144, 104);
38 entity.notify_position_changed();
39 hero.set_top_left_xy(200, 144);
40 hero.notify_position_changed();
41 // FIXME notify automatically from Entity::set_xy()
42
43 PathFinding path_finder(env.get_map(), entity, hero);
44 std::string path = path_finder.compute_path();
45
46 const bool traversable = !hero.is_obstacle_for(entity);
47 const std::string expected_path = traversable ? "7777700" : "77777";
48 Debug::check_assertion(path == expected_path,
49 std::string("Unexpected path: '") + path + "', expected '" + expected_path + "'");
50}
51
52/**
53 * \brief Checks path finding on a custom entity towards a hero.
54 */
55void custom_entity_test(TestEnvironment& env) {
56
57 CustomEntity& entity = *env.make_entity<CustomEntity>();
58
59 test_path_to_hero(env, entity);
60}

62/**

65void npc_test(TestEnvironment& env) {

67 Npc& entity = *env.make_entity<Npc>();
