34/**
35 * \brief Creates a block.
36 * \param name name identifying this block
37 * \param layer layer of the entity to create
38 * \param xy Coordinate of the entity to create.
39 * \param direction the only direction where the block can be moved
40 * or -1 to allow it to be pushed in any direction
41 * \param sprite_name animation set id of the sprite for this block
42 * \param can_be_pushed true to allow the hero to push this block
43 * \param can_be_pulled true to allow the hero to pull this block
44 * \param maximum_moves indicates how many times the block can
45 * be moved (0: none, 1: once, 2: infinite)
46 */
47Block::Block(
48 const std::string& name,
49 int layer,
50 const Point& xy,
51 int direction,
52 const std::string& sprite_name,
53 bool can_be_pushed,
54 bool can_be_pulled,
55 int maximum_moves
56):
57 Entity(name, direction, layer, xy, Size(16, 16)),
58 maximum_moves(maximum_moves),
59 sound_played(false),
60 when_can_move(System::now()),
61 last_position(xy),
62 initial_position(xy),
63 initial_maximum_moves(maximum_moves),
64 can_be_pushed(can_be_pushed),
65 can_be_pulled(can_be_pulled) {
66
67 Debug::check_assertion(maximum_moves >= 0 && maximum_moves <= 2,
68 "maximum_moves must be between 0 and 2");
69
70 set_collision_modes(CollisionMode::COLLISION_FACING);
71 set_origin(8, 13);
72 set_direction(direction);
73 const SpritePtr& sprite = create_sprite(sprite_name);
74 set_drawn_in_y_order(sprite->get_size().height > 16);
75}

121/**
122 * \brief Returns whether the hero is currently considered as an obstacle by this entity.
123 * \param hero the hero
124 * \return true if the hero is an obstacle for this entity.
125 */
126bool Block::is_hero_obstacle(Hero& hero) {
127
128 // The block is not an obstacle when the hero is already overlapping it,
129 // which is easily possible with blocks created dynamically.
130 if (hero.overlaps(*this)) {
131 return false;
132 }
133
134 // When the block is moved by the hero, one pixel can overlap.
135 return get_movement() == nullptr;
136}

200/**
201 * \brief This function is called when the player tries to push or pull this block.
202 * \return true if the player is allowed to move this block
203 */
204bool Block::start_movement_by_hero() {
205
206 Hero& hero = get_hero();
207 bool pulling = hero.is_grabbing_or_pulling();
208 int allowed_direction = get_direction();
209 int hero_direction = hero.get_animation_direction();
210 if (pulling) {
211 // the movement direction is backwards
212 hero_direction = (hero_direction + 2) % 4;
213 }
214
215 if (get_movement() != nullptr // the block is already moving
216 || maximum_moves == 0 // the block cannot move anymore
217 || System::now() < when_can_move // the block cannot move for a while
218 || (pulling && !can_be_pulled) // the hero tries to pull a block that cannot be pulled
219 || (!pulling && !can_be_pushed) // the hero tries to push a block that cannot be pushed
220 || (allowed_direction != -1 && hero_direction != allowed_direction)) { // incorrect direction
221 return false;
222 }
223
224 int dx = get_x() - hero.get_x();
225 int dy = get_y() - hero.get_y();
226
227 set_movement(std::make_shared<RelativeMovement>(
228 std::static_pointer_cast<Hero>(hero.shared_from_this()),
229 dx,
230 dy,
231 false
232 ));
233 sound_played = false;
234
235 return true;
236}

238/**
239 * \brief Notifies the block that it has just moved.
240 */
241void Block::notify_position_changed() {
242
243 Entity::notify_position_changed();
244
245 // Now we know that the block moves at least of 1 pixel:
246 // we can play the sound.
247 if (get_movement() != nullptr && !sound_played) {
248 Sound::play("hero_pushes");
249 sound_played = true;
250 }
251}

253/**
254 * \brief Notifies this entity that it has just failed to change its position
255 * because of obstacles.
256 */
257void Block::notify_obstacle_reached() {
258
259 // The block is stopped by an obstacle while being pushed or pulled.
260 get_hero().notify_grabbed_entity_collision();
261
262 Entity::notify_obstacle_reached();
263}

265/**
266 * \copydoc Entity::notify_ground_below_changed
267 */
268void Block::notify_ground_below_changed() {
269
270 Ground ground = get_ground_below();
271 switch (ground) {
272
273 case Ground::HOLE:
274 Sound::play("jump");
275 remove_from_map();
276 break;
277
278 case Ground::LAVA:
279 case Ground::DEEP_WATER:
280 Sound::play("splash");
281 remove_from_map();
282 break;
283
284 default:
285 break;
286 }
287}

289/**
290 * \brief This function is called when this entity stops being moved by the
291 * hero.
292 */
293void Block::stop_movement_by_hero() {
294
295 clear_movement();
296 when_can_move = System::now() + moving_delay;
297
298 // see if the block has moved
299 if (get_xy() != last_position) {
300
301 // the block has moved
302 last_position = get_xy(); // save the new position for next time
303
304 if (maximum_moves == 1) { // if the block could be moved only once,
305 maximum_moves = 0; // then it cannot move anymore
306 }
307 }
308}

326/**
327 * \brief Resets the block at its initial position.
328 */
329void Block::reset() {
330
331 if (get_movement() != nullptr) {
332 // the block was being pushed or pulled by the hero
333 clear_movement();
334 when_can_move = System::now() + moving_delay;
335 }
336
337 last_position = initial_position;
338 this->maximum_moves = initial_maximum_moves;
339 set_xy(initial_position);
340 notify_position_changed();
341}

377/**
378 * \brief Returns the initial maximum moves of this block.
379 *
380 * This value is the one passed to the constructor or to set_maximum_moves,

383 * \return How many times the block can be moved
384 * (0: none, 1: once, 2: infinite).
385 */
386int Block::get_maximum_moves() const {
387 return initial_maximum_moves;
388}
389
390/**
391 * \brief Sets the maximum moves of this block.
392 *
393 * This resets the remaining allowed moves.
394 *
395 * \param maximum_moves How many times the block can be moved
396 * (0: none, 1: once, 2: infinite).
397 */
398void Block::set_maximum_moves(int maximum_moves) {
399
400 Debug::check_assertion(maximum_moves >= 0 && maximum_moves <= 2,
401 "maximum_moves must be between 0 and 2");
402
403 this->initial_maximum_moves = maximum_moves;
404 this->maximum_moves = maximum_moves;
405}
