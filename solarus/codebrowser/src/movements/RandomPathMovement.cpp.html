21namespace Solarus {
22
23/**
24 * \brief Creates a random walk movement object.
25 * \param speed speed of the movement in pixels per second
26 */
27RandomPathMovement::RandomPathMovement(int speed):
28 PathMovement(create_random_path(), speed, false, false, false) {
29
30}
31
32/**
33 * \brief Updates the movements: detects the collisions
34 * in order to restart the movement.
35 */
36void RandomPathMovement::update() {
37
38 PathMovement::update();
39
40 if (!is_suspended() && PathMovement::is_finished()) {
41
42 // there was a collision or the random path is finished: restart with a new random path
43 set_path(create_random_path());
44 }
45}
46
47/**
48 * \brief Returns whether the movement is finished.
49 * \return always false because the movement is restarted as soon as the path is finished
50 * or an obstacle is reached
51 */
52bool RandomPathMovement::is_finished() const {
53 return false;
54}
55
56/**
57 * \brief Returns the name identifying this type in Lua.
58 * \return the name identifying this type in Lua
59 */
60const std::string& RandomPathMovement::get_lua_type_name() const {
61 return LuaContext::movement_random_path_module_name;
62}
