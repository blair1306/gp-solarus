17#ifndef SOLARUS_TEXT_SURFACE_H
18#define SOLARUS_TEXT_SURFACE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Color.h"
22#include "solarus/lowlevel/Point.h"
23#include "solarus/Drawable.h"
24#include <map>
25#include <string>
26#include <SDL_ttf.h>
27
28namespace Solarus {
29
30class Size;
31
32/**
33 * \brief Draws a line of text on a surface.
34 *
35 * This class encapsulates a graphic surface and handles text rendering,
36 * horizontal and vertical text alignment, color and other properties.
37 * This is the only class related to a font library (SDL_ttf).
38 *
39 * Two types of fonts are supported:
40 * - usual fonts (TTF and other formats are supported),
41 * - an image containing characters drawn.
42 */
43class TextSurface: public Drawable {
44
45 public:
46
47 /**
48 * Horizontal alignment of the text.
49 */
50 enum class HorizontalAlignment {
51 LEFT, /**< the text is aligned to the left */
52 CENTER, /**< the text is centered */
53 RIGHT /**< the text is aligned to the right */
54 };
55
56 /**
57 * Vertical alignment of the text.
58 *
59 */
60 enum class VerticalAlignment {
61 TOP, /**< the text is aligned to the top */
62 MIDDLE, /**< the text is aligned in the middle */
63 BOTTOM /**< the text is aligned to the bottom */
64 };
65
66 /**
67 * Rendering mode of the text.
68 */
69 enum class RenderingMode {
70 SOLID, /**< letters are drawn without effect (faster, visible pixels) */
71 ANTIALIASING /**< letters are drawn with a smooth effect (slower) */
72 };
73
74 TextSurface(int x, int y);
75 TextSurface(int x, int y,
76 HorizontalAlignment horizontal_alignment,
77 VerticalAlignment vertical_alignment);
78
79 const std::string& get_font() const;
80 void set_font(const std::string& font_id);
81 HorizontalAlignment get_horizontal_alignment() const;
82 void set_horizontal_alignment(HorizontalAlignment horizontal_alignment);
83 VerticalAlignment get_vertical_alignment() const;
84 void set_vertical_alignment(VerticalAlignment vertical_alignment);
85 void set_alignment(HorizontalAlignment horizontal_alignment,
86 VerticalAlignment vertical_alignment);
87 RenderingMode get_rendering_mode() const;
88 void set_rendering_mode(RenderingMode rendering_mode);
89 const Color& get_text_color() const;
90 void set_text_color(const Color& color);
91 int get_font_size() const;
92 void set_font_size(int font_size);
93
94 void set_position(int x, int y);
95 int get_x() const;
96 void set_x(int x);
97 int get_y() const;
98 void set_y(int y);
99 const std::string& get_text() const;
100 void set_text(const std::string& text);
101 bool is_empty() const;
102 void add_char(char c);
103
104 int get_width() const;
105 int get_height() const;
106 virtual Size get_size() const override;
107
108 virtual void raw_draw(Surface& dst_surface, const Point& dst_position) override;
109 virtual void raw_draw_region(const Rectangle& region,
110 Surface& dst_surface, const Point& dst_position) override;
111 virtual void draw_transition(Transition& transition) override;
112 virtual Surface& get_transition_surface() override;
113
114 virtual const std::string& get_lua_type_name() const override;
115
116 static constexpr int default_font_size = 11; /**< Default font size for outline fonts. */
117
118 private:
119
120 void rebuild();
121 void rebuild_bitmap();
122 void rebuild_ttf();
123
124 std::string font_id; /**< id of the font of the current text surface */
125 HorizontalAlignment horizontal_alignment; /**< horizontal alignment of the current text surface */
126 VerticalAlignment vertical_alignment; /**< vertical alignment of the current text surface */
127 RenderingMode rendering_mode; /**< rendering mode of the current text surface */
128 Color text_color; /**< color of the text */
129 int font_size; /**< size of the font */
130
131 int x; /**< x coordinate of where the text is aligned */
132 int y; /**< y coordinate of where the text is aligned */
133
134 SurfacePtr surface; /**< the surface to draw */
135 Point text_position; /**< position of the top-left corner of the surface on the screen */
136
137 std::string text; /**< the string to draw (only one line) */
138
139};
140
141}
142
143#endif
