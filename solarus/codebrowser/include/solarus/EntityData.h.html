17#ifndef SOLARUS_ENTITY_DATA_H
18#define SOLARUS_ENTITY_DATA_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityType.h"
22#include "solarus/lowlevel/Point.h"
23#include "solarus/lua/LuaData.h"
24#include <iosfwd>
25#include <map>
26#include <string>
27#include <vector>
28
29namespace Solarus {
30
31/**
32 * \brief Stores the properties of a map entity.
33 *
34 * Properties includes its name, layer, coordinates and properties specific
35 * to each entity type.
36 */
37class SOLARUS_API EntityData : public LuaData {
38
39 public:
40
41 /**
42 * \brief Type of a field of an entity in the map data file.
43 */
44 enum class EntityFieldType {
45 NIL, /**< A field that does not exist. */
46 STRING,
47 INTEGER,
48 BOOLEAN
49 };
50
51 /**
52 * \brief Stores the value of one field of the entity.
53 */
54 struct SOLARUS_API FieldValue {
55
56 FieldValue();
57 explicit FieldValue(const std::string& value);
58 explicit FieldValue(const char* value);
59 explicit FieldValue(int value);
60 explicit FieldValue(bool value);
61
62 bool operator==(const FieldValue& other) const;
63 bool operator!=(const FieldValue& other) const;
64
65 EntityFieldType value_type;
66 std::string string_value;
67 int int_value; // Also used for boolean.
68 };
69
70 /**
71 * \brief Whether a field is optional for an entity of some type.
72 */
73 enum class OptionalFlag {
74 MANDATORY,
75 OPTIONAL
76 };
77
78 /**
79 * \brief Describes a field of an entity type.
80 */
81 struct EntityFieldDescription {
82
83 std::string key; /**< Name of the field. */
84 OptionalFlag optional; /**< Whether the field is optional. */
85 FieldValue default_value; /**< Default value (also determines the value type). */
86
87 };
88
89 using EntityTypeDescription = std::vector<EntityFieldDescription>;
90
91 EntityData();
92 explicit EntityData(EntityType type);
93
94 EntityType get_type() const;
95 void set_type(EntityType type);
96 const std::string& get_type_name() const;
97 bool is_dynamic() const;
98
99 bool has_name() const;
100 std::string get_name() const;
101 void set_name(const std::string& name);
102 int get_layer() const;
103 void set_layer(int layer);
104 Point get_xy() const;
105 void set_xy(const Point& xy);
106
107 void initialize_fields();
108 const std::map<std::string, FieldValue>& get_fields() const;
109 FieldValue get_field(const std::string& key) const;
110 bool is_string(const std::string& key) const;
111 const std::string& get_string(const std::string& key) const;
112 void set_string(const std::string& key, const std::string& value);
113 bool is_integer(const std::string& key) const;
114 int get_integer(const std::string& key) const;
115 void set_integer(const std::string& key, int value);
116 bool is_boolean(const std::string& key) const;
117 bool get_boolean(const std::string& key) const;
118 void set_boolean(const std::string& key, bool value);
119 bool has_field(const std::string& key) const;
120 bool is_field_optional(const std::string& key) const;
121 bool is_field_unset(const std::string& key) const;
122
123 bool import_from_lua(lua_State* l) override;
124 bool export_to_lua(std::ostream& out) const override;
125
126 static EntityData check_entity_data(lua_State* l, int index, EntityType type);
127 static const std::map<EntityType, const EntityTypeDescription> get_entity_type_descriptions();
128
129 private:
130
131 EntityType type; /**< Type of entity. */
132 std::string name; /**< Unique name of the entity on the map. */
133 int layer; /**< Layer of the entity on the map. */
134 Point xy; /**< Entity position on the map. */
135
136 std::map<std::string, FieldValue>
137 fields; /**< Fields specific to the entity type. */
138
139};
140
141} // namespace Solarus
142
143#endif
