30/**
31 * \brief Constructor.
32 * \param hero the hero controlled by this state
33 * \param max_distance maximum distance of the movement in pixels
34 * \param speed speed of the movement in pixels per second
35 * \param tunic_preparing_animation animation name of the hero's tunic sprite
36 * when preparing the boomerang
37 * \param sprite_name animation set id that represents the boomerang
38 */
39Hero::BoomerangState::BoomerangState(
40 Hero& hero,
41 int max_distance,
42 int speed,
43 const std::string& tunic_preparing_animation,
44 const std::string& sprite_name):
45 HeroState(hero, "boomerang"),
46 direction_pressed8(-1),
47 max_distance(max_distance),
48 speed(speed),
49 tunic_preparing_animation(tunic_preparing_animation),
50 sprite_name(sprite_name) {
51
52}
53
54/**
55 * \brief Starts this state.
56 * \param previous_state the previous state
57 */
58void Hero::BoomerangState::start(const State* previous_state) {
59
60 HeroState::start(previous_state);
61
62 const bool boomerang_exists = !get_map().get_entities().get_entities_by_type<Boomerang>().empty();
63 if (boomerang_exists) {
64 Hero& hero = get_entity();
65 hero.set_state(new FreeState(hero));
66 }
67 else {
68 get_sprites().set_animation_boomerang(tunic_preparing_animation);
69 this->direction_pressed8 = get_commands().get_wanted_direction8();
70 }
71}
72
73/**
74 * \brief Updates this state.
75 */
76void Hero::BoomerangState::update() {
77
78 HeroState::update();
79
80 Hero& hero = get_entity();
81 if (hero.is_animation_finished()) {
82
83 if (direction_pressed8 == -1) {
84 // the player can press the diagonal arrows before or after the boomerang key
85 direction_pressed8 = get_commands().get_wanted_direction8();
86 }
87
88 int boomerang_direction8;
89 if (direction_pressed8 == -1 || direction_pressed8 % 2 == 0) {
90 boomerang_direction8 = get_sprites().get_animation_direction() * 2;
91 }
92 else {
93 boomerang_direction8 = direction_pressed8;
94 }
95 double angle = Geometry::degrees_to_radians(boomerang_direction8 * 45);
96 get_entities().add_entity(std::make_shared<Boomerang>(
97 std::static_pointer_cast<Hero>(get_entity().shared_from_this()),
98 max_distance,
99 speed,
100 angle,
101 sprite_name
102 ));
103
104 hero.set_state(new FreeState(hero));
105 }
106}

112 return true;
