26/**
27 * \brief Tests a path movement composed of one step.
28 */
29void one_step_test(TestEnvironment& env, Entity& entity) {
30
31 Point old_xy = entity.get_xy();
32
33 std::shared_ptr<PathMovement> movement = std::make_shared<PathMovement>(
34 "0", 100, false, false, false // 8 pixels to the right.
35 );
36 entity.set_movement(movement);
37
38 while (!movement->is_finished()) {
39 env.step();
40 }
41
42 Debug::check_assertion(entity.get_xy() - old_xy == Point(8, 0),
43 "Unexpected coordinates for 'one_step_test #1'");
44 Debug::check_assertion(movement->get_total_distance_covered() == 8,
45 "Unexpected distance covered for 'one_step_test #1'");
46
47 movement->set_path("4"); // 8 pixels to the left.
48 while (!movement->is_finished()) {
49 env.step();
50 }
51 Debug::check_assertion(entity.get_xy() == old_xy,
52 "Unexpected coordinates for 'one_step_test #2'");
53 Debug::check_assertion(movement->get_total_distance_covered() == 16,
54 "Unexpected distance covered for 'one_step_test #2'");
55
56 movement->set_path("3");
57 while (!movement->is_finished()) {
58 env.step();
59 }
60 Debug::check_assertion(entity.get_xy() - old_xy == Point(-8, -8),
61 "Unexpected coordinates for 'one_step_test #3'");
62 Debug::check_assertion(movement->get_total_distance_covered() == 24,
63 "Unexpected distance covered for 'one_step_test #3'");
64
65 movement->set_path("7");
66 while (!movement->is_finished()) {
67 env.step();
68 }
69 Debug::check_assertion(entity.get_xy() == old_xy,
70 "Unexpected coordinates for 'one_step_test #4'");
71 Debug::check_assertion(movement->get_total_distance_covered() == 32,
72 "Unexpected distance covered for 'one_step_test #4'");
73
74 entity.clear_movement();
75}

77/**
78 * \brief Tests that a path movement returns the correct current direction.
79 */
80void direction_test(TestEnvironment& env, Entity& entity) {
81
82 std::shared_ptr<PathMovement> movement = std::make_shared<PathMovement>(
83 "5", 100, false, false, false // 8 pixels to the right.
84 );
85 entity.set_movement(movement);
86
87 Debug::check_assertion(movement->get_current_direction() == 5,
88 "Unexpected current direction for 'direction_test #1'");
89
90 while (!movement->is_finished()) {
91 env.step();
92 }
93
94 // when the movement is finished, PathMovement::get_current_direction() must return the last direction
95 Debug::check_assertion(movement->get_current_direction() == 5,
96 "Unexpected last direction for 'direction_test #1'");
97}

99/**
100 * \brief Tests a path movement composed of several 8-pixel steps.
101 */
102void multi_step_test(TestEnvironment& env, Entity& entity) {
103
104 Point old_xy = entity.get_xy();
105
106 std::shared_ptr<PathMovement> movement = std::make_shared<PathMovement>(
107 "66", 100, false, false, false // 16 pixels downwards.
108 );
109 entity.set_movement(movement);
110
111 while (!movement->is_finished()) {
112 env.step();
113 }
114
115 Debug::check_assertion(entity.get_xy() - old_xy == Point(0, 16),
116 "Unexpected coordinates for 'multi_step_test #1'");
117 Debug::check_assertion(movement->get_total_distance_covered() == 16,
118 "Unexpected distance covered for 'multi_step_test #1'");
119
120 movement->set_path("220");
121
122 while (!movement->is_finished()) {
123 env.step();
124 }
125
126 Debug::check_assertion(entity.get_xy() - old_xy == Point(8, 0),
127 "Unexpected coordinates for 'multi_step_test #2'");
128 Debug::check_assertion(movement->get_total_distance_covered() == 40,
129 "Unexpected distance covered for 'multi_step_test #2'");
130
131 entity.clear_movement();
132}

134/**
135 * \brief Checks the snap-to-grid option of PathMovement.
136 */
137void snap_test(TestEnvironment& env, Entity& entity) {
138
139 entity.set_top_left_xy(155, 108); // Not aligned to the grid.
140 Debug::check_assertion(!entity.is_aligned_to_grid(),
141 "Entity should not be aligned on the grid");
142
143 // #1: a path movement that does not require the entity to snap to the grid
144 std::shared_ptr<PathMovement> movement = std::make_shared<PathMovement>(
145 "56", 100, false, false, false
146 );
147 entity.set_movement(movement);
148 while (!movement->is_finished()) {
149 env.step();
150 }
151 Debug::check_assertion(entity.get_top_left_xy() == Point(147, 124),
152 "Unexpected coordinates for 'snap_test #1'");
153 Debug::check_assertion(movement->get_total_distance_covered() == 16,
154 "Unexpected distance covered for 'snap_test #1'");
155 entity.clear_movement();
156
157 // #2: a path movement that requires the entity to snap to the grid
158 movement = std::make_shared<PathMovement>(
159 "21", 100, false, false, true
160 );
161 entity.set_movement(movement);
162 while (!movement->is_finished()) {
163 env.step();
164 }
165 Debug::check_assertion(entity.get_top_left_xy() == Point(152, 112),
166 "Unexpected coordinates for 'snap_test #2'");
167 Debug::check_assertion(movement->get_total_distance_covered() == 16,
168 "Unexpected distance covered for 'snap_test #2'");
169
170}

174/**
175 * Tests for the path movement.
176 */
177int main(int argc, char** argv) {
178
179 TestEnvironment env(argc, argv);
180
181 CustomEntity& entity = *env.make_entity<CustomEntity>({ 160, 177 });
182
183 one_step_test(env, entity);
184 multi_step_test(env, entity);
185 direction_test(env, entity);
186 snap_test(env, entity);
187
188 return 0;
189}
