30/**
31 * \brief Constructor.
32 * \param hero The hero controlled by this state.
33 * \param direction8 Direction of the jump (0 to 7).
34 * \param distance Distance of the jump in pixels.
35 * \param ignore_obstacles \c true to make the movement ignore obstacles.
36 * \param with_sound \c true to play the "jump" sound.
37 */
38Hero::JumpingState::JumpingState(
39 Hero& hero,
40 int direction8,
41 int distance,
42 bool ignore_obstacles,
43 bool with_sound
44):
45 HeroState(hero, "jumping"),
46 carried_object() {
47
48 if (get_previous_carried_object_behavior() == CarriedObject::BEHAVIOR_KEEP) {
49 // Keep the carried object of the previous state.
50 carried_object = hero.get_carried_object();
51 }
52
53 this->movement = std::make_shared<JumpMovement>(
54 direction8, distance, 0, ignore_obstacles
55 );
56 this->direction8 = direction8;
57 this->with_sound = with_sound;
58}
59
60/**
61 * \brief Starts this state.
62 * \param previous_state the previous state
63 */
64void Hero::JumpingState::start(const State* previous_state) {
65
66 HeroState::start(previous_state);
67
68 // update the sprites
69 HeroSprites& sprites = get_sprites();
70 sprites.set_animation_direction8(direction8);
71
72 if (carried_object == nullptr) {
73 sprites.set_animation_jumping();
74 }
75 else {
76 sprites.set_animation_walking_carrying();
77 sprites.set_lifted_item(carried_object);
78 }
79
80 // jump
81 get_entity().set_movement(movement);
82
83 if (with_sound) {
84 Sound::play("jump");
85 }
86}
87
88/**
89 * \brief Stops this state.
90 * \param next_state the next state
91 */
92void Hero::JumpingState::stop(const State* next_state) {
93
94 HeroState::stop(next_state);
95
96 get_entity().clear_movement();
97
98 if (carried_object != nullptr) {
99
100 switch (next_state->get_previous_carried_object_behavior()) {
101
102 case CarriedObject::BEHAVIOR_THROW:
103 carried_object->throw_item(get_sprites().get_animation_direction());
104 get_entities().add_entity(carried_object);
105 carried_object = nullptr;
106 get_sprites().set_lifted_item(nullptr);
107 break;
108
109 case CarriedObject::BEHAVIOR_DESTROY:
110 carried_object = nullptr;
111 get_sprites().set_lifted_item(nullptr);
112 break;
113
114 case CarriedObject::BEHAVIOR_KEEP:
115 // The next state is now the owner and has incremented the refcount.
116 carried_object = nullptr;
117 break;
118
119 default:
120 Debug::die("Invalid carried object behavior");
121 }
122 }
123}

125/**
126 * \brief Changes the map.
127 * \param map the new map
128 */
129void Hero::JumpingState::set_map(Map& map) {
130
131 HeroState::set_map(map);
132
133 // the hero may go to another map while jumping and carrying an item
134 if (carried_object != nullptr) {
135 carried_object->set_map(map);
136 }
137}

139/**
140 * \brief Updates this state.
141 */
142void Hero::JumpingState::update() {
143
144 HeroState::update();
145
146 if (carried_object != nullptr) {
147 carried_object->update();
148 }
149
150 if (movement->is_finished()) {
151 get_entity().start_state_from_ground();
152 }
153}

178/**
179 * \brief Returns the direction of the hero's movement as defined by the controls applied by the player
180 * and the movements allowed is the current state.
181 *
182 * If he is not moving, -1 is returned.
183 * This direction may be different from the real movement direction because of obstacles.
184 *
185 * \return the hero's wanted direction between 0 and 7, or -1 if he is stopped
186 */
187int Hero::JumpingState::get_wanted_movement_direction8() const {
188 return get_sprites().get_animation_direction8();
189}
