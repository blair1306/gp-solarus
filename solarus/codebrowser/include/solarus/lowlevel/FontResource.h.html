17#ifndef SOLARUS_FONT_RESOURCE_H
18#define SOLARUS_FONT_RESOURCE_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/SurfacePtr.h"
22#include <map>
23#include <memory>
24#include <string>
25#include <SDL_ttf.h>
26
27namespace Solarus {
28
29/**
30 * \brief Provides access to font files.
31 */
32class FontResource {
33
34 public:
35
36 static void initialize();
37 static void quit();
38
39 static std::string get_default_font_id();
40 static bool exists(const std::string& font_id);
41 static bool is_bitmap_font(const std::string& font_id);
42 static SurfacePtr get_bitmap_font(const std::string& font_id);
43 static TTF_Font& get_outline_font(const std::string& font_id, int size);
44
45 private:
46
47 struct SDL_RWops_Deleter {
48 void operator()(SDL_RWops* rw) {
49 SDL_FreeRW(rw);
50 }
51 };
52 using SDL_RWops_UniquePtr = std::unique_ptr<SDL_RWops, SDL_RWops_Deleter>;
53
54 struct TTF_Font_Deleter {
55 void operator()(TTF_Font* font) {
56 TTF_CloseFont(font);
57 }
58 };
59 using TTF_Font_UniquePtr = std::unique_ptr<TTF_Font, TTF_Font_Deleter>;
60
61 /**
62 * Reading an outline font for a given font size.
63 */
64 struct OutlineFontReader {
65 SDL_RWops_UniquePtr rw;
66 TTF_Font_UniquePtr outline_font;
67 };
68
69 /**
70 * This structure stores in memory the content of a font file.
71 */
72 struct FontFile {
73 std::string file_name; /**< Name of the font file, relative to the data directory. */
74 std::string buffer; /**< The font file loaded into memory. */
75
76 SurfacePtr bitmap_font; /**< The font bitmap. Only used for bitmap fonts. */
77 std::map<int, OutlineFontReader>
78 outline_fonts; /**< This font in any size it was loaded with.
79 * Only used for outline fonts. */
80 };
81
82 static void load_fonts();
83
84 static bool fonts_loaded;
85 static std::map<std::string, FontFile> fonts;
86
87};
88
89}
90
91#endif
