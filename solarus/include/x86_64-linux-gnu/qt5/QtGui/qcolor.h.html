213 Spec cspec;
214 union {
215 struct {
216 ushort alpha;
217 ushort red;
218 ushort green;
219 ushort blue;
220 ushort pad;
221 } argb;
222 struct {
223 ushort alpha;
224 ushort hue;
225 ushort saturation;
226 ushort value;
227 ushort pad;
228 } ahsv;
229 struct {
230 ushort alpha;
231 ushort cyan;
232 ushort magenta;
233 ushort yellow;
234 ushort black;
235 } acmyk;
236 struct {
237 ushort alpha;
238 ushort hue;
239 ushort saturation;
240 ushort lightness;
241 ushort pad;
242 } ahsl;
243 ushort array[5];
244 } ct;

253inline QColor::QColor()
254{ invalidate(); }
255
256inline QColor::QColor(int r, int g, int b, int a)
257{ setRgb(r, g, b, a); }
258
259inline QColor::QColor(const char *aname)
260{ setNamedColor(QLatin1String(aname)); }
261
262inline QColor::QColor(const QString& aname)
263{ setNamedColor(aname); }
264
265inline QColor::QColor(const QColor &acolor)
266 : cspec(acolor.cspec)
267{ ct.argb = acolor.ct.argb; }

269inline bool QColor::isValid() const
270{ return cspec != Invalid; }

272inline QColor QColor::lighter(int f) const
273{ return light(f); }
274
275inline QColor QColor::darker(int f) const
276{ return dark(f); }
