28/**

34 * \endcond

37/**
38 * \brief The Solarus version as a string.
39 *
40 * This string has the form "x.y.z" where x is the major version, y is the
41 * minor version and z is the patch version.
42 */
43#define SOLARUS_VERSION SOLARUS_STRINGIFY(SOLARUS_MAJOR_VERSION) "." SOLARUS_STRINGIFY(SOLARUS_MINOR_VERSION) "." SOLARUS_STRINGIFY(SOLARUS_PATCH_VERSION)
44
45/**
46 * \brief The Solarus version as a string, without patch number.
47 *
48 * This string has the form "x.y" where x is the major version and y is the
49 * minor version.
50 */
51#define SOLARUS_VERSION_WITHOUT_PATCH SOLARUS_STRINGIFY(SOLARUS_MAJOR_VERSION) "." SOLARUS_STRINGIFY(SOLARUS_MINOR_VERSION)

53// Windows specific.
54/**
55 * \def SOLARUS_API
56 * \brief Windows DLL import/export specifications for the Solarus library symbols.
57 */
58#ifndef SOLARUS_LIBRARY_EXPORT
59# ifdef solarus_EXPORTS // Define automatically added by cmake.
60# define SOLARUS_LIBRARY_EXPORT 1
61# else
62# define SOLARUS_LIBRARY_EXPORT 0
63# endif
64#else
65# define SOLARUS_LIBRARY_EXPORT 0
66#endif
67
68#ifdef _WIN32
69# if SOLARUS_LIBRARY_EXPORT == 1
70# define SOLARUS_API __declspec(dllexport)
71# else
72# define SOLARUS_API __declspec(dllimport)
73# endif
74
75# ifdef _MSC_VER
76// TODO MSVC: fix these warnings instead of disabling them
77# pragma warning( disable : 4251 4275 4458 4514 4710 4820 4244 4800)
78// 4251 needs to have dll-interface to be used by clients of class
79// 4275 non dll-interface class 'Foo::Bar' used as base for dll-interface class 'Foo::Baz'
80// 4458 declaration of 'foo' hides class member
81// 4514 unreferenced inline function has been removed
82// 4710 function not inlined
83// 4820 padding added after data member
84// 4244 'argument': conversion from 'foo' to 'bar', possible loss of data
85// 4800 'int': forcing value to bool 'true' or 'false' (performance warning)
86# endif
87
88#else
89# define SOLARUS_API
90#endif

111/**
112 * \def SOLARUS_HAVE_OPENGL
113 * \brief Whether the system supports OpenGL as renderer driver.
114 * This is optional, but allows to use advanced display features.
115 *
116 * This is not used for now and always disabled.
117 * TODO use the standard configuration way
118 */
119#ifdef SOLARUS_HAVE_OPENGL
120# undef SOLARUS_HAVE_OPENGL
121#endif
122#define SOLARUS_HAVE_OPENGL 0

125 * \def SOLARUS_DEFAULT_QUEST
126 * \brief Path of the quest to run is none is specified at runtime.

135 * \def SOLARUS_WRITE_DIR

143# endif
144#endif
