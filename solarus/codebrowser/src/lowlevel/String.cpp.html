17#include "solarus/lowlevel/String.h"
18#include <sstream>
19
20namespace Solarus {
21
22namespace String {
23
24/**
25 * \brief Converts an integer to a string.
26 *
27 * This function replaces std::to_string which is missing on some systems.
28 *
29 * \param value The value to convert.
30 * \return String representation of the value.
31 */
32SOLARUS_API std::string to_string(int value) {
33
34 std::ostringstream oss;
35 oss << value;
36 return oss.str();
37}
