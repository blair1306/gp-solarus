20#include "solarus/Common.h"
21#include "solarus/entities/Entity.h"
22#include <list>
23#include <string>
24
25namespace Solarus {
26
27/**
28 * \brief Represents an explosion on the map.
29 *
30 * An explosion can hurt the hero, the enemies and open weak walls.
31 */
32class Explosion: public Entity {
33
34 public:
35
36 static constexpr EntityType ThisType = EntityType::EXPLOSION;
37
38 Explosion(
39 const std::string& name,
40 int layer,
41 const Point& xy,
42 bool with_damage
43 );
44
45 virtual EntityType get_type() const override;
46
47 // state
48 virtual void update() override;
49 virtual void notify_sprite_frame_changed(Sprite& sprite, const std::string& animation, int frame) override;
50
51 // collisions
52 virtual void notify_collision(Entity& other_entity, Sprite& this_sprite, Sprite& other_sprite) override;
53 virtual void notify_collision_with_switch(Switch& sw, Sprite& sprite_overlapping) override;
54 virtual void notify_collision_with_crystal(Crystal& crystal, Sprite& sprite_overlapping) override;
55 virtual void notify_collision_with_sensor(Sensor& sensor, CollisionMode collision_mode) override;
56 virtual void notify_collision_with_enemy(Enemy& enemy, Sprite& enemy_sprite, Sprite& this_sprite) override;
57 void try_attack_enemy(Enemy& enemy, Sprite &enemy_sprite);
58 virtual void notify_attacked_enemy(
59 EnemyAttack attack,
60 Enemy& victim,
61 const Sprite* victim_sprite,
62 EnemyReaction::Reaction& result,
63 bool killed
64 ) override;
65
66 private:
67
68 std::list<Enemy*> victims; /**< list of enemies successfully hurt by this explosion */
69
70};
71
72}
