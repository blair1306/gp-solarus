17#ifndef SOLARUS_HERO_PUSHING_STATE_H
18#define SOLARUS_HERO_PUSHING_STATE_H
19
20#include "solarus/hero/HeroState.h"
21#include <memory>
22
23namespace Solarus {
24
25class PathMovement;
26
27/**
28 * \brief The state "Pushing" of the hero.
29 */
30class Hero::PushingState: public HeroState {
31
32 public:
33
34 explicit PushingState(Hero& hero);
35
36 virtual void start(const State* previous_state) override;
37 virtual void stop(const State* next_state) override;
38 virtual void update() override;
39 virtual bool can_avoid_stream(const Stream& stream) const override;
40 virtual bool can_start_sword() const override;
41 virtual bool is_moving_grabbed_entity() const override;
42 virtual void notify_grabbed_entity_collision() override;
43 virtual void notify_movement_finished() override;
44 virtual void notify_obstacle_reached() override;
45 virtual void notify_position_changed() override;
46
47 virtual bool can_be_hurt(Entity* attacker) const override;
48 virtual bool can_pick_treasure(EquipmentItem& item) const override;
49 virtual bool is_shallow_water_obstacle() const override;
50 virtual bool is_deep_water_obstacle() const override;
51 virtual bool is_hole_obstacle() const override;
52 virtual bool is_lava_obstacle() const override;
53 virtual bool is_prickle_obstacle() const override;
54
55 private:
56
57 void stop_moving_pushed_entity();
58
59 int pushing_direction4; /**< Direction where the hero is looking (0 to 3). */
60 Entity* pushed_entity; /**< The entity the hero is pushing or nullptr. */
61 std::shared_ptr<PathMovement>
62 pushing_movement; /**< The movement created by this state.
63 * The movement of the hero is also this object,
64 * unless a script decided to change it. */
65};
66
67}

69#endif
