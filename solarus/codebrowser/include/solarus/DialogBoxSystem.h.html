17#ifndef SOLARUS_DIALOG_BOX_SYSTEM_H
18#define SOLARUS_DIALOG_BOX_SYSTEM_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Point.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/lua/ScopedLuaRef.h"
24#include "solarus/Dialog.h"
25#include "solarus/GameCommand.h"
26#include <list>
27#include <memory>
28#include <string>
29
30namespace Solarus {
31
32class Game;
33class TextSurface;
34
35/**
36 * \brief Manages the dialog box where dialogs can be displayed.
37 *
38 * A very basic built-in dialog box is used by default, but the quest
39 * script can define its own one instead.
40 * This class handles both cases.
41 */
42class DialogBoxSystem {
43
44 public:
45
46 explicit DialogBoxSystem(Game& game);
47
48 Game& get_game();
49 bool is_enabled() const;
50
51 void open(
52 const std::string& dialog_id,
53 const ScopedLuaRef& info_ref,
54 const ScopedLuaRef& callback_ref
55 );
56 void close(const ScopedLuaRef& status_ref);
57 bool notify_command_pressed(GameCommand command);
58 const std::string& get_dialog_id() const;
59
60 void draw(const SurfacePtr& dst_surface);
61
62 private:
63
64 bool has_more_lines() const;
65 void show_more_lines();
66
67 Game& game; /**< The game this dialog box belongs to. */
68 std::string dialog_id; /**< Id of the current dialog or an empty string. */
69 Dialog dialog; /**< Current dialog. */
70 ScopedLuaRef callback_ref; /**< Lua ref of a function to call when the dialog finishes. */
71
72 // Fields only used by the built-in dialog box.
73 bool built_in; /**< Whether we are using the built-in dialog box. */
74 static constexpr int nb_visible_lines = 3; /**< Maximum number of visible lines. */
75 std::list<std::string> remaining_lines; /**< Text of each line still to be displayed. */
76 std::shared_ptr<TextSurface>
77 line_surfaces[nb_visible_lines]; /**< Text surface of each visible line. */
78 Point text_position; /**< Destination position of the first line. */
79 bool is_question; /**< Whether the dialog is a question with two possible answers. */
80 bool selected_first_answer; /**< If there is a question: whether the first or second answer is selected. */
81};
82
83}
84
85#endif
