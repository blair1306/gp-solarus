33/**
34 * \brief Constructor.
35 *
36 * Creates a bomb.
37 *
38 * \param name Name identifying the entity on the map or an empty string.
39 * \param layer layer of the entity to create
40 * \param xy Coordinates of the entity to create.
41 */
42Bomb::Bomb(const std::string& name, int layer, const Point& xy):
43 Entity(name, 0, layer, xy, Size(16, 16)),
44 explosion_date(System::now() + 6000) {
45
46 set_collision_modes(CollisionMode::COLLISION_FACING);
47 const SpritePtr& sprite = create_sprite("entities/bomb");
48 sprite->enable_pixel_collisions();
49 set_size(16, 16);
50 set_origin(8, 13);
51 set_drawn_in_y_order(true);
52}

128/**

141/**
142 * \copydoc Entity::notify_collision_with_stream
143 */
144void Bomb::notify_collision_with_stream(Stream& stream, int /* dx */, int /* dy */) {
145
146 if (get_movement() == nullptr) {
147 // TODO use a StreamAction, since it now works with any entity and not only the hero.
148
149 // Check that a significant part of the bomb is on the stream.
150 Rectangle center(get_center_point(), Size(2, 2));
151 center.add_xy(-1, -1);
152
153 if (stream.overlaps(center)) {
154 set_xy(stream.get_xy());
155
156 std::string path = " ";
157 path[0] = path[1] = '0' + stream.get_direction();
158 clear_movement();
159 set_movement(std::make_shared<PathMovement>(
160 path, 64, false, false, false
161 ));
162 }
163 }
164}

166/**
167 * \brief This function is called when the entity has just moved.
168 */
169void Bomb::notify_position_changed() {
170
171 Entity::notify_position_changed();
172
173 if (get_hero().get_facing_entity() == this
174 && get_commands_effects().get_action_key_effect() == CommandsEffects::ACTION_KEY_LIFT
175 && !get_hero().is_facing_point_in(get_bounding_box())) {
176
177 get_commands_effects().set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
178 }
179}

181/**
182 * \copydoc Entity::notify_action_command_pressed
183 */
184bool Bomb::notify_action_command_pressed() {
185
186 CommandsEffects::ActionKeyEffect effect = get_commands_effects().get_action_key_effect();
187
188 if (effect == CommandsEffects::ACTION_KEY_LIFT
189 && get_hero().get_facing_entity() == this
190 && get_hero().is_facing_point_in(get_bounding_box())) {
191
192 get_hero().start_lifting(std::make_shared<CarriedObject>(
193 get_hero(),
194 *this,
195 "entities/bomb",
196 "",
197 0,
198 explosion_date)
199 );
200 Sound::play("lift");
201 remove_from_map();
202 return true;
203 }
204
205 return false;
206}

208/**
209 * \brief This function is called by the map when the game is suspended or resumed.
210 * \param suspended true to suspend the entity, false to resume it
211 */
212void Bomb::set_suspended(bool suspended) {
213
214 Entity::set_suspended(suspended); // suspend the animation and the movement
215
216 if (!suspended && get_when_suspended() != 0) {
217 // recalculate the timer
218 uint32_t diff = System::now() - get_when_suspended();
219 explosion_date += diff;
220 }
221}
222
223/**
224 * \brief This function is called repeatedly.
225 */
226void Bomb::update() {
227
228 Entity::update();
229
230 if (is_suspended()) {
231 return;
232 }
233
234 // check the explosion date
235 uint32_t now = System::now();
236 if (now >= explosion_date) {
237 explode();
238 }
239 else if (now >= explosion_date - 1500) {
240 const SpritePtr& sprite = get_sprite();
241 if (sprite != nullptr &&
242 sprite->get_current_animation() != "stopped_explosion_soon") {
243 sprite->set_current_animation("stopped_explosion_soon");
244 }
245 }
246
247 // destroy the movement once finished
248 if (get_movement() != nullptr && get_movement()->is_finished()) {
249 clear_movement();
250 }
251
252 // check collision with explosions, streams, etc.
253 check_collision_with_detectors();
254}
255
256/**
257 * \brief Makes the bomb explode.
258 */
259void Bomb::explode() {
260
261 get_entities().add_entity(std::make_shared<Explosion>(
262 "", get_layer(), get_center_point(), true
263 ));
264 Sound::play("explosion");
265 remove_from_map();
266}
