17#ifndef SOLARUS_STREAM_ACTION_H
18#define SOLARUS_STREAM_ACTION_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/EntityPtr.h"
22#include "solarus/lowlevel/Point.h"
23#include <cstdint>
24#include <memory>
25
26namespace Solarus {
27
28class Entity;
29class Stream;
30
31/**
32 * \brief Manages a map entity being moved by a stream.
33 *
34 * An object of this class should be created when an entity has to be moved
35 * by a stream, and can be destroyed when the effect of this stream on
36 * this entity is finished. You can call is_active() to know if this happens.
37 */
38class StreamAction {
39
40 public:
41
42 StreamAction(Stream& stream, Entity& entity_moved);
43
44 const Stream& get_stream() const;
45 const Entity& get_entity_moved() const;
46 bool is_active() const;
47
48 void update();
49 bool is_suspended() const;
50 void set_suspended(bool suspended);
51
52 private:
53
54 void recompute_movement();
55 bool test_obstacles(int dx, int dy);
56 bool has_reached_target() const;
57
58 std::shared_ptr<Stream>
59 stream; /**< The stream applied,
60 * or nullptr if it was destroyed. */
61 EntityPtr entity_moved; /**< The entity the stream is applied to,
62 * or nullptr if it was destroyed. */
63 bool active; /**< Whether the stream effect is currently
64 * applied. */
65 bool suspended; /**< Whether the stream effect is currently
66 * suspended. */
67 uint32_t when_suspended; /**< Date when the stream effect was suspended. */
68
69 Point target; /**< Position where to stop. */
70 uint32_t next_move_date; /**< Date of the next one-pixel move. */
71 uint32_t delay; /**< Interval between two one-pixel moves. */
72
73};
74
75}
76
77#endif
