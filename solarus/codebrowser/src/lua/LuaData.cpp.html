28/**
29 * \brief Imports a Lua data file from memory to this object.
30 * \param[in] buffer A memory area with the content of a data file
31 * encoded in UTF-8.
32 * \param[in] file_name Name of a file to use in error messages.
33 * \return \c true in case of success, \c false if the file could not be loaded.
34 */
35bool LuaData::import_from_buffer(
36 const std::string& buffer,
37 const std::string& file_name
38) {
39 // Read the file.
40 lua_State* l = luaL_newstate();
41 if (luaL_loadbuffer(l, buffer.data(), buffer.size(), file_name.c_str()) != 0) {
42 Debug::error(std::string("Failed to load data file: ") + lua_tostring(l, -1));
43 lua_pop(l, 1);
44 return false;
45 }
46
47 bool success = import_from_lua(l);
48 lua_close(l);
49 return success;
50}

52/**
53 * \brief Imports a Lua data file from the filesystem to this object.
54 *
55 * Unlike import_from_quest_file(), this function acts on a regular file on the
56 * filesystem, independently of any notion of quest search path.
57 *
58 * \param[in] file_name Path of the file to load.
59 * The file must be encoded in UTF-8.
60 * \return \c true in case of success, \c false if the file could not be loaded.
61 */
62bool LuaData::import_from_file(const std::string& file_name) {
63
64 lua_State* l = luaL_newstate();
65 if (luaL_loadfile(l, file_name.c_str()) != 0) {
66 Debug::error(std::string("Failed to load data file '") + file_name + "': " + lua_tostring(l, -1));
67 lua_pop(l, 1);
68 return false;
69 }
70
71 bool success = import_from_lua(l);
72 lua_close(l);
73 return success;
74}

76/**
77 * \brief Imports a Lua data file from the current quest to this object.
78 *
79 * This function loads a file in the search path of the current quest.
80 * The actual file might be located in the physical quest data directory,
81 * in the quest write directory or in the quest data archive (see QuestFiles).
82 * This function does the search for you.
83 *
84 * \param[in] quest_file_name Path of the file to load, relative to the quest
85 * data path.
86 * \param[in] language_specific \c true to search in the language-specific
87 * directory of the current language.
88 * The file must be encoded in UTF-8.
89 * \return \c true in case of success, \c false if the file could not be loaded.
90 */
91bool LuaData::import_from_quest_file(
92 const std::string& quest_file_name,
93 bool language_specific
94) {
95 if (!QuestFiles::data_file_exists(quest_file_name, language_specific)) {
96 Debug::error(std::string("Cannot find quest file '") + quest_file_name + "'");
97 return false;
98 }
99
100 const std::string& buffer = QuestFiles::data_file_read(
101 quest_file_name, language_specific
102 );
103 return import_from_buffer(buffer, quest_file_name);
104}

106/**
107 * \brief Saves this object into memory as Lua.
108 * \param[out] buffer The buffer to write.
109 * Text will be encoded in UTF-8.
110 * \return \c true in case of success, \c false if the data
111 * could not be exported.
112 */
113bool LuaData::export_to_buffer(std::string& buffer) const {
114
115 std::ostringstream oss;
116 if (!export_to_lua(oss)) {
117 return false;
118 }
119
120 buffer = oss.str();
121 return true;
122}

124/**
125 * \brief Saves the data into a Lua file.
126 * \param[in] file_name Path of the file to save.
127 * The file will be encoded in UTF-8.
128 * \return \c true in case of success, \c false if the data
129 * could not be exported.
130 */
131bool LuaData::export_to_file(const std::string& file_name) const {
132
133 // Work on a temporary file to keep the initial one intact in case of failure.
134 std::string tmp_file_name = file_name + ".solarus_tmp";
135 std::ofstream tmp_out(tmp_file_name);
136 if (!tmp_out) {
137 return false;
138 }
139
140 if (!export_to_lua(tmp_out)) {
141 std::remove(tmp_file_name.c_str());
142 return false;
143 }
144 tmp_out.flush();
145 tmp_out.close();
146
147 std::ifstream in(tmp_file_name);
148 if (!in) {
149 return false;
150 }
151 std::ofstream out(file_name);
152 if (!out) {
153 return false;
154 }
155 out << in.rdbuf();
156
157 in.close();
158 std::remove(tmp_file_name.c_str());
159 out.flush();
160 return true;
161}

183/**
184 * \brief Protects a string so that it can safely be enclosed in double quotes.
185 *
186 * Characters \ and " are protected by a backslash.
187 *
188 * Note that there is no unescape_string() function because the Lua parser
189 * already replaces \\ and \".
190 *
191 * \param value The string value to escape.
192 * \return The escaped value.
193 */
194std::string LuaData::escape_string(std::string value) {
195
196 for (size_t i = 0; i < value.size(); ++i) {
197 if (value[i] == '\\') {
198 value.replace(i, 1, "\\\\");
199 ++i;
200 }
201 else if (value[i] == '"') {
202 value.replace(i, 1, "\\\"");
203 ++i;
204 }
205 }
206
207 return value;
208}
209
210/**
211 * \brief Protects a string so that it can safely be enclosed between [[ and ]].
212 *
213 * Characters \, [ and ] are protected by a backslash.
214 *
215 * \param value The string value to escape.
216 * \return The escaped value.
217 */
218std::string LuaData::escape_multiline_string(std::string value) {
219
220 for (size_t i = 0; i < value.size(); ++i) {
221 if (value[i] == '\\') {
222 value.replace(i, 1, "\\\\");
223 ++i;
224 }
225 else if (value[i] == '[') {
226 value.replace(i, 1, "\\[");
227 ++i;
228 }
229 else if (value[i] == ']') {
230 value.replace(i, 1, "\\]");
231 ++i;
232 }
233 }
234
235 return value;
236}
237
238/**
239 * \brief Restores an original string after a call to escape_multiline_string().
240 * \param value The string value to unescape.
241 * \return The unescaped value.
242 */
243std::string LuaData::unescape_multiline_string(std::string value) {
244
245 // The Lua parsing step already replaced \\.
246 for (int i = 0; i < (int) (value.size() - 1); ++i) {
247 if (value[i] == '\\' && value[i + 1] == '[') {
248 value.replace(i, 2, "[");
249 }
250 else if (value[i] == '\\' && value[i + 1] == ']') {
251 value.replace(i, 2, "]");
252 }
253 }
254
255 return value;
256}
