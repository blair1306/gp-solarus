113/**
114 * \brief Initializes the map features provided to Lua.

116void LuaContext::register_map_module() {
117
118 static const luaL_Reg methods[] = {

124 { "get_max_layer", map_api_get_max_layer },
125 { "get_size", map_api_get_size },
126 { "get_location", map_api_get_location },
127 { "get_floor", map_api_get_floor },
128 { "set_floor", map_api_set_floor },
129 { "get_tileset", map_api_get_tileset },
130 { "set_tileset", map_api_set_tileset },
131 { "get_music", map_api_get_music },
132 { "get_camera", map_api_get_camera },
133 { "get_camera_position", map_api_get_camera_position },
134 { "move_camera", map_api_move_camera },
135 { "get_ground", map_api_get_ground },
136 { "draw_visual", map_api_draw_visual },
137 { "draw_sprite", map_api_draw_sprite },
138 { "get_crystal_state", map_api_get_crystal_state },
139 { "set_crystal_state", map_api_set_crystal_state },
140 { "change_crystal_state", map_api_change_crystal_state },
141 { "open_doors", map_api_open_doors },
142 { "close_doors", map_api_close_doors },
143 { "set_doors_open", map_api_set_doors_open },
144 { "get_entity", map_api_get_entity },
145 { "has_entity", map_api_has_entity },
146 { "get_entities", map_api_get_entities },
147 { "get_entities_count", map_api_get_entities_count },
148 { "has_entities", map_api_has_entities },
149 { "get_entities_by_type", map_api_get_entities_by_type },
150 { "get_entities_in_rectangle", map_api_get_entities_in_rectangle },
151 { "get_entities_in_region", map_api_get_entities_in_region },
152 { "get_hero", map_api_get_hero },
153 { "set_entities_enabled", map_api_set_entities_enabled },
154 { "remove_entities", map_api_remove_entities },
155 { nullptr, nullptr }
156 };
157
158 static const luaL_Reg metamethods[] = {
159 { "__gc", userdata_meta_gc },
160 { "__newindex", userdata_meta_newindex_as_table },
161 { "__index", userdata_meta_index_as_table },
162 { nullptr, nullptr }
163 };
164
165 register_type(map_module_name, nullptr, methods, metamethods);
166
167 // Add map:create_* functions as closures because we pass the entity type as upvalue.
168 luaL_getmetatable(l, map_module_name.c_str());
169 for (const auto& kvp : EnumInfoTraits<EntityType>::names) {
170 EntityType type = kvp.first;
171 const std::string& type_name = kvp.second;
172 if (!EntityTypeInfo::can_be_created_from_lua_api(type)) {
173 continue;
174 }
175 std::string function_name = "create_" + type_name;
176 push_string(l, type_name);
177 lua_pushcclosure(l, map_api_create_entity, 1);
178 lua_setfield(l, -2, function_name.c_str());
179 }
180
181 // Add a Lua implementation of the deprecated map:move_camera() function.
182 int result = luaL_loadstring(l, move_camera_code);
183 if (result != 0) {
184 Debug::error(std::string("Failed to initialize map:move_camera(): ") + lua_tostring(l, -1));
185 lua_pop(l, 1);
186 }
187 else {
188 Debug::check_assertion(lua_isfunction(l, -1), "map:move_camera() is not a function");
189 lua_setfield(l, LUA_REGISTRYINDEX, "map.move_camera");
190 }
191}

229/**
230 * \brief Checks and returns the layer of an entity to be created.
231 *
232 * The layer is assumed to be specified in a field "layer".
233 * Throws a LuaException if the layer is invalid for the map.
234 *
235 * \param l A Lua state.
236 * \param index Index of the argument in the Lua stack.
237 * \param entity_data Description of the entity to create.
238 * \param map The map where to create the entity.
239 * \return The layer.
240 */
241int entity_creation_check_layer(
242 lua_State* l,
243 int index,
244 const EntityData& entity_data,
245 const Map& map) {
246
247 const int layer = entity_data.get_layer();
248
249 if (!map.is_valid_layer(layer)) {
250 std::ostringstream oss;
251 oss << "Invalid layer: " << layer;
252 LuaTools::arg_error(l, index, oss.str());
253 }
254
255 return layer;
256}

258/**
259 * \brief Checks and returns the size of an entity to be created.
260 *
261 * The size is assumed to be specified in fields "width" and "height".
262 * Throws a LuaException if the size is invalid.
263 *
264 * \param l A Lua state.
265 * \param index Index of the argument in the Lua stack.
266 * \param entity_data Description of the entity to create.
267 * \return The size.
268 */
269Size entity_creation_check_size(
270 lua_State* l,
271 int index,
272 const EntityData& entity_data) {
273
274 const Size size = {
275 entity_data.get_integer("width"),
276 entity_data.get_integer("height")
277 };
278
279 if (size.width < 0 || size.width % 8 != 0) {
280 std::ostringstream oss;
281 oss << "Invalid width " << size.width << ": should be a positive multiple of 8";
282 LuaTools::arg_error(l, index, oss.str());
283 }
284 if (size.height < 0 || size.height % 8 != 0) {
285 std::ostringstream oss;
286 oss << "Invalid height " << size.height << ": should be a positive multiple of 8";
287 LuaTools::arg_error(l, index, oss.str());
288 }
289
290 return size;
291}

293/**
294 * \brief Checks and returns a mandatory savegame variable field for an
295 * entity to be created.
296 *
297 * Throws a LuaException if the savegame variable name is invalid or empty.
298 *
299 * \param l A Lua state.
300 * \param index Index of the argument in the Lua stack.
301 * \param entity_data Description of the entity to create.
302 * \param field_name Name of the savegame variable field to check.
303 * \return The savegame variable name.
304 */
305std::string entity_creation_check_savegame_variable_mandatory(
306 lua_State* l,
307 int index,
308 const EntityData& entity_data,
309 const std::string& field_name
310) {
311
312 const std::string& savegame_variable = entity_data.get_string(field_name);
313
314 if (!LuaTools::is_valid_lua_identifier(savegame_variable)) {
315 LuaTools::arg_error(l, index,
316 "Bad field '" + field_name + "' (invalid savegame variable identifier: '"
317 + savegame_variable + "')"
318 );
319 }
320
321 return savegame_variable;
322}

352/**
353 * \brief Checks an enum field for an entity to be created.
354 *
355 * Throws a LuaException if the value is not a string or if the string is not
356 * a valid name for the enum.
357 * This is a useful function for mapping strings to C or C++ enums.
358 *
359 * \param l A Lua state.
360 * \param index Index of the argument in the Lua stack.
361 * \param entity_data Description of the entity to
362 * \param field_name Name of the field to check.
363 * \param names A mapping of enum values to strings to search in.
364 * \return The enumerated value corresponding to this string.
365 */
366template<typename E>
367E entity_creation_check_enum(
368 lua_State* l,
369 int index,
370 const EntityData& entity_data,
371 const std::string& field_name,
372 const std::map<E, std::string>& names
373) {
374 const std::string& name = entity_data.get_string(field_name);
375 for (const auto& kvp: names) {
376 if (kvp.second == name) {
377 return kvp.first;
378 }
379 }
380
381 // The value was not found. Build an error message with possible values.
382 std::string allowed_names;
383 for (const auto& kvp: names) {
384 allowed_names += "\"" + kvp.second + "\", ";
385 }
386 allowed_names = allowed_names.substr(0, allowed_names.size() - 2);
387
388 LuaTools::arg_error(l, index,
389 std::string("Invalid name '") + name + "'. Allowed names are: "
390 + allowed_names
391 );
392 return E(); // Make sure the compiler is happy.
393}
394
395template<typename E>
396E entity_creation_check_enum(
397 lua_State* l,
398 int index,
399 const EntityData& entity_data,
400 const std::string& field_name
401) {
402 return entity_creation_check_enum<E>(l, index, entity_data, field_name, EnumInfoTraits<E>::names);
403}

407/**
408 * \brief Creates a tile on the map.
409 * \param l The Lua context that is calling this function.
410 * \return Number of values to return to Lua.
411 */
412int LuaContext::l_create_tile(lua_State* l) {
413
414 return LuaTools::exception_boundary_handle(l, [&] {
415 Map& map = *check_map(l, 1);
416 EntityData& data = *(static_cast<EntityData*>(lua_touserdata(l, 2)));
417 const int layer = entity_creation_check_layer(l, 1, data, map);
418 const int x = data.get_xy().x;
419 const int y = data.get_xy().y;
420 const Size size = entity_creation_check_size(l, 1, data);
421 const std::string& tile_pattern_id = data.get_string("pattern");
422
423 const Tileset& tileset = map.get_tileset();
424 const TilePattern& pattern = tileset.get_tile_pattern(tile_pattern_id);
425 const Size& pattern_size = pattern.get_size();
426 Entities& entities = map.get_entities();
427
428 // If the tile is big, divide it in several smaller tiles so that
429 // most of them can still be optimized away.
430 // Otherwise, tiles expanded in big rectangles like a lake or a dungeon
431 // floor would be entirely redrawn at each frame if just one small
432 // animated tile overlapped them.
433
434 TileInfo tile_info;
435 tile_info.layer = layer;
436 tile_info.box = { Point(), pattern_size };
437 tile_info.pattern_id = tile_pattern_id;
438 tile_info.pattern = &pattern;
439
440 for (int current_y = y; current_y < y + size.height; current_y += pattern.get_height()) {
441 for (int current_x = x; current_x < x + size.width; current_x += pattern.get_width()) {
442 tile_info.box.set_xy(current_x, current_y);
443 // The tile will actually be created only if it cannot be optimized away.
444 entities.add_tile_info(
445 tile_info
446 );
447 }
448 }
449
450 return 0;
451 });
452}

559
560 if (map.is_started()) {
561 push_entity(l, *entity);
562 return 1;
563 }

607/**
608 * \brief Creates a chest on the map.
609 * \param l The Lua context that is calling this function.
610 * \return Number of values to return to Lua.
611 */
612int LuaContext::l_create_chest(lua_State* l) {
613
614 return LuaTools::exception_boundary_handle(l, [&] {
615 Map& map = *check_map(l, 1);
616 EntityData& data = *(static_cast<EntityData*>(lua_touserdata(l, 2)));
617
618 Chest::OpeningMethod opening_method = entity_creation_check_enum<Chest::OpeningMethod>(
619 l,
620 1,
621 data,
622 "opening_method",
623 Chest::opening_method_names
624 );
625
626 // Check the value of opening_condition depending on the opening method.
627 Game& game = map.get_game();
628 const std::string& opening_condition = data.get_string("opening_condition");
629 if (opening_method == Chest::OpeningMethod::BY_INTERACTION_IF_SAVEGAME_VARIABLE) {
630 entity_creation_check_savegame_variable_mandatory(l, 1, data, "opening_condition");
631 }
632 else if (opening_method == Chest::OpeningMethod::BY_INTERACTION_IF_ITEM) {
633 if (!game.get_equipment().item_exists(opening_condition)) {
634 LuaTools::arg_error(l, 1, "Bad field 'opening_condition' (no such equipment item: '"
635 + opening_condition + "')"
636 );
637 }
638 EquipmentItem& item = game.get_equipment().get_item(opening_condition);
639 if (!item.is_saved()) {
640 LuaTools::arg_error(l, 1, "Bad field 'opening_condition' (equipment item '"
641 + opening_condition + "' is not saved)"
642 );
643 }
644 }
645

1357/**
1358 * \brief Closure of an iterator over a list of entities.
1359 *
1360 * This closure expects 3 upvalues in this order:
1361 * - The array of entities.
1362 * - The size of the array (for performance).
1363 * - The current index in the array.
1364 *
1365 * \param l The Lua context that is calling this function.
1366 * \return Number of values to return to Lua.
1367 */
1368int LuaContext::l_entity_iterator_next(lua_State* l) {
1369
1370 return LuaTools::exception_boundary_handle(l, [&] {
1371
1372 // Get upvalues.
1373 const int table_index = lua_upvalueindex(1);
1374 const int size = lua_tointeger(l, lua_upvalueindex(2));
1375 int index = lua_tointeger(l, lua_upvalueindex(3));
1376
1377 if (index > size) {
1378 // Finished.
1379 return 0;
1380 }
1381
1382 // Get the next value.
1383 lua_rawgeti(l, table_index, index);
1384
1385 // Increment index.
1386 ++index;
1387 lua_pushinteger(l, index);
1388 lua_replace(l, lua_upvalueindex(3));
1389
1390 return 1;
1391 });
1392}
