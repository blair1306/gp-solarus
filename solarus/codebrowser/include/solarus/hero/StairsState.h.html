17#ifndef SOLARUS_HERO_STAIRS_STATE_H
18#define SOLARUS_HERO_STAIRS_STATE_H
19
20#include "solarus/entities/Stairs.h"
21#include "solarus/hero/HeroState.h"
22#include <cstdint>
23#include <memory>
24
25namespace Solarus {
26
27/**
28 * \brief The state "Stairs" of the hero.
29 */
30class Hero::StairsState: public HeroState {
31
32 public:
33
34 StairsState(
35 Hero& hero,
36 const std::shared_ptr<const Stairs>& stairs,
37 Stairs::Way way
38 );
39
40 virtual void start(const State* previous_state) override;
41 virtual void stop(const State* next_state) override;
42 virtual void set_map(Map& map) override;
43 virtual void update() override;
44 virtual void set_suspended(bool suspended) override;
45
46 virtual bool is_touching_ground() const override;
47 virtual bool can_come_from_bad_ground() const override;
48 virtual bool is_teletransporter_delayed() const override;
49 virtual int get_wanted_movement_direction8() const override;
50 virtual std::shared_ptr<CarriedObject> get_carried_object() const override;
51 virtual CarriedObject::Behavior get_previous_carried_object_behavior() const override;
52 virtual void notify_layer_changed() override;
53
54 private:
55
56 std::shared_ptr<const Stairs>
57 stairs; /**< The stairs the hero is currently taking. */
58 Stairs::Way way; /**< indicates the way the hero is taking the stairs:
59 * - for stairs inside a single floor, NORMAL_WAY means that the hero is going upstairs
60 * and REVERSE_WAY means the contrary;
61 * - for stairs between two floors, NORMAL_WAY means that the hero is on the stairs
62 * of the old map and REVERSE_WAY means that he is on the stairs on the new map */
63 int phase; /**< when walking on stairs between two floors, indicates the current
64 * phase of the animations (0: not started, 1: initial animation,
65 * 2: diagonal animation, 3: final animation) */
66 uint32_t next_phase_date; /**< date when the stairs phase changes */
67 std::shared_ptr<CarriedObject>
68 carried_object; /**< an item carried by the hero while taking the stairs, or nullptr */
69
70};
71
72}
73
74#endif
