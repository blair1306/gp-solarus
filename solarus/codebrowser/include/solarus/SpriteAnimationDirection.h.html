17#ifndef SOLARUS_SPRITE_ANIMATION_DIRECTION
18#define SOLARUS_SPRITE_ANIMATION_DIRECTION
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/PixelBits.h"
22#include "solarus/lowlevel/Rectangle.h"
23#include "solarus/lowlevel/Debug.h"
24#include <vector>
25
26namespace Solarus {
27
28class Surface;
29
30/**
31 * \brief A sequence of frames representing a sprite animated in a particular direction.
32 *
33 * An animation direction is a sequence of frames representing a sprite
34 * with a specific animation and in a specific direction.
35 */
36class SpriteAnimationDirection {
37
38 public:
39
40 // creation and destruction
41 SpriteAnimationDirection(
42 const std::vector<Rectangle>& frames,
43 const Point& origin
44 );
45
46 // size and origin point
47 Size get_size() const;
48 const Point& get_origin() const;
49
50 // frames
51 int get_nb_frames() const;
52 const Rectangle& get_frame(int frame) const;
53 void draw(Surface& dst_surface, const Point& dst_position,
54 int current_frame, Surface& src_image);
55
56 // pixel collisions
57 void enable_pixel_collisions(Surface& src_image);
58 void disable_pixel_collisions();
59 bool are_pixel_collisions_enabled() const;
60 const PixelBits& get_pixel_bits(int frame) const;
61
62 private:
63
64 std::vector<Rectangle> frames; /**< position of each frame of the sequence on the image */
65 Point origin; /**< coordinates of the sprite's origin from the
66 * upper-left corner of its image. */
67
68 std::vector<PixelBits>
69 pixel_bits; /**< bit masks representing the non-transparent pixels of each frame,
70 * computed only if enable_pixel_collisions() is called */
71};
72
73/**
74 * \brief Returns the origin point of a frame.
75 * \return The origin point of a frame.
76 */
77inline const Point& SpriteAnimationDirection::get_origin() const {
78 return origin;
79}
80
81/**
82 * \brief Returns the pixel bits object of a frame.
83 *
84 * It represents the transparent bits of the frame and permits to detect
85 * pixel-precise collisions.
86 * The pixel collisions must be enabled.
87 *
88 * \param frame A frame of the animation.
89 * \return The pixel bits object of a frame.
90 */
91inline const PixelBits& SpriteAnimationDirection::get_pixel_bits(int frame) const {
92
93 SOLARUS_ASSERT(are_pixel_collisions_enabled(),
94 "Pixel-precise collisions are not enabled for this sprite");
95 SOLARUS_ASSERT(frame >= 0 && frame < get_nb_frames(), "Invalid frame number");
96
97 return pixel_bits[frame];
98}
99
100}
101
102#endif
