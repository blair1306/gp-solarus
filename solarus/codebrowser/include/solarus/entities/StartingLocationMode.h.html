17#ifndef SOLARUS_STARTING_LOCATION_MODE_H
18#define SOLARUS_STARTING_LOCATION_MODE_H
19
20#include "solarus/Common.h"
21#include "solarus/EnumInfo.h"
22
23namespace Solarus {
24
25/**
26 * \brief Whether a destination updates the starting location when using it.
27 */
28enum class StartingLocationMode {
29
30 WHEN_WORLD_CHANGES, /**< Update if the current world has just changed. */
31 YES, /**< Update the starting location. */
32 NO /**< Don't update the starting location. */
33};
34
35template <>
36struct SOLARUS_API EnumInfoTraits<StartingLocationMode> {
37 static const std::string pretty_name;
38
39 static const EnumInfo<StartingLocationMode>::names_type names;
40};
41
42}

44#endif
