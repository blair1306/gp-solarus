17#ifndef SOLARUS_POINT_H
18#define SOLARUS_POINT_H
19
20#include "solarus/Common.h"
21#include <iosfwd>
22
23namespace Solarus {
24
25/**
26 * \brief Represents a 2D point.
27 *
28 * This class represents a point with its two integer coordinates, x and y.
29 */
30class Point {
31
32 public:
33
34 ////////////////////////////////////////////////////////////
35 // Constructors
36
37 constexpr Point();
38 constexpr Point(int x, int y);
39
40 ////////////////////////////////////////////////////////////
41 // Operators
42
43 Point& operator+=(const Point& other);
44 Point& operator-=(const Point& other);
45
46 Point& operator*=(int factor);
47 Point& operator/=(int divisor);
48
49 ////////////////////////////////////////////////////////////
50 // Member data
51
52 int x = 0; /**< First coordinate */
53 int y = 0; /**< Second coordinate */
54
55};
56
57////////////////////////////////////////////////////////////
58// Free functions
59
60// Comparison operators
61constexpr bool operator==(const Point& lhs, const Point& rhs);
62constexpr bool operator!=(const Point& lhs, const Point& rhs);
63
64// Unary arithmetic operators
65constexpr Point operator+(const Point& point);
66constexpr Point operator-(const Point& point);
67
68// Binary arithmetic operators
69constexpr Point operator+(const Point& lhs, const Point& rhs);
70constexpr Point operator-(const Point& lhs, const Point& rhs);
71
72constexpr Point operator*(const Point& point, int factor);
73constexpr Point operator*(int factor, const Point& point);
74constexpr Point operator/(const Point& point, int divisor);
75
76// Output to stream
77std::ostream& operator<<(std::ostream& stream, const Point& point);
78
79}
