64
65 inline int &rx();

68 inline QPoint &operator+=(const QPoint &p);
69 inline QPoint &operator-=(const QPoint &p);
70
71 inline QPoint &operator*=(float factor);

114Q_DECL_CONSTEXPR inline QPoint::QPoint() : xp(0), yp(0) {}
115
116Q_DECL_CONSTEXPR inline QPoint::QPoint(int xpos, int ypos) : xp(xpos), yp(ypos) {}
117
118Q_DECL_CONSTEXPR inline bool QPoint::isNull() const
119{ return xp == 0 && yp == 0; }
120
121Q_DECL_CONSTEXPR inline int QPoint::x() const
122{ return xp; }
123
124Q_DECL_CONSTEXPR inline int QPoint::y() const
125{ return yp; }
126
127inline void QPoint::setX(int xpos)
128{ xp = xpos; }
129
130inline void QPoint::setY(int ypos)
131{ yp = ypos; }
132
133inline int Q_DECL_CONSTEXPR QPoint::manhattanLength() const
134{ return qAbs(x())+qAbs(y()); }
135
136inline int &QPoint::rx()
137{ return xp; }
138
139inline int &QPoint::ry()
140{ return yp; }
141
142inline QPoint &QPoint::operator+=(const QPoint &p)
143{ xp+=p.xp; yp+=p.yp; return *this; }
144
145inline QPoint &QPoint::operator-=(const QPoint &p)
146{ xp-=p.xp; yp-=p.yp; return *this; }
147
148inline QPoint &QPoint::operator*=(float factor)
149{ xp = qRound(xp*factor); yp = qRound(yp*factor); return *this; }
150
151inline QPoint &QPoint::operator*=(double factor)
152{ xp = qRound(xp*factor); yp = qRound(yp*factor); return *this; }
153
154inline QPoint &QPoint::operator*=(int factor)
155{ xp = xp*factor; yp = yp*factor; return *this; }
156
157Q_DECL_CONSTEXPR inline bool operator==(const QPoint &p1, const QPoint &p2)
158{ return p1.xp == p2.xp && p1.yp == p2.yp; }
159
160Q_DECL_CONSTEXPR inline bool operator!=(const QPoint &p1, const QPoint &p2)
161{ return p1.xp != p2.xp || p1.yp != p2.yp; }
162
163Q_DECL_CONSTEXPR inline const QPoint operator+(const QPoint &p1, const QPoint &p2)
164{ return QPoint(p1.xp+p2.xp, p1.yp+p2.yp); }
165
166Q_DECL_CONSTEXPR inline const QPoint operator-(const QPoint &p1, const QPoint &p2)
167{ return QPoint(p1.xp-p2.xp, p1.yp-p2.yp); }
168
169Q_DECL_CONSTEXPR inline const QPoint operator*(const QPoint &p, float factor)
170{ return QPoint(qRound(p.xp*factor), qRound(p.yp*factor)); }
171
172Q_DECL_CONSTEXPR inline const QPoint operator*(const QPoint &p, double factor)
173{ return QPoint(qRound(p.xp*factor), qRound(p.yp*factor)); }
174
175Q_DECL_CONSTEXPR inline const QPoint operator*(const QPoint &p, int factor)
176{ return QPoint(p.xp*factor, p.yp*factor); }
177
178Q_DECL_CONSTEXPR inline const QPoint operator*(float factor, const QPoint &p)
179{ return QPoint(qRound(p.xp*factor), qRound(p.yp*factor)); }
180
181Q_DECL_CONSTEXPR inline const QPoint operator*(double factor, const QPoint &p)
182{ return QPoint(qRound(p.xp*factor), qRound(p.yp*factor)); }
183
184Q_DECL_CONSTEXPR inline const QPoint operator*(int factor, const QPoint &p)
185{ return QPoint(p.xp*factor, p.yp*factor); }
186
187Q_DECL_CONSTEXPR inline const QPoint operator+(const QPoint &p)
188{ return p; }
189
190Q_DECL_CONSTEXPR inline const QPoint operator-(const QPoint &p)
191{ return QPoint(-p.xp, -p.yp); }
192
193inline QPoint &QPoint::operator/=(qreal c)
194{
195 xp = qRound(xp/c);
196 yp = qRound(yp/c);
197 return *this;
198}
199
200Q_DECL_CONSTEXPR inline const QPoint operator/(const QPoint &p, qreal c)
201{
202 return QPoint(qRound(p.xp/c), qRound(p.yp/c));
203}

297 return yp;

300inline void QPointF::setX(qreal xpos)

302 xp = xpos;

305inline void QPointF::setY(qreal ypos)

307 yp = ypos;
308}
309
310inline qreal &QPointF::rx()

312 return xp;
313}

315inline qreal &QPointF::ry()
316{
317 return yp;

320inline QPointF &QPointF::operator+=(const QPointF &p)
321{
322 xp+=p.xp;
323 yp+=p.yp;
324 return *this;
325}
326
327inline QPointF &QPointF::operator-=(const QPointF &p)
328{
329 xp-=p.xp; yp-=p.yp; return *this;

332inline QPointF &QPointF::operator*=(qreal c)

334 xp*=c; yp*=c; return *this;

337Q_DECL_CONSTEXPR inline bool operator==(const QPointF &p1, const QPointF &p2)
338{
339 return qFuzzyIsNull(p1.xp - p2.xp) && qFuzzyIsNull(p1.yp - p2.yp);
340}
341
342Q_DECL_CONSTEXPR inline bool operator!=(const QPointF &p1, const QPointF &p2)
343{
344 return !qFuzzyIsNull(p1.xp - p2.xp) || !qFuzzyIsNull(p1.yp - p2.yp);
345}
346
347Q_DECL_CONSTEXPR inline const QPointF operator+(const QPointF &p1, const QPointF &p2)
348{
349 return QPointF(p1.xp+p2.xp, p1.yp+p2.yp);
350}
351
352Q_DECL_CONSTEXPR inline const QPointF operator-(const QPointF &p1, const QPointF &p2)
353{
354 return QPointF(p1.xp-p2.xp, p1.yp-p2.yp);
355}
356
357Q_DECL_CONSTEXPR inline const QPointF operator*(const QPointF &p, qreal c)
358{
359 return QPointF(p.xp*c, p.yp*c);
360}
361
362Q_DECL_CONSTEXPR inline const QPointF operator*(qreal c, const QPointF &p)
363{
364 return QPointF(p.xp*c, p.yp*c);
365}
366
367Q_DECL_CONSTEXPR inline const QPointF operator+(const QPointF &p)
368{
369 return p;
370}
371
372Q_DECL_CONSTEXPR inline const QPointF operator-(const QPointF &p)
373{
374 return QPointF(-p.xp, -p.yp);
375}
376
377inline QPointF &QPointF::operator/=(qreal divisor)
378{
379 xp/=divisor;
380 yp/=divisor;
381 return *this;
382}
383
384Q_DECL_CONSTEXPR inline const QPointF operator/(const QPointF &p, qreal divisor)
385{
386 return QPointF(p.xp/divisor, p.yp/divisor);
387}
388
389Q_DECL_CONSTEXPR inline QPoint QPointF::toPoint() const
390{
391 return QPoint(qRound(xp), qRound(yp));
392}
