55/**
56 * \brief Returns a property exists in this dialog.
57 * \param key Key of the property to check.
58 * \return true if a property exists with this key.
59 */
60bool DialogData::has_property(const std::string& key) const {
61
62 if (key.empty() || key == "text" || key == "id") {
63 return false;
64 }
65 return properties.find(key) != properties.end();
66}
67
68/**
69 * \brief Returns a custom properties of this dialog.
70 * \param key Key of the property to get. It must exist.
71 * \return The value of this property.
72 */
73const std::string& DialogData::get_property(const std::string& key) const {
74
75 Debug::check_assertion(has_property(key),
76 std::string("No such property: '") + key + "'");
77 return properties.at(key);
78}
79
80/**
81 * \brief Sets a custom property of this dialog.
82 * \param key Key of the property to set.
83 * \param value The value to set.
84 */
85void DialogData::set_property(const std::string& key, const std::string& value) {
86
87 if (key.empty() || key == "text" || key == "id") {
88 return;
89 }
90 properties[key] = value;
91}
92
93/**
94 * \brief Removes a custom property of this dialog.
95 * \param key Key of the property to remove.
96 * \return \c true in case of success.
97 */
98bool DialogData::remove_property(const std::string& key) {
99 return properties.erase(key) > 0;
100}

123/**
124 * \brief Returns whether there exists a dialog with the specified id.
125 * \param dialog_id The id to test.
126 * \return \c true if a dialog exists with this id.

128bool DialogResources::has_dialog(const std::string& dialog_id) const {

132/**
133 * \brief Returns the dialog with the specified id.
134 * \param dialog_id A dialog id. It must exist.
135 * \return The dialog with this id.
136 */
137const DialogData& DialogResources::get_dialog(
138 const std::string& dialog_id) const {
139
140 const auto& it = dialogs.find(dialog_id);
141 Debug::check_assertion(it != dialogs.end(),
142 std::string("No such dialog: '") + dialog_id + "'");
143
144 return it->second;
145}

147/**
148 * \brief Returns the dialog with the specified id.
149 *
150 * Non-const version.
151 *
152 * \param dialog_id A dialog id. It must exist.
153 * \return The dialog with this id.
154 */
155DialogData& DialogResources::get_dialog(
156 const std::string& dialog_id) {
157
158 const auto& it = dialogs.find(dialog_id);
159 Debug::check_assertion(it != dialogs.end(),
160 std::string("No such dialog: '") + dialog_id + "'");
161
162 return it->second;
163}

165/**
166 * \brief Adds a dialog.
167 * \param dialog_id Id of the new dialog.
168 * \param dialog The dialog to add.
169 * \return \c true in case of success.
170 */
171bool DialogResources::add_dialog(
172 const std::string& dialog_id, const DialogData& dialog) {
173
174 const auto& result = dialogs.emplace(dialog_id, dialog);
175 if (!result.second) {
176 // Insertion failed: the id already exists.
177 return false;
178 }
179
180 return true;
181}
182
183/**
184 * \brief Removes a dialog.
185 * \param dialog_id Id of the dialog to remove.
186 * \return \c true in case of success.
187 */
188bool DialogResources::remove_dialog(const std::string& dialog_id) {
189
190 return dialogs.erase(dialog_id) > 0;
191}
192
193/**
194 * \brief Changes the id of a dialog.
195 * \param old_dialog_id Old id of the dialog.
196 * \param new_dialog_id New id to set.
197 * \return \c true in case of success.
198 * In case of failure, the old dialog is unchanged.
199 */
200bool DialogResources::set_dialog_id(
201 const std::string& old_dialog_id, const std::string& new_dialog_id) {
202
203 if (!has_dialog(old_dialog_id)) {
204 // No dialog was found with the old id.
205 return false;
206 }
207
208 if (has_dialog(new_dialog_id)) {
209 // The new id is already used.
210 return false;
211 }
212
213 DialogData dialog = get_dialog(old_dialog_id);
214 remove_dialog(old_dialog_id);
215 add_dialog(new_dialog_id, dialog);
216
217 return true;
218}

220/**
221 * \brief Function called by Lua to add a dialog to the resource.
222 *
223 * - Argument 1 (table): properties of the dialog.
224 *
225 * \param l the Lua context that is calling this function
226 * \return Number of values to return to Lua.
227 */
228int DialogResources::l_dialog(lua_State* l) {
229
230 return LuaTools::exception_boundary_handle(l, [&] {
231 lua_getfield(l, LUA_REGISTRYINDEX, "dialogs");
232 DialogResources& dialogs = *static_cast<DialogResources*>(
233 lua_touserdata(l, -1));
234 lua_pop(l, 1);
235
236 LuaTools::check_type(l, 1, LUA_TTABLE);
237
238 std::string dialog_id = LuaTools::check_string_field(l, 1, "id");
239 std::string text = LuaTools::check_string_field(l, 1, "text");
240 text = unescape_multiline_string(text);
241 DialogData dialog;
242 dialog.set_text(text);
243
244 // Traverse the table, looking for custom properties.
245 lua_pushnil(l); // first key
246 while (lua_next(l, 1) != 0) {
247
248 const std::string& key = LuaTools::check_string(l, -2);
249 if (key == "id" || key == "text") {
250 lua_pop(l, 1);
251 continue;
252 }
253
254 // Custom property.
255 std::string value;
256 int type = lua_type(l, -1);
257 if (type == LUA_TSTRING || type == LUA_TNUMBER) {
258 value = lua_tostring(l, -1);
259 }
260 else if (type == LUA_TBOOLEAN) {
261 value = lua_toboolean(l, -1) ? "1" : "0";
262 }
263 else {
264 LuaTools::error(l, std::string("Invalid value '") + key + "' for dialog '"
265 + dialog_id + "'");
266 }
267 dialog.set_property(key, value);
268 lua_pop(l, 1);
269 }
270
271 if (dialog_id.empty()) {
272 LuaTools::error(l, "Missing value id");
273 }
274
275 if (dialogs.has_dialog(dialog_id)) {
276 LuaTools::error(l, std::string("Duplicate dialog '") + dialog_id + "'");
277 }
278
279 dialogs.add_dialog(dialog_id, dialog);
280
281 return 0;
282 });
283}

285/**
286 * \copydoc LuaData::import_from_lua
287 */
288bool DialogResources::import_from_lua(lua_State* l) {
289
290 lua_pushlightuserdata(l, this);
291 lua_setfield(l, LUA_REGISTRYINDEX, "dialogs");
292 lua_register(l, "dialog", l_dialog);
293 if (lua_pcall(l, 0, 0, 0) != 0) {
294 Debug::error(std::string("Error in dialog data file: ")
295 + lua_tostring(l, -1));
296 lua_pop(l, 1);
297 return false;
298 }
299
300 return true;
301}
302
303/**
304 * \copydoc LuaData::export_to_lua
305 */
306bool DialogResources::export_to_lua(std::ostream& out) const {
307
308 for (const auto kvp : dialogs) {
309 const std::string& id = kvp.first;
310 const DialogData& dialog = kvp.second;
311
312 out << "dialog{\n id = \"" << escape_string(id) << "\",\n";
313 for (const auto pkvp : dialog.get_properties()) {
314 out << " " << pkvp.first << " = \"" << pkvp.second << "\",\n";
315 }
316 const std::string& text = dialog.get_text();
317 out << " text = [[\n" << escape_multiline_string(text);
318 if (!text.empty() && text[text.size() - 1] != '\n') {
319 // Make sure that the closing ]] is always on a new line.
320 out << '\n';
321 }
322 out << "]]\n}\n\n";
323 }
324 return true;
325}
