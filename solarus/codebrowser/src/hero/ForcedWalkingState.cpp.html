23/**
24 * \brief Constructor.
25 * \param hero the hero controlled by this state
26 * \param path the succession of basic moves
27 * composing this movement (each character represents
28 * a direction between '0' and '7'), as in PathMovement
29 * \param loop true to make the movement return to the beginning
30 * once finished
31 * \param ignore_obstacles true to make the movement ignore obstacles
32 */
33Hero::ForcedWalkingState::ForcedWalkingState(
34 Hero& hero,
35 const std::string& path,
36 bool loop,
37 bool ignore_obstacles):
38
39 HeroState(hero, "forced walking"),
40 movement(std::make_shared<PathMovement>(
41 path, hero.get_walking_speed(), loop, ignore_obstacles, false
42 )) {
43
44}
45
46/**
47 * \brief Starts this state.
48 * \param previous_state the previous state
49 */
50void Hero::ForcedWalkingState::start(const State* previous_state) {
51
52 HeroState::start(previous_state);
53
54 // update the sprites
55 HeroSprites& sprites = get_sprites();
56 sprites.set_animation_walking_normal();
57
58 // walk
59 get_entity().set_movement(movement);
60}
61
62/**
63 * \brief Stops this state.
64 * \param next_state the next state
65 */
66void Hero::ForcedWalkingState::stop(const State* next_state) {
67
68 HeroState::stop(next_state);
69
70 get_entity().clear_movement();
71}
72
73/**
74 * \brief Updates this state.
75 */
76void Hero::ForcedWalkingState::update() {
77
78 HeroState::update();
79
80 get_sprites().set_animation_direction(movement->get_displayed_direction4());
81
82 if (movement->is_finished()) {
83 get_entity().start_state_from_ground();
84 }
85}
