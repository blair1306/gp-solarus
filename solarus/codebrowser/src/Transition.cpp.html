35 * \brief Creates a transition effect.

53/**
54 * \brief Creates a transition effect with the specified type and direction.
55 * \param style style of the transition: Transition::IMMEDIATE, Transition::FADE, etc.
56 * \param direction Direction of the transition.
57 * \param dst_surface The surface that will receive the transition
58 * (used by some kinds of transitions).
59 * \param game The current game if any (used by some kinds of transitions).
60 * \return the transition created
61 */
62Transition* Transition::create(
63 Transition::Style style,
64 Transition::Direction direction,
65 Surface& dst_surface,
66 Game* game) {
67
68 Transition* transition = nullptr;
69
70 switch (style) {
71
72 case Style::IMMEDIATE:
73 transition = new TransitionImmediate(direction);
74 break;
75
76 case Style::FADE:
77 transition = new TransitionFade(direction, dst_surface);
78 break;
79
80 case Style::SCROLLING:
81 transition = new TransitionScrolling(direction);
82 break;
83 }
84
85 transition->game = game;
86
87 return transition;
88}

118/**
119 * \brief Indicates the surface that was shown during the closing transition
120 * that was played before this opening transition.
121 * \param previous_surface The previous surface or nullptr.
122 */
123void Transition::set_previous_surface(Surface* previous_surface) {
124
125 Debug::check_assertion(previous_surface == nullptr
126 || get_direction() != Direction::CLOSING,
127 "Cannot show a previous surface with an closing transition effect");
128
129 this->previous_surface = previous_surface;
130}

140/**
141 * \brief Returns whether this transition is currently suspended.
142 * \return true if this transition is suspended.
143 */
144bool Transition::is_suspended() const {
145 return suspended;
146}
147
148/**
149 * \brief Suspends or resumes this transition.
150 * \param suspended true to suspend it, false to resume it.
151 */
152void Transition::set_suspended(bool suspended) {
153
154 if (suspended != this->suspended) {
155 this->suspended = suspended;
156 if (suspended) {
157 when_suspended = System::now();
158 }
159 notify_suspended(suspended);
160 }
161}
162
163/**
164 * \brief Returns the date when this transition was suspended if it is.
165 * \return The date when this transition was suspended or 0.
166 */
167uint32_t Transition::get_when_suspended() const {
168 return when_suspended;
169}
