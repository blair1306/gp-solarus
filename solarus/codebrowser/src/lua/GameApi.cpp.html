239/**
240 * \brief Implementation of game:start().
241 * \param l The Lua context that is calling this function.
242 * \return Number of values to return to Lua.
243 */
244int LuaContext::game_api_start(lua_State* l) {
245
246 return LuaTools::exception_boundary_handle(l, [&] {
247 std::shared_ptr<Savegame> savegame = check_game(l, 1);
248
249 if (CurrentQuest::get_resources(ResourceType::MAP).empty()) {
250 LuaTools::error(l, "Cannot start game: there is no map in this quest");
251 }
252
253 Game* game = savegame->get_game();
254 if (game != nullptr) {
255 // A game is already running with this savegame: restart it.
256 game->restart();
257 }
258 else {
259 // Create a new game to run.
260 MainLoop& main_loop = savegame->get_lua_context().get_main_loop();
261 if (main_loop.get_game() != nullptr) {
262 // Stop any previous game.
263 main_loop.get_game()->stop();
264 }
265 Game* game = new Game(main_loop, savegame);
266 main_loop.set_game(game);
267 }
268
269 return 0;
270 });
271}

426/**
427 * \brief Implementation of game:start_dialog().
428 * \param l The Lua context that is calling this function.
429 * \return Number of values to return to Lua.
430 */
431int LuaContext::game_api_start_dialog(lua_State* l) {
432
433 return LuaTools::exception_boundary_handle(l, [&] {
434 Savegame& savegame = *check_game(l, 1);
435 const std::string& dialog_id = LuaTools::check_string(l, 2);
436 ScopedLuaRef info_ref;
437 ScopedLuaRef callback_ref;
438
439 if (!CurrentQuest::dialog_exists(dialog_id)) {
440 LuaTools::arg_error(l, 2, std::string("No such dialog: '") + dialog_id + "'");
441 }
442
443 Game* game = savegame.get_game();
444 if (game == nullptr) {
445 LuaTools::error(l, "Cannot start dialog: this game is not running");
446 }
447
448 if (game->is_dialog_enabled()) {
449 LuaTools::error(l, "Cannot start dialog: another dialog is already active");
450 }
451
452 if (lua_gettop(l) >= 3) {
453
454 LuaContext& lua_context = get_lua_context(l);
455 int callback_index = 3;
456 if (lua_type(l, 3) != LUA_TFUNCTION) {
457 // There is an info parameter.
458 lua_pushvalue(l, 3);
459 info_ref = lua_context.create_ref();
460 ++callback_index;
461 }
462
463 callback_ref = LuaTools::opt_function(l, callback_index);
464 }
465 game->start_dialog(dialog_id, info_ref, callback_ref);
466
467 return 0;
468 });
469}

643/**
644 * \brief Implementation of game:set_value().
645 * \param l The Lua context that is calling this function.
646 * \return Number of values to return to Lua.
647 */
648int LuaContext::game_api_set_value(lua_State* l) {
649
650 return LuaTools::exception_boundary_handle(l, [&] {
651 Savegame& savegame = *check_game(l, 1);
652 const std::string& key = LuaTools::check_string(l, 2);
653
654 if (key[0] == '_') {
655 LuaTools::arg_error(l, 3,
656 std::string("Invalid savegame variable '") + key
657 + "': names prefixed by '_' are reserved for built-in variables");
658 }
659
660 if (!LuaTools::is_valid_lua_identifier(key)) {
661 LuaTools::arg_error(l, 3,
662 std::string("Invalid savegame variable '") + key
663 + "': the name should only contain alphanumeric characters or '_'"
664 + " and cannot start with a digit");
665 }
666
667 switch (lua_type(l, 3)) {
668
669 case LUA_TBOOLEAN:
670 savegame.set_boolean(key, lua_toboolean(l, 3));
671 break;
672
673 case LUA_TNUMBER:
674 savegame.set_integer(key, int(lua_tointeger(l, 3)));
675 break;
676
677 case LUA_TSTRING:
678 savegame.set_string(key, lua_tostring(l, 3));
679 break;
680
681 case LUA_TNIL:
682 savegame.unset(key);
683 break;
684
685 default:
686 LuaTools::arg_error(l, 3,
687 std::string("Expected string, number or boolean, got ")
688 + luaL_typename(l, 3)
689 );
690 }
691
692 return 0;
693 });
694}

697 * \brief Implementation of game:get_starting_location().
698 * \param l The Lua context that is calling this function.
699 * \return Number of values to return to Lua.
700 */
701int LuaContext::game_api_get_starting_location(lua_State* l) {
702
703 return LuaTools::exception_boundary_handle(l, [&] {
704 Savegame& savegame = *check_game(l, 1);
705
706 const std::string& starting_map = savegame.get_string(Savegame::KEY_STARTING_MAP);
707 const std::string& starting_point = savegame.get_string(Savegame::KEY_STARTING_POINT);
708
709 if (starting_map.empty()) {
710 lua_pushnil(l);
711 }
712 else {
713 push_string(l, savegame.get_string(Savegame::KEY_STARTING_MAP));
714 }
715 if (starting_point.empty()) {
