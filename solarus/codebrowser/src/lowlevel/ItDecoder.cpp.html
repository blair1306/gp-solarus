24/**
25 * \brief Creates an Impulse Tracker decoder.
26 */
27ItDecoder::ItDecoder():
28 modplug_file(nullptr) {
29
30 ModPlug_Settings settings;
31 ModPlug_GetSettings(&settings);
32 settings.mChannels = 2; // stereo
33 settings.mBits = 16; // 16 bits
34 settings.mLoopCount = -1; // Loop forever by default.
35 ModPlug_SetSettings(&settings);
36}
37
38/**
39 * \brief Loads an IT file from memory.
40 * \param sound_buffer The memory area to read
41 */
42void ItDecoder::load(const std::string& sound_buffer) {
43
44 Debug::check_assertion(modplug_file == nullptr,
45 "IT data is already loaded"
46 );
47
48 // Load the IT data into the IT library.
49 modplug_file = ModPlugFileUniquePtr(
50 ModPlug_Load((const void*) sound_buffer.data(), (int) sound_buffer.size())
51 );
52}
53
54/**
55 * \brief Unloads the IT file previously loaded.
56 */
57void ItDecoder::unload() {
58
59 Debug::check_assertion(modplug_file != nullptr,
60 "IT data is not loaded"
61 );
62
63 modplug_file = nullptr;
64}
65
66/**
67 * \brief Decodes a chunk of the previously loaded IT data into PCM data.
68 * \param decoded_data Pointer to where you want the decoded data to be written.
69 * \param nb_samples Number of samples to write.
70 * \return The number of bytes read, or 0 if the end is reached.
71 */
72int ItDecoder::decode(void* decoded_data, int nb_samples) {
73
74 // Decode from the IT data the specified number of PCM samples.
75 return ModPlug_Read(modplug_file.get(), decoded_data, nb_samples);
76}

78/**
79 * \brief Returns the number of channels in this music.
80 * \return The number of channels.
81 */
82int ItDecoder::get_num_channels() const {
83 return ModPlug_NumChannels(modplug_file.get());
84}
85
86/**
87 * \brief Returns the volume of a channel.
88 * \param channel A channel index.
89 * \return The volume of this channel.
90 */
91int ItDecoder::get_channel_volume(int channel) const {
92
93 const int num_patterns = ModPlug_NumPatterns(modplug_file.get());
94
95 Debug::check_assertion(channel >= 0 && channel < get_num_channels(),
96 "Invalid channel number");
97
98 if (num_patterns == 0) {
99 return 0;
100 }
101
102 unsigned int num_rows = 0;
103 ModPlugNote* notes = ModPlug_GetPattern(modplug_file.get(), 0, &num_rows);
104
105 if (num_rows == 0) {
106 return 0;
107 }
108
109 return notes[0].Volume;
110}
111
112/**
113 * \brief Sets the volume of a channel.
114 * \param channel A channel index.
115 * \param volume The volume to set.
116 */
117void ItDecoder::set_channel_volume(int channel, int volume) {
118
119 const unsigned int num_channels = get_num_channels();
120 const unsigned int num_patterns = ModPlug_NumPatterns(modplug_file.get());
121
122 for (unsigned int pattern = 0; pattern < num_patterns; ++pattern) {
123 unsigned int num_rows;
124 ModPlugNote* notes = ModPlug_GetPattern(modplug_file.get(), pattern, &num_rows);
125 for (unsigned int j = channel; j < num_rows * num_channels; j += num_channels) {
126 notes[j].Volume = volume;
127 }
128 }
129}

131/**
132 * \brief Returns the tempo of the music.
133 * \return The tempo.
134 */
135int ItDecoder::get_tempo() const {
136
137 return reinterpret_cast<CSoundFile*>(modplug_file.get())->GetMusicTempo();
138}

145
146 reinterpret_cast<CSoundFile*>(modplug_file.get())->SetTempo(tempo);
147}

149/**
150 * \brief Returns whether the decoder loops when reaching the end.
151 */
152bool ItDecoder::loops() const {
153
154 ModPlug_Settings settings;
155 ModPlug_GetSettings(&settings);
156 return settings.mLoopCount == -1; // -1 means looping forever.
157}
158
159/**
160 * \brief Sets whether the decoder should loop when reaching the end.
161 * \param loops \c true to make the decoder loop.
162 */
163void ItDecoder::set_loops(bool loops) {
164
165 ModPlug_Settings settings;
166 ModPlug_GetSettings(&settings);
167 settings.mLoopCount = loops ? -1 : 0;
168 ModPlug_SetSettings(&settings);
169}
