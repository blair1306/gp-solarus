185/******* inlines *****/
186inline QTransform::TransformationType QTransform::inline_type() const
187{
188 if (m_dirty == TxNone)
189 return static_cast<TransformationType>(m_type);
190 return type();
191}
192
193inline bool QTransform::isAffine() const
194{
195 return inline_type() < TxProject;
196}
197inline bool QTransform::isIdentity() const
198{
199 return inline_type() == TxNone;
200}
201
202inline bool QTransform::isInvertible() const
203{
204 return !qFuzzyIsNull(determinant());
205}
206
207inline bool QTransform::isScaling() const
208{
209 return type() >= TxScale;
210}
211inline bool QTransform::isRotating() const
212{
213 return inline_type() >= TxRotate;
214}
215
216inline bool QTransform::isTranslating() const
217{
218 return inline_type() >= TxTranslate;
219}
220
221inline qreal QTransform::determinant() const
222{
223 return affine._m11*(m_33*affine._m22-affine._dy*m_23) -
224 affine._m21*(m_33*affine._m12-affine._dy*m_13)+affine._dx*(m_23*affine._m12-affine._m22*m_13);
225}
226inline qreal QTransform::det() const
227{
228 return determinant();
229}

275inline QTransform &QTransform::operator*=(qreal num)
276{
277 if (num == 1.)
278 return *this;
279 affine._m11 *= num;
280 affine._m12 *= num;
281 m_13 *= num;
282 affine._m21 *= num;
283 affine._m22 *= num;
284 m_23 *= num;
285 affine._dx *= num;
286 affine._dy *= num;
287 m_33 *= num;
288 if (m_dirty < TxScale)
289 m_dirty = TxScale;
290 return *this;
291}
292inline QTransform &QTransform::operator/=(qreal div)
293{
294 if (div == 0)
295 return *this;
296 div = 1/div;
297 return operator*=(div);
298}
299inline QTransform &QTransform::operator+=(qreal num)
300{
301 if (num == 0)
302 return *this;
303 affine._m11 += num;
304 affine._m12 += num;
305 m_13 += num;
306 affine._m21 += num;
307 affine._m22 += num;
308 m_23 += num;
309 affine._dx += num;
310 affine._dy += num;
311 m_33 += num;
312 m_dirty = TxProject;
313 return *this;
314}
315inline QTransform &QTransform::operator-=(qreal num)
316{
317 if (num == 0)
318 return *this;
319 affine._m11 -= num;
320 affine._m12 -= num;
321 m_13 -= num;
322 affine._m21 -= num;
323 affine._m22 -= num;
324 m_23 -= num;
325 affine._dx -= num;
326 affine._dy -= num;
327 m_33 -= num;
328 m_dirty = TxProject;
329 return *this;
330}

332inline bool qFuzzyCompare(const QTransform& t1, const QTransform& t2)
333{
334 return qFuzzyCompare(t1.m11(), t2.m11())
335 && qFuzzyCompare(t1.m12(), t2.m12())
336 && qFuzzyCompare(t1.m13(), t2.m13())
337 && qFuzzyCompare(t1.m21(), t2.m21())
338 && qFuzzyCompare(t1.m22(), t2.m22())
339 && qFuzzyCompare(t1.m23(), t2.m23())
340 && qFuzzyCompare(t1.m31(), t2.m31())
341 && qFuzzyCompare(t1.m32(), t2.m32())
342 && qFuzzyCompare(t1.m33(), t2.m33());
343}
