17#include "solarus/lowlevel/Debug.h"
18#include "solarus/movements/PixelMovement.h"
19#include "test_tools/TestEnvironment.h"
20
21using namespace Solarus;
22
23namespace {
24
25/**
26 * \brief Tests that a simple movement results in the expected coordinates.
27 */
28void basic_test(TestEnvironment& env) {
29
30 PixelMovement m("1 -2 2 1", 50, false, false);
31
32 while (!m.is_finished()) {
33
34 m.update();
35 env.step();
36 }
37
38 Debug::check_assertion(m.get_x() == 3 && m.get_y() == -1,
39 "Unexpected coordinates for 'basic_test'");
40}
41
42/**
43 * \brief Tests that a pixel movement correctly loops as wanted.
44 */
45void loop_test(TestEnvironment& env) {
46
47 uint32_t start_date = env.now();
48 PixelMovement m("1 -2", 50, true, true);
49
50 while (!m.is_finished() && env.now() < start_date + 200) {
51
52 m.update();
53 env.step();
54 }
55
56 Debug::check_assertion(m.get_x() == 3 && m.get_y() == -6,
57 "Unexpected coordinates for 'loop_test'");
58}
59
60/**
61 * \brief Tests that a pixel movement with empty trajectory is always finished.
62 */
63void empty_test(TestEnvironment& /* env */) {
64
65 PixelMovement m("", 50, true, true);
66
67 Debug::check_assertion(m.is_finished(),
68 "Movement is not finished as expected in 'empty_test'");
69}
70
71/**
72 * \brief Tests restarting a pixel movement after use.
73 */
74void restart_test(TestEnvironment& env) {
75
76 PixelMovement m("2 1", 50, false, false);
77
78 while (!m.is_finished()) {
79
80 m.update();
81 env.step();
82 }
83
84 Debug::check_assertion(m.get_x() == 2 && m.get_y() == 1,
85 "Unexpected coordinates for 'restart_test #1'");
86
87 m.set_trajectory("0 2");
88
89 while (!m.is_finished()) {
90
91 m.update();
92 env.step();
93 }
94
95 Debug::check_assertion(m.get_x() == 2 && m.get_y() == 3,
96 "Unexpected coordinates for 'restart_test #2'");
97}
98
99/**
100 * \brief Tests a pixel movement whose trajectory is specified as a list.
101 */
102void list_test(TestEnvironment& env) {
103
104 std::list<Point> trajectory;
105 trajectory.push_back({ 3, 2 });
106 trajectory.push_back({ -4, -5 });
107
108 PixelMovement m("", 50, false, false);
109 m.set_trajectory(trajectory);
110
111 while (!m.is_finished()) {
112
113 m.update();
114 env.step();
115 }
116
117 Debug::check_assertion(m.get_x() == -1 && m.get_y() == -3,
118 "Unexpected coordinates for 'list_test'");
119}
120
121}
122
123/**
124 * \brief Tests for the pixel movement.
125 */
126int main(int argc, char** argv) {
127
128 TestEnvironment env(argc, argv);
129
130 basic_test(env);
131 loop_test(env);
132 empty_test(env);
133 restart_test(env);
134 list_test(env);
135
136 return 0;
137}
