34/**
35 * \brief Creates a text to draw with the default properties.
36 *
37 * The default properties are:
38 * - font: the default font defined in file text/fonts.dat
39 * - horizontal alignment: left
40 * - vertical alignment: middle
41 * - rendering mode: solid
42 * - text color: white
43 *
44 * \param x x position of the text on the destination surface
45 * \param y y position of the text on the destination surface
46 */
47TextSurface::TextSurface(int x, int y):
48 TextSurface(x, y, HorizontalAlignment::LEFT, VerticalAlignment::MIDDLE) {
49}
50
51/**
52 * \brief Creates a text to draw with the specified alignment properties.
53 * \param x x position of the text on the destination surface
54 * \param y y position of the text on the destination surface
55 * \param horizontal_alignment horizontal alignment of the text: ALIGN_LEFT,
56 * ALIGN_CENTER or ALIGN_RIGHT
57 * \param vertical_alignment vertical alignment of the text: ALIGN_TOP,
58 * ALIGN_MIDDLE or ALIGN_BOTTOM
59 */
60TextSurface::TextSurface(int x, int y,
61 TextSurface::HorizontalAlignment horizontal_alignment,
62 TextSurface::VerticalAlignment vertical_alignment):
63 Drawable(),
64 font_id(FontResource::get_default_font_id()),
65 horizontal_alignment(horizontal_alignment),
66 vertical_alignment(vertical_alignment),
67 rendering_mode(RenderingMode::SOLID),
68 text_color(Color::white),
69 font_size(11),
70 x(x),
71 y(y),
72 surface(nullptr),
73 text() {
74
75 if (font_id.empty()) {
76 Debug::error("This quest has no fonts");
77 }
78}

382/**
383 * \brief Redraws the text surface.
384 *
385 * This function is called when there is a change.
386 */
387void TextSurface::rebuild() {
388
389 surface = nullptr;
390
391 if (font_id.empty()) {
392 return;
393 }
394
395 if (is_empty()) {
396 // Empty string or only whitespaces: no surface to create.
397 // Some fonts make TTF_Font fail if the string contains only whitespaces.
398 return;
399 }
400
401 Debug::check_assertion(FontResource::exists(font_id),
402 std::string("No such font: '") + font_id + "'"
403 );
404
405 if (FontResource::is_bitmap_font(font_id)) {
406 rebuild_bitmap();
407 }
408 else {
409 rebuild_ttf();
410 }
411
412 // calculate the coordinates of the top-left corner
413 int x_left = 0, y_top = 0;
414
415 switch (horizontal_alignment) {
416
417 case HorizontalAlignment::LEFT:
418 x_left = x;
419 break;
420
421 case HorizontalAlignment::CENTER:
422 x_left = x - surface->get_width() / 2;
423 break;
424
425 case HorizontalAlignment::RIGHT:
426 x_left = x - surface->get_width();
427 break;
428 }
429
430 switch (vertical_alignment) {
431
432 case VerticalAlignment::TOP:
433 y_top = y;
434 break;
435
436 case VerticalAlignment::MIDDLE:
437 y_top = y - surface->get_height() / 2;
438 break;
439
440 case VerticalAlignment::BOTTOM:
441 y_top = y - surface->get_height();
442 break;
443 }
444
445 text_position = { x_left, y_top };
446}

448/**
449 * \brief Redraws the text surface in the case of a bitmap font.
450 *
451 * This function is called when there is a change.
452 */
453void TextSurface::rebuild_bitmap() {
454
455 // First count the number of characters in the UTF-8 string.
456 int num_chars = 0;
457 for (unsigned i = 0; i < text.size(); i++) {
458 char current_char = text[i];
459 if ((current_char & 0xE0) == 0xC0) {
460 // This character uses two bytes.
461 ++i;
462 }
463 ++num_chars;
464 }
465
466 // Determine the letter size from the surface size.
467 const SurfacePtr& bitmap = FontResource::get_bitmap_font(font_id);
468 const Size& bitmap_size = bitmap->get_size();
469 int char_width = bitmap_size.width / 128;
470 int char_height = bitmap_size.height / 16;
471
472 surface = Surface::create((char_width - 1) * num_chars + 1, char_height);
473
474 // Traverse the string again to draw the characters.
475 Point dst_position;
476 for (unsigned i = 0; i < text.size(); i++) {
477 char first_byte = text[i];
478 Rectangle src_position(0, 0, char_width, char_height);
479 if ((first_byte & 0xE0) != 0xC0) {
480 // This character uses one byte.
481 src_position.set_xy(first_byte * char_width, 0);
482 }
483 else {
484 // This character uses two bytes.
485 ++i;
486 char second_byte = text[i];
487 uint16_t code_point = ((first_byte & 0x1F) << 6) | (second_byte & 0x3F);
488 src_position.set_xy((code_point % 128) * char_width,
489 (code_point / 128) * char_height);
490 }
491 bitmap->draw_region(src_position, surface, dst_position);
492 dst_position.x += char_width - 1;
493 }
494}

496/**
497 * \brief Redraws the text surface in the case of a normal font.
498 *
499 * This function is called when there is a change.
500 */
501void TextSurface::rebuild_ttf() {
502
503 // create the text surface
504
505 SDL_Surface* internal_surface = nullptr;
506 TTF_Font& internal_font = FontResource::get_outline_font(font_id, font_size);
507 SDL_Color internal_color;
508 text_color.get_components(
509 internal_color.r, internal_color.g, internal_color.b, internal_color.a);
510
511 switch (rendering_mode) {
512
513 case RenderingMode::SOLID:
514 internal_surface = TTF_RenderUTF8_Solid(&internal_font, text.c_str(), internal_color);
515 break;
516
517 case RenderingMode::ANTIALIASING:
518 internal_surface = TTF_RenderUTF8_Blended(&internal_font, text.c_str(), internal_color);
519 break;
520 }
521
522 Debug::check_assertion(internal_surface != nullptr,
523 std::string("Cannot create the text surface for string '") + text + "': "
524 + SDL_GetError()
525 );
526
527 surface = std::make_shared<Surface>(internal_surface);
528}

530/**
531 * \brief Draws the text on a surface.
532 *
533 * This method blits the text at the position previously set
534 * (by set_x(), set_y() or set_position())
535 * plus the parameter dst_position.
536 *
537 * \param dst_surface The destination surface.
538 * \param dst_position Coordinates on the destination surface.
539 */
540void TextSurface::raw_draw(Surface& dst_surface,
541 const Point& dst_position) {
542
543 if (surface != nullptr) {
544 surface->set_blend_mode(get_blend_mode());
545 surface->raw_draw(dst_surface, dst_position + text_position);
546 }
547}
548
549/**
550 * \brief Draws a subrectangle of this text surface on another surface.
551 * \param region The subrectangle to draw in this object.
552 * \param dst_surface The destination surface.
553 * \param dst_position Coordinates on the destination surface.
554 */
555void TextSurface::raw_draw_region(const Rectangle& region,
556 Surface& dst_surface, const Point& dst_position) {
557
558 if (surface != nullptr) {
559 surface->set_blend_mode(get_blend_mode());
560 surface->raw_draw_region(
561 region, dst_surface,
562 dst_position + text_position);
563 }
564}
