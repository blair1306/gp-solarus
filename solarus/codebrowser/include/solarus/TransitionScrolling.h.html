17#ifndef SOLARUS_TRANSITION_SCROLLING_H
18#define SOLARUS_TRANSITION_SCROLLING_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Rectangle.h"
22#include "solarus/lowlevel/SurfacePtr.h"
23#include "solarus/Transition.h"
24
25namespace Solarus {
26
27/**
28 * \brief Scrolling transition between two maps.
29 *
30 * This kind of transition can only be used for the transition between two maps.
31 * The new map is drawn gradually, with a scrolling effect, and the hero
32 * is drawn between both maps.
33 * Nothing happens during the out transition. During the in transition,
34 * the previous map does not exist anymore but its surface was saved
35 * and we can still draw it.
36 */
37class TransitionScrolling: public Transition {
38
39 public:
40
41 explicit TransitionScrolling(Direction direction);
42
43 virtual bool needs_previous_surface() const override;
44
45 virtual void start() override;
46 virtual bool is_started() const override;
47 virtual bool is_finished() const override;
48 virtual void notify_suspended(bool suspended) override;
49 virtual void update() override;
50 virtual void draw(Surface& dst_surface) override;
51
52 private:
53
54 void scroll();
55 Rectangle get_previous_map_dst_position(int scrolling_direction);
56
57 SurfacePtr both_maps_surface; /**< an intermediate surface where the two map surfaces will be blitted */
58 int scrolling_direction; /**< direction of the scrolling (0 to 3) */
59 uint32_t next_scroll_date; /**< date of the next scrolling step */
60
61 int dx; /**< x change for each scrolling step */
62 int dy; /**< y change for each scrolling step */
63
64 Rectangle previous_map_dst_position; /**< position where the previous map is blitted on both_maps_surface */
65 Rectangle current_map_dst_position; /**< position where the current map is blitted on both_maps_surface */
66 Rectangle current_scrolling_position; /**< the rectangle from both_maps_surface that is currently visible */
67
68};
69
70}
71
72#endif
