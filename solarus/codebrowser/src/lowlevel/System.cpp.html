33uint32_t System::initial_time = 0;
34uint32_t System::ticks = 0;
35
36/**
37 * \brief Initializes the basic low-level system.
38 *
39 * Initializes the audio system, the video system,
40 * the data file system, etc.
41 *
42 * \param args Command-line arguments.
43 */
44void System::initialize(const Arguments& args) {
45
46 // initialize SDL
47 SDL_Init(SDL_INIT_VIDEO | SDL_INIT_JOYSTICK);
48 initial_time = get_real_time();
49 ticks = 0;
50
51 // audio
52 Sound::initialize(args);
53
54 // input
55 InputEvent::initialize();
56
57 // random number generator
58 Random::initialize();
59
60 // video
61 Video::initialize(args);
62 FontResource::initialize();
63 Sprite::initialize();
64}
65
66/**
67 * \brief Closes the low-level system.
68 *
69 * This closes all initializations made in initialize().
70 */
71void System::quit() {
72
73 Random::quit();
74 InputEvent::quit();
75 Sound::quit();
76 Sprite::quit();
77 FontResource::quit();
78 Video::quit();
79
80 SDL_Quit();
81}
82
83/**
84 * \brief This function is called repeatedly by the main loop.
85 *
86 * It calls the update function of low-level systems that need it.
87 */
88void System::update() {
89
90 // Use a constant timestep here to have deterministic updates.
91 ticks += timestep;
92 Sound::update();
93}
94
95/**
96 * \brief Returns the name of the running OS.
97 *
98 * Possible values are "Windows", "Mac OS X", "Linux", "iOS" and "Android".
99 *
100 * If the correct OS name is not available, returns a string
101 * beginning with the text "Unknown".
102 *
103 * \return the name of the running OS.
104 */
105std::string System::get_os() {
106
107 return SDL_GetPlatform();
108}
109
110/**
111 * \brief Returns the number of simulated milliseconds elapsed since the
112 * main loop started.
113 *
114 * Follows to the real time unless the system is too slow to play at
115 * normal speed.
116 *
117 * \return The number of simulated milliseconds elapsed since the
118 * initialization.
119 */
120uint32_t System::now() {
121 return ticks;
122}
123
124/**
125 * \brief Returns the number of real milliseconds elapsed since the
126 * initialization of the Solarus library.
127 *
128 * This function is not deterministic, so use it at your own risks.
129 *
130 * \return The number of milliseconds elapsed since the initialization.
131 */
132uint32_t System::get_real_time() {
133 return SDL_GetTicks() - initial_time;
134}
135
136/**
137 * \brief Makes the program sleep during some time.
138 *
139 * Due to the OS scheduling, the real delay may be longer.
140 *
141 * \param duration duration of the sleep in milliseconds
142 */
143void System::sleep(uint32_t duration) {
144 SDL_Delay(duration);
145}
