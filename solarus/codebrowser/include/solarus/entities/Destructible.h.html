17#ifndef SOLARUS_DESTRUCTIBLE_H
18#define SOLARUS_DESTRUCTIBLE_H
19
20#include "solarus/Common.h"
21#include "solarus/entities/Ground.h"
22#include "solarus/entities/Entity.h"
23#include "solarus/Treasure.h"
24#include <string>
25
26namespace Solarus {
27
28/**
29 * \brief An entity that the hero can destroy and that may contain a pickable
30 * treasure.
31 *
32 * Some destructible objects can be lifted and thrown (a pot, a stone, etc.),
33 * some of them can be cut with the sword (for example some grass)
34 * and others have both behaviors (for example a bush).
35 * Some others can explode when they are lifted (a bomb or a bomb flower).
36 * When the hero lifts a destructible object, it is removed from the map and
37 * replaced by an instance of CarriedObject that is attached to the hero.
38 */
39class Destructible: public Entity {
40
41 public:
42
43 static constexpr EntityType ThisType = EntityType::DESTRUCTIBLE;
44
45 // Creation and destruction.
46 Destructible(
47 const std::string& name,
48 int layer,
49 const Point& xy,
50 const std::string& animation_set_id,
51 const Treasure& treasure,
52 Ground modified_ground
53 );
54
55 // Properties overridden from Entity.
56 virtual EntityType get_type() const override;
57 virtual Ground get_modified_ground() const override;
58
59 // Properties specific to destructibles.
60 const Treasure& get_treasure() const;
61 void set_treasure(const Treasure& treasure);
62 const std::string& get_animation_set_id() const;
63 const std::string& get_destruction_sound() const;
64 void set_destruction_sound(const std::string& destruction_sound);
65 int get_weight() const;
66 void set_weight(int weight);
67 bool get_can_be_cut() const;
68 void set_can_be_cut(bool can_be_cut);
69 bool get_can_explode() const;
70 void set_can_explode(bool can_explode);
71 bool get_can_regenerate() const;
72 void set_can_regenerate(bool can_regenerate);
73 int get_damage_on_enemies() const;
74 void set_damage_on_enemies(int damage_on_enemies);
75
76 // State.
77 void explode();
78 bool is_waiting_for_regeneration() const;
79
80 // Collisions.
81 virtual bool is_obstacle_for(Entity& other) override;
82 virtual bool test_collision_custom(Entity& entity) override;
83 virtual void notify_collision(
84 Entity& entity_overlapping,
85 CollisionMode collision_mode
86 ) override;
87 virtual void notify_collision(
88 Entity& other_entity,
89 Sprite& this_sprite,
90 Sprite& other_sprite
91 ) override;
92 void notify_collision_with_hero(Hero& hero, CollisionMode collision_mode);
93 virtual bool notify_action_command_pressed() override;
94
95 // Game loop.
96 virtual void set_suspended(bool suspended) override;
97 virtual void update() override;
98
99 private:
100
101 void update_collision_modes();
102 void play_destroy_animation();
103 void create_treasure();
104
105 const Ground modified_ground; /**< The ground defined by this entity. */
106 Treasure treasure; /**< The pickable treasure that appears when this object is destroyed. */
107 const std::string
108 animation_set_id; /**< Animation set of the sprite of this object. */
109 std::string destruction_sound_id; /**< Sound played when this object is cut or broken
110 * or an empty string. */
111 bool can_be_cut; /**< Whether this object can be cut with the sword. */
112 bool can_explode; /**< Whether this object explodes after a delay when lifted. */
113 bool can_regenerate; /**< Whether this object regenerates once destroyed. */
114 int weight; /**< Weight of this object (level of "lift" ability required).
115 * -1 means an object that cannot be lifted. */
116 int damage_on_enemies; /**< Damage this object can cause to enemies. */
117
118 bool is_being_cut; /**< Whether this object is being cut */
119 uint32_t regeneration_date; /**< Date when this object starts regenerating. */
120 bool is_regenerating; /**< Whether this object is currently regenerating. */
121
122};
123
124}
125
126#endif
