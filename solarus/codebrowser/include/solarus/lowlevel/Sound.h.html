17#ifndef SOLARUS_SOUND_H
18#define SOLARUS_SOUND_H
19
20#include "solarus/Common.h"
21#include <string>
22#include <list>
23#include <map>
24#include <al.h>
25#include <alc.h>
26#include <vorbis/vorbisfile.h>
27
28namespace Solarus {
29
30class Arguments;
31
32/**
33 * \brief Represents a sound effect that can be played in the program.
34 *
35 * This class also handles the initialization of the whole audio system.
36 * To create a sound, prefer the Sound::play() method
37 * rather than calling directly the constructor of Sound.
38 * This class is the only one that depends on the sound decoding library (libsndfile).
39 * This class and the Music class are the only ones that depend on the audio mixer library (OpenAL).
40 */
41class SOLARUS_API Sound {
42
43 public:
44
45 // libvorbisfile
46
47 /**
48 * \brief Buffer containing an encoded sound file.
49 */
50 struct SoundFromMemory {
51 std::string data; /**< The OGG encded data. */
52 size_t position; /**< Current position in the buffer. */
53 bool loop; /**< \c true to restart the sound if it finishes. */
54 };
55
56 // functions to load the encoded sound from memory
57 static ov_callbacks ogg_callbacks; /**< vorbisfile object used to load the encoded sound from memory */
58
59 Sound();
60 explicit Sound(const std::string& sound_id);
61 ~Sound();
62 void load();
63 bool start();
64
65 static void load_all();
66 static bool exists(const std::string& sound_id);
67 static void play(const std::string& sound_id);
68
69 static void initialize(const Arguments& args);
70 static void quit();
71 static bool is_initialized();
72 static void update();
73
74 static int get_volume();
75 static void set_volume(int volume);
76
77 private:
78
79 ALuint decode_file(const std::string& file_name);
80 bool update_playing();
81
82 static ALCdevice* device;
83 static ALCcontext* context;
84
85 std::string id; /**< id of this sound */
86 ALuint buffer; /**< the OpenAL buffer containing the PCM decoded data of this sound */
87 std::list<ALuint> sources; /**< the sources currently playing this sound */
88 static std::list<Sound*> current_sounds; /**< the sounds currently playing */
89 static std::map<std::string, Sound> all_sounds; /**< all sounds created before */
90
91 static bool initialized; /**< indicates that the audio system is initialized */
92 static bool sounds_preloaded; /**< true if load_all() was called */
93 static float volume; /**< the volume of sound effects (0.0 to 1.0) */
94
95};
96
97}
98
99#endif
