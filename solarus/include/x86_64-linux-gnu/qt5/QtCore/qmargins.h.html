53class QMargins
54{
55public:
56 Q_DECL_CONSTEXPR QMargins();
57 Q_DECL_CONSTEXPR QMargins(int left, int top, int right, int bottom);
58
59 Q_DECL_CONSTEXPR bool isNull() const;
60
61 Q_DECL_CONSTEXPR int left() const;
62 Q_DECL_CONSTEXPR int top() const;
63 Q_DECL_CONSTEXPR int right() const;
64 Q_DECL_CONSTEXPR int bottom() const;
65
66 void setLeft(int left);
67 void setTop(int top);
68 void setRight(int right);
69 void setBottom(int bottom);
70
71 QMargins &operator+=(const QMargins &margins);
72 QMargins &operator-=(const QMargins &margins);
73 QMargins &operator+=(int);
74 QMargins &operator-=(int);
75 QMargins &operator*=(int);
76 QMargins &operator/=(int);
77 QMargins &operator*=(qreal);
78 QMargins &operator/=(qreal);
79
80private:
81 int m_left;
82 int m_top;
83 int m_right;
84 int m_bottom;
85
86 friend Q_DECL_CONSTEXPR inline bool operator==(const QMargins &, const QMargins &);
87 friend Q_DECL_CONSTEXPR inline bool operator!=(const QMargins &, const QMargins &);
88};

137Q_DECL_CONSTEXPR inline bool operator==(const QMargins &m1, const QMargins &m2)
138{
139 return
140 m1.m_left == m2.m_left &&
141 m1.m_top == m2.m_top &&
142 m1.m_right == m2.m_right &&
143 m1.m_bottom == m2.m_bottom;
144}
145
146Q_DECL_CONSTEXPR inline bool operator!=(const QMargins &m1, const QMargins &m2)
147{
148 return
149 m1.m_left != m2.m_left ||
150 m1.m_top != m2.m_top ||
151 m1.m_right != m2.m_right ||
152 m1.m_bottom != m2.m_bottom;
153}
