32/**
33 * \brief Creates a new dialog box.
34 * \param game The game this dialog box belongs to.
35 */
36DialogBoxSystem::DialogBoxSystem(Game& game):
37 game(game),
38 callback_ref(),
39 built_in(false),
40 is_question(false),
41 selected_first_answer(true) {
42
43 for (int i = 0; i < nb_visible_lines; i++) {
44 line_surfaces[i] = std::make_shared<TextSurface>(
45 0,
46 0,
47 TextSurface::HorizontalAlignment::LEFT,
48 TextSurface::VerticalAlignment::BOTTOM
49 );
50 }
51}

62 * \brief Returns whether the dialog box is currently active.

65bool DialogBoxSystem::is_enabled() const {
66
67 return !dialog_id.empty();

78/**
79 * \brief Opens the dialog box to show a dialog.
80 *
81 * No other dialog should be already running.
82 *
83 * \param dialog_id Id of the dialog to show.
84 * \param info_ref Lua ref to an optional info parameter to pass to the
85 * dialog box, or an empty ref.
86 * \param callback_ref Lua ref to a function to call when the dialog finishes,
87 * or an empty ref.
88 */
89void DialogBoxSystem::open(
90 const std::string& dialog_id,
91 const ScopedLuaRef& info_ref,
92 const ScopedLuaRef& callback_ref
93) {
94 Debug::check_assertion(!is_enabled(), "A dialog is already active");
95
96 this->dialog_id = dialog_id;
97 this->dialog = CurrentQuest::get_dialog(dialog_id);
98 this->callback_ref = callback_ref;
99
100 // Save commands.
101 CommandsEffects& keys_effect = game.get_commands_effects();
102 keys_effect.save_action_key_effect();
103 keys_effect.set_action_key_effect(CommandsEffects::ACTION_KEY_NONE);
104 keys_effect.save_sword_key_effect();
105 keys_effect.set_sword_key_effect(CommandsEffects::ATTACK_KEY_NONE);
106 keys_effect.save_pause_key_effect();
107 keys_effect.set_pause_key_effect(CommandsEffects::PAUSE_KEY_NONE);
108
109 // A dialog was just started: notify Lua.
110 LuaContext& lua_context = game.get_lua_context();
111 lua_State* l = lua_context.get_internal_state();
112 built_in = !lua_context.notify_dialog_started(
113 game, dialog, info_ref
114 );
115
116 if (built_in) {
117
118 // Show a built-in default dialog box.
119 keys_effect.set_action_key_effect(CommandsEffects::ACTION_KEY_NEXT);
120
121 // Prepare the text.
122 std::string text = dialog.get_text();
123 this->is_question = false;
124
125 if (dialog_id == "_shop.question") {
126 // Built-in dialog with the "do you want to buy" question and the price.
127 this->is_question = true;
128 size_t index = text.find("$v");
129 if (index != std::string::npos) {
130 // Replace the special sequence '$v' by the price of the shop item.
131 info_ref.push();
132 int price = LuaTools::check_int(l, -1);
133 lua_pop(l, -1);
134 std::ostringstream oss;
135 oss << price;
136 text = text.replace(index, 2, oss.str());
137 }
138 }
139
140 remaining_lines.clear();
141 std::istringstream iss(text);
142 std::string line;
143 while (std::getline(iss, line)) {
144 remaining_lines.push_back(line);
145 }
146
147 // Determine the position.
148 bool top = false;
149 const CameraPtr& camera = game.get_current_map().get_camera();
150 if (camera != nullptr) {
151 const Rectangle& camera_position = camera->get_bounding_box();
152 if (game.get_hero()->get_y() >= camera_position.get_y() + 130) {
153 top = true;
154 }
155 }
156
157 const Size& quest_size = Video::get_quest_size();
158 int x = quest_size.width / 2 - 110;
159 int y = top ? 32 : quest_size.height - 96;
160
161 text_position = { x, y };
162
163 // Start showing text.
164 show_more_lines();
165 }
166}

168/**
169 * \brief Closes the dialog box.
170 * \param status_ref Lua ref to a status value to return to the start_dialog
171 * callback, or an empty ref. "skipped" means that the dialog was canceled by
172 * the user.
173 */
174void DialogBoxSystem::close(const ScopedLuaRef& status_ref) {
175
176 Debug::check_assertion(is_enabled(), "No dialog is active");
177
178 ScopedLuaRef callback_ref = this->callback_ref;
179 this->callback_ref.clear();
180 this->dialog_id = "";
181
182 // Restore commands.
183 CommandsEffects& keys_effect = game.get_commands_effects();
184 keys_effect.restore_action_key_effect();
185 keys_effect.restore_sword_key_effect();
186 keys_effect.restore_pause_key_effect();
187
188 // A dialog was just finished: notify Lua.
189 game.get_lua_context().notify_dialog_finished(
190 game, dialog, callback_ref, status_ref
191 );
192}

194/**
195 * \brief Returns whether there are more lines remaining to display after the
196 * current 3 lines in the built-in dialog box.
197 * \return \c true if there are more lines.
198 */
199bool DialogBoxSystem::has_more_lines() const {
200 return !remaining_lines.empty();
201}
202
203/**
204 * \brief Shows a new group of 3 lines (if possible) in the built-in
205 * dialog box.
206 */
207void DialogBoxSystem::show_more_lines() {
208
209 // This function is only called in the built-in case.
210 Debug::check_assertion(built_in, "This dialog box is not the built-in one");
211
212 if (!has_more_lines()) {
213
214 ScopedLuaRef status_ref;
215 if (is_question) {
216 // Send the answer to the callback.
217 LuaContext& lua_context = game.get_lua_context();
218 lua_pushboolean(lua_context.get_internal_state(), selected_first_answer);
219 status_ref = lua_context.create_ref();
220 }
221 close(status_ref);
222 return;
223 }
224
225 CommandsEffects& keys_effect = game.get_commands_effects();
226 keys_effect.set_action_key_effect(CommandsEffects::ACTION_KEY_NEXT);
227
228 // Prepare the 3 lines.
229 int text_x = text_position.x;
230 int text_y = text_position.y;
231 for (int i = 0; i < nb_visible_lines; i++) {
232 text_y += 16;
233 line_surfaces[i]->set_x(text_x);
234 line_surfaces[i]->set_y(text_y);
235 line_surfaces[i]->set_text_color(Color::white);
236
237 if (has_more_lines()) {
238 line_surfaces[i]->set_text(*remaining_lines.begin());
239 remaining_lines.pop_front();
240 }
241 else {
242 line_surfaces[i]->set_text("");
243 }
244 }
245
246 if (built_in && is_question && !has_more_lines()) {
247 // If the dialog is a question, we assume that the last group of 3 lines
248 // are the question and the two possible answers.
249 // Remember that this is only the case of the built-in dialog box:
250 // if the user needs something more elaborate, he should make his own
251 // dialog box in Lua.
252 this->selected_first_answer = true;
253 line_surfaces[nb_visible_lines - 2]->set_text_color(Color::yellow);
254 }
255}

257/**
258 * \brief This function is called by the game when a command is pressed
259 * while a dialog is active.
260 *
261 * Nothing happens if the dialog is handled in Lua.
262 *
263 * \param command The command pressed.
264 * \return \c true if the command was handled (that is, if the dialog box
265 * is active and is the built-in one).
266 */
267bool DialogBoxSystem::notify_command_pressed(GameCommand command) {
268
269 if (!is_enabled()) {
270 return false;
271 }
272
273 if (!built_in) {
274 // The dialog box is handled by a Lua script.
275 return false;
276 }
277
278 if (command == GameCommand::ACTION) {
279 show_more_lines();
280 }
281 else if (command == GameCommand::UP || command == GameCommand::DOWN) {
282 if (is_question && !has_more_lines()) {
283 // Switch the selected answer.
284 selected_first_answer = !selected_first_answer;
285 int selected_line_index = selected_first_answer ? 1 : 2;
286 for (int i = 0; i < nb_visible_lines; i++) {
287 line_surfaces[i]->set_text_color(Color::white);
288 }
289 line_surfaces[selected_line_index]->set_text_color(Color::yellow);
290 }
291 }
292
293 return true;
294}

296/**
297 * \brief Draws the dialog box on a surface.
298 *
299 * Draws nothing if the dialog is handled by Lua.
300 *
301 * \param dst_surface The destination surface.
302 */
303void DialogBoxSystem::draw(const SurfacePtr& dst_surface) {
304
305 if (!built_in) {
306 // The dialog box is handled by a Lua script.
307 return;
308 }
309
310 // Draw the text.
311 for (int i = 0; i < nb_visible_lines; i++) {
312 line_surfaces[i]->draw(dst_surface);
313 }
314}
