27/**
28 * \brief Constructor.
29 * \param ignore_obstacles true to ignore obstacles of the map
30 * (used only when there is a map and the movement is attached to an entity of this map)
31 * \param smooth true to adjust the trajectory if an obstacle is reached
32 * (only when ignore_obstacles is false)
33 */

35 Movement(ignore_obstacles),
36 angle(0),
37 x_speed(0),
38 y_speed(0),
39 next_move_date_x(System::now()),

41 x_delay(0),
42 y_delay(0),
43 x_move(0),
44 y_move(0),
45 max_distance(0),
46 finished(false),
47 smooth(smooth) {

49}

76/**
77 * \brief Returns the total speed of the object.
78 *
79 * The speed is calculated as sqrt(x_speed^2 + y_speed^2).
80 *
81 * \return the speed in pixels per second
82 */
83double StraightMovement::get_speed() const {
84 return std::sqrt(x_speed * x_speed + y_speed * y_speed);
85}

87/**
88 * \brief Sets the x speed.
89 * \param x_speed the x speed of the object in pixels per second
90 */
91void StraightMovement::set_x_speed(double x_speed) {
92
93 if (std::abs(x_speed) <= 1E-6) {
94 x_speed = 0;
95 }
96
97 this->x_speed = x_speed;
98 uint32_t now = System::now();
99
100 // compute x_delay, x_move and next_move_date_x
101 if (x_speed == 0) {
102 x_move = 0;
103 }
104 else {
105 if (x_speed > 0) {
106 x_delay = (uint32_t) (1000 / x_speed);
107 x_move = 1;
108 }
109 else {
110 x_delay = (uint32_t) (1000 / (-x_speed));
111 x_move = -1;
112 }
113 set_next_move_date_x(now + x_delay);
114 }
115 angle = Geometry::get_angle(0.0, 0.0, x_speed * 100.0, y_speed * 100.0);
116 initial_xy = get_xy();
117 finished = false;
118
119 notify_movement_changed();
120}

123 * \brief Sets the y speed.
124 * \param y_speed the y speed of the object in pixels per second

126void StraightMovement::set_y_speed(double y_speed) {
127
128 if (std::abs(y_speed) <= 1E-6) {
129 y_speed = 0;

131
132 this->y_speed = y_speed;

135 // compute y_delay, y_move and next_move_date_y
136 if (y_speed == 0) {
137 y_move = 0;
138 }
139 else {
140 if (y_speed > 0) {
141 y_delay = (uint32_t) (1000 / y_speed);
142 y_move = 1;
143 }
144 else {
145 y_delay = (uint32_t) (1000 / (-y_speed));
146 y_move = -1;
147 }
148 set_next_move_date_y(now + y_delay);
149 }
150 angle = Geometry::get_angle(0.0, 0.0, x_speed * 100.0, y_speed * 100.0);
151 initial_xy = get_xy();
152 finished = false;
153
154 notify_movement_changed();
155}

157/**
158 * \brief Changes the speed, keeping the same direction of the movement.
159 *
160 * x_speed and y_speed are recomputed so that the movement direction is unchanged.
161 *
162 * \param speed the new speed
163 */
164void StraightMovement::set_speed(double speed) {
165
166 // compute the new speed vector
167 double old_angle = this->angle;
168 set_x_speed(speed * std::cos(old_angle));
169 set_y_speed(-speed * std::sin(old_angle));
170 this->angle = old_angle;
171
172 notify_movement_changed();
173}

175/**
176 * \brief Returns whether the speed is not zero.
177 * \return true if the object is moving, false otherwise
178 */
179bool StraightMovement::is_started() const {
180 return x_speed != 0 || y_speed != 0;
181}

183/**
184 * \brief Sets the speed to zero.
185 */
186void StraightMovement::stop() {
187
188 double old_angle = this->angle;
189 set_x_speed(0);
190 set_y_speed(0);
191 x_move = 0;
192 y_move = 0;
193 this->angle = old_angle;
194
195 notify_movement_changed();
196}

198/**
199 * \brief Sets the date of the next change of the x coordinate.
200 * \param next_move_date_x the date in milliseconds
201 */
202void StraightMovement::set_next_move_date_x(uint32_t next_move_date_x) {
203
204 if (is_suspended()) {
205 uint32_t delay = next_move_date_x - System::now();
206 this->next_move_date_x = get_when_suspended() + delay;
207 }
208 else {
209 this->next_move_date_x = next_move_date_x;
210 }
211}

219 if (is_suspended()) {

221 this->next_move_date_y = get_when_suspended() + delay;

236/**
237 * \brief Changes the direction of the movement vector, keeping the same speed.
238 *
239 * x_speed and y_speed are recomputed so that the total speed is unchanged.
240 *
241 * \param angle the new movement direction in radians
242 */
243void StraightMovement::set_angle(double angle) {
244
245 if (!is_stopped()) {
246 double speed = get_speed();
247 set_x_speed(speed * std::cos(angle));
248 set_y_speed(-speed * std::sin(angle));
249 }
250 this->angle = angle;
251
252 notify_movement_changed();

301/**
302 * \brief Returns true if the object is about to try to move.
303 *
304 * This function returns true if x_move is not equal to zero
305 * and next_move_date_x is past, or the same thing for y.
306 *
307 * \return true if the entity is about to try to move
308 */
309bool StraightMovement::has_to_move_now() const {
310
311 uint32_t now = System::now();
312
313 return (x_move != 0 && now >= next_move_date_x)
314 || (y_move != 0 && now >= next_move_date_y);
315}

317/**
318 * \brief Suspends or resumes the movement.
319 *
320 * This function is called by the entity when the game is suspended or resumed.
321 *
322 * \param suspended true to suspend the movement, false to resume it
323 */
324void StraightMovement::set_suspended(bool suspended) {
325
326 Movement::set_suspended(suspended);
327
328 if (!suspended) {
329
330 // recalculate the next move date
331 if (get_when_suspended() != 0) {
332 uint32_t diff = System::now() - get_when_suspended();
333 next_move_date_x += diff;
334 next_move_date_y += diff;
335 }
336 }
337}

357/**
358 * @brief Updates the x and y coordinates of one step
359 * in the smooth case.
360 */
361void StraightMovement::update_smooth_xy() {
362
363 // Save the current coordinates.
364 Point old_xy = get_xy();
365
366 uint32_t now = System::now();
367 bool x_move_now = x_move != 0 && now >= next_move_date_x;
368 bool y_move_now = y_move != 0 && now >= next_move_date_y;
369
370 if (x_move_now) {
371 // it's time to make an x move
372
373 if (y_move_now) {
374 // but it's also time to make a y move
375
376 if (next_move_date_x <= next_move_date_y) {
377 // x move first
378 update_smooth_x();
379 if (now >= next_move_date_y) {
380 update_smooth_y();
381 }
382 }
383 else {
384 // y move first
385 update_smooth_y();
386 if (now >= next_move_date_x) {
387 update_smooth_x();
388 }
389 }
390 }
391 else {
392 update_smooth_x();
393 }
394 }
395 else {
396 update_smooth_y();
397 }
398
399 if (!is_suspended() && get_entity() != nullptr && !finished) {
400
401 // the movement was successful if the entity's coordinates have changed
402 // and the movement was not stopped
403 bool success = (get_xy() != old_xy)
404 && (x_move != 0 || y_move != 0);
405
406 if (!success) {
407 notify_obstacle_reached();
408 }
409 }
410
411}

413/**
414 * \brief Updates the x position of the entity if it wants to move
415 * (smooth version).
416 */
417void StraightMovement::update_smooth_x() {
418
419 if (x_move != 0) { // The entity wants to move on x.
420
421 // By default, next_move_date_x will be incremented by x_delay,
422 // unless we modify below the movement in such a way that the
423 // x speed needs to be fixed.
424 uint32_t next_move_date_x_increment = x_delay;
425
426 if (!test_collision_with_obstacles(x_move, 0)) {
427
428 translate_x(x_move); // Make the move.
429
430 if (y_move != 0 && test_collision_with_obstacles(0, y_move)) {
431 // If there is also a y move, and if this y move is illegal,
432 // we still allow the x move and we give it all the speed.
433 next_move_date_x_increment = (int) (1000.0 / get_speed());
434 }
435 }
436 else {
437 if (y_move == 0) {
438 // The move on x is not possible and there is no y move:
439 // let's try to add a move on y to make a diagonal move,
440 // but only if the wall is really diagonal: otherwise, the hero
441 // could bypass sensors.
442
443 if (!test_collision_with_obstacles(x_move, 1) // Can move diagonally and:
444 && (test_collision_with_obstacles(0, -1) || // the wall is really diagonal
445 test_collision_with_obstacles(0, 1)) // or we don't have a choice anyway.
446 ) {
447 translate_xy(x_move, 1);
448 next_move_date_x_increment = (int) (x_delay * Geometry::SQRT_2); // Fix the speed.
449 }
450 else if (!test_collision_with_obstacles(x_move, -1)
451 && (test_collision_with_obstacles(0, 1) ||
452 test_collision_with_obstacles(0, -1))
453 ) {
454 translate_xy(x_move, -1);
455 next_move_date_x_increment = (int) (x_delay * Geometry::SQRT_2);
456 }
457 else {
458
459 // The diagonal moves didn't work either.
460 // So we look for a place (up to 8 pixels up and down)
461 // where the required move would be allowed.
462 // If we find a such place, then we move towards this place.
463
464 bool moved = false;
465 for (int i = 1; i <= 8 && !moved; i++) {
466
467 if (!test_collision_with_obstacles(x_move, i) && !test_collision_with_obstacles(0, 1)) {
468 translate_y(1);
469 moved = true;
470 }
471 else if (!test_collision_with_obstacles(x_move, -i) && !test_collision_with_obstacles(0, -1)) {
472 translate_y(-1);
473 moved = true;
474 }
475 }
476 }
477 }
478 else {
479 // The move on x is not possible, but there is also a vertical move.
480 if (!test_collision_with_obstacles(0, y_move)) {
481 // Do the vertical move right now, don't wait uselessly.
482 update_smooth_y();
483 }
484 else {
485 // The x move is not possible and neither is the y move.
486 // Last chance: do both x and y moves in one step.
487 // This case is only necessary in narrow diagonal passages.
488 // We do it as a last resort, because we want separate x and y
489 // steps whenever possible: otherwise, the hero could bypass sensors.
490 if (!test_collision_with_obstacles(x_move, y_move)) {
491 translate_xy(x_move, y_move);
492 next_move_date_y += y_delay; // Delay the next update_smooth_y() since we just replaced it.
493 }
494 }
495 }
496 }
497 next_move_date_x += next_move_date_x_increment;
498 }
499}

501/**
502 * \brief Updates the y position of the entity if it wants to move
503 * (smooth version).
504 */
505void StraightMovement::update_smooth_y() {
506
507 if (y_move != 0) { // The entity wants to move on y.
508
509 // By default, next_move_date_y will be incremented by y_delay,
510 // unless we modify the movement in such a way that the
511 // y speed needs to be fixed.
512 uint32_t next_move_date_y_increment = y_delay;
513
514 if (!test_collision_with_obstacles(0, y_move)) {
515
516 translate_y(y_move); // Make the move.
517
518 if (x_move != 0 && test_collision_with_obstacles(x_move, 0)) {
519 // If there is also an x move, and if this x move is illegal,
520 // we still allow the y move and we give it all the speed.
521 next_move_date_y_increment = (int) (1000.0 / get_speed());
522 }
523 }
524 else {
525 if (x_move == 0) {
526 // The move on y is not possible and there is no x move:
527 // let's try to add a move on x to make a diagonal move,
528 // but only if the wall is really diagonal: otherwise, the hero
529 // could bypass sensors.
530
531 if (!test_collision_with_obstacles(1, y_move) // Can move diagonally and:
532 && (test_collision_with_obstacles(-1, 0) || // the wall is really diagonal
533 test_collision_with_obstacles(1, 0)) // or we don't have a choice anyway.
534 ) {
535 translate_xy(1, y_move);
536 next_move_date_y_increment = (int) (y_delay * Geometry::SQRT_2); // Fix the speed.
537 }
538 else if (!test_collision_with_obstacles(-1, y_move)
539 && (test_collision_with_obstacles(1, 0) ||
540 test_collision_with_obstacles(-1, 0))
541 ) {
542 translate_xy(-1, y_move);
543 next_move_date_y_increment = (int) (y_delay * Geometry::SQRT_2);
544 }
545 else {
546 // The diagonal moves didn't work either.
547 // So we look for a place (up to 8 pixels on the left and on the right)
548 // where the required move would be allowed.
549 // If we find a such place, then we move towards this place.
550
551 bool moved = false;
552 for (int i = 1; i <= 8 && !moved; i++) {
553
554 if (!test_collision_with_obstacles(i, y_move) && !test_collision_with_obstacles(1, 0)) {
555 translate_x(1);
556 moved = true;
557 }
558 else if (!test_collision_with_obstacles(-i, y_move) && !test_collision_with_obstacles(-1, 0)) {
559 translate_x(-1);
560 moved = true;
561 }
562 }
563 }
564 }
565 else {
566 // The move on y is not possible, but there is also a horizontal move.
567 if (!test_collision_with_obstacles(x_move, 0)) {
568 // Do the horizontal move right now, don't wait uselessly.
569 update_smooth_x();
570 }
571 else {
572 // The y move is not possible and neither is the x move.
573 // Last chance: do both x and y moves in one step.
574 // This case is only necessary in narrow diagonal passages.
575 // We do it as a last resort, because we want separate x and y
576 // steps whenever possible: otherwise, the hero could bypass sensors.
577 if (!test_collision_with_obstacles(x_move, y_move)) {
578 translate_xy(x_move, y_move);
579 next_move_date_x += x_delay; // Delay the next update_smooth_x() since we just replaced it.
580 }
581 }
582 }
583 }
584 next_move_date_y += next_move_date_y_increment;
585 }

588/**
589 * @brief Updates the x and y coordinates of one step
590 * in the non-smooth case.
591 */
592void StraightMovement::update_non_smooth_xy() {
593
594 // Save the current coordinates.
595 Point old_xy = get_xy();
596
597 uint32_t now = System::now();
598 bool x_move_now = x_move != 0 && now >= next_move_date_x;
599 bool y_move_now = y_move != 0 && now >= next_move_date_y;
600
601 if (x_move_now) {
602 // it's time to make an x move
603
604 if (y_move_now) {
605 // but it's also time to make a y move
606
607 if (!test_collision_with_obstacles(x_move, y_move)) {
608 translate_xy(x_move, y_move);
609 }
610 next_move_date_x += x_delay;
611 next_move_date_y += y_delay;
612 }
613 else {
614 if (!test_collision_with_obstacles(x_move, 0)) {
615 translate_x(x_move);
616 }
617 next_move_date_x += x_delay;
618 }
619 }
620 else {
621 if (!test_collision_with_obstacles(0, y_move)) {
622 translate_y(y_move);
623 }
624 next_move_date_y += y_delay;
625 }
626
627 if (!is_suspended() &&
628 get_entity() != nullptr &&
629 !finished) {
630
631 // the movement was successful if the entity's coordinates have changed
632 // and the movement was not stopped
633 bool success = (get_xy() != old_xy)
634 && (x_move != 0 || y_move != 0);
635
636 if (!success) {
637 notify_obstacle_reached();
638 }
639 }
640
641}

643/**
644 * \brief Updates the position of the object controlled by this movement.
645 *
646 * This function is called repeatedly.
647 */
648void StraightMovement::update() {
649
650 if (!is_suspended()) {
651
652 uint32_t now = System::now();
653 bool x_move_now = x_move != 0 && now >= next_move_date_x;
654 bool y_move_now = y_move != 0 && now >= next_move_date_y;
655
656 while (x_move_now || y_move_now) { // while it's time to move
657
658 if (is_smooth()) {
659 update_smooth_xy();
660 }
661 else {
662 update_non_smooth_xy();
663 }
664
665 now = System::now();
666
667 if (!finished && max_distance != 0
668 && Geometry::get_distance(initial_xy, get_xy()) >= max_distance) {
669 set_finished();
670 }
671 else {
672 x_move_now = x_move != 0 && now >= next_move_date_x;
673 y_move_now = y_move != 0 && now >= next_move_date_y;
674 }
675 }
676 }
677
678 // Do this at last so that Movement::update() knows whether we are finished.
679 Movement::update();
680}
