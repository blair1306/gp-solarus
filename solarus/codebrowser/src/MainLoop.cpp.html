45/**
46 * \brief Checks that the quest is compatible with the current version of
47 * Solarus.
48 * \param solarus_required_version Version of the quest.
49 */
50void check_version_compatibility(const std::string& solarus_required_version) {
51
52 if (solarus_required_version.empty()) {
53 Debug::die("No Solarus version is specified in your quest.dat file!");
54 }
55
56 // TODO check the syntax of the version string
57
58 int dot_index_1 = solarus_required_version.find('.');
59 std::istringstream iss(solarus_required_version.substr(0, dot_index_1));
60 int required_major_version = 0;
61 iss >> required_major_version;
62
63 int dot_index_2 = solarus_required_version.find('.', dot_index_1 + 1);
64 std::istringstream iss2(solarus_required_version.substr(dot_index_1 + 1, dot_index_2));
65 int required_minor_version = 0;
66 iss2 >> required_minor_version;
67
68 // The third digit of the version (patch) is ignored because compatibility
69 // is not broken by patches.
70
71 // For now, we assume that any mismatch of major or minor version (first
72 // and second digits) breaks compatibility and we just die.
73 // This may change in the future, because all minor versions won't
74 // necessarily break compatibility.
75 if (required_major_version != SOLARUS_MAJOR_VERSION ||
76 required_minor_version != SOLARUS_MINOR_VERSION) {
77 std::ostringstream oss;
78 oss << "This quest is made for Solarus " << required_major_version << "."
79 << required_minor_version << ".x but you are running Solarus "
80 << SOLARUS_VERSION;
81 Debug::die(oss.str());
82 }
83}

85/**
86 * \brief Returns path of the quest to run.
87 *
88 * It may be the path defined as command-line argument,
89 * the path defined during the build process, or the current directory
90 * if nothing was specified.
91 *
92 * \param args Command-line arguments.
93 * \return The quest path.
94 */
95std::string get_quest_path(const Arguments& args) {
96
97 // If a quest command-line argument was specified, use it instead.
98 const std::vector<std::string>& options = args.get_arguments();
99 if (!options.empty()
100 && !options.back().empty()
101 && options.back()[0] != '-') {
102 // The last parameter is not an option: it is the quest path.
103 return options.back();
104 }
105
106 // The default quest path is defined during the build process.
107 return SOLARUS_DEFAULT_QUEST;
108}

112/**
113 * \brief Initializes the game engine.
114 * \param args Command-line arguments.
115 */
116MainLoop::MainLoop(const Arguments& args):
117 lua_context(nullptr),
118 root_surface(nullptr),
119 game(nullptr),
120 next_game(nullptr),
121 exiting(false),
122 debug_lag(0),
123 turbo(false),
124 lua_commands(),
125 lua_commands_mutex(),
126 num_lua_commands_pushed(0),
127 num_lua_commands_done(0) {
128
129 Logger::info(std::string("Solarus ") + SOLARUS_VERSION);
130
131 // Main loop settings.
132 const std::string lag_arg = args.get_argument_value("-lag");
133 if (!lag_arg.empty()) {
134 std::istringstream iss(lag_arg);
135 iss >> debug_lag;
136 }
137 const std::string& turbo_arg = args.get_argument_value("-turbo");
138 turbo = (turbo_arg == "yes");
139
140 // Try to open the quest.
141 const std::string& quest_path = get_quest_path(args);
142 Logger::info("Opening quest '" + quest_path + "'");
143 if (!QuestFiles::open_quest(args.get_program_name(), quest_path)) {
144 Debug::error("No quest was found in the directory '" + quest_path + "'");
145 return;
146 }
147
148 // Initialize engine features (audio, video...).
149 System::initialize(args);
150
151 if (turbo) {
152 Logger::info("Turbo mode: yes");
153 }
154 else {
155 Logger::info("Turbo mode: no");
156 }
157
158 // Read the quest resource list from data.
159 CurrentQuest::initialize();
160 TilePattern::initialize();
161
162 // Read the quest general properties.
163 load_quest_properties();
164
165 // Create the quest surface.
166 root_surface = Surface::create(
167 Video::get_quest_size()
168 );
169 root_surface->set_software_destination(false); // Accelerate this surface.
170
171 // Run the Lua world.
172 // Do this after the creation of the window, but before showing the window,
173 // because Lua might change the video mode initially.
174 lua_context = std::unique_ptr<LuaContext>(new LuaContext(*this));
175 lua_context->initialize();
176
177 // Set up the Lua console.
178 const std::string& lua_console_arg = args.get_argument_value("-lua-console");
179 const bool enable_lua_console = lua_console_arg.empty() || lua_console_arg == "yes";
180 if (enable_lua_console) {
181 Logger::info("Lua console: yes");
182 initialize_lua_console();
183 }
184 else {
185 Logger::info("Lua console: no");
186 }
187
188 // Finally show the window.
189 Video::show_window();
190}

192/**
193 * \brief Cleans everything.
194 */
195MainLoop::~MainLoop() {
196
197 if (game != nullptr) {
198 game->stop();
199 game.reset(); // While deleting the game, the Lua world must still exist.
200 }
201
202 // Clear the surface while Lua still exists,
203 // because it may point to other surfaces that have Lua movements.
204 root_surface = nullptr;
205
206 if (lua_context != nullptr) {
207 lua_context->exit();
208 }
209 TilePattern::quit();
210 CurrentQuest::quit();
211 QuestFiles::close_quest();
212 System::quit();
213 quit_lua_console();
214}

290/**
291 * \brief Schedules a Lua command to be executed at the next cycle.
292 *
293 * This function is thread safe, it can be called from a separate thread
294 * while the main loop is running.
295 *
296 * \param command The Lua string to execute.
297 * \return A number identifying your command.
298 */
299int MainLoop::push_lua_command(const std::string& command) {
300
301 std::lock_guard<std::mutex> lock(lua_commands_mutex);
302 lua_commands.push_back(command);
303 return num_lua_commands_pushed++;
304}

306/**
307 * \brief Runs the main loop until the user requests to stop the program.
308 *
309 * The main loop controls simulated time and repeatedly updates the world and
310 * redraws the screen.
311 *
312 * Does nothing if the quest is missing.
313 */
314void MainLoop::run() {
315
316 if (!QuestFiles::quest_exists()) {
317 return;
318 }
319
320 // Main loop.
321 Logger::info("Simulation started");
322
323 uint32_t last_frame_date = System::get_real_time();
324 uint32_t lag = 0; // Lose time of the simulation to catch up.
325 uint32_t time_dropped = 0; // Time that won't be caught up.
326
327 // The main loop basically repeats
328 // check_input(), update(), draw() and sleep().
329 // Each call to update() makes the simulated time advance one fixed step.
330
331 while (!is_exiting()) {
332
333 // Measure the time of the last iteration.
334 uint32_t now = System::get_real_time() - time_dropped;
335 uint32_t last_frame_duration = now - last_frame_date;
336 last_frame_date = now;
337 lag += last_frame_duration;
338 // At this point, lag represents how much late the simulated time with
339 // compared to the real time.
340
341 if (lag >= 200) {
342 // Huge lag: don't try to catch up.
343 // Maybe we have just made a one-time heavy operation like loading a
344 // big file, or the process was just unsuspended.
345 // Let's fake the real time instead.
346 time_dropped += lag - System::timestep;
347 lag = System::timestep;
348 last_frame_date = System::get_real_time() - time_dropped;
349 }
350
351 // 1. Detect and handle input events.
352 check_input();
353
354 // 2. Update the world once, or several times (skipping some draws)
355 // to catch up if the system is slow.
356 int num_updates = 0;
357 if (turbo) {
358 // Turbo mode: always update at least once.
359 step();
360 lag -= System::timestep;
361 ++num_updates;
362 }
363
364 while (lag >= System::timestep &&
365 num_updates < 10 && // To draw sometimes anyway on very slow systems.
366 !is_exiting()
367 ) {
368 step();
369 lag -= System::timestep;
370 ++num_updates;
371 }
372
373 // 3. Redraw the screen.
374 if (num_updates > 0) {
375 draw();
376 }
377
378 // 4. Sleep if we have time, to save CPU and GPU cycles.
379 if (debug_lag > 0 && !turbo) {
380 // Extra sleep time for debugging, useful to simulate slower systems.
381 System::sleep(debug_lag);
382 }
383
384 last_frame_duration = (System::get_real_time() - time_dropped) - last_frame_date;
385 if (last_frame_duration < System::timestep && !turbo) {
386 System::sleep(System::timestep - last_frame_duration);
387 }
388 }
389
390 Logger::info("Simulation finished");
391}

403/**
404 * \brief Detects whether there were input events and if yes, handles them.
405 */
406void MainLoop::check_input() {
407
408 // Check SDL events.
409 std::unique_ptr<InputEvent> event = InputEvent::get_event();
410 while (event != nullptr) {
411 notify_input(*event);
412 event = InputEvent::get_event();
413 }
414
415 // Check Lua requests.
416 if (!lua_commands.empty()) {
417 std::lock_guard<std::mutex> lock(lua_commands_mutex);
418 for (const std::string& command : lua_commands) {
419 std::cout << "\n"; // To make sure that the command delimiter starts on a new line.
420 Logger::info("====== Begin Lua command #" + String::to_string(num_lua_commands_done) + " ======");
421 const bool success = LuaTools::do_string(get_lua_context().get_internal_state(), command, "Lua command");
422 if (success) {
423 std::cout << "\n";
424 Logger::info("====== End Lua command #" + String::to_string(num_lua_commands_done) + ": success ======");
425 }
426 else {
427 std::cout << "\n";
428 Logger::info("====== End Lua command #" + String::to_string(num_lua_commands_done) + ": error ======");
429 }
430 ++num_lua_commands_done;
431 }
432 lua_commands.clear();
433 }
434}

436/**
437 * \brief This function is called when there is an input event.
438 *
439 * It forwards the event to Lua, and then to the game if Lua did not
440 * handle it.
441 */
442void MainLoop::notify_input(const InputEvent& event) {
443
444 if (event.is_window_closing()) {
445 set_exiting();
446 }
447 else if (event.is_keyboard_key_pressed()) {
448 // A key was pressed.
449#if defined(PANDORA)
450 // TODO make a clean flag
451 if (event.get_keyboard_key() == InputEvent::KEY_ESCAPE) {
452 exiting = true;
453 }
454#endif
455 }
456
457 // Send the event to Lua and to the current screen.
458 bool handled = lua_context->notify_input(event);
459 if (!handled && game != nullptr) {
460 game->notify_input(event);
461 }
462}

464/**
465 * \brief Updates the current screen.
466 *
467 * This function is called repeatedly by the main loop.
468 */
469void MainLoop::update() {
470
471 if (game != nullptr) {
472 game->update();
473 }
474 lua_context->update();
475 System::update();
476
477 // go to another game?
478 if (next_game != game.get()) {
479
480 game = std::unique_ptr<Game>(next_game);
481
482 if (game != nullptr) {
483 game->start();
484 }
485 else {
486 lua_context->exit();
487 lua_context->initialize();
488 Music::stop_playing();
489 }
490 }
491}

493/**
494 * \brief Redraws the current screen.
495 *
496 * This function is called repeatedly by the main loop.
497 */
498void MainLoop::draw() {
499
500 root_surface->clear();
501
502 if (game != nullptr) {
503 game->draw(root_surface);
504 }
505 lua_context->main_on_draw(root_surface);
506 Video::render(root_surface);
507}

509/**
510 * \brief Reads the quest properties file quest.dat and applies its settings.
511 */
512void MainLoop::load_quest_properties() {
513
514 const QuestProperties& properties = CurrentQuest::get_properties();
515
516 check_version_compatibility(properties.get_solarus_version());
517 const std::string& title = properties.get_title();
518 const std::string& quest_version = properties.get_quest_version();
519 if (!title.empty()) {
520 std::string window_title = title;
521 if (!quest_version.empty()) {
522 window_title += " " + quest_version;
523 }
524 window_title += std::string(" - Solarus ") + SOLARUS_VERSION;
525 Video::set_window_title(window_title);
526 }
527
528 Video::set_quest_size_range(
529 properties.get_normal_quest_size(),
530 properties.get_min_quest_size(),
531 properties.get_max_quest_size()
532 );
533
534}

536/**
537 * \brief Enables accepting standard input lines as Lua commands.
538 */
539void MainLoop::initialize_lua_console() {
540
541 // Watch stdin in a separate thread.
542 stdin_thread = std::thread([this]() {
543
544 std::string line;
545 while (!is_exiting()) {
546
547 if (std::getline(std::cin, line)) {
548
549 while (!line.empty() && std::isspace(line.at(line.size() - 1))) {
550 line.erase(line.size() - 1);
551 }
552
553 if (!line.empty()) {
554 push_lua_command(line);
555 }
556 }
557 }
558 });
559 stdin_thread.detach();
560}
561
562/**
563 * \brief Cleans resources started by initialize_lua_console().
564 */
565void MainLoop::quit_lua_console() {
566
567 exiting = true;
568 if (!stdin_thread.joinable()) {
569 return;
570 }
571
572 stdin_thread.join();
573}
