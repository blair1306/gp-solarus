61class Q_CORE_EXPORT QRegularExpression
62{
63public:
64 enum PatternOption {
65 NoPatternOption = 0x0000,
66 CaseInsensitiveOption = 0x0001,
67 DotMatchesEverythingOption = 0x0002,
68 MultilineOption = 0x0004,
69 ExtendedPatternSyntaxOption = 0x0008,
70 InvertedGreedinessOption = 0x0010,
71 DontCaptureOption = 0x0020,
72 UseUnicodePropertiesOption = 0x0040
73 };
74 Q_DECLARE_FLAGS(PatternOptions, PatternOption)
75
76 PatternOptions patternOptions() const;
77 void setPatternOptions(PatternOptions options);
78
79 QRegularExpression();
80 explicit QRegularExpression(const QString &pattern, PatternOptions options = NoPatternOption);
81 QRegularExpression(const QRegularExpression &re);
82 ~QRegularExpression();
83 QRegularExpression &operator=(const QRegularExpression &re);
84
85#ifdef Q_COMPILER_RVALUE_REFS
86 inline QRegularExpression &operator=(QRegularExpression &&re)
87 { d.swap(re.d); return *this; }
88#endif
89
90 inline void swap(QRegularExpression &re) { d.swap(re.d); }
91
92 QString pattern() const;
93 void setPattern(const QString &pattern);
94
95 bool isValid() const;
96 int patternErrorOffset() const;
97 QString errorString() const;
98
99 int captureCount() const;
100 QStringList namedCaptureGroups() const;
101
102 enum MatchType {
103 NormalMatch = 0,
104 PartialPreferCompleteMatch,
105 PartialPreferFirstMatch,
106 NoMatch
107 };
108
109 enum MatchOption {
110 NoMatchOption = 0x0000,
111 AnchoredMatchOption = 0x0001
112 };
113 Q_DECLARE_FLAGS(MatchOptions, MatchOption)
114
115 QRegularExpressionMatch match(const QString &subject,
116 int offset = 0,
117 MatchType matchType = NormalMatch,
118 MatchOptions matchOptions = NoMatchOption) const;
119
120 QRegularExpressionMatchIterator globalMatch(const QString &subject,
121 int offset = 0,
122 MatchType matchType = NormalMatch,
123 MatchOptions matchOptions = NoMatchOption) const;
124
125 static QString escape(const QString &str);
126
127 bool operator==(const QRegularExpression &re) const;
128 inline bool operator!=(const QRegularExpression &re) const { return !operator==(re); }
129
130private:
131 friend struct QRegularExpressionPrivate;
132 friend class QRegularExpressionMatch;
133 friend struct QRegularExpressionMatchPrivate;
134 friend class QRegularExpressionMatchIterator;
135
136 QRegularExpression(QRegularExpressionPrivate &dd);
137 QExplicitlySharedDataPointer<QRegularExpressionPrivate> d;
138};
139
140Q_DECLARE_SHARED(QRegularExpression)
141Q_DECLARE_OPERATORS_FOR_FLAGS(QRegularExpression::PatternOptions)
142Q_DECLARE_OPERATORS_FOR_FLAGS(QRegularExpression::MatchOptions)
