60// Prealloc = 256 by default, specified in qcontainerfwd.h
61template<class T, int Prealloc>
62class QVarLengthArray
63{
64public:
65 inline explicit QVarLengthArray(int size = 0);
66
67 inline QVarLengthArray(const QVarLengthArray<T, Prealloc> &other)
68 : a(Prealloc), s(0), ptr(reinterpret_cast<T *>(array))
69 {
70 append(other.constData(), other.size());
71 }
72
73 inline ~QVarLengthArray() {
74 if (QTypeInfo<T>::isComplex) {
75 T *i = ptr + s;
76 while (i-- != ptr)
77 i->~T();
78 }
79 if (ptr != reinterpret_cast<T *>(array))
80 free(ptr);
81 }
82 inline QVarLengthArray<T, Prealloc> &operator=(const QVarLengthArray<T, Prealloc> &other)
83 {
84 if (this != &other) {
85 clear();
86 append(other.constData(), other.size());
87 }
88 return *this;
89 }
90
91 inline void removeLast() {
92 Q_ASSERT(s > 0);
93 realloc(s - 1, a);
94 }
95 inline int size() const { return s; }
96 inline int count() const { return s; }
97 inline int length() const { return s; }
98 inline T& first() { Q_ASSERT(!isEmpty()); return *begin(); }
99 inline const T& first() const { Q_ASSERT(!isEmpty()); return *begin(); }
100 T& last() { Q_ASSERT(!isEmpty()); return *(end() - 1); }
101 const T& last() const { Q_ASSERT(!isEmpty()); return *(end() - 1); }
102 inline bool isEmpty() const { return (s == 0); }
103 inline void resize(int size);
104 inline void clear() { resize(0); }
105 inline void squeeze();
106
107 inline int capacity() const { return a; }
108 inline void reserve(int size);
109
110 inline int indexOf(const T &t, int from = 0) const;
111 inline int lastIndexOf(const T &t, int from = -1) const;
112 inline bool contains(const T &t) const;
113
114 inline T &operator[](int idx) {
115 Q_ASSERT(idx >= 0 && idx < s);
116 return ptr[idx];
117 }
118 inline const T &operator[](int idx) const {
119 Q_ASSERT(idx >= 0 && idx < s);
120 return ptr[idx];
121 }
122 inline const T &at(int idx) const { return operator[](idx); }
123
124 T value(int i) const;
125 T value(int i, const T &defaultValue) const;
126
127 inline void append(const T &t) {
128 if (s == a) // i.e. s != 0
129 realloc(s, s<<1);
130 const int idx = s++;
131 if (QTypeInfo<T>::isComplex) {
132 new (ptr + idx) T(t);
133 } else {
134 ptr[idx] = t;
135 }
136 }
137 void append(const T *buf, int size);
138 inline QVarLengthArray<T, Prealloc> &operator<<(const T &t)
139 { append(t); return *this; }
140 inline QVarLengthArray<T, Prealloc> &operator+=(const T &t)
141 { append(t); return *this; }
142
143 void prepend(const T &t);
144 void insert(int i, const T &t);
145 void insert(int i, int n, const T &t);
146 void replace(int i, const T &t);
147 void remove(int i);
148 void remove(int i, int n);
149
150
151 inline T *data() { return ptr; }
152 inline const T *data() const { return ptr; }
153 inline const T * constData() const { return ptr; }
154 typedef int size_type;
155 typedef T value_type;
156 typedef value_type *pointer;
157 typedef const value_type *const_pointer;
158 typedef value_type &reference;
159 typedef const value_type &const_reference;
160 typedef qptrdiff difference_type;
161
162
163 typedef T* iterator;
164 typedef const T* const_iterator;
165
166 inline iterator begin() { return ptr; }
167 inline const_iterator begin() const { return ptr; }
168 inline const_iterator cbegin() const { return ptr; }
169 inline const_iterator constBegin() const { return ptr; }
170 inline iterator end() { return ptr + s; }
171 inline const_iterator end() const { return ptr + s; }
172 inline const_iterator cend() const { return ptr + s; }
173 inline const_iterator constEnd() const { return ptr + s; }
174 iterator insert(const_iterator before, int n, const T &x);
175 inline iterator insert(const_iterator before, const T &x) { return insert(before, 1, x); }
176 iterator erase(const_iterator begin, const_iterator end);
177 inline iterator erase(const_iterator pos) { return erase(pos, pos+1); }
178
179 // STL compatibility:
180 inline bool empty() const { return isEmpty(); }
181 inline void push_back(const T &t) { append(t); }
182 inline void pop_back() { removeLast(); }
183 inline T &front() { return first(); }
184 inline const T &front() const { return first(); }
185 inline T &back() { return last(); }
186 inline const T &back() const { return last(); }
187
188private:
189 friend class QPodList<T, Prealloc>;
190 void realloc(int size, int alloc);
191
192 int a; // capacity
193 int s; // size
194 T *ptr; // data
195 union {
196 char array[Prealloc * sizeof(T)];
197 qint64 q_for_alignment_1;
198 double q_for_alignment_2;
199 };
200
201 bool isValidIterator(const const_iterator &i) const
202 {
203 return (i <= constEnd()) && (constBegin() <= i);
204 }
205};

207template <class T, int Prealloc>
208Q_INLINE_TEMPLATE QVarLengthArray<T, Prealloc>::QVarLengthArray(int asize)
209 : s(asize) {
210 Q_STATIC_ASSERT_X(Prealloc > 0, "QVarLengthArray Prealloc must be greater than 0.");
211 Q_ASSERT_X(s >= 0, "QVarLengthArray::QVarLengthArray()", "Size must be greater than or equal to 0.");
212 if (s > Prealloc) {
213 ptr = reinterpret_cast<T *>(malloc(s * sizeof(T)));
214 Q_CHECK_PTR(ptr);
215 a = s;
216 } else {
217 ptr = reinterpret_cast<T *>(array);
218 a = Prealloc;
219 }
220 if (QTypeInfo<T>::isComplex) {
221 T *i = ptr + s;
222 while (i != ptr)
223 new (--i) T;
224 }
225}

227template <class T, int Prealloc>
228Q_INLINE_TEMPLATE void QVarLengthArray<T, Prealloc>::resize(int asize)
229{ realloc(asize, qMax(asize, a)); }
230
231template <class T, int Prealloc>
232Q_INLINE_TEMPLATE void QVarLengthArray<T, Prealloc>::reserve(int asize)
233{ if (asize > a) realloc(s, asize); }
234
235template <class T, int Prealloc>
236Q_INLINE_TEMPLATE int QVarLengthArray<T, Prealloc>::indexOf(const T &t, int from) const
237{
238 if (from < 0)
239 from = qMax(from + s, 0);
240 if (from < s) {
241 T *n = ptr + from - 1;
242 T *e = ptr + s;
243 while (++n != e)
244 if (*n == t)
245 return n - ptr;
246 }
247 return -1;
248}
249
250template <class T, int Prealloc>
251Q_INLINE_TEMPLATE int QVarLengthArray<T, Prealloc>::lastIndexOf(const T &t, int from) const
252{
253 if (from < 0)
254 from += s;
255 else if (from >= s)
256 from = s - 1;
257 if (from >= 0) {
258 T *b = ptr;
259 T *n = ptr + from + 1;
260 while (n != b) {
261 if (*--n == t)
262 return n - b;
263 }
264 }
265 return -1;
266}

268template <class T, int Prealloc>
269Q_INLINE_TEMPLATE bool QVarLengthArray<T, Prealloc>::contains(const T &t) const
270{
271 T *b = ptr;
272 T *i = ptr + s;
273 while (i != b) {
274 if (*--i == t)
275 return true;
276 }
277 return false;
278}

280template <class T, int Prealloc>
281Q_OUTOFLINE_TEMPLATE void QVarLengthArray<T, Prealloc>::append(const T *abuf, int increment)
282{
283 Q_ASSERT(abuf);
284 if (increment <= 0)
285 return;
286
287 const int asize = s + increment;
288
289 if (asize >= a)
290 realloc(s, qMax(s*2, asize));
291
292 if (QTypeInfo<T>::isComplex) {
293 // call constructor for new objects (which can throw)
294 while (s < asize)
295 new (ptr+(s++)) T(*abuf++);
296 } else {
297 memcpy(&ptr[s], abuf, increment * sizeof(T));
298 s = asize;
299 }
300}

302template <class T, int Prealloc>
303Q_INLINE_TEMPLATE void QVarLengthArray<T, Prealloc>::squeeze()
304{ realloc(s, s); }

306template <class T, int Prealloc>
307Q_OUTOFLINE_TEMPLATE void QVarLengthArray<T, Prealloc>::realloc(int asize, int aalloc)
308{
309 Q_ASSERT(aalloc >= asize);
310 T *oldPtr = ptr;
311 int osize = s;
312
313 const int copySize = qMin(asize, osize);
314 if (aalloc != a) {
315 if (aalloc > Prealloc) {
316 T* newPtr = reinterpret_cast<T *>(malloc(aalloc * sizeof(T)));
317 Q_CHECK_PTR(newPtr); // could throw
318 // by design: in case of QT_NO_EXCEPTIONS malloc must not fail or it crashes here
319 ptr = newPtr;
320 a = aalloc;
321 } else {
322 ptr = reinterpret_cast<T *>(array);
323 a = Prealloc;
324 }
325 s = 0;
326 if (QTypeInfo<T>::isStatic) {
327 QT_TRY {
328 // copy all the old elements
329 while (s < copySize) {
330 new (ptr+s) T(*(oldPtr+s));
331 (oldPtr+s)->~T();
332 s++;
333 }
334 } QT_CATCH(...) {
335 // clean up all the old objects and then free the old ptr
336 int sClean = s;
337 while (sClean < osize)
338 (oldPtr+(sClean++))->~T();
339 if (oldPtr != reinterpret_cast<T *>(array) && oldPtr != ptr)
340 free(oldPtr);
341 QT_RETHROW;
342 }
343 } else {
344 memcpy(ptr, oldPtr, copySize * sizeof(T));
345 }
346 }
347 s = copySize;
348
349 if (QTypeInfo<T>::isComplex) {
350 // destroy remaining old objects
351 while (osize > asize)
352 (oldPtr+(--osize))->~T();
353 }
354
355 if (oldPtr != reinterpret_cast<T *>(array) && oldPtr != ptr)
356 free(oldPtr);
357
358 if (QTypeInfo<T>::isComplex) {
359 // call default constructor for new objects (which can throw)
360 while (s < asize)
361 new (ptr+(s++)) T;
362 } else {
363 s = asize;
364 }
365}

367template <class T, int Prealloc>
368Q_OUTOFLINE_TEMPLATE T QVarLengthArray<T, Prealloc>::value(int i) const
369{
370 if (uint(i) >= uint(size())) {
371 return T();
372 }
373 return at(i);
374}
375template <class T, int Prealloc>
376Q_OUTOFLINE_TEMPLATE T QVarLengthArray<T, Prealloc>::value(int i, const T &defaultValue) const
377{
378 return (uint(i) >= uint(size())) ? defaultValue : at(i);
379}

401template <class T, int Prealloc>
402inline void QVarLengthArray<T, Prealloc>::replace(int i, const T &t)
403{
404 Q_ASSERT_X(i >= 0 && i < s, "QVarLengthArray::replace", "index out of range");

406 data()[i] = copy;

410template <class T, int Prealloc>
411Q_OUTOFLINE_TEMPLATE typename QVarLengthArray<T, Prealloc>::iterator QVarLengthArray<T, Prealloc>::insert(const_iterator before, size_type n, const T &t)
412{
413 Q_ASSERT_X(isValidIterator(before), "QVarLengthArray::insert", "The specified const_iterator argument 'before' is invalid");
414
415 int offset = int(before - ptr);
416 if (n != 0) {
417 resize(s + n);
418 const T copy(t);
419 if (QTypeInfo<T>::isStatic) {
420 T *b = ptr + offset;
421 T *j = ptr + s;
422 T *i = j - n;
423 while (i != b)
424 *--j = *--i;
425 i = b + n;
426 while (i != b)
427 *--i = copy;
428 } else {
429 T *b = ptr + offset;
430 T *i = b + n;
431 memmove(i, b, (s - offset - n) * sizeof(T));
432 while (i != b)
433 new (--i) T(copy);
434 }
435 }
436 return ptr + offset;
437}

439template <class T, int Prealloc>
440Q_OUTOFLINE_TEMPLATE typename QVarLengthArray<T, Prealloc>::iterator QVarLengthArray<T, Prealloc>::erase(const_iterator abegin, const_iterator aend)
441{
442 Q_ASSERT_X(isValidIterator(abegin), "QVarLengthArray::insert", "The specified const_iterator argument 'abegin' is invalid");
443 Q_ASSERT_X(isValidIterator(aend), "QVarLengthArray::insert", "The specified const_iterator argument 'aend' is invalid");
444
445 int f = int(abegin - ptr);
446 int l = int(aend - ptr);
447 int n = l - f;
448 if (QTypeInfo<T>::isComplex) {
449 std::copy(ptr + l, ptr + s, ptr + f);
450 T *i = ptr + s;
451 T *b = ptr + s - n;
452 while (i != b) {
453 --i;
454 i->~T();
455 }
456 } else {
457 memmove(ptr + f, ptr + l, (s - l) * sizeof(T));
458 }
459 s -= n;
460 return ptr + f;
461}

463template <typename T, int Prealloc1, int Prealloc2>
464bool operator==(const QVarLengthArray<T, Prealloc1> &l, const QVarLengthArray<T, Prealloc2> &r)
465{
466 if (l.size() != r.size())
467 return false;
468 for (int i = 0; i < l.size(); i++) {
469 if (l.at(i) != r.at(i))
470 return false;
471 }
472 return true;
473}

475template <typename T, int Prealloc1, int Prealloc2>
476bool operator!=(const QVarLengthArray<T, Prealloc1> &l, const QVarLengthArray<T, Prealloc2> &r)
477{
478 return !(l == r);
479}
