17#ifndef SOLARUS_PIXEL_BITS_H
18#define SOLARUS_PIXEL_BITS_H
19
20#include "solarus/Common.h"
21#include <cstdint>
22#include <vector>
23
24namespace Solarus {
25
26class Point;
27class Rectangle;
28class Surface;
29
30/**
31 * \brief Provides pixel-perfect collision checks for a surface.
32 *
33 * This class stores efficiently the location of the non-transparent pixels of a surface.
34 * For each pixel of the image, a bit indicates whether this pixel is transparent.
35 * This class perform fast pixel-perfect collision checks.
36 */
37class PixelBits {
38
39 public:
40
41 PixelBits(const Surface& surface, const Rectangle& image_position);
42
43 bool test_collision(const PixelBits& other,
44 const Point& location1, const Point& location2) const;
45
46 private:
47
48 void print() const;
49 void print_mask(uint32_t mask) const;
50
51 int width; /**< width of the image in pixels */
52 int height; /**< height of the image in pixels */
53 int nb_integers_per_row; /**< number of uint32_t necessary to store
54 * the bits of a row of the image */
55
56 std::vector<std::vector<uint32_t>>
57 bits; /**< A two-dimensional array representing the
58 * transparency bit of each pixel in the image. */
59
60};
61
62}
63
64#endif
