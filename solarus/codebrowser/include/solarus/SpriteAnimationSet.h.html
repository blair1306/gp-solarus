17#ifndef SOLARUS_SPRITE_ANIMATION_SET_H
18#define SOLARUS_SPRITE_ANIMATION_SET_H
19
20#include "solarus/Common.h"
21#include "solarus/lowlevel/Rectangle.h"
22#include "solarus/lowlevel/Size.h"
23#include <map>
24#include <string>
25
26struct lua_State;
27
28namespace Solarus {
29
30class SpriteAnimation;
31class SpriteAnimationData;
32class Tileset;
33
34/**
35 * \brief A set of animations representing a sprite.
36 *
37 * Each animation correspond to a specific situation of a sprite
38 * and is an instance of SpriteAnimation.
39 * For example, an NPC usually has an animation "stopped"
40 * and an animation "walking".
41 */
42class SpriteAnimationSet {
43
44 public:
45
46 explicit SpriteAnimationSet(const std::string& id);
47
48 void set_tileset(const Tileset& tileset);
49
50 bool has_animation(const std::string& animation_name) const;
51 const SpriteAnimation& get_animation(const std::string& animation_name) const;
52 SpriteAnimation& get_animation(const std::string& animation_name);
53 const std::string& get_default_animation() const;
54
55 void enable_pixel_collisions();
56 bool are_pixel_collisions_enabled() const;
57 const Size& get_max_size() const;
58 const Rectangle& get_max_bounding_box() const;
59
60 private:
61
62 void load();
63
64 void add_animation(const std::string& animation_name,
65 const SpriteAnimationData& animation_data);
66
67 std::string id; /**< Id of this animation set. */
68 std::map<std::string, SpriteAnimation>
69 animations; /**< The animations. */
70 std::string default_animation_name; /**< Name of the default animation. */
71 Size max_size; /**< Size of this biggest frame. */
72 Rectangle max_bounding_box; /**< Rectangle big enough to contain any frame.
73 * Can be larger than max_size if
74 * the origin changes. */
75
76};
77
78}
79
80#endif
