17#include "solarus/lua/ScopedLuaRef.h"
18#include "solarus/lua/LuaTools.h"
19#include <lua.hpp>

21namespace Solarus {
22
23/**
24 * \brief Creates an empty scoped Lua ref.
25 */
26ScopedLuaRef::ScopedLuaRef():
27 l(nullptr),
28 ref(LUA_REFNIL) {
29}
30
31/**
32 * \brief Creates a scoped Lua ref.
33 * \param l The Lua state (cannot be nullptr).
34 * \param ref The Lua ref, possibly LUA_REFNIL or LUA_NOREF.
35 */
36ScopedLuaRef::ScopedLuaRef(lua_State* l, int ref):
37 l(l),
38 ref(ref) {
39 Debug::check_assertion(l != nullptr, "Missing Lua state");
40}
41
42/**
43 * \brief Copy constructor.
44 *
45 * This creates a new reference to the same Lua value.
46 *
47 * \param other The object to copy.
48 */
49ScopedLuaRef::ScopedLuaRef(const ScopedLuaRef& other):
50 l(other.l),
51 ref(LUA_REFNIL) {
52
53 *this = other;
54}
55
56/**
57 * \brief Move constructor.
58 *
59 * This transfers the ref to the new object and leaves the existing one empty.
60 *
61 * \param other The object to move.
62 */
63ScopedLuaRef::ScopedLuaRef(ScopedLuaRef&& other):
64 l(other.l),
65 ref(other.ref) {
66
67 other.l = nullptr;
68 other.ref = LUA_REFNIL; // Don't unref from the other one.
69}
70
71/**
72 * \brief Destroys the ref.
73 */
74ScopedLuaRef::~ScopedLuaRef() {
75 clear();
76}
77
78/**
79 * \brief Copy assignment operator.
80 * \param other The object to copy.
81 * \return The current ref after modification.
82 */
83ScopedLuaRef& ScopedLuaRef::operator=(const ScopedLuaRef& other) {
84
85 clear();
86 this->l = other.l;
87 if (l != nullptr) {
88 if (other.ref == LUA_REFNIL || other.ref == LUA_NOREF) {
89 this->ref = other.ref;
90 }
91 else {
92 // Get the value and make another ref of it.
93 lua_rawgeti(l, LUA_REGISTRYINDEX, other.ref);
94 this->ref = luaL_ref(l, LUA_REGISTRYINDEX);
95 }
96 }
97
98 return *this;
99}
100
101/**
102 * \brief Move assignment operator.
103 * \param other The object to move.
104 * \return The current ref after modification.
105 */
106ScopedLuaRef& ScopedLuaRef::operator=(ScopedLuaRef&& other) {
107
108 clear();
109 this->l = other.l;
110 this->ref = other.ref;
111 other.l = nullptr;
112 other.ref = LUA_REFNIL; // Don't unref from the other one.
113
114 return *this;
115}

117/**
118 * \brief Returns the Lua state this ref lives in.
119 * \return The Lua state (nullptr means that the ref is empty).
120 */
121lua_State* ScopedLuaRef::get_lua_state() const {
122 return l;
123}
124
125/**
126 * \brief Returns whether this ref is empty.
127 * \return \c true if the ref is empty.
128 */
129bool ScopedLuaRef::is_empty() const {
130
131 return l == nullptr || ref == LUA_REFNIL || ref == LUA_NOREF;
132}
133
134/**
135 * \brief Returns the ref.
136 * \return The encapsulated ref, possibly LUA_REFNIL or LUA_NOREF.
137 */
138int ScopedLuaRef::get() const {
139 return ref;
140}
141
142/**
143 * \brief Destroys the ref.
144 *
145 * This calls luaL_unref().
146 */
147void ScopedLuaRef::clear() {
148
149 if (l != nullptr) {
150 luaL_unref(l, LUA_REGISTRYINDEX, ref);
151 }
152 l = nullptr;
153 ref = LUA_REFNIL;
154}
155
156/**
157 * \brief Pushes the referenced value onto the Lua stack.
158 *
159 * The ref must not be empty.
160 */
161void ScopedLuaRef::push() const {
162
163 Debug::check_assertion(!is_empty(), "Attempt to push an empty ref");
164
165 lua_rawgeti(l, LUA_REGISTRYINDEX, ref);
166}
167
168/**
169 * \brief Calls the referenced function.
170 *
171 * If the referenced value is not empty, it must be a function.
172 * No parameters are passed to the function, and return values are ignored.
173 *
174 * \param function_name A name describing the Lua function (only used to
175 * print the error message if any).
176 */
177void ScopedLuaRef::call(const std::string& function_name) const {
178
179 if (!is_empty()) {
180 push();
181 LuaTools::call_function(l, 0, 0, function_name.c_str());
182 }
183}
184
185/**
186 * \brief Like call(), but clears the reference before calling
187 * the function.
188 *
189 * This avoids reentrant calls: your reference will become clean before
190 * the function is called.
191 *
192 * This is equivalent to:
193 *
194 * ScopedLuaRef copy = callback_ref;
195 * callback_ref.clear(); // Avoid nasty problems in case of reentrant calls.
196 * copy.call(s);
197 *
198 * If the reference is already empty, nothing happens.
199 * \param function_name A name describing the Lua function (only used to
200 * print the error message if any).
201 */
202void ScopedLuaRef::clear_and_call(const std::string& function_name) {
203
204 if (is_empty()) {
205 return;
206 }
207
208 lua_State* l = this->l;
209 push();
210 clear(); // The function is still alive, onto the stack.
211 LuaTools::call_function(l, 0, 0, function_name.c_str());
212}
