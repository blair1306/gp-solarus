42#include <QtCore/qglobal.h>

44#ifndef QSYSINFO_H
45#define QSYSINFO_H
46
47QT_BEGIN_NAMESPACE
48
49/*
50 System information
51*/
52
53class QString;
54class Q_CORE_EXPORT QSysInfo {
55public:
56 enum Sizes {
57 WordSize = (sizeof(void *)<<3)
58 };
59
60#if defined(QT_BUILD_QMAKE)
61 enum Endian {
62 BigEndian,
63 LittleEndian
64 };
65 /* needed to bootstrap qmake */
66 static const int ByteOrder;
67#elif defined(Q_BYTE_ORDER)
68 enum Endian {
69 BigEndian,
70 LittleEndian
71
72# ifdef Q_QDOC
73 , ByteOrder = <platform-dependent>
74# elif Q_BYTE_ORDER == Q_BIG_ENDIAN
75 , ByteOrder = BigEndian
76# elif Q_BYTE_ORDER == Q_LITTLE_ENDIAN
77 , ByteOrder = LittleEndian
78# else
79# error "Undefined byte order"
80# endif
81 };
82#endif
83#if defined(Q_OS_WIN) || defined(Q_OS_CYGWIN)
84 enum WinVersion {
85 WV_32s = 0x0001,
86 WV_95 = 0x0002,
87 WV_98 = 0x0003,
88 WV_Me = 0x0004,
89 WV_DOS_based= 0x000f,
90
91 /* codenames */
92 WV_NT = 0x0010,
93 WV_2000 = 0x0020,
94 WV_XP = 0x0030,
95 WV_2003 = 0x0040,
96 WV_VISTA = 0x0080,
97 WV_WINDOWS7 = 0x0090,
98 WV_WINDOWS8 = 0x00a0,
99 WV_WINDOWS8_1 = 0x00b0,
100 WV_NT_based = 0x00f0,
101
102 /* version numbers */
103 WV_4_0 = WV_NT,
104 WV_5_0 = WV_2000,
105 WV_5_1 = WV_XP,
106 WV_5_2 = WV_2003,
107 WV_6_0 = WV_VISTA,
108 WV_6_1 = WV_WINDOWS7,
109 WV_6_2 = WV_WINDOWS8,
110 WV_6_3 = WV_WINDOWS8_1,
111
112 WV_CE = 0x0100,
113 WV_CENET = 0x0200,
114 WV_CE_5 = 0x0300,
115 WV_CE_6 = 0x0400,
116 WV_CE_based = 0x0f00
117 };
118 static const WinVersion WindowsVersion;
119 static WinVersion windowsVersion();
120
121#endif
122#ifdef Q_OS_MAC
123# define Q_MV_IOS(major, minor) (QSysInfo::MV_IOS | major << 4 | minor)
124 enum MacVersion {
125 MV_Unknown = 0x0000,
126
127 /* version */
128 MV_9 = 0x0001,
129 MV_10_0 = 0x0002,
130 MV_10_1 = 0x0003,
131 MV_10_2 = 0x0004,
132 MV_10_3 = 0x0005,
133 MV_10_4 = 0x0006,
134 MV_10_5 = 0x0007,
135 MV_10_6 = 0x0008,
136 MV_10_7 = 0x0009,
137 MV_10_8 = 0x000A,
138 MV_10_9 = 0x000B,
139
140 /* codenames */
141 MV_CHEETAH = MV_10_0,
142 MV_PUMA = MV_10_1,
143 MV_JAGUAR = MV_10_2,
144 MV_PANTHER = MV_10_3,
145 MV_TIGER = MV_10_4,
146 MV_LEOPARD = MV_10_5,
147 MV_SNOWLEOPARD = MV_10_6,
148 MV_LION = MV_10_7,
149 MV_MOUNTAINLION = MV_10_8,
150 MV_MAVERICKS = MV_10_9,
151
152 /* iOS */
153 MV_IOS = 1 << 8,
154 MV_IOS_4_3 = Q_MV_IOS(4, 3),
155 MV_IOS_5_0 = Q_MV_IOS(5, 0),
156 MV_IOS_5_1 = Q_MV_IOS(5, 1),
157 MV_IOS_6_0 = Q_MV_IOS(6, 0),
158 MV_IOS_6_1 = Q_MV_IOS(6, 1),
159 MV_IOS_7_0 = Q_MV_IOS(7, 0),
160 MV_IOS_7_1 = Q_MV_IOS(7, 1)
161 };
162 static const MacVersion MacintoshVersion;
163 static MacVersion macVersion();
164#endif
165};
