259inline int QHeaderView::logicalIndexAt(int ax, int ay) const
260{ return orientation() == Qt::Horizontal ? logicalIndexAt(ax) : logicalIndexAt(ay); }

263inline void QHeaderView::hideSection(int alogicalIndex)
