322inline bool QFont::bold() const
323{ return weight() > Normal; }
324
325
326inline void QFont::setBold(bool enable)
327{ setWeight(enable ? Bold : Normal); }
328
329inline bool QFont::italic() const
330{
331 return (style() != StyleNormal);
332}
333
334inline void QFont::setItalic(bool b) {
335 setStyle(b ? StyleItalic : StyleNormal);
336}
