23/**
24 * \brief Converts an angle in radians into an angle in degrees.
25 * \param radians Angle in radians.
26 * \return The degrees value.
27 */
28int radians_to_degrees(double radians) {
29 return (int) (radians * 360.0 / TWO_PI);
30}
31
32/**
33 * \brief Converts an angle in degrees into an angle in radians.
34 * \param degrees Angle in degrees.
35 * \return The radians value.
36 */
37double degrees_to_radians(double degrees) {
38 return degrees * TWO_PI / 360.0;
39}
40
41/**
42 * \brief Returns the angle of the vector formed by two points.
43 * \param x1 X coordinate of the first point.
44 * \param y1 Y coordinate of the first point.
45 * \param x2 X coordinate of the second point.
46 * \param y2 Y coordinate of the second point.
47 * \return The angle in radians, between 0 and TWO_PI.
48 */
49double get_angle(double x1, double y1, double x2, double y2) {
50
51 double dx = x2 - x1;
52 double dy = y2 - y1;
53
54 // atan2 is undefined if (y == 0 and x == 0)
55 if (dx == 0.0 && dy == 0.0) {
56 return PI_OVER_2;
57 }
58
59 double angle = std::atan2(-dy, dx);
60
61 // Normalize.
62 if (angle < 0.0) {
63 angle += TWO_PI;
64 }
65
66 return angle;
67}
68
69/**
70 * \brief Returns the angle of the vector formed by two points.
71 * \param point1 First point.
72 * \param point2 Second point.
73 * \return The angle in radians, between 0 and TWO_PI.
74 */
75double get_angle(const Point& point1, const Point& point2) {
76 return get_angle(point1.x, point1.y, point2.x, point2.y);
77}
78
79/**
80 * \brief Returns the cartesian coordinates of a vector that starts from the origin, given its angle and distance.
81 * \param angle Angle of the vector in radians.
82 * \param distance Length of the vector in pixels.
83 * \return The coordinates of the second point.
84 */
85Point get_xy(double angle, int distance) {
86
87 return {
88 static_cast<int>(distance * std::cos(angle)),
89 static_cast<int>(-distance * std::sin(angle))
90 };
91}
92
93/**
94 * \brief Returns the cartesian coordinates of a vector, given its initial point, angle and distance.
95 * \param point1 Coordinates of the first point.
96 * \param angle Angle of the vector in radians.
97 * \param distance Length of the vector in pixels.
98 * \return The coordinates of the second point.
99 */
100Point get_xy(const Point& point1, double angle, int distance) {
101 return point1 + get_xy(angle, distance);
102}
