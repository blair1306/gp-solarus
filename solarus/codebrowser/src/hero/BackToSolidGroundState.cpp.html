35/**
36 * \brief Constructor.
37 * \param hero the hero controlled by this state
38 * \param use_specified_position true to get back to the place previously specified (if any),
39 * false to get back to the last coordinates with solid ground
40 * \param end_delay a delay to add at the end before returning control to the hero (default 0)
41 * \param with_sound true to play a sound when returning to solid ground
42 */
43Hero::BackToSolidGroundState::BackToSolidGroundState(
44 Hero& hero,
45 bool use_specified_position,
46 uint32_t end_delay,
47 bool with_sound):
48 HeroState(hero, "back to solid ground"),
49 target_position(),
50 end_delay(end_delay),
51 end_date(0),
52 with_sound(with_sound) {
53
54 if (use_specified_position && !hero.get_target_solid_ground_callback().is_empty()) {
55 // go back to a target position specified earlier
56 this->target_position = hero.get_target_solid_ground_callback();
57 }
58 else {
59 // just go back to the last solid ground location
60 this->target_position = hero.make_solid_ground_callback(
61 hero.get_last_solid_ground_coords(),
62 hero.get_last_solid_ground_layer()
63 );
64 }
65}

67/**
68 * \brief Starts this state.
69 * \param previous_state the previous state
70 */
71void Hero::BackToSolidGroundState::start(const State* previous_state) {
72
73 HeroState::start(previous_state);
74
75 Point xy;
76 int layer = 0;
77
78 Hero& hero = get_entity();
79 lua_State* l = hero.get_lua_context()->get_internal_state();
80
81 // Call the Lua function to get the coordinates and layer.
82 Debug::check_assertion(!target_position.is_empty(), "Missing solid ground callback");
83 target_position.push();
84 bool success = LuaTools::call_function(l, 0, 3, "Solid ground callback");
85 if (!success) {
86 // Fallback: use the last solid ground position.
87 xy = hero.get_last_solid_ground_coords();
88 layer = hero.get_last_solid_ground_layer();
89 }
90 else {
91 // Normal case: use the result of the function.
92 xy.x = LuaTools::check_int(l, -3);
93 xy.y = LuaTools::check_int(l, -2);
94 layer = LuaTools::check_int(l, -1);
95 lua_pop(l, 3);
96 }
97
98 hero.set_movement(std::make_shared<TargetMovement>(
99 nullptr, xy.x, xy.y, 144, true
100 ));
101 get_entities().set_entity_layer(hero, layer);
102
103 const std::set<std::shared_ptr<Boomerang>>& boomerangs =
104 get_entities().get_entities_by_type<Boomerang>();
105 for (const std::shared_ptr<Boomerang>& boomerang : boomerangs) {
106 boomerang->remove_from_map();
107 }
108}

121/**
122 * \brief Updates this state.
123 */
124void Hero::BackToSolidGroundState::update() {
125
126 HeroState::update();
127
128 if (is_suspended()) {
129 return;
130 }
131
132 // the current movement is an instance of TargetMovement
133 Hero& hero = get_entity();
134 if (hero.get_movement()->is_finished()) {
135
136 uint32_t now = System::now();
137 if (end_date == 0) {
138 end_date = now + end_delay;
139 get_sprites().set_animation_stopped_normal();
140 get_sprites().blink(2000);
141
142 if (with_sound) {
143 Sound::play("message_end"); // TODO rename this sound.
144 }
145 }
146
147 if (now >= end_date) {
148
149 if (get_equipment().get_life() <= 0 &&
150 !get_game().is_showing_game_over()) {
151 get_sprites().stop_blinking();
152 get_game().start_game_over();
153 return;
154 }
155
156 hero.start_state_from_ground();
157 }
158 }
159}

161void Hero::BackToSolidGroundState::set_suspended(bool suspended) {
162
163 HeroState::set_suspended(suspended);
164
165 if (!suspended && end_date != 0) {
166 end_date += System::now() - get_when_suspended();
167 }
168}
