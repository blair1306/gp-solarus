42#ifndef QTOOLBUTTON_H
43#define QTOOLBUTTON_H
44
45#include <QtWidgets/qabstractbutton.h>
46
47QT_BEGIN_NAMESPACE
48
49
50#ifndef QT_NO_TOOLBUTTON
51
52class QToolButtonPrivate;
53class QMenu;
54class QStyleOptionToolButton;
55
56class Q_WIDGETS_EXPORT QToolButton : public QAbstractButton
57{
58 Q_OBJECT
59 Q_ENUMS(Qt::ToolButtonStyle Qt::ArrowType ToolButtonPopupMode)
60#ifndef QT_NO_MENU
61 Q_PROPERTY(ToolButtonPopupMode popupMode READ popupMode WRITE setPopupMode)
62#endif
63 Q_PROPERTY(Qt::ToolButtonStyle toolButtonStyle READ toolButtonStyle WRITE setToolButtonStyle)
64 Q_PROPERTY(bool autoRaise READ autoRaise WRITE setAutoRaise)
65 Q_PROPERTY(Qt::ArrowType arrowType READ arrowType WRITE setArrowType)
66
67public:
68 enum ToolButtonPopupMode {
69 DelayedPopup,
70 MenuButtonPopup,
71 InstantPopup
72 };
73
74 explicit QToolButton(QWidget * parent=0);
75 ~QToolButton();
76
77 QSize sizeHint() const;
78 QSize minimumSizeHint() const;
79
80 Qt::ToolButtonStyle toolButtonStyle() const;
81
82 Qt::ArrowType arrowType() const;
83 void setArrowType(Qt::ArrowType type);
84
85#ifndef QT_NO_MENU
86 void setMenu(QMenu* menu);
87 QMenu* menu() const;
88
89 void setPopupMode(ToolButtonPopupMode mode);
90 ToolButtonPopupMode popupMode() const;
91#endif
92
93 QAction *defaultAction() const;
94
95 void setAutoRaise(bool enable);
96 bool autoRaise() const;
97
98public Q_SLOTS:
99#ifndef QT_NO_MENU
100 void showMenu();
101#endif
102 void setToolButtonStyle(Qt::ToolButtonStyle style);
103 void setDefaultAction(QAction *);
104
105Q_SIGNALS:
106 void triggered(QAction *);
107
108protected:
109 bool event(QEvent *e);
110 void mousePressEvent(QMouseEvent *);
111 void mouseReleaseEvent(QMouseEvent *);
112 void paintEvent(QPaintEvent *);
113 void actionEvent(QActionEvent *);
114
115 void enterEvent(QEvent *);
116 void leaveEvent(QEvent *);
117 void timerEvent(QTimerEvent *);
118 void changeEvent(QEvent *);
119
120 bool hitButton(const QPoint &pos) const;
121 void nextCheckState();
122 void initStyleOption(QStyleOptionToolButton *option) const;
123
124private:
125 Q_DISABLE_COPY(QToolButton)
126 Q_DECLARE_PRIVATE(QToolButton)
127#ifndef QT_NO_MENU
128 Q_PRIVATE_SLOT(d_func(), void _q_buttonPressed())
129 Q_PRIVATE_SLOT(d_func(), void _q_updateButtonDown())
130 Q_PRIVATE_SLOT(d_func(), void _q_menuTriggered(QAction*))
131#endif
132 Q_PRIVATE_SLOT(d_func(), void _q_actionTriggered())
133
134};
135
136#endif // QT_NO_TOOLBUTTON
137
138QT_END_NAMESPACE
139
140#endif // QTOOLBUTTON_H
