23/**
24 * \brief Creates a new separator.
25 * \param name Name of the entity to create.
26 * \param layer Layer of the entity to create on the map.
27 * \param xy Coordinate of the entity to create.
28 * \param size Size of the separator.
29 * A width of 16 means a vertical separator and a height of 16 means a
30 * horizontal one.
31 */
32Separator::Separator(
33 const std::string& name,
34 int layer,
35 const Point& xy,
36 const Size& size
37):
38 Entity(name, 0, layer, xy, size) {
39
40 Debug::check_assertion((size.width == 16 && size.height > 16)
41 || (size.width > 16 && size.height == 16), "Invalid separator size");
42
43 set_collision_modes(CollisionMode::COLLISION_CUSTOM);
44 set_layer_independent_collisions(true);
45}

54/**
55 * \brief Returns whether this is an horizontal separator.
56 * \return \c true if this separator is horizontal, \c false if it is vertical.
57 */
58bool Separator::is_horizontal() const {
59 return get_height() == 16;
60}
61
62/**
63 * \brief Returns whether this is an vertical separator.
64 * \return \c true if this separator is vertical, \c false if it is horizontal.
65 */
66bool Separator::is_vertical() const {
67 return get_width() == 16;
68}

77/**
78 * \copydoc Entity::test_collision_custom
79 */
80bool Separator::test_collision_custom(Entity& entity) {
81
82 // Trigger the collision if the center point crosses the middle of the
83 // separator.
84
85 const Point& separator_center = get_center_point();
86 const Point& center = entity.get_center_point();
87
88 if (!overlaps(center)) {
89 return false;
90 }
91
92 if (is_horizontal()) {
93 if (center.y < separator_center.y) {
94 // The entity is above the separator.
95 return center.y == separator_center.y - 1;
96 }
97 else {
98 // The entity is below the separator.
99 return center.y == separator_center.y;
100 }
101 }
102 else {
103 if (center.x < separator_center.x) {
104 // The entity is west of the separator.
105 return center.x == separator_center.x - 1;
106 }
107 else {
108 // The entity is east of the separator.
109 return center.x == separator_center.x;
110 }
111 }
112}
