56class Q_CORE_EXPORT QModelIndex
57{
58 friend class QAbstractItemModel;
59public:
60 Q_DECL_CONSTEXPR inline QModelIndex() : r(-1), c(-1), i(0), m(0) {}
61 // compiler-generated copy/move ctors/assignment operators are fine!
62 Q_DECL_CONSTEXPR inline int row() const { return r; }
63 Q_DECL_CONSTEXPR inline int column() const { return c; }
64 Q_DECL_CONSTEXPR inline quintptr internalId() const { return i; }
65 inline void *internalPointer() const { return reinterpret_cast<void*>(i); }
66 inline QModelIndex parent() const;
67 inline QModelIndex sibling(int row, int column) const;
68 inline QModelIndex child(int row, int column) const;
69 inline QVariant data(int role = Qt::DisplayRole) const;
70 inline Qt::ItemFlags flags() const;
71 Q_DECL_CONSTEXPR inline const QAbstractItemModel *model() const { return m; }
72 Q_DECL_CONSTEXPR inline bool isValid() const { return (r >= 0) && (c >= 0) && (m != 0); }
73 Q_DECL_CONSTEXPR inline bool operator==(const QModelIndex &other) const
74 { return (other.r == r) && (other.i == i) && (other.c == c) && (other.m == m); }
75 Q_DECL_CONSTEXPR inline bool operator!=(const QModelIndex &other) const
76 { return !(*this == other); }
77 Q_DECL_CONSTEXPR inline bool operator<(const QModelIndex &other) const
78 {
79 return r < other.r
80 || (r == other.r && (c < other.c
81 || (c == other.c && (i < other.i
82 || (i == other.i && m < other.m )))));
83 }
84private:
85 inline QModelIndex(int arow, int acolumn, void *ptr, const QAbstractItemModel *amodel)
86 : r(arow), c(acolumn), i(reinterpret_cast<quintptr>(ptr)), m(amodel) {}
87 Q_DECL_CONSTEXPR inline QModelIndex(int arow, int acolumn, quintptr id, const QAbstractItemModel *amodel)
88 : r(arow), c(acolumn), i(id), m(amodel) {}
89 int r, c;
90 quintptr i;
91 const QAbstractItemModel *m;
92};
93Q_DECLARE_TYPEINFO(QModelIndex, Q_MOVABLE_TYPE);

487inline QModelIndex QModelIndex::parent() const
488{ return m ? m->parent(*this) : QModelIndex(); }
489
490inline QModelIndex QModelIndex::sibling(int arow, int acolumn) const
491{ return m ? (r == arow && c == acolumn) ? *this : m->sibling(arow, acolumn, *this) : QModelIndex(); }
492
493inline QModelIndex QModelIndex::child(int arow, int acolumn) const
494{ return m ? m->index(arow, acolumn, *this) : QModelIndex(); }
495
496inline QVariant QModelIndex::data(int arole) const
497{ return m ? m->data(*this, arole) : QVariant(); }
498
499inline Qt::ItemFlags QModelIndex::flags() const
500{ return m ? m->flags(*this) : Qt::ItemFlags(0); }
501
502inline uint qHash(const QModelIndex &index)
503{ return uint((index.row() << 4) + index.column() + index.internalId()); }
