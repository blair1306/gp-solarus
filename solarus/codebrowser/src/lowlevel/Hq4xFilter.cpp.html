17#include "solarus/lowlevel/Hq4xFilter.h"
18#include "solarus/third_party/hqx/hqx.h"
19
20namespace Solarus {
21
22namespace {
23 bool hqx_initialized = false; /**< Whether the common hqx initialization was done. */
24}
25
26/**
27 * \brief Constructor.
28 */
29Hq4xFilter::Hq4xFilter():
30 PixelFilter() {
31}
32
33/**
34 * \copydoc PixelFilter::get_scaling_factor
35 */
36int Hq4xFilter::get_scaling_factor() const {
37 return 4;
38}
39
40/**
41 * \copydoc PixelFilter::filter
42 */
43void Hq4xFilter::filter(
44 const uint32_t* src,
45 int src_width,
46 int src_height,
47 uint32_t* dst) const {
48
49 // Make sure hqx is initialized.
50 initialize_hqx();
51
52 hq4x_32(const_cast<uint32_t*>(src), dst, src_width, src_height);
53}
54
55/**
56 * \brief Performs the initialization common to the 3 variants of hqx.
57 *
58 * Does nothing if the initialization was already done.
59 */
60void Hq4xFilter::initialize_hqx() {
61
62 if (!hqx_initialized) {
63 hqx_initialized = true;
64 hqxInit();
65 }
66}
