28/**
29 * \brief Constructor.
30 * \param moving_speed movement speed
31 */
32PlayerMovement::PlayerMovement(int moving_speed):
33 StraightMovement(false, true),
34 moving_speed(moving_speed),
35 direction8(-1),
36 blocked_by_stream(false) {
37
38}
39
40/**
41 * \brief Updates this movement.
42 */
43void PlayerMovement::update() {
44
45 StraightMovement::update();
46
47 const Entity* entity = get_entity();
48 if (entity == nullptr || !entity->is_on_map()) {
49 return; // the entity is not ready yet
50 }
51
52 // See if a stream blocks the control from the player.
53 // If yes, we will set the speed to zero but keep the direction
54 // wanted by the player.
55 blocked_by_stream =
56 entity->has_stream_action() &&
57 !entity->get_stream_action()->get_stream().get_allow_movement();
58
59 // Someone may have stopped the movement
60 // (e.g. Hero::reset_movement()).
61 if (is_stopped() &&
62 direction8 != -1 &&
63 !blocked_by_stream
64 ) {
65 direction8 = -1;
66 compute_movement();
67 }
68 else {
69 if (!is_stopped() && blocked_by_stream) {
70 stop();
71 }
72 // Check if the wanted direction has changed.
73 const GameCommands& commands = get_entity()->get_game().get_commands();
74 int wanted_direction8 = commands.get_wanted_direction8();
75 if (wanted_direction8 != direction8 && !is_suspended()) {
76 direction8 = wanted_direction8;
77 compute_movement();
78 }
79 }
80}
81
82/**
83 * \brief Returns the direction this movement is trying to move towards.
84 * \return the direction (0 to 7), or -1 if the player is not trying to go
85 * to a direction or the movement is disabled
86 */
87int PlayerMovement::get_wanted_direction8() const {
88 return direction8;
89}
90
91/**
92 * \brief Returns the moving speed of the entity.
93 * \return the moving speed of the entity
94 */
95int PlayerMovement::get_moving_speed() const {
96 return moving_speed;
97}
98
99/**
100 * \brief Sets the moving speed of the entity.
101 * \param moving_speed the moving speed of the entity
102 */
103void PlayerMovement::set_moving_speed(int moving_speed) {
104
105 this->moving_speed = moving_speed;
106 set_wanted_direction();
107 compute_movement();
108}
109
110/**
111 * \brief Determines the direction defined by the directional keys currently pressed
112 * and computes the corresponding movement.
113 */
114void PlayerMovement::set_wanted_direction() {
115
116 if (get_entity() != nullptr && get_entity()->is_on_map()) {
117 GameCommands& commands = get_entity()->get_game().get_commands();
118 direction8 = commands.get_wanted_direction8();
119 }
120 else {
121 direction8 = -1;
122 }
123}
124
125/**
126 * \brief Changes the movement of the entity depending on the direction wanted.
127 *
128 * This function is called when the direction is changed.
129 */
130void PlayerMovement::compute_movement() {
131
132 // Compute the speed vector corresponding to the direction wanted by the player
133
134 if (direction8 == -1) {
135 // No wanted movement.
136 stop();
137 }
138 else { // The directional keys currently pressed define a valid movement.
139 if (blocked_by_stream) {
140 stop();
141 }
142 else {
143 set_speed(moving_speed);
144 }
145 set_angle(Geometry::degrees_to_radians(direction8 * 45));
146 }
147}
