23/**
24 * @brief Creates a quests view.
25 * @param parent Parent object or nullptr.
26 */
27QuestsView::QuestsView(QWidget* parent) :
28 QListView(parent),
29 model(nullptr), itemDelegate(nullptr) {
30
31 model = new QuestsModel(this);
32 itemDelegate = new QuestsItemDelegate(this);
33 itemDelegate->setIconSize(QSize(32,32));
34 setItemDelegate(itemDelegate);
35 setModel(model);
36}
37
38/**
39 * @brief Finds the index of a quest in the list.
40 * @param path Path of the quest to look for.
41 * @return The quest index, or -1 if it is not in the list.
42 */
43int QuestsView::path_to_index(const QString& path) const {
44
45 return model->path_to_index(path);
46}
47
48/**
49 * @brief Returns the quest path at the given index.
50 * @param index Index of the quest to get.
51 * @return The path, or an empty string if the index is invalid.
52 */
53QString QuestsView::index_to_path(int index) const {
54
55 return model->index_to_path(index);
56}

58/**
59 * @brief Returns the number of quests in this view.

62int QuestsView::get_num_quests() const {

67/**

107/**
108 * @brief Returns the properties of the selected quest.
109 * @return The quest properties, or a default-constructed object
110 * if no quest is selected.
111 */
112Solarus::QuestProperties QuestsView::get_selected_quest_properties() const {
113
114 int index = get_selected_index();
115 if (index == -1) {
116 return Solarus::QuestProperties();
117 }
118 return get_quest_properties(index);
119}

131/**
132 * @brief Returns the logo of the selected quest.
133 * @return The quest logo, or the default image if no quest is selected.
134 */
135const QPixmap& QuestsView::get_selected_logo() const {
136

154/**
155 * @brief Returns the index of the currently selected quest.
156 * @return Return current index, or -1 if no quest is selected.
157 */
158int QuestsView::get_selected_index() const {
159
160 QModelIndexList model_indexes = selectionModel()->selectedIndexes();
161 if (model_indexes.isEmpty()) {
162 return -1;
163 }
164 return model_indexes.first().row();
165}
166
167/**
168 * @brief Returns the quest path currently selected.
169 * @return The current quest path, or an empty string if no quest is selected.
170 */
171QString QuestsView::get_selected_path() const {
172
173 return index_to_path(get_selected_index());
174}
175
176/**
177 * @brief Selects a quest.
178 * @param index Index of the quest to make selected.
179 */
180void QuestsView::select_quest(int index) {
181
182 if (index < 0 || index >= get_num_quests()) {
183 return;
184 }
185
186 selectionModel()->select(model->index(index, 0), QItemSelectionModel::ClearAndSelect);
187}
188
189/**
190 * @brief Selects the given quest if it is in the view.
191 * @param path Path of the quest to make selected.
192 */
193void QuestsView::select_quest(const QString& path) {
194
195 int index = path_to_index(path);
196 if (index == -1) {
197 return;
198 }
199 select_quest(index);
200}
