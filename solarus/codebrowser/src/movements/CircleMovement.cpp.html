29/**
30 * \brief Creates a circle movement.
31 * \param ignore_obstacles \c true to ignore obstacles.
32 */
33CircleMovement::CircleMovement(bool ignore_obstacles):
34
35 Movement(ignore_obstacles),
36 center_entity(nullptr),
37 current_angle(0),
38 initial_angle(0),
39 angle_increment(1),
40 next_angle_change_date(System::now()),
41 angle_change_delay(5),
42 current_radius(0),
43 wanted_radius(0),
44 previous_radius(0),
45 radius_increment(0),
46 next_radius_change_date(System::now()),
47 radius_change_delay(0),
48 duration(0),
49 end_movement_date(0),
50 max_rotations(0),
51 nb_rotations(0),
52 loop_delay(0),
53 restart_date(System::now()) {
54
55}

57/**
58 * \brief Sets the center of the circles.
59 *
60 * The movement will make circles around the specified fixed point.
61 *
62 * \param center_point Center of the circles to make.
63 */
64void CircleMovement::set_center(const Point& center_point) {
65
66 this->center_entity = nullptr;
67 this->center_point = center_point;
68 recompute_position();
69}

71/**
72 * \brief Sets the center of the circles.
73 *
74 * The movement will make circles around the specified (possibly moving) entity.
75 *
76 * \param center_entity the entity around which you want to make circles
77 * \param x X coordinate of where the center should be placed relative to the entity's origin.
78 * \param y Y coordinate of where the center should be placed relative to the entity's origin.
79 */
80void CircleMovement::set_center(
81 const EntityPtr& center_entity,
82 int x,
83 int y
84) {
85 this->center_entity = center_entity;
86 this->center_point = { x, y };
87 recompute_position();
88}

91 * \brief Returns the radius of the circles.
92 * \return The radius in pixels.
93 */
94int CircleMovement::get_radius() const {
95 return wanted_radius;
96}

98/**
99 * \brief Sets the radius of the circles.
100 * \param radius The radius in pixels.
101 */
102void CircleMovement::set_radius(int radius) {
103
104 if (radius < 0) {
105 std::ostringstream oss;
106 oss << "Invalid radius: " << radius;
107 Debug::die(oss.str());
108 }
109
110 this->wanted_radius = radius;
111 if (radius_change_delay == 0) {
112 if (is_started()) {
113 this->current_radius = wanted_radius;
114 }
115 }
116 else {
117 this->radius_increment = (radius > this->current_radius) ? 1 : -1;
118 if (is_started()) {
119 this->next_radius_change_date = System::now();
120 }
121 }
122 recompute_position();
123}

125/**
126 * \brief Returns the speed of the radius variations.
127 * \return The speed in pixels per second, or 0 if radius variations are immediate.
128 */
129int CircleMovement::get_radius_speed() const {
130
131 return radius_change_delay == 0 ? 0 : 1000 / radius_change_delay;
132}
133
134/**
135 * \brief Sets the radius to be updated immediately or gradually towards its wanted value.
136 *
137 * Use set_radius() to specify the wanted value.
138 *
139 * \param radius_speed Speed of the radius variation (number of pixels per second),
140 * or 0 to update it immediately.
141 */
142void CircleMovement::set_radius_speed(int radius_speed) {
143
144 if (radius_speed < 0) {
145 std::ostringstream oss;
146 oss << "Invalid radius speed: " << radius_speed;
147 Debug::die(oss.str());
148 }
149
150 if (radius_speed == 0) {
151 this->radius_change_delay = 0;
152 }
153 else {
154 this->radius_change_delay = 1000 / radius_speed;
155 }
156
157 set_radius(wanted_radius);
158}

160/**
161 * \brief Returns the speed of the angle variation.
162 * \return the number of degrees made per second
163 */
164int CircleMovement::get_angle_speed() const {
165 return 1000 / angle_change_delay;
166}
167
168/**
169 * \brief Sets the speed of the angle variation.
170 * \param angle_speed number of degrees to make per second
171 */
172void CircleMovement::set_angle_speed(int angle_speed) {
173
174 if (angle_speed <= 0) {
175 std::ostringstream oss;
176 oss << "Invalid angle speed: " << angle_speed;
177 Debug::die(oss.str());
178 }
179
180 this->angle_change_delay = 1000 / angle_speed;
181 this->next_angle_change_date = System::now();
182 recompute_position();
183}

194/**
195 * \brief Sets the angle from where the first circle starts.
196 * \param initial_angle Angle in degrees.
197 */
198void CircleMovement::set_initial_angle(int initial_angle) {
199
200 // Normalize the angle.
201 initial_angle = initial_angle % 360;
202 if (initial_angle < 0) {
203 initial_angle += 360;
204 }
205
206 this->initial_angle = initial_angle;
207
208 start();
209}

304/**
305 * \brief Makes the movement restart after a delay when it is finished.
306 * \param delay the movement will restart after this delay in milliseconds (0 to avoid this)
307 */
308void CircleMovement::set_loop(uint32_t delay) {
309
310 this->loop_delay = delay;
311 if (delay != 0 && is_stopped()) {
312 this->restart_date = System::now() + delay;

316/**
317 * \brief Updates the movement.
318 */
319void CircleMovement::update() {
320
321 if (center_entity != nullptr && center_entity->is_being_removed()) {
322 set_center(Point(
323 center_entity->get_x() + center_point.x,
324 center_entity->get_y() + center_point.y));
325 }
326
327 if (is_suspended()) {
328 return;
329 }
330
331 bool update_needed = false;
332 uint32_t now = System::now();
333
334 // maybe it is time to stop or to restart
335 if (current_radius != 0 &&
336 duration != 0 &&
337 now >= end_movement_date &&
338 wanted_radius != 0) {
339 stop();
340 }
341 else if (current_radius == 0 &&
342 loop_delay != 0 &&
343 now >= restart_date &&
344 wanted_radius == 0
345 ) {
346 set_radius(previous_radius);
347 start();
348 }
349
350 // update the angle
351 if (is_started()) {
352 while (now >= next_angle_change_date) {
353
354 current_angle += angle_increment;
355 current_angle = (360 + current_angle) % 360;
356 if (current_angle == initial_angle) {
357 nb_rotations++;
358
359 if (nb_rotations == max_rotations) {
360 stop();
361 }
362 }
363
364 next_angle_change_date += angle_change_delay;
365 update_needed = true;
366 }
367 }
368
369 // update the radius
370 while (current_radius != wanted_radius &&
371 now >= next_radius_change_date) {
372
373 current_radius += radius_increment;
374
375 next_radius_change_date += radius_change_delay;
376 update_needed = true;
377 }
378
379 // the center may have moved
380 if (center_entity != nullptr) {
381 update_needed = true;
382 }
383
384 if (update_needed) {
385 recompute_position();
386 }
387
388 // Do this at last so that Movement::update() knows whether we are finished.
389 Movement::update();
390}

392/**
393 * \brief Computes the position of the object controlled by this movement.
394 *
395 * This function should be called whenever the angle, the radius or the center changes.
396 */
397void CircleMovement::recompute_position() {
398
399 Point center = this->center_point;
400 if (center_entity != nullptr) {
401 center += center_entity->get_xy();
402 }
403
404 Point xy = Geometry::get_xy(center, Geometry::degrees_to_radians(current_angle), current_radius);
405 if (get_entity() == nullptr
406 || !test_collision_with_obstacles(xy - get_entity()->get_xy())) {
407 set_xy(xy);
408 notify_position_changed();
409 }
410 else {
411 notify_obstacle_reached();
412 }
413}

415/**
416 * \brief Suspends or resumes the movement
417 * \param suspended true to suspend the movement, false to resume it
418 */
419void CircleMovement::set_suspended(bool suspended) {
420
421 Movement::set_suspended(suspended);
422
423 if (get_when_suspended() != 0) {
424 uint32_t diff = System::now() - get_when_suspended();
425 next_angle_change_date += diff;
426 next_radius_change_date += diff;
427 end_movement_date += diff;
428 restart_date += diff;
429 }
430}

432/**
433 * \brief Starts the circle movement.
434 */
435void CircleMovement::start() {
436
437 current_angle = initial_angle;
438 next_angle_change_date = System::now();
439 nb_rotations = 0;
440
441 if (duration != 0) {
442 end_movement_date = System::now() + duration;
443 }
444
445 if (radius_change_delay == 0) {
446 current_radius = wanted_radius;
447 }
448 else {
449 next_radius_change_date = System::now();
450 }
451 recompute_position();
452}
453
454/**
455 * \brief Returns whether this movement is running.
456 * \return true if the movement is started
457 */
458bool CircleMovement::is_started() const {
459 return current_radius != 0 || wanted_radius != 0;
460}
461
462/**
463 * \brief Returns whether this movement is finished.
464 * \return true if this movement is finished
465 */
466bool CircleMovement::is_finished() const {
467 return is_stopped();
468}
469
470/**
471 * \brief Stops the movement.
472 */
473void CircleMovement::stop() {
474
475 previous_radius = current_radius;
476 set_radius(0);
477
478 if (loop_delay != 0) {
479 restart_date = System::now() + loop_delay;
480 }
481 recompute_position();
482}
